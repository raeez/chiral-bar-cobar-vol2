\chapter{Infinite Fingerprint Classification}
\label{chap:infinite-fingerprint}
\index{fingerprint classification|textbf}
\index{six-slot fingerprint}
\index{$\kappaChHodge$}

Fix a bicoloured primitive package $\primPkgBicolour$ on the log curve
$\logCurve = (X, D, \tau)$, conilpotent-complete and boundary-linear,
with conformal vector at non-critical or critical level, and set $A_b =
\mathrm{End}_{\openFactCat}(b)$. The fingerprint reads six scalars off
the \emph{ordered} bar complex $\barB^{\mathrm{ord}}(A_b) =
T^c(s^{-1}\bar{A_b})$: pole orders from the OPE (slot~1), shadow depth
of the Maurer--Cartan tower $\Theta_{A_b}$ (slot~2), $\mathbb{Z}/2$-grading
of the Hodge filtration (slot~3), strong-generator count (slot~4),
commutant class (slot~5), and the Arakelov bar-curvature $\kappaChHodge$
(slot~6). The symmetric shadow $\barB^{\Sigma}(A_b) = \mathrm{av}_*
\barB^{\mathrm{ord}}(A_b)$ discards slots~4--6: the spectral parameter
and the Arakelov class are killed by $\Sigma_n$-coinvariants, leaving
the ordered bar as the unique carrier of a complete fingerprint.

\section{The six-slot fingerprint}
\label{sec:six-slot-fingerprint}

\begin{definition}[Fingerprint slots]
\label{def:fingerprint-slots}
\index{$p_{\max}$}\index{$r_{\max}$}\index{$\chi_{\mathrm{VOA}}$}
\index{$n_{\mathrm{strong}}$}\index{coset invariant}\index{$\kappaChHodge$}
The fingerprint of $A_b$ is the tuple
\[
 \varphi'(A_b)
 \;=\;
 \bigl(p_{\max}(A_b),\;
 r_{\max}(A_b),\;
 \chi_{\mathrm{VOA}}(A_b),\;
 n_{\mathrm{strong}}(A_b),\;
 \mathrm{coset}(A_b),\;
 \kappaChHodge(A_b)\bigr),
\]
where:
\begin{itemize}
\item $p_{\max}(A_b) \in \mathbb{Z}_{\ge 0} \cup \{\infty\}$ is the
maximum OPE pole order among the open-colour fields of $A_b$;
\item $r_{\max}(A_b) \in \{2, 3, 4, \infty\}$ is the shadow-tower depth;
\item $\chi_{\mathrm{VOA}}(A_b) \in \{+, -\}$ is the Hodge-filtered
$\mathbb{Z}/2$-graded Euler character;
\item $n_{\mathrm{strong}}(A_b) \in \mathbb{Z}_{\ge 0}$ is the minimal
strong-generator count;
\item $\mathrm{coset}(A_b)$ is the commutant-pair class in the
Conway-invariant commutant lattice;
\item $\kappaChHodge(A_b)$ is the chiral-Hodge component of the
$\kappaTuple{A_b}$ tuple, pulled back from the Theorem~D universal
Arakelov class.
\end{itemize}
\end{definition}

\section{Five-slot insufficiency; sixth-slot completeness}
\label{sec:six-slot-completeness}

\begin{proposition}[Five slots insufficient]
\label{prop:five-slot-insufficient-ch}
The truncation $\varphi = (p_{\max}, r_{\max}, \chi_{\mathrm{VOA}},
n_{\mathrm{strong}}, \mathrm{coset})$ fails on the Heisenberg family
$\{\mathcal{H}_k\}_{k \in \mathbb{C}^*}$:
$\varphi(\mathcal{H}_k) = (2, 2, +, 1, \mathrm{trivial})$ independent
of $k$, whereas $\mathcal{H}_k \not\simeq \mathcal{H}_{k'}$ for
$k \ne k'$.
\end{proposition}

\begin{proof}
The Heisenberg OPE $\alpha(z) \alpha(w) \sim k / (z - w)^2$ has
$p_{\max} = 2$ independent of $k$, with shadow-tower truncation at
depth~2 (class G), purely bosonic Hodge sign, one strong generator,
and trivial commutant. The five invariants are $k$-independent; the
bar curvature $d_B^2 = k \cdot \omega_g$ (Theorem~D) separates the
distinct levels.
\end{proof}

\begin{proposition}[Six slots distinguish the Heisenberg family]
\label{prop:six-slot-distinguishes-ch}
$\kappaChHodge(\mathcal{H}_k) = k$. For $k \ne k'$,
$\varphi'(\mathcal{H}_k) \ne \varphi'(\mathcal{H}_{k'})$.
\end{proposition}

\begin{proof}
The bar curvature $d_B^2 = k \cdot \omega_g$ of the Heisenberg
chiral algebra on a genus-$g$ curve has class pulled back from the
universal Arakelov class at $\kappaChHodge = k$.
\end{proof}

\begin{theorem}[Completeness]
\label{thm:fingerprint-complete-ch}
Let $A_b, A_{b'}$ be open-colour endomorphism chiral algebras of two
bicoloured primitive packages in the $C_2$-cofinite standard landscape.
If $\varphi'(A_b) = \varphi'(A_{b'})$, then $A_b \simeq A_{b'}$ as
chiral algebras.
\end{theorem}

\begin{proof}[Sketch]
Case-by-case on the five shadow classes G/L/C/M/FF. For class G
(pole-free, $r_{\max} = 2$), $\varphi'$ pins $(\dim \mathrm{Cartan},
\mathrm{level})$ via $(n_{\mathrm{strong}}, \kappaChHodge)$, giving a
unique Heisenberg/lattice reconstruction. For class L (affine KM), the
Sugawara central charge is determined by $\kappaChHodge$; the Lie
algebra by $(n_{\mathrm{strong}}, \mathrm{coset})$ via the Weyl-group
structure. For class C (W-algebra at $r_{\max} = 3$) and class M
($r_{\max} = \infty$), the Feigin--Frenkel screening-charge
reconstruction determines $A_b$ from the Slodowy datum
$(\mathfrak{g}, f, k)$ encoded by the six slots. Class FF
($\kappaChHodge = 0$) is the critical-level fibre, uniquely
characterised by the coset slot and the Langlands-dual $\mathfrak{g}^L$.
\end{proof}

\section{Quadrichotomy as double coarse projection}
\label{sec:quadrichotomy-projection}

\begin{corollary}[Double coarse projection]
\label{cor:double-coarse-projection}
The quadrichotomy G/L/C/M and the fifth class FF are a double coarse
projection:
\[
 \varphi'(A_b)
 \;\twoheadrightarrow\;
 \varphi(A_b)
 \;\twoheadrightarrow\;
 \{\text{class}(A_b)\} \in \{G, L, C, M, FF\}.
\]
The first projection forgets $\kappaChHodge$; the second projects
$\varphi$ onto its $r_{\max}$ coordinate (with the $\kappaChHodge = 0$
FF stratum as the separate fibre).
\end{corollary}

\section{Coupling to Theorem D}
\label{sec:fingerprint-theoremD-coupling}

\begin{remark}[Theorem~G couples to Theorem~D via $\kappaChHodge$]
\label{rem:fingerprint-theoremD-coupling}
The sixth slot $\kappaChHodge$ of $\varphi'(A_b)$ is the pullback of
the universal Arakelov class of Theorem~D. Theorem~G (Infinite
Fingerprint Classification) and Theorem~D (Modular Characteristic) are
coupled by the identification of the sixth fingerprint slot with the
Arakelov pullback; the five-slot truncation discards Theorem~D's
content and fails on the Heisenberg family.
\end{remark}

\section{Cross-volume anchor: specialisations}
\label{sec:fingerprint-anchor}

The six-slot fingerprint $\varphi'$ admits compatible specialisations
across the three volumes.

\begin{remark}[Vol~III CY-d dimension stratification]
\label{rem:fingerprint-cy-d-stratification}
For Calabi--Yau categories of dimension $d \in \{1, 2, 3, 4, 5\}$, the
Vol~III dimension-stratified identification
\[
 \kappaChHodge(\SpCh \PhiFA_d(\mathcal{C}_X))
 \;=\;
 \sum_{q} (-1)^{q}\, h^{0,q}(X)
\]
(Hodge supertrace via HKR plus Mukai pairing plus $\mathrm{HC}^{-}_{d}$
trace) realises the sixth fingerprint slot as a topological invariant
of $X$. The conifold and local $\mathbb{P}^{2}$ at $d = 3$ exhibit the
stratification: $\kappaChHodge(\mathrm{conifold}) = 1$ (direct McKay)
and $\kappaChHodge(\mathrm{local}\,\mathbb{P}^{2}) = 3/2$. The conifold
is not a local surface at $d = 3$, separating $\kappaChHodge$ from
$\chi(\mathcal{O}_X)$ at odd CY-d.
\end{remark}

\begin{remark}[Borcherds-weight $\kappaBKM$ universal]
\label{rem:fingerprint-bkm-universal}
The Borcherds weight identity
\[
 \kappaBKM(\Phi_{N}) \;=\; c_{N}(0)/2
\]
holds universally across $N \in \{1, 2, 3, 4, 6\}$ (Borcherds 1995;
Gritsenko 1999). The additive coincidence $\kappaBKM = \kappaChHodge
+ \chi(\mathcal{O}_{\mathrm{fiber}})$ holds at $N = 1$ and fails for
$N \ge 2$; the Borcherds-weight expression $c_N(0)/2$ is the universal
formula. The sixth fingerprint slot of $A_b$ specialises through
$\kappaBKM$ on BKM-type chiral algebras attached to Siegel modular
forms of weight $N$, closing the Vol~III universal-weight-formula
question for $\kappaBKM$.
\end{remark}

\begin{remark}[FF stratum at the critical level]
\label{rem:fingerprint-FF-critical}
The fifth class FF ($\kappaChHodge = 0$) is the critical-level fibre
$k = -h^{\vee}$ for affine Kac--Moody. On this stratum the conformal
vector ceases to be inner, the $E_{3}$ structure collapses to $E_{2}$
realising the Feigin--Frenkel centre on opers, and the sixth slot
vanishes. The coset slot $\mathrm{coset}(A_b)$ on FF is replaced by
$\mathrm{coset}_{\mathrm{FF}}(A_b)$, taking values in opers on the
formal disk; the Langlands-dual algebra $\mathfrak{g}^{L}$ is recovered
through the Feigin--Frenkel isomorphism.
\end{remark}

\begin{remark}[Five frontiers at the fingerprint level]
\label{rem:fingerprint-five-frontiers}
The five open frontiers F1--F5 of the climax
(\cref{conj:f1-wp-tempering}--\cref{conj:f5-irregular-singular-genus-g})
manifest at the fingerprint level. F1 (logarithmic $\mathcal{W}(p)$
tempering): does the logarithmic stratum carry a sixth slot
$\kappaChHodge$ satisfying the Adamovi\'c--Milas temperedness bound?
F2 (periodic-CDG non-simply-laced): refinement of the $\mathrm{coset}$
slot at admissible level for non-simply-laced types. F3
(intermediate-topology chain-level class~$\mathbf{M}$): does the class-M
stratum carry a chain-level fingerprint in an explicitly topologized
analytic model stricter than the full weight-completion, the raw
direct-sum bar being excluded at generic central charge? F4
(super-complementarity): single canonical Koszul-dual pairing in place
of the coexisting super-trace and Berezinian pairings on the sixth
slot for super-Yangians. F5 (irregular-singular KZB at higher-genus
boundaries): boundary-divisor extension of the sixth slot on
$\overline{\mathcal{M}}_{g, n}$ for $g \ge 1$. None of F1--F5 is a gap
in Theorem~\ref{thm:fingerprint-complete-ch}; each is a parameter
question on the fingerprint domain.
\end{remark}
