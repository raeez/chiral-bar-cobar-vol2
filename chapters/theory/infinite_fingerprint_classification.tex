\chapter{Infinite Fingerprint Classification}
\label{chap:infinite-fingerprint}
\index{fingerprint classification|textbf}
\index{six-slot fingerprint}
\index{$\kappa_{\mathrm{ch}}$}

Let $\mathcal{A}$ be an $E_1$-chiral algebra on a smooth projective curve
$X$, conilpotent-complete, boundary-linear, with conformal vector at
non-critical level or at critical level. This chapter defines the
six-slot fingerprint $\varphi'(\mathcal{A})$ and proves it is a complete
invariant of the Koszul-bar-complex structure on the standard landscape.

\section{The six-slot fingerprint}
\label{sec:six-slot-fingerprint}

\begin{definition}[Fingerprint slots]
\label{def:fingerprint-slots}
\index{$p_{\max}$}\index{$r_{\max}$}\index{$\chi_{\mathrm{VOA}}$}
\index{$n_{\mathrm{strong}}$}\index{coset invariant}\index{$\kappa_{\mathrm{ch}}$}
The fingerprint of $\mathcal{A}$ is the tuple
\[
  \varphi'(\mathcal{A})
  \;=\;
  \bigl(p_{\max}(\mathcal{A}),\;
        r_{\max}(\mathcal{A}),\;
        \chi_{\mathrm{VOA}}(\mathcal{A}),\;
        n_{\mathrm{strong}}(\mathcal{A}),\;
        \mathrm{coset}(\mathcal{A}),\;
        \kappa_{\mathrm{ch}}(\mathcal{A})\bigr),
\]
where:
\begin{itemize}
\item $p_{\max}(\mathcal{A}) \in \mathbb{Z}_{\ge 0} \cup \{\infty\}$ is
the maximum OPE pole order of $\mathcal{A}$-fields;
\item $r_{\max}(\mathcal{A}) \in \{2, 3, 4, \infty\}$ is the shadow-tower
depth;
\item $\chi_{\mathrm{VOA}}(\mathcal{A}) \in \{+, -\}$ is the Hodge-filtered
$\mathbb{Z}/2$-graded Euler character;
\item $n_{\mathrm{strong}}(\mathcal{A}) \in \mathbb{Z}_{\ge 0}$ is the
minimal strong-generator count;
\item $\mathrm{coset}(\mathcal{A})$ is the commutant-pair class in the
Conway-invariant commutant lattice;
\item $\kappa_{\mathrm{ch}}(\mathcal{A})$ is the central-extension /
level / curvature class pulled back from the Theorem~D universal
Arakelov class.
\end{itemize}
\end{definition}

\section{Five-slot insufficiency; sixth-slot completeness}
\label{sec:six-slot-completeness}

\begin{proposition}[Five slots insufficient]
\label{prop:five-slot-insufficient-ch}
The truncation $\varphi = (p_{\max}, r_{\max}, \chi_{\mathrm{VOA}},
n_{\mathrm{strong}}, \mathrm{coset})$ fails on the Heisenberg family
$\{\mathcal{H}_k\}_{k \in \mathbb{C}^*}$:
$\varphi(\mathcal{H}_k) = (2, 2, +, 1, \mathrm{trivial})$ independent
of $k$, whereas $\mathcal{H}_k \not\simeq \mathcal{H}_{k'}$ for
$k \ne k'$.
\end{proposition}

\begin{proof}
The Heisenberg OPE $\alpha(z) \alpha(w) \sim k / (z - w)^2$ has
$p_{\max} = 2$ independent of $k$; shadow-tower truncation at depth 2
(class L); purely bosonic; one strong generator; trivial commutant.
The five invariants are $k$-independent. However, the bar curvature
$d_B^2 = k \cdot \omega_g$ (Theorem~D) detects the distinct levels.
\end{proof}

\begin{proposition}[Six slots distinguish the Heisenberg family]
\label{prop:six-slot-distinguishes-ch}
$\kappa_{\mathrm{ch}}(\mathcal{H}_k) = k$. For $k \ne k'$,
$\varphi'(\mathcal{H}_k) \ne \varphi'(\mathcal{H}_{k'})$.
\end{proposition}

\begin{proof}
The bar curvature $d_B^2 = k \cdot \omega_g$ of the Heisenberg
chiral algebra on a genus-$g$ curve has class pulled back from the
universal Arakelov class at $\kappa_{\mathrm{ch}} = k$.
\end{proof}

\begin{theorem}[Completeness]
\label{thm:fingerprint-complete-ch}
Let $\mathcal{A}, \mathcal{A}'$ be chiral algebras in the $C_2$-cofinite
standard landscape. If $\varphi'(\mathcal{A}) = \varphi'(\mathcal{A}')$,
then $\mathcal{A} \simeq \mathcal{A}'$ as chiral algebras.
\end{theorem}

\begin{proof}[Sketch]
Case-by-case on the five shadow classes G/L/C/M/FF. For class G
(pole-free, $r_{\max} = 2$), $\varphi'$ pins
$(\dim \mathrm{Cartan}, \mathrm{level})$ via $(n_{\mathrm{strong}},
\kappa_{\mathrm{ch}})$, giving a unique Heisenberg/lattice
reconstruction. For class L (affine KM), the Sugawara central charge
is determined by $\kappa_{\mathrm{ch}}$; the Lie algebra by
$(n_{\mathrm{strong}}, \mathrm{coset})$ via the Weyl-group structure.
For class C (W-algebra at $r_{\max} = 3$) and class M
($r_{\max} = \infty$), the Feigin-Frenkel screening-charge
reconstruction determines $\mathcal{A}$ from
$(\mathfrak{g}, f, k)$ encoded via the six slots. Class FF
($\kappa_{\mathrm{ch}} = 0$) is the critical-level fibre,
uniquely characterised by the coset slot and the Langlands-dual
$\mathfrak{g}^L$.
\end{proof}

\section{Quaternitomy as double coarse projection}
\label{sec:quaternitomy-projection}

\begin{corollary}[Double coarse projection]
\label{cor:double-coarse-projection}
The quaternitomy G/L/C/M and fifth class FF are a double coarse
projection:
\[
  \varphi'(\mathcal{A})
  \;\twoheadrightarrow\;
  \varphi(\mathcal{A})
  \;\twoheadrightarrow\;
  \{\text{class}(\mathcal{A})\} \in \{G, L, C, M, FF\}.
\]
The first projection forgets $\kappa_{\mathrm{ch}}$; the second
projects $\varphi$ onto its $r_{\max}$ coordinate (modulo the
$\kappa = 0$ FF stratum).
\end{corollary}

\section{Coupling to Theorem D}
\label{sec:fingerprint-theoremD-coupling}

\begin{remark}[Theorem G couples tightly to Theorem D via $\kappa_{\mathrm{ch}}$]
\label{rem:fingerprint-theoremD-coupling}
The sixth slot $\kappa_{\mathrm{ch}}$ of $\varphi'(\mathcal{A})$ is the
pullback of the universal Arakelov class of
Theorem~D. Hence the Infinite Fingerprint Classification (Theorem~G
of the programme's Seven Theorems) is NOT independent of the
Modular Characteristic theorem (Theorem~D): the two are tightly
coupled by the identification of the sixth fingerprint slot with
the Arakelov pullback. This coupling explains why the five-slot
fingerprint (which omits Theorem~D's content) is insufficient.
\end{remark}
