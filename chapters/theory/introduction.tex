\section*{The master theorem}

\noindent
The volume's central content is one functor:
\[
 \boxed{\;
 \Phi_{\mathrm{hol}}\colon
 \mathrm{ChirAlg}^{\omega,\mathrm{BL}}_{X}
 \longrightarrow
 \mathrm{HT\text{-}QFT}_{X \times \R},
 \qquad
 \cA \longmapsto T_\cA,
 \;}
\]
Universal Holography
(Theorem~\ref{thm:universal-holography-master},
Chapter~\ref{ch:programme-climax-platonic}). The source category
is chirally Koszul algebras with conformal vector at non-critical
level (notation $\omega$) and a chosen boundary linearisation
(notation $\mathrm{BL}$). The target is 3d holomorphic-topological
gauge theories on $X \times \R$. The functor sends $\cA$ to a
canonical $T_\cA$ with
\[
 \mathrm{Obs}^\partial(T_\cA) \simeq \cA,
 \qquad
 \mathrm{Obs}^{\mathrm{bulk}}(T_\cA) \simeq
 \cZ^{\mathrm{der}}_{\mathrm{ch}}(\cA),
\]
the second equivalence as $E_3$-topological factorisation
algebras in the ambient where the boundary algebra is closed:
on the original chain complex for finite-propagation strata, and on
the weight-completed / KS-$\rho$ Banach ambient for class~M unless
finite propagation is separately verified. This ambient qualifier is
part of the theorem: it is the mechanism that prevents the
holomorphic-topological bulk theorem from claiming raw direct-sum
chain-level strictness in the class~M regime before convergence has
been proved.

The functor is unconditional on the
\emph{non-logarithmic, $C_2$-cofinite standard landscape}:
Virasoro $\mathrm{Vir}_c$ at generic $c$, principal $W_N$ for
every $N$, representatives of all $71$ Schellekens $c = 24$
holomorphic rows, Monster $V^\natural$, irrational cosets
$\mathrm{Com}(B,C)$.
Logarithmic $W(p)$ tempering is open
(Conjecture~\ref{conj:tempered-stratum-contains-wp}): Gurarie
1993 and Flohr 1996 exhibit logarithmic-CFT amplitudes with
unbounded Massey products despite finite-dimensional Zhu, so the
Zhu-bounded-Massey route does not close $W(p)$.

\section*{The coproduct: topological factorisation along $\R$}

The same bar complex carries a coassociative coproduct from
deconcatenation:
\begin{equation}\label{eq:intro-delta}
\Delta[s^{-1}a_1 \,|\, \cdots \,|\, s^{-1}a_n]
\;=\;
\sum_{i=0}^{n}
[s^{-1}a_1 \,|\, \cdots \,|\, s^{-1}a_i]
\;\otimes\;
[s^{-1}a_{i+1} \,|\, \cdots \,|\, s^{-1}a_n].
\end{equation}
Place $n$ fields at ordered positions $t_1 < \cdots < t_n$ on
$\R$ and choose a cut $t_i < t_* < t_{i+1}$; the value of
$\Delta$ at $i$ records the factorisation. The ordered
configuration space $\Conf_n^<(\R) = \{t_1 < \cdots < t_n\}$ is
contractible (open simplex), so the $\Eone$-coalgebra structure
contracts to strict deconcatenation; no higher homotopies; the
coproduct is the unique coassociative coproduct on
$T^c(s^{-1}\bar\cA)$, determined by cogenerators.

\paragraph{Verification on the Heisenberg.}
Take $\cA = \cH_k$, OPE $J(z)J(w) \sim k/(z-w)^2$. Bar element
$[s^{-1}J | s^{-1}J]$ at $(z_1, t_1), (z_2, t_2)$ with
$t_1 < t_2$. Apply $\Delta$:
\[
\Delta[s^{-1}J | s^{-1}J]
\;=\;
[\,] \otimes [s^{-1}J | s^{-1}J]
\;+\;
[s^{-1}J] \otimes [s^{-1}J]
\;+\;
[s^{-1}J | s^{-1}J] \otimes [\,].
\]
Three terms = three cuts. Middle term is factorisation into
independent half-line observables (possible because $J_{(0)}J = 0$:
no simple-pole interaction). Apply $d_{\barB}$ to middle:
$d_{\barB}([s^{-1}J]) = 0$ (no unary OPE), commuting with the
cut. Coderivation property
$\Delta \circ d_{\barB} = (d_{\barB} \otimes \id +
\id \otimes d_{\barB}) \circ \Delta$ states: collisions in
$\C$ commute with cuts in $\R$.

\section*{The pole-order principle}
\label{sec:pole-order-principle}
\index{pole-order principle}

The $E_1$ chiral coassociative coalgebra on $\barB(\cA)$ exists
for every chiral algebra. Its complexity is controlled by the
maximal pole order of the OPE.
\[
 \begin{array}{l|l|l|l|l}
 \text{class} & \text{OPE} & \text{algebra} & m_k \text{ profile} & \text{shadow depth} \\
 \hline
 \mathbf{G} & \text{double, central coeff.} & \text{Heisenberg} & m_k = 0\ \forall k \ge 3 & 2 \\
 \mathbf{L} & \text{double, non-central residues} & \text{affine Kac--Moody} & m_3 \ne 0,\ m_4 = 0\ \text{(coh.)} & 3 \\
 \mathbf{C} & \text{quartic, sparse content} & \beta\gamma\text{ system} & m_3, m_4 \ne 0,\ m_{k\ge 5} = 0\ \text{(coh.)} & 4 \\
 \mathbf{M} & \text{quartic}+,\ \text{rich content} & \text{Virasoro},\ \cW_N & m_k \ne 0\ \forall k \ge 3 & \infty \\
 \end{array}
\]

\textbf{Class $\mathbf{G}$ (Gaussian).}
$\Ainf$ structure trivial; bar complex formal.

\textbf{Class $\mathbf{L}$ (Lie).}
$m_3$ from Jacobi identity; $\Ainf$-formality on cohomology.

\textbf{Class $\mathbf{C}$ (contact).}
First nonlinear correction beyond Lie; finitely generated.

\textbf{Class $\mathbf{M}$ (mixed).}
The quartic pole (in $T(z)T(w)$ for any algebra with conformal
vector; $\cW$-algebras reach order $2s$, but quartic from
Virasoro suffices) forces associator $A_3 \ne 0$. Borcherds
identity converts this to ternary cocycle $m_3$; each $m_k$ is
an independent cocycle controlled by the Kac determinant. The
shadow obstruction tower is infinite; the full perturbative
hierarchy of 3d quantum gravity unfolds.

\paragraph{Class transport via Drinfeld--Sokolov reduction.}
Sugawara:
$T(z) = \frac{1}{2(k + h^\vee)} \sum_a {:}J^a J^a{:}(z)$,
$c = k \dim \fg/(k + h^\vee)$. Double Wick contraction in
$T(z) T(w)$ produces a quartic pole proportional to $c$. DS
reduction is a BRST functor transporting the finite $\Ainf$
tower of $\widehat{\fg}_k$ to the infinite tower of the Virasoro
quotient (Theorem~\ref{thm:ds-hpl-transfer}):
$\mathbf{L} \xrightarrow{\;\mathrm{DS}\;} \mathbf{M}$ (class
transport, manufactured not postulated).

Canonical illustrations: Heisenberg ($\mathbf{G}$); affine
Kac--Moody ($\mathbf{L}$); $\beta\gamma$ ($\mathbf{C}$);
Virasoro and $\cW_N$ ($\mathbf{M}$); DS as L-to-M bridge. Full
development: \S\ref{def:gravity-shadow-depth-class}.

\paragraph{Spectral-depth gap migration.}
Quantitative refinement: each $T$-dependent monomial in $m_n$
carries derivative weight $w$ and spectral depth $d$,
constrained by $w + d = n - 1$. A depth-$d$ term requires a
weight-$(n - 1 - d)$ field; when no such field exists, that
depth is absent from the expansion. Virasoro has no weight-$1$
primary, so depth $n$ is absent from every $m_n$: gap at
$d_{\mathrm{gap}} = n$. For principal $\cW_N$, DS shifts the

\section*{Construction of the Swiss-cheese operad $\SCchtop$}

The structures above (differential, coproduct, $\Ainf$
operations, $R$-matrix, Koszul triangle) are governed by a
single two-coloured operad.

\paragraph{Colours and spaces of operations.}
Colours $\{\mathsf{ch}, \mathsf{top}\}$: closed (holomorphic)
and open (topological).

\emph{Closed-to-closed.}
$\SCchtop(\mathsf{ch}^k;\,\mathsf{ch}) = \FM_k(\C)$, the
Fulton--MacPherson compactification of $k$ labelled points in
$\C$. Smooth manifold with corners of real dimension
$2k - 3$ (for $k \ge 2$, after quotienting by translations and
dilations); codimension-1 strata $D_S$ ($|S| \ge 2$): collision
divisor with internal cluster configuration in $\FM_{|S|}(\C)$
and cluster position in $\FM_{k - |S| + 1}(\C)$:
\begin{equation}
\label{eq:intro-FM-fibration}
\FM_{|S|}(\C) \;\hookrightarrow\; D_S \;\twoheadrightarrow\;
\FM_{k - |S| + 1}(\C).
\end{equation}
Convention: $\FM_0(\C) = \FM_1(\C) = \mathrm{pt}$.

\emph{Mixed-to-open.}
$\SCchtop(\mathsf{ch}^k, \mathsf{top}^m;\,\mathsf{top}) =
\FM_k(\C) \times \Eone(m)$, the product with the
little-intervals operad $\Eone(m) = \Conf_m^{<}(\R)$. The
product decomposition is forced by locality: a local observable
factorising holomorphically in $z$ and locally constant in $t$
depends only on collision data in $\C$ and time-ordering in $\R$.

\emph{Open-to-closed (empty).}
$\SCchtop(\ldots, \mathsf{top}, \ldots;\,\mathsf{ch}) = \varnothing$.
Operadic expression of the physical principle: bulk
interactions restrict to boundaries, not conversely.

\paragraph{Composition.}
By substitution, componentwise.

\emph{Closed-into-closed.}
$\gamma_{\mathrm{ch}}\colon \FM_k(\C) \times \prod_i
\FM_{\ell_i}(\C) \to \FM_{\sum_i \ell_i}(\C)$ via FM blow-up
(Fulton--MacPherson~\cite{FM94}, Getzler--Jones~\cite{GJ94}).

\emph{Open-into-open.}
$\gamma_{\mathrm{mix}}$: product of $\gamma_{\mathrm{ch}}$ on
holomorphic factors with standard $\Eone$ composition on
topological factors.

\emph{Closed-into-open.}
Closed operations may be substituted into closed input slots of
mixed operations; no open operation may be substituted into a
closed input slot. Forced at three levels: geometrically (no
configuration space interpolates real-line data to holomorphic
collision data); D-module level (!-extension support vanishing
from codim-$1$ boundary); physically (the factorized singular
parametrix has separate holomorphic and topological factors, so
topological inputs do not contribute to holomorphic outputs).

\begin{center}
\emph{Composition is associative, equivariant, unital by
inheritance from $\FM_\bullet(\C)$ and $\Eone$
(Definition~\textup{\ref{def:SC-composition}}).}
\end{center}

\begin{observation}[Helicity as Swiss-cheese directionality]
\label{obs:helicity-directionality}
The two $\SCchtop$ colours are, in the celestial reading, the
two helicities of scattering amplitudes. Positive-helicity
insertions are holomorphic (closed colour, on $\FM_k(\C)$,
producing $\omega_k \in \Omega^\bullet_{\log}(\ov{\FM}_k(\C))$
generating the bar differential). Negative-helicity insertions
are topological (open colour, on $\Conf_m^{<}(\R)$, producing
the $\Eone$ coproduct). The no-open-to-closed rule IS the
helicity selection rule: negative-helicity gluons cannot produce
positive-helicity gluons by OPE alone. The MHV amplitude is the
minimal open--closed coupling under $\SCchtop$; all-plus and
single-minus amplitudes vanish for the same form-type
obstruction reason. Self-dual sector identification proved via
$\barBch(\cA)$ in
Theorem~\textup{\ref{thm:ch-core-helicity-splitting}(i)}
(Part~\textup{VII}); anti-self-dual via $\barBch(\cA^!)$,
mixed-helicity via twisting morphism, MHV $=$ minimal
open-closed coupling rest on heuristic form-type arguments;
their elevation to full proofs remains open.
\end{observation}

\section*{The $\Ainf$ chiral operations $m_k$}

A coderivation on the cofree coalgebra $\barB^{\mathrm{ch}}(\cA)
= T^c(s^{-1}\ov{\cA})$ is determined by cogenerator projections:
\[

A((\lambda_1))\cdots((\lambda_{k-1})),
\qquad k \geq 1, \quad \deg m_k = 2 - k,

\paragraph{The $\Ainf$ relations as boundary decompositions of
$\FM_n(\C)$.}
\begin{multline}
\label{eq:intro-ainfty}
\sum_{\substack{i+j = n+1 \\ i,j \geq 1}} \;\sum_{s=0}^{n-j}
(-1)^{|a_1| + \cdots + |a_s|}\;
m_i\bigl(a_1, \ldots, a_s,\,
m_j(a_{s+1}, \ldots, a_{s+j};\, \lambda_{s+1}, \ldots,
\lambda_{s+j-1}) \;\big|\;\\
\Lambda_{[s+1,\ldots,s+j]},\,
a_{s+j+1}, \ldots, a_n;\, \bm\lambda^{\mathrm{out}}\bigr) = 0,
\end{multline}
$\Lambda_{[s+1,\ldots,s+j]} = \lambda_{s+1} + \cdots +
\lambda_{s+j-1}$ effective spectral parameter of fused block.
The relation IS Stokes theorem on $\FM_n(\C)$, term-by-term:
\[
\begin{array}{l|l}
\text{Algebraic datum} & \text{Geometric meaning} \\
\hline
\text{divisor } D_S,\ |S|\ge 2 &
  \text{cluster of } |S| \text{ points colliding} \\
\text{inner } m_{|S|} & \text{residue along } D_S \\
\text{outer } m_{n-|S|+1} &
  \text{fused cluster interacting with remainder} \\
(-1)^{|a_1|+\cdots+|a_s|} &
  \text{Koszul sign (coderivation past first } s\text{)} \\
\lambda_{\mathrm{inner}} \mapsto
  \Lambda_{[s+1,\ldots,s+j]} &
  \text{momentum conservation (fused cluster)}
\end{array}
\]

For $n = 2$: $m_1 \circ m_2 + m_2 \circ
(m_1 \otimes \id + \id \otimes m_1) = 0$, $Q = m_1$ a graded
derivation of $m_2$.

For $n = 3$: codim-1 strata of $\FM_3(\C)$ are $D_{\{1,2\}}$,
$D_{\{2,3\}}$, $D_{\{1,3\}}$. In ordered $A_\infty$ convention,
only consecutive blocks contribute to the bar coderivation:
$D_{\{1,2\}}$ and $D_{\{2,3\}}$ give the two
$m_2(m_2(-,-),-)$ terms; $D_{\{1,3\}}$ (non-consecutive) does
not contribute at the bar-complex level, but contributes to the
PVA Jacobi identity after passing to the unordered shadow
(see Step~5 of Theorem~\ref{thm:stokes_arnold}). The relation
reads:
\begin{equation}
\label{eq:intro-n3}
m_2(m_2(a,b;\lambda_1), c;\, \lambda_1 + \lambda_2) \;-\;
m_2(a, m_2(b,c;\lambda_2);\, \lambda_1) \;+\;
m_1 m_3(a,b,c) \;+\; m_3(Q\cdot) \;=\; 0.
\end{equation}
On cohomology, $m_3$ terms vanish and the surviving identity is
the Jacobi identity for the $\lambda$-bracket. At chain level,
$m_3$ is the homotopy for failure of strict associativity.

\paragraph{Resolvent principle: one equation for all $m_k$.}
$\{m_k\}_{k \ge 2}$ are not independent. With $(H, p, \iota, h)$
a deformation retract of $(\cA, m_1)$ onto cohomology with
$h^2 = h\iota = ph = 0$, the dressed propagator
$\Phi\colon T^+(H) \to \cA$ satisfies
\begin{equation}
\label{eq:intro-resolvent}
\Phi \;=\; \iota \;+\; h \circ m_2 \circ \Phi^{\otimes 2}
\end{equation}
the tree-level bootstrap (Theorem~\ref{thm:resolvent-principle}).
Solution by iteration: $m_k^H$ as a sum over $C_{k-1}$ planar
binary trees (equivalently, binary trees with $n = k - 1$
internal vertices and $k$ leaves), with $m_2$ at vertices and
$h$ at edges. One binary input, one homotopy, all higher
operations.

\begin{equation}
\label{eq:intro-reg-sing}
m_2(a,b)(\lambda) \;=\;
\underbrace{m_2^{\mathrm{reg}}(a,b)}_{\in\, A[[\lambda]]}
\;+\;
\underbrace{m_2^{\mathrm{sing}}(a,b)}_{\in\, \lambda^{-1}
A[\lambda^{-1}]}.
\end{equation}
Regular: normally ordered product (commutative multiplication).
Singular: $\lambda$-bracket (OPE residue):
$a \cdot b := m_2^{\mathrm{reg}}(a,b)|_{\lambda=0}$,
$\{a \,{}_\lambda\, b\} := m_2^{\mathrm{sing}}(a,b)$.
Together they form a Poisson vertex algebra on
$H^\bullet(A, Q)$.


\section*{PVA descent: from $\Ainf$ to Poisson vertex algebra}

\paragraph{Step 1: the product.}
On $Q$-closed $a, b$, the regular part $a \cdot b$ is
well-defined commutative product (Vacuum: if $a = Qc$, $m_2(Qc,
b) = Q m_2(c,b) \pm m_2(c, Qb)$; regular at $\lambda = 0$ is
$Q$-exact). Commutativity from $\sigma\colon \FM_2(\C) \to
\FM_2(\C)$ exchanging the two points: $\lambda \to -\lambda - \partial$.

\paragraph{Step 2: the $\lambda$-bracket.}
Singular part satisfies sesquilinearity (chain-level
sesquilinearity of $m_2$, geometric translation covariance on
$\FM_2(\C)$): $\{\partial a {}_\lambda b\} = -\lambda \{a {}_\lambda b\}$,
$\{a {}_\lambda \partial b\} = (\partial + \lambda)\{a {}_\lambda b\}$.
Skew-symmetry: $\sigma$ on $\FM_2(\C)$ sends
$\lambda \to -\lambda - \partial$;
$\{a {}_\lambda b\} = -\{b {}_{-\lambda - \partial} a\}$.
Leibniz: $n = 2$ $\Ainf$ relation with product yields
$\{c {}_\lambda (a \cdot b)\} = \{c {}_\lambda a\} \cdot b +
a \cdot \{c {}_\lambda b\}$.

\paragraph{Step 3: vanishing of higher operations on cohomology.}
For $k \geq 3$, $m_k = 0$ on $H^\bullet(A, Q)$. The geometric
input is contractibility of $\Conf_k^{<}(\R)$ for $k \geq 3$
(homeomorphic to $\R^{k-1}_{>0}$, an open simplex; note
$\Conf_2^{<}(\R) \cong \R_{>0}$ is also contractible, but the
contractibility argument applies to the transferred higher
homotopies $m_k$ for $k \geq 3$; binary $m_2$ is the generating
operation of the $\Eone$-structure). The $\FM_k(\C) \times
\Conf_k^{<}(\R)$ parametrisation provides a chain homotopy
$h_k$, and $\Ainf$ relations give
$Q \circ m_k + m_k \circ Q^{\otimes} +
\sum (\text{lower-degree}) = 0$;
$[m_k(\alpha_1, \ldots, \alpha_k)] = Q(h_k \circ m_k) = 0$.

\paragraph{Step 4: Jacobi identity.}
Jacobi requires $n = 3$ $\Ainf$ relation and Stokes on
$\FM_3(\C)$. Apply~\eqref{eq:intro-n3} on $Q$-closed $a, b, c$;
$m_3$ terms $Q$-exact, $m_1 = Q$ vanishes. Surviving:
$m_2(m_2(a,b;\lambda_1), c;\, \lambda_1 + \lambda_2) =
m_2(a, m_2(b,c;\lambda_2);\, \lambda_1)$. Singular parts give
LHS and RHS of Jacobi identity; mixed terms cancel by Arnold
relation $\omega_{12}\wedge\omega_{23} + \omega_{23}\wedge\omega_{31}
+ \omega_{31}\wedge\omega_{12} = 0$ on $\Conf_3(\C)$ via Stokes
on boundary strata of $\FM_3(\C)$. Full verification:
Theorem~\ref{thm:cohomology_PVA}, Section~\ref{sec:PVA_descent}.

\begin{maintheorem}[Cohomological descent to PVA;
Theorem~\textup{\ref{thm:cohomology_PVA}}; \ClaimStatusProvedHere]
\label{thm:intro-PVA-descent}
Let $\cA$ be a logarithmic $\SCchtop$-algebra
\textup{(}Definition~\textup{\ref{def:log-SC-algebra}}: an
$\SCchtop$-algebra whose structure maps extend to logarithmic
differential forms on the FM compactification, i.e.\ propagator
is a log $1$-form with at worst simple poles along boundary

($\lambda$-bracket has cohomological degree $+1$; underlying Lie

\begin{enumerate}[label=\textup{(\roman*)}]
\item commutative product is the regular part of $m_2$;
\item $\lambda$-bracket is the singular part of $m_2$;
\item all higher operations $m_{k \geq 3}$ vanish on cohomology.
\end{enumerate}
PVA axioms (sesquilinearity, skew-symmetry, Jacobi, Leibniz,

\end{maintheorem}

The PVA is the classical shadow: it remembers the commutative

$m_3, m_4, \ldots$.

\section*{The spectral $R$-matrix from the $\lambda$-bracket}

The $\lambda$-bracket encodes the classical braiding of line
operators. Bridge from PVA to Yang--Baxter: Laplace transform.

\paragraph{The classical $r$-matrix.}
\begin{equation}
\label{eq:intro-laplace}
r(z) \;=\; \int_0^\infty e^{-\lambda z}\,
\{{\cdot}\,{}_\lambda\,{\cdot}\}\, d\lambda.
\end{equation}
Convergent for $\operatorname{Re}(z) > 0$ under the
physical-realization hypotheses of
Theorem~\ref{thm:physics-bridge}; meromorphic on $\C^\times$.
Algebraically: if $\{a {}_\lambda b\} = \sum_{n \ge 0} c_n \lambda^n / n!$,
then $r(z)(a \otimes b) = \sum_{n \ge 0} c_n / z^{n+1}$.

\emph{Classical Yang--Baxter.} PVA Jacobi transforms under
Laplace into CYBE with spectral parameter:
\begin{equation}
\label{eq:intro-CYBE}
[r_{12}(z_{12}),\, r_{13}(z_{13})] +
[r_{12}(z_{12}),\, r_{23}(z_{23})] +
[r_{13}(z_{13}),\, r_{23}(z_{23})] \;=\; 0.
\end{equation}
$\mu = \lambda_2$, $\lambda = \lambda_1$ becomes
$z_{13} = z_{12} + z_{23}$ under Laplace; the Jacobi terms map
to the three CYBE commutators.

\paragraph{The quantum $R$-matrix.}
$R(z)$ from bulk-to-boundary OPE of line operators: parallel
boundary lines $L_1(z_1), L_2(z_2)$ in 3d HT theory on
$\C \times \R_{\ge 0}$ with braiding
\[
R_{L_1, L_2}(z) \colon L_1 \otimes L_2
\;\xrightarrow{\;\sim\;}\; L_2 \otimes L_1,
\qquad z = z_1 - z_2,
\]
meromorphic, computed by configuration integrals over
$\FM_2(\C) \times \Eone(2)$
(Definition~\ref{def:spectral-braiding}). Semiclassical limit
$R(z) = \id + \h\, r(z) + O(\h^2)$ recovers $r(z)$.

\emph{Yang--Baxter.}
\begin{equation}
\label{eq:intro-YBE}
R_{12}(z_{12})\, R_{13}(z_{13})\, R_{23}(z_{23})
\;=\;
R_{23}(z_{23})\, R_{13}(z_{13})\, R_{12}(z_{12}),
\end{equation}
proved by Stokes on $\FM_3(\C)$. Three codim-1 strata contribute
the three double products on each side; vanishing of the
boundary integral
$\int_{\partial \FM_3(\C)} \omega_3^{\partial} = 0$ forces
equality. $D_{\{1,3\}}$ does not contribute independently:
spectral parameter $z_{13} = z_{12} + z_{23}$.

\emph{Classical limit.} CYBE is the semiclassical limit of YBE;
both follow from Stokes on $\FM_3(\C)$.
