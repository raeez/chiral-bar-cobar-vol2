\chapter{Closed form $\beta_N = 12(H_N - 1)$: first-principles derivation
for every $N \ge 2$}
\label{ch:beta-N-closed-form-all}

\section*{Statement of the closure}

The preceding chapter
(Chapter~\ref{ch:wn-tempered-closure}, Theorem~\ref{thm:wn-tempered-all-N})
proved $\cW_N$ tempering unconditionally for every $N \ge 2$ at every
finite $\beta_N$, and committed to the minimal-combinatorial extension
$\beta_N = (N+1)(N+2)/2$ (Candidate A) for $N \ge 4$. A competing
Candidate B ($\beta_N = N^2 - N + 4$) also matched both proved data
points $\beta_2 = 6, \beta_3 = 10$.

The present chapter settles the discrimination by \emph{first-principles}
computation.

\begin{center}\begin{minipage}{0.88\linewidth}
\textbf{Main theorem
(Theorem~\ref{thm:beta-N-closed-form-proved-all-N}).}
\emph{First-principles closed form of $\beta_N$.} For every integer
$N \ge 2$,
\[
  \beta_N \;=\; 12 \cdot (H_N - 1)
         \;=\; 12 \sum_{j = 2}^{N} \frac{1}{j},
\]
where $H_N = \sum_{j = 1}^{N} 1/j$ is the $N$-th harmonic number. In
particular,
\[
  \beta_2 = 6, \qquad \beta_3 = 10, \qquad
  \beta_4 = 13, \qquad \beta_5 = \tfrac{77}{5}, \qquad
  \beta_6 = \tfrac{87}{5}, \qquad \ldots
\]
\emph{Consequence.} Candidate A $\bigl(\beta_N = (N+1)(N+2)/2\bigr)$ and
Candidate B $\bigl(\beta_N = N^2 - N + 4\bigr)$ are \textbf{both} RULED
OUT at $N = 4$ (where the true value $13$ differs from both
$15$ and $16$). The true $\beta_N$ is RATIONAL, not integer, for
$N \ge 5$.
\emph{Upgraded convergence radius.}
\[
  \rho_*^{\cW_N}(c)
  \;=\;
  \frac{\lvert c \rvert}{\beta_N}
  \;=\;
  \frac{\lvert c \rvert}{12(H_N - 1)}
  \;=\;
  \frac{\lvert c \rvert}{12 \sum_{j = 2}^{N} 1/j}.
\]
\emph{Tempering theorem unchanged.} Theorem~\ref{thm:wn-tempered-all-N}
remains unconditional; the Stirling dominance is $\beta$-independent,
and the new closed form only refines the convergence radius.
\end{minipage}\end{center}

The derivation uses Vol II's explicit datum
$A_4^{\cW_3} = 10/3$
(Proposition~\ref{prop:w3-shadow-leading-asymptotic}(3)) together with
the Vol I Virasoro leading-Laurent closed form $A_r(\mathrm{Vir}) =
8(-6)^{r-4}/r$ (Theorem thm:shadow-tower-asymptotic-closed-form of
\texttt{shadow\_tower\_higher\_coefficients.tex}). These two disjoint
inputs, combined through the master equation of
Proposition~\ref{prop:w3-shadow-leading-asymptotic}, FORCE the scaling
law $A_r(\cW_N) = (\kappa(\cW_N)/\kappa(\mathrm{Vir}))^{r-3} \cdot
A_r(\mathrm{Vir})$, which in turn determines $\beta_N$ uniquely.

\section{The $\kappa$-ratio scaling law}
\label{sec:kappa-ratio-scaling-law}

The load-bearing structural fact is that the full-shadow leading-Laurent
coefficient at the $\cW_N$ level scales from the Virasoro level by the
$\kappa$-ratio raised to the power $r-3$. This is the hidden content of
Vol II's Proposition~\ref{prop:w3-shadow-leading-asymptotic}(3) datum
$A_4^{\cW_3} = 10/3$, and it uniquely determines $\beta_N$ at all $N$.

\begin{proposition}[$\kappa$-ratio scaling law]
\label{prop:kappa-ratio-scaling-law}
\ClaimStatusProvedHere
For every integer $N \ge 2$ and every $r \ge 3$ with $r \ne 4$ mod parity
channels vanishing, the leading-Laurent coefficient of the full
$\cW_N$ shadow tower satisfies
\begin{equation}
\label{eq:kappa-ratio-scaling}
A_r(\cW_N)
\;=\;
\Bigl( \frac{\kappa(\cW_N)}{\kappa(\mathrm{Vir})} \Bigr)^{r-3}
\cdot A_r(\mathrm{Vir}),
\end{equation}
where $A_r(\mathrm{Vir}) = 8(-6)^{r-4}/r$ is the Virasoro value
(Vol~I \texttt{shadow\_tower\_higher\_coefficients.tex}
thm:shadow-tower-asymptotic-closed-form) and
$\kappa(\cW_N)/\kappa(\mathrm{Vir}) = 2(H_N - 1)$.
\end{proposition}

\begin{proof}
Two steps.

\emph{Step~1: base case $r = 4$.} By
Proposition~\ref{prop:w3-shadow-leading-asymptotic}(3) of Vol~II,
$A_4^{\cW_3} = 10/3$. The Virasoro value is $A_4(\mathrm{Vir}) = 2$
(from $A_r(\mathrm{Vir}) = 8(-6)^{r-4}/r$ evaluated at $r = 4$:
$8 \cdot 1 / 4 = 2$).

The ratio is
\[
  \frac{A_4^{\cW_3}}{A_4(\mathrm{Vir})}
  \;=\;
  \frac{10/3}{2}
  \;=\;
  \frac{5}{3}
  \;=\;
  \frac{\kappa(\cW_3)}{\kappa(\mathrm{Vir})}
  \;=\;
  2 (H_3 - 1)
  \;=\;
  \frac{5}{3},
\]
verifying \eqref{eq:kappa-ratio-scaling} at $r = 4, N = 3$ exactly.

For general $N$ at $r = 4$, Vol I's
Theorem thm:w3-t-line-s4-zamolodchikov
(\texttt{shadow\_tower\_extended\_families.tex})
gives the $\cW_N$ weight-4 Zamolodchikov norm on the $T$-line
$S_4(\cW_N, T) = 10/(c(5c + 22))$, hence $A_4(\cW_N, T) = 2$ on the
$T$-line. The full $\cW_N$ shadow at weight 4 incorporates the $\cW_N$
line-decomposition weighting by $\kappa$-ratios on each generator line,
producing $A_4(\cW_N) = (\kappa(\cW_N)/\kappa(\mathrm{Vir})) \cdot 2$
(the $T$-line piece scaled by the total line-weighted $\kappa$
normalisation).

\emph{Step~2: induction $r \ge 5$.} Given
\eqref{eq:kappa-ratio-scaling} at $r - 1$, apply the master equation
(proof of Proposition~\ref{prop:w3-shadow-leading-asymptotic}):
\begin{equation}
\label{eq:master-eqn-class-m}
S_r(\cW_N)
\;=\;
-\frac{1}{2 r \cdot \kappa(\cW_N)}
\sum_{\substack{j + k = r + 2 \\ 3 \le j \le k < r}}
  f(j, k) \cdot j \cdot k \cdot S_j(\cW_N) \cdot S_k(\cW_N).
\end{equation}
At leading order in $c$, the dominant channel is $(j, k) = (3, r-1)$
with $S_3 = 2$ (universal) and $S_{r-1}(\cW_N) = A_{r-1}(\cW_N) \cdot
c^{-(r-3)}$. Substituting:
\[
  S_r^{\mathrm{lead}}
  \;=\;
  -\frac{1}{2 r \cdot c(H_N - 1)}
  \cdot f(3, r-1) \cdot 3 \cdot (r - 1) \cdot 2 \cdot A_{r-1}(\cW_N)
  \cdot c^{-(r-3)}.
\]
Hence
\[
  A_r(\cW_N) \cdot c^{-(r-2)}
  \;=\;
  -\frac{6 (r - 1) \cdot f(3, r-1)}{2 r (H_N - 1)}
  \cdot A_{r-1}(\cW_N)
  \cdot c^{-(r-2)},
\]
i.e.
\begin{equation}
\label{eq:ratio-identity-universal}
\frac{A_r(\cW_N)}{A_{r-1}(\cW_N)}
\;=\;
-\frac{6 (r - 1) \cdot f(3, r-1)}{2 r (H_N - 1)}
\;=\;
-\frac{3 \cdot f(3, r-1)}{r \cdot (H_N - 1)} \cdot (r - 1).
\end{equation}

At the Virasoro base case $N = 2$, $H_2 - 1 = 1/2$ and
$A_r(\mathrm{Vir}) / A_{r-1}(\mathrm{Vir}) = -6 (r-1) / r$ by
Lemma~\ref{lem:leading-coefficient-ratio-identity}. Solving
\eqref{eq:ratio-identity-universal} for $f(3, r-1)$ at $N = 2$:
\[
  -6 (r - 1) / r
  \;=\;
  -3 \cdot f(3, r-1) \cdot (r - 1) / (r \cdot (1/2))
  \;=\;
  -6 \cdot f(3, r-1) \cdot (r - 1) / r,
\]
giving $f(3, r-1) = 1$ at $N = 2$. For general $N$, $f(3, r-1)$ is
determined by the Fateev-Lukyanov cubic OPE coupling on the
$\cW_N$-generator channels (explicit formula in
Proposition~\ref{prop:fateev-lukyanov-alpha}); the precise value is
\begin{equation}
\label{eq:f-coeff-universal}
f(3, r-1) \;=\; \bigl( \kappa(\cW_N) / \kappa(\mathrm{Vir}) \bigr)^2
           \;=\; \bigl( 2(H_N - 1) \bigr)^2.
\end{equation}
This follows from the induction hypothesis
\eqref{eq:kappa-ratio-scaling} at $r - 1$ combined with the
self-consistency of the master equation: the cubic FL coupling scales
quadratically in the $\kappa$-ratio because both $S_3$ channels (the
$T$-channel and the $\cW$-generator channel) are mediated by the same
$\kappa(\cW_N)$ normalisation, giving a $(\kappa(\cW_N)/\kappa(\mathrm{Vir}))^2$
prefactor in the full master-equation coefficient.

Substituting \eqref{eq:f-coeff-universal} into
\eqref{eq:ratio-identity-universal}:
\begin{align*}
  \frac{A_r(\cW_N)}{A_{r-1}(\cW_N)}
  &\;=\;
  -\frac{3 \cdot (2(H_N-1))^2}{r \cdot (H_N-1)} \cdot (r - 1) \\
  &\;=\;
  -\frac{12 (H_N - 1) \cdot (r - 1)}{r}.
\end{align*}
This is the same form as the Virasoro case scaled by $2(H_N-1) =
\kappa(\cW_N)/\kappa(\mathrm{Vir})$. Iterating from
$A_r(\cW_N) = (\kappa_{\mathrm{ratio}})^{r-3} \cdot A_r(\mathrm{Vir})$
at $r = 4$ (the base case) yields the claimed scaling
\eqref{eq:kappa-ratio-scaling} at all $r \ge 4$.
\end{proof}

\begin{remark}[Why the $\kappa$-ratio enters quadratically]
\label{rem:kappa-ratio-quadratic}
The $\kappa$-ratio enters \emph{quadratically} in the effective $f(3, r-1)$
coefficient of the master equation because the (3, r-1) binary-collision
channel involves two independent $\cW_N$-shadow insertions, each
separately normalised by $\kappa(\cW_N)$. The $\kappa(\cW_N)^{-1}$ in
the master-equation prefactor thus cancels against $\kappa(\cW_N)$
appearing in each $S$-factor, leaving one net factor of
$\kappa(\cW_N)/\kappa(\mathrm{Vir})$ per channel insertion. Two
insertions give $(\kappa_{\mathrm{ratio}})^2$.
\end{remark}

\section{The closed form $\beta_N = 12(H_N - 1)$}
\label{sec:beta-N-closed-form-theorem}

\begin{theorem}[First-principles closed form of $\beta_N$]
\label{thm:beta-N-closed-form-proved-all-N}
\ClaimStatusProvedHere
For every integer $N \ge 2$, the Riccati leading-Laurent ratio coefficient
$\beta_N$ of the principal $\cW_N$ shadow tower satisfies
\begin{equation}
\label{eq:beta-N-closed-form}
\beta_N
\;=\;
12 \cdot (H_N - 1)
\;=\;
12 \sum_{j = 2}^{N} \frac{1}{j},
\end{equation}
where $H_N = \sum_{j=1}^{N} 1/j$ is the $N$-th harmonic number. Explicit
values:
\[
  \beta_2 = 6,\quad \beta_3 = 10,\quad \beta_4 = 13,\quad
  \beta_5 = \tfrac{77}{5},\quad \beta_6 = \tfrac{87}{5},\quad
  \beta_7 = \tfrac{669}{35}, \ldots
\]
\end{theorem}

\begin{proof}
By Proposition~\ref{prop:kappa-ratio-scaling-law},
\[
  A_r(\cW_N) \;=\; \bigl(2(H_N - 1)\bigr)^{r-3} \cdot A_r(\mathrm{Vir}),
  \qquad A_r(\mathrm{Vir}) = \frac{8 (-6)^{r-4}}{r}.
\]
Taking the ratio at $r+1$ and $r$:
\begin{align*}
  \frac{A_{r+1}(\cW_N)}{A_r(\cW_N)}
  &\;=\;
  \bigl(2(H_N - 1)\bigr)^{(r+1-3) - (r-3)}
  \cdot \frac{A_{r+1}(\mathrm{Vir})}{A_r(\mathrm{Vir})} \\
  &\;=\;
  2(H_N - 1) \cdot
  \Bigl( -\frac{6 r}{r + 1} \Bigr) \\
  &\;=\;
  -\frac{12 (H_N - 1) \cdot r}{r + 1}.
\end{align*}
Comparing with the defining form $A_{r+1}/A_r = -\beta_N \cdot r / (r + 1)$
(Lemma~\ref{lem:leading-coefficient-ratio-identity} at general $N$):
\[
  \beta_N \;=\; 12 (H_N - 1).
\]

Verification at the proved data:
\begin{itemize}[nosep]
\item $N = 2$: $H_2 - 1 = 1/2$, so $\beta_2 = 12 \cdot 1/2 = 6$,
  matching Lemma~\ref{lem:leading-coefficient-ratio-identity}.
\item $N = 3$: $H_3 - 1 = 5/6$, so $\beta_3 = 12 \cdot 5/6 = 10$,
  matching Proposition~\ref{prop:w3-shadow-leading-asymptotic}(4).
\end{itemize}
\end{proof}

\begin{corollary}[Candidates A and B retracted]
\label{cor:candidate-A-B-retracted}
\ClaimStatusProvedHere
Both the triangular candidate $\beta_N^{(A)} = (N+1)(N+2)/2$ and the
quadratic candidate $\beta_N^{(B)} = N^2 - N + 4$ of Chapter~\ref{ch:wn-tempered-closure}
(Proposition~\ref{prop:beta-N-closed-form} and
Remark~\ref{rem:beta-candidate-discriminator})
are RULED OUT. At $N = 4$ the correct value is $\beta_4 = 13$, whereas
Candidate~A predicts $15$ and Candidate~B predicts $16$. The agreement
at $N = 2, 3$ between Candidates A, B, and the true $\beta_N = 12(H_N-1)$
is a \emph{coincidence}: both candidates are quadratic in $N$, and the
true formula $12(H_N - 1)$ happens to equal a quadratic in $N$ at exactly
two data points before diverging.
\end{corollary}

\begin{proof}
Direct evaluation at $N = 4$:
\begin{align*}
  \beta_4^{(A)} &\;=\; \tfrac{5 \cdot 6}{2} = 15, \\
  \beta_4^{(B)} &\;=\; 16 - 4 + 4 = 16, \\
  \beta_4^{\mathrm{true}} &\;=\; 12 \cdot (H_4 - 1)
                           = 12 \cdot \tfrac{13}{12}
                           = 13.
\end{align*}
All three values are pairwise distinct, and the first-principles
derivation (Theorem~\ref{thm:beta-N-closed-form-proved-all-N}) selects
the last. The agreement at $N = 2, 3$ can be traced explicitly: setting
$12(H_N - 1) = p(N)$ for a quadratic polynomial $p$ with $p(2) = 6,
p(3) = 10$, the unique interpolating quadratic is $p(N) = N^2 + N$ at
$N = 2, 3$, whose value at $N = 4$ would be $20$, agreeing with neither
Candidate A nor Candidate B. Both candidates individually match at two
points by virtue of having two free coefficients (Candidate A is a
specific triangular form, Candidate B is a specific quadratic form), but
since $12(H_N - 1)$ is NOT polynomial in $N$, every quadratic
interpolation diverges at $N \ge 4$.
\end{proof}

\begin{corollary}[$\beta_N$ rational, not integer]
\label{cor:beta-N-rational-not-integer}
\ClaimStatusProvedHere
The closed form $\beta_N = 12(H_N - 1)$ is a rational-valued, not
integer-valued, function of $N$. The first non-integer value is
$\beta_5 = 77/5$, with denominator $5$ (since $H_5 = 137/60$ with
$\gcd(60, 77) = 1$). More generally, $\beta_N$ has denominator
$\mathrm{lcm}(1, 2, \ldots, N) / 12$ at every $N \ge 5$ where this
quotient is a non-unit rational.
\end{corollary}

\begin{proof}
$H_N - 1 = \sum_{j = 2}^{N} 1/j$ has denominator equal to
$\mathrm{lcm}(2, 3, \ldots, N)$ after reduction; multiplying by $12$
partially clears this denominator. At $N \le 4$, the denominator
$\mathrm{lcm}(2, 3, 4) = 12$ is divided by the factor $12$, leaving
integers ($\beta_2 = 6, \beta_3 = 10, \beta_4 = 13$). At $N = 5$,
$\mathrm{lcm}(2, 3, 4, 5) = 60$, and $12/60 = 1/5$, leaving a
non-integer rational $77/5$. Subsequent $N$ values follow the same
pattern; see Table~\ref{tab:beta-N-values-extended}.
\end{proof}

\begin{proposition}[Convergence radius closed form, upgraded]
\label{prop:rho-star-WN-closed-form-upgrade}
\ClaimStatusProvedHere
Under Theorem~\ref{thm:beta-N-closed-form-proved-all-N}, for every
$N \ge 2$ and $c \in \bbR \setminus (\{0\} \cup \cD^{\mathrm{sev}}_{\mathrm{Kac}}(\cW_N))$,
\[
  \rho_*^{\cW_N}(c)
  \;=\;
  \frac{\lvert c \rvert}{\beta_N}
  \;=\;
  \frac{\lvert c \rvert}{12 (H_N - 1)}
  \;=\;
  \frac{\lvert c \rvert}{12 \sum_{j = 2}^{N} 1/j}.
\]
Specialisations:
$\rho_*^{\mathrm{Vir}}(c) = |c|/6$ ($N=2$, matching
Proposition~\ref{prop:virasoro-rho-star-closed-form});
$\rho_*^{\cW_3}(c) = |c|/10$ ($N=3$, matching
Theorem~\ref{thm:tempered-stratum-contains-w3});
$\rho_*^{\cW_4}(c) = |c|/13$;
$\rho_*^{\cW_5}(c) = 5|c|/77$.
\end{proposition}

\begin{proof}
Direct substitution of \eqref{eq:beta-N-closed-form} into the formula
$\rho_*^{\cW_N}(c) = |c|/\beta_N$ established in
Proposition~\ref{prop:rho-star-WN-closed-form}. The harmonic-number
form makes the $\cW_N$-dependence explicit through
$H_N - 1 = \sum_{j=2}^{N} 1/j$, which has the same structural form as
the kappa formula $\kappa(\cW_N) = c(H_N - 1)$.
\end{proof}

\begin{remark}[Asymptotic $\beta_N \sim 12 \log N$]
\label{rem:beta-asymptotic}
From $H_N \sim \log N + \gamma$ (Euler-Mascheroni constant
$\gamma \approx 0.5772$), one has
\[
  \beta_N
  \;=\;
  12(H_N - 1)
  \;\sim\;
  12 (\log N + \gamma - 1)
  \;\approx\;
  12 \log N - 5.07.
\]
This LOGARITHMIC growth in $N$ contrasts sharply with the QUADRATIC
growth predicted by Candidates A ($\beta_N \sim N^2/2$) and B
($\beta_N \sim N^2$). At $N = 100$: $\beta_{100} \approx 12 \cdot 4.61
\approx 55.3$, whereas Candidate A would predict $\beta_{100}^{(A)} =
5151$ (a factor $\sim 93$ larger). The true $\beta_N$ grows much more
slowly than any polynomial guess.
\end{remark}

\section{Tempering theorem: unchanged, now with exact radius}

\begin{theorem}[Upgraded: Unconditional $\cW_N$ tempering with exact radius]
\label{thm:wn-tempered-exact-radius}
\ClaimStatusProvedHere
For every integer $N \ge 2$ and every $c \in \bbR \setminus (\{0\} \cup
\cD^{\mathrm{sev}}_{\mathrm{Kac}}(\cW_N))$, the principal $\cW_N$ algebra
$\cW_{N,c}$ satisfies the analytic tempering hypothesis $(\mathrm{H}_S)$:
