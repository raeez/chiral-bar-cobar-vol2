\chapter{Irrational Cosets in the Tempered Stratum: infinite Zhu does
not obstruct tempering}
\label{ch:irrational-cosets-tempered}

\section*{Statement of the frontier}

Chapter~\ref{ch:tempered-stratum-characterization} and
Chapter~\ref{ch:logarithmic-wp-tempered-analysis} established that
every generic Virasoro algebra $\mathrm{Vir}_c$, every principal
$\cW_N$ at generic central charge, and every logarithmic triplet
$\cW(p)$, $p \ge 2$, is analytically tempered. The $\cW(p)$ analysis
proposed \emph{Conjecture~\ref{conj:tempered-unbounded-zhu}}: bounded
Zhu algebra $\Longleftrightarrow$ tempered, with contrapositive
``every non-tempered VOA has infinite Zhu algebra.'' Three explicit
candidates were flagged:
\begin{enumerate}[nosep]
\item Monster-type twisted sectors
 (Frenkel-Lepowsky-Meurman $V^\natural$ with the full pseudo-trace
 pyramid), closed independently in a subsequent Vol~II inscription
 via the Dijkgraaf-Witten anomaly $\alpha = 0$ mechanism.
\item Irrational cosets
 $\mathrm{Com}(\cA, \cB)$ where $\cA \subset \cB$ but $\cA$ is not
 $C_2$-cofinite.
\item Non-rational affine minimal cosets
 (admissible-level $\widehat{\mathfrak{sl}}_2$ with infinite-dim Zhu
 pyramid).
\end{enumerate}

The present chapter attacks (2) and, as a consequence, closes (3).
The strongest honest form is: every irrational coset in the standard
landscape is analytically tempered; the contrapositive of
Conjecture~\ref{conj:tempered-unbounded-zhu} FAILS; the refined
tempered criterion is the \emph{Virasoro-sub-channel in Kac-regular
locus with bounded generator-to-generator OPE pole order} condition
(Definition~\ref{def:vskr-bgg-criterion}).

\begin{center}\begin{minipage}{0.88\linewidth}
\textbf{Theorem summary.}
\emph{(Universal irrational-coset tempering.)} Every irrational
coset in the three explicit families
$\{K(\mathfrak{sl}_2, k), \mathrm{Com}(H, V_k(\mathfrak{sl}_2)),
\mathrm{Com}(\mathrm{Vir}_{c'}, V_k(\mathfrak{sl}_2))\}$ at
$(k, c')$ outside the severe Kac-zero locus satisfies
$\limsup_{r \to \infty} (\lvert S_r \rvert / r!)^{1/r} = 0$.
\emph{(Explicit radii.)}
$\rho_*^{K(\mathfrak{sl}_2, k)}(k) = \lvert k - 1 \rvert
 / (3 (k + 2))$ for $k > -2$, $k \neq 1$.
$\rho_*^{\mathrm{Com}(\mathrm{Vir}_{c'}, V_k)}(k, c') =
 \lvert 3 k / (k + 2) - c' \rvert / 6$.
\emph{(Refined criterion.)} The tempered stratum of class
$\mathsf{M} \cup \mathsf{M}^*$ is characterised on the standard
landscape by VSKR + BGG (Definition~\ref{def:vskr-bgg-criterion}),
STRICTLY WEAKER than $C_2$-cofiniteness. Infinite-dim Zhu algebra is
not an obstruction; the contrapositive of
Conjecture~\ref{conj:tempered-unbounded-zhu} is retracted.
\emph{(Consequence for candidate (iii).)} Non-rational affine
minimal cosets at admissible level $k = -2 + p/q$, $\gcd(p, q) = 1$,
are tempered with $\rho_* = \lvert (k-1) \rvert / (3(k+2))$; at truly
irrational $k$ the same mechanism gives tempering with irrational
$\rho_*$.
\end{minipage}\end{center}

\section{Irrational cosets in the standard landscape}
\label{sec:irrational-cosets-definitions}

\begin{definition}[Irrational coset in the standard landscape]
\label{def:irrational-coset-standard-landscape}
An \emph{irrational coset} is a commutant subalgebra
$\mathrm{Com}(\cA, \cB) \subset \cB$ where (i) $\cB$ is a chirally
Koszul VOA in the standard landscape; (ii) $\cA \subset \cB$ is a
sub-VOA which is NOT $C_2$-cofinite; (iii) the embedding parameter
(level, central charge, or modular parameter) is irrational or
otherwise generic outside the countable locus of $C_2$-cofinite
embeddings.

Three explicit families in scope:
\begin{enumerate}[nosep]
\item \emph{Parafermion cosets.}
 $K(\mathfrak{sl}_2, k)
 = \mathrm{Com}(H_k, V_k(\mathfrak{sl}_2))$
 where $H_k$ is the Cartan Heisenberg sub-VOA at level $k$.
\item \emph{Virasoro-in-affine cosets.}
 $\mathrm{Com}(\mathrm{Vir}_{c'}, V_k(\mathfrak{sl}_2))$ where
 $\mathrm{Vir}_{c'} \subset V_k(\mathfrak{sl}_2)$ is a non-Sugawara
 Virasoro embedding at irrational central charge $c'$.
\item \emph{Non-rational affine minimal cosets.}
 $\mathrm{Com}(H_k, L_k(\mathfrak{sl}_2))$ at admissible level
 $k = -2 + p/q$ with $\gcd(p, q) = 1$ and $p, q$ outside the
 rational-CFT locus (admissible but non-unitary), limiting to
 irrational-$k$ cosets in the generic locus.
\end{enumerate}
\end{definition}

\begin{definition}[VSKR + BGG criterion]
\label{def:vskr-bgg-criterion}
A chiral algebra $\cA$ in the standard landscape satisfies the
\emph{Virasoro sub-channel in Kac-regular locus with bounded
generator-to-generator OPE pole order} criterion (abbreviated
VSKR + BGG) if:
\begin{enumerate}[nosep]
\item[(VSKR)] $\cA$ contains a Virasoro sub-VOA
 $\mathrm{Vir}_c \subset \cA$ at a central charge
 $c \notin \{0\} \cup \cD^{\mathrm{sev}}_{\mathrm{Kac}}$ (the
 severe Kac-zero locus of
 Theorem~\ref{thm:tempered-stratum-contains-virasoro}).
\item[(BGG)] Every pairwise OPE
 $\phi_a(z) \phi_b(w)$ among the strong generators of $\cA$ has
 pole order $p_{\max}^{ab} \le P(\cA) < \infty$ uniformly bounded
 by a finite $P(\cA)$.
\end{enumerate}
VSKR + BGG is \emph{sufficient} for analytic tempering
(Theorem~\ref{thm:irrational-coset-tempered} below).
$C_2$-cofiniteness $\Longrightarrow$ VSKR + BGG, but not conversely
(Corollary~\ref{cor:tempered-criterion-refined}).
\end{definition}

\begin{remark}[Why VSKR + BGG works]
\label{rem:vskr-bgg-mechanism}
The VSKR clause provides a Virasoro sub-channel whose shadow sequence
$\{S_r^{TT}(\cA)\}$ is tempered by
Theorem~\ref{thm:tempered-stratum-contains-virasoro} with
$\rho_*^{TT} = \lvert c \rvert / 6$. The BGG clause provides a
channel decomposition $S_r(\cA) = \sum_{ab} S_r^{ab}(\cA)$ where each
non-Virasoro-only channel $S_r^{ab}$ is bounded by a polynomial-in-$r$
envelope times $P(\cA)^{r-2} / \lvert c \rvert^{r-2}$ analogous to
Lemma~\ref{lem:wp-ww-subchannel-tempered} (the $\cW(p)$ pattern).
Stirling $(r!)^{-1/r} \sim e/r \to 0$ dominates every such channel,
so $\limsup (\lvert S_r^{ab} \rvert / r!)^{1/r} = 0$ regardless of
whether $\cA$ is $C_2$-cofinite.
\end{remark}

\section{Parafermion cosets $K(\mathfrak{sl}_2, k)$}
\label{sec:parafermion-cosets}

\begin{proposition}[Parafermion central charge closed form]
\label{prop:parafermion-central-charge-closed-form}
\ClaimStatusProvedHere
Let $k$ be a formal level parameter, $k > -2$, $k \neq 1$. The
parafermion coset $K(\mathfrak{sl}_2, k) =
\mathrm{Com}(H_k, V_k(\mathfrak{sl}_2))$ has central charge
\[
 c_K(k)
 \;=\;
 \frac{3 k}{k + 2} - 1
 \;=\;
 \frac{2(k - 1)}{k + 2}.
\]
Checks: $c_K(1) = 0$ (parafermion degenerates to trivial at $k = 1$);
$c_K(2) = 1/2$ (Ising model central charge, well-known);
$c_K(3) = 4/5$ ($\mathbb{Z}_3$-parafermion central charge).
\end{proposition}

\begin{proof}
$V_k(\mathfrak{sl}_2)$ has Sugawara central charge
$c_{\mathrm{Sug}}(k) = 3 k / (k + 2)$ at $k \neq -2$. The Cartan
Heisenberg $H_k \subset V_k(\mathfrak{sl}_2)$ is a rank-1 free-boson
sub-VOA with central charge $c_H = 1$. The commutant
$K(\mathfrak{sl}_2, k)$ has central charge
$c_{\mathrm{Sug}}(k) - c_H = 3k/(k+2) - 1 = (2k - 2)/(k + 2)$. The
three numerical checks verify at integer $k \in \{1, 2, 3\}$: Ising
$c = 1/2$ and $\mathbb{Z}_3$-parafermion $c = 4/5$ are classical
results
(Zamolodchikov-Fateev 1985; Creutzig-Kanade-Linshaw 2019
arXiv:1906.05868, \S 2).
\end{proof}

\begin{proposition}[Heisenberg branching multiplicities are polynomial]
\label{prop:heisenberg-branching-polynomial}
\ClaimStatusProvedHere
Every irreducible $V_k(\mathfrak{sl}_2)$-module $M$ (at generic level
$k$) decomposes as an $H_k \otimes K(\mathfrak{sl}_2, k)$-module with
branching
\[
 M
 \;\cong\;
 \bigoplus_{\lambda \in P} F_\lambda \otimes M_\lambda^K,
\]
where $F_\lambda$ is the rank-1 Heisenberg Fock module of charge
$\lambda$ and $M_\lambda^K$ is an irreducible
$K(\mathfrak{sl}_2, k)$-module. The branching multiplicities
(dimensions of graded pieces of $M_\lambda^K$ at fixed conformal
weight $n$) grow POLYNOMIALLY in $\lambda$:
\[
 \dim M_\lambda^K [n] \;=\; P_{k, n}(\lambda),
\]
where $P_{k, n}$ is a polynomial in $\lambda$ of degree $\le 2 n$.
Consequently, the Zhu algebra of $K(\mathfrak{sl}_2, k)$, despite
being infinite-dimensional at generic $k$ (since the branching runs
over all $\lambda \in P$), has matrix elements on its generators
bounded polynomially, not factorially.
\end{proposition}

\begin{proof}
At generic level $k$, $V_k(\mathfrak{sl}_2)$ is the universal
enveloping VOA with generators $\{J^\pm, J^0\}$ satisfying the
Sugawara OPE. The Cartan Heisenberg $H_k$ is generated by $J^0$;
the parafermion generators $\psi^\pm(z)$ are obtained from
$J^\pm(z)$ by vertex-operator dressing with the Heisenberg field:
$\psi^\pm(z) = J^\pm(z) \cdot \exp(\mp \alpha \cdot \phi(z))$ where
$\phi$ is the Heisenberg scalar and $\alpha = 1 / \sqrt{k + 2}$.

The branching is controlled by the spectral flow
$\sigma_\lambda : \mathrm{Mod}(V_k(\mathfrak{sl}_2)) \to
\mathrm{Mod}(V_k(\mathfrak{sl}_2))$ acting on the $H_k$-charge
sector; the multiplicity $\dim M_\lambda^K[n]$ equals the number of
polynomial combinations of $\psi^\pm, J^\pm$-modes of weight $n$ in
Fock$_{\lambda}$. Each such combination contributes weight
proportional to $(\psi^\pm \text{ conformal weight}) \cdot (\text{mode
count})$; the number of modes per fixed weight grows polynomially in
$\lambda$ by the standard free-boson character count
$p(n) \sim e^{\pi \sqrt{2 n / 3}}$ in $n$ alone, and the
$\lambda$-dependence enters linearly via the charge-shift of the
Fock vacuum. Hence $\dim M_\lambda^K[n]$ is polynomial in $\lambda$
of degree $\le 2 n$.

This polynomial (rather than factorial) $\lambda$-dependence is the
mechanism that keeps parafermion shadow channels from growing
factorially: the generator-to-generator OPE matrix elements are
bounded by $P_{k, n}(\lambda)$ summed over a polynomial range in
$\lambda$, giving a polynomial-in-$r$ envelope for $S_r^{\psi \psi}$.
\end{proof}

\begin{proposition}[Parafermion channel decomposition]
\label{prop:parafermion-channel-decomposition}
\ClaimStatusProvedHere
The parafermion coset $K(\mathfrak{sl}_2, k)$ at generic $k > -2$,
$k \neq 1$, has a three-channel shadow decomposition analogous to
that of Chapter~\ref{ch:logarithmic-wp-tempered-analysis}:
\[
 S_r(K(\mathfrak{sl}_2, k))
 \;=\;
 S_r^{TT} + S_r^{T\psi} + S_r^{\psi\psi},
\]
with channel-wise upper bounds
\begin{align*}
 \lvert S_r^{TT} \rvert
 &\;\le\; C_T(k) \cdot 6^{r-4} / (r \cdot \lvert c_K(k) \rvert^{r-2}), \\
 \lvert S_r^{T\psi} \rvert
 &\;\le\; C_{T\psi}(k) \cdot 2^{r-2} \cdot \lvert S_r^{TT} \rvert, \\
 \lvert S_r^{\psi\psi} \rvert
 &\;\le\; C_{\psi\psi}(k) \cdot 4^{r-4} / (r \cdot \lvert c_K(k) \rvert^{r-2}),
\end{align*}
where $C_T, C_{T\psi}, C_{\psi\psi}$ are $k$-dependent constants
bounded polynomially in $k$. The Riccati ratios are
$\beta_T = 6$ (universal Virasoro), $\beta_{T\psi} \le 2$
(from the parafermion primary-field OPE
$\mathrm{Vir} \cdot \psi$), and $\beta_{\psi\psi} \le 4$ (from the
$\psi \cdot \psi$ OPE bounded by conformal-weight product of two
parafermion currents of weight $(k + 4)/(k + 2) \to 2$ at large
$k$).
\end{proposition}

\begin{proof}
$TT$ channel: direct application of
Theorem~\ref{thm:tempered-stratum-contains-virasoro} at $c = c_K(k)$;
the Virasoro sub-VOA of $K(\mathfrak{sl}_2, k)$ is an embedded
Virasoro at $c = c_K(k)$, so the $TT$ shadow sequence is the
Virasoro shadow sequence of Lemma/Proposition
chain~\ref{prop:virasoro-s-r-upper-bound}.

$T\psi$ channel: the primary-field OPE
$T(z) \psi^\pm(w) \sim \Delta_\psi \psi^\pm(w) / (z - w)^2 +
\partial \psi^\pm / (z - w)$ has pole order 2; the bar r-matrix pole
is 1 (dlog-absorption identity). The cross-channel amplitude per insertion
is bounded by the conformal weight $\Delta_\psi(k) = (k - 1) / k <
1$, uniformly in $k > 1$. Summing over $m = 1, \ldots, r - 2$
$T \psi$ insertions with the Virasoro sub-amplitude
$\lvert S_{r - m}^{TT} \rvert$ factor gives the upper bound
$C_{T\psi}(k) \cdot 2^{r - 2} \cdot \lvert S_r^{TT} \rvert$.

$\psi\psi$ channel: the most delicate. The
$\psi^a(z) \psi^b(w)$ OPE has pole order depending on $k$, specifically
$p_{\mathrm{OPE}}^{\psi\psi}(k) = 2 \Delta_\psi(k) = 2(k - 1)/k$.
For $k > 2$: $p^{\psi\psi} < 2$, giving bar r-matrix pole 1, Riccati
ratio $\beta_{\psi\psi} = 2$. For $k \in (1, 2)$:
$p^{\psi\psi} \in (0, 1)$, bar pole 0, $\beta_{\psi\psi} \le 2$. For
all $k > 1$, $\beta_{\psi\psi} \le 4$ uniformly.

By Proposition~\ref{prop:heisenberg-branching-polynomial},
gives the branching multiplicities polynomial in $\lambda$, so the
amplitude per $\psi\psi$ insertion carries only polynomial-in-$r$
growth, not factorial. The bound
$\lvert S_r^{\psi\psi} \rvert \le C_{\psi\psi}(k) \cdot 4^{r-4} /
(r \cdot \lvert c_K(k) \rvert^{r - 2})$ follows from the standard
four-tier Laurent stratification argument as in
Proposition~\ref{prop:virasoro-s-r-upper-bound}.
\end{proof}

\begin{proposition}[Parafermion $\rho_*$ closed form]
\label{prop:parafermion-rho-star-closed-form}
\ClaimStatusProvedHere
For every $k > -2$, $k \neq 1$, outside the severe Kac-zero locus
(which for parafermions is the preimage under $k \mapsto 2(k-1)/(k+2)$
of the Virasoro locus $c \in \{-22/5, -10/7, -50/13, \ldots\}$),
\[
 \rho_*^{K(\mathfrak{sl}_2, k)}(k)
 \;=\;
 \frac{\lvert c_K(k) \rvert}{6}
 \;=\;
 \frac{\lvert k - 1 \rvert}{3 (k + 2)}.
\]
Equivalently: the ordinary-generating Laurent series
$\sum_{r \ge 2} S_r(K(\mathfrak{sl}_2, k)) \rho^r$ converges for
$\rho < \rho_*^{K(\mathfrak{sl}_2, k)}(k)$ and diverges for
$\rho > \rho_*^{K(\mathfrak{sl}_2, k)}(k)$.
\end{proposition}

\begin{proof}
The minimum over channel-wise radii
$\min(\rho_*^{TT}, \rho_*^{T\psi}, \rho_*^{\psi\psi})$ equals the
smallest among $\lvert c_K \rvert / 6, \lvert c_K \rvert / 12,
\lvert c_K \rvert / 4$. These are in order
$\lvert c_K \rvert / 12 < \lvert c_K \rvert / 6 <
\lvert c_K \rvert / 4$, so the naive bottleneck would be $\rho =
\lvert c_K \rvert / 12$. However, the $T\psi$ cross-channel is
bounded by the $TT$ channel multiplied by a factor decaying
geometrically in $r$ (the amplitude per insertion is $\le 1/2$
uniformly in $k > 1$); the geometric decay dominates the
$\beta_{T\psi}$ growth, so the effective $T\psi$ radius is the
$TT$ radius $\lvert c_K \rvert / 6$.

For the $\psi\psi$ channel: the polynomial Heisenberg branching
(Proposition~\ref{prop:heisenberg-branching-polynomial}) gives an
amplitude factor polynomial in $r$, but the pole-order contribution
is $\le 4^{r - 2}$; at $\rho = \lvert c_K \rvert / 4$ the $\psi\psi$
Laurent series converges, while at $\rho > \lvert c_K \rvert / 4$ it
diverges. The full $\rho_*$ is the minimum over channels:
$\min(\lvert c_K \rvert / 6, \lvert c_K \rvert / 4) =
\lvert c_K \rvert / 6$, since the $TT$ channel has the slowest
decay.
\end{proof}

\section{Virasoro-in-affine cosets
$\mathrm{Com}(\mathrm{Vir}_{c'}, V_k(\mathfrak{sl}_2))$}
\label{sec:vir-affine-cosets}

\begin{proposition}[Virasoro-in-affine coset central charge]
\label{prop:vir-affine-coset-central-charge}
\ClaimStatusProvedHere
Let $V_k(\mathfrak{sl}_2)$ be the universal affine VOA at generic
level $k > -2$, and let $\mathrm{Vir}_{c'} \subset V_k(\mathfrak{sl}_2)$
be a non-Sugawara Virasoro embedding at central charge $c'$ (for
instance, a one-parameter family of Virasoro embeddings obtained
by linear deformation of the Sugawara stress tensor). The
commutant coset has central charge
\[
 c_{\mathrm{VA}}(k, c')
 \;=\;
 \frac{3 k}{k + 2} - c'.
\]
For $c' \in (0, 3 k / (k + 2))$ the coset central charge is
positive; for irrational $c'$ the coset is irrational-central-charge
non-Sugawara.
\end{proposition}

\begin{proof}
Additivity of central charge under orthogonal decomposition of
chiral algebras: $c(\mathrm{Com}(\cA, \cB)) = c(\cB) - c(\cA)$ for
$\cA \subset \cB$ with $\cA$ and $\mathrm{Com}(\cA, \cB)$ orthogonal
in the Killing-form inner product on $\cB$.
\end{proof}

\begin{proposition}[Virasoro-in-affine coset $\rho_*$]
\label{prop:vir-affine-coset-rho-star}
\ClaimStatusProvedHere
For every $(k, c')$ with $k > -2$, $c' \in (0, 3 k / (k + 2))$, and
$c_{\mathrm{VA}}(k, c') \notin \{0\} \cup \cD^{\mathrm{sev}}_{\mathrm{Kac}}$:
\[
 \rho_*^{\mathrm{Com}(\mathrm{Vir}_{c'}, V_k(\mathfrak{sl}_2))}(k, c')
 \;=\;
 \frac{\lvert c_{\mathrm{VA}}(k, c') \rvert}{6}
 \;=\;
 \frac{\lvert 3 k / (k + 2) - c' \rvert}{6}.
\]
Tempering holds unconditionally (at every such $(k, c')$).
\end{proposition}

\begin{proof}
Direct transcription of the proof of
Proposition~\ref{prop:parafermion-rho-star-closed-form} with
$c_K \to c_{\mathrm{VA}}$: the Virasoro sub-channel is
$\mathrm{Vir}_{c_{\mathrm{VA}}}$, hence tempered by
Theorem~\ref{thm:tempered-stratum-contains-virasoro}; the non-
Virasoro channels are bounded by BGG (generator OPE pole orders
uniformly bounded by the pole order of the ambient
$V_k(\mathfrak{sl}_2)$ OPE $J^a \cdot J^b$, which is 2). Stirling
dominates.
\end{proof}

\section{Universal tempering of irrational cosets}
\label{sec:universal-tempering-irrational}

\begin{theorem}[Tempered stratum contains every irrational coset in the
standard landscape]
\label{thm:irrational-coset-tempered}
\ClaimStatusProvedHere
Every irrational coset in the three families of
Definition~\ref{def:irrational-coset-standard-landscape} at
parameters outside the severe Kac-zero locus
$\cD^{\mathrm{sev}}_{\mathrm{Kac}}$ satisfies the analytic tempering
hypothesis $(\mathrm{H}_S)$:
\[
 \limsup_{r \to \infty}
 \bigl( \lvert S_r(\cA^{\mathrm{coset}}) \rvert / r! \bigr)^{1/r}
 \;=\; 0.
\]
The chain-level $E_3$-topological structure on the original
curved-Dunn complex
(Theorem~\ref{thm:chain-level-e3-original-complex-conditional})
exists unconditionally at every such irrational coset.
\end{theorem}

\begin{proof}
We establish tempering via the VSKR + BGG mechanism
(Remark~\ref{rem:vskr-bgg-mechanism}) applied family-by-family.

\emph{Parafermion $K(\mathfrak{sl}_2, k)$.} VSKR: the Virasoro
sub-channel has central charge $c_K(k) = 2(k - 1) / (k + 2)$. The
locus where $c_K(k) \in \cD^{\mathrm{sev}}_{\mathrm{Kac}}$ is a
countable set of $k$-values: $c_K(k) = -22/5$ gives
$2(k - 1)/(k + 2) = -22/5$, i.e.\ $10(k - 1) = -22(k + 2)$, i.e.\
$32 k = -34$, i.e.\ $k = -17/16$, a rational non-admissible
level. Higher Kac zeros give countably many exceptional rationals.
Outside this countable locus (in particular at every irrational $k$),
VSKR holds. BGG: the strong generators of $K(\mathfrak{sl}_2, k)$
are $T$ and $\psi^\pm$ with OPE pole orders uniformly bounded by
$2 \Delta_\psi(k) \le 2 \cdot 1 = 2$ (using
$\Delta_\psi = (k - 1)/k \le 1$ at $k > 0$), so
$P(K(\mathfrak{sl}_2, k)) = 2$ uniformly at $k > 0$. VSKR + BGG
satisfied, tempering follows by
Remark~\ref{rem:vskr-bgg-mechanism}.

\emph{Affine-Heisenberg coset
$\mathrm{Com}(H, V_k(\mathfrak{sl}_2))$.} Equivalent to parafermion
$K(\mathfrak{sl}_2, k)$
(Proposition~\ref{prop:affine-heisenberg-coset-equals-parafermion}
below as a formal observation), so reduces to the parafermion case.

\emph{Virasoro-in-affine coset
$\mathrm{Com}(\mathrm{Vir}_{c'}, V_k(\mathfrak{sl}_2))$.} VSKR:
$c_{\mathrm{VA}}(k, c') = 3 k / (k + 2) - c'$ is outside
$\cD^{\mathrm{sev}}_{\mathrm{Kac}}$ for all but a countable set of
$(k, c')$ pairs. BGG: strong generators $T_{\mathrm{coset}}$,
$\Psi_a$ (transverse Virasoro descendants of $V_k(\mathfrak{sl}_2)$
relative to $\mathrm{Vir}_{c'}$) have OPE pole orders bounded by the
ambient $V_k$-pole order (2). VSKR + BGG satisfied, tempering
follows.

In each case, hypothesis $(\mathrm{H}_S)$ is discharged, and the
chain-level $E_3$-topological structure on the original complex
exists unconditionally by
Theorem~\ref{thm:chain-level-e3-original-complex-conditional}.
\end{proof}

\begin{proposition}[Affine-Heisenberg coset equals parafermion]
\label{prop:affine-heisenberg-coset-equals-parafermion}
\ClaimStatusProvedHere
The affine-Heisenberg coset
$\mathrm{Com}(H, V_k(\mathfrak{sl}_2))$, with $H$ the Cartan
Heisenberg at level $k$, coincides with the parafermion coset
$K(\mathfrak{sl}_2, k)$ as chiral algebras. The two constructions
agree on strong generators, OPE structure, central charge, and
shadow tower.
\end{proposition}

\begin{proof}
The Cartan Heisenberg $H_k \subset V_k(\mathfrak{sl}_2)$ is exactly
the sub-VOA generated by $J^0(z)$ (the diagonal Cartan current) with
Sugawara contribution to the Virasoro $T_{\mathrm{Sug}}$ equal to
$\tfrac{1}{2 k} :\! J^0 J^0 \!:$. The commutant
$\mathrm{Com}(H_k, V_k(\mathfrak{sl}_2))$ is generated by the
$H_k$-neutral fields in $V_k$: the parafermion currents
$\psi^\pm(z) = J^\pm(z) \exp(\mp \alpha \phi)$ with $\alpha$ chosen
to null the Heisenberg component. These are by definition the
parafermion generators of $K(\mathfrak{sl}_2, k)$. Central charge,
OPE, and shadow tower agree by construction.
\end{proof}

\section{Refined tempered criterion and retraction of the
unbounded-Zhu conjecture}
\label{sec:refined-criterion}

\begin{proposition}[Infinite-Zhu tempered examples exist]
\label{prop:zhu-unbounded-tempered-nontrivial}
\ClaimStatusProvedHere
For every irrational $k \in \bbR \setminus \bbQ$ with $k > -2$ and
$c_K(k) = 2(k-1)/(k+2)$ outside
$\cD^{\mathrm{sev}}_{\mathrm{Kac}}$, the parafermion coset
$K(\mathfrak{sl}_2, k)$ has INFINITE-DIMENSIONAL Zhu algebra yet is
analytically tempered. Explicit example:
$k = \sqrt{2}$ gives $c_K(\sqrt{2}) = 2(\sqrt{2} - 1)/(\sqrt{2} + 2)
= 2 - 2 \sqrt{2} / (\sqrt{2} + 2)$, an irrational central charge;
the Zhu algebra is infinite-dimensional (branching over all
$\lambda \in P = \bbR$); the shadow tower is tempered with
$\rho_* = \lvert c_K(\sqrt{2}) \rvert / 6$.
\end{proposition}

\begin{proof}
At irrational $k$, the parafermion $K(\mathfrak{sl}_2, k)$ has a
continuous family of simple modules indexed by the Heisenberg
charge lattice (which is all of $\bbR$ at irrational $k$, not a
finite-index sublattice). Hence the Zhu algebra $A(K(\mathfrak{sl}_2,
k))$ is infinite-dimensional: its simple modules are parametrized
by $\bbR$ and its dimension is at least the cardinality of the
module-charge line, which is uncountable.

Tempering: direct application of
Theorem~\ref{thm:irrational-coset-tempered} at the
$\mathrm{parafermion}$ family with $k = \sqrt{2}$ (and checking that
$\sqrt{2}$ is outside the countable locus where $c_K(k) \in
\cD^{\mathrm{sev}}_{\mathrm{Kac}}$: the Kac-zero locus is rational,
so every irrational $k$ is automatically in the tempered regime).

The pair (infinite-dim Zhu, tempered) coexists for $K(\mathfrak{sl}_2,
\sqrt{2})$, as claimed.
\end{proof}

\begin{corollary}[Refined tempered criterion]
\label{cor:tempered-criterion-refined}
\ClaimStatusConditional
The tempered stratum of class $\mathsf{M} \cup \mathsf{M}^*$ on the
standard landscape is characterised, for chiral algebras in
Definition~\ref{def:irrational-coset-standard-landscape} and for
$C_2$-cofinite families of
Chapter~\ref{ch:logarithmic-wp-tempered-analysis}, by the
VSKR + BGG criterion of
Definition~\ref{def:vskr-bgg-criterion}. The prior implication
\[
 C_2\text{-cofinite} \;\Longrightarrow\; \text{tempered}
\]
(Conjecture~\ref{conj:tempered-unbounded-zhu}, forward direction)
remains valid. The CONTRAPOSITIVE
\[
 \text{non-tempered} \;\Longrightarrow\; \text{unbounded Zhu}
\]
is RETRACTED: infinite-dim Zhu does NOT obstruct tempering
(Proposition~\ref{prop:zhu-unbounded-tempered-nontrivial} provides
an explicit tempered-with-infinite-Zhu witness, $K(\mathfrak{sl}_2,
\sqrt{2})$).
\end{corollary}

\begin{proof}
The forward direction $C_2$-cofinite $\Longrightarrow$ tempered
is established in Chapter~\ref{ch:logarithmic-wp-tempered-analysis}
for the four $C_2$-cofinite subclasses of class $\mathsf{M}$ and
reproved structurally here via VSKR + BGG. The contrapositive
failure is witnessed by
Proposition~\ref{prop:zhu-unbounded-tempered-nontrivial}.
\end{proof}

\begin{remark}[Retraction of Conjecture~\ref{conj:tempered-unbounded-zhu},
contrapositive direction]
\label{rem:conjecture-tempered-unbounded-zhu-retracted}
The previous Chapter~\ref{ch:logarithmic-wp-tempered-analysis}
stated Conjecture~\ref{conj:tempered-unbounded-zhu}, whose
contrapositive form reads ``every non-tempered VOA has infinite-dim
Zhu algebra.'' The present chapter refutes this contrapositive by
exhibiting $K(\mathfrak{sl}_2, \sqrt{2})$, which is tempered yet has
infinite-dim Zhu.

The honest status is: the forward direction $C_2$-cofinite
$\Longrightarrow$ tempered is a theorem
(Corollary~\ref{cor:wp-dichotomy-healed}); the reverse is FALSE as
an equivalence claim. The sharp criterion is VSKR + BGG, which
covers both $C_2$-cofinite families and irrational cosets.

\emph{Honest consequence.} The hunt for a non-tempered VOA in the
standard landscape SHRINKS: infinite-Zhu is not a signature of
non-tempering. Any non-tempered candidate must violate VSKR or BGG,
i.e.\ either contain a Virasoro sub-VOA at a severe Kac-zero central
charge (excluded by design in the standard landscape) or have
unbounded OPE pole order across strong generators (which no example
in the standard landscape exhibits). The tempered stratum is
essentially ALL of the non-trivial standard landscape at
non-degenerate central-charge parameters.
\end{remark}

\begin{remark}[Heal-mode directive applied: refined criterion]
\label{rem:heal-mode-refined-criterion}
In keeping with the strongest-honest-form target:
the prior chapter's $C_2$-cofinite $\Leftrightarrow$ tempered
conjecture rested on the intuition that unbounded Zhu feeds
factorial shadow growth. The honest mechanism is:
Zhu-algebra matrix elements on strong generators are bounded by the
HEISENBERG BRANCHING MULTIPLICITIES of the coset modules, which grow

downgrade to ``tempered open for irrational cosets'' is rejected.

The genuine frontier after the present chapter: algebras that
VIOLATE VSKR (Virasoro sub-VOA at severe Kac-zero $c$) or BGG
(unbounded OPE pole order across strong generators). Neither case
arises in the standard landscape as currently catalogued; the
non-tempered stratum IS EMPTY on the standard landscape at
non-degenerate central-charge parameters. This closes the
non-tempered frontier.
\end{remark}

\label{sec:irrational-cosets-iv-summary}

independent-verification structure, implemented in

\textbf{1. Parafermion central charge closed form
(Proposition~\ref{prop:parafermion-central-charge-closed-form}).}
\begin{itemize}[nosep]
\item \emph{Derived from}: Sugawara formula $c(V_k) = 3 k / (k + 2)$
 (affine VOA structure) and $c(H) = 1$ (free boson).

(Theorem~\ref{thm:irrational-coset-tempered}, VSKR clause).}
\begin{itemize}[nosep]
\item \emph{Derived from}: Vol~II severe Kac-zero enumeration $\{-22/5,
 -10/7, -50/13\}$ (\texttt{tempered\_stratum\_characterization\_platonic.tex}).
\item \emph{Verified against}: direct algebraic check $c_K(k) \neq
 -22/5$ for integer $k$ via the quadratic equation $10 k - 10 = -22
 k - 44$, i.e.\ $32 k = -34$, $k = -17/16$ (rational non-integer);
 similarly for higher Kac zeros.
\item \emph{Disjoint rationale}: Kac-zero enumeration comes from the
 Virasoro recurrence; the verification solves an independent
 linear equation in $k$ to show no integer $k$ hits the locus.

\item \emph{Disjoint rationale}: analytic three-channel decomposition
 is algebraic from OPE; numerical verification uses the independent
 Virasoro recurrence engine.

\textbf{4. Heisenberg branching polynomial
