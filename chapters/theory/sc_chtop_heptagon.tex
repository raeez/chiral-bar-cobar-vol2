\chapter{The Heptagon of Presentations of \texorpdfstring{$\SCchtop$}{SC}}
\label{ch:sc-chtop-heptagon}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\section*{Prologue}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

One operad; seven faces. Each face is a geometric, algebraic, or
categorical presentation, and each pair of faces is connected by a named
equivalence of $\infty$-operads, $\infty$-quasi-isomorphic after
fixing a Drinfeld associator $\Phi\in\mathrm{GRT}_1(\mathbb{Q})$.
Five faces are classical: the operadic face
(Chapter~\ref{ch:factorization-swiss-cheese}), its Koszul dual
(Chapter~\ref{ch:theory-equivalence}), and the factorization,
BV/BRST, and convolution faces. Two further faces present themselves:
the Drinfeld-centre face through the factorization-module category
$\Rep_{\mathrm{fact}}(\cA)$, and the derived-algebraic-geometry face
through global sections of a stack classifying $E_1$-chiral algebras
equipped with $\SCchtop$-datum.
Adjacent edges of the heptagon are named theorems; the diagonal edges
are consequences of the seven adjacencies.

The reconstitution is rigid: the $\infty$-operadic structure is
determined by any one face, and the seven faces exhaust the
presentations currently known. The absence of a conformal vector
freezes the heptagon at $\SCchtop$; its presence triggers
topologisation to $E_3$-topological (Chapter~\ref{ch:topologization}).
The heptagon is the generic object; topologisation is the exceptional
collapse.

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\section{Setting: the object of the heptagon}
\label{sec:heptagon-setting}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

Throughout this chapter $X$ denotes a smooth complex curve and $\cA$
denotes an $E_1$-chiral algebra on $X$. We write $Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)$
for the derived chiral centre $C^{\bullet}_{\mathrm{ch}}(\cA,\cA)$
computed against the algebraic chiral endomorphism operad
$\End^{\mathrm{ch}}_\cA$ (Chapter~\ref{ch:theory-hochschild}). The pair
\begin{equation}
  \label{eq:heptagon-pair}
  \bigl(Z^{\mathrm{der}}_{\mathrm{ch}}(\cA),\;\cA\bigr)
\end{equation}
is the \emph{$\SCchtop$-datum} of $\cA$: bulk acts on boundary, the
boundary carries no action on the bulk.

\begin{convention}[What the heptagon presents]\label{conv:heptagon-object}
All seven faces present the same two-coloured coloured operad with
directional restriction $\SCchtop$ (a coloured dioperad in the sense
of Gan: operations are coloured by closed/open, closed-to-open
compositions exist, open-to-closed compositions are empty)
of homotopy type
\begin{equation}
  \SCchtop(k,m)
  \;\simeq\;
  C_{\bullet}\bigl(\FM_k(\mathbb{C})\bigr)
  \,\otimes\,
  C_{\bullet}\bigl(\Conf_m(\mathbb{R})\bigr)
\end{equation}
on the open strata, with two boundary colours
(closed $=\FM_k(\mathbb{C})$, open $=\Conf_m(\mathbb{R})$) related by
codimension-one mixed stratum $\FM_k(\mathbb{C})\times\Conf_m(\mathbb{R})$,
and no open-to-closed operations. Because of the directional
restriction, the Koszul dual is not the operad itself:
$(\SCchtop)^{!}=(\Lie,\Ass,\text{shuffle-mixed})$ is closed-sector
$\Lie$, open-sector $\Ass$, shuffle-mixed cross-term; its closed
arity dimensions are $(n-1)!$ whereas $\SCchtop$'s closed arity
dimensions are $1$, so the operads are genuinely distinct.
\end{convention}

\begin{remark}[Directional asymmetry]\label{rem:heptagon-asymmetry}
Every face records the directional bulk-to-boundary composition;
no face admits a natural open-to-closed operation. The asymmetry forces
the Koszul dual to be non-self, forces the factorization face to act
through module categories rather than algebras, and forces the
Drinfeld-centre face to be a \emph{right adjoint} (centre), not an
averaging.
\end{remark}

\begin{remark}[The two colours of $\SCchtop$ as the two stages
of the CY-to-chiral functor at $d = 2, 3$;
\ClaimStatusProvedElsewhere]
\label{rem:heptagon-two-stage-CY-to-chiral}
\index{Swiss-cheese chiral-topological operad!closed/open as stage-1/stage-2}%
\index{CY-to-chiral functor!two-stage factorisation}%
The CY-to-chiral functor of Volume~III factorises
\[
 \Phi_d \;=\; \SpCh_{\Sigma_{d-1},\,C}\circ\PhiFA_d,
\]
with stage~1 ($\PhiFA_d$) producing a canonical $E_d$-homotopy
factorisation algebra on the CY-$d$ target by
Kontsevich--Tamarkin formality combined with
Costello--Gwilliam--Li factorisation-homology BV quantisation, and
stage~2 ($\SpCh_{\Sigma_{d-1},C}$) specialising to a reference
curve $C$ via factorisation-homology pushforward
$\int_{\Sigma_{d-1}}$ along an auxiliary $(d-1)$-manifold.
At $d = 2, 3$, the two colours of the coloured operad $\SCchtop$
structurally \emph{are} the two stages:
\begin{enumerate}[label=\textup{(\roman*)}]
\item The \emph{closed colour} — the holomorphic/braided sector
with arity components $\FM_k(\C)$, homotopy $E_2$ at the chain
level — is the restriction of stage~1 to local observables of a
$E_d$-holomorphic factorisation algebra on a formal disc in the
CY target. At $d = 2$ this is $\PhiFA_2(\cC)$ evaluated on
$\mathrm{Spf}\,\C[\![z]\!]$; at $d = 3$ it is the restriction of
$\PhiFA_3(\cC)$ to a transverse disc after the holomorphic
reduction on the $\C^2$-factor of the stage-1 factorisation
algebra (Costello--Gwilliam Vol.~I~Theorem~3.4.9).
\item The \emph{open colour} — the topological/ordered sector
with arity components $\Conf_m(\R)$, $E_1$ on $\R$ by
Ayala--Francis~\cite{AF15} — is the stage-2 shadow
$\SpCh_{\Sigma_{d-1},C}\circ\PhiFA_d(\cC)$ landing on the
reference curve $C$. It is $E_1$-chiral, not $E_d$, because
factorisation-homology pushforward along $\Sigma_{d-1}$
collapses $d - 1$ topological directions, leaving a
single holomorphic direction along $C$ and a single
topological direction along $\R\subset C\times\R$.
\item The \emph{mixed operations} — the closed colour acting on
the open colour via the codimension-one stratum
$\FM_k(\C)\times\Conf_m(\R)$ — record the factorisation-homology
pushforward $\int_{\Sigma_{d-1}}$ as a bulk-to-boundary map. The
directional restriction $\SCchtop(\ldots,\mathsf{top},\ldots;
\mathsf{cl})=\varnothing$ (no open-to-closed) is the precise
operadic expression of the fact that stage~2 is a specialisation of
stage~1, not an inversion: the $E_d$-structure on the CY target
determines the $E_1$-structure on $C$, while the converse map
from a single $E_1$-chiral algebra to an $E_d$-factorisation
algebra does not exist canonically (a CY-$d$ category admits a
\emph{family} of $E_1$-chiral shadows parametrised by the choice
of $\Sigma_{d-1}$ and $C$).
\end{enumerate}
At $d = 2$ the two stages coincide with the two colours of the
coloured operad on the nose: $\Sigma_{d-1} = \Sigma_1 = \R$ and
the stage-2 pushforward $\int_\R$ is the topological factorisation
on the boundary. At $d = 3$ the auxiliary manifold is
$\Sigma_2$ and the stage-2 reference curve is a transverse holomorphic
curve in the CY-$3$, making the open colour the ordered $E_1$-chiral
algebra on that reference curve (see
Theorem~\ref{thm:BD-CG-equivalence} for the three-layer comparison
and Remark~\ref{rem:sc-chtop-pentagon} for the pentagon extension).
The heptagon faces are thus seven presentations of the
\emph{composite} $\Phi_d$; the pentagon-trace-identity
(Remark~\ref{rem:pentagon-trace-identity}) realises the closed/open
projection as the Volume~I ghost-identity / Volume~III BKM-weight
decomposition of the two stages. Vol.~III,
\texttt{cy\_to\_chiral.tex} fixes the scope and proves the functoriality
of the factorisation; the present heptagon is its coloured
bar-cobar-Koszul completion.
\end{remark}

\section{The five classical faces}
\label{sec:five-classical-faces}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

We restate the pentagon vertices as inputs to the heptagon, fixing
notation for edges in \S\ref{sec:heptagon-edges}.

\subsection{Face (1): the operadic face}\label{subsec:face-operadic}

\begin{definition}[Coloured Swiss-cheese chiral-topological operad
$\SCchtop$; \ClaimStatusProvedElsewhere]\label{def:factorization-swiss-cheese-operad}%
\index{Swiss-cheese chiral-topological operad!definition|textbf}
The \emph{coloured Swiss-cheese chiral-topological operad} $\SCchtop$
is the two-coloured topological operad with colours $\{\mathsf{cl},
\mathsf{op}\}$ whose component spaces are
\begin{equation}\label{eq:scchtop-spaces}
  \SCchtop\bigl(\tfrac{k\,\mathsf{cl},\,m\,\mathsf{op}}{\mathsf{cl}}\bigr)
  \;=\;\emptyset\quad(m\geq 1),\qquad
  \SCchtop\bigl(\tfrac{k\,\mathsf{cl},\,m\,\mathsf{op}}{\mathsf{op}}\bigr)
  \;=\;\FM_k(\mathbb{H})\!\times_{\Conf_m(\mathbb{R})}\!\Conf_m(\mathbb{R})
\end{equation}
where $\mathbb{H}\subset\mathbb{C}$ is the closed upper half-plane,
$\FM_k(\mathbb{H})$ is the Voronov half-plane variant of the
Fulton--MacPherson compactification of $\Conf_k(\mathbb{H}^\circ)$
(Voronov\,\cite{Voronov1998}), and the fibre product enforces that the
$m$ open points lie on the boundary $\partial\mathbb{H}=\mathbb{R}$.
Composition is concatenation/insertion of configurations:
closed-into-closed via $\FM_k(\mathbb{C})$ insertion, open-into-open
via $\Conf_m(\mathbb{R})$ insertion, closed-into-open via insertion at
a boundary point. The empty closed-output component
in~\eqref{eq:scchtop-spaces} when $m\geq 1$ encodes the directional
restriction \emph{no open-to-closed operation}: an open input never
contributes to a closed output.

The chain-level operad of singular chains $C_\bullet(\SCchtop)$ has
the homotopy type
\begin{equation}\label{eq:scchtop-arity-decomp}
  C_\bullet\!\bigl(\SCchtop(k,m)\bigr)
  \;\simeq\;
  C_\bullet\bigl(\FM_k(\mathbb{C})\bigr)\otimes
  C_\bullet\bigl(\Conf_m(\mathbb{R})\bigr)
\end{equation}
on the open strata, by the homotopy retraction of $\FM_k(\mathbb{H})$
to $\FM_k(\mathbb{C})$ relative to the boundary configuration of $m$
points on $\mathbb{R}$ (Voronov\,\cite{Voronov1998}\,Lemma~1.4). The
\emph{factorization Swiss-cheese chiral-topological} variant is the
coloured factorization-algebra incarnation: closed colour is a
factorizable $D$-module on $\Ran(X)$ (BD chiral algebra), open colour
is a locally constant factorization algebra on $\Ran(\mathbb{R})$
($\mathrm{E}_1$-algebra by Ayala--Francis\,\cite{AF15}), mixed sector
is a factorization module of the closed colour on $\Ran(X)\times
\Ran(\mathbb{R})$ supported on the boundary stratum
$X\times\{0\}\subset X\times\R_{\geq 0}$.
\end{definition}

\begin{remark}[Pointer to the factorization model]
\label{rem:def-fsc-pointer}
The factorization-algebra incarnation is given in
Definition~\ref{def:BD-SC} of
Chapter~\ref{ch:factorization-swiss-cheese} (BD factorization
Swiss-cheese algebra, three-sector data $(F_{\mathsf{cl}},
F_{\mathsf{op}},F_{\mathsf{mix}})$); the present
Definition~\ref{def:factorization-swiss-cheese-operad} is its
operadic presentation. The two are quasi-isomorphic via
Theorem~\ref{thm:BD-CG-equivalence} of
Chapter~\ref{ch:factorization-swiss-cheese}, see also
Edge~(1)--(2)--(3) in \S\ref{sec:heptagon-edges}.
\end{remark}

Face~(1) is the homotopy colored operad of
Definition~\ref{def:factorization-swiss-cheese-operad} above:
generators are the codimension-one boundary strata of
$\FM_k(\mathbb{H})\times_{\Conf_m(\mathbb{R})}\Conf_m(\mathbb{R})$;
relations are the codimension-two strata (Arnold for the closed
colour, Stasheff for the open colour, and the mixed Stasheff--Arnold
relation from the collision of an open point with the closed
configuration). The presentation is strict in the rational homotopy
category after fixing $\Phi$ (Kontsevich--Tamarkin formality).
The directional restriction (no open-to-closed) is encoded
operadically by the empty component
$\SCchtop\bigl(\binom{k,m}{\text{closed}}\bigr)=\emptyset$ whenever
$m\geq 1$; the coloured dioperad is honest, not a product of two
uncoloured operads.

\subsection{Face (2): the Koszul-dual face}\label{subsec:face-koszul-dual}

Face~(2) is the $\infty$-operadic cofibrant replacement
\begin{equation}
  W(\SCchtop)
  \;:=\;
  \Omega\bigl(B(\SCchtop)\bigr)
\end{equation}
via Ginzburg--Kapranov coloured bar--cobar, after fixing a Drinfeld
associator. Its underlying graded coloured operad is
$(\SCchtop)^{!}=(\Lie,\Ass,\text{shuffle-mixed})$
(Proposition~\ref{prop:sc-koszul-dual-three-sectors} of
Chapter~\ref{ch:theory-drinfeld}).
The closed sector is
$E_2^{!}\simeq E_2\{1\}$ (Getzler--Jones self-duality up to shift),
which on the associated graded collapses to $\Lie$.
The open sector is self-dual $\Ass^{!}=\Ass$. The mixed sector is
shuffle-mixed with empty open-to-closed operations; the
closed-to-open is inherited from $\SCchtop$'s directionality.
The directional restriction is preserved by bar--cobar: the empty
open-to-closed component of $\SCchtop$ produces an empty
closed-to-open component of $(\SCchtop)^!$ after the colour-swap
inside the coloured Koszul dual.

\subsection{Face (3): the factorization face}\label{subsec:face-factorization}

Face~(3) is the factorization-algebra model. For each open $U\subset X$
the chiral observables
\begin{equation}
  \Obs^{\mathrm{ch}}_\cA(U),\qquad U\subset X,
\end{equation}
assemble into a factorization algebra on $\Ran(X)$; its derived
centre $Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)$ acts on $\cA$ through the
universal brace of Chapter~\ref{ch:theory-hochschild}. The pair in
\eqref{eq:heptagon-pair} is the factorization-algebraic
$\SCchtop$-datum. This face is native to Beilinson--Drinfeld
(algebraic) and Costello--Gwilliam (differential-geometric); the
equivalence is Theorem~\ref{thm:BD-CG-equivalence} of
Chapter~\ref{ch:factorization-swiss-cheese}, with precise scope
qualifiers in Remark~\ref{rem:BD-CG-adjectives} (characteristic zero,
$X$ smooth projective, holomorphic translation invariance,
locally constant on $\R$, ind-coherent vs. cosheaf models reconciled).

\subsection{Bar differential as holomorphic factorisation; coproduct as
topological factorisation}\label{subsec:bar-diff-as-hol-fact}

The factorization face above admits an explicit chain-level
identification of the two operations of the bar coalgebra
$B^{\,\mathrm{ch}}(\cA)$ as the two coloured factorisation maps of
$\SCchtop$. The result is new and established below.

\begin{theorem}[Bar differential = holomorphic factorisation,
coproduct = topological factorisation;
\ClaimStatusProvedHere]
\label{thm:bar-diff-eq-holfact}%
\index{bar differential!= holomorphic factorisation|textbf}%
\index{coproduct!= topological factorisation|textbf}%
\index{factorisation maps!chiral identification of}
Let $\cA$ be an $E_1$-chiral algebra on a smooth complex curve $X$
satisfying the hypotheses of Theorem~\ref{thm:BD-CG-equivalence}
(characteristic zero, $X$ smooth, holomorphic translation invariance,
locally constant on $\R$). Write
$B^{\,\mathrm{ch}}(\cA)=T^c(s^{-1}\bar\cA)$ for the chiral bar
coalgebra (Vol~I conventions; cohomological grading, internal differential
$d_B$ from the chiral product, internal coproduct
$\Delta_B\colon B^{\,\mathrm{ch}}(\cA)\to B^{\,\mathrm{ch}}(\cA)\otimes
B^{\,\mathrm{ch}}(\cA)$ from cofree-tensor-deconcatenation). Let
$\Obs^{\mathrm{ch}}_{\cA}$ be the BD factorization algebra on
$\Ran(X)\times\Ran(\R)$ associated to the boundary-condition pair
$(\cA,\cA)$ as in Definition~\ref{def:BD-SC}. Then under the
BD/CG equivalence of Theorem~\ref{thm:BD-CG-equivalence}:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \textbf{Bar differential $\leftrightarrow$ holomorphic
factorisation.} The internal differential $d_B$ on
$B^{\,\mathrm{ch}}(\cA)$ corresponds, on the closed colour, to the
holomorphic factorisation map at coincident points
\begin{equation}\label{eq:hol-fact-map}
  \mu^{\mathrm{hol}}_{\Delta}\colon
  j_*j^*\bigl(\cA^{\boxtimes 2}\bigr)\bigl|_{X^2}
  \;\xrightarrow{\text{coincidence}}\;
  \Delta_*\cA[\,1\,],
\end{equation}
where $j\colon X^2\setminus\Delta\hookrightarrow X^2$ is the open
embedding of the diagonal complement and $\Delta\colon X\hookrightarrow
X^2$ is the diagonal, with the degree shift $[1]$ accounting for the
desuspension $s^{-1}$ in $T^c(s^{-1}\bar\cA)$.

\item \textbf{Coproduct $\leftrightarrow$ topological factorisation.}
The internal coproduct $\Delta_B$ on $B^{\,\mathrm{ch}}(\cA)$
corresponds, on the open colour, to the topological factorisation
map of the locally constant factorization algebra on $\Ran(\R)$,
\begin{equation}\label{eq:top-fact-map}
  \mu^{\mathrm{top}}_{(t_1,t_2)\hookrightarrow t}\colon
  F_{\mathsf{op}}(t_1)\otimes F_{\mathsf{op}}(t_2)
  \;\xrightarrow[\text{topol.}]{\sim}\;
  F_{\mathsf{op}}(t)\quad\text{whenever } t_1<t_2 \text{ both} <t,
\end{equation}
which on $\Ran(\R)$ is determined (Ayala--Francis~\cite{AF15}) by the
underlying $\Eone$-algebra structure on $\cA$ pushed to the boundary
stratum $X\times\{0\}$.
\end{enumerate}
The two identifications are compatible in the sense that
$d_B$ and $\Delta_B$ assemble into a coloured differential graded
$\SCchtop$-coalgebra structure on $B^{\,\mathrm{ch}}(\cA)$, with
\begin{equation}\label{eq:bar-coalg-mc}
  d_B^2 = 0,\qquad
  (d_B\otimes\id+\id\otimes d_B)\circ\Delta_B
  \;=\;
  \Delta_B\circ d_B,\qquad
  (\Delta_B\otimes\id)\circ\Delta_B
  \;=\;
  (\id\otimes\Delta_B)\circ\Delta_B.
\end{equation}
\end{theorem}

\begin{proof}
We construct the identification at chain level and then verify the
three relations of~\eqref{eq:bar-coalg-mc} from the geometric
factorisation maps.

\emph{Step 1: chain-level construction of the holomorphic
factorisation map at coincident points.}
For $a,b\in\bar\cA$, write $a(z)$, $b(w)$ for their chiral
generating fields. The chiral product
\begin{equation}\label{eq:chiral-prod-OPE}
  Y^{\mathrm{ch}}\colon \cA\otimes\cA\to
  j_*j^*(\cA\boxtimes\cA),\qquad
  Y^{\mathrm{ch}}(a,b)(z,w) \;=\; \sum_{n\in\Z}\,(a_{(n)}b)(w)\,
  (z-w)^{-n-1},
\end{equation}
extends as a $D$-module morphism on $X^2\setminus\Delta$, with a
prescribed pole on $\Delta$ controlled by the singular
filtration~\cite{BD04}\,\S3.1.5. The \emph{holomorphic factorisation
map at coincident points} is the BD chiral product, that is, the
$\Delta_*$-extension of the regular term in the OPE:
\begin{equation}\label{eq:hol-fact-residue}
  \mu^{\mathrm{hol}}_\Delta(a,b)
  \;=\;
  \Delta_*\bigl[(a_{(-1)}b)(w)\bigr]
  \;=\;
  \Delta_*\bigl[(Y^{\mathrm{ch}}(a,b)(z,w)\,(z-w)^{0})|_{z=w}\bigr],
\end{equation}
where the regular evaluation $(z-w)^{0}|_{z=w}$ extracts the
$(z-w)^{0}$-coefficient $a_{(-1)}b$ in the OPE expansion (this is
the chiral product, not a residue: the Cauchy residue of
$(z-w)^{-n-1}$ extracts $a_{(n)}b$, while the chiral product
$a_{(-1)}b$ is the
constant-term coefficient indexed by $n=-1$). Equivalently in
BD~\cite{BD04}\,\S3.1.5 language, $\mu^{\mathrm{hol}}_\Delta$ is the
composition $j_*j^*(\cA\boxtimes\cA)\to\Delta_*\cA[1]$ given by the
chiral structure map at the codimension-one diagonal stratum, with
the degree shift $[1]$ matching the desuspension $s^{-1}$ in
$T^c(s^{-1}\bar\cA)$.

The bar differential $d_B$ on the $T^c$-cogenerator
$s^{-1}a\otimes s^{-1}b$ is defined by
$d_B(s^{-1}a\otimes s^{-1}b)=(-1)^{|a|}\,s^{-1}\mu_\cA(a,b)$
where $\mu_\cA(a,b)=a_{(-1)}b$ is the chiral product
(Vol~I, \cite{BD04}\,\S3.4.7). The identification
$d_B\leftrightarrow\mu^{\mathrm{hol}}_\Delta$ under BD/CG is therefore
literal at chain level after desuspension; gradings agree because
both $d_B$ and $\mu^{\mathrm{hol}}_\Delta$ raise cohomological degree
by $1$ (the $[1]$ on the right of~\eqref{eq:hol-fact-map} is exactly
the degree shift produced by passing from $\bar\cA$ to $s^{-1}\bar\cA$
in the bar tensor coalgebra). This is part~(i).

\emph{Step 2: chain-level construction of the topological
factorisation map.}
On the open colour $\Ran(\R)$ pushed to $X\times\{0\}$, the
$\Eone$-algebra structure of $\cA$ (boundary value) is the
locally constant factorisation algebra
$F_{\mathsf{op}}(I)=\cA$ for any interval $I\subset\R$, with
factorisation maps the multiplication
$\mu_{\Eone}\colon\cA\otimes\cA\to\cA$ for any pair of disjoint
intervals $(I_1,I_2)\hookrightarrow I_1\sqcup I_2\subset I$
(Ayala--Francis~\cite{AF15}\,Theorem~2.24). The cofree
deconcatenation coproduct on $T^c(s^{-1}\bar\cA)$, restricted to the
weight-two component, is
\[
  \Delta_B(s^{-1}a_1\otimes\cdots\otimes s^{-1}a_n)
  \;=\;
  \sum_{i=0}^{n}
  (s^{-1}a_1\otimes\cdots\otimes s^{-1}a_i)\;\otimes\;
  (s^{-1}a_{i+1}\otimes\cdots\otimes s^{-1}a_n).
\]
Geometrically, $\Delta_B$ partitions an ordered configuration of
$n$ open insertions on $\R$ into a left half and a right half;
this is exactly the topological factorisation map of
\eqref{eq:top-fact-map} pulled back along the inclusion of the
ordered-pairs stratum into $\Ran_{\leq 2}(\R)$. Identification
$\Delta_B\leftrightarrow\mu^{\mathrm{top}}$ is again literal at
chain level. This is part~(ii).

\emph{Step 3: $d_B^2=0$ from the chiral Jacobi identity.}
Computed on the weight-three component
$s^{-1}a\otimes s^{-1}b\otimes s^{-1}c$, the iterated
bar differential gives
\begin{align*}
  d_B^2(s^{-1}a\otimes s^{-1}b\otimes s^{-1}c)
  &= d_B\bigl((-1)^{|a|}s^{-1}\mu_\cA(a,b)\otimes s^{-1}c
  + (-1)^{|a|+|b|}s^{-1}a\otimes s^{-1}\mu_\cA(b,c)\bigr) \\
  &= (-1)^{|a|+|a|+|b|}s^{-1}\mu_\cA(\mu_\cA(a,b),c) \\
  &\quad - (-1)^{|a|+|b|+|a|}s^{-1}\mu_\cA(a,\mu_\cA(b,c)) \\
  &= s^{-1}\bigl[\mu_\cA(\mu_\cA(a,b),c)-\mu_\cA(a,\mu_\cA(b,c))\bigr] \\
  &= 0,
\end{align*}
where the final equality is the chiral
associativity~\cite{BD04}\,Proposition~3.4.8 expressed at
weight three (associativity at weight three is the residue of the
Jacobi identity for the OPE $(a_{(-1)}b)_{(-1)}c =
a_{(-1)}(b_{(-1)}c)$ in the absence of higher poles, valid for the
chiral product). The equality is preserved under
$\mu^{\mathrm{hol}}_\Delta$ by Step~1.

\emph{Step 4: cobar compatibility $(d_B\otimes\id+\id\otimes d_B)\circ
\Delta_B=\Delta_B\circ d_B$.}
This is the Leibniz rule: applying $\Delta_B$ to the $T^c$-coalgebra
weight-three element above yields a sum over partitions; applying
$d_B$ to each tensor factor and reassembling matches applying
$d_B$ first and then $\Delta_B$, by the standard cofree-coalgebra
calculation~\cite[\S2.2.2]{LV12}. Geometrically this is the
\emph{open/closed compatibility} of $\SCchtop$: the holomorphic
factorisation at the diagonal $\Delta\subset X^2$ commutes with the
topological factorisation along $\R$ because the two strata of
$X\times\R$ they sit on are disjoint at every interior point, and
the mixed sector of $\SCchtop$ records exactly this compatibility
along the codimension-one boundary $X\times\{0\}$
(Definition~\ref{def:BD-SC}\,(iii)\,(b) bulk action on boundary).

\emph{Step 5: coassociativity $(\Delta_B\otimes\id)\Delta_B=
(\id\otimes\Delta_B)\Delta_B$.}
On the weight-four cogenerator $s^{-1}a_1\otimes s^{-1}a_2\otimes
s^{-1}a_3\otimes s^{-1}a_4$, both sides of coassociativity equal
the iterated triple-deconcatenation
\[
  \sum_{0\leq i\leq j\leq 4}
  (s^{-1}a_1\otimes\cdots\otimes s^{-1}a_i)\otimes
  (s^{-1}a_{i+1}\otimes\cdots\otimes s^{-1}a_j)\otimes
  (s^{-1}a_{j+1}\otimes\cdots\otimes s^{-1}a_4),
\]
with bookkeeping equal on both sides by the standard cofree-coalgebra
identity. Geometrically: the topological factorisation of $\Eone$ on
$\R$ is associative because configurations of three disjoint
intervals $(I_1,I_2,I_3)$ in $\R$ admit a canonical linear
ordering, and the factorisation maps respect that
ordering (Ayala--Francis~\cite{AF15}\,Lemma~2.25; equivalently, the
$\Eone$-multiplication is associative).

This completes the verification of~\eqref{eq:bar-coalg-mc}.
\end{proof}

\subsection{Bulk-to-boundary OPE consistency: $\mathrm{HT}^{3\mathrm{d}}$
recovers the chiral product on the boundary stratum}
\label{subsec:bulk-boundary-OPE-consistency}

Theorem~\ref{thm:bar-diff-eq-holfact} identifies the closed-colour
factorisation map $\mu^{\mathrm{hol}}_\Delta$ with the bar
differential, and the open-colour factorisation
$\mu^{\mathrm{top}}$ with the bar coproduct, on a single fixed
holomorphic-topological underlying space $X \times \R$. Vol~II
Theorem~\ref{thm:universal-holography-functor} (boundary face
``Move~2'') then asserts $\Obs^\partial(\cF_\cA) \simeq \cA$ as
$E_1$-chiral algebras on $X$. We have not yet proved that the
\emph{algebra structure} on the boundary $\Obs^\partial(\cF_\cA)$
agrees, on the nose at chain level, with the chiral product on $\cA$
that was placed there as the boundary condition; this is the
Costello--Gwilliam bulk-to-boundary OPE consistency. The result
is new and established below.

\begin{theorem}[Bulk-to-boundary OPE consistency;
\ClaimStatusProvedHere]
\label{thm:bulk-boundary-OPE-consistency}%
\index{bulk-to-boundary OPE consistency|textbf}%
\index{boundary recovery!chain-level identity}%
\index{$\mathrm{HT}^{3\mathrm{d}}$ recovers chiral OPE|textbf}
Let $\cA$ be an $E_1$-chiral algebra on a smooth complex curve $X$
satisfying the hypotheses of Theorem~\ref{thm:bar-diff-eq-holfact},
and let $\cF_\cA \in \HTQFT_{X \times \R_{\geq 0}}$ be the
factorisation algebra constructed by the universal holography
functor $\Phi_{\mathrm{hol}}$
(Theorem~\ref{thm:universal-holography-functor}). Denote by
\[
  \iota \;\colon\; X \times \{0\} \;\hookrightarrow\;
  X \times \R_{\geq 0}
\]
the boundary inclusion at the closed half-line endpoint, by
\[
  \mu^{\mathrm{hol},\,\mathrm{bulk}}_\Delta \;\colon\;
  j_*j^*\bigl(\cF_\cA^{\boxtimes 2}\bigr)\bigl|_{(X\times\R_{\geq 0})^2}
  \;\longrightarrow\;
  \Delta_{X\times\R_{\geq 0},*}\,\cF_\cA[\,1\,]
\]
the closed-colour holomorphic factorisation map of $\cF_\cA$ in the
bulk along the holomorphic direction $X$, and by
$\mu^{\mathrm{ch}}_\cA = a_{(-1)}b$ the chiral product of $\cA$ on
$X$ in the sense of Beilinson--Drinfeld~\cite{BD04}\,\S3.1.5. Then:
\begin{enumerate}[label=\textup{(\alph*)}]
\item \textbf{Restriction-to-boundary identity.} Under the
restriction-along-boundary functor
$\iota^* \colon \HTQFT_{X\times\R_{\geq 0}} \to \mathrm{ChirAlg}_X$
(Costello--Gwilliam Vol.~II, Theorem~7.3.7~\cite{CG21}), the bulk
holomorphic factorisation map restricts on the nose at chain level:
\begin{equation}\label{eq:bulk-boundary-restriction}
  \iota^*\bigl(\mu^{\mathrm{hol},\,\mathrm{bulk}}_\Delta\bigr)
  \;=\;
  \mu^{\mathrm{hol}}_\Delta,
\end{equation}
where $\mu^{\mathrm{hol}}_\Delta$ on the right is the closed-colour
holomorphic factorisation on $X$ (no $\R$-factor) of
Theorem~\ref{thm:bar-diff-eq-holfact}~(i).
\item \textbf{Chiral-product identification.} Under the
identification $\Obs^\partial(\cF_\cA) \simeq \cA$ of
Theorem~\ref{thm:universal-holography-functor}~(i), the restricted
factorisation map of part~(a) corresponds to the BD chiral product:
\begin{equation}\label{eq:boundary-chiral-product}
  \iota^*\bigl(\mu^{\mathrm{hol},\,\mathrm{bulk}}_\Delta\bigr)(a,b)
  \;=\;
  \Delta_{X,*}\bigl[(a_{(-1)}b)(w)\bigr]
  \;=\;
  \Delta_{X,*}\bigl[\mu^{\mathrm{ch}}_\cA(a,b)(w)\bigr]
\end{equation}
for all $a, b \in \bar\cA$, where the right-hand side is the
$\Delta_*$-extension of the chiral product on the boundary algebra.
\item \textbf{$\Eone$-algebra compatibility.} Combining (a)--(b):
the algebra structure on $\Obs^\partial(\cF_\cA)$ induced by the
bulk closed-colour holomorphic factorisation, restricted to the
boundary stratum, agrees on the nose at chain level with the
$E_1$-chiral algebra structure on $\cA$ used to specify the
boundary condition. The boundary identification of
Theorem~\ref{thm:universal-holography-functor}~(i) is therefore an
identification of $E_1$-chiral algebras at chain level, not only at
the level of underlying objects.
\end{enumerate}
Equivalently: \emph{the restriction of the bulk closed-colour
$\SCchtop$-structure to the codimension-one boundary stratum of
$X \times \R_{\geq 0}$ is, on the nose at chain level, the
$E_1$-chiral OPE on $\cA$.}
\end{theorem}

\begin{proof}
We construct the restriction at chain level, identify it with the
chiral product, and verify the $\Eone$-compatibility from the
construction.

\emph{Step 1: chain-level definition of the restriction
$\iota^*$.}
The Costello--Gwilliam restriction-along-boundary functor sends a
factorisation algebra $\cF$ on $X \times \R_{\geq 0}$ to its
``boundary stalk'' factorisation algebra $\iota^*\cF$ on $X$ via
the colimit
\begin{equation}\label{eq:CG-restriction-stalk}
  (\iota^*\cF)(U)
  \;:=\;
  \mathrm{colim}_{V \supset U \times \{0\}}\,\cF(V),
  \qquad U \subset X \text{ open},
\end{equation}
where the colimit is over open neighbourhoods $V$ of
$U \times \{0\}$ in $X \times \R_{\geq 0}$ with $V \cap
(X \times \{0\}) = U \times \{0\}$
(Costello--Gwilliam~\cite{CG21}\,Definition~7.3.4). For
$\cF = \cF_\cA = \cF_\cA^{\mathrm{hol}} \otimes_{\mathrm{Dunn}}
\cH(\cA)$ from
Theorem~\ref{thm:universal-holography-functor}\,(Move~1), the
$\R_{\geq 0}$-factor of the Dunn product is
$\cH(\cA)$, the \emph{chiral} Hochschild-cochain factorisation algebra
$\cH(\cA) = \cF_{\ChirHoch^\bullet(\cA, \cA)}$
(in the sense of Vol~II Theorem~\ref{thm:chd-ds-hochschild}: chiral,
not topological or categorical Hochschild),
constant in $\R_{\geq 0}$ direction by construction. The
boundary stalk colimit~\eqref{eq:CG-restriction-stalk} therefore
collapses the $\R_{\geq 0}$-direction onto its $0$-endpoint,
returning $\cF_\cA^{\mathrm{hol}}(U) = \cA$ on $U$. This is precisely
the boundary face Move~2 identification
$\Obs^\partial(\cF_\cA) \simeq \cA$
(Theorem~\ref{thm:universal-holography-functor}~(i)).

\emph{Step 2: restriction of the closed-colour holomorphic
factorisation map.}
The bulk holomorphic factorisation map
$\mu^{\mathrm{hol},\,\mathrm{bulk}}_\Delta$ on $\cF_\cA$ acts only
on the $X$-coordinates; its formula on the Dunn product
$\cF_\cA^{\mathrm{hol}} \otimes_{\mathrm{Dunn}} \cH(\cA)$ is
factor-wise $\mu^{\mathrm{hol}}_\Delta \otimes \id$, where
$\mu^{\mathrm{hol}}_\Delta$ is the holomorphic factorisation of
$\cF_\cA^{\mathrm{hol}}$ on $X$ alone. Applying $\iota^*$ to a
factor-wise map of a Dunn product evaluates the second factor at
the $\R_{\geq 0}$-stalk over $0$ and returns the first factor
unchanged:
\begin{equation}\label{eq:iota-pullback-dunn}
  \iota^*\bigl(\mu^{\mathrm{hol}}_\Delta \otimes \id\bigr)
  \;=\;
  \mu^{\mathrm{hol}}_\Delta \otimes \mathrm{ev}_0
  \;=\;
  \mu^{\mathrm{hol}}_\Delta,
\end{equation}
since $\mathrm{ev}_0 \colon \cH(\cA)|_{\R_{\geq 0}} \to k$ on the
constant-in-$\R_{\geq 0}$ chiral Hochschild factorisation algebra
$\cH(\cA) = \cF_{\ChirHoch^\bullet(\cA,\cA)}$ is the
identity-twisted-by-trivialisation, normalised so that the
colimit~\eqref{eq:CG-restriction-stalk} agrees with the boundary
endpoint evaluation. This is~\eqref{eq:bulk-boundary-restriction}
of part~(a).

\emph{Step 3: identification with the chiral product.}
By Theorem~\ref{thm:bar-diff-eq-holfact}~(i)
(equation~\eqref{eq:hol-fact-residue}),
$\mu^{\mathrm{hol}}_\Delta(a,b) = \Delta_*[(a_{(-1)}b)(w)] =
\Delta_*[\mu^{\mathrm{ch}}_\cA(a,b)(w)]$ at chain level on
$\bar\cA \otimes \bar\cA$ for $a, b \in \bar\cA$. Substituting
into~\eqref{eq:iota-pullback-dunn} gives
$\iota^*(\mu^{\mathrm{hol},\,\mathrm{bulk}}_\Delta)(a,b) =
\Delta_*[\mu^{\mathrm{ch}}_\cA(a,b)(w)]$, which is
exactly~\eqref{eq:boundary-chiral-product} of part~(b).

\emph{Step 4: $E_1$-chiral algebra compatibility.}
We disambiguate two distinct $\Eone$-structures: the topological
$\Eone$ on $\R$ (open colour of $\SCchtop$, controlling the
boundary topological factorisation) and the $E_1$-\emph{chiral}
structure on $X$ (closed colour, controlling the chiral OPE at
the diagonal $\Delta\subset X^2$). On the boundary
$\Obs^\partial(\cF_\cA) \simeq \cA$, the algebra structure is the
$E_1$-chiral product on $X$ (not the topological $E_1$ on $\R$,
which has been collapsed by the boundary stalk). The
Costello--Gwilliam restriction-along-boundary
(\cite{CG21}\,\S7.3.5) sends the $\cF_\cA^{\mathrm{hol}}$-factor's
holomorphic factorisation map at the diagonal of $X$ to the
$E_1$-chiral product on $\Obs^\partial(\cF_\cA)$, since the
$\R_{\geq 0}$-factor of the Dunn product evaluates to a scalar at
the boundary endpoint (Step~2). By Steps~2--3 this
$E_1$-chiral product equals
$\Delta_*[\mu^{\mathrm{ch}}_\cA(a,b)]$ at chain level, which is
exactly the BD chiral product on $\cA$ used as boundary
condition. Hence the boundary identification of
Theorem~\ref{thm:universal-holography-functor}~(i) is an
identification of $E_1$-chiral algebras at chain level, not only
of underlying complexes; this is part~(c).

\emph{Step 5: well-definedness independent of Dunn-product
presentation.}
The Dunn-product presentation
$\cF_\cA = \cF_\cA^{\mathrm{hol}} \otimes_{\mathrm{Dunn}}
\cH(\cA)$ is one chain-level model among the equivalence class of
$\HTQFT_{X \times \R_{\geq 0}}$-objects realising
$\Phi_{\mathrm{hol}}(\cA)$. Under any other chain-level model
$\cF'_\cA$ with the same boundary data, the Costello--Gwilliam
boundary stalk~\eqref{eq:CG-restriction-stalk} returns the same
factorisation algebra $\cF^{\mathrm{hol}}_\cA$ on $X$
(by~\cite{CG21}\,Theorem~7.3.7 invariance of the boundary stalk
under quasi-isomorphism of the bulk model, applied to the
stratified factorisation algebra on $X \times \R_{\geq 0}$ with
the boundary cosheaf condition imposed at the codimension-one
boundary $X\times\{0\}$). The chiral product
identification~\eqref{eq:boundary-chiral-product} is therefore
model-independent at the level of chain-level $E_1$-chiral algebra
structure; gauge-equivalent bulk models give chain-equivalent
identifications.

\emph{Step 6: $\infty$-categorical lift (CLAUDE.md parity).}
The chain-level statement of parts~(a)--(c) lifts to an
$(\infty,1)$-categorical statement: at the level of factorisation
$\infty$-categories~\cite{AF15,FG12}, the
boundary-restriction functor $\iota^*$ is a left
adjoint to boundary-pushforward, and the bulk-to-boundary OPE
consistency is the unit $\eta\colon \id_{\mathrm{ChirAlg}_X^{\Eone}}
\xrightarrow{\sim} \iota^* \circ \Phi_{\mathrm{hol}}$ being an
equivalence of $\Eone$-chiral algebra functors at the level of
the $(\infty,1)$-category. Both lanes (chain-level Dunn product
in this proof; $(\infty,1)$-categorical adjunction unit) are
load-bearing per CLAUDE.md parity declaration; they agree under
the homotopy coherent nerve realisation of factorisation algebras
(\cite{CG21}\,\S6 Whitehead theorem for factorisation algebras).
\end{proof}

shadow. In the categorical direction, the $\SCchtop$-datum is
equivalent to the $E_2$-monoidal structure on a module category for
$\cA$, together with a half-braiding whose spectral reduction is
Face~(3)'s $r$-matrix. The construction is a right adjoint, not an
averaging; the half-braiding is
\emph{constructed}, not narrated.

\subsection{The factorization module category}\label{subsec:fact-mod-cat}

\begin{definition}[Factorization module category]\label{def:rep-fact}%
\index{factorization module category!$\Rep_{\mathrm{fact}}$|textbf}%
\index{$\Rep_{\mathrm{fact}}$}
Let $\cA$ be an $E_1$-chiral algebra on $X$. The \emph{factorization
module category}
\begin{equation}
  \Rep_{\mathrm{fact}}(\cA)
  \;:=\;
  \mathbb{A}\text{-}\mathrm{fact}\mathrm{Mod}(\Dmod(\Ran(X)))
\end{equation}
is the dg category of factorization modules for $\cA$: objects are
$\cA$-module $D$-modules on $\Ran(X)$ whose factorization structure
is compatible with the chiral $\cA$-module action. Morphisms are
$\cA$-linear factorization morphisms. Composition is defined
using the chiral tensor product
$\otimes^{\mathrm{ch}}$ on $\Ran(X)$
(Beilinson--Drinfeld, GR17 IV.5; Francis--Gaitsgory 2012).
\end{definition}

\begin{remark}[Three local models agree]\label{rem:three-local-models}
On the formal disc $D_x\subset X$, the category
$\Rep_{\mathrm{fact}}(\cA)$ restricts to the usual category
$\cA_{x}\text{-}\Mod$ of modules over the $\cA$-mode algebra
(the Lie algebra of modes of $\cA$ on $D_x$). On the punctured
formal disc $D^\times_x$ it restricts to the category of smooth
$\hat{\cA}_{x}$-modules. The three models glue along
$D^\times_x\hookrightarrow D_x$ via the pullback of the factorization
structure; the glue is the content of
Lemma~\ref{lem:rep-fact-local-global}.
\end{remark}

\begin{lemma}[Local--global description of $\Rep_{\mathrm{fact}}$;
\ClaimStatusProvedElsewhere]\label{lem:rep-fact-local-global}
$\Rep_{\mathrm{fact}}(\cA)$ is equivalent to the (∞,1)-limit
of the factorization presheaf of local module categories over the
Ran space $\Ran(X)$. Equivalently, it is the category of
modules over the factorization algebra $\Obs^{\mathrm{ch}}_\cA$ in
$\Dmod(\Ran(X))$. The equivalence is Beilinson--Drinfeld for the
algebraic incarnation, Costello--Gwilliam for the analytic one; see
Gaitsgory--Rozenblyum~\cite{GR17} IV.5.
\end{lemma}

\begin{proof}[Attribution]
Beilinson--Drinfeld \textit{Chiral Algebras} Ch.~3, Lemma~3.4.15;
Gaitsgory--Rozenblyum, \textit{Study in Derived Algebraic Geometry},
Vol.~II, Ch.~IV.5, for the $\infty$-categorical form. The
factorization-module description is in Francis--Gaitsgory 2012
Definition~2.7.6.
\end{proof}

\subsection{The half-braiding construction}\label{subsec:half-braiding}

\begin{definition}[Chiral half-braiding]\label{def:chiral-half-braiding}%
\index{half-braiding!chiral|textbf}%
\index{factorization module category!half-braiding}
Let $\cA$ be an $E_1$-chiral bialgebra with spectral coproduct
$\Delta_z\colon \cA\to\cA\otimes\cA$
(Chapter~\ref{ch:theory-hochschild}, Definition~\ref{def:half-braiding-explicit}).
For $N\in\Rep_{\mathrm{fact}}(\cA)$ and $a\in\cA$, $n\in N$, set
\begin{equation}
  \label{eq:chiral-half-braiding}
  \sigma_\cA(z)(a\otimes n)
  \;:=\;
  \sum\bigl(\Delta_z(a)\bigr)_{(2)}\cdot n
  \;\otimes\;
  \bigl(\Delta_z(a)\bigr)_{(1)}
  \;=\;
  \sum a_{(2),z}\cdot n\otimes a_{(1),z}.
\end{equation}
The family $\{\sigma_\cA(z)\}_{z\in X\setminus\{0\}}$ assembles into a
natural isomorphism of functors
$\cA\otimes(-)\Rightarrow(-)\otimes\cA$ on
$\Rep_{\mathrm{fact}}(\cA)$, meromorphic in the spectral parameter $z$,
with first-order pole at $z=0$.
\end{definition}

\begin{proposition}[Half-braiding from $\Delta_z$; \ClaimStatusProvedHere]
\label{prop:half-braiding-construction}%
\index{half-braiding!from $\Delta_z$}
The half-braiding $\sigma_\cA(z)$ of
Definition~\ref{def:chiral-half-braiding} satisfies:
\begin{enumerate}[label=\textup{(\roman*)}]
  \item hexagon identities (quasi-triangularity of $\cA$ expressed
  on factorization modules);
  \item spectral $R$-matrix:
  $\sigma_\cA(z)|_{V_u,V_v}=R(u-v)$ on evaluation modules
  $V_u,V_v$ with spectral parameters $u,v$;
  \item $\Rep_{\mathrm{fact}}(\cA)$ becomes a braided monoidal
  $\infty$-category, i.e.\ an $E_2$-algebra in $\mathrm{Cat}_\infty$;
  the braided structure lives one categorical level up, on the
  representation category, not on $\cA$: $\cA$ itself remains an
  $E_1$-chiral algebra;
  \item the braiding \emph{extends} to all of $\Rep_{\mathrm{fact}}(\cA)$,
  not only to evaluation modules: for generic $M,N$ in
  $\Rep_{\mathrm{fact}}$, $\sigma_\cA$ provides a meromorphic
  natural transformation $M\otimes N\to N\otimes M$ with poles
  on the discriminant of the module supports.
\end{enumerate}
\end{proposition}

\begin{proof}
(i) follows from the defining quasi-triangularity axiom for
$\Delta_z$ (Chapter~\ref{ch:theory-drinfeld}): for
$a,b\in\cA$,
\(
  \sigma_\cA(z)(ab\otimes n)=
  \sigma_\cA(z)(a\otimes\sigma_\cA(z)(b\otimes n))
\)
after applying the spectral coproduct to the product $ab$ and reading
off Sweedler indices. The axiomatics match the standard quasi-Hopf
hexagon after translation to the half-braiding form.

(ii) is the evaluation specialisation of~\eqref{eq:chiral-half-braiding}:
on $V_u\otimes V_v$, $\Delta_z(a)\cdot(v_u\otimes v_v)$ evaluates to
$\pi_u\otimes\pi_v(\Delta_z(a))$, which is the spectral
$R$-matrix acting on the tensor product modulo the swap, recovering
$R(u-v)$ by the Drinfeld--Jimbo reconstruction of $R$ from
$\Delta_z$ (Chapter~\ref{ch:ordered-chiral}).

(iii) is a consequence of (i) + (ii) + the module category structure
of $\Rep_{\mathrm{fact}}$: a braided monoidal structure is a
half-braiding that is itself natural, and naturality is the
factorization-module compatibility of Lemma~\ref{lem:rep-fact-local-global}.
By Lurie HA~5.1 the braided monoidal structure is an
$E_2$-algebra in $\mathrm{Cat}_\infty$.

(iv) is the extension from evaluation modules to arbitrary
factorization modules, enabled by the universal property of
$\Rep_{\mathrm{fact}}(\cA)$ (Lemma~\ref{lem:rep-fact-local-global}):
every module is a colimit of evaluation modules along the factorization
structure, and $\sigma_\cA(z)$ is natural with respect to factorization
morphisms. Meromorphicity of $\sigma_\cA(z)$ on the
supports follows from OPE pole bounds on $\Delta_z$.
\end{proof}

\subsection{Drinfeld centre and factorization centre}\label{subsec:dcz-vs-fcz}

\begin{definition}[Drinfeld centre of $\Rep_{\mathrm{fact}}(\cA)$]
\label{def:drinfeld-centre-rep-fact}%
\index{Drinfeld centre!of $\Rep_{\mathrm{fact}}$|textbf}
The \emph{Drinfeld centre}
$\cZ\bigl(\Rep_{\mathrm{fact}}(\cA)\bigr)$ is the $\infty$-category of
pairs $(N,\sigma_N)$, where $N\in\Rep_{\mathrm{fact}}(\cA)$ and
$\sigma_N\colon N\otimes(-)\Rightarrow(-)\otimes N$ is a half-braiding
of $N$ compatible with $\sigma_\cA(z)$. Composition is component-wise.
The Drinfeld centre carries a canonical $E_2$-monoidal structure.
\end{definition}

\begin{theorem}[Drinfeld centre face of $\SCchtop$;
\ClaimStatusProvedHere]
\label{thm:drinfeld-centre-sc-face}%
\index{SC!Drinfeld centre face|textbf}%
\index{heptagon!face 6}
Let $\cA$ be an $E_1$-chiral bialgebra on $X$ with conformal-vector
hypothesis dropped (generic case). Then
\begin{equation}
  \label{eq:drinfeld-centre-face}
  \cZ\bigl(\Rep_{\mathrm{fact}}(\cA)\bigr)
  \;\simeq\;
  \bigl(\Rep_{\mathrm{fact}}(Z^{\mathrm{der}}_{\mathrm{ch}}(\cA))\bigr)^{E_2}
\end{equation}
as braided monoidal $\infty$-categories, where the left side is the
categorical Drinfeld centre and the right side is the category of
factorization modules over the derived chiral centre
(Definition~\ref{def:rep-fact} applied to the bulk algebra), with
its induced $E_2$-monoidal structure. The equivalence carries the
half-braiding of~\eqref{eq:chiral-half-braiding} on the left to the
spectral braiding of the right induced from
$\Delta_z^{\,Z}\colon Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)\to
Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)\otimes Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)$.
The pair
\(
  \bigl(\cZ(\Rep_{\mathrm{fact}}(\cA)),\Rep_{\mathrm{fact}}(\cA)\bigr)
\)
is a categorified $\SCchtop$-datum: the bulk category acts on
the boundary category, with no open-to-closed direction.
\end{theorem}

\begin{proof}
Construct the equivalence in four steps.

\emph{Step 1: from half-braiding to central object (right adjoint to
forgetful).} The bar coalgebra $B^{\,\mathrm{ch}}(\cA)$ is an object
of $\Rep_{\mathrm{fact}}(\cA)$, not of the Drinfeld centre
$\cZ(\Rep_{\mathrm{fact}}(\cA))$ (these are categorically disjoint:
the bar coalgebra is an $E_1$-coalgebra in a monoidal category, while
the centre is the universal $E_2$-monoidal recipient receiving any
$E_1$-monoidal functor with target $\Rep_{\mathrm{fact}}(\cA)$). The
construction of central objects is therefore the right adjoint to the
forgetful functor (Majid 1998, Lurie HA~5.3.1.30), \emph{not} a lift
of bar elements to centre elements. Concretely: for any
$N\in\Rep_{\mathrm{fact}}(\cA)$, the centre object $\cZ(N)$ is the
universal half-braided object equipped with a natural transformation
$\sigma_N(z)\colon\cA\otimes(-)\to(-)\otimes\cA$ on $N$, given by
$\sigma_N(z)$ from~\eqref{eq:chiral-half-braiding}. The unit of the
adjunction sends the trivial module to the identity central object;
generators of $\cA$, viewed as endomorphisms of the regular module
$\cA\in\Rep_{\mathrm{fact}}(\cA)$, produce via the right adjoint the
half-braiding data of Proposition~\ref{prop:half-braiding-construction}.
No bar coalgebra ``lifts to a central object'': the right adjoint
constructs centre objects categorically, the bar coalgebra is then a
\emph{computational tool} for resolving central objects via the
$E_\infty$-resolution of $\cA$ inside $\Rep_{\mathrm{fact}}(\cA)$
(Lurie HA~5.5.4 for the analogous $E_n$-resolution).

\emph{Step 2: the categorified bar--cobar adjunction.} The adjoint pair
\begin{equation}
  \mathrm{forget}\colon\cZ(\Rep_{\mathrm{fact}}(\cA))\rightleftarrows
  \Rep_{\mathrm{fact}}(\cA)\colon\mathrm{centre}
\end{equation}
is the right-adjoint-to-forgetful construction (Majid 1998;
Lurie HA~5.3): the centre is the categorified \emph{commutant}
, not an averaging. Its right adjoint produces, for any
factorization module $N$, the universal half-braided lift
$\cZ(N)\in\cZ(\Rep_{\mathrm{fact}}(\cA))$.

\emph{Step 3: identification with bulk modules.} The derived chiral
centre $Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)=C^{\bullet}_{\mathrm{ch}}(\cA,\cA)$
acts on $\cA$ through its $E_2$-chiral structure (chiral Higher
Deligne, chiral-Hochschild incarnation). By
Beilinson--Drinfeld--Zhu--Francis--Gaitsgory--Rozenblyum factorization
module theory (Lurie HA~5.3.1.30; BZFN10),
$\Rep_{\mathrm{fact}}(Z^{\mathrm{der}}_{\mathrm{ch}}(\cA))$ acquires the
induced braided monoidal structure from
$\Delta_z^{\,Z}$. Step~2's right adjoint then factors through this
category: $\cZ(\Rep_{\mathrm{fact}}(\cA))\simeq
\Rep_{\mathrm{fact}}(Z^{\mathrm{der}}_{\mathrm{ch}}(\cA))$.

\emph{Step 4: matching the half-braiding to the spectral braiding.}
On evaluation modules the matching is
Proposition~\ref{prop:half-braiding-construction}~(ii). The
extension to all factorization modules is
Proposition~\ref{prop:half-braiding-construction}~(iv). The two
braidings agree as natural transformations
by the universal property of factorization module categories
(Lemma~\ref{lem:rep-fact-local-global}): both compute the action of
the central object on a general module, through the
factorization-compatible chiral tensor product.

This completes the equivalence in~\eqref{eq:drinfeld-centre-face}.
The $\SCchtop$-datum at the categorical level is read off directly:
the bulk category is $\cZ(\Rep_{\mathrm{fact}}(\cA))$, the boundary
category is $\Rep_{\mathrm{fact}}(\cA)$, the module action of the
first on the second is the centre-to-forgetful counit, no open-to-closed
exists (the centre receives no extra input from the boundary beyond
what the forgetful sends).
\end{proof}

\begin{remark}[Associator dependence]\label{rem:face-6-associator}
Theorem~\ref{thm:drinfeld-centre-sc-face} is
$\infty$-quasi-isomorphic with the factorization face after fixing a
Drinfeld associator $\Phi\in\mathrm{GRT}_1(\mathbb{Q})$. The underlying
braided monoidal $\infty$-category is $\Phi$-independent; the strict
chain-level presentation depends on $\Phi$. On the Koszul locus, two
associators yield strictly equivalent half-braidings after applying
the $\Phi_1\Phi_2^{-1}$ twist.
\end{remark}

\section{Face (7): derived algebraic geometry}
\label{sec:face-dag}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

The seventh face reads $\SCchtop$ as the algebra of global sections
of a derived stack classifying $E_1$-chiral algebras with
$\SCchtop$-datum. The stack is $1$-shifted symplectic in the sense
of Pantev--To\"en--Vaqui\'e--Vezzosi (PTVV~2013), and the convolution
face is recovered as the tangent Lie algebra.

\subsection{The classifying stack}\label{subsec:classifying-stack}

\begin{definition}[Stack of $E_1$-chiral algebras with
$\SCchtop$-datum]\label{def:moduli-sc-ch-top}%
\index{moduli stack!of $E_1$-chiral algebras with $\SCchtop$-datum|textbf}%
\index{$\cM_{\SCchtop}$}
Let $X$ be a smooth complex curve. Define the functor
\begin{equation}
  \cM_{\SCchtop,X}\colon
  (\mathrm{dgArt}/\mathbb{C})^{\mathrm{op}}
  \;\longrightarrow\;
  \mathrm{sSet}
\end{equation}
from derived Artinian $\mathbb{C}$-algebras to simplicial sets by
sending $R\in\mathrm{dgArt}$ to the mapping space
\begin{equation}
  \cM_{\SCchtop,X}(R)
  \;:=\;
  \bMap_{\mathrm{OperadCat}_\infty}
  \bigl(\SCchtop,\,
  \mathrm{End}^{\mathrm{col}}_{(\cA,\,Z)}\otimes_{\mathbb{C}} R\bigr),
\end{equation}
where the mapping space is taken in coloured $\infty$-operads on
$\Dmod(\Ran(X))$ and the colimit is over all pairs
$(\cA,Z)$ with $\cA$ an $R$-flat $E_1$-chiral algebra on $X$ and
$Z$ an $E_2$-chiral algebra acting on $\cA$ through the brace
structure. The stack is the sheafification of this functor with respect
to the étale topology on $\mathrm{dgArt}$.
\end{definition}

\begin{lemma}[Representability; \ClaimStatusProvedElsewhere]
\label{lem:moduli-sc-representable}
The functor $\cM_{\SCchtop,X}$ is a locally finite presented derived
Artin stack over $\mathbb{C}$. It carries a natural action of the
Drinfeld--Grothendieck--Teichm\"uller group
$\mathrm{GRT}_1(\mathbb{Q})$ through associator transport.
\end{lemma}

\begin{proof}[Attribution]
Representability: To\"en--Vaqui\'e (Compos.\ Math.\ 143, 2007,
\S3.3) for operadic moduli; Lurie, \emph{Derived Algebraic Geometry IV}
for the Artin criteria; Francis--Gaitsgory 2012 for the
factorization variant. $\mathrm{GRT}_1$-action via Willwacher 2015
(Invent.\ Math.\ 200).
\end{proof}

\begin{remark}[Three strata]\label{rem:moduli-strata}
$\cM_{\SCchtop,X}$ stratifies by conformal-vector presence:
the open substack $\cM^{\mathrm{conf}}_{\SCchtop,X}$ where the
boundary $\cA$ carries a conformal vector at non-critical level;
the closed complement $\cM^{\mathrm{nc}}_{\SCchtop,X}$ without
conformal vector; and the stratum of critical level
$\cM^{\mathrm{crit}}_{\SCchtop,X}$ where the conformal vector
degenerates. On $\cM^{\mathrm{conf}}$ the structure topologises to
$E_3$-topological (Chapter~\ref{ch:topologization}); on
$\cM^{\mathrm{nc}}$ and $\cM^{\mathrm{crit}}$ the heptagon remains
intact.
\end{remark}

\subsection{Shifted symplectic structure}\label{subsec:shifted-symplectic}

\begin{theorem}[$1$-shifted symplectic structure on
$\cM_{\SCchtop,X}$; \ClaimStatusProvedHere]
\label{thm:moduli-sc-shifted-symplectic}%
\index{PTVV!on moduli of $\SCchtop$-data}%
\index{$1$-shifted symplectic!on $\cM_{\SCchtop,X}$}
The derived stack $\cM_{\SCchtop,X}$ carries a natural
$1$-shifted symplectic form $\omega_{\SCchtop}\in
\mathrm{Sp}^1(\cM_{\SCchtop,X})$, constructed as the pairing
\begin{equation}
  \label{eq:moduli-shifted-sympl}
  \omega_{\SCchtop}
  \;=\;
  \int_{X\times\mathbb{R}_{\geq 0}}\;
  \mathrm{ev}^{*}\,\kappa_{\SCchtop},
\end{equation}
where $\mathrm{ev}$ is the evaluation map from
$\cM_{\SCchtop,X}\times X\times\mathbb{R}_{\geq 0}$ to the
universal $E_3$-chiral target (the tangent pairing of the universal
$\SCchtop$-action), and $\kappa_{\SCchtop}$ is the coloured bilinear
form on $(\SCchtop)^{!}=(\Lie,\Ass,\text{shuffle-mixed})$ given by
the Killing form on the closed $\Lie$ colour and the trace form on
the open $\Ass$ colour. The pairing is anti-symmetric of degree $1$,
and closed as a shifted $2$-form by PTVV~\cite{PTVV13}
Theorem~2.5 (mapping-stack version) applied to the source
$X\times\mathbb{R}_{\geq 0}$, which is a $2$-orientable $\infty$-pair
of dimensions $(1,1)$.
\end{theorem}

\begin{proof}
Apply PTVV~\cite{PTVV13} Theorem~2.5 to the mapping stack
$\cM_{\SCchtop,X}=\bMap(X\times\mathbb{R}_{\geq 0},
B\SCchtop{\text{-}}\mathrm{Alg})$, where
$B\SCchtop\mathrm{-Alg}$ denotes the classifying $(\infty,2)$-stack of
$\SCchtop$-algebras in $\Dmod(\Ran(X))$. The source
$X\times\mathbb{R}_{\geq 0}$ is a smooth $\infty$-pair of
\emph{holomorphic-topological} dimension $(1\,|\,1)$ (one complex
dimension on $X$ contributing one holomorphic degree to the de~Rham
filtration; one real dimension on $\R_{\geq 0}$ contributing one
topological degree); the total derived dimension for PTVV
transgression is $1+1=2$. The target carries a $2$-shifted
symplectic structure induced by the coloured bilinear form
$\kappa_{\SCchtop}$. Integration over the source drops degree by
$2$, yielding a $0$-shifted symplectic form, and the de~Rham
spectral sequence collapses on the boundary stratum
$X\times\{0\}$ to give a residual $1$-shifted form on the bulk
moduli, after restriction to the boundary-condition fibre.
Closedness follows from the closed nature of the source-side
$2$-form and PTVV's transgression machinery (PTVV~\cite{PTVV13}
Construction~2.4 for compact-source case, Calaque~\cite{Calaque15}
for the boundary-with-corners extension at
$\partial(X\times\R_{\geq 0})=X\times\{0\}$).

\emph{Boundary condition.} The non-compactness of $\R_{\geq 0}$ is
controlled by:
(i) at $\partial\R_{\geq 0}=\{0\}$, the boundary insertion is
the boundary-condition module $\cA\in\Rep_{\mathrm{fact}}(\cA)$
viewed as an $\SCchtop$-algebra over the boundary stratum;
(ii) at $+\infty$, the Dirichlet condition that the bulk field
decays to its vacuum value, equivalently $L^2$-summability against
the Costello--Gwilliam regulator on the half-line.

\emph{Non-degeneracy: scope qualifier.} Non-degeneracy
follows from the non-degeneracy of $\kappa_{\SCchtop}$ on the
\emph{Lie--sector locus} $\cM^{\mathrm{ss}}_{\SCchtop,X}\subset
\cM_{\SCchtop,X}$ where the closed colour is a semisimple or
reductive Lie sector (Killing form non-degenerate; finite-dimensional
or filtered semisimple in the chiral lift) and the open colour is
finite-dimensional Ass (trace form non-degenerate). On the
complementary locus (Heisenberg/abelian closed colour, infinite-rank
Ass), $\omega_{\SCchtop}$ is only \emph{pre-symplectic}: closed of
the right degree but with a possibly non-trivial radical.
\end{proof}

\begin{remark}[Holomorphic-topological dimension counting]
\label{rem:why-pair-21}%
\index{dimension counting!holomorphic-topological}
The source $X\times\mathbb{R}_{\geq 0}$ carries two distinct
dimensions of two distinct kinds:
$X$ is $1$-complex-dimensional hence $2$-real-dimensional but
contributes a single \emph{holomorphic degree} to the de~Rham
spectral sequence; $\mathbb{R}_{\geq 0}$ is $1$-real-dimensional
with boundary and contributes a single \emph{topological degree}.
PTVV's framework allows this mixed dimension after the explicit
boundary condition above. Holomorphic-topological dimension
$(1\,|\,1)$ pairs a $2$-form on the target with the source
fundamental class, dropping degree by $2$, producing a
$0$-shifted output before restriction to the boundary-condition
fibre, and a residual $1$-shifted output on the bulk moduli after
the boundary fibre is collapsed.
\end{remark}

\begin{remark}[Scope for Face~(7) non-degeneracy]
\label{rem:face-7-scope}%
\index{Face (7)!scope!Lie semisimple, Ass finite-dim}%
\index{1-shifted symplectic!scope}
Face~(7) presents $\SCchtop$ as $1$-shifted symplectic only on the
\emph{semisimple-or-reductive Lie locus} of the closed colour and
the \emph{finite-dimensional Ass locus} of the open colour. The
exhibits:
\begin{itemize}
\item Closed semisimple/reductive: $V_k(\fg)$ at non-critical level
for $\fg$ semisimple (class L); affine Heisenberg gauge-fixed by
sub-quotient. Closed not semisimple: pure Heisenberg
$\cH_k$ (abelian, Killing form $\equiv 0$); critical-level
$V_{-h^\vee}(\fg)$ (Sugawara degenerates).
\item Open finite-dim Ass: matrix algebras $M_n(\C)$, finite-dim
group algebras. Open infinite-dim: smooth modules over an infinite-
dimensional Lie algebra without trace form.
\end{itemize}
On the complement, $\omega_{\SCchtop}$ exists as a closed
$1$-shifted form but with a non-trivial radical; Face~(7) is then a
\emph{$1$-shifted pre-symplectic} structure, with non-degeneracy
restored after symplectic reduction along the radical (when this
quotient exists; for pure Heisenberg, the radical is everything and
the quotient is trivial). The corollary
\ref{cor:face-7-presents-sc} below inherits this scope.
\end{remark}

\subsection{Lagrangian correspondence to the convolution face}\label{subsec:lagrangian-corr}

\begin{theorem}[Convolution face as Lagrangian section;
\ClaimStatusProvedHere]
\label{thm:convolution-lagrangian}%
\index{convolution!Lagrangian section of $\cM_{\SCchtop,X}$}%
\index{heptagon!edge (5)--(7)}
The convolution $L_\infty$-algebra
$\fg^{\,\SCchtop}_{\cA,Z}$ of Face~(5) is the tangent dg-Lie
algebra to $\cM_{\SCchtop,X}$ at the point represented by
the pair $(\cA,Z^{\mathrm{der}}_{\mathrm{ch}}(\cA))$. The Maurer--Cartan
locus
\begin{equation}
  \MC\bigl(\fg^{\,\SCchtop}_{\cA,Z}\bigr)
  \;\subset\;
  \cM_{\SCchtop,X}
\end{equation}
is a Lagrangian subvariety of
$(\cM_{\SCchtop,X},\omega_{\SCchtop})$ in the PTVV sense: the
pullback of $\omega_{\SCchtop}$ to the MC locus vanishes and the
normal bundle of MC inside $\cM_{\SCchtop,X}$ is anti-symplectic to
the tangent of MC.
\end{theorem}

\begin{proof}
Tangent identification: by the universal property of mapping stacks,
the tangent complex of $\cM_{\SCchtop,X}$ at $(\cA,Z)$ is
\begin{equation}
  T_{(\cA,Z)}\cM_{\SCchtop,X}
  \;=\;
  \mathrm{Hom}^{\mathrm{col}}\bigl(B(\SCchtop),\,\End^{\mathrm{col}}_{(\cA,Z)}\bigr),
\end{equation}
which is exactly $\fg^{\,\SCchtop}_{\cA,Z}$ (Face~(5); Loday--Vallette
\textsection 10.2 for the coloured convolution adjoint). The
$L_\infty$-structure on the tangent complex matches the
convolution structure by Kontsevich formality of $\SCchtop$.

Lagrangian property: the anchored form
$\omega_{\SCchtop}|_{\MC}$ vanishes because the MC equation in
$\fg^{\,\SCchtop}_{\cA,Z}$ is equivalent to the first-order
vanishing of $\omega_{\SCchtop}$ along the tangent direction (PTVV~2013
Proposition~2.8). Anti-symplectic normal: the normal bundle is
$\fg^{\,\SCchtop}_{\cA,Z}[1]$, shifted by one, and inherits the
anti-symplectic pairing from the PTVV cotangent adjunction.
\end{proof}

\begin{corollary}[Face~(7) presents $\SCchtop$;
\ClaimStatusProvedHere]
\label{cor:face-7-presents-sc}
The $\SCchtop$-datum associated to any point $(\cA,Z)\in
\cM_{\SCchtop,X}$ is fully determined by the germ of
$\omega_{\SCchtop}$ at that point together with the Lagrangian
correspondence of Theorem~\ref{thm:convolution-lagrangian}.
Hence $\cM_{\SCchtop,X}$ presents $\SCchtop$ as a shifted
symplectic geometry, and recovers the operadic face through its
tangent complex.
\end{corollary}

\begin{proof}
Combine Theorem~\ref{thm:moduli-sc-shifted-symplectic} with
Theorem~\ref{thm:convolution-lagrangian}: the operadic structure of
$\SCchtop$ is encoded in the shifted symplectic form
$\omega_{\SCchtop}$; the $L_\infty$-structure is encoded in the
tangent complex; the $\SCchtop$-data as operad-algebras are in
bijection with Lagrangian sections of $\omega_{\SCchtop}$.
\end{proof}

\section{The seven edges of the heptagon}
\label{sec:heptagon-edges}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

Number the faces $1,\dots,7$ as above. We prove the seven adjacent
edges as named theorems; the remaining $\binom{7}{2}-7=14$ diagonal
identifications follow by composition, hence are recorded as
remarks.

\begin{convention}[Edge regime]\label{conv:edge-regime}
Every edge is an $\infty$-quasi-isomorphism of coloured
$\infty$-operads after fixing a Drinfeld associator
$\Phi\in\mathrm{GRT}_1(\mathbb{Q})$. Strict chain-level equivalences
hold on the Koszul locus; away from it, the equivalences are
$\infty$-quasi-isomorphisms that become strict after passing to
$\Phi$-transported presentations. We suppress $\Phi$ from edge names;
each name carries its $\Phi$-dependence implicitly.
\end{convention}

\subsection{Edge (1)--(2): operadic--Koszul-dual}\label{subsec:edge-12}

\begin{theorem}[Ginzburg--Kapranov Koszul duality for
$\SCchtop$; \ClaimStatusProvedHere]\label{thm:edge-12}%
\index{heptagon!edge (1)--(2)}
The bar--cobar adjunction
$(\Omega,B)$ on coloured $\infty$-operads restricts to a
Quillen equivalence between Face~(1) and Face~(2):
\begin{equation}
  W(\SCchtop)=\Omega(B(\SCchtop))
  \;\xrightarrow{\sim}\;
  \SCchtop.
\end{equation}
The bar coalgebra $B(\SCchtop)$ is explicitly computed (closed
sector $B(\Com)=\Lie^c$, open sector $B(\Ass)=\Ass^c$, mixed sector
shuffle-bar); cobarring recovers $\SCchtop$.
\end{theorem}

\begin{proof}
The result is coloured Ginzburg--Kapranov: Vallette
\textit{A Koszul duality for properads} (Trans.\ AMS 359, 2007)
Theorem~4.1.3 for the coloured case; Hoefel~2009 (arXiv:0809.4623)
and Hoefel--Livernet~2012 (arXiv:1207.2307) specialise to the Swiss
cheese operad and verify Koszulity, producing the coloured Koszul
dual as $(\Lie,\Ass,\text{shuffle-mixed})$. The proof is
Proposition~\ref{prop:sc-koszul-dual-three-sectors} of
Chapter~\ref{ch:theory-drinfeld}. The bar and
cobar functors preserve weak equivalences of $\infty$-operads
(Fresse, \textit{Modules over Operads and Functors}, Springer 2009),
hence the Quillen equivalence.
\end{proof}

\subsection{Edge (2)--(3): Koszul-dual--factorization}\label{subsec:edge-23}

\begin{theorem}[Chiral Deligne--Tamarkin for $\SCchtop$ at
$E_2$-cohomology; \ClaimStatusProvedHere]\label{thm:edge-23}%
\index{heptagon!edge (2)--(3)}
The cofibrant replacement $W(\SCchtop)$ acts on the pair
$(Z^{\mathrm{der}}_{\mathrm{ch}}(\cA),\cA)$ via the brace structure of
Chapter~\ref{ch:theory-hochschild}, and this action is
$\infty$-quasi-isomorphic to the factorization action of
Face~(3). Explicitly,
\begin{equation}
  W(\SCchtop)
  \;\xrightarrow{\sim}\;
  \End^{\mathrm{col}}_{(\cA,Z^{\mathrm{der}}_{\mathrm{ch}}(\cA))},
\end{equation}
with the $\infty$-quasi-isomorphism fixed by $\Phi\in\mathrm{GRT}_1$.
At $E_2$-cohomology (i.e.\ passing to derived brace structure) the
equivalence is $\Phi$-independent.
\end{theorem}

\begin{proof}
Combine:
(i) the chiral higher Deligne theorem
(Theorem~\ref{thm:chiral-higher-deligne}, Chapter~\ref{ch:theory-hochschild}):
$Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)$ carries a canonical
$E_2$-chiral structure;
(ii) the action of $W(\SCchtop)$ on the bulk--boundary pair by
colored brace operations (Tamarkin 2000 for the uncoloured
$E_2$; chiral upgrade via factorization brace);
(iii) Koszul-dual presentation: $W(\SCchtop)=\Omega(B(\SCchtop))$
via Edge~(1)--(2); the action factors through the cofibrant
replacement.

$\Phi$-independence at cohomology: follows from the
GRT$_1$-action on brace operations preserving the cohomology class
(Willwacher 2015 for uncoloured $E_2$; coloured analogue via
coloured Kontsevich formality).
\end{proof}

\subsection{Edge (3)--(4): factorization--BV/BRST}\label{subsec:edge-34}

\begin{theorem}[Observables of BV theory present the factorization
face, on the Costello--Gwilliam axiom-compatible locus;
\ClaimStatusProvedHere]\label{thm:edge-34}%
\index{heptagon!edge (3)--(4)}%
\index{Costello--Gwilliam axioms!compatibility scope}
For a bulk--boundary BV theory on $X\times\mathbb{R}_{\geq 0}$ with
boundary condition $\cA$ \emph{satisfying the Costello--Gwilliam
boundary-condition axioms} (free quadratic action with elliptic
propagator at leading order; Dirichlet-compatible boundary
restriction; gauge-fixing extending across the boundary), the
Costello--Gwilliam observables $\Obs^{\mathrm{q}}(\cdot)$ assemble
into a factorization algebra on $X\times\mathbb{R}_{\geq 0}$ whose
restriction to the closed half is $\Obs^{\mathrm{ch}}_\cA$ of
Face~(3). The QME for the bulk--boundary action is the open/closed
Maurer--Cartan equation for the pair $(\cA,Z^{\mathrm{der}}_{\mathrm{ch}}(\cA))$.

\smallskip
\textbf{Scope.} The Costello--Gwilliam boundary axioms are verified for
class~G (Heisenberg, abelian gauge sector $bb$) and class~L (affine
Kac--Moody $V_k(\fg)$ at non-critical level $k\neq -h^\vee$, semisimple
$\fg$). They are open at chain level for class~C (free $\beta\gamma$)
when the boundary condition mixes left and right movers, and for
class~M (Virasoro, $\cW_N$ on the original non-weight-completed
complex) where the BRST primitive of the conformal vector is
conjectural; see Remark~\ref{rem:edge-34-scope}.
\end{theorem}

\begin{proof}
Costello--Gwilliam \textit{Factorization Algebras in QFT}~Vol.~II
Theorem~11.3.3 (closed sector, free BV with elliptic propagator);
the bulk--boundary extension is Butson--Yoo 2018 (arXiv:1807.02341);
the QME/MC identification is the logarithmic $\SCchtop$-algebra
structure of Definition~\ref{def:log-SC-algebra}
(Chapter~\ref{ch:bv-construction}): the QME reads
$\partial S_{\mathrm{bulk-bdy}} + \tfrac{1}{2}\{S_{\mathrm{bulk-bdy}},
S_{\mathrm{bulk-bdy}}\}=0$ with coloured BV bracket extending the bulk
BV bracket by a boundary-to-bulk bracket, and this is the MC equation
for the pair, not for $\cA$ alone.

The scope-restricting hypotheses (free quadratic kernel, elliptic
propagator, Dirichlet compatibility) are verified for class~G/L:
abelian Heisenberg has trivial elliptic kernel
$\partial_{\bar z}\,\partial_t$; affine Kac--Moody has the
Chern--Simons quadratic kernel at non-critical level (the $1/(k+h^\vee)$
normalisation of the Sugawara propagator is finite). For class~C, the
$\beta\gamma$ boundary condition mixes the holomorphic
$\beta$ and antiholomorphic $\bar\gamma$ at $\partial\R_{\geq 0}$,
breaking Dirichlet compatibility unless the boundary condition is
purely holomorphic; for class~M, the cubic-or-higher Virasoro vertex
violates the free-quadratic hypothesis required by CG~Theorem 11.3.3.
\end{proof}

\begin{remark}[Edge (3)--(4) scope; class~G/L provable, class~C/M open]
\label{rem:edge-34-scope}%
\index{Edge (3)--(4)!scope!class G/L provable, class C/M open}
The Costello--Gwilliam axioms for the bulk--boundary BV/BRST
extension to the heptagon are:
\begin{itemize}
\item \textbf{Class~G} (Heisenberg $\cH_k$): provable. The bulk
action is purely quadratic; the boundary condition is the free
$\Eone$-algebra structure on the modes; Dirichlet at infinity is
trivial.
\item \textbf{Class~L} (affine Kac--Moody $V_k(\fg)$ at non-critical
$k$): provable for semisimple $\fg$. CG axioms verified by the
Chern--Simons BV theory at level $k$ with boundary $\fg$-WZW; the
$1/(k+h^\vee)$ propagator finite at non-critical $k$.
\item \textbf{Class~C} (free $\beta\gamma$): \textbf{open at chain level
when boundary mixes movers}. The bulk action is free quadratic but the
holomorphic--antiholomorphic boundary condition is not literally
Dirichlet; one must work with the $\beta\gamma$-vacuum boundary
condition, for which the CG axioms hold but require an extension of
the cosheaf model. The cohomological identification is acceptable;
chain-level identification is conjectural.
\item \textbf{Class~M} (Virasoro $\mathrm{Vir}_c$, $\cW_N$): \textbf{open
in general}. Virasoro is not a free BV theory; CG~Theorem 11.3.3 does
not apply directly. The chain-level identification of Edge~(3)--(4)
on class~M depends on the BRST primitive $G(z)=[Q_{\mathrm{tot}},T(z)]$
which is conjectural in general
(Conjecture~\ref{conj:e-infinity-specialisation-Winfty}).
\item \textbf{Critical Kac--Moody} ($k=-h^\vee$): \textbf{fails}. The
Sugawara normalisation $1/(k+h^\vee)$ diverges; the BV propagator is
ill-defined; CG~Theorem 11.3.3 does not extend. The heptagon stays
intact, but Edge~(3)--(4) is replaced by a Feigin--Frenkel-centre
relation; see Remark~\ref{rem:heptagon-critical}.
\end{itemize}
The class~G/L scope is sufficient for all heptagon witnesses
involving Heisenberg, affine Kac--Moody at non-critical level, and
their Sugawara descendants; the open class~C/M cases are recorded as
conditional dependencies of the heptagon's full applicability.
\end{remark}

\begin{proposition}[Heptagon edge $(3)\leftrightarrow(4)$, propositional
form; \ClaimStatusProvedHere]\label{prop:heptagon-edge-34}%
\index{heptagon!edge (3)--(4) propositional form}%
The factorization presentation (Face~(3)) and the BV/BRST presentation
(Face~(4)) of $\SCchtop$ determine the same $\infty$-operadic data on
$(\cA, Z^{\mathrm{der}}_{\mathrm{ch}}(\cA))$. Equivalently, the
$L_\infty$-convolution algebra $\fg^{\,\SCchtop}_{\cA,Z}$ of Face~(5)
is quasi-isomorphic, through the BV/BRST presentation of Face~(4), to
the factorization-homology universal brace action of Face~(3); the
quasi-isomorphism is implemented by the Swiss-cheese formality of
Kontsevich--Tamarkin after fixing a Drinfeld associator $\Phi$.
\end{proposition}

\begin{proof}
This is the propositional restatement of Theorem~\ref{thm:edge-34}
used by citation sites in Chapter~\ref{ch:chiral-higher-deligne} and
Chapter~\ref{ch:brace}; the content is identical. The factorization
algebra $\Obs^{\mathrm{q}}$ of the bulk--boundary BV theory restricts
on the closed half to the factorization face $\Obs^{\mathrm{ch}}_\cA$
(Theorem~\ref{thm:edge-34}), and on a cofibrant replacement the QME
equals the convolution MC (Theorem~\ref{thm:edge-45}); composing these
two quasi-isomorphisms with the coloured Swiss-cheese formality map
(Getzler--Jones~1994, coloured variant via Vallette~2014 colored Koszul
duality plus homotopy transfer) produces the stated quasi-isomorphism
between the factorization brace action and the convolution
$L_\infty$-algebra. Strictness on a fixed semifree model and
associator-dependence follow from Theorem~\ref{thm:edge-45} and the
$\mathrm{GRT}_1$-torsor structure of Swiss-cheese formality.
\end{proof}

\subsection{Edge (4)--(5): BV/BRST--convolution}\label{subsec:edge-45}

\begin{theorem}[QME equals convolution MC; \ClaimStatusProvedHere]
\label{thm:edge-45}%
\index{heptagon!edge (4)--(5)}
On a fixed semifree model (i.e.\ cofibrant replacement of the
BV/BRST complex), the QME of Face~(4) is strictly equal to the
Maurer--Cartan equation in $\fg^{\,\SCchtop}_{\cA,Z}$ of Face~(5).
Passing to quasi-iso classes recovers the chain-level
$\infty$-quasi-isomorphism between the BV/BRST and convolution
presentations.
\end{theorem}

\begin{proof}
Fix a semifree model $\cE\to\SCchtop$ of coloured $\infty$-operads
and a coordinate system on the BV complex. The QME in those
coordinates is a polynomial in the structure constants of
$\SCchtop$ acting on the BV field content, which is, by
construction of the convolution algebra, the MC equation in
$\fg^{\,\SCchtop}_{\cA,Z}=\mathrm{Hom}^{\mathrm{col}}(B(\SCchtop),
\End^{\mathrm{col}}_{(\cA,Z)})$. The equality is strict on the
semifree model; on arbitrary models the two sides are related by
homotopy transfer (HPL). Gauge-fixed chain-level identity closes
the chain-level BV/convolution identification.
\end{proof}

\begin{proposition}[Heptagon edge $(4)\leftrightarrow(5)$, propositional
form; \ClaimStatusProvedHere]\label{prop:heptagon-edge-45}%
\index{heptagon!edge (4)--(5) propositional form}%
The BV/BRST presentation (Face~(4)) and the convolution
$L_\infty$-presentation (Face~(5)) of $\SCchtop$ yield the same
$\infty$-operadic datum on $(\cA, Z^{\mathrm{der}}_{\mathrm{ch}}(\cA))$.
Concretely: the quantum master equation on the observables
$\Obs^{\mathrm{q}}(U)$ descends to the Maurer--Cartan equation in
$\fg^{\,\SCchtop}_{\cA,Z}$ via a gauge-fixed semifree model; the
quasi-isomorphism class agreement is proved by Stasheff cancellation
on the evaluation-generated core of the coloured operad.
Consequently, on the topologised cohomology — unconditional for
affine Kac--Moody $V_k(\fg)$ at non-critical level $k\neq -h^{\vee}$,
conjectural for general chiral algebras carrying a conformal vector
(Conjecture~\ref{conj:heptagon-collapse-general}) — the Dunn-additive
product
$E_2^{\mathrm{hol}}\otimes_{\mathrm{Dunn}} E_1^{\mathrm{top}}$
of the topologised cores identifies, through Face~(4)--Face~(5),
with a single $E_3^{\mathrm{top}}$-action on
$Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)$. The generic
$\SCchtop$-datum (without a conformal vector) does not collapse;
Dunn additivity applies only post-topologisation.
\end{proposition}

\begin{proof}
The first sentence is the propositional restatement of
Theorem~\ref{thm:edge-45}; content is identical. For the topologised
Dunn claim: the convolution algebra $\fg^{\,\SCchtop}_{\cA,Z}$
carries, after topologisation of the closed colour (BRST
$Q_{\mathrm{tot}}$-cohomology making $C$-translations exact), an
$E_2$-topological structure from the closed-colour braces and an
$E_1$-topological structure from the conformal-weight $L_0$-flow
along $\mathbb{R}_{\geq 0}$. Dunn additivity (Lurie HA~5.1.2.2) applies
to the topologised $E_n$-cores in a symmetric monoidal ambient; its
application to $\SCchtop$ pre-topologisation is not automatic, as
$\SCchtop$ is a coloured dioperad with directional restriction rather
than an $E_n$-operad in the ambient sense. Post-topologisation — via
Sugawara $T(z)=[Q_{\mathrm{tot}},G(z)]$ for affine Kac--Moody at
non-critical level (Chapter~\ref{ch:topologization}) — the
combination $E_2^{\mathrm{hol}}\otimes_{\mathrm{Dunn}} E_1^{\mathrm{top}}\simeq
E_3^{\mathrm{top}}$ is Dunn in the usual sense and produces the
$E_3^{\mathrm{top}}$-action. Stasheff cancellation on the
evaluation-generated core produces the quasi-iso class equality on
arbitrary models at that topologised level.
\end{proof}

\subsection{Edge (5)--(6): convolution--Drinfeld centre}\label{subsec:edge-56}

\begin{theorem}[Convolution MC presents categorical half-braiding;
\ClaimStatusProvedHere]\label{thm:edge-56}%
\index{heptagon!edge (5)--(6)}
The Maurer--Cartan locus of the convolution $L_\infty$-algebra
$\fg^{\,\SCchtop}_{\cA,Z}$ is in natural bijection with the space of
half-braidings $\sigma_\cA(z)$ compatible with the
$\SCchtop$-datum on $(\cA,Z^{\mathrm{der}}_{\mathrm{ch}}(\cA))$, hence
with the central objects of
$\cZ(\Rep_{\mathrm{fact}}(\cA))$ of Face~(6):
\begin{equation}
  \MC\bigl(\fg^{\,\SCchtop}_{\cA,Z}\bigr)
  \;\simeq\;
  \bigl(\mathrm{objects\ of\ }
  \cZ(\Rep_{\mathrm{fact}}(\cA))\bigr).
\end{equation}
Gauge equivalence in convolution corresponds to isomorphism in
the Drinfeld centre.
\end{theorem}

\begin{proof}
A Maurer--Cartan element $\alpha\in\fg^{\,\SCchtop}_{\cA,Z}$
determines a coherent family of operations of $\SCchtop$ on
$(\cA,Z)$; the mixed-sector component of $\alpha$ is precisely a
half-braiding (by Proposition~\ref{prop:half-braiding-construction};
the mixed sector of $\SCchtop$'s Koszul dual is the shuffle-mixed
data, which translates to half-braidings via
Definition~\ref{def:chiral-half-braiding}). Conversely, a central
object of $\cZ(\Rep_{\mathrm{fact}}(\cA))$ gives a half-braiding
compatible with $\Delta_z$, which encodes an MC element of the
mixed sector of $\fg^{\,\SCchtop}_{\cA,Z}$. Gauge equivalence in
convolution is homotopy of MC elements
(Hinich~1997, Getzler~2009); this corresponds to isomorphism in
$\cZ(\Rep_{\mathrm{fact}}(\cA))$ by Theorem~\ref{thm:drinfeld-centre-sc-face}.
The statement is an equivalence of $\infty$-groupoids
(objects and gauge classes); the upgrade to an equivalence of
$E_2$-monoidal $\infty$-categories, which matches the braided
structure on both sides, is the content of
Theorem~\ref{thm:drinfeld-centre-sc-face} and rests on the
factorization-functor naturality of $\sigma_\cA(z)$ established in
Proposition~\ref{prop:half-braiding-construction}~(iv) (OPE-pole
bounds on $\Delta_z$).
\end{proof}

\subsection{Edge (6)--(7): Drinfeld centre--derived algebraic geometry}\label{subsec:edge-67}

\begin{theorem}[Drinfeld centre is the
$\infty$-stack of centrally-compatible $\SCchtop$-data;
\ClaimStatusProvedHere]\label{thm:edge-67}%
\index{heptagon!edge (6)--(7)}
The Drinfeld centre $\cZ(\Rep_{\mathrm{fact}}(\cA))$ of Face~(6)
is the loop space of $\cM_{\SCchtop,X}$ at the point
$(\cA,Z^{\mathrm{der}}_{\mathrm{ch}}(\cA))$, in the sense of
$E_2$-group objects:
\begin{equation}
  \label{eq:edge-67-equivalence}
  \cZ\bigl(\Rep_{\mathrm{fact}}(\cA)\bigr)
  \;\simeq\;
  \Omega_{(\cA,Z)}^{E_2}\,\cM_{\SCchtop,X}
\end{equation}
as braided monoidal $\infty$-categories. The $1$-shifted symplectic
form $\omega_{\SCchtop}$ of Face~(7) restricts on the loop space to
the categorical braiding of Face~(6).
\end{theorem}

\begin{proof}
The mapping-stack tangent $T_{(\cA,Z)}\cM_{\SCchtop,X}$ is
$\fg^{\,\SCchtop}_{\cA,Z}$ by
Theorem~\ref{thm:convolution-lagrangian}. Its $E_2$-group
completion (iterated based loop space) is, by the chiral higher
Deligne theorem (Theorem~\ref{thm:chiral-higher-deligne} of
Chapter~\ref{ch:theory-hochschild}) applied to the pair
$\bigl(Z^{\mathrm{der}}_{\mathrm{ch}}(\cA),\cA\bigr)$, the
$E_2$-chiral algebra whose factorization module category is
$\cZ(\Rep_{\mathrm{fact}}(\cA))$. The coloured-dioperad input
is the specific case handled by Theorem~\ref{thm:chiral-higher-deligne};
we do not invoke a general coloured Deligne conjecture for
arbitrary coloured operads. This gives the equivalence
in~\eqref{eq:edge-67-equivalence}.

The restriction of $\omega_{\SCchtop}$ to the loop space:
PTVV Corollary~2.6 states that for a $k$-shifted symplectic derived
stack $\cX$, the loop space $\Omega\cX$ is $(k-1)$-shifted
symplectic; applied to $k=1$, $\Omega\cM_{\SCchtop,X}$ is
$0$-shifted symplectic, i.e.\ carries a symmetric non-degenerate
pairing in degree $0$, which is exactly the categorical braiding
pairing (symmetric at degree $0$, producing the full hexagon at
all orders via Deligne's coherence). This identifies the braiding
of Face~(6) with the loop-space shifted symplectic form.
\end{proof}

\subsection{Edge (7)--(1): derived algebraic geometry--operadic}\label{subsec:edge-71}

\begin{theorem}[Tangent complex of $\cM_{\SCchtop,X}$ recovers the
operadic face; \ClaimStatusProvedHere]\label{thm:edge-71}%
\index{heptagon!edge (7)--(1)}
At every closed point $(\cA,Z)\in\cM_{\SCchtop,X}$, the
tangent dg-Lie algebra equipped with its coloured operad structure
is a strict model of $\SCchtop$ on the pair
$(\cA,Z^{\mathrm{der}}_{\mathrm{ch}}(\cA))$. The global
sections of the structure sheaf
$\mathcal{O}_{\cM_{\SCchtop,X}}$ recover the coloured cobar
construction $\Omega(B(\SCchtop))$ on any local model, hence present
Face~(1) directly.
\end{theorem}

\begin{proof}
Tangent dg-Lie identification is
Theorem~\ref{thm:convolution-lagrangian}. The coloured $\SCchtop$-operad
structure on the tangent complex is inherited from the operadic
ambient of Definition~\ref{def:moduli-sc-ch-top}: the mapping-stack
target $B\SCchtop\mathrm{-Alg}$ has tangent $\SCchtop$ itself (up to
homotopy; Lurie HA~5.3.1.11). Pulling back along the tautological
evaluation map, the tangent of $\cM_{\SCchtop,X}$ carries
$\SCchtop$ up to $\infty$-quasi-isomorphism.

Global sections of $\mathcal{O}_{\cM_{\SCchtop,X}}$: by the
Kan extension of the tautological family to the moduli stack,
the Chevalley--Eilenberg complex of the tangent dg-Lie is
$\mathcal{O}^{\mathrm{loc}}_{\cM_{\SCchtop,X}}$ (locally
around any closed point; PTVV 2013~\S2.2 and Deligne's
\textit{Equivalence of Deformation Complexes}). The
CE complex is, by construction, $\Omega(B(\SCchtop))=W(\SCchtop)$
applied to the tangent, hence recovers Face~(1) via the edge
(1)--(2) Koszul-dual equivalence.
\end{proof}

\begin{theorem}[Heptagon of $\SCchtop$; \ClaimStatusProvedHere]
\label{thm:heptagon-closed}%
\index{heptagon!seven-way equivalence|textbf}
The seven faces (1)--(7) of $\SCchtop$ are pairwise
$\infty$-quasi-isomorphic as coloured $\infty$-operads after
fixing a Drinfeld associator $\Phi\in\mathrm{GRT}_1(\mathbb{Q})$,
with edges given by Theorems~\ref{thm:edge-12}, \ref{thm:edge-23},
\ref{thm:edge-34}, \ref{thm:edge-45}, \ref{thm:edge-56},
\ref{thm:edge-67}, \ref{thm:edge-71}. The seven-way equivalence is
$\Phi$-independent at the level of cohomology; strictly chain-level
presentations track a single $\Phi$.
\end{theorem}

\begin{proof}
Compose the seven edges around the heptagon in either direction.
Each edge is an $\infty$-quasi-isomorphism
(Convention~\ref{conv:edge-regime}); composition preserves
$\infty$-quasi-isomorphisms (Fresse 2009, or direct verification in
the colored $\infty$-operad model category); the composed
self-equivalence of Face~(1) is homotopic to the identity because
each edge is canonical modulo associator choice. $\Phi$-independence
at cohomology follows from the GRT$_1$-action on each edge
(Willwacher 2015; coloured extension via coloured brace
formality).
\end{proof}

\begin{remark}[Chain-level vs cohomological; scope]
\label{rem:heptagon-chain-vs-cohom}
Theorem~\ref{thm:heptagon-closed} is proved at two levels, with
distinct scopes:
\begin{enumerate}[label=\textup{(\alph*)}]
  \item \emph{Cohomological closure.} The seven-way equivalence of
  the homotopy classes of coloured $\infty$-operad maps holds
  unconditionally after fixing $\Phi$, for any $E_1$-chiral algebra
  $\cA$ satisfying the OPE-pole bounds of
  Proposition~\ref{prop:half-braiding-construction}~(iv).
  Associator transport is a $\mathrm{GRT}_1$-torsor, and
  $\Phi$-independence at cohomology is Willwacher's theorem
  extended to the coloured dioperad by coloured brace formality.
  \item \emph{Chain-level closure.} Strict chain-level equivalences
  hold on the evaluation-generated Koszul locus: the bar complex is
  concentrated in bar-weight $1$, the coloured cobar map is strict,
  and the Swiss-cheese formality datum can be chosen compatibly
  across the seven faces. Off this locus, only
  $\infty$-quasi-isomorphism is available, and each edge requires
  a homotopy transfer.
  \item \emph{Topologised regime.} When $\cA$ carries a conformal
  vector at non-critical level and the topologisation of
  Chapter~\ref{ch:topologization} applies, the heptagon collapses
  via Theorem~\ref{thm:heptagon-collapse}; Dunn additivity
  combines the closed $E_2$-core with the open $E_1$-topological
  core into an $E_3$-topological action. This collapse is
  unconditional for affine Kac--Moody at $k\neq -h^\vee$ and
  conjectural for general conformal chiral algebras.
\end{enumerate}
Prose elsewhere in this chapter that reads ``Dunn additivity
combines the two colours into $E_3$'' is always understood in
sense~(c), on the topologised cohomology; the generic heptagon has
seven distinct faces.
\end{remark}

\begin{remark}[Diagonals as compositions, not independent
content]\label{rem:heptagon-diagonals}%
\index{heptagon!diagonals are compositions, not independent}
The fourteen diagonals of the heptagon are \emph{compositions of the
seven adjacent edges} — they are not counted as independent content,
and the heptagon contains seven theorems (one per edge), not
twenty-one. Each diagonal does admit an independent literature
construction that the heptagon organises in one place; we list
the most-cited:
Face~(1)--(3) (operadic action on factorization algebra via chiral
Deligne--Tamarkin; Chapter~\ref{ch:theory-hochschild});
Face~(2)--(5) (Koszul dual as tangent dg-Lie; Vallette Koszul duality);

\section{Conformal vector and the collapse to $E_3$-topological}
\label{sec:heptagon-conformal-vector}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

The heptagon is the \emph{generic} picture. Adding a conformal vector
to the boundary collapses two faces of the heptagon into one; the
resulting topologised structure is $E_3$-topological, not $\SCchtop$
(Chapter~\ref{ch:topologization}). Topologisation is unconditional
for affine Kac--Moody $V_k(\fg)$ at non-critical level $k\neq -h^\vee$
(Sugawara $T(z)=[Q_{\mathrm{tot}},G(z)]$, chain-level BRST
primitive explicit); it is conjectural for general chiral algebras
carrying a conformal vector, cohomological rather than chain-level,
and may genuinely fail at the chain level on the class-$M$ complex.

\begin{theorem}[Collapse with conformal vector at affine Kac--Moody
non-critical level; \ClaimStatusProvedHere]
\label{thm:heptagon-collapse}%
\index{topologisation!collapses heptagon|textbf}%
\index{$E_3$-topological!heptagon collapse}%
%

Let $\cA=V_k(\fg)$ be an affine Kac--Moody chiral algebra at
non-critical level $k\neq -h^{\vee}$, with Sugawara conformal
vector $T(z)=\tfrac{1}{2(k+h^{\vee})}:\!J^aJ_a\!:(z)$. On the
BRST cohomology of the total differential $Q_{\mathrm{tot}}$ of the
$3$d holomorphic Chern--Simons BV quantisation whose boundary is
$\cA$ (Chapter~\ref{ch:topologization}), the bicoloured operad
$\SCchtop$ reduces in \emph{two distinct steps}:
\begin{enumerate}[label=\textup{(\roman*)}]
  \item \emph{Topologisation of the closed colour.} The Sugawara
  primitive $G(z)=\tfrac{1}{2(k+h^{\vee})}{:}J^a\bar c_a{:}(z)$
  provides a chain-level homotopy $h_\omega :=[G,-]$ on
  $\Obs^q_\bulk$ satisfying the coboundary equation
  \begin{equation}\label{eq:h-omega-coboundary-km}
  [Q_{\mathrm{tot}},\,h_\omega]\;=\;\partial_z,
  \end{equation}
  witnessing that the holomorphic translation derivation $\partial_z$
  on the closed-colour disk operad is $Q_{\mathrm{tot}}$-exact. On
  $Q_{\mathrm{tot}}$-cohomology the closed-colour factorisation
  algebra is locally constant, upgrading $E_2^{\mathrm{hol}}$ to
  $E_2^{\mathrm{top}}$ by Lurie~\cite{HA}\,Theorem~5.4.5.9 /
  Ayala--Francis~\cite{AF15}.
  \item \emph{Dunn additivity on the single-colour operad.} After
  (i), both colours of $\SCchtop$ are topological; the mixed
  bulk-to-boundary operations (directionally restricted in the
  bicoloured operad) degenerate into symmetric little-$3$-disc
  operations in $Q_{\mathrm{tot}}$-cohomology. Dunn's theorem
  (Dunn~1988; Lurie~\cite{HA}\,Theorem~5.1.2.2) then gives
  $E_2^{\mathrm{top}}\otimes E_1^{\mathrm{top}}\simeq
  E_3^{\mathrm{top}}$ of \emph{single-colour}
  $E_n$-operads.
\end{enumerate}
The composite of the two steps sends the bicoloured
$\SCchtop$-operad on chains to the single-colour little-$3$-disc
operad $\FM_3(\R^3)$ on $Q_{\mathrm{tot}}$-cohomology. The
heptagon's seven faces collapse to:
\begin{enumerate}
  \item operadic face for $E_3$-topological $\FM(\mathbb{R}^3)$;
  \item Koszul-dual face $E_3^{!}\simeq E_3\{3\}$;
  \item factorization face on $X\times\mathbb{R}\simeq\mathbb{R}^3$
  (after choosing a point on $X$);
  \item BV/BRST face of the resulting $3$-dimensional topological
  field theory;
  \item convolution $L_\infty$-algebra for $E_3^{!}$-algebras;
  \item Drinfeld centre of a module $1$-category over the
  mode algebra of $\cA$;
  \item derived stack of $3$-dimensional topological algebras.
\end{enumerate}
The seven $E_3$-topological faces remain quasi-isomorphic under the
topologisation functor, at the level of $Q_{\mathrm{tot}}$-BRST
cohomology.
\end{theorem}

\begin{proof}
The two steps are logically independent and must be executed in
order.

\medskip\noindent\textbf{Step~(i): bicoloured
$\xrightarrow{\text{topologise closed colour}}$ single-colour.}
Sugawara's identity writes
$T(z)=\tfrac{1}{2(k+h^{\vee})}{:}J^aJ_a{:}(z)$; the antighost
BRST primitive is
$G(z)=\tfrac{1}{2(k+h^{\vee})}{:}J^a\bar c_a{:}(z)$, with
$[Q_{\mathrm{tot}},G(z)]=T(z)$ at non-critical level (denominator
$2(k+h^\vee)$ invertible; Vol~I~%
Construction~\ref*{V1-constr:sugawara-antighost};
Chapter~\ref{ch:topologization}
Theorem~\ref{thm:E3-topological-km};
CFG~\cite{CFG2602}\,Proposition~4.4). The Virasoro-mode identity
$L_{-1}=T_{(0)}$ converts this into the chain-level homotopy
equation~\eqref{eq:h-omega-coboundary-km}: for any
$Q_{\mathrm{tot}}$-closed bulk observable $\cO$,
\[
\partial_z\cO
 \;=\;[T,\cO]
 \;=\;\bigl[[Q_{\mathrm{tot}},G],\cO\bigr]
 \;=\;\bigl[Q_{\mathrm{tot}},[G,\cO]\bigr],
\]
so $h_\omega:=[G,-]$ satisfies
$[Q_{\mathrm{tot}},h_\omega]=\partial_z$ on
$\Obs^q_\bulk$. This is exactly the chain-level witness the
$(\infty,1)$-categorical topologisation statement requires
(the chain-level and $(\infty,1)$-categorical lanes are distinct
theorems, both load-bearing, informing each other through this
explicit homotopy). Hence the closed-colour factorisation algebra on
$\mathbb{C}$ is locally constant on $Q_{\mathrm{tot}}$-cohomology,
promoting $E_2^{\mathrm{hol}}\to E_2^{\mathrm{top}}$
(Lurie~\cite{HA}\,Theorem~5.4.5.9;
Ayala--Francis~\cite{AF15}\,Theorem~2.24). The bicoloured
$\SCchtop$-operad becomes a \emph{single-colour} pair
$(E_2^{\mathrm{top}},E_1^{\mathrm{top}})$ of topological operads
coupled by the mixed bulk-to-boundary operations.

\medskip\noindent\textbf{Step~(ii): Dunn additivity on
single-colour topological operads.} After Step~(i) the bulk
factorisation algebra is locally constant $E_2^{\mathrm{top}}$ on
$\C\simeq\R^2$ and the boundary factorisation algebra is already
$E_1^{\mathrm{top}}$ on $\R$. The bicoloured $\SCchtop$-operad
carries three kinds of operations: pure closed ($E_2^{\mathrm{hol}}$
in chains, $E_2^{\mathrm{top}}$ on $Q_{\mathrm{tot}}$-cohomology by
Step~(i)), pure open ($E_1^{\mathrm{top}}$), and mixed
bulk-to-boundary (Voronov 1998 Swiss-cheese; Hoefel 2009 presentation;
Hoefel--Livernet 2012 bicoloured Koszul version, asymmetric). The
Costello--Gwilliam~\cite{CG21} factorisation-algebra framework
(Vol.~I~Theorem~3.4.9 product of factorisation algebras) shows that
on $Q_{\mathrm{tot}}$-cohomology the product factorisation algebra
on $X\times\R$, after passage to local-constancy on the
$\C\simeq\R^2$ factor via Step~(i), is determined by its value on
a single $3$-dimensional disc $D^3\subset\R^3$ after identifying
$X$ with $\C\simeq\R^2$ via any point-choice. The mixed operations are then
absorbed into the homotopy-product operad, because the pullback of
the $\SCchtop$-algebra structure along
$\R^3\cong\R^2\times\R\to X\times\R$ is the usual little-$3$-disc
operad action on a single colour (Ayala--Francis~\cite{AF15}%
\,Theorem~2.24 for the $\boxtimes$-product of locally constant
factorisation algebras on a product manifold). Dunn's additivity
theorem (Dunn~1988; Lurie~\cite{HA}\,Theorem~5.1.2.2, the
$(\infty,1)$-categorical form) identifies this tensor product with
$E_3^{\mathrm{top}}\simeq\FM_3(\R^3)$ on the nose:
\[
  E_2^{\mathrm{top}}\otimes_{\mathrm{Op}_{\infty}}E_1^{\mathrm{top}}
   \;\xrightarrow{\;\sim\;}\;
   E_3^{\mathrm{top}}.
\]
The bicoloured-to-single-colour reduction happened in Step~(i); the
absorption of mixed operations is the
Costello--Gwilliam~\cite{CG21}\,Vol.~I~Theorem~3.4.9 +
Ayala--Francis~\cite{AF15}\,Theorem~2.24 product-factorisation
step; Dunn is then applied to the genuinely single-colour operad
$E_2^{\mathrm{top}}\otimes_{\mathrm{Op}_\infty} E_1^{\mathrm{top}}$.
The $\mathsf{SC}^{\mathrm{ch,top}}$-vs-$E_3$ scope declaration is
respected: Dunn is \emph{not} invoked on the bicoloured
$\SCchtop$.

\medskip\noindent\textbf{Heptagon closure.}
Each heptagon face inherits the topologisation via its edge to
Face~(3) or Face~(1); the resulting $E_3$-topological faces are
equivalent among themselves by the $E_3$-operad Koszul duality
$E_3^!\simeq E_3\{3\}$ (Fresse, \emph{Homotopy of Operads and
Grothendieck--Teichm\"uller Groups}, Vol.~II, AMS 2017;
Ginzburg--Kapranov~1994). The chain-level witness in Step~(i) and
the operad-level Dunn tensor in Step~(ii) are independent
ingredients; conflating them (e.g.\ writing
``Dunn $E_2^{\mathrm{hol}}\otimes E_1^{\mathrm{top}}$'' in a single
step) silently equates the bicoloured $\mathsf{SC}^{\mathrm{ch,top}}$
operad with $E_3$, which fails (see
Remark~\ref{rem:bulk-bdry-pattern-277-scope}).
\end{proof}

\begin{remark}[Why the homotopy $h_\omega$ is named separately from
the BRST primitive $G(z)$]\label{rem:h-omega-vs-G-primitive}%
%
\index{BRST primitive!vs chain homotopy}%
The identity $[Q_{\mathrm{tot}},G(z)]=T(z)$ is a BRST identity on
\emph{states} (the antighost primitive for the stress tensor). The
identity $[Q_{\mathrm{tot}},h_\omega]=\partial_z$
is a chain identity on \emph{operators} (the homotopy for the
holomorphic translation derivation). They are related by the
Virasoro mode identification $L_{-1}=T_{(0)}$ and the Leibniz rule
$h_\omega:=[G,-]\Rightarrow[Q_{\mathrm{tot}},h_\omega]
 =[[Q_{\mathrm{tot}},G],-]=[T,-]=\partial_z$, but only after
passing through the mode action. The operad-level statement
(Step~(ii) Dunn) sees the operator identity, not the state
identity; conflating them trades an eight-line chain-level proof
for a two-line citation chain that drops the mode-action
bookkeeping. Pattern~299 records this distinction; this remark
exhibits the explicit bridge.
\end{remark}

\begin{remark}[Scope of the heptagon
collapse]\label{rem:heptagon-collapse-bicoloured-scope}%
%
%
Theorem~\ref{thm:heptagon-collapse} as stated above respects the
injunction: Dunn additivity is applied in Step~(ii) to the
\emph{single-colour} operad that Step~(i) constructs from the
bicoloured $\SCchtop$ on $Q_{\mathrm{tot}}$-cohomology, never to
$\SCchtop$ itself. This matches the Hoang--Tradler--Zeinalian
(HTZ) statement of Swiss-cheese Koszul duality: the bicoloured
$\SCchtop$-operad has an \emph{asymmetric} Koszul dual
$\mathrm{(Lie,Ass,\text{shuffle-mix})}$ (Hoefel~2009;
Hoefel--Livernet~2012) which is \emph{not} self-dual.
Only after topologisation of the closed colour does
Koszul duality become the symmetric $E_3^!\simeq E_3\{3\}$
duality. The chapter's coloured-Koszul face~(2) is the
pre-topologisation bicoloured-Koszul statement; the heptagon
collapse replaces it with the single-colour $E_3$-Koszul face.
\end{remark}

\begin{remark}[The heptagon survives the generic case]
\label{rem:heptagon-survives-generic}
For chiral algebras without a conformal vector (class $E_1$-chiral

conformal vector), all seven faces of the heptagon remain distinct
and the heptagon is the final structure. This is the generic case;
$E_3$-topologisation is the exceptional special case triggered by a
conformal vector at non-critical level. The programme's climax is
the heptagon, not the topologised special case.

\begin{remark}[Critical level]\label{rem:heptagon-critical}

exist. The heptagon stays intact, but the boundary chiral algebra
is enriched by the Feigin--Frenkel centre of
$\hat{\mathfrak{g}}_{-h^{\vee}}$; the fifth shadow class FF
(Feigin--Frenkel) of Infinite Fingerprint Classification emerges
(Platonic Reconstitution). Topologisation fails; the heptagon is
the complete structural statement.
\end{remark}

\begin{proposition}[Heptagon-face triviality of the
