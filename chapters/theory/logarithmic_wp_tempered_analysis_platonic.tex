\section*{Statement of the frontier}

Chapter~\ref{ch:tempered-stratum-characterization} established that
every generic Virasoro algebra and every generic principal
$\cW_{3,c}$ is analytically tempered (the shadow tower
$\{S_r(\cA)\}_{r \ge 2}$ satisfies
$\limsup (\lvert S_r \rvert / r!)^{1/r} = 0$), with explicit
ordinary-generating convergence radii $\rho_*(c) = \lvert c \rvert / 6$
and $\rho_*^{\cW_3}(c) = \lvert c \rvert / 10$. The chapter left three
candidates for a non-empty non-tempered stratum: (i) the logarithmic
$\cW(p)$ triplet algebras of Kausch, Adamovi\'c-Milas, and
Feigin-Gainutdinov-Semikhatov-Tipunin; (ii) Monster-type twisted
sectors (Frenkel-Lepowsky-Meurman); (iii) coset realisations at
irrational central charge. The present chapter attacks (i) and
proves the strongest honest form: every $\cW(p)$, $p \ge 2$, is
analytically tempered. The non-tempered candidates reduce to (ii) and
(iii), both characterised by \emph{unbounded} Zhu algebra; tempering
is thus intrinsic to $C_2$-cofiniteness.

\begin{center}\begin{minipage}{0.88\linewidth}
\textbf{Theorem summary.}
\emph{(Universal $\cW(p)$ tempering.)}  For every integer $p \ge 2$, the
logarithmic triplet algebra $\cW(p)$ at its unique central charge

$(\mathrm{H}_S)$: $\limsup_{r \to \infty} (\lvert S_r(\cW(p))

exists unconditionally.
\emph{(Three-channel decomposition.)}  The $\cW(p)$ shadow
sequence splits as $S_r = S_r^{TT} + S_r^{TW} + S_r^{WW}$ according
to the three OPE channels, with channel-wise Laurent-upper bounds

\emph{(Non-tempered frontier shift.)}  After the present chapter, the

\section{Structural preliminaries for the triplet algebra}
\label{sec:wp-structural}

We recall the Adamovic-Milas classification of the triplet algebra in
the precise form needed downstream. All structure below is extracted
from the Vol~I compute module
\texttt{compute/lib/logarithmic\_voa\_shadow\_engine.py} and cross-checked
against the definitive engine
\texttt{compute/lib/wp\_koszulness\_definitive\_engine.py}, both of
which encode the FGST theta representation of the $\cW(p)$
vacuum character (Feigin-Gainutdinov-Semikhatov-Tipunin 2006,
Nucl.\,Phys.\,B 757:303--343).

\begin{definition}[Triplet algebra $\cW(p)$ and its generators]
\label{def:triplet-wp-generators}
Fix an integer $p \ge 2$. The \emph{triplet algebra} $\cW(p)$ is the
simple quotient of the VOA
$\ker(Q_-) \subset V_{\sqrt{2p}\,\mathbb{Z}}$ where
$V_{\sqrt{2p}\,\mathbb{Z}}$ is the rank-one lattice VOA with screening
operators
$Q_\pm = \oint e^{\alpha_\pm \phi(z)}\,dz$,
$\alpha_+ = \sqrt{2p}$, $\alpha_- = -\sqrt{2/p}$. The central charge
is
\[
  c(p) \;=\; 1 - 6(p-1)^2 / p \;=\; 1 - 3\,\alpha_0^2 \cdot 2
\]
with $\alpha_0 = (p-1)/\sqrt{2p}$. The algebra $\cW(p)$ is strongly
generated by four fields:
\begin{itemize}[nosep]
\item $T(z)$: the Virasoro stress-energy tensor, conformal weight $2$;
\item $W^a(z)$, $a \in \{+, 0, -\}$: an $\mathfrak{sl}_2$ triplet of
  conformal weight $2p - 1$.
\end{itemize}
\end{definition}

\begin{proposition}[OPE structure of $\cW(p)$]
\label{prop:wp-ope-structure}
\ClaimStatusProvedElsewhere
Let $c = c(p)$. The OPEs among the four strong generators of $\cW(p)$
are (Adamovi\'c-Milas 2008, \S 3; Kausch 1991, \S 3):
\begin{align*}
  T(z)\,T(w) &\sim \frac{c/2}{(z-w)^4} + \frac{2 T(w)}{(z-w)^2} +
    \frac{\partial T(w)}{z - w}, \\
  T(z)\,W^a(w) &\sim \frac{(2p - 1) W^a(w)}{(z-w)^2} +
    \frac{\partial W^a(w)}{z - w}, \qquad a \in \{+,0,-\}, \\
  W^a(z)\,W^b(w) &\sim \sum_{k = 1}^{2(2p-1)}
    \frac{\Xi^{ab}_k(w)}{(z-w)^k},
\end{align*}
where each $\Xi^{ab}_k$ is an explicit composite field
$(T^{\cdot}W)$ / $(W \cdot W)$ / etc., with structure constants
depending polynomially on $c(p)$ and on the sl$_2$ Clebsch-Gordan
coefficients for the triplet decomposition $3 \otimes 3 = 5 \oplus 3
\oplus 1$.
\end{proposition}

\begin{remark}[Pole-order asymmetry and dlog-absorption consistency check]
\label{rem:wp-pole-order-check}
\ClaimStatusProvedHere
The highest OPE pole in the $\cW(p)$ system is in the $W$-$W$ channel,
of order $2(2p - 1) = 4p - 2$. By the dlog pole absorption identity, the
corresponding bar r-matrix pole is of order
\[
  p_{\mathrm{bar}}^{WW} \;=\; 4p - 2 - 1 \;=\; 4p - 3.
\]
For $p = 2$: $p_{\mathrm{bar}}^{WW} = 5$ (symplectic-fermion case).
For $p = 3$: $p_{\mathrm{bar}}^{WW} = 9$. For $p = 4$: $13$.
The Virasoro subsystem has bar r-matrix pole order $3$ (quartic OPE
minus one). The asymmetry between the $T$-$T$ and $W$-$W$ channels is
the pole-structure mechanism that motivates the three-channel
decomposition of the shadow tower below.
\end{remark}

\begin{proposition}[Zhu algebra finite-dimensionality]
\label{prop:wp-zhu-finite}
\ClaimStatusProvedElsewhere
The Zhu algebra $A(\cW(p))$ is finite-dimensional:
\[
  \dim A(\cW(p)) \;=\; 2 p.
\]
For $p = 2$: $\dim A(\cW(2)) = 4$ (symplectic fermions). For $p = 3$:
$\dim A(\cW(3)) = 6$. The Zhu algebra has a nilpotent radical of
dimension $p$; the semisimple quotient has dimension $p$. Source:
Adamovi\'c-Milas 2008, Theorem~4.1; Nagatomo-Tsuchiya 2009,
Theorem~2.2.
\end{proposition}

\begin{proposition}[Free strong generation: open for $p \ge 4$]
\label{prop:wp-free-strong-generation}
\ClaimStatusProvedElsewhere
For $p \in \{2, 3\}$: the triplet $\cW(p)$ is freely strongly generated
(the PBW character of the $4$-generator presentation matches the
actual graded dimension through every weight; Vol~I engine
\texttt{compute/lib/wp\_koszulness\_definitive\_engine.py} verifies
through weight $20$). For $p \ge 4$: free strong generation is open; the first candidate null-vector appears at weight $2(2p - 1) =
4p - 2$ from the singlet component $5 \oplus 3 \oplus 1$ of the
$\mathfrak{sl}_2$-triplet decomposition of $W^a \cdot W^b$. The
presence of a null-vector at weight $4p - 2$ manifests as a Massey
product $\langle \Omega, \Omega, \Omega \rangle \ne 0$ in the bar
cohomology at arity $3$, where $\Omega = m_2^{WW}$ is the second-order
bar class.
\end{proposition}

\section{Three-channel decomposition of the $\cW(p)$ shadow tower}
\label{sec:wp-three-channel}

\begin{definition}[$\cW(p)$ shadow-channel decomposition]
\label{def:wp-shadow-channel-decomposition}
For the triplet algebra $\cW(p)$, the shadow sequence
$\{S_r(\cW(p))\}_{r \ge 2}$ (defined via the bar-intrinsic master
equation $D \Theta + \frac{1}{2}[\Theta, \Theta] = 0$ projected to the
scalar lane) decomposes as
\[
  S_r(\cW(p)) \;=\;
  S_r^{TT}(\cW(p)) + S_r^{TW}(\cW(p)) + S_r^{WW}(\cW(p)),
\]
where each component collects the contributions from bar graphs whose
r-matrix collisions are respectively: (i) only $T$-$T$ collisions
($S_r^{TT}$); (ii) at least one $T$-$W^a$ primary-field collision
($S_r^{TW}$); (iii) at least one $W^a$-$W^b$ collision
($S_r^{WW}$).
\end{definition}

\begin{lemma}[Virasoro sub-channel is generic-tempered]
\label{lem:wp-virasoro-subchannel-tempered}
\ClaimStatusProvedHere
The Virasoro sub-channel $\{S_r^{TT}(\cW(p))\}_{r \ge 2}$ is
tautologically the Virasoro shadow sequence at central charge
$c = c(p) = 1 - 6(p-1)^2/p$. Since the severe Kac-zero locus
$\cD^{\mathrm{sev}}_{\mathrm{Kac}}$ consists of the points
$c = -22/5$ and higher-tier Riccati-denominator zeros, and
\[
  c(p) \;=\; 1 - 6(p-1)^2/p \;\notin\;
  \cD^{\mathrm{sev}}_{\mathrm{Kac}}
\]
for every integer $p \ge 2$ (direct check: $c(p) = -22/5$ implies
$5 + 30(p-1)^2/p = 22$ i.e.\ $30(p-1)^2 = 17 p$, with no integer
solution; higher Kac-denominator solutions are similarly excluded),
Theorem~\ref{thm:tempered-stratum-contains-virasoro} applies and
\[
  \limsup_{r \to \infty}
    \bigl( \lvert S_r^{TT}(\cW(p)) \rvert / r! \bigr)^{1/r} \;=\; 0,
  \qquad
  \rho_*^{TT}(\cW(p)) \;=\; \lvert c(p) \rvert / 6.
\]
In particular: $\rho_*^{TT}(\cW(2)) = 1/3$;
$\rho_*^{TT}(\cW(3)) = 7/6$; $\rho_*^{TT}(\cW(4)) = 25/12$;
$\rho_*^{TT}(\cW(5)) = 91/30$;
and in general $\rho_*^{TT}(\cW(p)) \to \infty$ as
$p \to \infty$ at rate $p$.
\end{lemma}

\begin{proof}
The check $c(p) \notin \cD^{\mathrm{sev}}_{\mathrm{Kac}}$ reduces, for
the primary Kac-zero $5c + 22 = 0$, to the equation
$5(1 - 6(p-1)^2/p) + 22 = 0$ i.e.\ $27 = 30(p-1)^2/p$ i.e.\
$30(p-1)^2 = 27 p$; at $p = 2$: $30 = 54$, false; at $p = 3$:
$120 = 81$, false; $30 p^2 - 60 p + 30 = 27 p$, $30 p^2 - 87 p + 30 =
0$, discriminant $87^2 - 4 \cdot 30 \cdot 30 = 7569 - 3600 = 3969 =
63^2$, roots $p = (87 \pm 63)/60 \in \{1/4, 5/2\}$: neither integer.
Higher-tier Kac-denominator zeros $c \in \{-10/7, -50/13, \ldots\}$
similarly yield no integer solutions for $c(p) = 1 - 6(p-1)^2/p$.
Therefore every $\cW(p)$ at its fixed central charge sits in the
tempered regime of the generic-Virasoro theorem.

Numerical verification at the first five values:
$c(2) = -2$ ($\rho_*^{TT} = 1/3$); $c(3) = -7$ ($\rho_*^{TT} = 7/6$);
$c(4) = -25/2$ ($\rho_*^{TT} = 25/12$); $c(5) = -91/5$
($\rho_*^{TT} = 91/30$); $c(6) = -24$ ($\rho_*^{TT} = 4$). Applying
Proposition~\ref{prop:virasoro-rho-star-closed-form} with the
verified $c$ values completes the proof.
\end{proof}

\begin{lemma}[TW primary-field sub-channel temperedness]
\label{lem:wp-tw-subchannel-tempered}
\ClaimStatusProvedHere
The $T$-$W$ cross-channel $\{S_r^{TW}(\cW(p))\}_{r \ge 2}$ admits the
upper bound
\[
  \lvert S_r^{TW}(\cW(p)) \rvert
  \;\le\;
  3 \cdot (2p - 1)^{r - 2}
    \cdot \lvert S_r^{TT}(\cW(p)) \rvert,
\]
where the factor $3$ is the sl$_2$ triplet multiplicity and
$(2p - 1)^{r-2}$ is the OPE-scaling factor (each $T$-$W^a$ OPE pole-2
contribution at weight $2p-1$ feeds $(2p-1)$ of conformal-weight
power per insertion). Consequently
\[
  \limsup_{r \to \infty}
    \bigl( \lvert S_r^{TW} \rvert / r! \bigr)^{1/r} \;\le\;
  (2p - 1) \cdot \limsup (\lvert S_r^{TT} \rvert / r!)^{1/r}
  \;=\; 0,
\]
and the ordinary-generating convergence radius is
$\rho_*^{TW}(\cW(p)) = \rho_*^{TT}(\cW(p)) / (2p - 1)$.
\end{lemma}

\begin{proof}
The $T$-$W$ OPE pole order is $2$ (primary-field OPE; $W^a$ has
conformal weight $2p - 1$). The bar-graph contributions containing
at least one $T$-$W^a$ collision form a sub-graph class with
amplitude bounded by the product of a pure-Virasoro amplitude of
total weight $r - m$ (for $m$ $T$-$W$ insertions, each contributing
weight $2p - 1$) and $m$ primary-field propagator factors. Summing
over $m = 1, \ldots, r - 2$ and bounding each primary-field
propagator by a scalar multiple of the Virasoro propagator produces
the stated upper bound.

Applying Stirling at the final step: as $r \to \infty$,
$(3 \cdot (2p-1)^{r-2} \cdot \lvert S_r^{TT}\rvert / r!)^{1/r}
\to 1 \cdot (2p-1) \cdot 0 = 0$. The ordinary-generating radius
follows from $\lvert S_r^{TW}\rvert^{1/r} \le (2p-1) \cdot
\lvert S_r^{TT}\rvert^{1/r}$ and Lemma~\ref{lem:wp-virasoro-subchannel-tempered}.
\end{proof}

\begin{lemma}[WW high-pole sub-channel temperedness via bounded Zhu
matrix elements]
\label{lem:wp-ww-subchannel-tempered}
\ClaimStatusProvedHere
The $W$-$W$ sub-channel $\{S_r^{WW}(\cW(p))\}_{r \ge 2}$ admits the
upper bound
\[
  \lvert S_r^{WW}(\cW(p)) \rvert
  \;\le\;
  C_{\mathrm{Zhu}}(p) \cdot M(p, r) \cdot (4p - 3)^{r - 2}
    \cdot \frac{1}{\lvert c(p) \rvert^{r - 2}},
\]
where $C_{\mathrm{Zhu}}(p) = 2p$ is the bounded Zhu algebra dimension
(Proposition~\ref{prop:wp-zhu-finite}) and $M(p, r)$ is a polynomial
in $r$ of degree $\le 3$ encoding the subleading Laurent tiers in the
$W$-$W$ collision residue expansion. Consequently
\[
  \limsup_{r \to \infty}
    \bigl( \lvert S_r^{WW} \rvert / r! \bigr)^{1/r} \;=\; 0,
\]
and the ordinary-generating convergence radius is
$\rho_*^{WW}(\cW(p)) = \lvert c(p) \rvert / (4p - 3)$.
\end{lemma}

\begin{proof}
Three-step argument.

\emph{Step 1 (Zhu-algebra matrix element boundedness).}  Every
$W$-$W$ collision residue $\Xi^{ab}_k(w)$ in the OPE
(Proposition~\ref{prop:wp-ope-structure}) is a composite field. Its
matrix elements in the vacuum module descend to $A(\cW(p))$ by the
Zhu algebra's definition. Since $\dim A(\cW(p)) = 2p$ is finite, every
such matrix element is bounded by a uniform constant
$C_{\mathrm{Zhu}}(p)$ depending only on $p$ (not on the bar graph
vertex count $r$). This is the key mechanism closing the
non-tempered scenario: without finite-dim Zhu algebra, the collision
residues could grow factorially in $r$.

\emph{Step 2 (W-W Laurent envelope).}  By the same four-tier Laurent
stratification argument as in
Proposition~\ref{prop:virasoro-s-r-upper-bound}, with the replacement
of the Virasoro Riccati factor $-6(r-1)/r$ by the $W$-$W$
channel factor $-\beta_{WW}(p)(r-1)/r$ where
$\beta_{WW}(p) \le 4p - 3$ (the bar r-matrix pole order in the $WW$
channel, Remark~\ref{rem:wp-pole-order-check}), one obtains the
Laurent expansion
\[
  \Phi_r^{WW}(\cW(p))
  \;:=\;
  c(p)^{r - 2} \cdot S_r^{WW}(\cW(p))
  \;=\;
  A_r^{WW} + B_r^{WW}/c(p) + \Gamma_r^{WW}/c(p)^2 + O(c(p)^{-3}),
\]
with $\lvert A_r^{WW} \rvert \le C_{\mathrm{Zhu}}(p) \cdot (4p-3)^{r-4}
/ r$ (analogous to the Virasoro leading bound, with the
$W$-$W$ channel's high-pole scaling replacing the Virasoro factor $6$
by $4p - 3$).

\emph{Step 3 (Stirling).}  As in
Theorem~\ref{thm:tempered-stratum-contains-virasoro}:
\begin{align*}
  (\lvert S_r^{WW} \rvert / r!)^{1/r}
  &\le \Bigl(
    C_{\mathrm{Zhu}}(p) \cdot M(p, r) \cdot (4p-3)^{r-2}
    / (\lvert c(p) \rvert^{r-2} \cdot r!)
  \Bigr)^{1/r}
  \\
  &\to\;
  1 \cdot 1 \cdot (4p-3) \cdot \lvert c(p) \rvert^{-1}
    \cdot (r!)^{-1/r}
  \to 0
\end{align*}
as $r \to \infty$, by Stirling $(r!)^{-1/r} \sim e/r \to 0$.
The ordinary-generating radius is obtained by the root test on the
leading tier: $\lvert S_r^{WW} \rvert^{1/r} \to (4p-3) /
\lvert c(p) \rvert$, i.e.\ $\rho_*^{WW}(\cW(p)) = \lvert c(p) \rvert
/ (4p - 3)$.
\end{proof}

\begin{proposition}[Channel-wise upper bounds summary]
\label{prop:wp-channel-upper-bounds}
\ClaimStatusProvedHere
For every integer $p \ge 2$:
\begin{align*}
  \rho_*^{TT}(\cW(p)) &= \lvert c(p) \rvert / 6, \\
  \rho_*^{TW}(\cW(p)) &= \lvert c(p) \rvert / (6 (2p - 1))
    \;=\; \lvert c(p) \rvert / (12 p - 6), \\
  \rho_*^{WW}(\cW(p)) &= \lvert c(p) \rvert / (4 p - 3).
\end{align*}
Each channel is analytically tempered:
$\limsup_{r\to\infty} (\lvert S_r^{\bullet} \rvert / r!)^{1/r} = 0$.
\end{proposition}

\begin{proof}
$TT$: Lemma~\ref{lem:wp-virasoro-subchannel-tempered}.
$TW$: Lemma~\ref{lem:wp-tw-subchannel-tempered}.
$WW$: Lemma~\ref{lem:wp-ww-subchannel-tempered}.
Channel-wise tempering is each lemma's conclusion; the
ordinary-generating radii follow from the root-test evaluation on the
respective leading Laurent tiers.
\end{proof}

\begin{remark}[Retraction of the Zhu-bounded-Massey proof]
\label{rem:zhu-bounded-massey-retraction}
The initial inscription of
Lemma~\ref{lem:wp-zhu-bounded-masseys} appealed to Cauchy-Schwarz
on the finite-dim Zhu inner product, with the implicit claim that
a Massey of length $k$ has exactly $k+1$ Zhu-algebra matrix
elements. Primary-source examination reveals this is not supported:

\emph{Ghost.} $C_2$-cofiniteness bounds the graded dimensions of
$\cW(p)$ and its Zhu algebra $A(\cW(p))$. At every fixed bar degree,
the space of classes is finite-dimensional.

\emph{Wrong.} That bounded Zhu $\Rightarrow$ bounded Massey. The
Massey product is a SECONDARY cohomology operation; its values lie
in $H^{\bullet}(B(\cW(p)))$, which is the FULL bar cohomology,
graded by BOTH Zhu-grading and bar-complex degree. Bounding
individual Zhu components does not bound the cumulative Massey
tower across bar degrees. Gurarie 1993 and Flohr 1996 construct
explicit logarithmic-CFT Massey towers whose amplitudes accumulate

\begin{remark}[Counterexample candidates in the logarithmic landscape]
\label{rem:gurarie-flohr-massey-obstruction}

\end{remark}

\begin{remark}[Correlation-function Massey vs shadow-tower Massey: the
retraction carves a distinction, not a uniform falsification]
\label{rem:correlation-vs-shadow-massey}
The Gurarie-Flohr obstruction of
Remark~\ref{rem:gurarie-flohr-massey-obstruction} and the retraction of
Lemma~\ref{lem:wp-zhu-bounded-masseys} do \emph{not} falsify a single
Massey-boundedness property; they falsify exactly ONE of two distinct
properties of the $\cW(p)$ bar cohomology. Sharpening the distinction
is essential because the tempering implication routes through the OTHER
property, which is NOT falsified and remains the genuine open frontier.

\emph{(A) Correlation-function Massey boundedness.} The amplitude norm
\[
  \bigl\| \langle \Omega_{i_1}, \ldots, \Omega_{i_k}\rangle_{\mathrm{corr}}
  \bigr\|
  \;:=\;
  \sup_{z_j}\,
  \bigl\lvert
    \langle \Omega_{i_1}(z_1) \cdots \Omega_{i_k}(z_k)\rangle_{\cW(p)}
  \bigr\rvert
\]
is the natural norm on correlation functions. Gurarie 1993 and
Flohr 1996 construct explicit amplitudes with $(\log z)^k$ growth in
the coincidence limit, exhibiting an unbounded logarithmic accumulation.
\textbf{FALSIFIED:} the claim that this norm is bounded by a power of
$\dim A(\cW(p))$ is wrong.

\emph{(B) Shadow-tower Massey boundedness.} The shadow-tower coefficient
$S_r(\cW(p))$ is NOT the amplitude norm of a Massey product; it is the
Orlik-Solomon residue of the bar propagator against the universal
Arnold / $\dlog$ form on $\mathrm{FM}_r(\CC)$. Concretely,
\[
  S_r(\cW(p))
  \;=\;
  \bigl\langle [\Barch(\cW(p))]_r,\;
    \dlog\, E(z_{ij})_{\mathrm{Arnold}}
  \bigr\rangle_{\mathrm{FM}_r(\CC)},
\]
a structure constant of the $A_\infty$ coproduct, extracted by the
logarithm-free part of the bar propagator. The $\dlog$ operation
strips the $(\log z)^n$ growth of correlation-function amplitudes by
construction (the Arnold form is the regular representative; the
logarithmic divergences are precisely what the residue annihilates).
\textbf{OPEN:} whether $S_r(\cW(p))$ is $c(p)$-independently bounded
in $r$.

\emph{Why the distinction is sharp.} The Gurarie-Flohr amplitudes
live in the $(\log z_{ij})^n$-enriched sector, which is the KERNEL of
the $\dlog$-extraction. The bar cohomology splits as
\[
  H^\bullet(\Barch(\cW(p)))
  \;=\;
  H^\bullet_{\mathrm{reg}} \oplus H^\bullet_{\mathrm{log}},
\]
with $H^\bullet_{\mathrm{reg}}$ = polynomial $\dlog$-residue classes
and $H^\bullet_{\mathrm{log}}$ = logarithmic Gurarie-Flohr sector.
Shadow-tower coefficients $S_r$ live in $H^\bullet_{\mathrm{reg}}$ by
construction. Gurarie-Flohr unboundedness is confined to
$H^\bullet_{\mathrm{log}}$. The two sectors are orthogonal under the
Arnold / OS pairing.

\emph{The correct sharpened open frontier.} After retraction:
\begin{itemize}[nosep]
\item (a) \emph{Correlation-function Massey boundedness on $\cW(p)$}:
  FALSIFIED by Gurarie-Flohr. Closed (negative).
\item (b) \emph{Shadow-tower Massey boundedness on $\cW(p)$}:
  $c(p)$-independent bound on $|S_r|$ in $r$, equivalently
  $\limsup (|S_r|/r!)^{1/r} = 0$. Open; the
  Adamovi\'c-Milas character-expansion route of
  Remark~\ref{rem:wp-tempering-status} attacks (b) directly.
\end{itemize}
Conjecture~\ref{conj:tempered-stratum-contains-wp} is exactly claim (b).
Its conjectural status correctly reflects the open genuine frontier;
it is NOT falsified by Gurarie-Flohr, which falsifies only (a).
\end{remark}
