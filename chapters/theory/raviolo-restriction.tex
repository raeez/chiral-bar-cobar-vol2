\section{Raviolo restriction: Čech/Thom--Sullivan model and coinvariants}
\label{sec:raviolo-expanded}

The \v{C}ech/Thom--Sullivan totalization of the raviolo factorization algebra (\S\ref{sec:raviolo}) computes derived global sections; the resulting homotopy coinvariant spaces furnish the generic raviolo coinvariant package, and on finite-type comparison surfaces their linear duals recover the corresponding conformal blocks.

\subsection{Raviolo geometry and derived global sections}
\label{subsec:raviolo-geometry}
Let $\mathrm{Rav}:=D_+\cup_{D^\times}D_-$ denote the raviolo obtained by gluing two copies of the formal disk along the punctured disk. Cover by $\{D_+,D_-,D^\times\}$ and write the Čech nerve.
\begin{definition}[Raviolo factorization algebra]
\label{def:raviolo-FA}
Given a $W(\mathsf{SC}^{\mathrm{ch,top}})$–algebra $(A_{\mathsf{ch}},A_{\mathsf{top}})$, define the raviolo factorization algebra $\mathsf V_{\mathrm{rav}}$ by restricting to time–slices as in Section~\ref{sec:raviolo} and glue along $D^\times$ using the $E_1$–homotopy. Then
\[
\Gamma(\mathrm{Rav},\mathsf V_{\mathrm{rav}})
:= \Tot\bigl(\check{C}^\bullet(\{D_+,D_-,D^\times\};\mathsf V_{\mathrm{rav}})\bigr)
\]
with Thom–Sullivan realization yields the derived global sections.
\end{definition}
\begin{proposition}[Čech exactness; \ClaimStatusProvedElsewhere]
\label{prop:Cech-exactness}
For holomorphic–topological $\mathsf V_{\mathrm{rav}}$, the Čech/Thom–Sullivan totalization computes derived global sections on $\mathrm{Rav}$.
\end{proposition}
\begin{proof}
Use cover descent and the standard argument for factorization algebras; cf.\ \cite{CG17,GW23}.
\end{proof}

\subsection{Coinvariants and conformal blocks}
\label{subsec:coinvariants}
\begin{definition}[Coinvariants]
\label{def:coinvariants}
For a raviolo VA $\mathsf V_{\mathrm{rav}}$, define the coinvariants on a test curve $X\to\mathrm{Rav}$ by the homotopy quotient
\[
\mathrm{Coinv}_X(\mathsf V_{\mathrm{rav}}) := \bigl(\Gamma(X,\mathsf V_{\mathrm{rav}})\bigr)_{G}^{\mathrm{h}}
\]
with respect to the global symmetries $G$ acting by currents (where $G$ is a Lie algebra acting on~$\mathsf V_{\mathrm{rav}}$ by chiral currents, when such an action is part of the data). Conformal blocks are the linear duals when finite type.
\end{definition}

\begin{remark}
The holomorphic–topological pairing ensures that distributions supported on $D^\times$ implement the usual vertex–algebra state–field correspondence after passing to cohomology.
\end{remark}

\subsection{Chain-level comparison: punctured disk versus circle}
\label{subsec:chain-level-D-vs-S1}

The state--field correspondence implemented by the raviolo is a statement at the level of cohomology.
At the chain level, two distinct models of the gluing locus $D^\times$ are available, and their comparison is the source of the $E_2/E_1$ gap that governs the passage from holomorphic to topological data.

\phantomsection\label{rem:chain-D-vs-S1}

\begin{proposition}[Chain-level comparison: $D^\times$ versus $S^1$;
\ClaimStatusProvedHere]%
\label{prop:chain-D-vs-S1}%
\index{punctured disk!chain-level comparison with circle|textbf}%
\index{factorisation homology!punctured disk vs circle}%
\index{E2/E1 gap@$E_2/E_1$ gap!chain-level source}%
Let $\cA$ be a strongly admissible $E_\infty$-chiral algebra
and let $A$ denote the underlying holomorphic--topological
factorization algebra on $\mathrm{Rav}$.
\begin{enumerate}[label=\textup{(\roman*)}]
\item \textbf{Punctured-disk model.}
  The chiral factorization homology $C^{\mathrm{ch}}(D^\times,\cA)$
  computes the derived global sections of~$\cA$ on the punctured
  formal disk.
  Configuration spaces on~$D^\times$ are two-dimensional:
  $\mathrm{Conf}_n(D^\times)\simeq
  \mathrm{Conf}_n(\mathbf{C}^\times)$, and the
  $\mathbf{C}^\times$-equivariant Fulton--MacPherson
  compactification
  $\overline{\mathrm{FM}}_n(D^\times)$ governs the
  operadic structure.
  The resulting chain complex carries a natural $E_2$-algebra
  structure: the two-dimensional configurations encode the
  full little-disk operad~$E_2$.

\item \textbf{Circle model.}
  The topological factorization homology $C^{\mathrm{top}}(S^1,\cA)$
  coincides with the cyclic bar complex $B^{S^1}_\bullet(\cA)$
  of Definition~\textup{\ref{def:circle-bar-chain-model}}.
  The configurations are one-dimensional
  (cyclically ordered points on the circle),
  and the resulting chain complex carries an $E_1$-algebra
  structure: cyclic order provides the data of an
  associative (up to coherent homotopy) multiplication.

\item \textbf{Quasi-isomorphism.}
  The radial projection
  $r\colon D^\times\to S^1$, $z\mapsto z/|z|$,
  is a deformation retraction with section
  $\iota\colon S^1\hookrightarrow D^\times$ (the
  inclusion of the unit circle) and homotopy
  $h_t(z)=(1-t)z + t\,z/|z|$, $t\in[0,1]$.
  There is an induced chain map
  \begin{equation}\label{eq:rho-chain-map}
    \rho\colon C^{\mathrm{ch}}(D^\times,\cA)
    \xrightarrow{\;\sim\;}
    C^{\mathrm{top}}(S^1,\cA)
  \end{equation}
  that is a quasi-isomorphism of chain complexes.
  In particular, both models compute the same homology:
  \[
    H_\bullet\bigl(C^{\mathrm{ch}}(D^\times,\cA)\bigr)
    \;\cong\;
    H_\bullet\bigl(C^{\mathrm{top}}(S^1,\cA)\bigr)
    \;\cong\;
    Z(\cA),
  \]
  the derived center (chiral Hochschild homology)
  of~$\cA$. This is the chain-level content of the
  state--field correspondence.

\item \textbf{$E_1$-compatibility, $E_2$-incompatibility.}
  The map~$\rho$ is an $E_1$-algebra map but \emph{not}
  an $E_2$-algebra map: the target carries only an
  $E_1$-structure.
  The discrepancy---the fiber of the forgetful functor
  $E_2\text{-}\mathsf{Alg}\to E_1\text{-}\mathsf{Alg}$---is
  the holomorphic datum that homology discards.
  The higher $E_2$-operations encode the braiding
  (monodromy of two points in $D^\times$ around each
  other), while the $E_1$-structure retains only the
  cyclic ordering.
\end{enumerate}
\end{proposition}

\begin{proof}
The argument proceeds in four steps: construction of the
map on configuration spaces, verification that it is a
chain map, the quasi-isomorphism property, and the
$E_1/E_2$ analysis.

\medskip
\noindent\textit{Step 1: Map on configuration spaces.}
The radial projection $r\colon D^\times\to S^1$,
$z\mapsto z/|z|$, is a smooth $\mathbf{C}^\times$-equivariant
map (with respect to the rotation action on both sides).
It induces a map on unordered configuration spaces
\begin{equation}\label{eq:config-retract}
  r_n\colon \mathrm{Conf}_n(D^\times)
  \longrightarrow \mathrm{Conf}_n(S^1),
  \qquad
  \{z_1,\dots,z_n\}\longmapsto
  \bigl\{z_1/|z_1|,\dots,z_n/|z_n|\bigr\}.
\end{equation}
This map is well-defined on configurations (distinct
points in~$D^\times$ may collide under~$r_n$ only when
they share the same angular coordinate), and the
subset of colliding configurations has real
codimension~$1$ in~$\mathrm{Conf}_n(D^\times)$.
To handle this, we work with the Fulton--MacPherson
compactifications on both sides.

On the source, $\overline{\mathrm{FM}}_n(D^\times)$
compactifies $\mathrm{Conf}_n(D^\times)$ by recording
relative positions of colliding points as
screen data (both radial and angular).
On the target, $\overline{\mathrm{Conf}}_n(S^1)$
compactifies $\mathrm{Conf}_n(S^1)$ by allowing
points to collide with recorded relative cyclic order.
The radial projection extends to a continuous map
\begin{equation}\label{eq:FM-retract}
  \bar r_n\colon
  \overline{\mathrm{FM}}_n(D^\times)
  \longrightarrow
  \overline{\mathrm{Conf}}_n(S^1)
\end{equation}
by sending screen data at a collision to the angular
component of the screen: a configuration of $k$~points
colliding in~$D^\times$ records their relative positions
in~$T_{z_0}D^\times\cong\mathbf{C}$; the radial
projection extracts the angular ordering, which is
the screen datum in~$\overline{\mathrm{Conf}}_n(S^1)$.
This extension is well-defined because the angular
ordering is the data that survives radial collapse,
and it is compatible with the face maps (further
collisions) by functoriality of the angular extraction.

\medskip
\noindent\textit{Step 2: Chain map.}
The factorization homology chain complexes are assembled
from the configuration spaces by integrating the
algebraic data (OPE kernels in the chiral direction,
coproduct faces in the topological direction) over the
compactified configuration spaces.
Concretely,
\[
  C^{\mathrm{ch}}_n(D^\times,\cA)
  \;=\;
  C_\ast\!\bigl(\overline{\mathrm{FM}}_n(D^\times);\,
  \omega_{\mathrm{OPE}}\bigr),
  \qquad
  C^{\mathrm{top}}_n(S^1,\cA)
  \;=\;
  C_\ast\!\bigl(\overline{\mathrm{Conf}}_n(S^1);\,
  \omega_{\mathrm{cyc}}\bigr),
\]
where $\omega_{\mathrm{OPE}}$ encodes the full OPE data
(poles, residues, regular terms) and
$\omega_{\mathrm{cyc}}$ encodes the cyclic bar data
(coproduct faces and wraparound monodromy).
The map $\rho$ is defined by pushforward:
\begin{equation}\label{eq:rho-pushforward}
  \rho(\sigma\otimes\omega_{\mathrm{OPE}})
  \;:=\;
  (\bar r_n)_\ast(\sigma)\otimes
  \bigl(\bar r_n^\ast\bigr)^{-1}(\omega_{\mathrm{OPE}}),
\end{equation}
where $(\bar r_n)_\ast$ is the pushforward on chains
and $(\bar r_n^\ast)^{-1}(\omega_{\mathrm{OPE}})$ denotes
the image of the OPE coefficient system under the
projection: all radial (holomorphic modulus) information
is integrated out, and the remaining angular data is
precisely the cyclic bar coefficient $\omega_{\mathrm{cyc}}$.

To verify the chain-map property
$\rho\circ d^{\mathrm{ch}}=d^{\mathrm{top}}\circ\rho$,
observe that the differentials on both sides are
determined by codimension-$1$ boundary strata of
their respective configuration spaces. The map~$\bar r_n$
sends each boundary stratum of
$\overline{\mathrm{FM}}_n(D^\times)$ to the
corresponding stratum of
$\overline{\mathrm{Conf}}_n(S^1)$:
\begin{itemize}
\item A two-dimensional collision in~$D^\times$
  (two holomorphic points approach each other, with
  the full $\mathbf{C}$-valued screen) maps to a
  one-dimensional collision on~$S^1$ (two points
  approach on the circle, with only the cyclic-order
  screen). The OPE residue $d_{\mathrm{res}}$ on the
  source restricts to the collision differential on the
  target. By the residue theorem, integrating the full
  OPE kernel over the radial fiber at a collision
  extracts the leading residue, which is the datum
  recorded by the topological collision face.
\item The coproduct faces $d_\Delta$ (a point
  approaches the boundary of the FM compactification
  without colliding with another point) map identically
  on both sides: they are purely topological
  (combinatorial face maps on the simplicial/cosimplicial
  direction) and the radial projection acts trivially.
\item The wrap-around face $d_{\mathrm{wrap}}$ (the
  cyclic identification $0\sim 1$ on $S^1$) is present
  only on the circle side. On the punctured-disk side,
  the corresponding operation is the monodromy of a
  point traversing a loop in~$D^\times$ around the
  origin. The map $\bar r_n$ sends this loop to the
  full circle traversal on~$S^1$, and the monodromy
  $\mathrm{Mon}(R)$ is transported faithfully, since
  the angular component of the loop is preserved by
  radial projection.
\end{itemize}
Hence $\rho$ commutes with all components of the
differential and is a chain map.

\medskip
\noindent\textit{Step 3: Quasi-isomorphism.}
The deformation retraction
$h_t\colon D^\times\to D^\times$,
$h_t(z)=(1-t)z+t\,z/|z|$, satisfies $h_0=\mathrm{id}$,
$h_1=\iota\circ r$. It induces a homotopy
\[
  (h_t)_n\colon
  \mathrm{Conf}_n(D^\times)\times[0,1]
  \longrightarrow\mathrm{Conf}_n(D^\times)
\]
by applying $h_t$ pointwise. For generic configurations,
this homotopy stays within
$\mathrm{Conf}_n(D^\times)$ (no collisions occur
during the retraction when the angular coordinates of
the $n$~points are distinct).
The locus where collisions occur during retraction has
codimension~$1$ in the product
$\mathrm{Conf}_n(D^\times)\times[0,1]$
and is handled by the FM compactification: the homotopy
extends to
\[
  \bar h_n\colon
  \overline{\mathrm{FM}}_n(D^\times)\times[0,1]
  \longrightarrow
  \overline{\mathrm{FM}}_n(D^\times)
\]
by the universal property of the compactification
(recording screen data for any incipient collisions
created by the retraction).
The chain homotopy operator
$K_n:=(\bar h_n)_\ast\circ\mathrm{pr}_\ast$
(where $\mathrm{pr}\colon
\overline{\mathrm{FM}}_n(D^\times)\times[0,1]
\to\overline{\mathrm{FM}}_n(D^\times)$ is the
projection) satisfies
$\iota_\ast\circ\rho - \mathrm{id}
= d^{\mathrm{ch}}\circ K + K\circ d^{\mathrm{ch}}$
by the standard argument for deformation retracts at
the chain level (prism decomposition). Since
$\rho\circ\iota_\ast=\mathrm{id}$ on
$C^{\mathrm{top}}(S^1,\cA)$
(the section followed by the retraction is the identity
on~$S^1$), the pair $(\rho,\iota_\ast)$ exhibits a
deformation retract of chain complexes.
In particular, $\rho$ is a quasi-isomorphism.

\medskip
\noindent\textit{Step 4: $E_1$-map, not $E_2$-map.}
The $E_1$-structure on both sides is governed by the
cyclic ordering of points.
Since the radial projection preserves angular ordering
($\arg(z_i)<\arg(z_j)$ implies
$\arg(r(z_i))<\arg(r(z_j))$), the map~$\rho$ preserves
all $E_1$-operations: the operadic compositions
(governed by nested intervals on $S^1\subset D^\times$)
are transported faithfully.

The $E_2$-operations, by contrast, are governed by the
full two-dimensional positioning of points in~$D^\times$.
The $E_2$-multiplication $\mu_{E_2}$ involves two
little disks in~$D^\times$ that can wind around each
other: the fundamental group
$\pi_1(\mathrm{Conf}_2(D^\times))$ is free on two
generators (the braid generator and the winding number
around the puncture), while
$\pi_1(\mathrm{Conf}_2(S^1))$ is infinite cyclic
(only cyclic order).
The radial projection collapses the radial separation
and identifies configurations that differ only in the
moduli of their points; this quotient kills the
generator corresponding to radial separation and
thus cannot preserve the $E_2$-braiding.
More precisely, the braiding on
$C^{\mathrm{ch}}(D^\times,\cA)$ is encoded by chains
on $\mathrm{Conf}_2(D^\times)$ swept out by one point
encircling another in the full two-dimensional sense;
the image of such a chain under~$r_2$ is a
degenerate cycle on $\mathrm{Conf}_2(S^1)$ (the
radial component collapses), and the $E_2$-operation
is not recoverable from its image.
\end{proof}

\begin{remark}[The $E_1$ case]%
\label{rem:chain-D-vs-S1-E1}%
The proposition above is stated for $E_\infty$-chiral
algebras, but the quasi-isomorphism extends to all
strongly admissible $E_1$-chiral algebras: see
Proposition~\ref{prop:punctured-disk-S1-qiso}.  The two
additional verifications required for the $E_1$ case are:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Chain map for $d_{\mathrm{wrap}}$.}
  For $E_\infty$ algebras, the $R$-matrix monodromy
  factors through the angular projection automatically
  (the $R$-matrix is derived from the local OPE).
  For genuinely $E_1$ algebras, $R(z)$ is an independent
  datum, but it is meromorphic on~$\C^\times$ with poles
  only at $z=0$
  (Proposition~\ref{prop:field-theory-r}).
  By Cauchy's theorem, the monodromy
  $\mathrm{Mon}(R)$ is independent of the radius of
  the integration contour, so the chain map property
  of $d_{\mathrm{wrap}}$ holds at every radius.
\item \emph{Acyclicity of the kernel.}
  The radial deformation retraction $\rho_t$ preserves
  collision divisors (it does not alter relative
  local coordinates within clusters) and commutes with
  the monodromy insertion (by the same Cauchy argument).
  The contracting homotopy $h=\int_0^1\rho_t^*\,dt$ is
  therefore compatible with the full differential
  $d^{D^\times}$ and yields acyclicity of~$\ker\varphi$.
\end{enumerate}
The distinction between $E_\infty$ and $E_1$ is not about
the existence of the quasi-isomorphism but about what it
discards: the circle model retains only the monodromy
$\mathrm{Mon}(R)$, not the full spectral family $R(z)$
(Remark~\ref{rem:punctured-disk-S1-E1-gap}).
\end{remark}

\begin{remark}[Scope and limitations]%
\label{rem:chain-D-vs-S1-scope}%
\begin{enumerate}[label=\textup{(\alph*)}]
\item The extension of the pointwise homotopy~$h_t$ to
  the FM compactification~$\bar h_n$ in Step~3 uses the
  universal property of the real-oriented blowup at
  collision diagonals. The argument is standard for
  $\overline{\mathrm{FM}}_n(\mathbf{C})$
  (cf.\ \cite{Sinha2004,LV12}), but the equivariant
  version for~$D^\times$ requires the
  $\mathbf{C}^\times$-equivariant FM compactification
  as constructed in the annular configuration space
  framework of
  Definition~\textup{\ref{def:circle-bar-chain-model}}.
  The compatibility of~$\bar h_n$ with the coefficient
  system~$\omega_{\mathrm{OPE}}$ follows from the fact
  that the OPE kernels extend meromorphically across
  the collision divisors (strong admissibility of~$\cA$).
\item The quasi-isomorphism $\rho$ is \emph{not}
  canonical at the $E_2$ level: different choices of
  retraction (e.g.\ $z\mapsto z/|z|$ versus
  $z\mapsto e^{i\arg(z)}$, which agree but illustrate
  that the choice of radial collapse is essentially
  unique up to homotopy) all produce the same map on
  homology but may differ at the cochain level in ways
  that reflect the $E_2$-ambiguity.
\item The fiber of~$\rho$ (the chain-level kernel of
  ``forgetting holomorphic data'') is the chain-level
  home of the $E_2/E_1$ gap. Describing this fiber
  explicitly as a module over the $E_2$-operations is
  an open problem related to the formality of the
  little-disk operad and the Deligne conjecture
  (Proposition~\textup{\ref{prop:derived-center-E2}}).
\end{enumerate}
\end{remark}

\begin{remark}[Role in the geometric ladder]
\label{rem:E2-E1-ladder}
The chain-level $E_2/E_1$ comparison is load-bearing for the
$E_1$-to-$E_2$ step in the geometric hierarchy of
\S\ref{sec:raviolo-expanded}.
At the coarsest level, the raviolo construction produces
an $E_1$-algebra (the topological direction along $\mathbf{R}$
in the Swiss-cheese operad $\mathsf{SC}^{\mathrm{ch,top}}$).
Recovering $E_2$-structure requires remembering that the gluing locus
is not merely $S^1$ but $D^\times$: a surface, not a curve.
The holomorphic data encoded by the $E_2$-operations on
$C^{\mathrm{ch}}(D^\times,A)$ is the obstruction to promoting
$E_1$-associativity to $E_2$-commutativity.
At the level of homology this obstruction vanishes
(the derived center is symmetric), but at chains the two levels
of structure are genuinely distinct.
\end{remark}

\subsection{Shifted Poisson vertex structure on cohomology}
\label{subsec:PVA-cohomology-expanded}
Let $H^\bullet(\mathsf V_{\mathrm{rav}})$ be the BV cohomology. By Section~\ref{sec:Ainfty-to-PVA} and Theorem~\ref{thm:raviolo-PVA}, we have:

\begin{theorem}[Cohomological PVA on raviolo; \ClaimStatusProvedElsewhere]
\label{thm:raviolo-PVA-expanded}
\textup{(Reproduced from Theorem~\ref{thm:raviolo-PVA} for the reader's convenience.)}
$H^\bullet(\mathsf V_{\mathrm{rav}})$ is a $(-1)$–shifted Poisson vertex algebra; its $\lambda$–bracket is given by the residue of the antisymmetric singular part of $m_2$, and its product is the symmetric regular part of $m_2$. Higher $m_{k\ge3}$ vanish in cohomology by Stokes/AOS identities.
\end{theorem}

\begin{corollary}[Agreement with HT constructions; \ClaimStatusConditional]
\label{cor:agreement-HT}
For HT twists of 3d $\mathcal N=2,4$ theories whose observables lie in the scope of Theorem~\ref{thm:physics-bridge}, $H^\bullet(\mathsf V_{\mathrm{rav}})$ agrees with the PVA extracted by the secondary products in \cite{OY20,CDG20} and the 2--shifted Poisson structures in \cite{GW23,GRW21}. The two shifts are reconciled by the relationship between the derived algebraic geometry perspective (which counts the shift of the symplectic form on the derived stack, giving~$+2$) and the vertex algebra perspective (which counts the intrinsic degree of the $\lambda$-bracket as an operation on the underlying graded space, giving~$-1$). The total shift $2 + (-1) = 1$ accounts for the passage from the stack-theoretic to the operadic description, with the~$+1$ absorbed by the parity of the raviolo.
\end{corollary}
