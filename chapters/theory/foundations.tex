\section{The open primitive and the Swiss-cheese operad}
\label{sec:foundations}
\label{sec:operads-factorization}

\subsection{The primitive datum}

Vol~I worked pointwise: the bar complex of a single vertex algebra,
all constructions local to one marked point on the curve. The
three-dimensional holomorphic--topological setting forces
globalization. Insertions spread over $\C_z$ with holomorphic
collisions and over $\R_t$ with topological ordering; the two
directions cannot be studied in isolation because the defining
relations couple them. What is the correct primitive datum for
the open sector of a 3d HT theory, and what is its operadic
home?

The answer is the two-colored Swiss-cheese operad~$\SCchtop$, with
operation spaces $\FM_k(\C) \times E_1(m)$: a point in
$\FM_k(\C)$ records the collision pattern of~$k$ holomorphic
insertions, a point in $E_1(m)$ records the linear ordering
of~$m$ topological insertions, and the Swiss-cheese composition
enforces bulk-to-boundary directionality (no open-to-closed map).
The primitive datum is not any particular algebra but the
open/closed factorization dg-category~$\cC$ on the bordified curve
(\S\ref{subsec:categorical-open-sector}); boundary algebras are
charts on~$\cC$ recovered by evaluation at a compact generator.

The bar complex $\barB^{\mathrm{ch}}(\cA)$ of a boundary
algebra~$\cA$ (constructed in Volume~I) is the explicit
coalgebraic presentation of the twisting data of~$\cC$: the
tensor coalgebra $T^c(s^{-1}\bar{\cA})$ equipped with
\begin{enumerate}[label=(\roman*)]
\item a \emph{differential} $d_{\barB}$, built from OPE
 residues along collision divisors in $\FM_k(X)$, encoding the
 chiral product (the closed color of $\SCchtop$);
\item a \emph{coassociative coproduct}
 $\Delta[a_1|\cdots|a_n]
 = \sum_{i=0}^{n} [a_1|\cdots|a_i]
 \otimes [a_{i+1}|\cdots|a_n]$,
 the ordered deconcatenation of tensor factors (the open color).
\end{enumerate}
The differential lives on $\FM_k(\C)$. The coproduct lives on
$\Conf_k(\R)$: the ordered splitting of a bar element into two
sub-sequences is the algebraic expression of cutting an interval at
a point. Together, a bar element of degree~$k$ is parametrised by
$\FM_k(\C) \times \Conf_k(\R)$, the product of holomorphic and
topological configuration spaces.

\subsection{The open sector as factorization category}
\label{subsec:categorical-open-sector}

The primitive datum is the open/closed factorization dg-category on
a bordified curve, not any particular algebra
extracted from it. The boundary
algebra $A_b = \RHom_{\cC(J)}(b,b)$ is a \emph{chart} on this
category: it presents~$\cC$ from the viewpoint of a single compact
generator~$b$, as a coordinate chart presents a manifold from
a single open set. Different compact generators yield
Morita-equivalent presentations, and the bar complex
$\barB(\cA)$ encodes the twisting data of the category. The
bar-cobar adjunction is the representability theorem for
deformations of a vacuum object.

\begin{definition}[Bordified curve and boundary intervals]
\label{def:bordified-curve}
Let $(X,D)$ be a smooth algebraic curve $X$ with a finite set of
marked points $D = \{p_1,\ldots,p_n\} \subset X$. The \emph{real
oriented blowup}
\[
 \pi \colon \widetilde{X}_D \;\longrightarrow\; X
\]
replaces each $p_i$ by the circle $S^1_{p_i}$ of real tangent
directions at~$p_i$; it is a smooth manifold with boundary
$\partial \widetilde{X}_D = \bigsqcup_{i=1}^n S^1_{p_i}$. A choice
of \emph{tangential basepoints} $\tau = \{\tau_{p_i} \in
S^1_{p_i}\}_{i=1}^n$ determines open boundary intervals
\[
 I_{p_i} \;:=\; S^1_{p_i} \setminus \{\tau_{p_i}\}
 \;\cong\; \R,
\]
which inherit a canonical orientation from the complex structure
of~$X$. The data $(X,D,\tau)$ is called a \emph{pointed curve with
tangential basepoints}. When $(X,D) = (\C, \{0\})$ and $\tau = \tau_0$
is the positive-real direction, the interval $I_0$ recovers the
time axis~$\R_t$ of the holomorphic--topological setting.
\end{definition}

\begin{definition}[Mixed configuration spaces]
\label{def:mixed-config}
Let $(X,D,\tau)$ be as above. For a finite set $I$ and a tuple
$\mathbf{m} = (m_{p_1},\ldots,m_{p_n}) \in \N^n$, the
\emph{mixed open/closed configuration space} is
\[
 \Conf^{\mathrm{oc}}_{I,\mathbf{m}}(X, D, \tau)
 \;:=\;
 \Conf_I(X \setminus D)
 \;\times\;
 \prod_{p \in D} \Conf^{\mathrm{ord}}_{m_p}(I_p),
\]
where $\Conf_I(X \setminus D)$ parametrises $|I|$~labelled
points in the open complement and
$\Conf^{\mathrm{ord}}_{m_p}(I_p) = \{t_1 < \cdots < t_{m_p}
\in I_p\} \cong \Conf^{<}_{m_p}(\R)$ parametrises $m_p$ ordered
open insertions on the boundary interval at~$p$. The factor
$\Conf_I(X \setminus D)$ carries the Fulton--MacPherson
compactification; the boundary factors carry the topology of
ordered real configuration spaces.

The \emph{mixed Ran space} is the colimit
\[
 \Ran^{\mathrm{oc}}(X,D,\tau)
 \;:=\;
 \varinjlim_{(I,\mathbf{m})}
 \Conf^{\mathrm{oc}}_{I,\mathbf{m}}(X,D,\tau) \,/\, \Sigma_I,
\]
over finite sets $I$ and tuples $\mathbf{m}$, with transition maps
given by forgetting points.
\end{definition}

\begin{definition}[Open/closed factorization category on $(X,D,\tau)$]
\label{def:oc-factorization-category}
An \emph{open/closed factorization dg-category} $\cC$ on
$(X, D, \tau)$ consists of the following data:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \textbf{Cosheaf structure.} A constructible dg-cosheaf
 $\cC$ on $\Ran^{\mathrm{oc}}(X,D,\tau)$: for each stratum
 $(I,\mathbf{m})$, a dg-category
 $\cC(I,\mathbf{m})$ with restriction functors for inclusions of
 strata.

\item \textbf{Factorization.} For every decomposition of
 $(I,\mathbf{m})$ into disjoint sub-configurations
 $(I_\alpha,\mathbf{m}_\alpha)$ supported on pairwise disjoint
 open subsets, an equivalence
 \[
 \bigotimes_\alpha \cC(I_\alpha,\mathbf{m}_\alpha)
 \;\xrightarrow{\;\sim\;}\;
 \cC(I,\mathbf{m}),
 \]
 natural in the decomposition.

\item \textbf{Holomorphic locality.} In the closed (bulk)
 variables, the factorization structure is holomorphic: the
 cosheaf is constructible with respect to the
 Fulton--MacPherson stratification of $\Conf_I(X \setminus D)$,
 and the structure maps extend meromorphically along collision
 divisors with OPE poles.

\item \textbf{Local constancy on boundary intervals.} In the open
 (boundary) variables, the restriction functors along inclusions
 $I_p \supset J \supset J'$ of subintervals are equivalences:
 $\cC$ is locally constant in the boundary directions, encoding
 the $E_1$-category structure.

\item \textbf{Local model.} At the formal neighbourhood of each
 mixed collision stratum, $\cC$ is locally modelled by the
 $W(\SCchtop)$-algebra structure (where
 $W(-)$ is the Boardman--Vogt cofibrant resolution of a coloured
 operad, replacing strict composition by homotopy-coherent
 composition along intervals): the closed collisions are governed
 by $\FM_k(\C)$ and the open collisions by $E_1(m)$, as in
 Definition~\ref{def:SC-operations}.

\item \textbf{Clutching compatibility.} For a closed boundary
 circle $S^1_p$ (i.e.\ restoring the tangential basepoint
 $\tau_p$), the clutching functor
 \[
 \mathrm{cl}_p \colon
 \cC\big|_{I_p} \;\longrightarrow\;
 \cC\big|_{S^1_p}
 \]
 extends to a trace on the dg-category, implementing the
 sewing/factorization-homology gluing along~$S^1_p$.
 Precisely: for each node $p$ created by identifying two marked
 points on the normalization, the clutching functor $\mathrm{cl}_p$
 is defined by the coend
 $\mathrm{cl}_p(\mathcal{F}) := \mathcal{F} \otimes_{\cC(I_p)} \mathbf{1}$,
 where $\mathbf{1}$ is the monoidal unit of $\cC(S^1_p)$ and
 the tensor is taken over the restriction to the open interval.
 The
 cosheaf extends compatibly over the boundary strata of
 $\overline{\cM}_{g,n}$: genus-$g$ amplitudes are obtained by
 composing clutching along $g$~cycles.
\end{enumerate}
\end{definition}

\begin{remark}[Relation to known factorization structures]
\label{rem:oc-vs-known}
When $D = \varnothing$ (no boundary), the closed sector of
$\cC$ recovers a chiral factorization algebra on~$X$ in the
sense of \cite{BD04,CG17}. When $X = \C$ and $D = \{0\}$,
the boundary interval $I_0 \cong \R$ recovers the topological
direction~$\R_t$, and $\cC$ encodes the full two-colored
Swiss-cheese data of Definition~\ref{def:SC-algebra} promoted
to a category. The local constancy axiom~(iv) is the
categorical avatar of the locally-constant factorization
structure on~$\R$ that produces the $E_1$ (associative) color.
\end{remark}

\begin{remark}[The Swiss-cheese operad and the three bar complexes]
\label{rem:sc-three-bar-complexes}
\index{three bar complexes!Swiss-cheese decomposition}
\index{Swiss-cheese operad!two colours and bar complexes}
The two-coloured Swiss-cheese operad $\SCchtop$ acts on a boundary
algebra $\cA$ through two distinct channels, each of which produces
its own bar complex. The \emph{closed colour} (holomorphic,
$\FM_k(\C)$) assembles collision residues into the symmetric bar
complex $\barB^{\Sigma}(\cA)$, the factorization coalgebra of
Volume~I, Theorem~A. The \emph{open colour} (topological,
$E_1(m) \simeq \Conf^<_m(\R)$) assembles ordered tensor factors
into the ordered bar complex $\barB^{\mathrm{ord}}(\cA)$, the
$E_1$-coalgebra of the ordered associative chiral Koszul duality
programme (Volume~I, Theorem~\textup{\ref*{thm:bar-swiss-cheese}}).
Together, $(\barB^{\Sigma}(\cA), \barB^{\mathrm{ord}}(\cA))$ is
the two-coloured $E_1$ dg coalgebra datum attached to~$\cA$: the symmetric bar $\barB^{\Sigma}(\cA)$
classifies chiral twisting morphisms, the ordered bar $\barB^{\mathrm{ord}}(\cA)$
classifies ordered (associative) ones, and the mixed sector encodes
bulk-to-boundary intertwiners. The $\SCchtop$ structure emerges in the chiral derived center pair
$(C^\bullet_{\mathrm{ch}}(\cA,\cA),\, \cA)$. The ordered-to-symmetric descent
$\barB^{\mathrm{ord}} \to \barB^{\Sigma}$ is the $R$-matrix-twisted
$\Sigma_n$-coinvariant projection of Volume~I,
Theorem~\textup{\ref*{V1-thm:three-bar-complexes}}: it is the bar-level
incarnation of the averaging map
$\mathrm{av}\colon \mathfrak{g}^{E_1}_{\cA} \to \mathfrak{g}^{\mathrm{mod}}_{\cA}$.
\end{remark}

\begin{remark}[The ambient for bar--cobar: symmetric monoidal factorization vs.\ pseudo-tensor]
\label{rem:bar-cobar-ambient-clarification}
\index{bar--cobar!ambient clarification}
\index{pseudo-tensor!vs.\ symmetric monoidal}
\index{factorization ambient!symmetric monoidal}
Throughout this volume the phrase ``symmetric monoidal category of
chiral algebras'' is \emph{never} used in the strict sense. The
distinction is a standing hypothesis of the programme; this remark
records it once, canonically, for all downstream Koszul-duality and
bar--cobar statements.
\begin{enumerate}[label=\textup{(\alph*)}]
\item \emph{The ambient for bar--cobar is symmetric monoidal.}
 Bar--cobar is developed inside the symmetric monoidal
 $(\infty,1)$-category $\bigl(\Dmod(\Ran(X)),\, \otimes^{\star}\bigr)$
 of $D$-modules on the Ran space, equipped with the star
 (chiral/convolution) tensor product. Its full sub-$(\infty,1)$-category
 of \emph{factorizable} $D$-modules $\Fact(X) \subset \Dmod(\Ran(X))$
 is again symmetric monoidal under the external-product tensor
 $\boxtimes$, and it is this sub-ambient, together with its
 canonical $(\infty,2)$-enhancement \cite[IV.5]{GR17}, in which the
 bar and cobar functors
 \(
 \bar{B}^{\mathrm{ch}}_X \colon \Fact^{\mathrm{aug}}(X)
 \rightleftarrows \mathrm{CoFact}^{\mathrm{conil,comp}}(X)
 \colon \Omega^{\mathrm{ch}}_X
 \)
 are defined and exhibited as an adjunction.
\item \emph{Chiral algebras themselves form a pseudo-tensor
 category, not a symmetric monoidal one.}
 Following Beilinson--Drinfeld~\cite[\S 1.1, \S 3.4]{BD04}, the
 category of chiral algebras on $X$ carries only an
 \emph{operadic} / pseudo-tensor structure:
 there is no bifunctor
 $\cA \otimes^{\mathrm{ch}} \cB$ of chiral algebras and no unit
 object for such a bifunctor. The chiral datum is an
 $\infty$-operad (equivalently, a pseudo-tensor category in the
 sense of BD04) whose multilinear operations
 $\Hom^{\mathrm{ch}}(\cA_1,\dots,\cA_n;\,\cB)$ are parametrised by
 Ran-space configuration data.
\item \emph{The compatibility.}
 The pseudo-tensor structure of chiral algebras is encoded as a
 map of $\infty$-operads
 \(
 \mathsf{ChirAlg}(X) \longhookrightarrow \Fact(X)
 \)
 into the symmetric monoidal factorization ambient of~(a). Under
 this embedding, bar--cobar as an adjunction in the sym.\ monoidal
 ambient \emph{pulls back} to the Koszul-duality statement for
 chiral algebras as pseudo-tensor objects. The adjunction lives
 in~(a); the pseudo-tensor structure of the source category is
 respected but not promoted to a symmetric monoidal structure.
\end{enumerate}
\emph{Three quasi-equivalent reformulations of the ambient.} The
three standard realisations of this ambient are quasi-equivalent in
the precise sense of $(\infty,1)$-categorical equivalence of the
underlying presentable stable symmetric monoidal targets:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Beilinson--Drinfeld}~\cite[\S 3.4]{BD04}: factorization
 algebras on $X$ defined via a pseudo-tensor structure on
 $\Dmod(\Ran(X))$ with chiral operations encoded by the
 disjointness locus~$U \subset \Ran(X)\times\Ran(X)$.
\item \emph{Francis--Gaitsgory}~\cite{FG12}: $\infty$-categorical
 factorization algebras on $\Ran(X)$ with the star-tensor
 symmetric monoidal structure and a model structure for which
 bar--cobar is a Quillen adjunction on conilpotent-complete
 objects.
\item \emph{Gaitsgory--Rozenblyum}~\cite[IV.5]{GR17}: the
 presentable stable $(\infty,2)$-category of factorization
 algebras, equipped with its $(\infty,2)$-enhancement of star
 tensor, under which both forms are promoted to properads (cf.\
 Theorem~\ref{thm:properad-bar-cobar} in factorization ambient).
\end{enumerate}
Each of (i)--(iii) embeds the pseudo-tensor structure of
Beilinson--Drinfeld chiral algebras as a distinguished
sub-$\infty$-operad. All Koszul-duality, bar--cobar, and
$\mathsf{SC}^{\mathrm{ch,top}}$-pentagon theorems of this volume are
stated in the symmetric monoidal ambient (i)--(iii) and specialise
to the pseudo-tensor sub-category via this embedding.

\emph{Canonical convention (applies globally).} Whenever prose in
this volume refers to ``the symmetric monoidal category in which the
bar complex lives,'' the target is (i)--(iii). Whenever prose refers
to ``chiral algebras as a category,'' the target is the pseudo-tensor
sub-$\infty$-operad. These two referents are never identified; the
embedding of~(c) mediates between them.
\end{remark}

\begin{remark}[Rigid tensor structure, ribbon categories, and Tannaka reconstruction for chiral algebras]
\label{rem:rigid-tensor-structure-chir-alg}
\index{rigid category!factorization ambient}
\index{ribbon category!Rep\_q(A)}
\index{Tannaka reconstruction!chiral algebra}
\index{Verdier duality!autonomy of Fact(X)}
The symmetric monoidal ambient
$\bigl(\Fact(X),\,\boxtimes\bigr)$ of
Remark~\ref{rem:bar-cobar-ambient-clarification} carries a canonical
\emph{autonomous} (= rigid symmetric monoidal) structure, and the
representation category $\Repq(\cA)$ of a chirally Koszul algebra
$\cA$ carries a ribbon structure (rigid braided plus balancing
twist) precisely on the MTC locus. We record the four layers
once, canonically.
\begin{enumerate}[label=\textup{(\alph*)}]
\item \emph{$\Fact(X)$ is autonomous.} Every factorization algebra
 $\cF \in \Fact(X)$ admits a Verdier dual $\mathbb{D}\cF \in
 \Fact(X)$, with evaluation
 $\ev_\cF\colon \cF \boxtimes \mathbb{D}\cF \to \omega_X$ and
 coevaluation
 $\coev_\cF\colon \omega_X \to \mathbb{D}\cF \boxtimes \cF$
 given by the Verdier pairing on $\Dmod(\Ran(X))$ restricted to the
 factorizable locus. Triangle identities hold by biduality
 $\mathbb{D}^2 \simeq \id$ on coherent factorization algebras
 \cite[IV.5.3]{GR17}. Thus
 $(\Fact(X),\boxtimes,\omega_X,\mathbb{D})$ is a rigid symmetric
 monoidal $(\infty,1)$-category; its $(\infty,2)$-enhancement is
 autonomous in the coherent sense of \cite{GR17}. All bar--cobar
 theorems of this volume live in an autonomous ambient, a fact
 used implicitly in every Verdier-duality argument and explicitly
 in Theorem~C (genus-wise Lagrangian complementarity) and
 Theorem~H (chiral Higher Deligne).
\item \emph{$\Repq(\cA)$ is rigid braided iff $\cA$ is
 $C_2$-cofinite.} For $\cA$ chirally Koszul, the representation
 category $\Repq(\cA) = \cA\text{-}\mathsf{Mod}$ inside
 $\Fact(X)$ inherits a braided monoidal structure from the spectral
 $R$-matrix $R(z)$ on collision residues. Rigidity (existence of
 left and right duals) holds iff every object is dualizable;
 by Huang--Lepowsky--Zhang~\cite{HLZ14} and Huang~\cite{Hua08},
 this is equivalent to $C_2$-cofiniteness of $\cA$. A pivotal
 \textup{structure} (canonical iso $V \simeq V^{**}$) exists iff
 $\cA$ carries a conformal vector $\omega$ and the balancing twist
 $\theta_V = e^{2\pi i L_0}$ from the $T$-matrix phase trivialises
 the double dual. The MTC datum requires simplicity $+$ finite
 semisimplicity $+$ non-degeneracy of the $S$-matrix.
\item \emph{$\Repq(\cA)/I_{\neg}$ is ribbon.} In the MTC regime,
 quotienting the integrable level representation category by the
 negligible ideal $I_{\neg} \subset \Repq(\cA)$ yields a ribbon
 fusion category $\Repq(\cA)/I_{\neg}$ --- rigid $+$ braided $+$
 balanced $+$ simple-finite. This is the input to the
 Reshetikhin--Turaev functor; on the chiral Koszul side, it is the
 MTC quotient appearing in
 Vol~III's $\prop$\textup{:rt-functor-via-negligible-quotient}
 and in the affine monodromy--braiding identification for
 integrable level $k \in \Z_{\geq 0}$ on the
 evaluation-generated core (cf.~AP47).
\item \emph{Tannaka reconstruction.} For $\cA$ chirally Koszul,
 choose a fiber functor
 $\omega\colon \Repq(\cA)/I_{\neg} \to \Vect$ (fusion at a
 smooth point of $X$). Then
 \[
 \cA \;\simeq\; \End^{\otimes}(\omega)
 \]
 recovers $\cA$ as the algebra of tensor-natural endomorphisms of
 $\omega$; this is the Kazhdan--Lusztig equivalence~\cite{KL93} at
 generic level, extended to the ribbon MTC quotient at integrable
 level. In the Francis--Gaitsgory ambient (ii) of
 Remark~\ref{rem:bar-cobar-ambient-clarification}, the
 reconstruction is an $(\infty,1)$-equivalence of ribbon fusion
 categories; cf.~the meromorphic Tannakian reconstruction theorem
 in the RTT setting
 (chapters/connections/shifted\_rtt\_duality\_orthogonal\_coideals.tex,
 \S Tannakian reconstruction).
\end{enumerate}
\emph{Scope boundaries.} Rigidity of $\Repq(\cA)$ without
$C_2$-cofiniteness is false (Yangian: $\Repq$ has infinite-dim
modules without duals); ribbon structure without a conformal vector
is false (level-stripped $r(z)$ at $k=0$ has trivial $T$-matrix;
cf.~AP126). Autonomy of $\Fact(X)$ is unconditional.
\end{remark}

\begin{remark}[Chiral fusion product at chain level]
\label{rem:chiral-fusion-chain-level}
\index{fusion product!chain-level}%
\index{Huang--Lepowsky--Zhang tensor product}%
\index{Kunneth formula!fusion product}%
The Huang--Lepowsky--Zhang tensor product $\boxtimes_{\mathrm{HLZ}}$
of~\cite{HLZ14,Hua08} is by construction a COHOMOLOGICAL operation on
$\Repq(\cA)$: the associator isomorphism of
$M_1 \boxtimes_{\mathrm{HLZ}} (M_2 \boxtimes_{\mathrm{HLZ}} M_3)
\simeq (M_1 \boxtimes_{\mathrm{HLZ}} M_2) \boxtimes_{\mathrm{HLZ}} M_3$
is proved at the level of generalized modules, not at chain level. The
programme's chain-level status is as follows.
\begin{enumerate}
\item[(a)] \emph{Cohomological fusion (classical).} For $\cA$
  $C_2$-cofinite of CFT type, HLZ14 + Huang08 establish fusion on
  $\Repq(\cA)_{\mathrm{MTC}\text{-quotient}}$ as a rigid braided tensor
  category with associator iso; at integrable level this is
  quasi-triangular and matches the spectral $R(z)$ on eigenspace
  decompositions through the KL correspondence.
\item[(b)] \emph{Chain-level fusion and K\"unneth.} For $\cA$ chirally
  Koszul and $C_2$-cofinite with conilpotent augmented modules
  $M_1, M_2$, we DEFINE the chain-level chiral fusion as the derived
  cotensor
  \[
    M_1 \boxtimes^{\mathrm{ch}}_\cA M_2
      \;:=\; \Omega^{\mathrm{ch}}\bigl(B(M_1) \,\otimes_{B(\cA)}^{\mathbb{L}}\, B(M_2)\bigr),
  \]
  computed inside the Francis--Gaitsgory ambient of
  Remark~\ref{rem:bar-cobar-ambient-clarification}. The
  K\"unneth-type formula
  \[
    B\bigl(M_1 \boxtimes^{\mathrm{ch}}_\cA M_2\bigr)
      \;\simeq\;
    B(M_1) \,\otimes_{B(\cA)}^{\mathbb{L}}\, B(M_2)
  \]
  holds on the conilpotent sub-category as a direct consequence of
  bar--cobar Quillen equivalence
  (Theorem~A${}^{\infty,2}$); it is the module-level avatar of the
  HDC on $\Repq(\cA)$. For non-rational $\cA$ (including logarithmic
  families $W(p)$), chain-level associativity is conjectural; the
  cohomological associator of HLZ14 lifts via the standard spectral
  sequence.
\item[(c)] \emph{Chain-level associativity.} The associator
  isomorphism
  $\alpha_{M_1,M_2,M_3}\colon
   M_1 \boxtimes^{\mathrm{ch}} (M_2 \boxtimes^{\mathrm{ch}} M_3)
   \simeq (M_1 \boxtimes^{\mathrm{ch}} M_2) \boxtimes^{\mathrm{ch}} M_3$
  is CONDITIONAL on the chiral Higher Deligne theorem
  (cf.~\textsf{chiral\_higher\_deligne.tex}).
  Under that theorem, $\alpha$ equals the $E_2$-brace composed with a
  Drinfeld associator $\Phi \in \GRT_1(\Q)$; under $\Phi = \Phi_{\KZ}$
  it reduces to the Huang iterated composition associator, matching
  HLZ14 on $H^\bullet$.
\end{enumerate}
\emph{Scope boundaries.} (a) unconditional under $C_2$-cofiniteness;
(b) proved as derived-cotensor formula on conilpotent modules; (c)
conditional on chiral Higher Deligne, reducing to HLZ14 at
cohomology.
\end{remark}

\begin{remark}[Factorisation ambient category atlas]
\label{rem:factorisation-ambient-category-atlas}
\index{factorisation ambient!atlas}%
\index{Ran space!ambient categories}%
\index{IndCoh@$\IndCoh(\Ran(X))$}%
\index{D-mod@$\Dmod(\Ran(X))$}%
\index{constructible derived category}%
Several derived-category ambients surround the Ran space; each
backbone theorem of this volume lives natively in one of them. This
atlas fixes the assignment once for all, records the bridges, and
resolves the potential conflict between the BD~\cite{BD04} ambient
$\Dmod(\Ran(X))$ used for the factorisation adjunction in
Remark~\ref{rem:bar-cobar-ambient-clarification} and the
$\IndCoh(\Ran(X))$ ambient used in the convolution $L_\infty$-algebra
of the physics-bridge proof (tangent complex of the $R$-family FMP
produced in $\mathrm{IndCoh}(\mathrm{Ran}(X))$).
\begin{enumerate}[label=\textup{(\Alph*)}]
\item \emph{$\Dmod(\Ran(X))$ (Beilinson--Drinfeld~\cite[\S 3.4]{BD04}).}
 Primary ambient for \emph{factorisation algebras} and for the
 classical (Quillen) form of Theorem~A. Bar--cobar as a Quillen
 adjunction (conilpotent-complete locus) and the pseudo-tensor
 Koszul-duality statements are stated here.
\item \emph{$\IndCoh(\Ran(X))$ (Gaitsgory--Rozenblyum~\cite[IV.5, V]{GR17}).}
 Correct ambient for the \emph{$(\infty,2)$-categorical} form
 (Theorem~A${}^{\infty,2}$, properad-level bar--cobar), for the
 unbounded-sheaf regime, and for the convolution dg-Lie
 $\mathfrak g^{\mathrm{SC}}_X$ controlling the formal moduli
 problem of logarithmic $\SCchtop$-algebras. Needed for
 convergence of twisting-cochain limits and for the
 Ind-completion required by Pridham--Lurie formal-moduli
 correspondences.
\item \emph{Constructible $D^b_c(\Ran(X))$
 (Beilinson--Bernstein--Deligne + Saito).}
 Correct ambient for the \emph{modular-bootstrap} and
 \emph{weight-filtration} theorems (Theorem~D tensor-curvature,

 route to Koszulness). Mixed Hodge modules (Saito weight

 not in $\Dmod$-without-weight; the weight filtration on

 belongs to this ambient.
\item \emph{Bridges.}
 \begin{enumerate}[label=\textup{(B\arabic*)}]
 \item $\Dmod(\Ran(X)) \hookrightarrow \IndCoh(\Ran(X))$
 on the bounded holonomic / factorisation sub-category:

 $\Dmod(Y)^{\mathrm{hol},\mathrm{bdd}} \simeq

 restricts to factorisation objects compatibly with the
 star-tensor; on this locus (A) and (B) compute the same

 \item $D^b_c(\Ran(X)) \hookrightarrow \Dmod(\Ran(X))$ via

 \item The three bridges assemble into a commutative triangle
 of $(\infty,1)$-functors after restriction to the regular
 holonomic factorisation locus, on which (A), (B), (C)
 compute the same cohomological invariants.
 \end{enumerate}
\item \emph{Programme-level assignments (canonical).}
 \begin{itemize}
 \item Theorem~A (classical Quillen form) and
 Koszul--bar--cobar pentagon: ambient (A),

 \item Theorem~A${}^{\infty,2}$ (properad, $(\infty,2)$-cat,

this atlas.
\end{remark}

\begin{remark}[Three monoidal levels: ambient, chiral algebras, representations]
\label{rem:chir-alg-three-monoidal-levels}
\index{monoidal category!three levels}
\index{braided monoidal!representation category}
\index{pseudo-tensor!chiral algebras}
Refining Remark~\ref{rem:bar-cobar-ambient-clarification}, the word
``monoidal'' attaches to chiral algebra theory at three strictly
distinct categorical levels; confusing them is the source of
several common slips (FM56; V2-AP1).
\begin{enumerate}
\item \emph{Ambient $(\Dmod(\Ran(X)), \otimes^\star)$: symmetric
 monoidal.} The Francis--Gaitsgory factorisation ambient
 $\Fact(X) \subset \Dmod(\Ran(X))$ is symmetric monoidal under the
 $\star$-tensor product
 (\cite[IV.5.2]{GR17}), with symmetry induced by the obvious
 $\Sigma_n$-action on Ran-strata. Bar--cobar adjunction is
 formulated at this level.
\item \emph{Chiral algebras $\mathrm{ChirAlg}(X)$ themselves:
 pseudo-tensor, \emph{not} symmetric monoidal.} Chiral algebras
 form a pseudo-tensor category in the sense of
 Beilinson--Drinfeld~\cite[\S 1.1]{BD04}, equivalently a
 coloured $\infty$-operad; there is no bifunctor
 $\mathrm{ChirAlg}(X) \times \mathrm{ChirAlg}(X) \to
 \mathrm{ChirAlg}(X)$ realising a tensor of two chiral algebras.
 Promoting this pseudo-tensor structure to a symmetric monoidal
 structure would require a unit object and a bifunctor, both
 absent. An intermediate \emph{braided} monoidal structure on
 $\mathrm{ChirAlg}(X)$ is likewise absent: there is no chiral
 swap $\cA \otimes \cB \to \cB \otimes \cA$ because there is no
 underlying tensor. The correct statement is that
 $\mathrm{ChirAlg}(X)$ is a pseudo-tensor category full stop;
 braidings appear one categorical level up.
\item \emph{Representations $\Rep^{E_1}_q(\cA)$ for $\cA$ chirally
 Koszul: braided monoidal in $\mathrm{Cat}$.} When $\cA$ carries a
 spectral $R$-matrix $R(z)$ (as in the affine lineage, or more
 generally via the chiral coproduct of the Unified Chiral Quantum
 Group Theorem; cf.~\cite[\S 4]{DNP25}), the category of
 $E_1$-chiral modules $\Rep^{E_1}_q(\cA)$ is braided monoidal with
 braiding $\beta_{M,N}(z) = R(z) \circ \sigma$ of meromorphic
 dependence on the spectral parameter $z$; Yang--Baxter is the
 braid-relation coherence. This is the
 \emph{Drinfeld-centre-style} braiding, obtained via
 half-braidings
 $\sigma_A(z)\colon a \otimes n \mapsto \sum \Delta_z(a)_{(2)} n
 \otimes \Delta_z(a)_{(1)}$, and lives on the
 representations, never on $\cA$ itself.
 If $\cA$ carries a conformal vector $T(z)$ at
 non-critical level, the $\Theta$-twist $\theta_M = q^{2L_0}$
 furnishes a balancing, upgrading the braiding to a
 \emph{balanced} (ribbon) structure; this is the categorical
 shadow of $E_3$-topologisation
 (cf.~Theorem~\ref{thm:topologization} and FM57/FM215).
\item \emph{Critical level $k = -h^\vee$: degeneration.} At the
 critical level Sugawara degenerates, the fusion ring becomes
 finite abelian (Feigin--Frenkel centre), and the braiding of
 item~(iii) becomes degenerate on $\Rep_q(\cA)$ itself; only the
 Drinfeld centre $Z(\Rep_q(\cA))$ retains a non-degenerate
 braiding via half-braidings. Thus the braided structure
 survives at the level of centres even when it collapses on the
 representation category directly.
\end{enumerate}
In particular: writing ``$\mathrm{ChirAlg}(X)$ is symmetric
monoidal'' (or ``braided monoidal'') is an FM56-class error;
level~(i) is symmetric monoidal, level~(ii) is pseudo-tensor only,
level~(iii) is where braiding appears.
\end{remark}

\begin{remark}[Model structure ambient for classical bar--cobar]
\label{rem:model-structure-ambient-for-bar-cobar}
The classical Quillen-equivalence form of Volume~I Theorem~A lives
in a specific model-categorical ambient that we now make explicit,
since chiral algebras do not inherently carry a model structure
(they form only a pseudo-tensor category, cf.\ item~(b) of
Remark~\ref{rem:ambient-category-Koszul-bar-cobar}). The
$(\infty,2)$-upgrade of Theorem~\ref{thm:properad-bar-cobar}
(properad-level bar--cobar in the factorisation ambient of
Gaitsgory--Rozenblyum~\cite[IV.5]{GR17}) is strictly stronger than
the classical Quillen form and is obtained from it by
$\infty$-categorical localisation plus properadic extension.

\smallskip
\noindent\emph{Francis--Gaitsgory model structure on $\Fact(X)$.}
Following~\cite[\S 2-3]{FG12} (Francis--Gaitsgory
\textsc{arXiv}:1110.5802) and~\cite[IV.5.3]{GR17}, the category
$\Fact(X)$ of factorisation algebras on the Ran space $\Ran(X)$
of a smooth curve carries a cofibrantly generated model structure
in which
\begin{itemize}
\item \emph{weak equivalences} are levelwise quasi-isomorphisms on
 each finite-arity stratum
 $\Ran(X)_{\leq n} = \coprod_{k\leq n} X^k / \Sigma_k$ (equivalently,
 maps inducing quasi-isomorphisms on all finite unordered
 products),
\item \emph{fibrations} are levelwise surjections in the underlying
 $\Dmod(X^k/\Sigma_k)$ categories (equivalently, underlying
 Reedy fibrations after the $\Sigma$-equivariant unfolding),
\item \emph{cofibrations} are retracts of generalised free
 extensions (cell attachments along $\star$-tensor-free
 factorisation algebras).
\end{itemize}
This model structure is compatible with the $\star$-tensor
symmetric monoidal structure of~\cite[IV.5.2]{GR17}, making
$\Fact(X)$ a symmetric monoidal model category in the sense
of Hovey~\cite[Definition~4.2.6]{Hovey99}.

\smallskip
\noindent\emph{Transfer to chiral algebras.} The category
$\mathrm{ChirAlg}^{\mathrm{aug},\mathrm{conil},\mathrm{comp}}(X)$ of
augmented, conilpotent, complete chiral algebras arises as the
comma category over the augmentation morphism
$\epsilon \colon \mathbf{1} \to \cA$ in the pseudo-tensor category
$\mathrm{ChirAlg}(X)$ of Beilinson--Drinfeld; under the embedding
of~(c) above it sits as a full sub-$\infty$-operad of $\Fact(X)$
with morphisms the ambient factorisation morphisms. A model
structure is \emph{transferred} along this embedding via the
standard right-induction criterion of
Hinich~\cite[\S 2]{Hinich97} (verified for chiral algebras
in~\cite[\S 4]{FG12}):
\begin{itemize}
\item \emph{weak equivalences} of chiral algebras are maps whose
 underlying factorisation morphism is a Francis--Gaitsgory weak
 equivalence,
\item \emph{fibrations} are maps whose underlying factorisation
 morphism is a Francis--Gaitsgory fibration,
\item \emph{cofibrations} are retracts of free extensions along
 semifree chiral algebras
 $\mathrm{Free}^{\mathrm{ch}}(V) = \bigoplus_n V^{\otimes^{\mathrm{ch}} n}_{\Sigma_n}$
 on a cofibrant $\Dmod(X)$-object~$V$.
\end{itemize}
The transfer is valid because (i) the ambient Francis--Gaitsgory
model structure is cofibrantly generated with small generating
(trivial-)cofibrations, and (ii) the forgetful functor
$\mathrm{ChirAlg}(X) \hookrightarrow \Fact(X)$ preserves filtered
colimits and detects weak equivalences. The pseudo-tensor structure
on chiral algebras is preserved (not promoted to a symmetric
monoidal structure): the transferred model structure records
homotopical information along the embedding without asserting a
bifunctor $\otimes^{\mathrm{ch}}$ on chiral algebras.

\smallskip
\noindent\emph{Theorem~A as Quillen equivalence.} In this transferred
model-categorical ambient, the bar and cobar functors
\[
  \Omega^{\mathrm{ch}} \colon
  \mathrm{CoAlg}_{E_1}^{\mathrm{ch,conil,comp}}(X)
  \;\rightleftarrows\;
  \mathrm{ChirAlg}^{\mathrm{aug},\mathrm{conil},\mathrm{comp}}(X)
  \colon \bar{B}^{\mathrm{ch}}
\]
form a \emph{Quillen pair} between the transferred model structures
on conilpotent-complete $E_1$-chiral coalgebras and
augmented-conilpotent-complete $E_1$-chiral algebras; this Quillen
pair is a \emph{Quillen equivalence} on the chirally-Koszul locus
of Theorem~\ref{thm:filtered-koszul}. This is the precise
model-categorical content of Volume~I Theorem~A specialised to the
pseudo-tensor sub-$\infty$-operad.

\smallskip
\noindent\emph{Relation to the $(\infty,2)$-upgrade.} Passage to the
underlying $\infty$-category via Dwyer--Kan hammock localisation
recovers the $(\infty,1)$-categorical bar--cobar adjunction on
factorisation (co)algebras; passage to the $(\infty,2)$-enhancement
of~\cite[IV.5]{GR17} (which records the 2-morphisms between
bar--cobar functor pairs, not merely invertible natural
transformations) recovers the properad-level statement of
Theorem~\ref{thm:properad-bar-cobar}. The $(\infty,2)$-upgrade is
strictly stronger than the classical Quillen form in two respects:
(a)~it extends bar--cobar from operads (one output) to properads
(multiple outputs and inputs, cf.\ Hackney--Robertson~\cite{HR17}),
and (b)~it captures the coherent 2-morphism data between bar--cobar
functor pairs that is discarded by passing to homotopy categories.
The classical Quillen form is the $(1,1)$-truncation of the
$(\infty,2)$-form; conversely, the $(\infty,2)$-form is the
properadic $\infty$-categorical completion of the Quillen form.

\smallskip
\noindent\emph{Pointer for curved genus~$g \geq 1$.} The curved
extension to $g \geq 1$ uses Positselski's CDG model structure on
curved coalgebras~\cite{Positselski2011} transferred along the
same Francis--Gaitsgory embedding; see
Theorem~\ref{thm:curved-bar-cobar-genus-ge-1}(i) for the precise
CDG Quillen-equivalence statement and its conilpotent-complete
inversion locus.
\end{remark}

\begin{remark}[Triangulated category vs.\ stable $\infty$-category;
programme flavour choice]
\label{rem:triangulated-vs-stable-infty-cat}
\index{triangulated category!vs.\ stable $\infty$-category}
\index{stable infinity-category@stable $\infty$-category!vs.\ triangulated}
\index{Verdier!triangulated category}
\index{Lurie!stable $\infty$-category}
The programme manipulates categories of bar complexes, factorization
algebras, and curved coalgebras in two related but strictly different
formalisms. Conflation is silent and frontier-relevant; we fix the
assignment once.
\begin{enumerate}[label=\textup{(\alph*)}]
\item \emph{Triangulated category (Verdier~\cite{Verdier1996}, orig.\
1967).} A category equipped with a shift functor $[1]$ and a class
of \emph{distinguished triangles} satisfying (TR1)--(TR4). Obtained
as the homotopy category $\mathrm{Ho}(\cC)$ of a stable
$\infty$-category, or as $K(\cA)/\text{acyclics}$ for a dg-category
$\cA$. Carries no higher homotopy-coherent information: cones are
defined only up to non-canonical isomorphism; no functoriality of
the cone construction.
\item \emph{Stable $\infty$-category (Lurie~\cite{LurieHA}).} An
$(\infty,1)$-category with a zero object in which every morphism
admits a fibre and cofibre sequence, and the loop/suspension
adjunction $\Omega \dashv \Sigma$ is an equivalence. Carries full
homotopy-coherent data: cones are canonical (functorial) up to
contractible choice; mapping spectra (not just mapping sets) are
intrinsic; all limits/colimits are homotopy-coherent.
\item \emph{Bridge.} For every stable $\infty$-category $\cC$, the
homotopy category $\mathrm{Ho}(\cC)$ is naturally a triangulated
category (\cite[\S 1.1.2]{LurieHA}). The collapse
$\cC \rightsquigarrow \mathrm{Ho}(\cC)$ forgets higher coherence
and retains only distinguished triangles. Every triangulated
category of geometric origin in this programme arises as
$\mathrm{Ho}(-)$ of a canonical stable $\infty$-category (via a
dg-enhancement or a factorization $\infty$-category model).
\end{enumerate}
\emph{Programme assignments.}
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Theorem~A (classical form).} Statable at the
triangulated level: the bar--cobar Quillen equivalence descends to
an equivalence of triangulated derived categories
$D(E_1\text{-}\mathrm{ChirAlg}^{\mathrm{aug}}) \simeq
D^{\mathrm{conil}}(E_1\text{-}\mathrm{ChirCoAlg})$. Loday--Vallette
\cite{LV12} operate at this level.
\item \emph{Theorem~A$^{\infty,2}$ upgrade
(Francis--Gaitsgory~\cite{GR17}).} Requires the stable
$\infty$-categorical refinement: the $(\infty,2)$-categorical
equivalence of factorization algebras on $\Ran(X)$ is not visible
at the triangulated level. Properad-level extension, $R$-twisted
$\Sigma_n$-descent on $\Ran(X)^{\mathrm{ord}} \to \Ran(X)$, and
all $(\infty,2)$-coherent statements are intrinsically stable
$\infty$-categorical.
\item \emph{Curved bar--cobar at genus $g \ge 1$.} Lives in the
coderived category $D^{\mathrm{co}}$ of
Remark~\ref*{rem:positselski-three-derived-categories}(b),
which is itself the homotopy category of a stable $\infty$-category
(Positselski's curved-CDG model, a variant of the Lurie stable
$\infty$-framework accommodating $d^2 = m_0 \cdot (-)$ with
non-zero curvature $m_0 = \kappa \cdot \omega_g$). Curved quasi-isomorphisms
are intrinsically higher-coherent; triangulated-level statements
suffice for cone/cofibre arguments but not for modular operad
composition at $g \ge 2$.
\item \emph{Modular operad composition, chiral Higher Deligne,
half-braiding constructions.} Stable $\infty$-categorical;
triangulated-level statements lose the coherence data (Drinfeld
associator, Tamarkin zigzag, half-braiding natural transformations)
these arguments manipulate.
\end{enumerate}
\emph{Compatibility.} Every theorem in Volumes I--II is statable at
the triangulated level via $\mathrm{Ho}(-)$; the stable
$\infty$-categorical refinement is required only where coherence
data appear explicitly (Theorem~A$^{\infty,2}$, chiral Higher
Deligne, $(\infty,2)$-properad bar--cobar, modular operad
composition at $g \ge 2$). Every triangulated statement in the
manuscript admits a canonical stable $\infty$-categorical lift via
its dg-enhancement; conversely the collapse
$\cC \rightsquigarrow \mathrm{Ho}(\cC)$ preserves all
equivalence statements but strictly loses coherence data. See the
Vol~II first-principles cache entry on triangulated-vs-stable
consistency.
\end{remark}

\begin{remark}[Positselski's three derived categories; programme
flavour choice]
\label{rem:positselski-three-derived-categories}
Positselski's nonhomogeneous Koszul-duality framework
\cite{Positselski2011} operates in three distinct derived
categories, each with a precise role in this volume.
Inconsistency between them is silent and load-bearing; we fix the
flavour of each theorem here.
\begin{enumerate}[label=\textup{(\alph*)}]
\item \emph{Derived category $D$.} The bounded derived category
 $D^b(\mathrm{CoAlg}_{E_1}^{\mathrm{ch}})$ of conilpotent
 $E_1$-chiral dg coalgebras, obtained by inverting
 quasi-isomorphisms. This is the home of the flat bar complex
 $\barB(\cA)$ (with $d^2 = 0$) at genus $0$.
\item \emph{Coderived category $D^{\mathrm{co}}$.} The quotient
 of the homotopy category of CDG coalgebras $(C, d, \kappa)$ by
 the thick subcategory of \emph{coacyclic} objects (the smallest
 triangulated subcategory containing totalisations of short
 exact sequences of CDG coalgebras and closed under coproducts).
 Home of the curved bar complex
 $\barB^{\mathrm{curv}}(\cA) = (\barB(\cA), d, \kappa \cdot
 \omega_g)$ at genus $g \ge 1$, where $d^2 \neq 0$ precludes
 $D$.
\item \emph{Filtered derived category $D^{\mathrm{filt}}$.} The
 filtered derived category of Beilinson--Bernstein--Deligne
 \cite{BBD82} (or its mixed-Hodge-module refinement
 \cite{Saito90}), with weight filtration $W_\bullet$ whose
 graded pieces are objects of $D$. Home of the modular
 bootstrap complex: the conformal-weight filtration on $\cA$
 and the genus tower $\{F_g\}$ live here.
\end{enumerate}
\emph{Embeddings.} Over characteristic zero
\cite[\S 6.1--6.3]{Positselski2011}, for conilpotent dg coalgebras
acyclicity coincides with coacyclicity, yielding a fully faithful
embedding $\iota\colon D \hookrightarrow D^{\mathrm{co}}$ on the
conilpotent-complete locus; $\iota$ is \emph{not} essentially
surjective (curved objects lie outside its image, by
Proposition~\ref*{prop:coderived-vs-derived} in
\texttt{modular\_swiss\_cheese\_operad.tex}). Forgetting the
filtration gives a triangulated functor $\mathrm{forget}\colon
D^{\mathrm{filt}} \to D$; its graded-piece refinement
$\mathrm{gr}^W_\bullet\colon D^{\mathrm{filt}} \to
D^{\mathbb Z}$ is exact. The three are related by the zigzag
\[
 D^{\mathrm{filt}}
 \xrightarrow{\mathrm{forget}} D
 \xhookrightarrow{\iota} D^{\mathrm{co}}.
\]
\emph{Programme flavour assignments.}
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Theorem~A (bar--cobar Quillen equivalence, genus
 $0$).} Lives in $D$. Both sides flat ($d^2 = 0$); the classical
 bar--cobar adjunction of Loday--Vallette~\cite{LV12} specialised
 to the factorization ambient of the preceding remark applies
 unchanged.
\item \emph{Theorem~B (bar--cobar inversion on the Koszul
 locus).} The underlying adjunction lives in $D$; at critical
 (affine KM at $k = -h^\vee$), admissible, or curved
 (minimal-model Virasoro) loci where $d^2 \neq 0$, inversion is
 re-interpreted in $D^{\mathrm{co}}$ via $\iota$. The unified
 Positselski--Grothendieck pro-category of Theorem~B's UPGRADE
 form is precisely $D^{\mathrm{co}}$ with a compatible
 conformal-weight filtration.
\item \emph{Curved bar--cobar at genus $g \ge 1$}
 (\texttt{modular\_swiss\_cheese\_operad.tex}, W13 install).
 Lives in $D^{\mathrm{co}}$. Curvature
 $m_0 = \kappa \cdot \omega_g$ is Positselski's curvature datum;
 quasi-isomorphisms are curved (cones coacyclic, not acyclic).
\item \emph{Cross-genus MC equation and modular bootstrap.} Lives
 in $D^{\mathrm{filt}}$. Genus filtration and conformal-weight
 filtration are compatible; each associated graded piece is an
 object of $D^{\mathrm{co}}$, and at $g = 0$ descends through
 $\iota$ to $D$. The modular-bootstrap $H^2 = 0$ result
 (Theorem~\ref*{thm:curved-dunn-H2-vanishing-all-genera} in
 \texttt{curved\_dunn\_higher\_genus.tex}) is a statement in
 $D^{\mathrm{filt}}$; its image under $\mathrm{forget}$ is the
 genus-wise curved Dunn theorem in $D^{\mathrm{co}}$.
\end{enumerate}
\emph{Compatibility.} Every theorem of Volume~II manipulating bar
complexes declares (explicitly or by inheritance from Theorem~A /
Theorem~B / curved bar--cobar / modular bootstrap) which flavour
it uses; cross-flavour comparisons are mediated by $\iota$ or
$\mathrm{forget}$ above. Violations are the silent failure mode
this remark installs guards against; see the W13
\texttt{appendices/first\_principles\_cache.md} entry on
three-derived-category consistency.
\end{remark}

\begin{remark}[Three flavors of $C_2$-cofiniteness; canonical
registry]
\label{rem:c2-cofiniteness-flavors}
\index{C2-cofiniteness@$C_2$-cofiniteness!three flavors}
\index{Arakawa!$C_2$-cofiniteness flavors}
The programme invokes Arakawa-type $C_2$-cofiniteness at multiple
sites (DS--Hochschild bridge, chain-level E\textsubscript{3}
topologization, HKR scope, minimal-model class transport, universal
holography). In the literature these are three distinct properties;
conflation is mathematical, not cosmetic. This remark fixes the
nomenclature once and binds each downstream invocation to exactly one
flavor.
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Classical $C_2$-cofiniteness (Zhu finite-dimensionality).}
$\dim_k V/C_2(V) < \infty$, equivalently Zhu's algebra $A(V)$ is
finite-dimensional (Zhu~\cite{Zhu96}; see
Dong--Li--Mason~\cite{DLM98}). On a rational VOA this forces
modular invariance of characters (Huang~\cite{Hua05}) and a
finite, semisimple module category. This is the flavor used in
classical rationality statements (Arakawa's semisimplicity
theorem~\cite{Arakawa15}).
\item \emph{Generalised (logarithmic) $C_2$-cofiniteness.}
$V/C_2(V)$ finite-dimensional \emph{without} rationality, so that
$A(V)$ is Artinian but not semisimple; equivalently the
associated-graded commutative algebra $\operatorname{gr}_C(V) =
\bigoplus_n C_n(V)/C_{n+1}(V)$ is a finitely generated Poisson
algebra, but the module category is non-semisimple (logarithmic).
Examples: the triplet $W(p)$ at $p\ge 2$, the $N{=}2$ superconformal
algebra at admissible $c_{p,q}$, and the minimal-model Virasoro
quotients at non-generic $c$
(Adamovi\'c--Milas~\cite{AdamovicMilas08}).
\item \emph{Strong (Arakawa--Khan--Zeng) $C_2$-cofiniteness.}
$V$ is $C_2$-cofinite in sense~(i) \emph{and} the strong generators
of $V$ are \emph{freely} strongly generating modulo $C_2(V)$, i.e.\
$\operatorname{gr}_F V$ is a free polynomial (super)algebra on a
finite set of weighted generators (Arakawa~\cite{Arakawa15}
for DS-reduced $\cW_k(\fg, f)$; Khan--Zeng for the $3$d Poisson
sigma-model input). This is strictly stronger than~(i): it
provides the polynomial $\lambda$-bracket datum required for the
chain-level chiral HKR degeneration and the $E_3$-topologization
of the freely-generated PVA lane (AP67, FM82, FM216).
\end{enumerate}
\emph{Binding of downstream invocations.}
\begin{itemize}
\item \textbf{DS--Hochschild compatibility bridge}
(Theorem~\ref{thm:chd-ds-hochschild}, class~$\mathsf{M}$ global
triangle, FM126/FM185/FM214): flavor~(iii). Arakawa's theorem for
$\cW_k(\fg, f_{\mathrm{prin}})$ supplies free strong generation,
which is what drives the HPL strong-deformation-retract on
$\operatorname{gr}_F$ and the chain-level finiteness of each
weight space.
\item \textbf{Chain-level HKR degeneration of the chiral Hochschild
spectral sequence}
(Theorem~\ref{thm:chd-deligne-tamarkin} and
Remark on HKR scope in
Chapter~\ref{ch:chiral-higher-deligne}): flavor~(iii) at generic
non-critical $c$; flavor~(ii) at minimal-model $c_{p,q}$ and at
logarithmic loci, where the degeneration holds after replacing
$\operatorname{gr}_F V$ with the Artinian Poisson algebra
$V/C_2(V)$ and tracking the logarithmic obstruction via the
modular-bootstrap complex.
\item \textbf{Universal Holography functor} (class coverage
G/L/C/$\mathsf{M}$, Theorem~F in the climax): flavor~(iii) on
the freely-generated lane (class~$\mathsf{G}$/$\mathsf{L}$/$\mathsf{C}$
and $\cW_k(\fg, f)$ in class~$\mathsf{M}$); flavor~(ii) suffices on
the Virasoro minimal-model lane via the Feigin--Fuchs
null-vector-corrected shadow tower (FM76 heal).
\item \textbf{Minimal-model class transport and the PBW range of the
bar complex} (logarithmic W-algebras,
Chapter~\ref{ch:log-w-algebras}): flavor~(ii). Free strong
generation generically fails (V2-AP22, AP67); the bar complex is
controlled by the Artinian Poisson algebra $V/C_2(V)$ rather than
a polynomial $\lambda$-bracket, and class-M assignment requires
the full shadow tower.
\item \textbf{Classical rationality / modular-invariance inputs}
(Arakawa semisimplicity theorem cited in
Chapter~\ref{ch:hochschild}, thm:vertex-completion): flavor~(i).
\end{itemize}
\emph{Directive.} Every downstream invocation of ``Arakawa
$C_2$-cofiniteness'' must specify one of~\textup{(i)}, \textup{(ii)},
\textup{(iii)} at the cite site; ambiguity resolves to the
\emph{strongest} flavor required by the argument and must be verified
against the hypothesis catalogue of the cited theorem. Conflations of
flavors across~\textup{(i)}/\textup{(ii)}/\textup{(iii)} are audit
failures (AP40, AP67, V2-AP22).
\end{remark}

\begin{theorem}[Boundary algebra from a vacuum object;
\ClaimStatusProvedHere]
\label{thm:boundary-algebra-from-vacuum}
Let\/ $\cC$ be an open/closed factorization dg-category on
$(X, D, \tau)$ in the sense of
Definition~\textup{\ref{def:oc-factorization-category}}, and let\/
$b \in \cC(J)$ be a compact generator\/ for a boundary interval
$J \subset I_p$. Then:
\begin{enumerate}[label=\textup{(\alph*)}]
\item The endomorphism algebra
 \[
 A_b \;:=\; \RHom_{\cC(J)}(b, b)
 \]
 is an\/ $\Ainf$ chiral algebra in the sense of
 Definition~\textup{\ref{def:ainfty_chiral}}. The higher
 products\/ $m_k$ arise by restricting the factorization functors
 to mixed collision strata with\/ $k$ closed insertions
 evaluated on the endomorphism complex of\/~$b$.

\item The assignment\/ $b \mapsto A_b$ is Morita-functorial: a
 different compact generator\/ $b' \in \cC(J')$ yields an\/
 $\Ainf$ chiral algebra\/ $A_{b'}$ that is\/ $\Ainf$-Morita
 equivalent to\/~$A_b$. The true invariant is the category\/
 $\cC$, not a chosen algebra presentation.
\end{enumerate}
\end{theorem}

\begin{proof}
\textit{Part}~(a).
By local constancy~(iv), the dg-category $\cC(J)$ is independent
of the choice of interval $J \subset I_p$ up to canonical
equivalence, so it suffices to work with a single $J$. The
endomorphism complex $A_b = \RHom_{\cC(J)}(b,b)$ inherits an
associative product from composition in~$\cC$.

The construction of the higher products $m_k$ for $k \ge 2$ proceeds
by restricting the factorization structure to the endomorphism
complex of~$b$, keeping explicit track of both the holomorphic and
topological variables. The factorization structure of~$\cC$
provides, for each $k$-element configuration
$(z_1,\ldots,z_k) \in \FM_k(\C)$ and each ordered open
configuration $(t_1 < \cdots < t_k) \in \Conf^{\mathrm{ord}}_k(\R)$,
a multilinear functor
\[
 \mu_k \colon
 \cC(J_1) \times \cdots \times \cC(J_k)
 \;\longrightarrow\; \cC(J),
\]
where the $k$ ordered boundary insertions at $t_1 < \cdots < t_k$ divide $J$ into $k+1$ subintervals; $k$ of these carry the input morphisms $a_1, \ldots, a_k \in A_b$, and the remaining subinterval (to the right of $t_k$) carries the output. Restricting to $\Hom(b,b)$ in each slot gives the
$k$-ary operation:
\[
 m_k(a_1,\ldots,a_k)(z_1,\ldots,z_k;\, t_1,\ldots,t_k)
 \;:=\;
 \mu_k(a_1,\ldots,a_k)\big|_{b,\ldots,b},
\]
where $a_i \in A_b = \Hom(b,b)$ for each~$i$.

The two sets of variables play distinct geometric roles:
\begin{itemize}
\item \textbf{Holomorphic parameters} (from the $\C$-colour). The
 $k-1$ spectral
 parameters $\lambda_i = z_i - z_{i+1}$ (using the consecutive-difference convention of Notation~\ref{not:spectral}) arise from the
 holomorphic dependence on the $\FM_k(\C)$ factor. These are the
 OPE spectral variables: the operation $m_k$ is a formal Laurent
 series in $\lambda_1,\ldots,\lambda_{k-1}$ with poles from
 collisions $z_i \to z_{i+1}$.
\item \textbf{Topological ordering} (from the $\R$-colour). The
 $E_1$ ordering from the
 $\Conf^{\mathrm{ord}}_k(\R)$ factor ensures the planar
 $\Ainf$ convention: inputs are ordered $a_1,\ldots,a_k$
 according to the total order $t_1 < \cdots < t_k$ on boundary
 insertions. Local constancy~(iv) ensures the result depends
 only on the combinatorial ordering, not on the specific
 positions~$t_i$.
\end{itemize}
The resulting operation has the type
\[
 m_k \colon
 C_\ast\!\bigl(\FM_k(\C)\bigr) \otimes A_b^{\otimes k}
 \;\longrightarrow\; A_b((\lambda_1))\cdots((\lambda_{k-1})).
\]
The $\Ainf$ identity $\sum_{i+j=n+1} m_i \circ m_j = 0$ now
follows from Stokes' theorem on the operation space
$\FM_k(\C) \times \Conf^{\mathrm{ord}}_k(\R)$ of $\SCchtop$.
Concretely, the codimension-one boundary strata of this product
are products of lower-dimensional operation spaces, and they are
exactly the Stasheff faces: each boundary stratum
$\partial_\sigma\!\bigl(\FM_k(\C) \times
\Conf^{\mathrm{ord}}_k(\R)\bigr)$ decomposes as
$\bigl(\FM_i(\C) \times \Conf^{\mathrm{ord}}_i(\R)\bigr)
\times
\bigl(\FM_j(\C) \times \Conf^{\mathrm{ord}}_j(\R)\bigr)$
with $i + j = k + 1$, where the boundary decomposition uses the Fulton--MacPherson compactification on the $\C$-factor and the Stasheff associahedron on the $\R$-factor, and the Stokes identity on the total
boundary $\sum_\sigma \int_{\partial_\sigma} = 0$ is the $\Ainf$
relation (Theorem~\ref{thm:FM-calculus}). The local model
axiom~(v) ensures these boundary identifications are compatible
with the $W(\SCchtop)$-algebra coherence data.
Sesquilinearity follows from translation covariance in the
holomorphic coordinate.

\smallskip

\textit{Part}~(b).
Let $b' \in \cC(J)$ be another compact generator (transported from any other boundary interval via the local-constancy equivalence of axiom~(iv)). The bimodule
$M = \RHom_{\cC(J)}(b, b')$ defines an $\Ainf$-bimodule over $(A_b,
A_{b'})$. Since both $b$ and $b'$ are compact generators, the
natural maps
\[
 M \otimes^{\mathbf{L}}_{A_{b'}} M^\vee \;\xrightarrow{\;\sim\;}\; A_b,
 \qquad
 M^\vee \otimes^{\mathbf{L}}_{A_b} M \;\xrightarrow{\;\sim\;}\; A_{b'},
\]
are quasi-isomorphisms, where $M^\vee = \RHom_{\cC(J)}(b', b)$: this is
the standard Morita equivalence criterion for compact generators of
a dg-category \cite{Kel06,HA}.
\end{proof}

\medskip

The boundary algebra determines the open sector. The bulk is
determined by the categorical derived center: the endomorphism
algebra of the identity functor of~$\cC$, the universal object
acting on every module.

\begin{corollary}[Bulk as chiral Hochschild cochains;
 \ClaimStatusProvedHere]
\label{cor:bulk-derived-center-categorical}
Let\/ $\cC$ be an open/closed factorization dg-category on
$(X,D,\tau)$ with compact generator\/ $b \in \cC(J)$ and boundary
algebra\/ $A_b = \RHom_{\cC(J)}(b,b)$. Then there is a
quasi-isomorphism of cochain complexes
\[
 \End_{\operatorname{Fun}(\cC,\cC)}(\id_\cC)
 \;\simeq\;
 \mathrm{C}^\bullet_{\mathrm{ch}}(A_b, A_b),
\]
where the left side is the endomorphism algebra of the identity
functor\/ \textup{(}the categorical derived center\textup{)} and the
right side is the chiral Hochschild cochain complex of\/~$A_b$.
This is a chain-level identification: both sides are dg algebras
and the quasi-isomorphism is compatible with the brace structures.
In particular, the chain-level bulk is Morita-invariant: it depends
on\/ $\cC$ but not on the choice of compact generator.

On cohomology, this yields the derived center\/
$\cZ^{\mathrm{der}}_{\mathrm{ch}}(\cC)
= H^*(\mathrm{C}^\bullet_{\mathrm{ch}}(A_b, A_b), \delta)$
as a Gerstenhaber algebra.
\end{corollary}

\begin{proof}
For any dg-category $\cC$ with compact generator $b$, the derived
Morita equivalence $\cC \simeq \Perf\text{-}A_b$ (the dg-category
of perfect right $A_b$-modules) induces an equivalence on
endomorphism algebras of the identity functor:
\[
 \End_{\operatorname{Fun}(\cC,\cC)}(\id_\cC)
 \;\simeq\;
 \End_{\operatorname{Fun}(\Perf\text{-}A_b,\,\Perf\text{-}A_b)}(\id)
 \;\simeq\;
 \HH^\bullet(A_b, A_b).
\]
The first equivalence is Morita invariance of the derived center
\cite{Kel06}; the second is the identification of the derived
center of $\Perf\text{-}A_b$ with the Hochschild cochain complex.
In the holomorphic--topological setting, the holomorphic locality
axiom~(iii) of $\cC$ refines $\HH^\bullet(A_b, A_b)$ to the
\emph{chiral} Hochschild cochain complex
$\mathrm{C}^\bullet_{\mathrm{ch}}(A_b,A_b)$, where the
Hochschild differential uses the meromorphic OPE product rather
than a purely algebraic one. The refinement is not a change of
category but a change of product: the underlying cochain complex
is the same bimodule bar complex, but the multiplication maps are
upgraded from algebraic to meromorphic.
Concretely, the Hochschild bar
complex $\barB(A_b, A_b, A_b) = \bigoplus_n A_b \otimes
A_b^{\otimes n} \otimes A_b$ computes $\HH^\bullet$ when equipped
with the algebraic differential; the holomorphic locality of
$\cC$ replaces the algebraic product with the meromorphic OPE
product $m_2(a,b;\lambda)$ on the inputs, upgrading each
$A_b^{\otimes n}$ factor to
$A_b((\lambda_1))\cdots((\lambda_{n-1}))$ and the differential to
$d_{\mathrm{ch}} = \delta_Q + \delta_{\mathrm{Hoch}}^{\mathrm{ch}}$,
where $\delta_{\mathrm{Hoch}}^{\mathrm{ch}}$ is the alternating sum of
face maps defined by the chiral binary product $m_2(-,-;\lambda)$
in place of the algebraic product.
This recovers the chiral Hochschild complex of
Definition~\ref{def:chiral_hochschild}. All three
quasi-isomorphisms in the chain are quasi-isomorphisms of cochain
complexes, not merely identifications of cohomology: Keller's
derived Morita equivalence~\cite{Kel06} operates at the chain
level, and the refinement from $\HH^\bullet$ to chiral Hochschild
cochains replaces the product but preserves the cochain-complex
structure. This recovers
Theorem~\ref{thm:bulk_hochschild} from the categorical viewpoint:
the bulk--Hochschild identification is a consequence of Morita
invariance.
\end{proof}

\begin{remark}[Two presentations of a single theory]
\label{rem:corrected-big-picture}
The bar complex and the open factorization category are two
presentations of the same content, operating at
different levels. Neither is subordinate to the other; each
accesses structures invisible from the other side.

\medskip\noindent\textbf{The bar complex as computational engine.}
The bar complex $\barB^{\mathrm{ch}}(A_b)$ is where the proofs
live. Theorems~A--D+H of Volume~I, the shadow obstruction tower,
the modular characteristic~$\kappa$, the genus expansion
$F_g = \kappa \cdot \lambda_g^{\mathrm{FP}}$, and the graph-sum
calculus through all degrees are stated and proved at the
bar-complex level. The identity $D_A^2 = 0$, which is the
source of the Maurer--Cartan element $\Theta_A$, is a theorem
about the bar differential. Over two thousand pages of proved
mathematics take place in the bar complex.

\medskip\noindent\textbf{The open category as categorical context.}
The open/closed factorization dg-category $\cC$ on $(X,D,\tau)$
is the categorical setting that \emph{explains} the bar complex's
properties. It reveals that the bar complex classifies twisting
morphisms (Theorem~\ref{thm:boundary-algebra-from-vacuum}), that
the chain-level bulk is the Morita-invariant chiral Hochschild
cochain complex
$C^\bullet_{\mathrm{ch}}(A_b, A_b)$
(Corollary~\ref{cor:bulk-derived-center-categorical}), that line
operators are objects of~$\cC$
(Theorem~\ref{thm:lines_as_modules}), and that genus-$g$
amplitudes arise from traces and clutching on~$\cC$
(Definition~\ref{def:oc-factorization-category}(vi)). Different
compact generators $b \in \cC(J)$ yield Morita-equivalent
boundary algebras $A_b = \RHom_{\cC(J)}(b,b)$; the categorical
level is where this invariance is visible. Throughout, the
identification operates at the chain level: the cochain complex
$C^\bullet_{\mathrm{ch}}(A_b, A_b)$ with its brace dg algebra
structure, its $L_\infty$ deformation theory, and its MC moduli
is the primary object. The cohomological derived center
$\cZ^{\mathrm{der}}_{\mathrm{ch}}$ is its Gerstenhaber shadow.

\medskip\noindent\textbf{The universal property relating them.}
The two levels are connected by the universal property of the bar
construction: $\barB^{\mathrm{ch}}(A_b)$ \emph{represents} the
twisting morphism functor $\operatorname{Tw}(-, A_b)$. The category~$\cC$
parametrises objects that couple to~$A_b$ via twisting morphisms;
the bar complex is the explicit coalgebra that classifies these
couplings. The bar complex is to the open category as coordinates
are to a manifold: the computations happen in coordinates; the
geometry lives on the manifold. The bar-cobar adjunction
(Volume~I, Theorem~A) is the representability theorem for
deformations of a vacuum object inside~$\cC$.
\end{remark}

\subsubsection{The universal bulk and its specializations}
\label{subsec:universal-bulk-specializations}
% -------------------------------------------------------------------

The Swiss-cheese theorem
(Volume~I, Theorem~\ref*{V1-thm:thqg-swiss-cheese})
establishes an abstract universality: for any curved
$A_\infty$-chiral algebra~$\cA$, the chiral Hochschild cochain
complex $C^\bullet_{\mathrm{ch}}(\cA, \cA)$ with its brace dg
algebra structure is the universal bulk at the chain level. The
cohomological derived center
$\cZ^{\mathrm{der}}_{\mathrm{ch}}(\cA)
= H^*(C^\bullet_{\mathrm{ch}}(\cA, \cA), \delta)$
is its shadow: a Gerstenhaber algebra that retains the cup product
and Lie bracket but forgets the higher homotopies, the full
$L_\infty$ deformation theory, and the Maurer--Cartan moduli
that classify deformations beyond the formal neighborhood.
The question ``is this the \emph{known}
bulk of a specific physical theory?'' requires additional input,
and must be answered at the chain level.

\begin{proposition}[Specialization to the boundary-linear sector;
 \ClaimStatusProvedHere]
\label{prop:bulk-specialization-boundary-linear}
For a logarithmic $\SCchtop$-algebra with boundary algebra
$A_\partial$ in the boundary-linear exact sector, there is a
quasi-isomorphism of brace dg algebras
\[
 \cO_{\mathrm{bulk}}
 \;\simeq\;
 C^\bullet_{\mathrm{ch}}(A_\partial, A_\partial),
\]
and a quasi-isomorphism of the underlying cochain complexes
\[
 C^\bullet_{\mathrm{ch}}(A_\partial, A_\partial)
 \;\simeq\;
 \End_{\operatorname{Fun}(\cC_\partial,\cC_\partial)}(\id),
\]
where $\cC_\partial$ is the open-sector dg-category and the
second quasi-isomorphism requires compact generation. Both
identifications hold at the chain level\textup{;} the
cohomological derived center
$\cZ^{\mathrm{der}}_{\mathrm{ch}}(\cC_\partial)
= H^*(C^\bullet_{\mathrm{ch}}(A_\partial, A_\partial), \delta)$
is the Gerstenhaber shadow obtained by passing to cohomology.
\end{proposition}

\begin{proof}
The first quasi-isomorphism is
Theorem~\ref{thm:bulk_hochschild} (unconditional for any
logarithmic $\SCchtop$-algebra); it preserves the brace
structure because the map is induced by evaluation of bulk
observables on boundary insertions, which intertwines the
operadic composition on both sides. The second follows from
Keller's identification of the derived center of a compactly
generated dg-category with the Hochschild cochains of any compact
generator's endomorphism algebra~\cite{Kel06}.
\end{proof}

\begin{remark}[The checkable computation]
\label{rem:checkable-bulk-computation}
For Chern--Simons at integrable level~$k$ with gauge group~$G$:
the passage from the chain-level bulk
$C^\bullet_{\mathrm{ch}}(V_k(\fg), V_k(\fg))$ to the known
Verlinde ring requires:
\begin{enumerate}[label=\textup{(\roman*)}]
\item computing $HH^0(V_k(\fg), V_k(\fg))$, which should equal
 the center $Z(\operatorname{Rep}(V_k(\fg)))$ of the
 module category;
\item at integrable levels, the module category is semisimple
 (Huang--Lepowsky), so its center is the Verlinde algebra
 $\bigoplus_\lambda \C \cdot [\lambda]$.
\end{enumerate}
For $\fg = \mathfrak{sl}_2$ at $k = 1$: the module category has
$2$ simple objects ($L_0, L_1$), so $\dim Z = 2$. The
computation $\dim HH^0(V_1(\mathfrak{sl}_2)) = 2$ would give
the derived-center-as-bulk claim its first checkable confirmation.
This computation is within reach of the existing compute
infrastructure.
\end{remark}

\begin{theorem}[Chain-level structure of the universal bulk;
\ClaimStatusProvedHere]
\label{thm:chain-level-bulk}
\label{thm:derived-center-gerstenhaber}
\index{universal bulk!chain-level structure|textbf}
\index{brace dg algebra!chain-level bulk|textbf}
\index{Gerstenhaber algebra!derived center|textbf}
Let\/ $\cC$ be an open/closed factorization dg-category with compact
generator\/ $b$ and boundary algebra\/ $A_b$. The chain-level
object is the chiral Hochschild cochain complex\/
$\mathrm{C}^\bullet_{\mathrm{ch}}(A_b, A_b)$, equipped with three
layers of algebraic structure:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \textbf{Brace dg algebra.}
 The operadic braces from the chiral endomorphism operad\/
 $\cE\!nd^{\mathrm{ch}}_{A_b}$
 \textup{(}Remark~\textup{\ref{rem:operadic-formulation}}\textup{)}
 make\/ $\mathrm{C}^\bullet_{\mathrm{ch}}(A_b, A_b)$
 a brace dg algebra. This structure is strictly richer than a
 Gerstenhaber algebra: it encodes higher homotopies that are
 invisible on cohomology. The pair
 $(\mathrm{C}^\bullet_{\mathrm{ch}}(A_b, A_b),\; A_b)$
 is the terminal local chiral open/closed pair over~$A_b$
 \textup{(}Volume~I,
 Theorem~\textup{\ref*{V1-thm:thqg-swiss-cheese}}\textup{)},
 a chain-level universal property that does not descend to
 cohomology.

\item \textbf{Shifted $L_\infty$-algebra.}
 The shifted complex\/
 $\mathrm{C}^\bullet_{\mathrm{ch}}(A_b, A_b)[1]$
 carries a filtered\/ $L_\infty$-structure whose binary bracket
 is the Gerstenhaber bracket and whose higher brackets encode
 Massey-type obstructions. The Maurer--Cartan space
 \[
 \MC\bigl(\mathrm{C}^\bullet_{\mathrm{ch}}(A_b, A_b)[1]
 \otimes \mathfrak{m}\bigr)
 \]
 \textup{(}for any pronilpotent dg algebra~$\mathfrak{m}$\textup{)}
 classifies deformations of the\/ $\SCchtop$-algebra structure
 on~$A_b$ over the base\/ $\mathfrak{m}$, up to gauge equivalence.
 On cohomology, only the \emph{formal neighborhood} of the
 trivial deformation survives\textup{;} the chain-level moduli
 space is the full derived deformation space.

\item \textbf{Maurer--Cartan element.}
 The universal MC element\/
 $\Theta_\cA \in \MC(\gAmod)$
 \textup{(}Volume~I,
 Theorem~\textup{\ref*{V1-thm:mc2-bar-intrinsic}}\textup{)}
 lives in the completed product\/
 $\prod_{r \ge 2}
 \mathrm{C}^\bullet_{\mathrm{ch}}(A_b, A_b)_{r,\bullet}$,
 satisfying the master equation
 $\delta \Theta + \tfrac{1}{2}[\Theta, \Theta] = 0$
 at the chain level. The shadow obstruction tower projections
 $\kappa$~\textup{(}degree~$2$\textup{)},
 $\mathfrak{C}$~\textup{(}degree~$3$\textup{)},
 $\mathfrak{Q}$~\textup{(}degree~$4$\textup{)}
 are chain-level representatives\textup{;} on cohomology they
 project to elements of\/ $\cZ^1$, $\cZ^1$, $\cZ^2$.
 The all-degree master equation
 $\nabla_{H}(\mathrm{Sh}_r) + \mathfrak{o}^{(r)} = 0$
 holds at the chain level and descends to the Gerstenhaber Jacobi
 identity on cohomology
 \textup{(}Volume~I, Appendix~\textup{\ref*{V1-app:nonlinear-modular-shadows}}\textup{)}.
\end{enumerate}
\end{theorem}

\begin{proof}
(i) The brace dg algebra structure is
Volume~I, Theorem~\ref*{V1-thm:thqg-brace-dg-algebra}.
The terminality of the pair
$(\mathrm{C}^\bullet_{\mathrm{ch}}(A_b, A_b),\; A_b)$
is Volume~I, Theorem~\ref*{V1-thm:thqg-swiss-cheese}: the
universal property is stated and proved at the chain level,
before any passage to cohomology.

(ii) The $L_\infty$-structure on
$\mathrm{C}^\bullet_{\mathrm{ch}}(A_b, A_b)[1]$ is the
homotopy convolution algebra associated to the bar cooperad
of~$\SCchtop$ and the endomorphism operad of~$A_b$. By the
Kontsevich--Soibelman deformation theorem
\cite[Theorem~3.1]{KS09}, Maurer--Cartan elements in
$\mathrm{C}^\bullet_{\mathrm{ch}}(A_b, A_b)[1] \otimes
\mathfrak{m}$ classify deformations of the
$\SCchtop$-algebra structure, up to gauge equivalence given
by the higher brackets. The classification at the chain level
is strictly finer than the cohomological one: a gauge
equivalence at the chain level may project to a nontrivial
automorphism on cohomology, and conversely, two non-gauge-equivalent
chain-level deformations may become equivalent on cohomology.

(iii) The MC element $\Theta_\cA$ is constructed at the chain
level in the completed convolution algebra
(Volume~I, Theorem~\ref*{V1-thm:mc2-bar-intrinsic}).
The master equation
$\delta \Theta + \tfrac{1}{2}[\Theta, \Theta] = 0$ is an
identity of chains; projection to cohomology yields the
degree-graded equations on the shadow invariants.
\end{proof}

\begin{corollary}[Gerstenhaber algebra on cohomology;
\ClaimStatusProvedHere]
\label{cor:gerstenhaber-on-cohomology}
\index{Gerstenhaber algebra!as cohomological shadow}
The cohomology\/
$\cZ^{\mathrm{der}}_{\mathrm{ch}}(\cC)
= H^*(\mathrm{C}^\bullet_{\mathrm{ch}}(A_b, A_b), \delta)$
is a Gerstenhaber algebra: the brace dg algebra structure of
Theorem~\textup{\ref{thm:chain-level-bulk}(i)} descends to a
cup product and a Lie bracket of degree~$-1$ on cohomology.
On the chirally Koszul locus\/
\textup{(}Volume~I, Theorem~H\textup{)},
$\cZ^{\mathrm{der}}_{\mathrm{ch}}(\cC)$ is concentrated in
degrees $\{0, 1, 2\}$, giving a three-term Gerstenhaber algebra
$\cZ^0 \oplus \cZ^1 \oplus \cZ^2$
where\/~$\cZ^0$ is the center of\/ $A_b$,
$\cZ^1$ carries the Lie bracket of bulk derivations, and\/
$\cZ^2$ carries obstructions dual to\/~$\cZ^0$ by Koszul duality.
\end{corollary}

\begin{proof}
The braces on $\mathrm{C}^\bullet_{\mathrm{ch}}$ make it a
homotopy Gerstenhaber algebra; the Gerstenhaber bracket
$[f,g] = f\{g\} - (-1)^{(\deg f - 1)(\deg g - 1)} g\{f\}$
descends to cohomology because $\delta$ is a derivation of the
brace operations
(Theorem~\ref*{V1-thm:thqg-brace-dg-algebra}).
The polynomial growth bound $\ChirHoch^n(\cA) = 0$ for
$n \notin \{0,1,2\}$ on the Koszul locus is Theorem~H of
Volume~I, at generic level; the critical level $k = -h^\vee$
is excluded because $\dim \ChirHoch^0$ can be infinite there
(see Vol~I Remark~\ref*{rem:critical-level-lie-vs-chirhoch}).
\end{proof}

\begin{remark}[What is lost by passing to cohomology]
\label{rem:chain-vs-cohomology}
\index{chain-level vs.\ cohomological structure}
Three structures on\/
$\mathrm{C}^\bullet_{\mathrm{ch}}(A_b, A_b)$ do not survive
the passage to\/ $\cZ^{\mathrm{der}}_{\mathrm{ch}}(\cC)$:
\begin{enumerate}[label=\textup{(\alph*)}]
\item \emph{The terminal universal property.}
 The pair $(\mathrm{C}^\bullet_{\mathrm{ch}}, A_b)$ is the
 terminal local chiral open/closed pair. No analogous
 universal property holds for the cohomological pair
 $(\cZ^{\mathrm{der}}_{\mathrm{ch}}, H^*(A_b))$: the
 terminality requires chain-level maps and chain-level
 homotopies.

\item \emph{Higher obstructions.}
 The full $L_\infty$-structure on
 $\mathrm{C}^\bullet_{\mathrm{ch}}[1]$ encodes higher-order
 obstructions to deformations
 \textup{(}Massey products, $A_\infty$ and $L_\infty$ transfer
 data\textup{)} that collapse to zero on cohomology. Two
 deformations that are obstructed at the chain level may appear
 unobstructed on cohomology; conversely, a cohomological
 deformation class may fail to lift to the chain level.

\item \emph{The derived moduli space.}
 The Maurer--Cartan space
 $\MC(\mathrm{C}^\bullet_{\mathrm{ch}}[1] \otimes \mathfrak{m})$
 is a derived stack whose tangent complex at the trivial MC
 element recovers the cohomological deformation theory. Away
 from the trivial element, the derived moduli space carries
 information
 \textup{(}automorphisms, higher isotopies, wall-crossing
 phenomena\textup{)} that is invisible on the formal
 neighborhood detected by the Gerstenhaber bracket on
 $\cZ^{\mathrm{der}}_{\mathrm{ch}}$.
\end{enumerate}
The organizing principle is Kapranov's: the chain-level object
is the primary mathematical datum; its cohomology is a
shadow that retains the leading-order information.
\end{remark}

\begin{principle}[Modular trace principle]
\label{princ:modular-trace}
\index{modular trace principle|textbf}
\index{modularity!as trace on open sector|textbf}
The modular structure of an open/closed chiral homotopy theory
on $(X,D,\tau)$ is not an additional axiom on the bulk algebra:
it is not a property that some theories have and others lack, but
a consequence of the open-sector structure that is present whenever
the open sector carries a trace.
It is determined entirely by three ingredients:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Cyclic structure}: a nondegenerate trace on the
 local endomorphism algebras of the open sector,
 \[
 \operatorname{Tr}_b \colon
 \RHom_{\cC(J)}(b,b) \;\longrightarrow\; k[-d],
 \]
 inducing a Calabi--Yau structure on $\cC(J)$.

\item \emph{Annulus identification}: the factorization homology
 of the open sector along a boundary circle $S^1_p$ is the
 Hochschild chain complex,
 \[
 \int_{S^1_p} \cC \;\simeq\; \HH_\bullet(\cC),
 \]
 which is the first \emph{traced} shadow of the open sector.
 The proof uses the Ayala--Francis $\otimes$-excision axiom
 \cite{AF15}: the standard Weiss cover $S^1_p = U_+ \cup U_-$
 \textup{(}two intervals whose intersection is the disjoint
 union of two intervals\textup{)} decomposes factorization
 homology as
 $\int_{S^1_p} \cC \simeq
 \int_{U_+} \cC \otimes^{\mathbf{L}}_{\int_{U_+ \cap U_-} \cC}
 \int_{U_-} \cC$;
 evaluating on intervals via local constancy gives
 $A_b \otimes^{\mathbf{L}}_{A_b^e} A_b = \HH_\bullet(A_b)$,
 the two-sided bar complex that computes Hochschild homology
 \textup{(}Volume~I,
 Theorem~\ref*{V1-thm:thqg-annulus-trace}\textup{)}.

\item \emph{Clutching}: the codimension-one nodal degeneration
 strata of the log-Fulton--MacPherson compactification implement
 sewing by composing the clutching maps
 $\cC(\Sigma_{g_1,n_1}) \otimes \cC(\Sigma_{g_2,n_2})
 \to \cC(\Sigma_{g,n})$
 along shared boundary circles. The modular Maurer--Cartan
 equation
 \begin{equation}\label{eq:modular-mc-trace}
 d\Theta_\cC + \tfrac{1}{2}[\Theta_\cC,\Theta_\cC]
 + \Delta_{\mathrm{clutch}}(\Theta_\cC) = 0
 \end{equation}
 is the boundary-of-boundary-is-zero identity on the
 compactified bordered moduli spaces.
\end{enumerate}
The four classical modular adjectives organize as successive
shadows of this structure:
\begin{center}
\begin{tabular}{lll}
\textbf{Adjective} & \textbf{Open-sector origin}
 & \textbf{Closed-sector shadow} \\
\midrule
Cyclic & trace on $\End(b)$
 & invariant pairing on $\cA$ \\
Traced & Hochschild chains $\HH_\bullet(\cC)$
 & open-to-closed map \\
Ribbon & trace compatible with framing/Virasoro
 & ribbon twist \\
Modular & traces survive all clutching degenerations
 & genus-$g$ amplitudes \\
\end{tabular}
\end{center}
In particular, the genus-one curvature $\kappa(\cA)$ is the image
of the annular trace $\operatorname{Tr}_b$ under the clutching map
that degenerates an annulus to a node; and the full shadow
obstruction tower of Volume~I is the genus- and degree-graded
expansion of the universal twisting element~$\Theta_\cC$.
The entire genus tower $\{F_g\}_{g \geq 0}$ is thus
\emph{determined} by the open-sector trace and the clutching
combinatorics of the compactified moduli space; modularity is not
an axiom placed on the closed algebra but a consequence of the open
sector carrying a trace.
\end{principle}

\begin{remark}[Proved status of the modular trace principle]
\label{rem:modular-trace-proved-status}
The modular trace principle is not a conjecture: it is a
consequence of three proved theorems.
\begin{enumerate}[label=\textup{(\roman*)}]
\item The bar-intrinsic MC element
 $\Theta_\cA := D_\cA - d_0$
 satisfies the clutching factorization at \emph{all} genera
 (Volume~I, Theorem~\ref*{V1-thm:mc2-bar-intrinsic}(iii)):
 separating degenerations give
 $\xi_{\mathrm{sep}}^*(\Theta^{(g)})
 = \sum_{g_1+g_2=g} \Theta^{(g_1)} \star \Theta^{(g_2)}$
 and non-separating degenerations give
 $\xi_{\mathrm{ns}}^*(\Theta^{(g+1)})
 = \Delta_{\mathrm{ns}}(\Theta^{(g)})$.
\item The open/closed MC equation
 (Volume~I, Theorem~\ref*{V1-thm:thqg-oc-mc-equation})
 packages closed-sector, open-sector, mixed, and clutching
 data into a single identity at all genera.
\item The projection principle
 (Volume~I, Theorem~\ref*{V1-thm:thqg-oc-projection})
 shows that both the closed-sector genus tower and the
 open-sector module structures are projections of the
 same element $\Theta^{\mathrm{oc}}_\cA$.
\end{enumerate}
Consequently, the consistency between the closed-sector
computation (Theorem~D of Volume~I: $F_1 = \kappa/24$
unconditionally, with the higher-genus scalar continuation
$F_g = \kappa \cdot \lambda_g$ conditional) and the
open-sector clutching is enforced by the MC equation itself.
What remains is the \emph{independent open-sector verification}:
computing genus-$g$ amplitudes from iterated clutching on
$\cC_{\mathrm{op}}$ without invoking the bar-intrinsic
construction, and confirming agreement. This is a consistency
check of the proved framework, not a gap in it.
\end{remark}

\subsubsection{The line category: MC couplings vs.\ Koszul
 modules}
\label{subsec:line-mc-vs-koszul}
% -------------------------------------------------------------------

The open sector of a 3d HT theory supports line operators, extended
operators wrapping the topological direction~$\R_t$. Algebraically,
a line operator is a module over the boundary algebra~$\cA$, and the
category of such modules is the third vertex of the
bulk/boundary/line triangle
(Theorem~\ref{thm:boundary-linear-bulk-boundary}). The key
distinction is between two presentations of this category: the
intrinsic Maurer--Cartan presentation (always well-defined) and the
Koszul presentation (available only on the chirally Koszul locus).

\begin{definition}[MC coupling category]
\label{def:mc-coupling-category}
\index{MC coupling category|textbf}
\index{line operators!MC coupling presentation|textbf}
For any $A_\infty$-chiral algebra~$\cA$, the \emph{MC coupling
category} $\operatorname{Line}(\cA)$ has:
\begin{enumerate}[label=\textup{(\alph*)}]
\item \emph{Objects}: pairs $(V, \mu)$ where $V$ is a dg vector
 space and $\mu \in \End(V) \widehat\otimes \cA$ of degree~$1$
 satisfies the $A_\infty$ Maurer--Cartan equation
 $\sum_{k \ge 1} m_k(\mu, \ldots, \mu) = 0$.

\item \emph{Morphisms}:
 $\Hom((V,\mu), (W,\nu))
 = \Hom(V, W) \widehat\otimes \cA$
 with twisted differential
 $d_{\mu,\nu}(f)
 = \sum_{r,s \ge 0} m_{r+s+1}(\nu^{\otimes r}, f,
 \mu^{\otimes s})$.
\end{enumerate}
This category is \emph{always} well-defined: it requires only the
$A_\infty$ structure on~$\cA$ and convergence of the completed
tensor product, not Koszulness or any other structural hypothesis.
It is the intrinsic primitive object.
\end{definition}

\begin{theorem}[Koszul comparison; \ClaimStatusProvedHere]
\label{thm:koszul-comparison-mc}
\index{Koszul comparison!MC coupling|textbf}
If $\cA$ is chirally Koszul
\textup{(}bar cohomology concentrated in bar degree~$1$;
any of the $10$ unconditional equivalent conditions of
Volume~I, Theorem~\textup{\ref*{V1-thm:koszul-equivalences-meta}}\textup{)},
then the Koszul dual $\cA^!$ is well-defined and the
comparison functor
\begin{align*}
 \Phi \colon \operatorname{Line}(\cA)
 &\;\longrightarrow\; \Perf(\cA^!) \\
 (V, \mu) &\;\longmapsto\;
 (V,\, \rho = (\mathrm{id}_V \otimes \pi)(\mu))
\end{align*}
is a quasi-equivalence, where
$\pi \colon \cA \twoheadrightarrow \cA^!$ is the bar-cohomology
projection.
\end{theorem}

\begin{proof}
On the Koszul locus, bar-cobar inversion
(Theorem~B) gives a quasi-isomorphism
$\Omega(\barB(\cA)) \xrightarrow{\sim} \cA$. The Koszul dual
$\cA^! = H^\bullet(\barB(\cA))$ is concentrated in bar degree~$1$
by the Koszul hypothesis, and the universal
twisting morphism $\tau \in \cA^! \otimes \cA$ mediates between
the MC coupling $(V, \mu)$ and the $\cA^!$-module
$(V, \rho)$ via $\mu = (\rho \otimes \mathrm{id})(\tau)$: the
MC element $\mu$ assembles into an $\cA^!$-module structure because
the bar cohomology concentration forces the MC series to terminate
at linear order.
The quasi-equivalence follows from the bar-cobar Quillen
equivalence applied to modules: the functor $\Phi$ is the derived
restriction-of-scalars along the quasi-isomorphism
$\Omega(\barB(\cA)) \xrightarrow{\sim} \cA$, and its quasi-inverse
is derived extension-of-scalars.
\end{proof}

\begin{remark}[Off the Koszul locus]
\label{rem:off-koszul-locus}
For algebras whose Koszulness is open or fails
\textup{(}e.g.\ simple quotients $L_k(\fg)$ at admissible levels,
whose bar-Ext vs.\ ordinary-Ext gap is unresolved\textup{)},
the MC coupling category $\operatorname{Line}(\cA)$ is still
well-defined and is still the correct primitive. What fails
\textup{(}or may fail\textup{)} is the algebraic presentation
as $\Perf(\cA^!)$: the Koszul dual may not exist, or it may
exist but the comparison functor may not be an equivalence.
The Koszulness characterization programme
\textup{(}$10{+}1{+}1$ conditions\textup{)} classifies precisely
when the algebraic presentation works.
\end{remark}

\subsection{From local collars to global factorization}
\label{subsec:local-to-global-factorization}
% -------------------------------------------------------------------

\begin{construction}[Local collar categories]
\label{constr:local-collar-categories}
\index{collar category|textbf}
Let $(X, D, \tau)$ be a tangential log curve and
$I_p \cong \R$ the boundary interval at a puncture~$p$.
For a small subinterval $J \subset I_p$, the \emph{collar category}
$\cC_J := \operatorname{Line}(\cA_J)$
is the MC coupling category of the restriction of the bulk algebra
to the formal neighborhood of~$J$.
\end{construction}

\begin{proposition}[MC homotopy invariance; \ClaimStatusProvedHere]
\label{prop:mc-homotopy-invariance}
\index{MC coupling category!homotopy invariance}
Let $f \colon \cA \xrightarrow{\sim} \cA'$ be a quasi-isomorphism
of $A_\infty$ algebras. Then the induced functor
\[
 f_* \colon \operatorname{Line}(\cA)
 \;\longrightarrow\;
 \operatorname{Line}(\cA')
\]
is a quasi-equivalence of dg-categories.
\end{proposition}

\begin{proof}
Write $\operatorname{Line}(\cA)$ for the dg-category whose objects
are Maurer--Cartan elements $\mu \in \MC(\cA \otimes \End(V))$ and
whose morphism complexes are the twisted Hom spaces
$(\Hom(V, W),\, d_{\mu,\nu})$.

\emph{Essential surjectivity on $H^0$.}
A quasi-isomorphism $f \colon \cA \xrightarrow{\sim} \cA'$
induces a bijection on gauge-equivalence classes of
Maurer--Cartan elements
\cite[Theorem~4.6]{KS09}.
Concretely, $f$ extends to an $A_\infty$ functor
$f_* \colon \operatorname{Line}(\cA) \to \operatorname{Line}(\cA')$
by $f_*(\mu) = \sum_{n \ge 1} (f_n \otimes \mathrm{id}_{
\End(V)})(\mu^{\otimes n})$.
The MC bijection gives essential surjectivity on~$H^0$.

\emph{Quasi-isomorphism on morphism complexes.}
For MC elements $\mu, \nu$ in $\cA \otimes \End(V)$
and $\cA \otimes \End(W)$ respectively, the morphism complex
$(\Hom(V, W),\, d_{\mu,\nu})$ has differential twisted by $\mu$
and~$\nu$.
The map $f$ induces a chain map
$f_* \colon (\Hom(V, W), d_{\mu,\nu})
\to (\Hom(V, W), d_{f_*\mu, f_*\nu})$.
Since $f_1$ is a quasi-isomorphism, the homological perturbation
lemma \cite{Berglund15} provides a deformation retract of the
twisted complexes: the perturbation series converges because
$\mu, \nu$ are Maurer--Cartan \textup{(}nilpotent in the
completed tensor product\textup{)}, and the resulting chain map
is a quasi-isomorphism on each $\Hom$ complex.
\end{proof}

\begin{theorem}[$E_1$-factorization on disjoint intervals;
 \ClaimStatusProvedHere]
\label{thm:e1-factorization-disjoint}
For disjoint subintervals $J_1, J_2 \subset I_p$ with
$\sup J_1 < \inf J_2$, the external tensor product
\[
 \cC_{J_1} \boxtimes \cC_{J_2}
 \;\xrightarrow{\;\sim\;}
 \cC_{J_1 \cup J_2}
\]
is an equivalence. This is the $E_1$-factorization structure of
the open color: the ordered interval composition on~$\R$ gives
the associative monoidal structure on the line category.
\end{theorem}

\begin{proof}
The interval $J_1 \cup J_2$ (with a gap between $J_1$ and $J_2$)
retracts onto $J_1 \sqcup J_2$ (disjoint union). The MC
coupling category is homotopy-invariant
(Proposition~\ref{prop:mc-homotopy-invariance}), so
$\cC_{J_1 \cup J_2} \simeq \cC_{J_1 \sqcup J_2}
= \cC_{J_1} \boxtimes \cC_{J_2}$.
\end{proof}

\begin{theorem}[Extension across collisions;
 \ClaimStatusProvedHere{} on the chirally Koszul locus]
\label{thm:collision-extension}
On the chirally Koszul locus, the meromorphic tensor product
\[
 \ell \otimes_z \ell'
 \;:=\;
 (V_\ell \otimes V_{\ell'},\;
 \mu_\ell(z) + \mu_{\ell'}(0) + r_{\ell,\ell'}(z))
\]
extends the $E_1$-factorization structure across the collision
strata of the ordered boundary configuration space, where
$r_{\ell,\ell'}(z)$ is the universal quantum correction computed
by tree-level ladder diagrams. The extension is associative:
\[
 (\ell_1 \otimes_{z_1} \ell_2) \otimes_{z_2} \ell_3
 \;\simeq\;
 \ell_1 \otimes_{z_1 + z_2} (\ell_2 \otimes_{z_2} \ell_3).
\]
\end{theorem}

\begin{proof}
The correction $r_{\ell,\ell'}(z)$ is meromorphic in~$z$ by the
one-loop exactness property of the perturbative 3d~HT theory
(Theorem~\ref{thm:one-loop-koszul}). Associativity follows from
the Yang--Baxter equation for the spectral $R$-matrix
$R(z) = 1 + r(z)/z + \cdots$, which is proved by Stokes' theorem
on $\FM_3(\C)$
(Theorem~\ref{thm:spectral_R_YBE}).
\end{proof}

\begin{theorem}[Meromorphic tensor $=$ factorization;
 \ClaimStatusProvedHere{} on the chirally Koszul locus]
\label{thm:meromorphic-factorization-equivalence}
\index{meromorphic tensor!factorization equivalence|textbf}
The meromorphic tensor presentation
$(\operatorname{Line}(\cA), \otimes_z)$
\textup{(}Theorem~\textup{\ref{thm:meromorphic-tensor-dgcat}}
in~\S\textup{\ref{sec:chiral-Hochschild-brace}}\textup{)}
and the factorization dg-cosheaf presentation
$\cC_{\mathrm{op},\partial}$ on ordered boundary configuration
spaces determine each other:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Tensor to factorization.} On each chamber of
 $\operatorname{Conf}_n^{\mathrm{ord}}(I_p)$, assign the
 external tensor product
 $\cC_J^{\mathrm{op}} \boxtimes \cdots \boxtimes
 \cC_J^{\mathrm{op}}$. On codimension-one collision faces,
 glue by the iterated meromorphic tensor product~$\otimes_z$.
 Codimension-two coherence holds because:
 \textup{(a)}~disjoint collisions commute tautologically;
 \textup{(b)}~triple collisions reduce to the associativity
 of~$\otimes_z$
 \textup{(}Theorem~\textup{\ref{thm:collision-extension}}\textup{)}.

\item \emph{Factorization to tensor.} Given a factorization
 dg-cosheaf, restrict to two-point collision faces to recover
 the binary meromorphic tensor product. Higher fusion functors
 are recovered from multi-point collision strata by
 factorization on each stratum.
\end{enumerate}
\end{theorem}

\begin{proof}
Direction (i): the cocycle condition for a constructible cosheaf
on the FM-type compactification of ordered configurations requires
consistency at all codimension-two strata. These strata are of two
types: disjoint collisions (commute tautologically) and triple
collisions with two parenthesizations. The second type is exactly
the associativity of~$\otimes_z$
(Theorem~\ref{thm:collision-extension}).
Direction (ii): the factorization axiom
$\cC(U_1 \sqcup U_2) \simeq \cC(U_1) \boxtimes \cC(U_2)$
determines the cosheaf on chambers; the restriction to
codimension-one faces gives the binary fusion functor, i.e.\
the meromorphic tensor product.
\end{proof}

\begin{conjecture}[Global Weiss descent]
\label{conj:weiss-descent}
\index{Weiss descent!boundary factorization|textbf}
The local collar categories $\{\cC_J\}_{J \subset I_p}$,
equipped with the $E_1$-factorization on disjoint intervals and the
meromorphic tensor extension across collisions, satisfy Weiss
descent on the boundary configuration space: the natural map
\[
 \cC_{\mathrm{global}}
 \;\longrightarrow\;
 \holim_{J \in \mathrm{Weiss}(I_p)} \cC_J
\]
is an equivalence. This is the categorical boundary analogue of
the Beilinson--Drinfeld factorization axiom for algebras.
\end{conjecture}

\begin{remark}[What is proved vs.\ what is conjectured]
\label{rem:local-vs-global-honest}
The local theory is a theorem:
\begin{itemize}[nosep]
\item Collar categories are well-defined
 (Definition~\ref{def:mc-coupling-category}).
\item $E_1$-factorization on disjoint intervals is proved
 (Theorem~\ref{thm:e1-factorization-disjoint}).
\item Extension across collisions is proved on the Koszul locus
 (Theorem~\ref{thm:collision-extension}).
\item The Koszul comparison $\operatorname{Line}(\cA)
 \simeq \Perf(\cA^!)$ is proved on the Koszul locus
 (Theorem~\ref{thm:koszul-comparison-mc}).
\end{itemize}
The global theory is a conjecture:
\begin{itemize}[nosep]
\item Weiss descent on the boundary Ran space
 (Conjecture~\ref{conj:weiss-descent}).
\item The categorical Ran-space formalism in the boundary setting
 (requires the Raskin--Beraldo framework adapted to
 real-oriented blowups).
\end{itemize}
The honest description: the primitive object is defined locally
as a theorem and is expected to globalize by general categorical
arguments that have not yet been carried out in the boundary
setting.
\end{remark}

\subsection{The two--colored Swiss--cheese operad}
\label{subsec:SC-operad}

For the detailed Weiss descent argument and operadic properties, see Section~\ref{sec:SC-operad-foundations}.

\begin{definition}[Colors]
\label{def:SC-colors}
Consider the set of two colors $\mathbf{Col}=\{\mathsf{ch},\mathsf{top}\}$, thought of as the
\emph{closed} (holomorphic) and \emph{open} (topological) colors respectively.
\end{definition}

\begin{definition}[Spaces of operations]
\label{def:SC-operations}
Define the topological 2--colored operad
$\mathsf{SC}^{\mathrm{ch,top}}$ as follows.
\begin{enumerate}[label=(\roman*)]
\item \textbf{Closed outputs (holomorphic):} For closed output and only closed inputs,
 set
 \[
 \mathsf{SC}^{\mathrm{ch,top}}\!\bigl( (\underbrace{\mathsf{ch},\ldots,\mathsf{ch}}_{k});\,\mathsf{ch} \bigr)
 \;:=\; \FM_k(\mathbb C),
 \]
 the Fulton--MacPherson compactification of the configuration space of $k$ points in $\mathbb C$.
 If any input is open, the space of operations with closed output is \emph{empty}:
 $\mathsf{SC}^{\mathrm{ch,top}}\bigl( (\ldots,\mathsf{top},\ldots);\,\mathsf{ch} \bigr)=\varnothing$.
 By convention, $\FM_0(\C)$ and $\FM_1(\C)$ are single points, and $E_1(0) = \emptyset$ (the empty configuration).
\item \textbf{Open outputs (topological):} For open output and a mixture of open and closed inputs, set
 \[
 \mathsf{SC}^{\mathrm{ch,top}}\!\bigl( (\underbrace{\mathsf{ch},\ldots,\mathsf{ch}}_{k}, \underbrace{\mathsf{top},\ldots,\mathsf{top}}_{m});\,\mathsf{top} \bigr)
 \;:=\; \FM_k(\mathbb C)\times E_1(m) .
 \]
 This encodes that an open operation may depend on closed inputs (holomorphic insertions)
 and on open inputs (time--ordered insertions), and its structure is the product of
 holomorphic and topological composition spaces.
\end{enumerate}
\end{definition}

\begin{remark}[Why the product structure is forced]
\label{rem:product-forced-by-locality}
The product decomposition
$\FM_k(\C) \times E_1(m)$ in
Definition~\ref{def:SC-operations}(ii) is forced by locality: a
local observable that factorizes holomorphically in~$z$ and is
locally constant in~$t$ can only depend on collision data in~$\C$
(parametrised by $\FM_k(\C)$) and time-ordering in~$\R$
(parametrised by $\Conf_k^{<}(\R) \subset E_1(m)$). No other
topological data is available. The argument has three steps.
First, the holomorphic dependence on the $z$-variables is
controlled by $\FM_k(\C)$ because the factorisation structure
along~$\C$ is holomorphic (meromorphic at collisions); the
FM compactification records precisely the collision patterns and
their blowup data. Second, the topological dependence on the
$t$-variables is controlled by $E_1(m)$ because local constancy
along~$\R$ reduces the continuous family of ordered
configurations to a discrete combinatorial datum: only the
linear ordering $t_1 < \cdots < t_m$ matters, not the metric
positions. Third, there is no mixed
configuration space interpolating between the holomorphic and
topological directions: the kernel of each $m_k$ integral
factorises as a meromorphic function of $z_{ij} = z_i - z_j$ times
a step function of $t_{ij} = t_i - t_j$, because the equations of
motion in the HT gauge decouple the $\C$- and $\R$-dependences
of the propagator (the Green's function is the Cauchy kernel
$dz/(z-w)$ in~$\C$ times the Heaviside function
$\theta(t_1 - t_2)$ in~$\R$). For physical realisations, this
decoupling is packaged by Theorem~\ref{thm:physics-bridge}.
\end{remark}

\begin{remark}[Swiss--cheese directionality]
The absence of mixed operations $(\ldots,\mathsf{top},\ldots)\to\mathsf{ch}$ ensures a
\emph{closed--output isolation}: holomorphic ("closed") outputs cannot depend on open inputs.
This directionality is the defining property of Swiss--cheese operads and reflects the physical constraint:
bulk (closed) interactions can restrict to boundaries (open), but boundary interactions
cannot produce new bulk structures.
\end{remark}

\begin{definition}[Composition]
\label{def:SC-composition}
Composition in $\mathsf{SC}^{\mathrm{ch,top}}$ is induced by the strict operadic insertion
maps of $\FM_\bullet(\mathbb C)$ and $E_1$, composed \emph{componentwise} as follows.
\begin{itemize}
 \item For closed--into--closed insertions,
 \[
 \FM_k(\mathbb C)\times\prod_{i=1}^k\FM_{\ell_i}(\mathbb C)
 \;\longrightarrow\;
 \FM_{\sum_i\ell_i}(\mathbb C),
 \]
 we use the standard operadic insertion of configurations of points.
 \item For open--into--open with $k$ closed and $m$ open inputs,
 \[
 \bigl(\FM_k(\mathbb C)\times E_1(m)\bigr)\times
 \prod_{i=1}^k\FM_{\ell_i}(\mathbb C)\times\prod_{j=1}^m E_1(n_j)
 \;\longrightarrow\;
 \FM_{\sum_i\ell_i}(\mathbb C)\times E_1(\sum_j n_j) ,
 \]
 taking the product of insertion maps on $\FM_\bullet(\mathbb C)$ and on $E_1$.
 \item For closed--into--open, we allow insertion of closed operations in the closed $\FM$--slots
 inside open operations, but never the reverse (no open--into--closed).
\end{itemize}
The symmetric groups act by permuting labeled inputs of each color, and units are given by
the degree--1 elements in $\FM_1(\mathbb C)$ and~$E_1(1)$.
\end{definition}

\begin{proposition}[Well-definedness; \ClaimStatusProvedHere]
\label{prop:SC-well-defined}
The data of Definitions~\ref{def:SC-colors}--\ref{def:SC-composition}
endows $\mathsf{SC}^{\mathrm{ch,top}}$ with the structure of a
strict two--colored topological operad.
\end{proposition}

\begin{proof}
Associativity, unitality, and symmetric--group compatibility follow from the corresponding
properties of the operads $\FM_\bullet(\mathbb C)$ and~$E_1$, together with the product structure of the
mixed composition. The "no open-to-closed" restriction is
stable under composition by construction.
\end{proof}

\begin{remark}[Homotopy-Koszulity and two-color Koszul duality]
\label{rem:two-color-koszul-duality-operadic}
\index{two-color Koszul duality!operadic input}
The homotopy-Koszulity of $\SCchtop$
(Theorem~\ref{thm:homotopy-Koszul}) is the operadic input to
the master two-color Koszul duality theorem
(Theorem~\ref{thm:two-color-master}). Together with Koszul
self-duality (\S\ref{subsec:dg-yangian-operadic}), it ensures
that on the chirally Koszul locus the open-colour Koszul dual
$\cA^!_{\mathrm{line}}$ is an ${\SCchtop}^{!}$-algebra (i.e.\ a $(\mathrm{Lie},\mathrm{Ass})$-algebra)
\textup{(}Theorem~\textup{\ref{thm:dual-sc-algebra}}\textup{)}
and hence a dg-shifted Yangian
\textup{(}Theorem~\textup{\ref{thm:yangian-recognition}}\textup{)}.

The associated graded splitting
$\mathrm{gr}\;\SCchtop \cong \mathsf{P}_{\mathrm{ch}}
\amalg E_1$
(Proposition~\ref{prop:gr-chiral}) shows that the two colors
decouple at the quadratic level; the full $\SCchtop$-algebra
is a \emph{filtered deformation} of this decoupled structure.
The holomorphic weight filtration is the organizing filtration
that separates the ``easy'' (decoupled) data from the
``interacting'' data: the differentials in the associated
spectral sequence encode successively higher-order corrections,
starting from the classical $r$-matrix at $E_1$ and converging
to the full quantum $R(z)$.
\end{remark}

\begin{construction}[$E_n$ formality hierarchy as MC truncation]
\label{constr:vol2-en-formality}
The $E_n$ formality hierarchy corresponds to truncation of the MC obstruction tower, as in the $E_n$-formality/MC-truncation proposition of Volume~I:
\begin{enumerate}[label=\textup{(\roman*)}]
\item $E_\infty$-formality: all higher brackets $\ell_n^{(g)} = 0$ for $n \geq 3$ in $\gAmod$.
\item $E_2$-formality: genus-$0$ convolution algebra $\mathfrak{g}^{(0)}_\cA$ is formal (chiral Koszulness).
\item $E_1$-formality: ordered convolution algebra $\mathfrak{g}^{E_1}_\cA$ carries a formal $A_\infty$-model.
\end{enumerate}
The Swiss-cheese operad $\mathrm{SC}^{\mathrm{ch,top}}$ is homotopy-Koszul (Theorem~\ref{thm:homotopy-Koszul}), placing the present volume at the $E_1$ level.

The mechanism is as follows. The MC obstruction tower for the
deformation complex $\gAmod$ has obstructions living on
configuration spaces $\FM_k$ at each degree~$k$. Truncating the tower
at degree~$n$ retains only configurations of $\le n$ points, which is
the defining data of an $E_n$-algebra. At level~(i), $E_\infty$
means all higher $L_\infty$ brackets vanish and the MC equation
linearizes; at level~(ii), $E_2$ means only genus-$0$
(planar/commutative) obstructions survive, corresponding to the
commutativity of the $E_2$-operad; at level~(iii), $E_1$ means the
full $A_\infty$ obstruction tower is retained, with obstructions on
ordered configurations governing the associative (non-commutative)
sector. The $E_1$-level data is classified by the bar complex:
its differential is the MC element of the ordered convolution algebra.
\end{construction}

\label{rem:three-koszul-dualities}
Three distinct Koszul duality frameworks in the chiral setting are unified by the two-coloured operad $\SCchtop$:

\end{remark}

\subsection{Algebras and the Boardman--Vogt resolution}
\label{subsec:WSC-algebras}

\begin{definition}[Algebras over $\mathsf{SC}^{\mathrm{ch,top}}$]
\label{def:SC-algebra}
Let $(\mathcal C,\otimes,\mathbf 1)$ be a symmetric monoidal presentable stable dg--category
(e.g.\ chain complexes over a field $k$ of characteristic zero). An algebra over the operad
$C_\ast(\mathsf{SC}^{\mathrm{ch,top}})$ in $\mathcal C$ is a pair of objects
$(A_{\mathsf{ch}},A_{\mathsf{top}})\in\mathcal C^2$
together with structure maps
\[
 C_\ast(\FM_k(\mathbb C))\otimes A_{\mathsf{ch}}^{\otimes k}\longrightarrow A_{\mathsf{ch}} ,
 \qquad
 C_\ast(\FM_k(\mathbb C)\times E_1(m))\otimes
 A_{\mathsf{ch}}^{\otimes k}\otimes A_{\mathsf{top}}^{\otimes m}\longrightarrow A_{\mathsf{top}} ,
\]
satisfying the operadic identities.
\end{definition}

\begin{definition}[Boardman--Vogt resolution]
\label{def:WSC}
Let $W(\mathsf{SC}^{\mathrm{ch,top}})$ denote the Boardman--Vogt resolution of the
topological operad $\mathsf{SC}^{\mathrm{ch,top}}$.
Its object $W(\mathsf{SC}^{\mathrm{ch,top}})$ is a cofibrant replacement in the model category of two-colored topological operads \cite{BM03,CM14}, encoding
the data of \emph{homotopy--coherent} operations and relations.
An algebra over $C_\ast\!\bigl(W(\mathsf{SC}^{\mathrm{ch,top}})\bigr)$ is called a
\emph{homotopy holomorphic--topological algebra}.
\end{definition}

\begin{remark}[Homotopy--coherent composition]
\label{rmk:W-homotopy}
The $W$--resolution introduces \emph{edge lengths} on the trees that index compositions,
tracking higher homotopies between different composition orders.
These encode $A_\infty$--type structures in the topological color
and chiral associativity up to homotopy in the closed color.
\end{remark}

\subsection{HT prefactorization algebras and Swiss--cheese algebras}
\label{subsec:prefact-vs-operad}

\begin{definition}[HT prefactorization algebra]
\label{def:HT-prefactorization}
A \emph{holomorphic--topological prefactorization algebra} on $\mathbb C\times\mathbb R$ with values
in $\mathcal C$ assigns to every sufficiently nice open set $U\subset\mathbb C\times\mathbb R$ an
object $\mathsf{Obs}(U)\in\mathcal C$ together with
\begin{enumerate}[label=(\alph*)]
\item \emph{multiplicative structure} for disjoint unions
 $U=\bigsqcup_i U_i$:
 \(
 \bigotimes_i \mathsf{Obs}(U_i)\to \mathsf{Obs}(U),
 \)
\item \emph{restriction structure} for inclusions $U\subset V$:
 \(
 \mathsf{Obs}(U)\to \mathsf{Obs}(V),
 \)
\end{enumerate}
which is (i) holomorphic in the $\mathbb C$--direction, with meromorphic singularities (OPE poles) arising from the collision of operator insertions,
(ii) locally constant in the $\mathbb R$--direction (topological factorization),
and (iii) compatible with iterated decomposition by rectangles $D\times I$.
\end{definition}

\begin{theorem}[Swiss--cheese recognition; \ClaimStatusProvedHere]
\label{thm:recognition-foundations}
\label{thm:recognition}
Let $\mathcal C$ be a symmetric monoidal presentable stable
dg--category over a field $k$ of characteristic zero.
The following data are equivalent:
\begin{enumerate}[label=(\roman*)]
\item A homotopy holomorphic--topological algebra over
 $C_\ast\!\bigl(W(\mathsf{SC}^{\mathrm{ch,top}})\bigr)$ in
 $\mathcal C$;
\item A holomorphic--topological prefactorization algebra
 $\mathsf{Obs}$ on~$\mathbb C\times\mathbb R$ with values in
 $\mathcal C$, satisfying:
 \begin{enumerate}[label=(\alph*)]
 \item \emph{Holomorphic factorization along $\mathbb C$:} the
 assignment $D\mapsto\mathsf{Obs}(D\times I)$ for discs
 $D\subset\mathbb C$ forms a prefactorization algebra
 on~$\mathbb C$ governed by
 $C_\ast(\FM_\bullet(\mathbb C))$, with meromorphic OPE
 singularities along collision diagonals;
 \item \emph{Local constancy along $\mathbb R$:} the assignment
 $I\mapsto\mathsf{Obs}(D\times I)$ for intervals
 $I\subset\mathbb R$ is locally constant (restriction along
 inclusions of intervals is a quasi-isomorphism), yielding an
 $E_1$--algebra structure;
 \item \emph{Color compatibility:} the closed--color
 (holomorphic) factorization acts on the open--color
 (topological) factorization via the product structure
 $\FM_k(\mathbb C)\times E_1(m)$, but no open--to--closed
 operations exist ($\mathsf{SC}^{\mathrm{ch,top}}(\ldots,
 \mathsf{top},\ldots;\mathsf{ch})=\varnothing$).
 \end{enumerate}
\end{enumerate}
\end{theorem}

\begin{proof}
The proof assembles three independently established recognition
results via the product decomposition of the operation spaces of
$\mathsf{SC}^{\mathrm{ch,top}}$. The Weiss cosheaf descent lemma
(Lemma~\ref{lem:product-weiss-descent}), which shows that HT
prefactorization algebras on rectangles satisfy descent and hence
determine operadic algebra structures.

\medskip\noindent
\textbf{Step~1: Closed-color recognition (holomorphic).}
Costello--Gwilliam~\cite{CG17} prove that the $(\infty,1)$-category
of holomorphic prefactorization algebras on~$\mathbb C$ with
meromorphic OPE singularities is equivalent to the category of
algebras over the chains
$C_\ast(\FM_\bullet(\mathbb C); \mathscr{L}_{\log})$, where
$\mathscr{L}_{\log}$ is the logarithmic local system encoding OPE
poles. Since the closed-color component of
$\mathsf{SC}^{\mathrm{ch,top}}$ is exactly $\FM_k(\mathbb C)$
(Definition~\ref{def:SC-operations}(i)), condition~(ii)(a) is
equivalent to giving the closed-color component of a
$C_\ast(W(\mathsf{SC}^{\mathrm{ch,top}}))$-algebra structure on
$A_{\mathsf{ch}}$.

\medskip\noindent
\textbf{Step~2: Open-color recognition (topological).}
The Ayala--Francis recognition theorem~\cite{AF15} identifies
locally constant factorization algebras on~$\mathbb R$ satisfying
Weiss cosheaf descent with
$E_1$-algebras (associative algebras up to coherent homotopy).
Condition~(ii)(b) is therefore equivalent to giving the open-color
component of the algebra structure on $A_{\mathsf{top}}$.
The descent condition is verified by
Lemma~\ref{lem:product-weiss-descent}, Step~3: locally constant
prefactorization algebras on~$\R$ automatically satisfy Weiss descent
because the \v{C}ech nerve of any Weiss cover by intervals is
contractible and all restriction maps are quasi-isomorphisms.

\medskip\noindent
\textbf{Step~3: Two-colored assembly via the product decomposition.}
The operation spaces of
$\mathsf{SC}^{\mathrm{ch,top}}$ are \emph{products}:
\[
\mathsf{SC}^{\mathrm{ch,top}}
\bigl((\mathsf{ch}^k,\mathsf{top}^m);\mathsf{top}\bigr)
= \FM_k(\mathbb C)\times E_1(m).
\]
The no-open-to-closed rule
($\mathsf{SC}^{\mathrm{ch,top}}(\ldots,\mathsf{top},\ldots;
\mathsf{ch})=\varnothing$) implies that the closed color is an
\emph{operadic retract}: restricting
$\mathsf{SC}^{\mathrm{ch,top}}$ to the closed color recovers
$\FM_\bullet(\mathbb C)$ as a sub-operad, and this restriction
admits a section (the inclusion of closed-only operations). The
open-color operations see the closed color only through the product
factor $\FM_k(\mathbb C)$, never the reverse.

In the model category of $\Sigma$-cofibrant two-colored topological
operads~\cite{BM03,CM14}, the Boardman--Vogt resolution
$W(\mathsf{SC}^{\mathrm{ch,top}})$ is cofibrant. The product
decomposition of the operation spaces is preserved by the
$W$-construction because the cubical edge-length parameters act
independently on the $\FM_k(\mathbb C)$ and $E_1(m)$ factors: a
$W$-tree with internal edges decorated by $s_e \in [0,1]$ decomposes
into its holomorphic ($\FM$-decorated) and topological
($E_1$-decorated) parts, with the same edge parameters governing
both. This independence holds because the operation spaces are
\emph{set-theoretic products} $\FM_k(\mathbb C) \times E_1(m)$, and
the Boardman--Vogt $W$-construction acts on these products factor by
factor: an edge parameter $s_e$ governs the interpolation between
sequential and simultaneous composition, and since compositions in the
$\FM$ factor (holomorphic insertions) commute with compositions in
the $E_1$ factor (interval concatenations); the two types never
interact by the no-open-to-closed rule, so the $W$-resolution of the
product is the product of the $W$-resolutions. In the language
of~\cite{BM03}, the product decomposition is a distributive law
between the two single-colored sub-operads, and the $W$-construction
preserves distributive law decompositions.

Therefore, giving a
$C_\ast(W(\mathsf{SC}^{\mathrm{ch,top}}))$-algebra
$(A_{\mathsf{ch}}, A_{\mathsf{top}})$ is equivalent to giving:
\begin{itemize}
\item a $C_\ast(W(\FM_\bullet(\mathbb C)))$-algebra structure on
 $A_{\mathsf{ch}}$ (the closed-color data, governing holomorphic
 factorization);
\item a $C_\ast(W(E_1))$-algebra structure on
 $A_{\mathsf{top}}$ (the open-color data, governing topological
 factorization);
\item a compatible action of $A_{\mathsf{ch}}$ on
 $A_{\mathsf{top}}$ via the mixed operations
 $C_\ast(\FM_k(\mathbb C) \times E_1(m))$, coherent with the
 Boardman--Vogt homotopies.
\end{itemize}
Steps~1 and~2 identify the first two items with
conditions~(ii)(a) and~(ii)(b). Condition~(ii)(c) is exactly the
third item: the closed-color holomorphic structure acts on the
open-color topological structure via the product operation spaces,
with no open-to-closed flow. The no-open-to-closed rule ensures
that the closed-color recognition (Step~1) is not modified by the
presence of the open color, so the two single-colored recognitions
compose without obstruction.

\medskip\noindent
\textbf{Step~4: Weiss descent for the mixed operations.}
The remaining content is the converse direction: given an HT
prefactorization algebra $\Obs$ satisfying (ii)(a)--(c), we must
produce the mixed operadic structure maps
$C_\ast(\FM_k(\C) \times E_1(m)) \otimes A_{\mathsf{ch}}^{\otimes k}
\otimes A_{\mathsf{top}}^{\otimes m} \to A_{\mathsf{top}}$.
By Lemma~\ref{lem:product-weiss-descent}, $\Obs$ satisfies Weiss
cosheaf descent on the basis of rectangles
$\{D \times I\}$. The product Weiss descent argument
(Lemma~\ref{lem:product-weiss-descent}, Step~4) shows that the
\v{C}ech nerve of any Weiss cover of $D \times I$ by sub-rectangles
computes $\Obs(D \times I)$ via an iterated hocolimit that factors
through the holomorphic and topological directions independently.
The mixed structure maps are then read off from the factorization
structure on rectangles $D_1 \sqcup \cdots \sqcup D_k \times
I_1 \sqcup \cdots \sqcup I_m \hookrightarrow D \times I$:
the holomorphic dependence on disk positions is governed by
$C_\ast(\FM_k(\C))$ (by Step~1 and condition~(ii)(a)),
the topological dependence on interval positions is governed by
$C_\ast(E_1(m))$ (by Step~2 and condition~(ii)(b)),
and the K\"unneth quasi-isomorphism
$C_\ast(\FM_k(\C)) \otimes C_\ast(E_1(m)) \simeq
C_\ast(\FM_k(\C) \times E_1(m))$
(valid in characteristic zero with finite-type homology)
assembles these into the required mixed operation maps.
Higher coherences are supplied by the $W$-resolution as in Step~3.

\medskip\noindent
\textbf{Conclusion.}
Assembling the four steps yields the equivalence between~(i)
and~(ii): the product decomposition of the operation spaces
(Step~3), the factor-by-factor recognition (Steps~1--2), the Weiss
cosheaf descent (Step~4 via
Lemma~\ref{lem:product-weiss-descent}), and the K\"unneth assembly
of mixed operations complete the proof.\qedhere
\end{proof}

\begin{remark}[Structure of the proof]
\label{rmk:recognition-proof-structure}
The proof rests on three independently established results: the
Costello--Gwilliam holomorphic recognition~\cite{CG17}, the
Ayala--Francis topological recognition~\cite{AF15}, and the
Berger--Moerdijk / Caviglia model structure for colored
operads~\cite{BM03,CM14}. Two structural features reduce
the two-colored recognition to the product of two single-colored
recognitions plus a compatibility datum: the
\emph{product decomposition} of the operation spaces of
$\mathsf{SC}^{\mathrm{ch,top}}$ and the \emph{retract
property} of the closed color. The compatibility datum
is condition~(ii)(c). The Weiss cosheaf descent lemma
(Lemma~\ref{lem:product-weiss-descent}) closes the remaining gap by
showing that the prefactorization data on small rectangles determine
the full operadic structure.
\end{remark}

\begin{remark}[Six operations on the Ran space]
\label{rem:six-operations-on-ran}
The Ran space $\Ran(X) = \mathrm{colim}_n \Sym^n X$ is an
ind--scheme (a filtered colimit of schemes $\Sym^{\le n} X$ along
closed embeddings) rather than a scheme; $\infty$--categorical
functorialities are therefore not available from standard
Grothendieck--six--operations on a single scheme, and must be
assembled from a colimit presentation.

The package of operations $(f^*, f_*, f_!, f^!, \otimes^*, \uHom)$
on the stable $\infty$--category $\DMod(\Ran(X))$ is constructed by
Gaitsgory--Rozenblyum, Volume~I, Chapter IV.5, in the framework of
correspondences of prestacks. The external tensor $\otimes^*$ is the
symmetric monoidal structure arising from pullback along the
diagonal and product functoriality; the dualising object is
$\omega_{\Ran(X)}$, and Verdier duality on the ind--scheme is the
contravariant endofunctor
$\D_{\Ran}\colon \DMod(\Ran(X))^{\op}\to \DMod(\Ran(X))$,
$\mathcal F \mapsto \uHom(\mathcal F, \omega_{\Ran(X)})$.

The chiral tensor $\otimes^{\mathrm{ch}}$ is distinct from the
external tensor $\otimes^*$. Where $\otimes^*$ is external (pullback
of the two factors to $\Ran(X)\times \Ran(X)$ followed by the
external product), $\otimes^{\mathrm{ch}}$ is a pseudo--tensor
structure built from the union map
$\union\colon \Ran(X)\times \Ran(X) \to \Ran(X)$,
$(S,T)\mapsto S\cup T$, restricted to the open substack of disjoint
pairs; it is not a bifunctor on $\DMod(\Ran(X))$ but a family of
multilinear operations indexed by partitions, exactly as in
Beilinson--Drinfeld, Chapter~3.4.10.

Compatibilities. Proper base change on $\Ran(X)$ (Gaitsgory--Rozenblyum
IV.5, $\S 3$) gives $f_! \otimes^*$ preservation and exchange of $f^!$
with $\otimes^*$ up to canonical isomorphism; Verdier duality $\D_{\Ran}$
intertwines $(f^*, f_!)$ with $(f^!, f_*)$. The chiral tensor
$\otimes^{\mathrm{ch}}$ factors through the factorization structure on
$\Ran(X)$: on the disjoint locus $(\Ran(X)\times \Ran(X))_{\mathrm{disj}}$
the union map is an open embedding, so $\otimes^{\mathrm{ch}}$ is
computed by $\union_! \boxtimes$, and the factorization isomorphisms
of Beilinson--Drinfeld 3.4.10 express the compatibility of this
presentation with $\otimes^*$.

Usage in the programme. Theorem~A (bar--cobar) is stated in the
factorization ambient $(\Fact(X),\otimes^*)$: the bar and cobar
functors are endofunctors of augmented factorization algebras, and
the equivalence uses $\otimes^*$ on the conilpotent--complete locus.
Chiral algebras are algebras over the pseudo--tensor
$\otimes^{\mathrm{ch}}$, not over $\otimes^*$; Verdier duality
$\D_{\Ran}$ supplies the Koszul--Verdier compatibility used in the
coderived formulation of Theorem~A and in the definition of the
Lagrangian pairing of Theorem~C. The distinction $\otimes^*$ versus
$\otimes^{\mathrm{ch}}$ is the precise reason for FM56 (chiral algebras
form a pseudo--tensor category, not a symmetric monoidal one), and the
six--operations package above is the ambient in which the repair lives.
\end{remark}

\subsection{The correspondence structure}
\label{subsec:correspondence-structure}

The coalgebraic
shadow (the bar complex) carries a natural correspondence
structure. The bar complex $\barB^{\mathrm{ch}}(\cA) =
T^c(s^{-1}\bar{\cA})$ on $\FM_k(\C) \times \Conf_k(\R)$ is a
correspondence in the precise sense that its differential arises from
convolution and its coproduct arises from the diagonal, exactly as
the Hecke algebra arises from convolution on $\widetilde{\mathcal N}
\times_{\mathfrak g} \widetilde{\mathcal N}$ and restriction along
the groupoid diagonal.

\begin{proposition}[Bar complex as correspondence algebra;
 \ClaimStatusProvedHere]
\label{prop:bar-correspondence}
Let\/ $\cA$ be a chiral algebra on\/ $\C$ with bar complex\/
$\barB^{\mathrm{ch}}(\cA) = T^c(s^{-1}\bar{\cA})$. The bar data
$(d_{\barB}, \Delta)$ admit the following correspondence-geometric
description:
\begin{enumerate}[label=\textup{(\alph*)}]
\item \textbf{Convolution differential.}
 Consider the triple correspondence
 \[
 \FM_k(\C)
 \xleftarrow{\;\iota_S\;}
 D_S \;\cong\; \FM_{k-|S|+1}(\C) \times \FM_{|S|}(\C)
 \xrightarrow{\;\mathrm{compose}\;}
 \FM_{k-|S|+1}(\C),
 \]
 where for each subset\/ $S\subset \{1,\dots,k\}$ with\/
 $|S|\ge 2$, the codimension-$1$ boundary face\/
 $D_S \subset \partial\FM_k(\C)$ is the stratum where the points
 labelled by\/ $S$ collide. The Fulton--MacPherson
 compactification identifies\/ $D_S \cong \FM_{k-|S|+1}(\C)
 \times \FM_{|S|}^{\mathrm{red}}(\C)$: the first factor records the positions
 of the\/ $k-|S|+1$ clusters in the ambient plane, and the second
 records the relative positions of the\/ $|S|$ colliding points at
 infinitesimal scale (modulo dilations, since the cluster size is
 absorbed by the normal coordinate to\/ $D_S$). The bar differential\/ $d_{\barB}$ is a sum
 over such faces: restrict to\/ $D_S$ via the boundary inclusion\/
 $\iota_S$, apply the chiral product\/ $m_{|S|}$ on the\/
 $\FM_{|S|}(\C)$ factor, and re-embed the result into\/
 $\FM_{k-|S|+1}(\C)$. Each summand in\/ $d_{\barB}$ is an OPE
 residue along the collision stratum\/ $D_S$, computed by Stokes'
 theorem as the integral of the bar kernel over\/ $\partial\FM_k$
 (Section~\ref{sec:FM_calculus}).
 (Here ``convolution'' means the boundary-restriction-compose
 recipe on FM strata, not the pushforward-pullback convolution of
 Chriss--Ginzburg; the structural parallel is that both assemble a
 product from a correspondence.)

\item \textbf{Diagonal coproduct.}
 The deconcatenation coproduct\/
 $\Delta\colon T^c(s^{-1}\bar{\cA}) \to
 T^c(s^{-1}\bar{\cA}) \otimes T^c(s^{-1}\bar{\cA})$
 is the pullback\/ $\delta^*$ along the diagonal\/
 $\Conf_k(\R) \to \Conf_k(\R) \times \Conf_k(\R)$
 (here the ``diagonal'' is the universal property of the cofree
 conilpotent coalgebra, not a literal pullback of a geometric map).
 Because the
 ordered configuration space\/ $\Conf_k^{<}(\R)$ is the open
 simplex\/ $\{t_1 < \cdots < t_k\}$, hence contractible,
 the only coassociative coalgebra structure on the cofree
 conilpotent coalgebra\/ $T^c(s^{-1}\bar{\cA})$ compatible with
 the cogenerators is deconcatenation, the diagonal of
 the groupoid of ordered interval decompositions.

\item \textbf{Coderivation = groupoid compatibility.}
 The identity
 \[
 d_{\barB} \circ \Delta
 \;=\;
 (d_{\barB} \otimes \mathrm{id}
 + \mathrm{id} \otimes d_{\barB})
 \circ \Delta
 \]
 states that the convolution differential is a coderivation of the
 diagonal coproduct: the diagonal is a groupoid morphism compatible
 with convolution. This is the content of the bar construction:
 giving a coderivation of the cofree coalgebra is equivalent to
 giving a collection of maps\/
 $\{m_k \colon \bar{\cA}^{\otimes k} \to \cA\}_{k \geq 1}$,
 which are the\/ $A_\infty$ operations of~$\cA$.
\end{enumerate}
\end{proposition}

\begin{proof}
\textbf{Part~(a).}
The bar differential $d_{\barB}$ is the coderivation of
$T^c(s^{-1}\bar{\cA})$ determined by the collection of
$A_\infty$ operations $\{m_k\}_{k \geq 1}$. Its $m_2$-component
acts on a bar element $[a_1|\cdots|a_n]$ by applying $m_2$ to each
\emph{consecutive} pair:
\[
 d_{\barB}^{(2)}[a_1|\cdots|a_n]
 = \sum_{i=1}^{n-1}
 \pm\, [a_1|\cdots|m_2(a_i,a_{i+1})|\cdots|a_n],
\]
where $m_2(a_i,a_{i+1}) = \Res_{z_i = z_{i+1}} \mu(a_i,a_{i+1})$
is the OPE residue along the collision divisor
$D_{\{i,i+1\}} \subset \partial\FM_n(\C)$. This is the
convolution recipe: form the external product of $a_i$ and
$a_{i+1}$ on the two-point space $\FM_2(\C)$, restrict to the
fiber product over the boundary stratum
$D_{\{i,i+1\}} \hookrightarrow \FM_n(\C)$, and push forward along
the collapse map $\FM_n(\C) \to \FM_{n-1}(\C)$ that merges
$z_i$ and $z_{i+1}$. The higher components $d_{\barB}^{(k)}$
for $k \geq 3$, built from the operations $m_k$, arise from
deeper boundary strata (simultaneous collisions of $k$
consecutive points) which correspond to iterated convolutions
along higher fiber products.

\textbf{Part~(b).}
The ordered configuration space $\Conf_k^{<}(\R) = \{t_1 < \cdots
< t_k\} \subset \R^k$ is an open simplex, hence contractible.
On a contractible space, a locally constant cosheaf is determined by
its value at a point. The cofree conilpotent coalgebra
$T^c(V)$ on a graded vector space $V$ admits a unique coassociative
coproduct compatible with the coaugmentation filtration, namely
deconcatenation (this is the defining universal property of the
cofree coalgebra; see e.g.\ Loday--Vallette~\cite{LV12},
Proposition~1.2.9). Geometrically, the deconcatenation
$[a_1|\cdots|a_n] \mapsto \sum_{i=0}^{n} [a_1|\cdots|a_i] \otimes
[a_{i+1}|\cdots|a_n]$ is the diagonal of the self-intersection
groupoid: cutting the interval $[t_1,t_n]$ at $t_i$ gives two
sub-intervals whose ordered configurations assemble into the two
tensor factors.

\textbf{Part~(c).}
A linear map $d\colon T^c(V) \to T^c(V)$ of degree~$+1$ is a
coderivation of the deconcatenation coproduct if and only if it is
determined by its components $\pi \circ d|_{V^{\otimes k}} \colon
V^{\otimes k} \to V$, where $\pi\colon T^c(V) \to V$ is the
cogenerator projection (Loday--Vallette~\cite{LV12},
Proposition~1.3.3). These components are the
$A_\infty$ operations $\{m_k\}_{k \geq 1}$. The condition
$d_{\barB}^{\,2} = 0$ then encodes the $A_\infty$ relations
$\sum_{p+q+r=n} m_{p+1+r} \circ (\mathrm{id}^{\otimes p} \otimes
m_q \otimes \mathrm{id}^{\otimes r}) = 0$, which are exactly the
Stasheff identities. This is Vol~I, Theorem~A: the bar differential
is a coderivation, and the bar-cobar adjunction is the translation
between coderivations of the cofree coalgebra and $A_\infty$
structure maps.
\end{proof}

\label{cor:R-factorization-lagrangian}
\textup{(\ClaimStatusProvedHere.)}

$\R$-factorization is determined as the unique

\begin{proof}
By Proposition~\ref{prop:bar-correspondence}(b), the

\end{proof}

\subsection{The five theorems as Lagrangian geometry}
\label{subsec:five-theorems-lagrangian}

\providecommand{\cL}{\mathcal L}
\providecommand{\cM}{\mathcal M}
\providecommand{\HH}{\operatorname{HH}}
\providecommand{\Steinb}{\mathfrak{S}}

Volume~I establishes five theorems about the categorical logarithm.
Theorem~\ref{thm:bar-is-self-intersection} (the bar complex is the
formal bar/Koszul model for the Lagrangian self-intersection)
reveals their geometric content.
Each theorem is a shadow of the same Lagrangian embedding
$\cL \hookrightarrow \cM$, read through different invariants.

\begin{interpretation}[The five theorems and their Lagrangian-geometric shadow;
\ClaimStatusHeuristic]
\label{prop:five-theorems-geometry}
Let\/ $\cM$ be a $(-2)$-shifted symplectic derived stack and\/
$\cL \hookrightarrow \cM$ a Lagrangian, with\/
$\Steinb = \cL \times_{\cM}^h \cL$ the derived self-intersection.
Write\/ $B = \cO(\cL)$. Then Vol~I's five theorems admit the
following geometric readings:
\begin{enumerate}[label=\textup{(\roman*)},leftmargin=2em]
\item \textbf{Theorem~A \textup{(}Bar-cobar adjunction\textup{)}.}
 The self-intersection\/ $\Steinb = \cL \times_{\cM}^h \cL$ is a
 groupoid object over~$\cL$: source and target are the two
 projections $\Steinb \rightrightarrows \cL$, composition is the
 fiber product over~$\cM$, and the unit is the diagonal
 $\cL \to \Steinb$. Every groupoid object in a stable category
 determines an adjunction between its category of comodules\/
 \textup{(}the bar side: cofree coalgebra on the conormal
 fibers\textup{)} and its category of modules\/
 \textup{(}the cobar side: free algebra on the normal
 fibers\textup{)}. The bar-cobar adjunction of Volume~I is
 read on this surface through the corresponding formal
 groupoid comodule-module adjunction.

\item \textbf{Theorem~B \textup{(}Koszul inversion\textup{)}.}
 On the clean locus, where the derived intersection has no
 higher Tor, the self-intersection groupoid\/
 $\Steinb$ collapses to its base and the bar-cobar round-trip
 is an equivalence. Off the clean locus, the higher-Tor
 directions persist, and the bar-cobar map is a
 completion rather than an inversion.

\item \textbf{Theorem~C \textup{(}Complementarity\textup{)}.}
 Two complementary Lagrangians\/ $\cL, \cL'$ in~$\cM$ intersect
 in a derived scheme\/ $\cL \times_{\cM}^h \cL'$ carrying a
 $(-1)$-shifted symplectic structure \textup{(}PTVV
 \cite{PTVV13}\textup{)}. The complementarity potential\/
 $S_{\cA}$ of Volume~I is read geometrically as the action
 functional attached to this $(-1)$-shifted symplectic
 intersection.
 The complementary pair decomposes the tangent space\/
 $T_p\cM \simeq T_p\cL \oplus T_p\cL'$ at every intersection
 point~$p$, and this decomposition is the complementary
 Lagrangian splitting of the obstruction complex that governs the
 genus-$g$ deformation theory.

\item \textbf{Theorem~D \textup{(}Leading coefficient\textup{)}.}
 The genus-$1$ curvature, and conditionally its higher-genus
 scalar continuation\/ $\kappa(\cA) \cdot \omega_g$, are read as
 the first-order deformation of the
 Lagrangian embedding\/ $\cL \hookrightarrow \cM_{\mathrm{vac}}$
 as the moduli of curves varies over\/ $\overline{\mathcal M}_g$.
 In this dictionary, the equation
 $\dfib^2 = \kappa(\cA) \cdot \omega_g$ is expected to record the failure of the
 displaced Lagrangian to remain isotropic to first order.

\item \textbf{Theorem~H \textup{(}Hochschild ring\textup{)}.}
 The HKR theorem for Lagrangian embeddings gives
 \[
 \HH^\bullet(B)
 \;\simeq\;
 \cO\bigl(T^*[-1]\cL\bigr).
 \]
 The Hochschild cohomology of the boundary algebra~$B$ is the
 ring of functions on the shifted cotangent bundle of~$\cL$,
 which supplies the shifted-cotangent formal local bulk model.
 Recovering the actual formal neighborhood of~$\cL$ in~$\cM$
 requires the additional $(-1)$-shifted closed $1$-form from the
 formal Darboux theorem. This is the geometric content of
 Theorem~H: the derived center of the boundary fixes the
 shifted-cotangent side of the bulk reconstruction, and the BV
 bracket on bulk observables is the $(-1)$-shifted Poisson bracket
 on $T^*[-1]\cL$ induced by the ambient $(-2)$-shifted symplectic
 structure.
\end{enumerate}
\end{interpretation}

\begin{proof}[Justification]
This interpretation is a geometric dictionary, not a new package of
global derived-stack theorems. Item~\textup{(i)} uses the standard
groupoid structure on\/ $\cL \times_{\cM}^h \cL$ together with
Theorem~\ref{thm:bar-is-self-intersection}(ii), which identifies the
derived tensor product with its formal bar/Koszul model. Item~\textup{(iii)}
is the PTVV geometry of intersections of Lagrangians, read through the
complementarity package of Volume~I. Item~\textup{(v)} is the HKR
identification of Theorem~\ref{thm:bar-is-self-intersection}(iii).

Items~\textup{(ii)} and~\textup{(iv)} are intentionally weaker: they
record the expected global geometry suggested by the formal model.
Item~\textup{(ii)} would require showing that the higher Tor terms
$\Tor^{\cO(\cM)}_i(B,B)$ vanish for a global derived stack
embedding, not just for the formal bar/Koszul model; this is the
content of the \'etale groupoid conjecture
(Conjecture~\ref{conj:etale-groupoid-koszul}).
Item~\textup{(iv)} would require a Kodaira--Spencer class
computation for a family of Lagrangian embeddings parametrised
by~$\overline{\mathcal M}_g$, establishing that the first-order
displacement class equals $\kappa(\cA) \cdot \omega_g$; the formal
model suggests this, but the global derived-stack argument has not
been carried out.
\end{proof}

\begin{remark}[The single-phenomenon principle]
\label{rem:single-phenomenon}
Interpretation~\ref{prop:five-theorems-geometry} is not five separate
theorems unified by analogy. It is a single formal-local principle:
the derived self-intersection admits the standard bar/Koszul model
$\barB(B)$ and its Hochschild shadow is the shifted-cotangent model
(formalized as Theorem~\ref{thm:steinberg-presentation}):
the groupoid structure (A), the expected \'etaleness picture (B), the
$(-1)$-shifted symplectic form on a complementary intersection (C),
the deformation over moduli of curves (D), and the HKR isomorphism
(H). The Lagrangian self-intersection is the organizing geometry,
and the five theorems are its five faces.
\end{remark}

\begin{remark}[\'Etale groupoid and Koszul Morita theorem]
\label{rem:etale-cross-ref}
Heuristically, for Koszul algebras the self-intersection groupoid
$\Steinb \rightrightarrows \cL$ is expected to be \'etale
\textup{(}Conjecture~\textup{\ref{conj:etale-groupoid-koszul}}\textup{)}:
both source and target maps are expected to be quasi-isomorphisms of
derived stacks, because the higher Tor terms
$\operatorname{Tor}^{\cO(\cM)}_i(B,B) = 0$ vanish for $i > 0$.
Koszul inversion is then expected to be the corresponding groupoid
Morita theorem \textup{(}Conjecture~\textup{\ref{conj:koszul-morita}}\textup{)}.
Off the Koszul locus, the groupoid acquires genuine derived
fibers (the higher $A_\infty$ operations), and the Morita
equivalence is replaced by completion. The transition from the
unconditional categorical equivalence of
Theorem~\ref{thm:steinberg-presentation}(ii) to the Koszul
specialisation of~(iii) is the passage from the formal derived
groupoid model to its conjectural \'etale shadow.
\end{remark}

\begin{proposition}[Koszul duality as complementary Lagrangian symmetry;
\ClaimStatusProvedHere]
\label{prop:koszul-serre}
\index{Koszul duality!as complementary Lagrangian symmetry|textbf}
\index{complementary Lagrangians!Koszul involution}
Koszul duality $\cA \leftrightarrow \cA^!$ is the exchange of
complementary Lagrangians in the derived symplectic category.
Precisely:
\begin{enumerate}[label=\textup{(\roman*)}]
\item The Lagrangians $\cL_\cA$ and $\cL_{\cA^!}$ are complementary
 in $\cM$: $T_{\cL_\cA}\cM \simeq T\cL_\cA \oplus T\cL_{\cA^!}$.
\item Their intersection $\cL_\cA \times_\cM^h \cL_{\cA^!}$
 carries the $(-1)$-shifted symplectic form that is the bar-cobar
 pairing
 $\langle\barB(\cA),\, \Omegach(\barB(\cA))\rangle$.
\item The involution $(\text{-})^{!!} \simeq \id$
 \textup{(}double Koszul duality is the identity on the Koszul
 locus\textup{)} follows from the \emph{self-duality} of the
 bar-cobar pairing, which is symmetric because the ambient
 $(-2)$-shifted symplectic form on~$\cM$ is symmetric.
 Geometrically, this is the statement that complementarity is
 symmetric: if\/ $\cL_\cA$ and\/ $\cL_{\cA^!}$ are complementary
 Lagrangians, then\/ $\cL_{\cA^!}$ and\/ $\cL_{(\cA^!)^!}$ are
 also complementary, and the resulting Lagrangian\/
 $\cL_{(\cA^!)^!}$ coincides with\/ $\cL_\cA$.
\end{enumerate}
\end{proposition}

\begin{proof}
(i) is Vol~I, Theorem~C: the deformation and obstruction complexes
of $\cA$ and $\cA^!$ decompose $T\cM|_\cL$ into complementary
Lagrangian summands. (ii)~follows from PTVV, Theorem~2.9: the
intersection of complementary Lagrangians inherits a $(-1)$-shifted
symplectic form, and this form is the non-degenerate pairing between
bar and cobar.

(iii)~The $(-2)$-shifted symplectic form $\omega_{-2}$ on $\cM$ is
symmetric (even shift), so the bar-cobar pairing
$\langle \barB(\cA), \Omegach(\barB(\cA)) \rangle$, which is the
restriction of $\omega_{-2}$ to the Lagrangian intersection, is
a \emph{self-dual} pairing. Algebraically: on the Koszul locus, the
bar-cobar adjunction is a Quillen equivalence
(Vol~I, Theorem~B): $\Omegach(\barB(\cA)) \xrightarrow{\sim} \cA$.
Applying the bar construction and then Koszul duality twice gives
$(\cA^!)^! = \Omegach(\barB(\cA^!))$. Since $\cA^!$ is Koszul when
$\cA$ is (by the symmetry of the complementary Lagrangian
decomposition), the bar-cobar unit for $\cA^!$ is also an
equivalence, so $(\cA^!)^! \simeq \cA$. Geometrically: the
complementary decomposition $T\cM|_\cL \simeq T\cL_\cA \oplus
T\cL_{\cA^!}$ is symmetric in its two summands (because
$\omega_{-2}$ is symmetric, so the roles of the two Lagrangian
complements are interchangeable), and
exchanging the roles of $\cA$ and $\cA^!$ gives
$T\cM|_{\cL_{\cA^!}} \simeq T\cL_{\cA^!} \oplus T\cL_\cA$, with
the second complement recovering $\cL_\cA$.

\smallskip
\noindent\emph{Clarification.}
The involution $(\text{-})^{!!} \simeq \mathrm{id}$ does \emph{not}
follow from $S^2 = \mathrm{id}$ for a Serre functor. For a
Calabi--Yau variety of dimension~$n$, the Serre functor is
$S = [n] \otimes \omega$ and $S^2 = [2n] \otimes \omega^2$, which is
generally nontrivial. The mechanism here is different: the
$(-2)$-shifted symplectic form is symmetric (even degree), so the
induced pairing on the bar-cobar complex is self-dual, and this
self-duality forces $(-)^{!!} \simeq \mathrm{id}$ on the Koszul
locus.
\end{proof}

\begin{remark}[The formal/non-formal dichotomy as the perturbative/non-perturbative divide]
\label{rem:koszul-perturbative-watershed}
\index{Swiss-cheese formality!perturbative/non-perturbative dichotomy|textbf}
\index{perturbative!Lagrangian intersection characterization}
\index{non-perturbative!excess intersection}
The formal/non-formal Swiss-cheese dichotomy is the mathematical form
of the perturbative/non-perturbative divide. An algebra with formal
Swiss-cheese structure has a \emph{clean} Lagrangian
self-intersection: the derived intersection
$\cL_\cA \times_\cM^h \cL_\cA$ is concentrated in degree zero, the
QFT is perturbatively exact (one-loop, no higher corrections), genus
corrections are scalar multiples of $\omega_g$, and the bar-complex
coproduct is genus-independent (though the $R$-matrix may still
acquire genus dependence through the propagator channel when the
Coisson bracket $c_0 = \{a\,{}_0\,b\}$ is nonzero).
An algebra with non-formal Swiss-cheese structure has an \emph{excess}
intersection: the derived self-intersection carries higher cohomology,
higher $\Ainf$ operations proliferate like instanton corrections,
curvature and braiding entangle
(Theorem~\ref{thm:pole-structure-dichotomy}), and the genus tower
carries genuinely new data at each level. The OPE pole structure
determines the intersection type: simple poles give transverse meeting
(perturbative); higher-order poles create excess dimension in the
derived fiber (non-perturbative).
(Note: all standard-landscape algebras, including Virasoro and
$\mathcal{W}_N$, are chirally Koszul by PBW universality;
the formal/non-formal distinction classifies complexity
\emph{within} the Koszul locus.)
\end{remark}

\subsection{Curved looping and the coderived category}
\label{subsec:curved-looping-coderived}

At genus~$0$ the bar complex is honest: $d_{\barB}^2 = 0$.
At genus~$1$, and conditionally on the higher-genus scalar
continuation, Volume~I Theorem~D identifies the square of the
bar differential with a scalar multiple of a moduli-space
cohomology class: $d_{\barB}^2 = \kappa(\cA) \cdot \omega_g$.
This curvature is not a defect but
a structure: it measures exactly how much monodromy the
$D$-module connection acquires around the $B$-cycles
of~$\Sigma_g$, and it controls the genus tower of the theory.
This subsection develops the consequences: the
curved bar complex lives in the coderived category (Positselski's
framework), the bar-cobar
adjunction lifts to the curved setting, and the delooping of a
curved algebra determines a line bundle over the moduli space whose
first Chern class is the modular characteristic.

\begin{definition}[Curved $\Ainf$ chiral algebra]
\label{def:curved-Ainf-chiral}
A logarithmic $\SCchtop$-algebra $\cA$
(Definition~\ref{def:log-SC-algebra}) is \emph{curved}
if its bar differential satisfies
\[
 d_{\barB}^2 \;=\; \kappa(\cA) \cdot \omega
 \;\neq\; 0,
\]
where $\kappa(\cA) \in \mathbb{k}$ is the \emph{modular
characteristic} of~$\cA$ and $\omega$ is a cohomology class on the
moduli space of curves.
\begin{enumerate}[label=\textup{(\roman*)}]
\item At genus~$0$, $\omega = 0$: the moduli space
 $\overline{\mathcal M}_{0,n}$ contributes no obstruction, and the
 bar complex is honest ($d^2 = 0$).
\item At genus~$g \geq 1$, $\omega = \omega_g \in
 H^2(\overline{\mathcal M}_g, \mathbb{k})$ is the Hodge class
 $\lambda_1 = c_1(\det \lambda)$, where
 $\lambda = \pi_* \omega_{C/\overline{\mathcal M}_g}$ is the
 rank-$g$ Hodge bundle.
 At genus~$1$ this equation is the content of Vol~I,
 Theorem~D; the higher-genus scalar extension is conditional
 after the Vol~I rectification. The leading coefficient of the
 categorical logarithm is the modular characteristic.
\end{enumerate}
An algebra with $\kappa(\cA) = 0$ is \emph{uncurved};
it admits an honest bar complex with $d^2 = 0$ at all genera.
\end{definition}

In Positselski's framework, the curvature element is
$m_0 := \kappa(\cA) \cdot \omega_g
 \in H^2(\overline{\mathcal M}_g, \mathbb{k})$,
a degree-$2$ element with no inputs. The identity
$d_{\barB}^2$ acts by the zero-ary operation~$m_0$: the identity
$d_{\barB}^2(x) = m_2(m_0, x) \pm m_2(x, m_0)$ follows from the
curved $\Ainf$ relations. When
$m_0 = \kappa(\cA) \cdot \omega_g$ lies in the base direction (a
scalar times a moduli class), the action $m_2(m_0, x) = \kappa(\cA)
\cdot \omega_g \cdot x$ is multiplication by the curvature scalar,
because $m_2$ with a base-direction input reduces to scalar
multiplication on fibers.

\begin{lemma}[Centrality of the curvature element]
\label{lem:curvature-centrality}
The element $m_0 = \kappa(\cA) \cdot \omega_g$ commutes with all\/
$\Ainf$ operations $m_k$ $(k \geq 1)$: for every bar element~$x$,
$[m_0, m_k(\ldots)] = m_k(\ldots, m_0 \cdot -, \ldots)$ vanishes.
\end{lemma}

\begin{proof}
The class $\omega_g \in H^*(\overline{\mathcal M}_g)$ acts on the
total complex $\barB^{(g)}(\cA)$ by multiplication in the base
direction. The operations $m_k$ ($k \geq 1$) act fiberwise: they
are defined on $\mathrm{FM}_k(\mathbb{C}) \times
\mathrm{Conf}_k(\mathbb{R})$ and do not involve the moduli-space
direction. Since base multiplication and fiberwise operations act on
complementary tensor factors, they commute. As $\kappa(\cA) \in
\mathbb{k}$ is a scalar, $m_0 = \kappa \cdot \omega_g$ commutes with
every $m_k$.
\end{proof}

\begin{proposition}[Modular characteristic as Euler class; \ClaimStatusProvedHere]
\label{prop:kappa-euler-class}
Let $\cA$ be a curved $A_\infty$ chiral algebra with cyclic structure $\langle -, - \rangle_{\mathrm{cyc}}$ (Construction~\textup{\ref{constr:cyclic-structure}}). The modular characteristic $\kappa(\cA)$ is the \emph{Euler class} of the cyclic $A_\infty$ structure: it is the unique scalar such that
\[
d_{\barB}^2 \;=\; \kappa(\cA) \cdot \omega_g \;\in\; H^2(\overline{\cM}_g, \mathbb{k}),
\]
where the left-hand side is computed using the cyclic trace $\Tr_\cC$ and the right-hand side is the Hodge class.

More precisely: the cyclic trace $\Tr_\cC \colon \HH_\ast(\cC) \to \mathbb{k}$ determines a section of the determinant line bundle $\det(\lambda)$ over $\overline{\cM}_g$ (by evaluating on the fundamental Hochschild cycle of the family $\cA|_C$ as $C$ varies over $\overline{\cM}_g$). The vanishing locus of this section has class $\kappa(\cA) \cdot c_1(\det\lambda) = \kappa(\cA) \cdot \omega_g$. Thus $\kappa(\cA)$ is the degree of the section, the Euler class of the cyclic structure.
\end{proposition}

\begin{proof}
The cyclic trace on $\HH_\ast(\cC|_C)$ varies with $C \in \overline{\cM}_g$: at a smooth curve, it evaluates the genus-$g$ partition function $Z_g(\cA|_C)$. The partition function $Z_g$ is a section of $\cL_\kappa^{-1} = (\det\lambda)^{-\kappa}$ (Construction~\ref{constr:determinant-line-bundle}), so its vanishing locus has class $\kappa \cdot \omega_g$. At genus~$1$, and conditionally on the higher-genus scalar continuation, the identification of this vanishing class with $d_{\barB}^2$ follows from Volume~I, Theorem~D: the curvature of the bar differential equals $\kappa \cdot \omega_g$, and the curvature IS the Euler class of the determinant line bundle by the Chern--Weil correspondence (Construction~\ref{constr:determinant-line-bundle}, Step~5).
\end{proof}

\begin{proposition}[Coderived vs derived; \ClaimStatusProvedHere]
\label{prop:coderived-vs-derived}
Let\/ $\cA$ be a chiral algebra with modular characteristic\/
$\kappa(\cA)$.
\begin{enumerate}[label=\textup{(\roman*)}]
\item If\/ $\kappa(\cA) = 0$, the bar complex\/
 $\barB(\cA)$ is an object of the derived category\/
 $D^b(\mathrm{CoAlg}_{E_1}^{\mathrm{ch}})$ of conilpotent\/
 $E_1$ dg coassociative coalgebras: the differential squares to zero, and
 quasi-isomorphisms invert.
\item If\/ $\kappa(\cA) \neq 0$, the bar complex\/
 $\barB^{\mathrm{curv}}(\cA)$ is an object of the coderived
 category\/
 $D^{\mathrm{co}}(\mathrm{CoAlg}_{E_1}^{\mathrm{ch}})$, defined as the
 quotient of the homotopy category of curved dg coalgebras by
 coacyclic objects.
\item The bar-cobar adjunction lifts to an adjunction between
 curved algebras and coderived coalgebras:
 \[
 \Omega^{\mathrm{ch,co}}
 \colon
 D^{\mathrm{co}}(\mathrm{CoAlg}_{E_1}^{\mathrm{ch}})
 \rightleftarrows
 D(\mathrm{Alg}_{E_1}^{\mathrm{ch,curv}})
 :\!
 \barB^{\mathrm{curv}}.
 \]
\end{enumerate}
\end{proposition}

\begin{proof}
\textbf{Part~(i).}
When $\kappa(\cA) = 0$, the bar differential satisfies
$d_{\barB}^2 = 0$ at every genus. The bar complex is therefore an
ordinary dg coalgebra, and it lives in the derived category by the
content of Vol~I, Theorem~A (the bar-cobar adjunction for honest
$\Ainf$ algebras).

\textbf{Part~(ii).}
The total differential $D = \dfib + d_{\mathrm{base}}$ on the
family $\barB^{(g)}(\cA)$ over $\overline{\mathcal M}_g$ satisfies
\[
 D^2 \;=\; \dfib^2 + [\dfib, d_{\mathrm{base}}]
 + d_{\mathrm{base}}^2
 \;=\; \kappa(\cA) \cdot \omega_g,
\]
where $d_{\mathrm{base}}^2 = 0$ because the de~Rham
differential on $\overline{\mathcal M}_g$ is flat, and the
cross-term $[\dfib, d_{\mathrm{base}}] = 0$ for the following reason:
the bar differential $\dfib$ is defined fiberwise: it depends on
the chiral algebra structure of $\cA$, not on the moduli parameter
$[C] \in \overline{\mathcal M}_g$. The base differential
$d_{\mathrm{base}}$ is the de~Rham differential on
$\overline{\mathcal M}_g$, which does not see the fiber operations.
Therefore $[\dfib, d_{\mathrm{base}}] = 0$ as operators on the
total complex, by the Leibniz rule for the tensor product
differential on a trivial fibration. Thus the total
complex $(\barB^{(g)}(\cA), D)$ is a \emph{curved} dg coalgebra
with curvature $\kappa(\cA) \cdot \omega_g$. It is \emph{not} an
object of the derived category $D^b$ (where $d^2 = 0$ is required),
but \emph{is} a legitimate object of the coderived category
$D^{\mathrm{co}}$, which admits curved objects by definition. The
coderived category is the quotient of the homotopy category of curved
dg coalgebras by coacyclic objects (those belonging to the smallest
triangulated subcategory closed under infinite coproducts and
containing total complexes of short exact sequences).

\textbf{Part~(iii).}
The uncurved bar-cobar adjunction (Vol~I, Theorem~A) has unit
and counit built from universal twisting morphisms
$\tau \colon \barB(\cA) \to \cA$. In the curved setting,
replace $\barB$ by $\barB^{\mathrm{curv}}$ and $\Omegach$ by
$\Omega^{\mathrm{ch,co}}$, the curved cobar construction. The
universal twisting morphism still exists: the Maurer--Cartan
equation $d\tau + \tau \star \tau = 0$ is replaced by the curved
Maurer--Cartan equation
$d\tau + \tau \star \tau + m_0 = 0$, i.e.\
$d\tau + \tau \star \tau = -m_0 = -\kappa \cdot \omega_g$
(Positselski's sign: $m_0$ is the curvature with
$d^2 = m_0 \cdot {-}$, so the MC equation acquires
$-m_0$ on the right-hand side), whose
solvability is guaranteed by the centrality of $\kappa$ (Vol~I,
Theorem~D: $\kappa$ is a scalar, so $\kappa \cdot \omega_g$
commutes with all operations). The adjunction unit and counit
are curved quasi-isomorphisms: maps whose cones are coacyclic.
For conilpotent coalgebras over a field of characteristic zero,
coacyclic objects coincide with acyclic objects, so $D^{\mathrm{co}}$
agrees with $D^b$ in the uncurved case. In the curved case
($d^2 \neq 0$), acyclicity is not defined, and the coderived
category provides the natural home. The adjunction descends to
the stated localization.
\end{proof}

\begin{theorem}[Curved bar--cobar at genus $g \geq 1$;
\ClaimStatusProvedHere]
\label{thm:curved-bar-cobar-genus-ge-1}
Let $\cA$ be a logarithmic $\SCchtop$-algebra
(Definition~\textup{\ref{def:log-SC-algebra}}) with modular
characteristic $\kappa = \kappa(\cA) \in \mathbb{k}$, and let
$g \geq 1$. Denote by $\barB^{\mathrm{curv}}_g(\cA) =
(T^c(s^{-1}\bar\cA), \dfib, m_0)$ the curved bar coalgebra with
scalar curvature $m_0 = \kappa \cdot \omega_g \in
H^2(\overline{\mathcal M}_g, \mathbb{k})$
\textup{(}Definition~\textup{\ref{def:curved-Ainf-chiral}}\textup{)},
and by $\Omega^{\mathrm{ch,co}}_g$ the curved cobar construction of
Positselski~\cite{Positselski2011}.
\begin{enumerate}[label=\textup{(\roman*)}]
\item \textbf{Quillen equivalence of CDG categories.} The pair
 \[
 \Omega^{\mathrm{ch,co}}_g \colon
 \Dco\!\bigl(\mathrm{CoAlg}_{E_1}^{\mathrm{ch,curv}}\bigr)
 \;\rightleftarrows\;
 D\!\bigl(\mathrm{Alg}_{E_1}^{\mathrm{ch,curv}}\bigr)
 :\! \barB^{\mathrm{curv}}_g
 \]
 is a Quillen adjunction of CDG model categories
 \textup{(}Positselski's nonhomogeneous framework, specialised to
 the BD-chiral pseudo-tensor\textup{)} and a Quillen equivalence
 on the subcategory of conilpotent-complete curved coalgebras
 with finite-dimensional graded pieces.
\item \textbf{Inversion on the Koszul locus.} On the chirally
 Koszul locus \textup{(}in the sense of
 Theorem~\textup{\ref{thm:filtered-koszul}}\textup{)}, the counit
 of the curved adjunction is a curved quasi-isomorphism
 \[
 \Omega^{\mathrm{ch,co}}_g\,\barB^{\mathrm{curv}}_g(\cA)
 \;\simeq\;
 \cA^{[\kappa]},
 \]
 where $\cA^{[\kappa]}$ denotes $\cA$ twisted by the determinant
 line bundle $\mathcal{L}_\kappa =
 (\det\lambda)^{\otimes\kappa}$ of
 Construction~\textup{\ref{constr:determinant-line-bundle}}.
\item \textbf{Reduction to Theorem~A at $g=0$.} When $g = 0$,
 $\omega_0 = 0$, the curvature $m_0$ vanishes,
 $\mathcal{L}_\kappa$ is trivial, the CDG category reduces to the
 ordinary dg category, and the statement specialises to
 Volume~I Theorem~A: $\Omegach \barB(\cA) \simeq \cA$ on the
 Koszul locus.
\end{enumerate}
\end{theorem}

\begin{proof}
\textbf{Part (i).} The curved bar $\barB^{\mathrm{curv}}_g(\cA)$
is a CDG coalgebra in Positselski's sense
(Proposition~\ref{prop:coderived-vs-derived}(ii);
Theorem~\ref{thm:curved-koszul-genus-g}(i)):
conilpotent-complete graded coalgebra, degree-$+1$ coderivation
$\dfib$, centrality of the scalar curvature
$m_0 = \kappa \cdot \omega_g$
(Lemma~\ref{lem:curvature-centrality}), and structural relation
$\dfib^2 = m_0 \cdot \id$. The curved cobar construction
$\Omega^{\mathrm{ch,co}}_g$ is defined by the dual formulas:
free graded algebra on $s\bar C$, derivation from the coproduct,
and correction term absorbing the curvature
(Positselski~\cite{Positselski2011}, \S6.1--6.3). The
nonhomogeneous CDG Quillen pair
$(\Omega^{\mathrm{ch,co}}_g \dashv \barB^{\mathrm{curv}}_g)$
transfers along the BD-chiral pseudo-tensor enrichment via
Francis--Gaitsgory factorization categories~\cite{GR17} under
the $\star$-tensor structure: the Francis--Gaitsgory model
structure on factorization (co)algebras is compatible with the
curved extension because the curvature $m_0$ is central
(no obstruction to lifting cofibrations). Fibrant replacement
in the coderived model structure uses Positselski's
coacyclic-object criterion; cofibrant replacement uses
conilpotent completion.

\textbf{Part (ii).} On the chirally Koszul locus the uncurved
unit and counit of Theorem~A are quasi-isomorphisms. The curved
extension reuses the same universal twisting morphism
$\tau_\cA \colon \barB(\cA) \to \cA$ now solving the curved
Maurer--Cartan equation $d\tau + \tau \star \tau + m_0 = 0$
(Proposition~\ref{prop:coderived-vs-derived}(iii)). Centrality of
$m_0$ guarantees solvability. Conilpotent completion converts the
formal-series twisting morphism into an honest element of
$\Omega^{\mathrm{ch,co}}_g \barB^{\mathrm{curv}}_g(\cA)$; the
counit comparison map is a curved quasi-isomorphism. The residual
$\kappa$-twist arises because the Maurer--Cartan solution lives
over $\overline{\mathcal M}_g$: passage through the coderived
category records the holonomy of the determinant line bundle
$\mathcal{L}_\kappa$, exactly the twist of
Proposition~\ref{prop:curved-delooping}(ii). Thus
$\Omega^{\mathrm{ch,co}}_g \barB^{\mathrm{curv}}_g(\cA) \simeq
\cA^{[\kappa]}$.

\textbf{Part (iii).} At $g = 0$ the moduli class $\omega_0 = 0$
(genus-zero Deligne--Mumford space is a point for $n \leq 3$ and
rationally connected with trivial $H^2$ for the tautological
subring we use), so $m_0 = 0$. The curved MC equation degenerates
to the ordinary MC equation; the CDG coalgebra is an ordinary dg
coalgebra; coderived equals derived for conilpotent dg coalgebras
in characteristic zero
(Proposition~\ref{prop:coderived-vs-derived} proof, part (iii));
$\mathcal{L}_\kappa$ is trivial so the twist disappears; the
statement is Volume~I Theorem~A.
\end{proof}

\begin{proposition}[Curved delooping; \ClaimStatusProvedHere]
\label{prop:curved-delooping}
Let\/ $\cA$ be a curved\/ $\Ainf$ chiral algebra with modular
characteristic\/ $\kappa \neq 0$.
\begin{enumerate}[label=\textup{(\roman*)}]
\item The delooping\/ $B(\cA)$ determines a line bundle\/
 $\mathcal{L}_\kappa$ on\/ $\overline{\mathcal M}_g$ with first
 Chern class
 \[
 c_1(\mathcal{L}_\kappa)
 \;=\;
 \kappa(\cA) \cdot [\omega_g]
 \;\in\;
 H^2(\overline{\mathcal M}_g,\, \mathbb{Z}).
 \]
 For the affine lineage $\cA = \hat{\mathfrak{g}}_k$, this is the\/
 $k$-th power of the determinant of the Hodge bundle:
 $\mathcal{L}_k = (\det \lambda)^{\otimes k}$.
\item The loop space\/ $\Omega(B(\cA))$ recovers\/ $\cA$ up to a
 twist by\/ $\mathcal{L}_\kappa$: there is an equivalence of
 curved\/ $\Ainf$ chiral algebras
 \[
 \Omega(B(\cA))
 \;\simeq\;
 \cA^{[\kappa]},
 \]
 where\/ $\cA^{[\kappa]}$ denotes\/ $\cA$ twisted by\/
 $\mathcal{L}_\kappa$. When\/ $\kappa = 0$, the line bundle is
 trivial and\/ $\Omega(B(\cA)) \simeq \cA$ as an honest
 equivalence.
\item When\/ $\kappa = 0$, the line bundle is trivial\/
 ($c_1 = 0$), the loop-deloop equivalence is honest:
 $\Omega B(\cA) \simeq \cA$, and the bar-cobar adjunction
 reduces to Vol~I, Theorem~A.
 When\/ $\kappa \neq 0$, $\Omega(B(\cA))$ is\/ $\cA$ twisted
 by\/ $\mathcal{L}_\kappa$.
\end{enumerate}
\end{proposition}

\begin{proof}
\textbf{Part~(i).}
The bar complex $\barB^{\mathrm{curv}}(\cA)$ over
$\overline{\mathcal M}_g$ has fiberwise curvature
$\dfib^2 = \kappa \cdot \omega_g$. The curvature class
$\kappa(\cA) \cdot [\omega_g] \in H^2(\overline{\mathcal M}_g,
\mathbb{Z})$ is a degree-$2$ cohomology class, and therefore
determines a line bundle $\mathcal{L}_\kappa$ with first Chern
class $c_1(\mathcal{L}_\kappa) = \kappa \cdot [\omega_g]$.
(A gerbe would require an $H^3$ obstruction class, but
$\kappa \cdot [\omega_g]$ lives in $H^2$.) When the class vanishes
($\kappa = 0$), the line bundle is trivial and the bar complex is
an honest dg coalgebra bundle; when $\kappa \neq 0$, the bar
complex is twisted by $\mathcal{L}_\kappa$.

For the affine lineage $\cA = \hat{\mathfrak{g}}_k$,
the Verlinde line bundle is $\mathcal{L}_k = (\det \lambda)^{\otimes k}$
where $\lambda = \det R\pi_* \omega_{C/\overline{\mathcal M}_g}$ is the
Hodge bundle. The twisting is by the level~$k$, which controls conformal
blocks. The Vol~I modular characteristic
$\kappa(\hat{\fg}_k) = (k+h^\vee)\dim\fg/(2h^\vee)$ is a different
normalization: it measures the shadow obstruction tower coefficient
rather than the Verlinde level. (The Heisenberg algebra is not
an affine Kac--Moody algebra: its modular characteristic
$\kappa(\cH_k) = k$ is computed independently, not by taking
$h^\vee \to 0$ in the affine formula, which is undefined at
$h^\vee = 0$.)

\textbf{Part~(ii).}
The loop space $\Omega(B(\cA))$ is the fiber of the evaluation map
$B(\cA) \to B(\cA)|_{\mathrm{pt}}$ over a basepoint. When
$c_1(\mathcal{L}_\kappa) \neq 0$, the fiber is not $\cA$ itself
but $\cA$ twisted by the holonomy of $\mathcal{L}_\kappa$.
This twist is a $\kappa$-dependent automorphism of $\cA$ as a
curved $\Ainf$ algebra, corresponding to the monodromy of the
connection $D = \dfib + d_{\mathrm{base}}$ around loops in
$\overline{\mathcal M}_g$. The equivalence
$\Omega(B(\cA)) \simeq \cA^{[\kappa]}$ follows from the lifted
bar-cobar adjunction
(Proposition~\ref{prop:coderived-vs-derived}(iii)): the counit of
the adjunction
$\Omega^{\mathrm{ch,co}}(\barB^{\mathrm{curv}}(\cA))
\xrightarrow{\sim} \cA$ is a curved quasi-isomorphism, and the
twist by $\kappa$ records the passage through the coderived
category. (Part~(ii) is proved in the formal setting; the
analytic extension requires additional input from the PTVV
framework.)

\textbf{Part~(iii).}
When $\kappa = 0$, $c_1(\mathcal{L}_\kappa) = 0$, the line bundle
is trivial, the twist disappears, and the bar-cobar adjunction
reduces to the uncurved version (Vol~I, Theorem~A). The
loop-deloop equivalence $\Omega B(\cA) \simeq \cA$ is then an
honest equivalence of $\Ainf$ chiral algebras.
\end{proof}

\begin{construction}[The determinant line bundle $L_\kappa$; \ClaimStatusProvedHere]
\label{constr:determinant-line-bundle}

We construct the line bundle $\mathcal{L}_\kappa$ on $\overline{\cM}_g$ whose first Chern class is $\kappa(\cA) \cdot [\omega_g]$.

\textbf{Step 1: The Hodge bundle.}
Over $\overline{\cM}_g$, the Hodge bundle $\lambda = \pi_*(\omega_{\cC/\overline{\cM}_g})$ is a rank-$g$ vector bundle whose fiber at a smooth curve $[C]$ is $H^0(C, \omega_C)$ (the space of holomorphic differentials). Its determinant is the line bundle $\det(\lambda) = \Lambda^g(\lambda)$.

\textbf{Step 2: The Mumford isomorphism.}
The first Chern class of the Hodge bundle is $c_1(\lambda) = \omega_g \in H^2(\overline{\cM}_g, \Z)$, where $\omega_g$ is the Hodge class (the first tautological class $\lambda_1$). By Mumford's formula, $c_1(\det \lambda) = \lambda_1 = \omega_g$.

\textbf{Step 3: The $\kappa$-th power.}
Define:
\[
 L_\kappa \;:=\; (\det \lambda)^{\otimes \kappa(\cA)}.
\]
This is the $\kappa$-th tensor power of the determinant of the Hodge bundle. Its first Chern class is:
\[
 c_1(L_\kappa) = \kappa(\cA) \cdot c_1(\det \lambda) = \kappa(\cA) \cdot \omega_g.
\]

\textbf{Step 4: The connections and curvature.}
The Hodge bundle carries two natural connections. The \emph{Gauss--Manin connection} $\nabla^{\mathrm{GM}}$ is flat ($F^{\mathrm{GM}} = 0$): its flat sections are the period vectors $\int_{\gamma_i} \omega_j$, encoding the variation of Hodge structure. The \emph{Chern connection} $\nabla^{\mathrm{Q}}$ on $\det(\lambda)$, induced by the Quillen metric (the $L^2$ metric regularised via analytic torsion), has non-trivial curvature given by the Wolpert--Quillen formula:
\[
 F^{\mathrm{Q}} \;=\; -\frac{2\pi i}{12} \cdot \omega_{\mathrm{WP}},
\]
where $\omega_{\mathrm{WP}}$ is the Weil--Petersson form on $\overline{\cM}_g$. Its first Chern class is $c_1(\det\lambda) = \frac{i}{2\pi}F^{\mathrm{Q}} = \frac{1}{12}\omega_{\mathrm{WP}} = \omega_g$ (the Hodge class $\lambda_1$, which equals $\frac{1}{12}\omega_{\mathrm{WP}}$ by Wolpert's theorem). The connection on $L_\kappa = (\det\lambda)^{\otimes\kappa}$ is $\nabla_{L_\kappa} = \kappa \cdot \nabla^{\mathrm{Q}}$, with curvature:
\[
 F_{L_\kappa}
 \;=\; \kappa(\cA) \cdot F^{\mathrm{Q}}
 \;=\; -\frac{2\pi i}{12} \cdot \kappa(\cA) \cdot \omega_{\mathrm{WP}},
\]
and $c_1(L_\kappa) = \frac{i}{2\pi} F_{L_\kappa} = \kappa(\cA) \cdot \omega_g$ as claimed.

\textbf{Step 5: Integrality and the affine lineage.}
When $\kappa(\cA) \in \Z$ (as for the Heisenberg algebra at
integer level, where $\kappa(\cH_k) = k$),
$L_\kappa = (\det\lambda)^{\otimes\kappa}$ is a
genuine line bundle: the tensor power of a line bundle by an
integer is well-defined in
$\mathrm{Pic}(\overline{\cM}_g)$. For the affine lineage
$\cA = \hat{\fg}_k$, the modular characteristic
$\kappa = \dim(\fg)(k+h^\vee)/(2h^\vee)$ is
generally non-integral; the Verlinde line bundle
$(\det\lambda)^{\otimes k}$ at level~$k$ is a different object
(its sections over $\overline{\cM}_g$ are the partition functions
of the WZW model, and $k$ is the level, not the modular
characteristic). For general $\kappa(\cA) \in \mathbb{k}$, $L_\kappa$ is
defined as a formal object in
$\mathrm{Pic}(\overline{\cM}_g) \otimes_{\Z} \mathbb{k}$: the
formula $c_1(L_\kappa) = \kappa(\cA) \cdot \omega_g$ determines a
class in $H^2(\overline{\cM}_g, \mathbb{k})$ but does not
correspond to an honest line bundle unless $\kappa(\cA)$ is
integral.
\end{construction}

\begin{remark}[The genus tower as a filtration of the coderived
 category]
\label{rem:genus-tower-coderived}
The genus tower of Volume~I organizes the curved bar complex by
genus:
\[
 \barB^{\mathrm{curv}}(\cA)
 \;=\;
 \barB^{(0)}(\cA)
 \;\xleftarrow{\;\kappa \cdot \omega_1\;}\;
 \barB^{(1)}(\cA)
 \;\xleftarrow{\;\kappa \cdot \omega_2\;}\;
 \cdots
\]
At genus~$0$, $\barB^{(0)}(\cA)$ is an honest dg coalgebra (the
uncurved bar complex). The passage to genus~$1$ introduces the
curvature $\kappa \cdot \omega_1$, promoting the object from the
derived to the coderived category. Each subsequent genus level
introduces higher obstructions in
$H^*(\overline{\mathcal M}_g, \mathbb{k})$, refining the
first Chern class of the line bundle $\mathcal{L}_\kappa$. The full curved bar complex is
the inverse limit of this tower; its home is the completed coderived
category $\hat{D}^{\mathrm{co}}(\mathrm{CoAlg}_{\SCchtop})$,
filtered by genus.

This filtration is the algebraic expression of the physical genus
expansion: the perturbative ($g = 0$) sector lives in the derived
category and is governed by tree-level Feynman diagrams; the
non-perturbative ($g \geq 1$) sector lives in the coderived category
and is governed by loop corrections, with each genus contributing a
new curvature obstruction. The passage from derived to coderived is
the passage from trees to loops.
\end{remark}

\begin{theorem}[Tree-level KK transfer;
 \ClaimStatusProvedElsewhere]
\label{thm:tree-level-kk}
\index{Kaluza--Klein transfer!tree-level|textbf}
The Kaluza--Klein reduction of $5$d holomorphic Chern--Simons on
$\C \times S^3$ to a $3$d HT theory on $\C \times \R$ is
controlled at tree level by the $A_\infty$ structure on
$H^{b_0, \bullet}(S^3)$ obtained by homotopy transfer from the dg
algebra $\Omega^{0,\bullet}(S^3)$. The transferred products
$m_2^{\mathrm{tree}}, m_3^{\mathrm{tree}}, \ldots$ are computed by
tree-level Feynman diagrams on~$S^3$.
\end{theorem}

\begin{conjecture}[Non-planar KK transfer]
\label{conj:nonplanar-kk}
\index{Kaluza--Klein transfer!non-planar|textbf}
The full non-planar structure \textup{(}at all loop orders in
$1/N$\textup{)} is controlled by a \emph{deformed} $A_\infty$
structure on $H^{b_0, \bullet}(S^3)$, where the deformation
parameter is $\varepsilon_1 \sim 1/N$
\textup{(}the $\Omega$-background parameter\textup{)}. The
deformed operations $m_k^{(g)}$ at genus~$g$ are the $g$-loop
corrections to the KK transfer. The identification of these
deformed operations with the known non-planar corrections to the
twisted holographic OPE is a checkable prediction of the
programme.
\end{conjecture}

\begin{remark}[Scope]
\label{rem:kk-transfer-scope}
Theorem~\ref{thm:tree-level-kk} is standard
\textup{(}Costello--Li, Costello--Gaiotto\textup{)}.
Conjecture~\ref{conj:nonplanar-kk} is the programme's own
construction: it uses the general theory of $A_\infty$ algebras
in families to produce the deformed structure, but the
identification of the deformation with the physical non-planar
corrections has not been verified by explicit computation at any
loop order beyond tree level.
\end{remark}
