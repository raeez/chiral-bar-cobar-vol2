\chapter{The Koszulness Moduli Scheme $M_{\mathrm{Kosz}}$}
\label{chap:koszulness-moduli}
\index{Koszulness Moduli Scheme|textbf}
\index{$M_{\mathrm{Kosz}}$}
\index{GRT torsor}

Fix a bicoloured primitive package $\primPkgBicolour$ on the log curve
$\logCurve = (X, D, \tau)$ at non-critical level, conilpotent-complete,
boundary-linear, and write $A_b = \mathrm{End}_{\openFactCat}(b)$ for the
open-colour endomorphism chiral algebra. Every family in the standard
landscape — affine Kac--Moody at non-critical level, $\mathcal{W}_N$,
Heisenberg/lattice, Virasoro, free-field — is Koszul on its own
Drinfeld-associator chart; Koszulness is the existence of a Maurer--Cartan
bijection between the chiral bar adjunction $\Omega^{\mathrm{ch}} \dashv
\BarTwc{\cdot}$ evaluated on $(A_b, A_b^{!})$, in the sense of
\effKoszul-effectiveness (cf.\ slogan~17 of the bicoloured voice
table: quadratic dual is an injection in general; the bijection requires
chart-local effectiveness). Koszulness is \emph{not} SC$^{\mathrm{ch,top}}$
formality: the latter is a Kontsevich-stratum statement on the
two-coloured Swiss-cheese-chiral-topological operad, distinct from
chart-local Koszulness at $A_b$.
\par
Fourteen classical characterizations (Positselski, Frenkel--Gaitsgory,
Arinkin--Gaitsgory, Francis--Gaitsgory, Hoefel--Livernet, Kontsevich,
Tamarkin, Willwacher, Vallette, Fresse, Loday--Vallette, and four
Vol~I-specific) sit on fourteen affine charts of the
$\mathrm{GRT}_1(\mathbb{Q})$-equivariant Koszulness Moduli Scheme
$M_{\mathrm{Kosz}}(A_b)$. The partition into ``unconditional'' and
``scoped'' is coordinate-chart ambiguity under associator choice: each
characterization is unconditional on its own chart and acquires scope
conditions only across non-rational chart transitions.

\section{Definition of $M_{\mathrm{Kosz}}(A_b)$}
\label{sec:mkosz-definition}

\begin{definition}[Koszulness Moduli Scheme]
\label{def:mkosz}
For $A_b = \mathrm{End}_{\openFactCat}(b)$ conilpotent-complete on
$\logCurve$, the Koszulness Moduli Scheme is the functor
\[
 M_{\mathrm{Kosz}}(A_b) :
 \mathrm{CAlg}_{\mathbb{Q}} \to \mathrm{Set},
 \qquad
 R \mapsto
 \{\text{Koszul-duality data } (A_b, A_b^{!}, \Phi) \text{ over } R\},
\]
where $\Phi$ is a Drinfeld associator fixing the chain-level bar-cobar
duality coordinate. The group $\mathrm{GRT}_1(\mathbb{Q})$ acts on
$M_{\mathrm{Kosz}}(A_b)$ by transporting $\Phi$ through the
Grothendieck--Teichm\"uller group action on associators.
\end{definition}

\begin{remark}
$M_{\mathrm{Kosz}}(A_b)$ is a $\mathrm{GRT}_1(\mathbb{Q})$-torsor over
the quotient by associator-gauge. A $\mathbb{Q}$-point of
$M_{\mathrm{Kosz}}(A_b)$ exists if and only if $A_b$ is Koszul on some
chart; the specific choice of $\mathbb{Q}$-point names a coordinate
chart. Verdier duality determines $A_b^{!}$; the bar twisting cochain
$\BarTwc{A_b}$ and the universal-bulk Hochschild object
$\bulkChirHoch{A_b}$ are separate invariants of the package, not
conflated with $A_b^{!}$.
\end{remark}

\section{Fourteen-chart atlas}
\label{sec:mkosz-fourteen-charts}

\begin{theorem}[Fourteen-chart atlas]
\label{thm:mkosz-fourteen-charts}
The moduli scheme $M_{\mathrm{Kosz}}(A_b)$ admits a finite affine
atlas of fourteen charts:
\[
 M_{\mathrm{Kosz}}(A_b)
 \;=\;
 \bigcup_{i=1}^{14} U_i,
 \qquad
 U_i \;=\; \mathrm{Spec}\,R_i,
\]
where each chart $U_i$ corresponds to one of the fourteen classical
Koszulness characterizations. Every characterization is
unconditionally equivalent to Koszulness on its own chart; the
``scope'' of each classical characterization is its chart-domain in
$M_{\mathrm{Kosz}}(A_b)$.
\end{theorem}

\begin{proof}[Sketch]
Each characterization (cobar-is-free, bar-is-conilpotent, Quillen
equivalence, Ext-vanishing, $\mathrm{Tor}$-finiteness, etc.) is a
polynomial condition in a specific Drinfeld-associator coordinate
on $M_{\mathrm{Kosz}}(A_b)$. On its own chart the characterization is
unconditionally equivalent to Koszulness. ``Scoped'' characterizations
are non-rational transitions between charts: the associator-change
cocycle on a chart overlap introduces $\mathrm{GRT}_1$-torsor
obstructions to a universal global statement.
\end{proof}

\section{Shadow-class stratification}
\label{sec:mkosz-shadow-stratification}

\begin{theorem}[Shadow-class stratification]
\label{thm:mkosz-shadow-stratification}
$M_{\mathrm{Kosz}}(A_b)$ decomposes as a disjoint union
\[
 M_{\mathrm{Kosz}}(A_b)
 \;=\;
 \bigsqcup_{c \in \{G, L, C, M, FF\}} M_{\mathrm{Kosz}}^{(c)}(A_b),
\]
where each $M_{\mathrm{Kosz}}^{(c)}(A_b)$ is the locus on which $A_b$
has shadow class $c$. The fingerprint $\varphi'(A_b) = (p_{\max},
r_{\max}, \chi_{\mathrm{VOA}}, n_{\mathrm{strong}}, \mathrm{coset},
\kappaChHodge)$ of Theorem~\ref{thm:fingerprint-complete-ch} provides
complete coordinates on each $M_{\mathrm{Kosz}}^{(c)}(A_b)$.
\end{theorem}

\begin{proof}[Sketch]
Shadow class is an invariant of the bar-cobar Koszul-duality data,
hence locally constant on $M_{\mathrm{Kosz}}(A_b)$. The strata
$M_{\mathrm{Kosz}}^{(c)}(A_b)$ are the connected components of the
locally constant shadow class. Completeness of $\varphi'$ on each
stratum follows from Theorem~\ref{thm:fingerprint-complete-ch}.
\end{proof}

\section{Relation to the chiral Deligne-Tamarkin GRT torsor}
\label{sec:mkosz-grt-torsor}

\begin{remark}[$M_{\mathrm{Kosz}}$ and the associator-independent form]
The $\mathrm{GRT}_1(\mathbb{Q})$-action on $M_{\mathrm{Kosz}}(A_b)$ is the
Koszulness-level manifestation of the chiral Deligne--Tamarkin torsor.
By Theorem~\ref{thm:higher-massey-tower}, the Higher Massey tower
$\{\langle r^{\times(2k+1)}\rangle\}_{k \ge 1}$ obstructs
associator-independent chain-level chiral Deligne--Tamarkin at every odd
weight; the universal form is a $\mathrm{GRT}_1$-torsor of
associator-dependent statements. The fourteen charts of
$M_{\mathrm{Kosz}}(A_b)$ realise fourteen Drinfeld-associator coordinates
within this torsor.
\end{remark}

\section{Consequences}
\label{sec:mkosz-consequences}

\begin{corollary}[Fourteen characterizations unconditional on own chart]
The fourteen Koszulness characterizations of
Theorem~\ref{thm:mkosz-fourteen-charts} are unconditional on their own
Drinfeld-associator coordinate. Scope conditions are chart-transition
cocycles, not defects of the characterizations themselves.
\end{corollary}

\begin{corollary}[Classification via charts]
The five shadow classes G/L/C/M/FF are the five connected components of
$\bigcup_{A_b} M_{\mathrm{Kosz}}(A_b)$ (union over all open-colour
endomorphism algebras in the standard landscape). The fingerprint
$\varphi'(A_b)$ is a complete coordinate on each component.
\end{corollary}

\section{Cross-volume anchor: four irreducible opens}
\label{sec:mkosz-anchor}

The fourteen-chart atlas of Theorem~\ref{thm:mkosz-fourteen-charts}
hosts four irreducible opens on the non-degenerate locus, each
inhabiting a named associator chart.

\begin{itemize}
\item \emph{Curved-Dunn $H^{2}=0$ at $g\ge 2$.} The chart
 $U_{13}$ at the Enriquez elliptic associator
 $\Phi_{\mathrm{ell}}$ carries the conditional transport theorem:
 modular-bootstrap \(H^2\)-acyclicity plus the degree-\(2\)
 curved-Dunn comparison imply the vanishing
 (Theorem~\ref{thm:curved-dunn-H2-vanishing-all-genera}). The
 irregular-singular KZB theorem is the separate boundary-composition
 input under formal-type and mixed Stokes hypotheses
 (Theorem~\ref{thm:irregular-singular-kzb-regularity}).
\item \emph{DS--Hochschild compatibility for class~$\mathbf{M}$.}
 Closed via HPL transfer through Drinfeld--Sokolov strong
 deformation retract plus Arakawa $C_{2}$-cofiniteness
 (Theorem~\ref{thm:chd-ds-hochschild} +
 Corollary~\ref{cor:universal-holography-class-M}). Lives on the
 chart $U_{12}$ at the Hodge--Betti associator $\Phi_{dR,B}$.
\item \emph{Periodic-CDG admissible KL.} Closed via Arkhipov--Gaitsgory
 twisted Satake plus Finkelberg tilting semisimplification
 (Vol~I Chapter~\ref{V1-chap:periodic-cdg-admissible}). Lives on
 the chart $U_{1}$ at the KZ associator $\Phi_{\mathrm{KZ}}$.
\item \emph{Chiral Deligne--Tamarkin chain-level associator-dependence.}
 Closed after fixing a Drinfeld associator
 (Theorem~\ref{thm:chd-deligne-tamarkin}). Lives on the chart
 $U_{14}$ at the Kontsevich associator $\Phi_{\mathrm{Kon}}$.
\end{itemize}

The five open frontiers F1--F5 of the climax
(\cref{conj:f1-wp-tempering}--\cref{conj:f5-irregular-singular-genus-g})
correspond to non-rational chart-transition cocycles within
$M_{\mathrm{Kosz}}$:

\begin{itemize}
\item F1 (logarithmic $\mathcal{W}(p)$ tempering): a non-rational
 transition between the non-logarithmic chart $U_{1}$ and a
 conjectural logarithmic chart $U_{1, \log}$; closure requires
 the Adamovi\'c--Milas character-amplitude bound.
\item F2 (periodic-CDG non-simply-laced): a non-rational
 transition refining $U_{1}$ from simply-laced to non-simply-laced
 type; closure requires Arkhipov--Gaitsgory twisted Satake plus
 Brylinski--Deligne reciprocity.
\item F3 (intermediate-topology chain-level class~$\mathbf{M}$): a
 refinement of $U_{12}$ from the full weight-completed category to an
 explicitly topologized analytic model stricter than that completion;
 the raw direct-sum bar is excluded because the generic class-M
 direct-sum statement is false.
\item F4 (super-complementarity canonical pairing): a refinement
 of the Koszul-dual scalar coordinate on every chart, requiring a
 single bilinear pairing in place of the coexisting super-trace
 and Berezinian pairings.
\item F5 (irregular-singular KZB at
 $\overline{\mathcal{M}}_{g,n}$ boundaries for $g \ge 1$):
 the boundary-stratum extension of $U_{13}$, currently established
 on the smooth locus only.
\end{itemize}

\begin{remark}[Atlas boundary questions]
\label{rem:mkosz-atlas-boundary-questions}
On the non-degenerate locus each chart in the atlas
$\{U_{j}\}_{j=1}^{14}$ is specified by its defining equations; the
five boundary questions F1--F5 are explicit chart-transition or
sub-chart questions on $M_{\mathrm{Kosz}}(A_b)$.
\end{remark}
