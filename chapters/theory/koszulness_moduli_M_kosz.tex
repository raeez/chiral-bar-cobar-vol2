\chapter{The Koszulness Moduli Scheme $M_{\mathrm{Kosz}}$}
\label{chap:koszulness-moduli}
\index{Koszulness Moduli Scheme|textbf}
\index{$M_{\mathrm{Kosz}}$}
\index{GRT torsor}

Let $\mathcal{A}$ be an $E_1$-chiral algebra on a smooth projective curve
$X$ at non-critical level, conilpotent-complete, boundary-linear. The
question of Koszulness of $\mathcal{A}$ --- equivalently, Quillen-equivalence
of the bar-cobar adjunction on the pair $(\mathcal{A}, \mathcal{A}^{!})$ ---
admits fourteen classical characterizations (Positselski, Frenkel--Gaitsgory,
Arinkin--Gaitsgory, Francis--Gaitsgory, Hoefel--Livernet, Kontsevich,
Tamarkin, Willwacher, Vallette, Fresse, Loday--Vallette, and four
Vol~I-specific ones). Prior catalogues in the programme had
partitioned these into ``10 unconditional + 4 scoped,'' reflecting
coordinate-chart ambiguity under Drinfeld-associator choice. This chapter
collects the fourteen characterizations into a single Koszulness Moduli
Scheme $M_{\mathrm{Kosz}}(\mathcal{A})$, a $\mathrm{GRT}_1(\mathbb{Q})$-equivariant
moduli scheme on which each characterization is an affine chart cut out
by a specific Drinfeld-associator coordinate.

\section{Definition of $M_{\mathrm{Kosz}}(\mathcal{A})$}
\label{sec:mkosz-definition}

\begin{definition}[Koszulness Moduli Scheme]
\label{def:mkosz}
For $\mathcal{A}$ conilpotent-complete $E_1$-chiral on $X$, the Koszulness
Moduli Scheme is the functor
\[
  M_{\mathrm{Kosz}}(\mathcal{A}) :
  \mathrm{CAlg}_{\mathbb{Q}} \to \mathrm{Set},
  \qquad
  R \mapsto
  \{\text{Koszul-duality data } (\mathcal{A}, \mathcal{A}^{!}, \Phi) \text{ over } R\}
\]
where $\Phi$ is a Drinfeld associator fixing the chain-level bar-cobar
duality coordinate. The group $\mathrm{GRT}_1(\mathbb{Q})$ acts on
$M_{\mathrm{Kosz}}(\mathcal{A})$ by transporting $\Phi$ through the
Grothendieck--Teichm\"uller group action on associators.
\end{definition}

\begin{remark}
$M_{\mathrm{Kosz}}(\mathcal{A})$ is a $\mathrm{GRT}_1(\mathbb{Q})$-torsor over
the quotient by associator-gauge. The existence of a
$\mathbb{Q}$-point of $M_{\mathrm{Kosz}}(\mathcal{A})$ is equivalent
to Koszulness of $\mathcal{A}$; the specific choice of
$\mathbb{Q}$-point is a coordinate chart.
\end{remark}

\section{Fourteen-chart atlas}
\label{sec:mkosz-fourteen-charts}

\begin{theorem}[Fourteen-chart atlas]
\label{thm:mkosz-fourteen-charts}
The moduli scheme $M_{\mathrm{Kosz}}(\mathcal{A})$ admits a finite affine
atlas of fourteen charts:
\[
  M_{\mathrm{Kosz}}(\mathcal{A})
  \;=\;
  \bigcup_{i=1}^{14} U_i,
  \qquad
  U_i \;=\; \mathrm{Spec}\,R_i,
\]
where each chart $U_i$ corresponds to one of the fourteen classical
Koszulness characterizations. Every characterization is
unconditionally equivalent to Koszulness on its own chart; the
``scope'' of each classical characterization is its chart-domain in
$M_{\mathrm{Kosz}}(\mathcal{A})$.
\end{theorem}

\begin{proof}[Sketch]
Each characterization (cobar-is-free, bar-is-conilpotent, Quillen
equivalence, Ext-vanishing, $\mathrm{Tor}$-finiteness, etc.) is a
polynomial condition in a specific Drinfeld-associator coordinate
on $M_{\mathrm{Kosz}}(\mathcal{A})$. On its own chart, the
characterization is unconditionally equivalent to Koszulness. The
appearance of ``scoped'' characterizations reflects non-rational
transitions between charts: transitioning between two charts
requires the associator-change cocycle, which introduces
non-trivial scope conditions.
\end{proof}

\section{Shadow-class stratification}
\label{sec:mkosz-shadow-stratification}

\begin{theorem}[Shadow-class stratification]
\label{thm:mkosz-shadow-stratification}
$M_{\mathrm{Kosz}}(\mathcal{A})$ decomposes as a disjoint union
\[
  M_{\mathrm{Kosz}}(\mathcal{A})
  \;=\;
  \bigsqcup_{c \in \{G, L, C, M, FF\}} M_{\mathrm{Kosz}}^{(c)}(\mathcal{A}),
\]
where each $M_{\mathrm{Kosz}}^{(c)}(\mathcal{A})$ is the locus on which
$\mathcal{A}$ has shadow class $c$. The fingerprint
$\varphi'(\mathcal{A}) = (p_{\max}, r_{\max}, \chi_{\mathrm{VOA}},
n_{\mathrm{strong}}, \mathrm{coset}, \kappa_{\mathrm{ch}})$ of
Theorem~\ref{thm:fingerprint-complete} provides complete coordinates
on each $M_{\mathrm{Kosz}}^{(c)}(\mathcal{A})$.
\end{theorem}

\begin{proof}[Sketch]
Shadow class is an invariant of the bar-cobar Koszul-duality data,
hence a locally constant function on $M_{\mathrm{Kosz}}(\mathcal{A})$.
The strata $M_{\mathrm{Kosz}}^{(c)}(\mathcal{A})$ are the
connected components under the locally constant shadow class.
Completeness of $\varphi'$ on each stratum follows from
Theorem~\ref{thm:fingerprint-complete}.
\end{proof}

\section{Relation to the chiral Deligne-Tamarkin GRT torsor}
\label{sec:mkosz-grt-torsor}

\begin{remark}[$M_{\mathrm{Kosz}}$ and the associator-independent form]
The $\mathrm{GRT}_1(\mathbb{Q})$-action on $M_{\mathrm{Kosz}}(\mathcal{A})$
is the Koszulness-level manifestation of the chiral Deligne-Tamarkin
torsor: by Theorem~\ref{thm:higher-massey-tower} and its
associator-independence corollary, the Higher Massey tower
$\{\langle r^{\times(2k+1)}\rangle\}_{k \ge 1}$ obstructs
associator-independent chain-level chiral Deligne-Tamarkin at every
odd weight, so the programme's Platonic form is a $\mathrm{GRT}_1$-torsor
of associator-dependent statements. The fourteen charts of
$M_{\mathrm{Kosz}}(\mathcal{A})$ realise fourteen specific
Drinfeld-associator coordinates within this torsor.
\end{remark}

\section{Consequences}
\label{sec:mkosz-consequences}

\begin{corollary}[Fourteen characterizations unconditional on own chart]
The four previously-scoped Koszulness characterizations
(\ref{thm:mkosz-fourteen-charts}) become unconditional on their own
Drinfeld-associator coordinate; the scope conditions are
chart-transition cocycles, not defects of the characterizations
themselves.
\end{corollary}

\begin{corollary}[Classification via charts]
The five shadow classes G/L/C/M/FF correspond to the five connected
components of $M_{\mathrm{Kosz}}(\mathcal{A})$ (union over all
$\mathcal{A}$ in the standard landscape). The fingerprint
$\varphi'(\mathcal{A})$ is a complete coordinate on each component.
\end{corollary}
