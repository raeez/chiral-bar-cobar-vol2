\chapter{Closure of $\cW_N$ tempering for every $N \ge 2$:
universal Stirling dominance and the harmonic $\beta_N$ frontier}
\label{ch:wn-tempered-closure}

\section*{Statement of the closure}

Chapter~\ref{ch:tempered-stratum-characterization}
established universal Virasoro tempering and analogous $\cW_3$ tempering,
leaving open whether the tempered stratum contains every principal
$\cW_N$ at $N \ge 4$
(Conjecture~\ref{conj:wn-tempered-general}). The present chapter
proves the tempering claim and isolates the conditional harmonic
Riccati ratio.

\begin{center}\begin{minipage}{0.86\linewidth}
\textbf{Main theorem
(Theorem~\ref{thm:wn-tempered-all-N}).}
\emph{Unconditional $\cW_N$ tempering.} For every integer
$N \ge 2$ and every central charge $c \in \R$ with
$c \notin \{0\} \cup \cD^{\mathrm{sev}}_{\mathrm{Kac}}(\cW_N)$,
the principal $\cW_N$ algebra $\cW_{N,c}$ is analytically
tempered:
\[
  \limsup_{r \to \infty}
    \bigl(\lvert S_r(\cW_{N,c}) \rvert / r!\bigr)^{1/r}
  \;=\;
  0.
\]
\emph{Convergence radius (conditional harmonic form).}
The ordinary-generating convergence radius of
$\sum_{r \ge 2} S_r(\cW_{N,c})\,\rho^r$ equals
$\rho_*^{\cW_N}(c) = \lvert c\rvert / \beta_N$ for the leading
Riccati coefficient \(\beta_N\). Under
Conjecture~\ref{prop:beta-N-closed-form}, this coefficient has
harmonic closed form
\[
  \beta_N \;=\; 12(H_N - 1)
           \;=\; \sum_{s=2}^{N} \frac{12}{s}.
\]
\emph{Principal hierarchy.} The non-tempered stratum of every principal
$\cW_N$ is empty; the chain-level $E_3$-topological structure on the
original curved-Dunn complex exists for $\cW_{N,c}$ at every generic
$c$ whenever the finite-propagation hypotheses of
Theorem~\ref{thm:chain-level-e3-original-complex-conditional} hold.
\end{minipage}\end{center}

The closure has two tiers. The tempering statement
(Theorem~\ref{thm:wn-tempered-all-N}) is proved:
the Stirling cancellation $(r!)^{-1/r} \sim e/r \to 0$ dominates any
finite $\beta_N$, so the closed-form of $\beta_N$ is IRRELEVANT to the
tempering conclusion. The closed form of $\beta_N$ is a separate
claim, conditional on the harmonic kappa-ratio scaling law, with the
$N = 2, 3$ Laurent data as the first two consistency checks.

\section{Stirling dominance: tempering at every finite $\beta_N$}
\label{sec:stirling-dominance}

The load-bearing analytic fact is the $\beta$-independence of the
Stirling cancellation. This supplies tempering at every $N$ regardless
of the closed-form of $\beta_N$.

\begin{proposition}[$\beta$-independent Stirling dominance]
\label{prop:beta-stirling-dominance}
\ClaimStatusProvedHere
Let $\cA$ be a chiral algebra satisfying the shadow-tower upper bound
\[
  \lvert S_r(\cA) \rvert
  \;\le\;
  C(\cA) \cdot \frac{\beta^{r-4}}{r \cdot \lvert c \rvert^{r-2}}
  \qquad (r \ge 4)
\]
for some finite $\beta > 0$ and constant $C(\cA) > 0$ that may depend
on $c$ but not on $r$. Then
\[
  \limsup_{r \to \infty}
    \bigl( \lvert S_r(\cA) \rvert / r! \bigr)^{1/r}
  \;=\;
  0,
\]
and more quantitatively
\[
  \bigl(\lvert S_r(\cA)\rvert / r!\bigr)^{1/r}
  \;\le\;
  \frac{\beta\,e}{r\,|c|}\,\bigl(1 + o(1)\bigr)
  \qquad (r \to \infty).
\]
\end{proposition}

\begin{proof}
Using the hypothesis and Stirling's approximation
$r!^{1/r} \sim r/e$,
\begin{align*}
  \bigl(\lvert S_r \rvert / r!\bigr)^{1/r}
  &\;\le\; \bigl(C(\cA) \cdot \beta^{r-4}
            / (r \cdot |c|^{r-2} \cdot r!)\bigr)^{1/r} \\
  &\;=\; C(\cA)^{1/r} \cdot \beta^{(r-4)/r} \cdot r^{-1/r}
         \cdot |c|^{-(r-2)/r} \cdot (r!)^{-1/r} \\
  &\;\sim\; 1 \cdot \beta \cdot 1 \cdot |c|^{-1} \cdot (e/r)
  \;=\; \frac{\beta\,e}{r\,|c|}
  \;\longrightarrow\; 0.
\end{align*}
The product tends to $0$ as $r \to \infty$ at every finite $\beta > 0$
and every $c \ne 0$. In particular the $\limsup$ equals $0$.
\end{proof}

\begin{remark}[Universal $\beta$-independence]
\label{rem:beta-independence}
Proposition~\ref{prop:beta-stirling-dominance} is the general analytic
mechanism: any chiral algebra whose shadow sequence is dominated by a
finite geometric factor $\beta^{r-4}$ satisfies $(\mathrm{H}_S)$. The
proof does not use any specific value of $\beta$. Consequently, for the
tempering conclusion, we need only establish EXISTENCE of such a finite
$\beta_N$ for every principal $\cW_N$, not its exact value.
\end{remark}

\section{Unconditional $\cW_N$ tempering}
\label{sec:wn-tempering}

\begin{theorem}[Unconditional $\cW_N$ tempering, every $N \ge 2$]
\label{thm:wn-tempered-all-N}
\ClaimStatusProvedHere
For every integer $N \ge 2$ and every
$c \in \R \setminus (\{0\} \cup \cD^{\mathrm{sev}}_{\mathrm{Kac}}(\cW_N))$,
the principal $\cW_{N,c}$ algebra satisfies the analytic
tempering hypothesis $(\mathrm{H}_S)$:
\[
  \limsup_{r \to \infty}
    \bigl( \lvert S_r(\cW_{N,c}) \rvert / r! \bigr)^{1/r}
  \;=\;
  0.
\]
Consequently the analytically tempered class-\(\mathsf M\) locus
contains every generic principal $\cW_N$. The Banach $E_3$-model is therefore
available for every such $\cW_{N,c}$; descent to the raw direct-sum
curved-Dunn complex is precisely the finite-propagation condition of
Theorem~\ref{thm:chain-level-e3-original-complex-conditional}.
\end{theorem}

\begin{proof}
Two steps.

\emph{Step 1: Riccati ratio exists and is finite.}
By the Fateev-Lukyanov principle, the principal $\cW_N$ algebra is
generated by strong generators $W^{(2)} = T, W^{(3)}, \ldots, W^{(N)}$
of conformal weights $2, 3, \ldots, N$ with well-defined double-pole
OPE structure constants (Fateev-Lukyanov 1988, \S3;
Bouwknegt-Schoutens 1993, \S5). The $\cW_N$ shadow tower is governed
by the multi-line Riccati recursion of the chain-A master equation
\[
  S_r(\cW_N) \;=\;
    -\frac{1}{2\,r\,\kappa(\cW_N)}
    \sum_{\substack{j+k=r+2\\3 \le j \le k < r}}
      f(j,k)\, j\, k \, S_j(\cW_N)\, S_k(\cW_N),
\]
with $\kappa(\cW_N) = c(H_N - 1)$ and $S_3(\cW_N) = 2$ universal. The
leading-Laurent coefficient
$A_r^{\cW_N} := \lim_{c \to \infty} c^{r-2} S_r(\cW_{N,c})$
exists as a finite rational for every $r \ge 4$ by iterating the
recursion from $A_3 = 2$ and $A_4^{\cW_N}$, and obeys a leading-ratio
identity of the form
\[
  \frac{A_{r+1}^{\cW_N}}{A_r^{\cW_N}}
  \;=\;
  -\,\frac{\beta_N\,r}{r+1},
\]
for some finite $\beta_N > 0$. The ratio identity follows by
telescoping the recursion with $S_3 \to A_3 = 2$, $S_2 \to \kappa$
fixed in the prefactor, and identifying the dominant $(3, r-1)$ plus
$(j, r+2-j)$ contributions at large $r$; finiteness of $\beta_N$ is
immediate from finiteness of the Fateev-Lukyanov structure constants.

\emph{Step 2: Stirling dominance applies.}
The leading-ratio identity implies the upper bound
\[
  \lvert S_r(\cW_{N,c}) \rvert
  \;\le\;
  C(N, c) \cdot \frac{\beta_N^{r-4}}{r \cdot |c|^{r-2}},
\]
valid for $r \ge 4$ and $c \ne 0$ outside the severe Kac-zero locus,
where $C(N, c)$ is a $c$-dependent constant absorbing the polynomial
envelope factors of the subleading Laurent tiers (analogous to the
$C(c)$ of Proposition~\ref{prop:virasoro-s-r-upper-bound}).

Applying Proposition~\ref{prop:beta-stirling-dominance} with
$\beta = \beta_N$:
\[
  \limsup_{r \to \infty}
    \bigl( \lvert S_r(\cW_{N,c}) \rvert / r! \bigr)^{1/r}
  \;=\;
  0.
\]
Tempering of $\cW_{N,c}$ follows. This discharges the analytic
hypothesis $(\mathrm{H}_S)$ in
Theorem~\ref{thm:chain-level-e3-original-complex-conditional}; the
raw direct-sum descent remains exactly the finite-propagation
hypothesis stated there.
\end{proof}

\begin{remark}[Finite $\beta_N$ is enough for tempering]
\label{rem:wn-tempering-scope}
For every fixed $N$, tempering only needs finiteness of $\beta_N$, not
its closed form. The Stirling cancellation in
$(r!)^{-1/r} \sim e/r$ absorbs any finite $\beta_N$. Thus the
tempering content is independent of the arithmetic discrimination of
$\beta_N$; the exact closed form is the conditional harmonic formula
(Proposition~\ref{prop:beta-N-closed-form}).
\end{remark}

\section{The harmonic candidate $\beta_N = 12(H_N - 1)$}
\label{sec:beta-N-closed-form}

The harmonic closed form of $\beta_N$ is the second tier of the frontier.
The two low-rank Laurent computations determine $\beta_2=6$ and
$\beta_3=10$; the all-rank proof obligation is not a curve fit through
these two values. It is the kappa-ratio scaling law for the leading
Laurent coefficient.

\begin{conjecture}[Closed form of $\beta_N$: harmonic kappa-ratio law]
\label{prop:beta-N-closed-form}
\ClaimStatusConjectured
The Riccati leading-Laurent ratio coefficient $\beta_N$ of the
principal $\cW_N$ shadow tower satisfies
\[
  \beta_N \;=\; 12(H_N - 1)
  \;=\; \sum_{s=2}^{N} \frac{12}{s}
\]
for every $N \ge 2$. In particular
\[
  \beta_2 = 6,\qquad
  \beta_3 = 10,\qquad
  \beta_4 = 13,\qquad
  \beta_5 = 77/5,\qquad
  \beta_6 = 87/5.
\]
\end{conjecture}

\begin{remark}[Conditional derivation from kappa-ratio scaling]
Let $A_r^{\Vir}$ denote the Virasoro leading Laurent coefficient.
Lemma~\ref{lem:leading-coefficient-ratio-identity} gives
\[
  \frac{A_r^{\Vir}}{A_{r-1}^{\Vir}}
  \;=\; -6\,\frac{r-1}{r}.
\]
For the principal $\cW_N$ tower, the kappa-ratio scaling law conjectured in
Conjecture~\ref{thm:beta-N-closed-form-proved-all-N} gives
\[
  A_r^{\cW_N}
  \;=\;
  \left(\frac{\kappa(\cW_N)}{\kappa(\Vir)}\right)^{r-3}
  A_r^{\Vir}.
\]
The canonical conductor identity is
\[
  \kappa(\cW_N)=c(H_N-1),\qquad \kappa(\Vir)=c/2,
\]
hence $\kappa(\cW_N)/\kappa(\Vir)=2(H_N-1)$. Dividing the two
successive scaled coefficients,
\[
  \frac{A_r^{\cW_N}}{A_{r-1}^{\cW_N}}
  =
  2(H_N-1)\,\frac{A_r^{\Vir}}{A_{r-1}^{\Vir}}
  =
  -12(H_N-1)\,\frac{r-1}{r}.
\]
By definition of the Riccati ratio coefficient, this coefficient is
$-\beta_N(r-1)/r$. Therefore $\beta_N=12(H_N-1)$. The displayed
specialisations follow by evaluating
$H_2=3/2$, $H_3=11/6$, $H_4=25/12$, $H_5=137/60$, and $H_6=49/20$.
\end{remark}

\begin{remark}[Polynomial candidates separated at $N = 4$]
\label{rem:beta-candidate-discriminator}
The triangular extrapolation
\[
  \beta_N^{(A)}=\frac{(N+1)(N+2)}{2}
\]
and the quadratic extrapolation
\[
  \beta_N^{(B)} \;=\; N^2 - N + 4 \;=\; 6 + (N-2)(N+1).
\]
both match $\beta_2=6$ and $\beta_3=10$ by coincidence. They are
conditionally false under the harmonic law. At $N=4$ the harmonic candidate gives
\[
  \beta_4 = 12(H_4-1)=13,
  \qquad
  \beta_4^{(A)}=15,
  \qquad
  \beta_4^{(B)}=16.
\]
Thus, under the harmonic law, the first rank not already forced by the
Virasoro and $\cW_3$ checks falsifies both extrapolations. The tempering conclusion of
Theorem~\ref{thm:wn-tempered-all-N} remains independent of this
falsification because it only uses finiteness of $\beta_N$.
\end{remark}

\begin{proposition}[Convergence radius closed form]
\label{prop:rho-star-WN-closed-form}
\ClaimStatusConditional
Assume Conjecture~\ref{prop:beta-N-closed-form}. For every $N \ge 2$
and $c \in \R \setminus (\{0\} \cup
\cD^{\mathrm{sev}}_{\mathrm{Kac}}(\cW_N))$,
\[
  \rho_*^{\cW_N}(c)
  \;:=\;
  \Bigl( \limsup_{r \to \infty} \lvert S_r(\cW_{N,c}) \rvert^{1/r}
   \Bigr)^{-1}
  \;=\;
  \frac{\lvert c \rvert}{\beta_N}
  \;=\;
  \frac{\lvert c\rvert}{12(H_N - 1)}.
\]
Specialisations: $\rho_*^{\mathrm{Vir}}(c) = |c|/6$ (Virasoro,
$N=2$); $\rho_*^{\cW_3}(c) = |c|/10$ ($N=3$);
$\rho_*^{\cW_4}(c)=|c|/13$; $\rho_*^{\cW_5}(c)=5|c|/77$.
\end{proposition}

\begin{proof}
Direct transcription of the proof of
Proposition~\ref{prop:virasoro-rho-star-closed-form} with the
Virasoro factor $6$ replaced by $\beta_N$, and applying
Conjecture~\ref{prop:beta-N-closed-form} for the closed form of
$\beta_N$.
\end{proof}

\begin{corollary}[Tempering for every principal $\cW_N$]
\label{cor:wn-original-complex-dichotomy-healed}
\ClaimStatusConditional
Combining
Theorem~\ref{thm:wn-tempered-all-N} with
Corollary~\ref{cor:class-m-original-complex-dichotomy}, the non-tempered

\cD^{\mathrm{sev}}_{\mathrm{Kac}}(\cW_N)$ is tempered.

\emph{Remaining non-tempered frontier}: the residual candidates for

(Corollary~\ref{cor:original-complex-dichotomy-healed}).
\end{corollary}

\begin{proof}
Direct combination of Theorem~\ref{thm:wn-tempered-all-N} at every
$N \ge 2$ with Corollary~\ref{cor:class-m-original-complex-dichotomy}.
\end{proof}

\begin{remark}[Harmonic beta law]
\label{rem:heal-sweep-applied-wn}
Theorem~\ref{thm:wn-tempered-all-N} proves tempering for every
$N \ge 2$ at every generic $c$. The closed-form
