\chapter{Closure of $\cW_N$ tempering for every $N \ge 2$:
universal Stirling dominance and the harmonic $\beta_N$ frontier}
\label{ch:wn-tempered-closure}

\section*{Statement of the closure}

Chapter~\ref{ch:tempered-stratum-characterization}
established universal Virasoro tempering and analogous $\cW_3$ tempering,
leaving open whether the tempered stratum contains every principal
$\cW_N$ at $N \ge 4$
(Conjecture~\ref{conj:wn-tempered-general}). The present chapter
proves the finite-\(\beta_N\) Stirling implication and isolates the
two remaining inputs: the finite Riccati envelope for all ranks and the
conditional harmonic Riccati ratio.

\begin{center}\begin{minipage}{0.86\linewidth}
\textbf{Main theorem
(Theorem~\ref{thm:wn-tempered-all-N}).}
\emph{Finite-\(\beta_N\) criterion.} Fix an integer
$N \ge 2$ and a central charge $c \in \R$ with
$c \notin \{0\} \cup \cD^{\mathrm{sev}}_{\mathrm{Kac}}(\cW_N)$.
Assume that the principal $\cW_N$ shadow tower has a finite Riccati
envelope
\[
 \lvert S_r(\cW_{N,c}) \rvert
 \;\le\;
 C(N,c)\,\frac{\beta_N^{r-4}}{r\,|c|^{r-2}}
 \qquad (r\ge 4)
\]
for some finite $\beta_N>0$. Then $\cW_{N,c}$ is analytically tempered:
\[
 \limsup_{r \to \infty}
 \bigl(\lvert S_r(\cW_{N,c}) \rvert / r!\bigr)^{1/r}
 \;=\;
 0.
\]
\emph{Convergence radius (conditional harmonic form).}
The ordinary-generating convergence radius of
$\sum_{r \ge 2} S_r(\cW_{N,c})\,\rho^r$ equals
$\rho_*^{\cW_N}(c) = \lvert c\rvert / \beta_N$ for the leading
Riccati coefficient \(\beta_N\). Under
Conjecture~\ref{prop:beta-N-closed-form}, this coefficient has
harmonic closed form
\[
 \beta_N \;=\; 12(H_N - 1)
 \;=\; \sum_{s=2}^{N} \frac{12}{s}.
\]
\emph{Principal hierarchy.} The non-tempered stratum of a principal
$\cW_N$ is empty on every locus where the finite Riccati envelope is
proved; the chain-level $E_3$-topological structure on the original
curved-Dunn complex exists for such $\cW_{N,c}$ whenever the
finite-propagation hypotheses of
Theorem~\ref{thm:chain-level-e3-original-complex-conditional} hold.
\end{minipage}\end{center}

The closure has two tiers. The tempering statement
(Theorem~\ref{thm:wn-tempered-all-N}) is proved as a finite-\(\beta_N\)
criterion:
the Stirling cancellation $(r!)^{-1/r} \sim e/r \to 0$ dominates any
finite $\beta_N$, so the closed-form of $\beta_N$ is irrelevant once
the envelope is known. The existence of such an envelope for all
principal $\cW_N$ is a separate proof obligation. The closed form of
$\beta_N$ is another separate claim, conditional on the harmonic
kappa-ratio scaling law, with the $N = 2, 3$ Laurent data as the first
two consistency checks.

\section{Stirling dominance: tempering at every finite $\beta_N$}
\label{sec:stirling-dominance}

The load-bearing analytic fact is the $\beta$-independence of the
Stirling cancellation. This supplies tempering at every $N$ regardless
of the closed-form of $\beta_N$.

\begin{proposition}[$\beta$-independent Stirling dominance]
\label{prop:beta-stirling-dominance}
\ClaimStatusProvedHere
Let $\cA$ be a chiral algebra satisfying the shadow-tower upper bound
\[
 \lvert S_r(\cA) \rvert
 \;\le\;
 C(\cA) \cdot \frac{\beta^{r-4}}{r \cdot \lvert c \rvert^{r-2}}
 \qquad (r \ge 4)
\]
for some finite $\beta > 0$ and constant $C(\cA) > 0$ that may depend
on $c$ but not on $r$. Then
\[
 \limsup_{r \to \infty}
 \bigl( \lvert S_r(\cA) \rvert / r! \bigr)^{1/r}
 \;=\;
 0,
\]
and more quantitatively
\[
 \bigl(\lvert S_r(\cA)\rvert / r!\bigr)^{1/r}
 \;\le\;
 \frac{\beta\,e}{r\,|c|}\,\bigl(1 + o(1)\bigr)
 \qquad (r \to \infty).
\]
\end{proposition}

\begin{proof}
Using the hypothesis and Stirling's approximation
$r!^{1/r} \sim r/e$,
\begin{align*}
 \bigl(\lvert S_r \rvert / r!\bigr)^{1/r}
 &\;\le\; \bigl(C(\cA) \cdot \beta^{r-4}
 / (r \cdot |c|^{r-2} \cdot r!)\bigr)^{1/r} \\
 &\;=\; C(\cA)^{1/r} \cdot \beta^{(r-4)/r} \cdot r^{-1/r}
 \cdot |c|^{-(r-2)/r} \cdot (r!)^{-1/r} \\
 &\;\sim\; 1 \cdot \beta \cdot 1 \cdot |c|^{-1} \cdot (e/r)
 \;=\; \frac{\beta\,e}{r\,|c|}
 \;\longrightarrow\; 0.
\end{align*}
The product tends to $0$ as $r \to \infty$ at every finite $\beta > 0$
and every $c \ne 0$. In particular the $\limsup$ equals $0$.
\end{proof}

\begin{remark}[Universal $\beta$-independence]
\label{rem:beta-independence}
Proposition~\ref{prop:beta-stirling-dominance} is the general analytic
mechanism: any chiral algebra whose shadow sequence is dominated by a
finite geometric factor $\beta^{r-4}$ satisfies $(\mathrm{H}_S)$. The
proof does not use any specific value of $\beta$. Consequently, for the
tempering conclusion, we need only establish EXISTENCE of such a finite
$\beta_N$ for every principal $\cW_N$, not its exact value.
\end{remark}

\section{Finite-envelope $\cW_N$ tempering criterion}
\label{sec:wn-tempering}

\begin{theorem}[Finite-\(\beta_N\) criterion for principal $\cW_N$ tempering]
\label{thm:wn-tempered-all-N}
\ClaimStatusConditional
Fix an integer $N \ge 2$ and
$c \in \R \setminus (\{0\} \cup \cD^{\mathrm{sev}}_{\mathrm{Kac}}(\cW_N))$,
and assume that the principal $\cW_{N,c}$ shadow tower satisfies a
finite Riccati envelope
\[
 \lvert S_r(\cW_{N,c}) \rvert
 \;\le\;
 C(N,c)\,\frac{\beta_N^{r-4}}{r\,|c|^{r-2}}
 \qquad (r\ge 4)
\]
for some finite $\beta_N>0$ and $C(N,c)>0$. Then
$\cW_{N,c}$ satisfies the analytic tempering hypothesis
$(\mathrm{H}_S)$:
\[
 \limsup_{r \to \infty}
 \bigl( \lvert S_r(\cW_{N,c}) \rvert / r! \bigr)^{1/r}
 \;=\;
 0.
\]
Consequently the analytically tempered class-\(\mathsf M\) locus
contains every principal $\cW_N$ point for which this finite envelope is
proved. The Banach $E_3$-model is available on that locus; descent to
the raw direct-sum curved-Dunn complex is precisely the
finite-propagation condition of
Theorem~\ref{thm:chain-level-e3-original-complex-conditional}.
\end{theorem}

\begin{proof}
Two steps.

\emph{Step 1: the finite envelope is the hypothesis.}
For \(N=2\) and \(N=3\) the envelope is supplied by
Theorem~\ref{thm:tempered-stratum-contains-virasoro} and
Theorem~\ref{thm:tempered-stratum-contains-w3}. For \(N\ge4\), this
chapter does not derive it from Fateev--Lukyanov structure constants.
The missing input is a full Miura/OPE computation of the shadow
Riccati coefficients, already visible at the \(\cW_4\) bridge
\(A_5^{\cW_4}/A_4^{\cW_4}=-52/5\). The theorem therefore takes the
finite envelope as an explicit assumption.

\emph{Step 2: Stirling dominance applies.}
Applying Proposition~\ref{prop:beta-stirling-dominance} with
$\beta = \beta_N$:
\[
 \limsup_{r \to \infty}
 \bigl( \lvert S_r(\cW_{N,c}) \rvert / r! \bigr)^{1/r}
 \;=\;
 0.
\]
Tempering of $\cW_{N,c}$ follows on the finite-envelope locus. This
discharges the analytic
hypothesis $(\mathrm{H}_S)$ in
Theorem~\ref{thm:chain-level-e3-original-complex-conditional}; the
raw direct-sum descent remains exactly the finite-propagation
hypothesis stated there.
\end{proof}

\begin{remark}[Finite $\beta_N$ is enough for tempering]
\label{rem:wn-tempering-scope}
For every fixed $N$, tempering only needs finiteness of $\beta_N$, not
its closed form. The Stirling cancellation in
$(r!)^{-1/r} \sim e/r$ absorbs any finite $\beta_N$. Thus the
tempering implication is independent of the arithmetic discrimination
of $\beta_N$ once the finite envelope is proved. For \(N\ge4\), proving
that envelope is a separate Riccati/Miura obligation; the exact closed
form is the conditional harmonic formula
(Proposition~\ref{prop:beta-N-closed-form}).
\end{remark}

\section{The harmonic candidate $\beta_N = 12(H_N - 1)$}
\label{sec:beta-N-closed-form}

The harmonic closed form of $\beta_N$ is the second tier of the frontier.
The two low-rank Laurent computations determine $\beta_2=6$ and
$\beta_3=10$; the all-rank proof obligation is not a curve fit through
these two values. It is the kappa-ratio scaling law for the leading
Laurent coefficient.

\begin{conjecture}[Closed form of $\beta_N$: harmonic kappa-ratio law]
\label{prop:beta-N-closed-form}
\ClaimStatusConjectured
The Riccati leading-Laurent ratio coefficient $\beta_N$ of the
principal $\cW_N$ shadow tower satisfies
\[
 \beta_N \;=\; 12(H_N - 1)
 \;=\; \sum_{s=2}^{N} \frac{12}{s}
\]
for every $N \ge 2$. In particular
\[
 \beta_2 = 6,\qquad
 \beta_3 = 10,\qquad
 \beta_4 = 13,\qquad
 \beta_5 = 77/5,\qquad
 \beta_6 = 87/5.
\]
\end{conjecture}

\begin{remark}[Conditional derivation from kappa-ratio scaling]
Let $A_r^{\Vir}$ denote the Virasoro leading Laurent coefficient.
Lemma~\ref{lem:leading-coefficient-ratio-identity} gives
\[
 \frac{A_r^{\Vir}}{A_{r-1}^{\Vir}}
 \;=\; -6\,\frac{r-1}{r}.
\]
For the principal $\cW_N$ tower, the kappa-ratio scaling law conjectured in
Conjecture~\ref{thm:beta-N-closed-form-proved-all-N} gives
\[
 A_r^{\cW_N}
 \;=\;
 \left(\frac{\kappa(\cW_N)}{\kappa(\Vir)}\right)^{r-3}
 A_r^{\Vir}.
\]
The canonical conductor identity is
\[
 \kappa(\cW_N)=c(H_N-1),\qquad \kappa(\Vir)=c/2,
\]
hence $\kappa(\cW_N)/\kappa(\Vir)=2(H_N-1)$. Dividing the two
successive scaled coefficients,
\[
 \frac{A_r^{\cW_N}}{A_{r-1}^{\cW_N}}
 =
 2(H_N-1)\,\frac{A_r^{\Vir}}{A_{r-1}^{\Vir}}
 =
 -12(H_N-1)\,\frac{r-1}{r}.
\]
By definition of the Riccati ratio coefficient, this coefficient is
$-\beta_N(r-1)/r$. Therefore $\beta_N=12(H_N-1)$. The displayed
specialisations follow by evaluating
$H_2=3/2$, $H_3=11/6$, $H_4=25/12$, $H_5=137/60$, and $H_6=49/20$.
\end{remark}

\begin{remark}[Polynomial candidates separated at $N = 4$]
\label{rem:beta-candidate-discriminator}
The triangular extrapolation
\[
 \beta_N^{(A)}=\frac{(N+1)(N+2)}{2}
\]
and the quadratic extrapolation
\[
 \beta_N^{(B)} \;=\; N^2 - N + 4 \;=\; 6 + (N-2)(N+1).
\]
both match $\beta_2=6$ and $\beta_3=10$ by coincidence. They are
separated by the harmonic law. At $N=4$ the harmonic candidate gives
\[
 \beta_4 = 12(H_4-1)=13,
 \qquad
 \beta_4^{(A)}=15,
 \qquad
 \beta_4^{(B)}=16.
\]
Thus, under the harmonic law, the first rank not already forced by the
Virasoro and $\cW_3$ checks separates both extrapolations. The tempering conclusion of
Theorem~\ref{thm:wn-tempered-all-N} remains independent of this
separation because it only uses finiteness of $\beta_N$ after the
Riccati envelope is supplied.
\end{remark}

\begin{proposition}[Convergence radius closed form]
\label{prop:rho-star-WN-closed-form}
\ClaimStatusConditional
Assume Conjecture~\ref{prop:beta-N-closed-form}. For every $N \ge 2$
and $c \in \R \setminus (\{0\} \cup
\cD^{\mathrm{sev}}_{\mathrm{Kac}}(\cW_N))$,
\[
 \rho_*^{\cW_N}(c)
 \;:=\;
 \Bigl( \limsup_{r \to \infty} \lvert S_r(\cW_{N,c}) \rvert^{1/r}
 \Bigr)^{-1}
 \;=\;
 \frac{\lvert c \rvert}{\beta_N}
 \;=\;
 \frac{\lvert c\rvert}{12(H_N - 1)}.
\]
Specialisations: $\rho_*^{\mathrm{Vir}}(c) = |c|/6$ (Virasoro,
$N=2$); $\rho_*^{\cW_3}(c) = |c|/10$ ($N=3$);
$\rho_*^{\cW_4}(c)=|c|/13$; $\rho_*^{\cW_5}(c)=5|c|/77$.
\end{proposition}

\begin{proof}
Direct transcription of the proof of
Proposition~\ref{prop:virasoro-rho-star-closed-form} with the
Virasoro factor $6$ replaced by $\beta_N$, and applying
Conjecture~\ref{prop:beta-N-closed-form} for the closed form of
$\beta_N$.
\end{proof}

\section{Principal $\cW_N$ tempered stratum}
\label{sec:wn-dichotomy-healed}

\begin{corollary}[Tempering for every principal $\cW_N$]
\label{cor:wn-original-complex-dichotomy-healed}
\ClaimStatusConditional
Combining
Theorem~\ref{thm:wn-tempered-all-N} with
Corollary~\ref{cor:class-m-original-complex-dichotomy}, the non-tempered
generic stratum of a principal $\cW_N$, $N \ge 2$, is empty on the
finite-envelope locus: every generic $\cW_{N,c}$ with
$c \notin \{0\} \cup \cD^{\mathrm{sev}}_{\mathrm{Kac}}(\cW_N)$ and a
proved finite Riccati envelope is tempered.

\emph{Remaining non-tempered frontier}: the residual candidates for
a non-empty generic stratum include the unproved finite-envelope
problem for principal $\cW_N$ at \(N\ge4\), logarithmic $\cW(p)$
(Adamovic-Milas), Monster-type orbifolds (Frenkel-Lepowsky-Meurman
twisted sectors), and coset realisations at irrational central charge.
Each requires its own hypotheses; none is supplied by the finite-beta
Stirling estimate alone.
\end{corollary}

\begin{proof}
Direct combination of the finite-envelope criterion
Theorem~\ref{thm:wn-tempered-all-N} with
Corollary~\ref{cor:class-m-original-complex-dichotomy}.
\end{proof}

\begin{remark}[Harmonic beta law]
\label{rem:heal-sweep-applied-wn}
Theorem~\ref{thm:wn-tempered-all-N} proves that a finite Riccati
envelope implies tempering for every fixed $N \ge 2$ at every generic
$c$. The closed-form
$\beta_N = 12(H_N - 1)$ is conditional on
Conjecture~\ref{prop:beta-N-closed-form}. Under that conjecture, the
triangular and quadratic polynomial extrapolations are excluded at
$N = 4$, where the harmonic value is $\beta_4=13$ and the polynomial
values are $15$ and $16$. The tempering conclusion depends only on finite
$\beta_N$: Stirling dominates every finite $\beta_N$ once the envelope
is available, regardless of its exact arithmetic form.
\end{remark}

\label{sec:wn-tempered-closure-iv-summary}

\textbf{1. Conditional $\beta_N$ closed form
(Proposition~\ref{prop:beta-N-closed-form}).}
\begin{itemize}[nosep]
\item \emph{Derived from}: the kappa-ratio scaling law
 $A_r(\cW_N)=(\kappa(\cW_N)/\kappa(\Vir))^{r-3}A_r(\Vir)$, the

 $\cW_4$ lane calculation are finite-rank tests. They do not yet

(Proposition~\ref{prop:beta-stirling-dominance}).}
\begin{itemize}[nosep]
\item \emph{Derived from}: Stirling approximation $(r!)^{-1/r} \sim
 e/r$.
\item \emph{Verified against}: Direct numerical evaluation of the
 critical radius $r_{\mathrm{crit}} = \beta_N \cdot e / |c|$; empirical
 bound at $r = 100$ for $N \le 6$ and $c = 100$.
\end{itemize}

\textbf{3. Finite-envelope $\cW_N$ tempering criterion
(Theorem~\ref{thm:wn-tempered-all-N}).}
\begin{itemize}[nosep]

\item \emph{Verified against}: Direct numerical evaluation of the rate
 bound at $r = 4, 10, 20, 30$ for $N = 2..8$ and $c = 100$; monotone

 every $N \le 6$.
\item \emph{Scope}: these tests verify the finite-\(\beta_N\) implication,
 not the all-rank existence of the Riccati envelope.
\end{itemize}

\textbf{4. Conditional convergence radius closed form
(Proposition~\ref{prop:rho-star-WN-closed-form}).}
\begin{itemize}[nosep]

\textbf{5. Polynomial interpolation exclusion
(Remark~\ref{rem:beta-candidate-discriminator}).}
\begin{itemize}[nosep]
\item \emph{Derived from}: comparison of the harmonic conjecture
 $12(H_N-1)$ with the two polynomial interpolants $(N+1)(N+2)/2$ and

\item \emph{Verified against}: Direct evaluation at $N = 2, 3, 4$;
 all three agree at $N = 2, 3$ only by coincidence, while at $N = 4$
 the harmonic value is $13$ and the polynomial interpolants give
 $15$ and $16$.
\end{itemize}
