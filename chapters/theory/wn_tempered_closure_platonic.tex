\chapter{Closure of $\cW_N$ tempering for every $N \ge 2$:
universal Stirling dominance and the harmonic $\beta_N$ frontier}
\label{ch:wn-tempered-closure}

\section*{Statement of the closure}

Chapter~\ref{ch:tempered-stratum-characterization} establishes
universal Virasoro tempering and the analogous $\cW_3$ tempering;
Conjecture~\ref{conj:wn-tempered-general} is the open principal
$\cW_N$ case at $N \ge 4$. The finite-$\beta_N$ Stirling
implication is proved below; the two remaining inputs are the finite
Riccati envelope for all ranks and the conditional harmonic Riccati
ratio.

\emph{Finite-$\beta_N$ criterion (Theorem~\ref{thm:wn-tempered-all-N}).}
Fix $N \ge 2$ and $c \in \R$ with
$c \notin \{0\} \cup \cD^{\mathrm{sev}}_{\mathrm{Kac}}(\cW_N)$. On the
finite Riccati envelope
\[
 \lvert S_r(\cW_{N,c}) \rvert
 \;\le\;
 C(N,c)\,\frac{\beta_N^{r-4}}{r\,|c|^{r-2}}
 \qquad (r\ge 4)
\]
for finite $\beta_N>0$, the algebra $\cW_{N,c}$ is analytically
tempered:
\[
 \limsup_{r \to \infty}
 \bigl(\lvert S_r(\cW_{N,c}) \rvert / r!\bigr)^{1/r}
 \;=\;
 0.
\]
The ordinary-generating convergence radius of
$\sum_{r \ge 2} S_r(\cW_{N,c})\,\rho^r$ equals
$\rho_*^{\cW_N}(c) = \lvert c\rvert / \beta_N$ for the leading
Riccati coefficient $\beta_N$; under
Conjecture~\ref{prop:beta-N-closed-form}, this coefficient takes the
harmonic closed form
\[
 \beta_N \;=\; 12(H_N - 1)
 \;=\; \sum_{s=2}^{N} \frac{12}{s}.
\]
The non-tempered stratum of a principal $\cW_N$ is empty on every
locus with the finite Riccati envelope; the chain-level
$E_3$-topological structure on the original curved-Dunn complex
exists for such $\cW_{N,c}$ under the finite-propagation hypotheses
of Theorem~\ref{thm:chain-level-e3-original-complex-conditional}.

The closure has two tiers. The Stirling cancellation
$(r!)^{-1/r} \sim e/r \to 0$ dominates any finite $\beta_N$, so the
closed form of $\beta_N$ does not enter the tempering implication.
The existence of the envelope for all principal $\cW_N$ is a separate
proof obligation. The closed form of $\beta_N$ is conditional on the
harmonic kappa-ratio scaling law, with $N = 2, 3$ Laurent data giving
the first two consistency checks.

\section{Stirling dominance: tempering at every finite $\beta_N$}
\label{sec:stirling-dominance}

The Stirling cancellation is $\beta$-independent, giving tempering at
every $N$ regardless of the closed form of $\beta_N$.

\begin{proposition}[$\beta$-independent Stirling dominance]
\label{prop:beta-stirling-dominance}
\ClaimStatusProvedHere
Let $\cA$ be a chiral algebra satisfying the shadow-tower upper bound
\[
 \lvert S_r(\cA) \rvert
 \;\le\;
 C(\cA) \cdot \frac{\beta^{r-4}}{r \cdot \lvert c \rvert^{r-2}}
 \qquad (r \ge 4)
\]
for some finite $\beta > 0$ and constant $C(\cA) > 0$ that may depend
on $c$ but not on $r$. Then
\[
 \limsup_{r \to \infty}
 \bigl( \lvert S_r(\cA) \rvert / r! \bigr)^{1/r}
 \;=\;
 0,
\]
and more quantitatively
\[
 \bigl(\lvert S_r(\cA)\rvert / r!\bigr)^{1/r}
 \;\le\;
 \frac{\beta\,e}{r\,|c|}\,\bigl(1 + o(1)\bigr)
 \qquad (r \to \infty).
\]
\end{proposition}

\begin{proof}
Using the hypothesis and Stirling's approximation
$r!^{1/r} \sim r/e$,
\begin{align*}
 \bigl(\lvert S_r \rvert / r!\bigr)^{1/r}
 &\;\le\; \bigl(C(\cA) \cdot \beta^{r-4}
 / (r \cdot |c|^{r-2} \cdot r!)\bigr)^{1/r} \\
 &\;=\; C(\cA)^{1/r} \cdot \beta^{(r-4)/r} \cdot r^{-1/r}
 \cdot |c|^{-(r-2)/r} \cdot (r!)^{-1/r} \\
 &\;\sim\; 1 \cdot \beta \cdot 1 \cdot |c|^{-1} \cdot (e/r)
 \;=\; \frac{\beta\,e}{r\,|c|}
 \;\longrightarrow\; 0.
\end{align*}
The product tends to $0$ as $r \to \infty$ at every finite $\beta > 0$
and every $c \ne 0$. In particular the $\limsup$ equals $0$.
\end{proof}

\begin{remark}[Universal $\beta$-independence]
\label{rem:beta-independence}
Any chiral algebra whose shadow sequence is dominated by a finite
geometric factor $\beta^{r-4}$ satisfies $(\mathrm{H}_S)$, with the
proof independent of the value of $\beta$. The tempering conclusion
requires only existence of a finite $\beta_N$ for principal $\cW_N$,
not the exact value.
\end{remark}

\section{Finite-envelope $\cW_N$ tempering criterion}
\label{sec:wn-tempering}

\begin{theorem}[Finite-\(\beta_N\) criterion for principal $\cW_N$ tempering]
\label{thm:wn-tempered-all-N}
\ClaimStatusConditional
Fix an integer $N \ge 2$ and
$c \in \R \setminus (\{0\} \cup \cD^{\mathrm{sev}}_{\mathrm{Kac}}(\cW_N))$,
and assume that the principal $\cW_{N,c}$ shadow tower satisfies a
finite Riccati envelope
\[
 \lvert S_r(\cW_{N,c}) \rvert
 \;\le\;
 C(N,c)\,\frac{\beta_N^{r-4}}{r\,|c|^{r-2}}
 \qquad (r\ge 4)
\]
for some finite $\beta_N>0$ and $C(N,c)>0$. Then
$\cW_{N,c}$ satisfies the analytic tempering hypothesis
$(\mathrm{H}_S)$:
\[
 \limsup_{r \to \infty}
 \bigl( \lvert S_r(\cW_{N,c}) \rvert / r! \bigr)^{1/r}
 \;=\;
 0.
\]
The analytically tempered class~$\mathsf M$ locus contains every
principal $\cW_N$ point with this finite envelope. The Banach
$E_3$-model holds on that locus; descent to the raw direct-sum
curved-Dunn complex is the finite-propagation condition of
Theorem~\ref{thm:chain-level-e3-original-complex-conditional}.
\end{theorem}

\begin{proof}
Two steps.

\emph{Step 1: the finite envelope is the hypothesis.}
For $N=2$ and $N=3$ the envelope is supplied by
Theorem~\ref{thm:tempered-stratum-contains-virasoro} and
Theorem~\ref{thm:tempered-stratum-contains-w3}. For $N\ge4$ the
envelope comes from a full Miura/OPE computation of the shadow
Riccati coefficients, visible at the $\cW_4$ bridge
$A_5^{\cW_4}/A_4^{\cW_4}=-52/5$. The theorem takes the finite
envelope as an explicit hypothesis.

\emph{Step 2: Stirling dominance applies.}
Proposition~\ref{prop:beta-stirling-dominance} with
$\beta = \beta_N$ gives
\[
 \limsup_{r \to \infty}
 \bigl( \lvert S_r(\cW_{N,c}) \rvert / r! \bigr)^{1/r}
 \;=\;
 0.
\]
Tempering of $\cW_{N,c}$ follows on the finite-envelope locus,
discharging the analytic hypothesis $(\mathrm{H}_S)$ in
Theorem~\ref{thm:chain-level-e3-original-complex-conditional}. The
raw direct-sum descent is the finite-propagation hypothesis of that
theorem.
\end{proof}

\begin{remark}[Finite $\beta_N$ suffices for tempering]
\label{rem:wn-tempering-scope}
At each fixed $N$, tempering requires only finiteness of $\beta_N$.
The Stirling cancellation $(r!)^{-1/r} \sim e/r$ absorbs every finite
$\beta_N$. For $N\ge4$ the finite envelope is a Riccati/Miura
obligation; the closed form is the conditional harmonic formula
(Proposition~\ref{prop:beta-N-closed-form}).
\end{remark}

\section{The harmonic candidate $\beta_N = 12(H_N - 1)$}
\label{sec:beta-N-closed-form}

The harmonic closed form of $\beta_N$ is the second tier of the
frontier. The Laurent computations determine $\beta_2=6$ and
$\beta_3=10$; the all-rank closed form comes from the kappa-ratio
scaling law for the leading Laurent coefficient.

\begin{conjecture}[Closed form of $\beta_N$: harmonic kappa-ratio law]
\label{prop:beta-N-closed-form}
\ClaimStatusConjectured
The Riccati leading-Laurent ratio coefficient $\beta_N$ of the
principal $\cW_N$ shadow tower satisfies
\[
 \beta_N \;=\; 12(H_N - 1)
 \;=\; \sum_{s=2}^{N} \frac{12}{s}
\]
for every $N \ge 2$. In particular
\[
 \beta_2 = 6,\qquad
 \beta_3 = 10,\qquad
 \beta_4 = 13,\qquad
 \beta_5 = 77/5,\qquad
 \beta_6 = 87/5.
\]
\end{conjecture}

\begin{remark}[Conditional derivation from kappa-ratio scaling]
Let $A_r^{\Vir}$ denote the Virasoro leading Laurent coefficient.
Lemma~\ref{lem:leading-coefficient-ratio-identity} gives
\[
 \frac{A_r^{\Vir}}{A_{r-1}^{\Vir}}
 \;=\; -6\,\frac{r-1}{r}.
\]
For the principal $\cW_N$ tower, the kappa-ratio scaling law conjectured in
Conjecture~\ref{thm:beta-N-closed-form-proved-all-N} gives
\[
 A_r^{\cW_N}
 \;=\;
 \left(\frac{\kappaChHodge(\cW_N)}{\kappaChHodge(\Vir)}\right)^{r-3}
 A_r^{\Vir}.
\]
The canonical conductor identity is
\[
 \kappaChHodge(\cW_N)=c(H_N-1),\qquad \kappaChHodge(\Vir)=c/2,
\]
hence $\kappaChHodge(\cW_N)/\kappaChHodge(\Vir)=2(H_N-1)$. Dividing the two
successive scaled coefficients,
\[
 \frac{A_r^{\cW_N}}{A_{r-1}^{\cW_N}}
 =
 2(H_N-1)\,\frac{A_r^{\Vir}}{A_{r-1}^{\Vir}}
 =
 -12(H_N-1)\,\frac{r-1}{r}.
\]
By definition of the Riccati ratio coefficient, this coefficient is
$-\beta_N(r-1)/r$. Therefore $\beta_N=12(H_N-1)$. The displayed
specialisations follow by evaluating
$H_2=3/2$, $H_3=11/6$, $H_4=25/12$, $H_5=137/60$, and $H_6=49/20$.
\end{remark}

\begin{remark}[Polynomial candidates separated at $N = 4$]
\label{rem:beta-candidate-discriminator}
The triangular extrapolation
\[
 \beta_N^{(A)}=\frac{(N+1)(N+2)}{2}
\]
and the quadratic extrapolation
\[
 \beta_N^{(B)} \;=\; N^2 - N + 4 \;=\; 6 + (N-2)(N+1)
\]
match $\beta_2=6$ and $\beta_3=10$ and separate from the harmonic law
at $N=4$:
\[
 \beta_4 = 12(H_4-1)=13,
 \qquad
 \beta_4^{(A)}=15,
 \qquad
 \beta_4^{(B)}=16.
\]
The tempering conclusion of
Theorem~\ref{thm:wn-tempered-all-N} uses only finiteness of $\beta_N$
once the Riccati envelope is supplied, independently of the
separation.
\end{remark}

\begin{proposition}[Convergence radius closed form]
\label{prop:rho-star-WN-closed-form}
\ClaimStatusConditional
Assume Conjecture~\ref{prop:beta-N-closed-form}. For every $N \ge 2$
and $c \in \R \setminus (\{0\} \cup
\cD^{\mathrm{sev}}_{\mathrm{Kac}}(\cW_N))$,
\[
 \rho_*^{\cW_N}(c)
 \;:=\;
 \Bigl( \limsup_{r \to \infty} \lvert S_r(\cW_{N,c}) \rvert^{1/r}
 \Bigr)^{-1}
 \;=\;
 \frac{\lvert c \rvert}{\beta_N}
 \;=\;
 \frac{\lvert c\rvert}{12(H_N - 1)}.
\]
Specialisations: $\rho_*^{\mathrm{Vir}}(c) = |c|/6$ (Virasoro,
$N=2$); $\rho_*^{\cW_3}(c) = |c|/10$ ($N=3$);
$\rho_*^{\cW_4}(c)=|c|/13$; $\rho_*^{\cW_5}(c)=5|c|/77$.
\end{proposition}

\begin{proof}
Transcribe the proof of
Proposition~\ref{prop:virasoro-rho-star-closed-form} with the
Virasoro factor $6$ replaced by $\beta_N$, applying
Conjecture~\ref{prop:beta-N-closed-form} for the closed form.
\end{proof}

\section{Principal $\cW_N$ tempered stratum}
\label{sec:wn-dichotomy-healed}

\begin{corollary}[Tempering for every principal $\cW_N$]
\label{cor:wn-original-complex-dichotomy-healed}
\ClaimStatusConditional
Theorem~\ref{thm:wn-tempered-all-N} and
Corollary~\ref{cor:class-m-original-complex-dichotomy} give the
non-tempered generic stratum of a principal $\cW_N$, $N \ge 2$, as
empty on the finite-envelope locus: every generic $\cW_{N,c}$ with
$c \notin \{0\} \cup \cD^{\mathrm{sev}}_{\mathrm{Kac}}(\cW_N)$ and
finite Riccati envelope is tempered.

\emph{Remaining open candidates}: the finite-envelope problem for
principal $\cW_N$ at $N\ge4$; logarithmic $\cW(p)$ (Adamovic--Milas);
Monster-type orbifolds (Frenkel--Lepowsky--Meurman twisted sectors);
coset realisations at irrational central charge.
\end{corollary}

\begin{proof}
Combine the finite-envelope criterion of
Theorem~\ref{thm:wn-tempered-all-N} with
Corollary~\ref{cor:class-m-original-complex-dichotomy}.
\end{proof}

\begin{remark}[Harmonic beta law]
\label{rem:heal-sweep-applied-wn}
Theorem~\ref{thm:wn-tempered-all-N} gives the finite-envelope
tempering implication at every fixed $N \ge 2$. The closed form
$\beta_N = 12(H_N - 1)$ is conditional on
Conjecture~\ref{prop:beta-N-closed-form}; under that conjecture, the
triangular and quadratic polynomial extrapolations separate from the
harmonic value at $N = 4$ ($\beta_4=13$ versus $15$ and $16$). The
tempering conclusion depends only on finite $\beta_N$.
\end{remark}

\section{Independent verification summary}
\label{sec:wn-tempered-closure-iv-summary}

Tests at \texttt{compute/tests/test\_wn\_tempered\_closure.py}
with \texttt{@independent\_verification} decorators requiring
disjoint \texttt{derived\_from} and \texttt{verified\_against}
source families.

\textbf{1. Conditional $\beta_N$ closed form
(Proposition~\ref{prop:beta-N-closed-form}).}
\begin{itemize}[nosep]
\item \emph{Derived from}: the kappa-ratio scaling law
 $A_r(\cW_N)=(\kappaChHodge(\cW_N)/\kappaChHodge(\Vir))^{r-3}A_r(\Vir)$, the
 conductor identity $\kappaChHodge(\cW_N)=c(H_N-1)$, and the Virasoro ratio
 $A_r^{\Vir}/A_{r-1}^{\Vir}=-6(r-1)/r$.
\item \emph{Verified against}: Direct numerical match at $N = 2$
 ($\beta = 6$ from Virasoro leading Laurent) and $N = 3$ ($\beta =
 10$ from $\cW_3$ leading Laurent); the finite-rank $\cW_4$ spin-lane
 attack gives the compatible lane sum $6+4+3=13$ against the polynomial
 values $15$ and $16$.
\item \emph{Disjoint rationale}: The kappa-ratio derivation computes
 the whole family, while the low-rank Laurent checks and the
 $\cW_4$ lane calculation are finite-rank tests. They do not yet
 compute $A_5^{\cW_4}/A_4^{\cW_4}$ from the full Miura OPE, so they are
 evidence for the harmonic law, not a proof.
\end{itemize}

\textbf{2. $\beta$-Stirling dominance
(Proposition~\ref{prop:beta-stirling-dominance}).}
\begin{itemize}[nosep]
\item \emph{Derived from}: Stirling approximation $(r!)^{-1/r} \sim
 e/r$.
\item \emph{Verified against}: Direct numerical evaluation of the
 critical radius $r_{\mathrm{crit}} = \beta_N \cdot e / |c|$; empirical
 bound at $r = 100$ for $N \le 6$ and $c = 100$.
\end{itemize}

\textbf{3. Finite-envelope $\cW_N$ tempering criterion
(Theorem~\ref{thm:wn-tempered-all-N}).}
\begin{itemize}[nosep]
\item \emph{Derived from}: a conditional Riccati upper bound with finite
 $\beta_N$; Proposition~\ref{prop:beta-stirling-dominance}.
\item \emph{Verified against}: Direct numerical evaluation of the rate
 bound at $r = 4, 10, 20, 30$ for $N = 2..8$ and $c = 100$; monotone
 decreasing bound; values already below $0.2$ at $r = 30$ for
 every $N \le 6$.
\item \emph{Scope}: these tests verify the finite-\(\beta_N\) implication,
 not the all-rank existence of the Riccati envelope.
\end{itemize}

\textbf{4. Conditional convergence radius closed form
(Proposition~\ref{prop:rho-star-WN-closed-form}).}
\begin{itemize}[nosep]
\item \emph{Derived from}: the conditional harmonic $\beta_N$ closed form plus
 the Cauchy-Hadamard root test.
\item \emph{Verified against}: Algebraic match at $N = 2$ ($|c|/6$
 from Vol~II Proposition~\ref{prop:virasoro-rho-star-closed-form}) and $N = 3$
 ($|c|/10$ from \texttt{thm:tempered-stratum-contains-w3}); direct
 arithmetic at $N = 4, 5, 6$ gives $|c|/13$, $5|c|/77$, and
 $5|c|/87$.
\end{itemize}

\textbf{5. Polynomial interpolation exclusion
(Remark~\ref{rem:beta-candidate-discriminator}).}
\begin{itemize}[nosep]
\item \emph{Derived from}: comparison of the harmonic conjecture
 $12(H_N-1)$ with the two polynomial interpolants $(N+1)(N+2)/2$ and
 $N^2 - N + 4$.
\item \emph{Verified against}: Direct evaluation at $N = 2, 3, 4$;
 all three agree at $N = 2, 3$ only by coincidence, while at $N = 4$
 the harmonic value is $13$ and the polynomial interpolants give
 $15$ and $16$.
\end{itemize}
