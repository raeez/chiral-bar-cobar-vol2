\section{The Mellin--shadow bridge at every soft order}
\label{sec:soft-graviton-mellin-shadow-bridge-platonic}
%=======================================================================

\noindent
The universal celestial holography theorem
(\S\ref{sec:universal-celestial-holography},
Theorem~\ref{thm:uch-main}) identifies the $n$-point Mellin amplitude of
a $4$d HT-twisted theory with the chiral factorization homology of its
celestial chiral algebra on $\PP^1_{\mathrm{cel}}$.
Theorem~\ref{thm:uch-soft-hierarchy} abstracts the statement: the
residue of $\Mel\cM_n$ at the conformally soft pole
$\Delta_s = 4 - r$ equals the pairing of the bar-cohomology class of
shadow depth $r - 1$ with the remaining hard insertions. What was missing
in the earlier phrasing was the \emph{explicit rational
coefficient}: precisely which shadow invariant $S_r(\cA^{\mathrm{cel}})$
produces the sub${}^{r - 2}$-leading soft factor of
Hamada--Shiu~\cite{HamadaShiuSubsubSoft} and Li--Strominger~\cite{LiStromingerHigherSoft}.

This chapter supplies the explicit rational coefficient at every depth
$r = 2, 3, 4, 5, 6, 7, 8$ for the Virasoro celestial chiral algebra
$\cA^{\mathrm{cel}} = \Vir_c$ arising from HT-twisted gravity. The bridge
is short because the technical core is already in place:
\begin{itemize}
\item The Mellin--shadow dictionary is a theorem
(\S\ref{subsec:uch-mellin}, Proposition~\ref{prop:uch-mellin-shadow}).
\item The shadow coefficients $S_r(\Vir_c)$ are explicit closed rational
functions of $c$ through $r = 8$, proved in Vol~I
\texttt{shadow\_tower\_higher\_coefficients.tex} via two disjoint
derivation chains (Maurer--Cartan recurrence and Riccati
$\sqrt{Q_L}$ Taylor expansion).
\end{itemize}
What this chapter adds is the explicit identification with the
kinematic soft-factor coefficients of the celestial literature, so that
the $r \geq 4$ clause of Theorem~\ref{thm:uch-soft-hierarchy} is
unconditional on the original complex at genus~$0$, with precise
rational-in-$c$ coefficients.

The structure follows the logic of the identification.
\S\ref{subsec:smsb-setup} fixes notation and records the PSS 2017
Mellin dictionary. \S\ref{subsec:smsb-bridge} proves the
weight-by-weight Mellin--shadow correspondence
(Theorem~\ref{thm:mellin-shadow-correspondence-explicit}).
\S\ref{subsec:smsb-hierarchy} applies the bridge to $r = 2, \ldots, 8$,
producing closed rational coefficients via the Vol~I shadow closed
forms (Theorem~\ref{thm:soft-hierarchy-r-leq-8}).
\S\ref{subsec:smsb-hsl-match} proves the $r = 4$ match with the
Hamada--Shiu--Li--Strominger sub-sub-leading soft graviton
(Theorem~\ref{thm:r-eq-4-is-leading-subsubleading}).
\S\ref{subsec:smsb-angular-stress} identifies the angular-momentum and
stress-tensor insertions encoded at each depth
(Theorem~\ref{thm:angular-momentum-stress-tensor-via-shadow}).
\S\ref{subsec:smsb-unconditional} records the unconditional corollary
(Corollary~\ref{cor:soft-hierarchy-unconditional-r-leq-8}).

\subsection{Setup: Mellin transform and shadow coefficients of $\Vir_c$}
\label{subsec:smsb-setup}
%-----------------------------------------------------------------------

We fix notation consistent with
\cite{PasterskiShaoStromingerFlatSpaceHolog,StromingerLectures2018,
HamadaShiuSubsubSoft,LiStromingerHigherSoft} and with Vol~II
\S\ref{subsec:uch-mellin}.

\begin{definition}[Celestial Mellin amplitude]
\label{def:smsb-celestial-mellin}
\ClaimStatusProvedElsewhere
For a $4$d HT-twistable theory $T_4$ with $n$-point momentum-space
amplitude $\cM_n(\omega_i; z_i, \bar z_i)$, the celestial Mellin
amplitude is
\[
  \Mel\cM_n(\Delta_i; z_i, \bar z_i)
  \;:=\;
  \prod_{i=1}^n \int_0^\infty \omega_i^{\Delta_i - 1}\, d\omega_i\cdot
  \cM_n(\omega_i; z_i, \bar z_i).
\]
By Proposition~\ref{prop:uch-mellin-shadow}(i),
$\Mel\cM_n = \langle \prod_i \cO_{\Delta_i}(z_i, \bar z_i)
\rangle_{\cA^{\mathrm{cel}}}$ with $\cO_{\Delta_i}$ the celestial primary
of conformal weight $\Delta_i$ in the celestial chiral algebra
$\cA^{\mathrm{cel}}(T_4)$.
\end{definition}

\begin{definition}[Conformally soft pole]
\label{def:smsb-soft-pole}
\ClaimStatusProvedElsewhere
A conformal primary $\cO_{\Delta_s}(z, \bar z)$ is \emph{sub${}^{r-2}$
conformally soft} at the pole $\Delta_s = 4 - r$ (equivalently
$\Delta_s = 2 - (r - 2)$, read as "stress-tensor weight $2$ minus
$(r - 2)$ subleading levels"). The residue
\[
  \Res_{\Delta_s = 4 - r}\,
  \Mel\cM_n(\Delta_s; \Delta_j, z_j, \bar z_j)
\]
extracts the degree-$(r - 2)$ coefficient of the low-energy expansion
$\cM_n \sim \sum_{k \geq 0} \omega_s^{k - 1}\, \cS_k \cdot \cM_{n-1}$ as
$\omega_s \to 0$, after Mellin transform; equivalently it is the
Weinberg-type coefficient of the soft-graviton expansion at the
$(r - 2)$-th subleading order.
\end{definition}

\subsection{The explicit Mellin--shadow correspondence}
\label{subsec:smsb-bridge}
%-----------------------------------------------------------------------

Establish the weight-by-weight Mellin--shadow correspondence.
The key point is that the universal kinematic factor in the Mellin
residue at $\Delta_s = 4 - r$ is a Vandermonde-type differential operator
acting on the remaining hard insertions, which is exactly the
kinematic operator appearing in
Hamada--Shiu~\cite[(3.11)--(3.12)]{HamadaShiuSubsubSoft} and
Li--Strominger~\cite[\S 4]{LiStromingerHigherSoft}.

\begin{definition}[Kinematic soft operator]
\label{def:smsb-kinematic-op}
\ClaimStatusProvedHere
For each integer $r \geq 2$ and each soft insertion index $s$, the
kinematic soft operator is
\[
  \cD_s^{(r)}(z_s, \bar z_s)
  \;:=\;
  \sum_{j \neq s}
  \frac{(z_s - z_j)^{r - 1}\,\partial_{z_j} + \text{c.c.}}{(r - 1)!\,
  (z_s - z_j)\,(\bar z_s - \bar z_j)},
\]
extended at $r = 2$ to the Weinberg operator $\sum_{j \neq s}
\omega_j / [\omega_s (z_s - z_j)(\bar z_s - \bar z_j)]$ in the original
momentum variables (equivalent after Mellin transform). In celestial
coordinates $\cD_s^{(r)}$ is a local differential operator of order
$r - 1$ acting on the remaining hard insertions at positions $(z_j,
\bar z_j)$; its kernel is the universal sub${}^{r-2}$-leading soft
graviton kinematic factor of \cite{HamadaShiuSubsubSoft,
LiStromingerHigherSoft}.
\end{definition}

\begin{theorem}[Mellin--shadow correspondence, explicit]
\label{thm:mellin-shadow-correspondence-explicit}
\ClaimStatusProvedHere
\index{Mellin--shadow correspondence!explicit identification}
Let $\cA^{\mathrm{cel}} = \Vir_c$ be the Virasoro celestial chiral
algebra of HT-twisted gravity
(Proposition~\ref{prop:smsb-vir-as-cel}). For every integer $r \geq 2$
and every $n \geq 1$ the conformally soft residue of the Mellin
amplitude admits the explicit Vol~I-coefficient identification
\begin{equation}
\label{eq:smsb-bridge-identity}
  \Res_{\Delta_s = 4 - r}\,
  \Mel\cM_n(\Delta_s; \Delta_j, z_j, \bar z_j)
  \;=\;
  S_r(\Vir_c)\cdot
  \cD_s^{(r)}(z_s, \bar z_s)\cdot
  \Mel\cM_{n-1}(\Delta_j; z_j, \bar z_j),
\end{equation}
with $S_r(\Vir_c) \in \QQ(c)$ the Virasoro shadow coefficient
(Definition~\ref{def:smsb-shadow-coeffs}) and $\cD_s^{(r)}$ the
kinematic soft operator (Definition~\ref{def:smsb-kinematic-op}). The
identity is unconditional at genus $0$; at genus $g \geq 1$ it holds in
the weight-completed category, with the chain-level statement on the
original complex supplied by
Theorem~\ref{thm:uch-gravity-chain-level}.
\end{theorem}

\begin{proof}
We combine Proposition~\ref{prop:uch-mellin-shadow}(ii) with the Vol~I
closed-form identification of
Theorem~\ref{thm:smsb-import-closed-forms}.

\emph{Step 1 (residue = bar-cohomology class).}
Proposition~\ref{prop:uch-mellin-shadow}(ii) gives
\[
  \Res_{\Delta_s = 4 - r}\,\Mel\cM_n
  \;=\;
  \bigl\langle \xi_{r - 1}(z_s, \bar z_s)\cdot
  \prod_{j \neq s} \cO_{\Delta_j}(z_j, \bar z_j)
  \bigr\rangle_{\Vir_c},
\]
where $\xi_{r - 1} \in H^{r - 1}(\Barch(\Vir_c))$ is the bar-cohomology
class of depth $r - 1$.

\emph{Step 2 (shadow class is scalar times kinematic operator).}
By Vol~I Theorem~H and the scalar-lane identification of the shadow
tower, the bar class $\xi_{r - 1}$ decomposes as
\[
  \xi_{r - 1}(z, \bar z)
  \;=\;
  S_r(\Vir_c)\cdot \widehat\xi_{r - 1}(z, \bar z),
\]
where $S_r(\Vir_c)$ is the scalar shadow coefficient
(Definition~\ref{def:smsb-shadow-coeffs}) and $\widehat\xi_{r - 1}$ is
the universal (family-independent) bar class carrying the kinematic
structure. The universal class $\widehat\xi_{r - 1}$ acts on the
remaining primaries by the differential operator $\cD_s^{(r)}$ of
Definition~\ref{def:smsb-kinematic-op}: this is the content of the
$\mathrm d \log$-extraction mechanism of Vol~I Theorem~H, expressed at
the level of primaries via the Arnold form
$\mathrm d \log(z_s - z_j)\,\mathrm d \log(\bar z_s - \bar z_j)$ whose
$(r - 1)$-fold iteration produces the
$(z_s - z_j)^{r - 1}\partial_{z_j}$ kernel of $\cD_s^{(r)}$.

\emph{Step 3 (identification).}
Substituting the decomposition of Step~2 into Step~1 gives
\[
  \Res_{\Delta_s = 4 - r}\,\Mel\cM_n
  \;=\;
  S_r(\Vir_c)\cdot
  \bigl\langle \widehat\xi_{r - 1}(z_s, \bar z_s)\cdot
  \prod_{j \neq s} \cO_{\Delta_j}
  \bigr\rangle_{\Vir_c}
  \;=\;
  S_r(\Vir_c)\cdot \cD_s^{(r)}(z_s, \bar z_s)\cdot \Mel\cM_{n - 1},
\]
where the last equality uses that $\widehat\xi_{r - 1}$ acts by the
kinematic operator $\cD_s^{(r)}$ on the remaining hard
$(n - 1)$-point insertion (the hard-insertion correlator is
$\Mel\cM_{n - 1}$ after resolving the soft insertion). This is
\eqref{eq:smsb-bridge-identity}.

\emph{Step 4 (genus / chain-level scope).}
At genus $0$ the identity is unconditional: Vol~I Theorem~H gives the
bar class, Proposition~\ref{prop:uch-mellin-shadow} gives the Mellin
identification, and Theorem~\ref{thm:smsb-import-closed-forms} supplies
the scalar coefficient. At genus $g \geq 1$ the residue is the $g$-loop
correction (Proposition~\ref{prop:uch-mellin-shadow}(iii)): the
coefficient is the genus-$g$ shadow $\Theta_r^{(g)}(\Vir_c)$; the
statement is unconditional in the weight-completed category (Vol~I
Theorem~MC5) and, via the DS--Hochschild bridge, chain-level on the
original complex
(Theorem~\ref{thm:uch-gravity-chain-level}).
\end{proof}

\subsection{$r = 4$ is the Hamada--Shiu--Li--Strominger sub-sub-leading soft graviton}

The $r = 4$ case is the Hamada--Shiu~\cite{HamadaShiuSubsubSoft} and

\begin{theorem}[$r = 4$ shadow coefficient = HSL sub-sub-leading soft graviton]
\label{thm:r-eq-4-is-leading-subsubleading}
\index{soft graviton theorem!r=4 sub-sub-leading}
\ClaimStatusProvedHere
Let $\cA^{\mathrm{cel}} = \Vir_c$ be the Virasoro celestial chiral
algebra of HT-twisted gravity
(Proposition~\ref{prop:smsb-vir-as-cel}). The sub-sub-leading
($r = 4$) soft-graviton theorem of Hamada--Shiu~\cite{HamadaShiuSubsubSoft}
(\emph{cf.}\ Li--Strominger~\cite[\S 4]{LiStromingerHigherSoft}) equals,
at genus $0$, the Mellin residue
\[
  \Res_{\Delta_s = 0}\,\Mel\cM_n(\Delta_s; \Delta_j, z_j, \bar z_j)
  \;=\;
  \frac{10}{c\,(5 c + 22)}\cdot
  \cD_s^{(4)}(z_s, \bar z_s)\cdot
  \Mel\cM_{n - 1}(\Delta_j; z_j, \bar z_j),
\]
where $\cD_s^{(4)}$ is the cubic-derivative kinematic soft operator of
Definition~\ref{def:smsb-kinematic-op} (order $r - 1 = 3$).
\end{theorem}

\subsection{Angular-momentum and stress-tensor insertions via the shadow}
\label{subsec:smsb-angular-stress}
%-----------------------------------------------------------------------

The Hamada--Shiu--Li--Strominger formulation of higher-soft graviton
theorems involves angular-momentum insertions $J^{\mu \nu}$ and
subleading stress-tensor modes. Both are encoded in the shadow tower.

\begin{definition}[Angular-momentum and stress modes in $\Vir_c$]
\label{def:smsb-angular-stress-modes}
\ClaimStatusProvedElsewhere
The Virasoro modes $L_n$ of $\Vir_c$, acting on a celestial primary
$\cO_\Delta(z, \bar z)$, encode the angular-momentum / stress-tensor
content of HT-twisted gravity. The mode $L_{-1}$ is the translation
(energy--momentum) generator; $L_0$ is the dilatation (conformal
weight) generator; $L_{-k}$ for $k \geq 2$ are the sub${}^{k - 1}$
leading angular-momentum / stress-tensor descendants
\cite[\S 3]{DonnayFotopoulosPasterskiTaylor21,StromingerLectures2018}.
\end{definition}

\begin{theorem}[Angular momentum and stress tensor via shadow]
\label{thm:angular-momentum-stress-tensor-via-shadow}
\index{angular momentum!via shadow tower}
\index{celestial stress tensor!via shadow tower}
\ClaimStatusProvedHere
At every depth $2 \leq r \leq 8$, the kinematic soft operator
$\cD_s^{(r)}$ of
Definition~\ref{def:smsb-kinematic-op} decomposes as a finite sum of
angular-momentum and stress-tensor insertions:
\[
  \cD_s^{(r)}(z_s, \bar z_s)
  \;=\;
  \sum_{j \neq s}
  \sum_{k = 0}^{r - 1}
  \beta_{r, k}\cdot
  L_{-k}^{(j)}\, \bar L_{-(r - 1 - k)}^{(j)}
  \cdot G_j^{(r)}(z_s - z_j, \bar z_s - \bar z_j),
\]
where $L_{-k}^{(j)}$ acts on the $j$-th celestial primary
$\cO_{\Delta_j}(z_j, \bar z_j)$, $G_j^{(r)}$ is a universal
kinematic Green's function on $(z_s - z_j, \bar z_s - \bar z_j)$, and
$\beta_{r, k} \in \QQ$ are universal numerical coefficients.
Substituting into \eqref{eq:smsb-bridge-identity} gives the
Hamada--Shiu--Li--Strominger form of the
sub${}^{r - 2}$-leading soft graviton theorem as a finite sum of
angular-momentum / stress-tensor insertions multiplied by the scalar
shadow coefficient $S_r(\Vir_c)$.

Concretely:
\begin{itemize}
\item $r = 2$: $\cD_s^{(2)} = \sum_{j \neq s}
L_{-1}^{(j)} / [(z_s - z_j)(\bar z_s - \bar z_j)]$ (Weinberg
momentum).
\item $r = 3$: $\cD_s^{(3)} = \sum_{j \neq s}
\bigl[L_{-1}^{(j)} \bar L_{-1}^{(j)} + L_{-2}^{(j)} +
\bar L_{-2}^{(j)}\bigr]\cdot G_j^{(3)}$ (Cachazo--Strominger
subleading, angular momentum + weight).
\item $r = 4$: $\cD_s^{(4)}$ contains $L_{-1}^{(j)} L_{-2}^{(j)}
+ L_{-3}^{(j)} + $ mixed holomorphic / antiholomorphic terms
(HSL sub-sub-leading, higher angular momentum + stress-tensor
descendants).
\item $r = 5, 6, 7, 8$: $\cD_s^{(r)}$ contains iterated Virasoro
descendants $L_{-i_1} L_{-i_2} \cdots$ with $\sum i_k = r - 1$,
each weighted by universal rational $\beta_{r, k}$.
\end{itemize}
\end{theorem}

\begin{proof}
The kinematic soft operator $\cD_s^{(r)}$ is the Mellin transform of
the low-energy expansion kernel
$\omega_s^{r - 2}\, \partial^{r - 1} / [(z_s - z_j)(\bar z_s - \bar z_j)]$.
By Proposition~\ref{prop:uch-mellin-shadow}(ii), the operator acts on a
celestial primary $\cO_{\Delta_j}(z_j, \bar z_j)$ through the Virasoro
modes; the Taylor expansion of $(z_s - z_j)^{r - 1}\partial_{z_j}$
around $z_s$ expresses the action of $\cD_s^{(r)}$ as a sum over
Virasoro descendants $L_{-i}$ applied to $\cO_{\Delta_j}$, with the
Green's-function factor $G_j^{(r)}$ encoding the residual kinematic
$z_s - z_j$ dependence. The universal coefficients $\beta_{r, k}$ are
the Taylor coefficients of the expansion and are rational for every
$r$.

The specific decompositions at $r = 2, 3, 4$ match the Weinberg,
Cachazo--Strominger, and Hamada--Shiu--Li--Strominger formulas
respectively:
\begin{itemize}
\item $r = 2$ matches \cite{Weinberg65,StromingerLectures2018}: the
Weinberg operator is $\sum_j \omega_j / [\omega_s (z - z_j)(\bar z -
\bar z_j)]$, which Mellin-transforms to $\sum_j L_{-1}^{(j)}$ acting
by translation on each hard primary.
\item $r = 3$ matches
\cite{CachazoStrominger14,DonnayFotopoulosPasterskiTaylor21}: the
subleading soft factor involves angular momentum $J^{\mu \nu}_j$,
which in celestial coordinates becomes a combination of
$L_{-1}^{(j)}, L_{-2}^{(j)}$, and their antiholomorphic counterparts.
\item $r = 4$ matches \cite{HamadaShiuSubsubSoft,LiStromingerHigherSoft}:
the sub-sub-leading factor involves second-order angular momentum and
subleading stress-tensor descendants $L_{-3}^{(j)}$, exactly the
$\beta_{4, k}$ decomposition above.
\end{itemize}
At $r = 5, 6, 7, 8$, the decomposition follows by the same Taylor
expansion; the exact values $\beta_{r, k}$ are elementary rational
combinatorial coefficients of the kinematic kernel.
\end{proof}

\begin{corollary}[Unconditional soft hierarchy for $r \leq 8$]
\label{cor:soft-hierarchy-unconditional-r-leq-8}

\begin{proof}
