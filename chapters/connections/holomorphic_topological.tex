A four-dimensional holomorphic-topological field theory restricts to a
chiral algebra on each holomorphic boundary. The bar complex of that
chiral algebra classifies twisting morphisms (couplings to the
Koszul dual); the chiral derived centre
$C^\bullet_{\mathrm{ch}}(\cA,\cA)$ computes the universal
bulk; bar-cobar inversion $\Omega(B(\cA)) \simeq \cA$
recovers the boundary algebra itself. The modular
convolution algebra~$\gAmod$
(Definition~\ref{def:modular-convolution-dg-lie}) organizes the full boundary-to-bulk correspondence
at all genera. The Swiss-cheese operad, descent
calculus, and recognition theorem are developed in Volume~II\@. The bar complex of
this volume, equipped with its coassociative coproduct
(Theorem~\ref{thm:bar-swiss-cheese}), provides the algebraic input.

\begin{proposition}[Bar complex and genus-zero HT data; \ClaimStatusProvedHere]
\label{prop:ht-bar-genus-zero}
Let $T$ be a $4$d $\mathcal{N}=2$ theory with HT boundary chiral
algebra~$\cA_T$. The bar complex $B(\cA_T)$ carries:
\begin{enumerate}[label=\textup{(\roman*)}]
\item a coassociative coproduct
 \textup{(}Theorem~\textup{\ref{thm:bar-swiss-cheese}}\textup{)}
 encoding the topological-direction factorization;
\item a differential $d_B$ whose cohomology is the Koszul dual
 coalgebra; the Verdier dual
 $\mathbb{D}_{\Ran}\,B(\cA_T) \simeq B(\cA_T^!)$
 recovers the homotopy Koszul dual algebra $(\cA_T^!)_\infty$
 \textup{(}Theorem~\textup{\ref{thm:bar-cobar-isomorphism-main}},
 Convention~\textup{\ref{conv:bar-coalgebra-identity}}\textup{)};
\item a modular operad algebra structure over the Feynman transform of
 the commutative modular operad~$\mathrm{FCom}$
 \textup{(}Theorem~\textup{\ref{thm:bar-modular-operad}}\textup{)},
 with $\partial^2 = 0$ at all genera a formal consequence.
\end{enumerate}
\end{proposition}

\begin{proof}
Specialization of the general bar--modular operad theorem
(Theorem~\ref{thm:bar-modular-operad}) to the boundary chiral
algebra $\cA_T$ of the HT twist. The coassociative coproduct is
the $\Eone$ dg coassociative coalgebra structure of
Theorem~\ref{thm:bar-swiss-cheese}. The Koszul dual identification
$\mathbb{D}_{\Ran}\,B(\cA_T) \simeq B(\cA_T^!)$ is Verdier
duality on $\Ran(X)$ applied to the bar coalgebra; bar-cobar
inversion on the Koszul locus
(Theorem~\ref{thm:bar-cobar-isomorphism-main}) supplies the
compatibility $\Omega(B(\cA_T)) \simeq \cA_T$ needed to transfer
the statement between the two sides.
\end{proof}

\begin{equation}\label{eq:ht-theta}
\Theta_{\cA_T}
\;:=\; D_{\cA_T} - d_0
\;\in\; \MC(\gAmod)

\textup{(}Theorem~\textup{\ref{thm:mc2-bar-intrinsic}}\textup{)}
satisfies
$D_{\cA_T}\,\Theta_{\cA_T}
+ \tfrac{1}{2}[\Theta_{\cA_T},\Theta_{\cA_T}] = 0$

$\Theta_{\cA_T}^{\leq r}$ are the shadow obstruction tower
\textup{(}Definition~\textup{\ref{def:shadow-postnikov-tower}}\textup{)};
the scalar shadow on the uniform-weight locus is
$\kappa(\cA_T) \cdot \lambda_g$
\textup{(}Theorem~\textup{\ref{thm:genus-universality}};

Theorem~\ref{thm:mc2-bar-intrinsic} to $\cA = \cA_T$. The MC

\section{S-duality as Koszul duality}
\label{sec:ht-s-duality-koszul}
\index{S-duality!Koszul duality}

\begin{conjecture}[S-duality = Koszul duality]
\label{conj:ht-s-duality-koszul}
\index{S-duality!bar-cobar}
Let $T$ be a $4$d $\mathcal{N}=4$ theory with S-dual~$T^\vee$.
The HT twist produces a Koszul pair:
\begin{equation}\label{eq:ht-s-duality-koszul}
\cA_{T^\vee} \;\simeq\; \cA_T^!.
\end{equation}
In particular, for $T = \mathcal{N}=4$ $\mathrm{SYM}$ with gauge group~$G$:
\begin{equation}\label{eq:ht-langlands-koszul}
\cA^!(\widehat{\mathfrak{g}}_k)
\;\simeq\;
\cA(\widehat{{}^L\!\mathfrak{g}}_{k^\vee}),
\qquad
k^\vee = -h^\vee_{{}^L\!G} + 1/k.
\end{equation}
\end{conjecture}

\begin{remark}[Discrepancy at finite coupling]
\label{rem:ht-s-duality-discrepancy}
\index{S-duality!discrepancy from Koszul duality}
The conjecture is generically \emph{false} as a strict identity.
For $\mathcal{N}=4$ $\mathrm{SU}(N)$ SYM at coupling $\Psi = \theta/(2\pi) + 4\pi i/g^2$,
the S-duality and Koszul duality operations on the HT chiral algebra
differ by an explicit discrepancy
\[
 \Delta(\Psi, N)
 \;=\;
 \frac{N^2 + 2N - 1}{2N}\,\Bigl(\Psi + \frac{1}{\Psi}\Bigr),
\]
which vanishes only when $\Psi^2 = -1$ (no real solutions).
In particular, the two operations coincide on the self-dual locus
$\Psi = \pm i$ (the $\mathbb{Z}_4$ fixed point of S-duality),
but are generically distinct: S-duality permutes the full
$\mathrm{SL}(2,\mathbb{Z})$-orbit of coupling constants,
while Koszul duality acts on the algebraic structure of the
chiral algebra at a \emph{fixed} coupling.
Conjecture~\ref{conj:ht-s-duality-koszul} should therefore
be read as asserting an equivalence at the level of
\emph{universal} (coupling-independent) algebraic structures,
not as a pointwise identity at finite~$\Psi$.
\end{remark}

\begin{remark}[Proved instances]
\label{rem:ht-s-duality-proved}
For $\cA_T = \widehat{\mathfrak{g}}_k$ at generic level, the Koszul
dual $\cA_T^!$ is obtained by applying Verdier duality on
$\mathrm{Ran}(X)$ to the bar coalgebra $B(\cA_T)$ (the Koszul-dual
functor), and the bar-cobar inversion
$\Omega(B(\cA_T)) \simeq \cA_T$
(Theorem~\ref{thm:bar-cobar-isomorphism-main}) recovers the boundary
algebra itself from its bar coalgebra, supplying the compatibility
between the two sides. The S-duality
conjecture reduces to comparing this Verdier-dual algebra with the
HT twist of the Langlands dual theory. The holographic datum
$\mathcal{H}(T)$
(Definition~\ref{def:holographic-modular-koszul-datum}) packages
this comparison: the MC element~$\Theta_{\cA_T}$ and the shadow
connection~$\nabla^{\mathrm{hol}}_T$ are invariant data of the
S-duality equivalence class.
\end{remark}

\begin{remark}[Four dualities comparison]
\label{rem:ht-four-dualities-comparison}
\index{duality!four operations comparison|textbf}
\index{Klein four-group!duality operations}
Four duality operations act on the boundary chiral algebra
$\cA_T$ of a~$4$d HT system. They are generically distinct:
on $\mathrm{gl}_N$ at coupling~$\Psi$,
the Feigin--Frenkel involution $\Psi \mapsto -\Psi$
and S-duality $\Psi \mapsto 1/\Psi$

\section{Twistor anomaly and the Deligne exceptional series}
\label{sec:ht-twistor-deligne}
\index{twistor space!anomaly cancellation|textbf}
\index{Deligne exceptional series!twistor anomaly|textbf}
\index{quartic Casimir!twistor selection rule}
\index{Green--Schwarz mechanism!twistor space}

Holomorphic Chern--Simons theory on twistor space
$\mathbb{P}^3$~\cite{Costello2111} furnishes the
holomorphic-topological origin of self-dual Yang--Mills in four
dimensions. Quantum consistency of this theory imposes a gauge
group selection rule whose algebraic content is quartic, not
quadratic: it operates at the shadow obstruction tower level of
the genus-$2$ shell, not at the genus-$1$ level where~$\kappa$
lives.

\begin{proposition}[Twistor anomaly selects the Deligne exceptional series;
\ClaimStatusProvedElsewhere]
\label{prop:ht-twistor-deligne}
\index{Costello!twistor anomaly}
\index{Deligne exceptional series!Costello selection}
The gauge algebras~$\mathfrak{g}$ for which holomorphic
Chern--Simons theory on twistor space
$\mathbb{P}^3 \setminus \mathbb{P}^1$ admits a consistent
one-loop quantization \textup{(}with a single holomorphic axion
field implementing the Green--Schwarz
mechanism\textup{)}
are exactly the Deligne exceptional
series~\cite{Deligne96, Cvitanovic08}:
\begin{equation}\label{eq:ht-deligne-series}
\mathfrak{g} \;\in\;
\bigl\{\,
 \mathfrak{sl}_2,\;
 \mathfrak{sl}_3,\;
 \mathfrak{so}_8,\;
 G_2,\;
 F_4,\;
 E_6,\;
 E_7,\;
 E_8
\,\bigr\}.
\end{equation}
\end{proposition}

\begin{proof}[Source]
Costello~\cite{Costello2111}.
The one-loop anomaly polynomial of holomorphic Chern--Simons
on~$\mathbb{P}^3$ decomposes as
$I_6 = a_2\,\operatorname{tr}(F^2)^2
+ b_4\,\operatorname{tr}'(F^4)$,
where $\operatorname{tr}'(F^4)$ denotes the component of the
adjoint fourth-order invariant independent of
$(\operatorname{tr} F^2)^2$.
The Green--Schwarz axion cancels the factorized
$(\operatorname{tr} F^2)^2$ term. The residual anomaly
$b_4\,\operatorname{tr}'(F^4)$ vanishes if and only if
$\mathfrak{g}$ has no independent quartic Casimir operator, or
equivalently,
$\operatorname{tr}_{\mathrm{adj}}(F^4)$ is proportional to
$(\operatorname{tr}_{\mathrm{adj}} F^2)^2$.
This condition characterizes the Deligne exceptional
series.
\end{proof}

\begin{remark}[Genus-level mechanism: quartic, not quadratic]
\label{rem:ht-twistor-genus-level}
\index{twistor space!genus-level mechanism}
At the $\kappa$~level (genus~$1$), the anomaly cancels for
\emph{every} simple gauge group: the effective modular
characteristic
$\kappa_{\mathrm{eff}} = \kappa(\widehat{\mathfrak{g}}_0)
+ \kappa(\text{axion}) = 0$
is achievable by tuning the axion coupling, since
$\kappa(\widehat{\mathfrak{g}}_0) = \dim(\mathfrak{g})/2$
is nonzero for all~$\mathfrak{g}$.
The selection operates one level higher: the quartic
anomaly $\operatorname{tr}'(F^4)$ is the genus-$2$
obstruction, living in the quartic shell of the shadow
obstruction tower. Groups with
$\operatorname{tr}'(F^4) \neq 0$ have a non-cancellable
obstruction at this level.
\end{remark}

\begin{remark}[Shadow interpretation:
$Q^{\mathrm{contact}}$ factors through~$\kappa$]
\label{rem:ht-twistor-shadow}
\index{quartic contact invariant!twistor selection}
\index{shadow obstruction tower!twistor anomaly}
In the language of the shadow obstruction tower
(\S\ref{sec:shadow-depth-classification}), the three
conditions
\begin{enumerate}[label=\textup{(\alph*)}]
\item $\mathfrak{g}$ is Costello-allowed on twistor space,
\item $\mathfrak{g}$ lies in the Deligne exceptional series,
\item $Q^{\mathrm{contact}}_{\widehat{\mathfrak{g}}}$
 factors through~$\kappa$, i.e.\
 the quartic shadow is determined by the quadratic data alone,
\end{enumerate}
are equivalent.
For all affine Kac--Moody algebras
$\widehat{\mathfrak{g}}_k$ the shadow class is~$\mathsf{L}$
and $r_{\max} = 3$
(Theorem~\ref{thm:cubic-gauge-triviality}), so
the quartic shadow vanishes identically: the distinction
between Deligne and non-Deligne is invisible at the shadow
depth level. It becomes visible in the \emph{ambient}
anomaly polynomial, where the independent quartic Casimir
$\operatorname{tr}'(F^4)$ contributes a genus-$2$ term
that the Green--Schwarz axion cannot absorb.

Computational verification:
\texttt{compute/lib/twistor\_anomaly\_koszul\_engine.py}
(99~tests, three-way equivalence
\textup{(a)}$\Leftrightarrow$\textup{(b)}$\Leftrightarrow$\textup{(c)}
verified for all simple algebras through rank~$8$).
\end{remark}

\begin{remark}[Three regimes of quartic factorization]
\label{rem:ht-twistor-three-regimes}
\index{Deligne exceptional series!three regimes}
\index{quartic Casimir!three regimes}
The quartic factorization condition partitions the Deligne
series into three regimes, distinguished by the mechanism
preventing an independent degree-$4$ Casimir invariant.

\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Low rank}
 ($\mathfrak{sl}_2$, $\mathfrak{sl}_3$, $G_2$).
 The rank is at most~$2$, so the ring of Casimir invariants
 has no generator in degree~$4$: the Casimir degrees of
 $\mathfrak{sl}_2$ are $\{2\}$, of $\mathfrak{sl}_3$ are
 $\{2,3\}$, and of $G_2$ are $\{2,6\}$.

\item \emph{No exponent~$3$}
 ($F_4$, $E_6$, $E_7$, $E_8$).
 The rank is $\geq 4$, but the smallest exponent $> 1$ is
 $\geq 4$ (the exponents of $F_4$ are
 $\{1,5,7,11\}$; of $E_6$, $\{1,4,5,7,8,11\}$;
 of $E_7$, $\{1,5,7,9,11,13,17\}$;
 of $E_8$, $\{1,7,11,13,17,19,23,29\}$).
 In particular, $3$~is not an exponent, so $4$~is not a
 Casimir degree.

\item \emph{Triality}
 ($\mathfrak{so}_8 = D_4$).
 The rank is~$4$ and the exponents $\{1,3,3,5\}$
 \emph{do} include~$3$, giving Casimir degrees
 $\{2,4,4,6\}$ with \emph{two} independent quartic
 invariants.
 The $S_3$ outer automorphism group (triality) forces
 both quartic invariants to decompose through
 $(\operatorname{tr} F^2)^2$: any
 $S_3$-invariant quartic on the Cartan subalgebra of
 $D_4$ is a polynomial in the elementary symmetric
 functions, and the only $S_3$-invariant of degree~$4$
 in three variables of degree~$2$ is the square
 $(\sigma_1)^2 = (\operatorname{tr} F^2)^2$.
\end{enumerate}

\noindent
Every non-Deligne simple algebra of rank~$\geq 2$ has
$3$~as an exponent (equivalently, $4$~as a Casimir
degree) and lacks a triality symmetry to enforce
factorization, producing an independent
$\operatorname{tr}'(F^4)$ and an uncancellable twistor
anomaly.
\end{remark}

\begin{remark}[Three invariants: $b_0^{4\mathrm{d}}$,
$b_0^{\mathrm{hol}}$, and~$\kappa$]
\label{rem:beta-function-kappa-distinction}
\index{beta function!modular characteristic distinction|textbf}
\index{modular characteristic!beta function distinction}
\index{asymptotic freedom!modular}
\index{one-loop level shift}
The one-loop beta function coefficient~$b_0$, the holomorphic
one-loop level shift~$b_0^{\mathrm{hol}}$, and the modular
characteristic~$\kappa$ are three \emph{distinct} invariants
of a gauge-theoretic system:
\begin{enumerate}[label=\textup{(\roman*)}]
\item $b_0^{4\mathrm{d}} = (11/3)\,N$ for pure $\mathrm{SU}(N)$
 Yang--Mills: the standard coefficient governing the
 running of the four-dimensional coupling. The factor $11/3$
 requires the full $4\mathrm{d}$ computation (gluon, ghost,
 and matter loops).

\item $b_0^{\mathrm{hol}} = h^\vee$ for the holomorphic twist:
 the one-loop level shift of the boundary current algebra
 $\widehat{\mathfrak{g}}_{k}$, computable as a holomorphic
 Euler characteristic on the twistor
 fiber~\cite{Costello2111}. This shifts the effective level to
 $k_{\mathrm{eff}} = k - h^\vee$.

\item $\kappa(\widehat{\mathfrak{g}}_k)
 = \dim(\mathfrak{g})(k + h^\vee)/(2h^\vee)$: the modular
 characteristic of the boundary VOA
 \textup{(}Theorem~\textup{\ref{thm:modular-characteristic}}\textup{)},
 controlling the genus expansion
 $F_g = \kappa \cdot \lambda_g^{\mathrm{FP}}$ At $k = 0$: $\kappa = \dim(\mathfrak{g})/2$.
\end{enumerate}
There is no universal proportionality constant between these
invariants. The ratio $b_0^{4\mathrm{d}}/\kappa$
depends on~$N$: for $\mathrm{SU}(2)$ it is $22/9$; for
$\mathrm{SU}(3)$ it is $11/4$; as $N \to \infty$ it tends
to zero.

The correct bridge is the \emph{one-loop level shift}:
$\delta k = -b_0^{\mathrm{hol}} = -h^\vee$ shifts the level of
the boundary VOA, inducing
$\delta\kappa = -\dim(\mathfrak{g})/2$. The condition
$\kappa(k_{\mathrm{eff}}) = 0$ (vanishing of the effective
modular characteristic) is equivalent to
$k_{\mathrm{eff}} = -h^\vee$, i.e.\ the critical level. This
is \emph{modular asymptotic freedom}: the one-loop correction
drives~$\kappa$ toward zero, the uncurved point of the bar
complex.
Computational verification:
\texttt{compute/lib/beta\_function\_kappa\_engine.py}
(149~tests).
\end{remark}

\section{Celestial holography and boundary transfer}
\label{subsec:celestial-boundary-transfer}

Kaluza--Klein reduction of six-dimensional holomorphic theories to
three-dimensional holomorphic-topological theories transfers
$A_\infty$- and $L_\infty$-structures to the tangential Cauchy--Riemann
cohomology $H_b^{0,\bullet}(S^3)$. These transferred structures form
the algebraic skeleton of celestial holography in the twisted setting.

The \emph{wedge $A_\infty$-algebra} construction quotients by
composite operators. The single-particle subalgebra is formal,
depending only on the linearized dg algebra; all genuinely
nonlinear content concentrates in the composite sector, measured by
obstruction classes $\operatorname{Ob}_r$ in the bar-filtered
coderivation complex.

For holomorphic BF, the transferred $A_\infty$-algebra admits closed
highest-weight ladder formulas in every degree, governed by simplex
integrals and a simplex summation law. For Poisson BF, the
odd highest-weight ladder of the transferred $L_\infty$-deformation of
$T^*[1]\operatorname{Ham}(\mathbb{C}^2)$ is realized by
first-order differential operators
$D_m = -\partial_t \circ M_{t^{2m+2}}$ on $\mathbb{C}[t]$, the
odd positive Witt modes acting on the density line (the
\emph{Airy--Witt normal form}).

For $W$-type Poisson vertex algebras, the first nonlinear
obstruction class $\operatorname{Ob}_1$ depends only on the linearized
Lie conformal structure and the lowest nonlinear homogeneous part of
the generator $\lambda$-brackets. For the classical $W_3$ PVA this
means the first celestial higher-spin correction is controlled entirely
by the composite field $\frac{16}{22+5c}\Lambda$ in the
$\{W_\lambda W\}$ bracket, a precise connection to the $W_3$
composite-field structure computed in
\S\ref{sec:w3-composite-complete}.

\begin{remark}[Celestial obstruction tower in the convolution algebra]
\label{rem:ht-celestial-obstruction-convolution}
\index{celestial holography!convolution obstruction}
The obstruction classes $\operatorname{Ob}_r$ in the bar-filtered
coderivation complex are the genus-$0$ shadow obstruction tower projections
$\Theta_{\cA_T}^{(0,r+1)}$ restricted to the single-particle
sector. The wedge construction is the composite-free quotient of the
collision filtration
(Definition~\ref{def:ht-collision-filtration}).
\end{remark}

\begin{proposition}[Witten diagrams as genus-$0$ shadow projections]
\label{prop:witten-diagram-shadow-projection}
\ClaimStatusProvedHere
\index{Witten diagram!shadow projection|textbf}
\index{celestial holography!Witten diagram identification}
\index{BCFW recursion!MC recursion}
\index{soft graviton!shadow coefficient}
\index{Parke--Taylor formula!shadow projection}
Let~$\cA$ be a chirally Koszul algebra with universal MC element
$\Theta_\cA \in \operatorname{MC}(\gAmod)$
\textup{(}Theorem~\textup{\ref{thm:mc2-bar-intrinsic})}.
The $n$-point tree-level Witten diagram amplitude in $4\mathrm{d}$
holomorphic Chern--Simons theory~\cite{CostelloP2201}
equals the genus-$0$, degree-$n$ shadow projection:
\begin{equation}\label{eq:witten-shadow-identification}
 \mathcal{A}_n^{\mathrm{tree}}(z_1, \ldots, z_n)
 \;=\;
 \mathrm{Sh}_{0,n}(\Theta_\cA).
\end{equation}
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{MC recursion $=$ BCFW\@.}
 The MC equation at genus~$0$, decomposed by degree, gives:
 $D_2\,\Theta_n
 + \sum_{j+k=n} [\Theta_j,\, \Theta_k]_n = 0$.
 The bracket $[\Theta_j,\, \Theta_k]$ glues two lower-degree
 amplitudes at a shared marked point with a propagator
 insertion. This recursion is the
 Britto--Cachazo--Feng--Witten on-shell recursion for the
 holomorphic twist.

\item \emph{Soft limits.}
 The $(n{+}1)$-point amplitude in the conformally soft limit
 $z_{n+1} \to \infty$ satisfies
 $\mathcal{A}_{n+1} \to S^{(0)} \cdot \mathcal{A}_n$
 with leading soft factor
 $S^{(0)} = \sum_k (z_{n+1} - z_k)^{-1}$.
 The coefficient of the leading soft factor is
 $\kappa = S_2$\textup{;}
 the subleading soft graviton theorem is controlled by the
 cubic shadow
 $S_3 = 2$
 \textup{(}universal for all chiral algebras with a stress
 tensor\textup{)}.

\item \emph{Color-ordered amplitudes.}
 For affine $\mathfrak{sl}_N$ at level~$k$, the color-ordered
 partial amplitude in the trace basis recovers the
 Parke--Taylor formula:
 $\mathcal{A}_n^{\mathrm{MHV}}
 = \langle ij \rangle^4 / \prod_{k=1}^{n}
 \langle k,\,k{+}1\rangle$.
 The MC recursion at genus~$0$ produces exactly the
 Berends--Giele current recursion for color-ordered gluon
 amplitudes.
\end{enumerate}
\end{proposition}

\begin{proof}
The genus-$0$ shadow projection
$\mathrm{Sh}_{0,n}(\Theta_\cA)$ is a map
$V^{\otimes n} \to H_*(\overline{\mathcal{M}}_{0,n})$
(Construction~\ref{constr:shadow-extraction-explicit});
at genus~$0$ the moduli space
$\overline{\mathcal{M}}_{0,n} \cong \mathrm{FM}_n(\mathbb{P}^1)$
parametrizes configurations of~$n$ marked points on~$\mathbb{P}^1$.
The tree-level Witten diagrams in $4\mathrm{d}$ holomorphic
Chern--Simons compute the same object: for each tree~$\Gamma$
with $n$~external legs, the Feynman integrand is
$\prod_{\text{edges}\,(i,j)} (z_i - z_j)^{-1}$, and the sum
over planar trees gives the graph sum
realizing~$\mathrm{Sh}_{0,n}$.

Part~(i): the MC equation
$D\Theta + \frac{1}{2}[\Theta,\Theta] = 0$ at genus~$0$,
projected to degree~$n$, reads
$D_2\,\Theta_n^{(0)}
= -\sum_{j+k=n} [\Theta_j^{(0)},\,\Theta_k^{(0)}]$.
The bracket inserts a propagator $(z_a - z_b)^{-1}$ between
genus-$0$ amplitudes of degrees $j$ and~$k$ and sums over all
ways to partition the external legs. This is the content of the
BCFW recursion for the color-stripped amplitude.

Part~(ii): the soft limit
extracts the degree-$(n{+}1)$ projection at a degenerate
configuration where one marked point escapes to infinity.
The leading contribution
$S^{(0)} = \sum_k (z_{n+1} - z_k)^{-1}$ has coefficient
$S_2 = \kappa$ (the degree-$2$ shadow). The subleading soft
factor involves $S_3$, the degree-$3$ shadow coefficient. For
any chiral algebra with a stress tensor, $S_3 = 2$
(Virasoro: the $L_{-1}$ Ward identity forces
$S_3 = 2$ universally).

Part~(iii): for affine $\mathfrak{sl}_N$, the r-matrix
$r(z) = k\,\Omega/z$
and the color-ordering restrict the sum over trees to planar
binary trees with leaves in cyclic order. The resulting
expression is the Parke--Taylor formula by direct computation.
Computational verification:
\texttt{compute/lib/celestial\_ope\_shadow\_projections\_engine.py}
(73~tests).
\end{proof}

\begin{remark}[Comparison with the Gaiotto programme]
\label{rem:gaiotto-programme-comparison}
\index{Gaiotto programme!comparison|textbf}
\index{CoHA!bar duality|textbf}
\index{Y-algebra@$Y$-algebra!Gaiotto programme}
\index{GKW higher operations!comparison}

The body of work by Gaiotto and
collaborators~\cite{gaiotto-rapchak,CostelloGaiotto2020,GKW2025}
develops HT field theory from the gauge-theoretic side. The
relationship with this monograph has three layers.

\emph{Structural identification.}
The $E_1$ dg coalgebra structure on the bar complex
$B(\cA_T)$
(Theorem~\ref{thm:bar-swiss-cheese}: coassociative coproduct
in the $\bR$-direction, bar differential from the chiral product in
the $\bC$-direction) is the algebraic skeleton of the HT
twist on $\bC \times \bR$. The $\bC$-direction encodes the
chiral algebra; the $\bR$-direction encodes the topological
coproduct. The $\SCchtop$ structure emerges in the chiral derived center
pair $(C^\bullet_{\mathrm{ch}}(\cA_T,\cA_T),\, \cA_T)$; the descent calculus, recognition theorem, and
operadic structure are developed in
Volume~II\@.

\emph{Formality and shadow depth.}
The higher operations of Gaiotto--Kulp--Wu~\cite{GKW2025}
classify boundary VOAs as \emph{formal} (all higher
$A_\infty$-operations vanish on cohomology) or
\emph{non-formal}. The shadow depth classification
(Theorem~\ref{thm:shadow-archetype-classification})
strictly refines this binary partition into four classes:
formal boundary VOAs split into
$\mathsf{G}$ ($r_{\max} = 2$, Heisenberg),
$\mathsf{L}$ ($r_{\max} = 3$, affine), and
$\mathsf{C}$ ($r_{\max} = 4$, $\beta\gamma$), while
the non-formal class corresponds to
$\mathsf{M}$ ($r_{\max} = \infty$, Virasoro/$W$-type).
The $Y$-algebras of Gaiotto--Rap\v{c}\'ak~\cite{gaiotto-rapchak}
are all chirally Koszul
(Theorem~\ref{thm:y-algebra-koszulness}); their shadow
classes are $\mathsf{G}$ and $\mathsf{M}$ only (the
intermediate classes $\mathsf{L}$, $\mathsf{C}$ require
matter content beyond the corner configuration).

\emph{CoHA--bar duality.}
For an ADE quiver~$Q$ with associated affine algebra
$\widehat{\mathfrak{g}}_Q$, the cohomological Hall
algebra multiplication (extension of quiver
representations) dualizes to the bar comultiplication
(factorization splitting on Ran):
\begin{equation}\label{eq:coha-bar-character-duality}
 \chi_{\mathrm{CoHA}(Q)}(q)
 \;=\;
 \chi_{B(\widehat{\mathfrak{g}}_Q)}(q)
 \;=\;
 \prod_{n \geq 1}(1 - q^n)^{-\dim \mathfrak{g}_Q}.
\end{equation}
This character-level identity lifts to a structural duality
of vertex bialgebras: the vertex coproduct on CoHA
constructed by Latyntsev~\cite{Latyntsev23} dualizes to the
bar coproduct on $B(\widehat{\mathfrak{g}}_Q)$, and both
produce the Yangian $Y(\mathfrak{g}_Q)$ as the associated
graded of compatible filtrations. For the $A_2$ quiver at
dimension vector $(1,1,0)$, the chain-level identification
(extension correspondence = deconcatenation on $B^2$)
is verified directly: the vertex bialgebra axiom
$\Delta^v(Y(a,z)b) = Y^{(2)}(\Delta^v(a),z)\,\Delta^v(b)$
is exactly the statement that the bar differential is a
coderivation of the tensor coalgebra.
Computational verification:
\texttt{compute/lib/theorem\_coha\_bar\_duality\_engine.py}
(84~tests).

The chain-level duality is verified for the $A_2$ quiver at
dimension vector~$(1,1,0)$
(\texttt{coha\_chain\_level\_duality\_engine.py}, 86~tests).
At this dimension vector, the representation space
$\operatorname{Rep}(A_2, (1,1,0)) = \operatorname{Hom}(\bC, \bC)
\cong \bC$ is one-dimensional, the extension space
$\operatorname{Ext}^1_Q((0,1,0),(1,0,0)) \cong \bC$ is
one-dimensional (from the single arrow $0 \to 1$), and the
duality reduces to a scalar identity. The structural content
lies in the vertex bialgebra axiom
(Latyntsev~\cite{Latyntsev23}):
\begin{equation}\label{eq:jkl-vertex-bialgebra}
 \Delta^v(Y(a,z)\,b)
 \;=\;
 Y^{(2)}(\Delta^v(a),z)\,\Delta^v(b),
\end{equation}
which is the bar coderivation property in vertex-algebraic
language. The vertex coproduct $\Delta^v$ at~$(1,1,0)$ has a
singular term $z^{-1}$ arising from the Euler form
$\langle (1,0,0), (0,1,0) \rangle = -1$: the pole order in
the vertex coproduct is controlled by the
Euler pairing of the quiver, exactly as the bar differential
pole order is controlled by the OPE. This is the
chain-level incarnation of the CoHA--bar duality beyond
character matching: the Hecke correspondence
(extension of representations) dualizes to bar deconcatenation
as chain maps, not merely as graded dimensions.

\emph{Complementary scope.}
Three items in the Gaiotto programme have no counterpart here:
\begin{enumerate}[label=\textup{(\alph*)}]
\item \emph{Explicit boundary VOA identification from gauge data.}
 Gaiotto--Rap\v{c}\'ak~\cite{gaiotto-rapchak} identify the
 boundary VOA of a specific gauge configuration (e.g.\
 Neumann $\to$ symplectic bosons,
 Nahm pole $\to$ principal $\mathcal{W}$-algebra) by
 direct gauge-theoretic construction. Our framework takes
 $\cA_T$ as input and computes invariants; it does not
 produce $\cA_T$ from the Lagrangian.

\item \emph{Coulomb branch quantization.}
 The mathematical Coulomb branch
 $\cM_C(T)$~\cite{BFN18} carries a
 deformation-quantization structure computed by gauge-theoretic
 methods (monopole operators, Hecke modifications). Our MC
 moduli $\operatorname{MC}(\gAmod)$ parametrize the
 corresponding algebraic deformation space, but the
 identification $\cM_C(T) \cong \operatorname{MC}(\gAmod)$
 requires the gauge-theoretic input.

\item \emph{$S_3$ triality.}
 The corner triality permuting $N_1, N_2, N_3$ and acting on
 $\Psi$ by M\"obius
 transformations~\cite{gaiotto-rapchak}
 is a three-dimensional gauge symmetry. It preserves
 Koszulness and shadow class, but does not arise from bar-cobar
 operations alone.
\end{enumerate}
Conversely, three structural results here have no counterpart
in the Gaiotto programme:
\begin{enumerate}[label=\textup{(\alph*)}]
\item \emph{All-genera shadow tower.}
 The MC element $\Theta_{\cA_T}$ at genus $g \geq 2$,
 the uniform-weight genus expansion
 $\operatorname{obs}_g = \kappa \cdot \lambda_g$
 (Theorem~\ref{thm:genus-universality};
 uniform-weight locus),
 and the multi-weight correction $\delta F_g^{\mathrm{cross}}$
 (Theorem~\ref{thm:multi-weight-genus-expansion};
 cross-channel correction off the uniform-weight locus) are
 algebraic results with no gauge-theoretic antecedent.

\item \emph{Twelve-fold Koszulness programme.}
 The equivalence of twelve characterizations of chiral
 Koszulness
 (Theorem~\ref{thm:koszul-equivalences-meta}) and the
 shadow depth classification
 (Theorem~\ref{thm:shadow-archetype-classification}) are
 intrinsic to the bar complex.

\item \emph{Modular characteristic~$\kappa$ and complementarity.}
 The universal invariant~$\kappa(\cA_T)$, its duality
 constraints (Theorem~\ref{thm:modular-characteristic}),
 and the complementarity theorem
 (Volume~I, Theorem~\ref{V1-thm:quantum-complementarity-main}) are
 algebraic structures not visible from perturbative gauge
 theory.
\end{enumerate}
\end{remark}

\begin{remark}[The BCOV identification]

a separate quantization procedure (the holomorphic anomaly
equation); in our framework the full genus tower is encoded
in $\Theta_\cA$ from the outset.
\end{remark}
