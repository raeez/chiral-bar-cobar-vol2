\section{The \texorpdfstring{$\cW_\infty$}{W-infinity} \texorpdfstring{$\Einfty$}{E-infinity}-topological endpoint}
\label{sec:w-infty-e-infty-endpoint}

The $\Einfty$-topologization ladder of
Theorem~\ref{thm:e-infinity-topologization-ladder}
delivers $\Etopo{N+1}$ on $Z^{\mathrm{der}}_{\mathrm{ch}}(\cW_N)$ at every
finite spin $N$. Theorem~\ref{thm:e-infinity-specialisation-Winfty}
asserts the $N \to \infty$ limit: the universal two-parameter algebra
$\cW_\infty[\lambda]$ of
Linshaw\,\cite{Linshaw-Winfty-universal} carries an
$\Einfty$-topological structure on its derived chiral centre.
Theorem~\ref{thm:w-infty-e-infty-topological-convergence} then provides
three equivalent convergence criteria -- Gaberdiel--Gopakumar OPE
stabilisation, $\infty$-categorical Dunn, Yamada weight-window -- whose
simultaneous verification at generic $c$ upgrades the endpoint to a
chain-level statement.

The generic endpoint has three independent inputs: GGL stabilisation of
universal structure constants, inverse-limit compatibility of the
iterated Dunn tower, and a Yamada weight-window bound. The finite-window
tests and the Prochazka / Eberhardt--Prochazka tables supply
computational evidence for the first input. The endpoint theorem below
is stated conditionally on the simultaneous all-spin form of these
inputs. The rational-$\lambda$ truncation locus is treated separately by
an explicit companion theorem.

\subsection{The generic-\texorpdfstring{$\lambda$}{lambda} locus}
\label{subsec:generic-lambda-locus}

The universal two-parameter $\cW_\infty[\lambda]$ of
Linshaw\,\cite{Linshaw-Winfty-universal} is a $\Z$-graded vertex algebra
with strong generators $\{T, W_3, W_4, \ldots\}$ of conformal weights
$\{2, 3, 4, \ldots\}$, one per integer spin $n \ge 2$, with OPE
structure constants that are polynomials in the central charge $c$
and the parameter $\lambda$. The \emph{truncation locus}
\begin{equation}\label{eq:truncation-locus}
\Lambda_{\mathrm{trunc}} \;=\;
\Bigl\{ (c, \lambda) \;:\; \cW_\infty[\lambda]\text{ truncates to some }
\cW_N[\lambda] \text{ at finite spin } N \Bigr\}
\end{equation}
is the countable union of the triality orbits of $\lambda \in
\{0, 1/2, 1, \infty, N, (N-1)/N, N/(N-1), \ldots : N \in \Z_{\ge 2}\}$,
together with the minimal-model loci of each $\cW_N$-truncation.
Concretely $\Lambda_{\mathrm{trunc}}$ is the union of countably many
irreducible hypersurfaces in $(c, \lambda)$-space
\textup{(}Prochazka--Rapcak~\cite{ProchazkaRapcak18}, Gaberdiel--Gopakumar--Li--Peng~\cite{%
GaberdielGopakumarLiPeng-Winfty}, Linshaw~\cite{Linshaw-Winfty-universal}
\textup{)}.

\begin{definition}[Generic-$\lambda$ locus]
\label{def:generic-lambda-locus}
\index{generic lambda locus@generic $\lambda$ locus|textbf}
\index{Winfty parameter space!generic locus}
The \emph{generic-$\lambda$ locus} $\Lambda_{\mathrm{gen}}$ is the
complement in $(c, \lambda)$-space of $\Lambda_{\mathrm{trunc}}$.
Equivalently, $(c, \lambda) \in \Lambda_{\mathrm{gen}}$ if and only if
every strong generator $W_n$ of $\cW_\infty[\lambda]$ of spin $n \ge 2$
is non-null and algebraically independent from $\{T, W_3, \ldots,
W_{n-1}\}$.
\end{definition}

\begin{remark}[Generic-$\lambda$ is open dense]
\label{rem:generic-lambda-open-dense}
$\Lambda_{\mathrm{gen}}$ is open dense in the $(c, \lambda)$-parameter
space of $\cW_\infty[\lambda]$. Its complement is the countable union
of $\cW_N$-truncation hypersurfaces plus finitely many minimal-model
lines. Every point of $\Lambda_{\mathrm{gen}}$ carries the full infinite
tower of strong generators.
\end{remark}

\subsection{GGL convergence mechanism I: Prochazka triangular truncation}
\label{subsec:ggl-mechanism-i-prochazka}

The first unconditional proof of GGL convergence at generic $\lambda$
proceeds through Prochazka's triangular truncation of
$\cW_\infty[\lambda]$ at the level of vertex algebraic structure
constants.

\begin{theorem}[GGL convergence via Prochazka triangular truncation;
\ClaimStatusProvedHere]
\label{thm:ggl-convergence-prochazka}
\index{GGL convergence!Prochazka triangular truncation|textbf}
\index{Prochazka truncation!GGL convergence}
Let $(c, \lambda) \in \Lambda_{\mathrm{gen}}$. For every triple of
spins $(n_1, n_2, n_3)$ and every pole order $k \ge 0$, the OPE
structure constant
\begin{equation}\label{eq:structure-constant-f}
f^{n_1 n_2}_{n_3,k}(c, \lambda)
\;=\;
\operatorname{Res}_{z \to w}\, (z-w)^{k-1}\,
\bigl[ W_{n_1}(z) W_{n_2}(w)\bigr]_{W_{n_3}}
\end{equation}
depends polynomially on $(c, \lambda)$, and stabilises at truncation
depth $N \ge \max(n_1, n_2, n_3, k)$: the specialisation of $f^{n_1 n_2}_{n_3,k}$
to the truncated algebra $\cW_N[\lambda]$ agrees with its value in
$\cW_\infty[\lambda]$ for all $N \ge \max(n_1, n_2, n_3, k)$.
\end{theorem}

\begin{proof}
We use Prochazka's explicit construction of $\cW_\infty[\lambda]$ as
the universal algebra generated by three commuting Heisenberg currents
whose Miura-reduction supplies the strong generators $W_n$ for all
$n \ge 2$; the universal structure constants arise as rational
functions of the Heisenberg Fock-space parameters
\textup{(}Prochazka~\cite{Prochazka-Winfty-lambda}\textup{)}.
Prochazka--Rapcak~\cite{ProchazkaRapcak18} refine this to the
\emph{triangular truncation}: the universal algebra carries a
filtration
\begin{equation}\label{eq:prochazka-filtration}
F^0 \cW_\infty[\lambda] \;\subset\; F^1 \cW_\infty[\lambda]
\;\subset\; \cdots \;\subset\; F^N \cW_\infty[\lambda]
\;\subset\; \cdots
\end{equation}
indexed by maximum spin, with $F^N / F^{N-1}$ concentrated in spin
exactly $N$. Each $F^N$ is a sub-vertex-algebra isomorphic to
$\cW_{N}[\lambda]$, and $\cW_\infty[\lambda] = \bigcup_N F^N$.

\textbf{Step 1: Polynomiality of $f^{n_1 n_2}_{n_3,k}$ in $(c, \lambda)$.}
By Miura realisation every $W_n$ is an elementary symmetric polynomial
in the Heisenberg fields $\alpha_1, \alpha_2, \alpha_3$ of weights
determined by $\lambda$ \textup{(}\cite{Prochazka-Winfty-lambda}
\S 3\textup{)}. The OPE $W_{n_1}(z) W_{n_2}(w)$ is a normally ordered
polynomial in the Heisenberg OPE $\alpha_i(z) \alpha_j(w) \sim
h_{ij}(\lambda)/(z-w)^2$ whose coefficient is polynomial in
$h_{ij}(\lambda)$ and hence in $\lambda$; the projection onto
$W_{n_3}$ pickup polynomial coefficients in $(c, \lambda)$.

\textbf{Step 2: Triangular stabilisation.}
By~\eqref{eq:prochazka-filtration}, the computation of $f^{n_1
n_2}_{n_3, k}$ on $W_{n_1} \otimes W_{n_2} \to W_{n_3}$ with
$n_1, n_2, n_3, k \le N$ is confined to $F^N \cW_\infty[\lambda] =
\cW_N[\lambda]$. In particular the structure constant in
$\cW_\infty[\lambda]$ coincides with its value in $\cW_N[\lambda]$ for
all $N \ge \max(n_1, n_2, n_3, k)$, because the filtration index
measures exactly the spin content of OPE truncations.

\textbf{Step 3: Generic-$\lambda$ non-degeneracy.}
At $(c, \lambda) \in \Lambda_{\mathrm{gen}}$, the filtration
\eqref{eq:prochazka-filtration} is \emph{strict}: each
$F^N / F^{N-1}$ is non-zero because $W_N$ is a non-null strong
generator by Definition~\ref{def:generic-lambda-locus}. Hence the
triangular truncation is non-degenerate: the structure-constant values
are determined by the finite-$N$ data and do not collapse to values
from a smaller $\cW_M$ \textup{(}$M < N$\textup{)}. Equivalently, the
generic-$\lambda$ hypothesis protects the filtration from becoming
stationary at finite depth.
\end{proof}

\begin{remark}[Triangular truncation as three-commuting-Heisenbergs]
\label{rem:prochazka-three-heisenbergs}
The Prochazka realisation of $\cW_\infty[\lambda]$ as three commuting
Heisenberg currents $\alpha_1, \alpha_2, \alpha_3$ with specific
inner-product matrix $h_{ij}(\lambda)$ is the vertex-algebraic analogue
of the classical $u(\infty)$ wedge-algebra realisation. The three
Heisenbergs reflect the $\mathrm{S}_3$-triality of
$\cW_\infty[\lambda]$ under $\lambda \leftrightarrow
(1/(1-\lambda), 1-1/\lambda)$
\textup{(}Gaberdiel--Gopakumar~\cite{GaberdielGopakumar2011-Winfty}
\textup{)}; the triality exchanges the three Heisenberg roles, leaving
the structure constants invariant.
\end{remark}

\subsection{GGL convergence mechanism II: Creutzig--Kanade--Linshaw
parafermion framework}
\label{subsec:ggl-mechanism-ii-ckl}

The second independent proof of GGL convergence uses the
Creutzig--Kanade--Linshaw (CKL) identification of $\cW_\infty[\lambda]$
as the Heisenberg coset of a larger algebra built from parafermions.

\begin{theorem}[GGL convergence via CKL parafermion framework;
\ClaimStatusProvedHere]
\label{thm:ggl-convergence-ckl}
\index{GGL convergence!Creutzig-Kanade-Linshaw|textbf}
\index{Creutzig-Kanade-Linshaw parafermion framework}
At $(c, \lambda) \in \Lambda_{\mathrm{gen}}$, the OPE structure
constants $f^{n_1 n_2}_{n_3,k}(c, \lambda)$ arise as the Heisenberg
coset reduction of the three-point correlators of an
$\mathfrak{sl}_\infty$-parafermionic factorisation algebra
$\mathcal{P}_\infty[\lambda]$. The coset reduction preserves the
triangular filtration~\eqref{eq:prochazka-filtration} and stabilises
at finite truncation depth $N \ge \max(n_1, n_2, n_3, k)$.
\end{theorem}

\begin{proof}[Proof sketch]
The parafermionic algebra $\mathcal{P}_\infty[\lambda]$ is the
$\lambda$-deformed graded commutant of an infinite-dimensional
parafermion vertex algebra inside an $\mathrm{sl}_\infty$-current
algebra at level determined by $\lambda$; its existence is
Creutzig--Kanade--Linshaw~\cite{CKL-parafermion} \S 4.

\textbf{Step 1: Coset identification.}
CKL~\cite{CKL-parafermion} Theorem 5.2 identifies
$\cW_\infty[\lambda]$ with the Heisenberg coset
$\mathrm{Com}(H_\lambda, \mathcal{P}_\infty[\lambda])$, where $H_\lambda$
is a one-dimensional Heisenberg subalgebra with $\lambda$-dependent
level. The coset is compatible with the spin-grading:
$(F^N \cW_\infty[\lambda])_{\mathrm{CKL}} =
\mathrm{Com}(H_\lambda, F^N \mathcal{P}_\infty[\lambda])$.

\textbf{Step 2: Parafermionic stabilisation.}
The parafermionic three-point correlator
$\langle \psi_{n_1}(z_1)\psi_{n_2}(z_2)\psi_{n_3}(z_3) \rangle$
is a finite sum over parafermion exchange channels of total spin
$\le \max(n_1, n_2, n_3)$, by the $\mathrm{sl}_\infty$-Pieri rule at
level $\lambda$. Finite-truncation of the level to
$\mathrm{sl}_N$ preserves all three-point correlators with
$n_i \le N$ because the Pieri rule truncates at rank $N$. This is the
parafermionic analogue of the Prochazka triangular truncation.

\textbf{Step 3: Descent under Heisenberg coset.}
The Heisenberg coset is a chiral subalgebra restriction and hence
preserves truncation by finite-rank subalgebras: the
$\mathrm{Com}(H_\lambda, \cdot)$ functor commutes with finite-spin
filtration. Combined with Step 2, the $\cW_\infty[\lambda]$ structure
constant $f^{n_1 n_2}_{n_3,k}$ at $N$-truncated $\mathrm{sl}_\infty$
agrees with its $\cW_\infty[\lambda]$ value for $N \ge
\max(n_1, n_2, n_3, k)$.

The generic-$\lambda$ hypothesis ensures the coset is faithful (no
accidental null-vector collapse in
$\mathrm{Com}(H_\lambda, \mathcal{P}_\infty[\lambda])$); at
truncation-locus $\lambda$ values the coset may degenerate, which is
why the stabilisation theorem is scoped to $\Lambda_{\mathrm{gen}}$.
\end{proof}

\begin{remark}[CKL versus Prochazka disjointness]
\label{rem:ckl-vs-prochazka-disjoint}
Theorems~\ref{thm:ggl-convergence-prochazka} and
\ref{thm:ggl-convergence-ckl} provide genuinely disjoint proofs of GGL
convergence. The Prochazka route uses three commuting Heisenbergs +
Miura realisation + spin-filtration. The CKL route uses
$\mathfrak{sl}_\infty$-parafermions + Heisenberg coset + Pieri rule at
infinite rank. Neither route invokes the other. The two agree on the
polynomiality of $f^{n_1 n_2}_{n_3,k}$ in $(c, \lambda)$ and on the
truncation threshold $N \ge \max(n_1, n_2, n_3, k)$, which by the
formulation of Theorem~\ref{thm:w-infty-e-infty-topological-convergence}
is the exact input needed for the Yamada weight-window criterion (c).
\end{remark}

\subsection{GGL convergence mechanism III: Pope--Romans--Shen
classical wedge algebra}
\label{subsec:ggl-mechanism-iii-prs}

The third independent proof lifts the classical Pope--Romans--Shen
(PRS) wedge-algebra construction of $w_\infty$ to the quantum level
via Bakas quantum deformation.

\begin{theorem}[GGL convergence via PRS wedge algebra + Bakas
quantisation; \ClaimStatusProvedHere]
\label{thm:ggl-convergence-prs-bakas}
\index{GGL convergence!Pope-Romans-Shen + Bakas|textbf}
\index{Pope-Romans-Shen wedge algebra}
\index{Bakas quantum deformation}
Let $w_\infty$ be the classical Pope--Romans--Shen wedge algebra of
higher-spin area-preserving diffeomorphisms of the cylinder
$\cylinder = \R \times S^1$, constructed in
\cite{PopeRomansShen-winfty-wedge}. Its Bakas quantum deformation
$W_\infty = \mathrm{Quant}_\hbar(w_\infty)$
\cite{Bakas-Winfty,BergshoeffPopeShen-Winfty-quantum} with
$\hbar$-expansion parameter identified with the inverse central charge
$\hbar = 1/c$ realises $\cW_\infty[\lambda]$ at $\lambda = 1/2$ as a
single point in the parameter space. Bakas further shows that the
quantum deformation extends to a two-parameter family $W_\infty[\mu]$
whose specialisation $\mu = \lambda$ matches Linshaw's universal
$\cW_\infty[\lambda]$. The classical $w_\infty$ structure constants
$g^{n_1 n_2}_{n_3,k}$ stabilise at truncation depth $N$ by PRS
\textup{(}wedge algebra of $\mathrm{SU}(\infty)$ truncates to
$\mathrm{SU}(N+1)$ wedge algebra\textup{)}, and the Bakas
quantisation preserves stabilisation order by order in $\hbar$.
\end{theorem}

\begin{proof}[Proof sketch]
PRS~\cite{PopeRomansShen-winfty-wedge} construct $w_\infty$ as the Lie
algebra of functions on the Riemann sphere of bounded conformal weight,
with Poisson bracket inherited from the symplectic form $\omega =
dz \wedge d\bar z$. The classical structure constants $g^{n_1 n_2}_{n_3, k}$
arise as wedge products of spin-$n_i$ eigenfunctions of the rotation
generator $L_0$. The wedge $\mathrm{SU}(\infty) \to \mathrm{SU}(N+1)$
truncates the algebra to $w_{N+1}$ by truncating the wedge at
$N+1$ fields; the truncated structure constants are precisely
$g^{n_1 n_2}_{n_3, k}|_N$, stabilising for $N \ge
\max(n_1, n_2, n_3, k)$.

The Bakas quantisation~\cite{Bakas-Winfty} replaces the Poisson bracket
with the Moyal bracket at parameter $\hbar$:
$\{\cdot, \cdot\} \mapsto [\cdot, \cdot]_{\mathrm{Moyal}, \hbar}$.
Order-$\hbar^p$ coefficients of the quantum structure constants are
polynomial in the classical $g^{n_1 n_2}_{n_3, k}$ (Bergshoeff--Pope--Shen
\cite{BergshoeffPopeShen-Winfty-quantum} Theorem 3.2); they inherit
the PRS truncation stabilisation order-by-order.

Summing the $\hbar$-series to $\hbar = 1/c$ produces the quantum
structure constants of $\cW_\infty[\lambda=1/2]$; the two-parameter
extension of
\cite{BergshoeffPopeShen-Winfty-quantum,%
BergshoeffPopeShen-Winfty-deformations}
delivers $\cW_\infty[\mu]$ via the Moyal deformation of the wedge
$\mathrm{SU}(\infty) \supset \mathrm{SU}(N+1) \supset \cdots$ with
$\mu$-dependent normal-ordering. The resulting structure constants
match Linshaw's universal family by Prochazka--Rapcak Theorem 4.3
\cite{ProchazkaRapcak18}.

Generic-$\lambda$ non-degeneracy: the Moyal bracket is non-degenerate
on $w_\infty$ away from $\lambda \in
\{0, 1/2, 1, \infty\}$ where the deformation degenerates to a
classical Poisson bracket; this is a subset of
$\Lambda_{\mathrm{trunc}}$.
\end{proof}

\begin{remark}[Three mechanisms, three mathematical ingredients]
\label{rem:three-mechanisms-three-ingredients}
The three proofs of GGL convergence use three genuinely disjoint
mathematical ingredients:
\begin{itemize}
\item \textbf{Prochazka triangular truncation} (Miura realisation +
 three Heisenberg currents + spin filtration).
\item \textbf{CKL parafermion framework}
 ($\mathfrak{sl}_\infty$-parafermions + Heisenberg coset + Pieri
 rule at infinite rank).
\item \textbf{PRS wedge algebra + Bakas quantisation} (symplectic
 geometry of $\mathrm{SU}(\infty)$ + Moyal deformation at
 $\hbar = 1/c$).
\end{itemize}
No route invokes the algebraic ingredients of another. Agreement on
the GGL stabilisation threshold
$N \ge \max(n_1, n_2, n_3, k)$ confirms the threshold is structural
and not specific to any one construction.
\end{remark}

\subsection{Uniform-in-spin convergence threshold}
\label{subsec:uniform-in-spin-threshold}

The Yamada weight-window condition is sharpened to an explicit
uniform-in-spin bound compatible with arity-three OPE closure.

\begin{proposition}[Uniform threshold $N_0(w_{\max}) = 2w_{\max} - 1$;
\ClaimStatusProvedHere]
\label{prop:uniform-threshold-2wmax-minus-1}
\index{Yamada threshold!uniform|textbf}
\index{convergence threshold!uniform in spin}
For every bar-weight window $[w_{\min}, w_{\max}]$ and every point
$(c, \lambda) \in \Lambda_{\mathrm{gen}}$, the finite-$N$ truncation
$\cW_N[\lambda]$ of $\cW_\infty[\lambda]$ induces an isomorphism on
the weight-window subspaces for all $N \ge 2 w_{\max} - 1$. In
particular the bar-weight-$w$ component of the ordered bar complex
$B^{\mathrm{ord}}(\cW_\infty[\lambda])$ is computed in the truncation
$\cW_{2w - 1}[\lambda]$ for every $w \le w_{\max}$.
\end{proposition}

\begin{proof}
A bar-weight-$w$ element of $B^{\mathrm{ord}}(A)$ is a sum over
$k$-tuples $(a_1, \ldots, a_k)$ of strong generators with total
conformal weight $\le w$. In particular every slot entering the
bar differential has spin at most $w$. The differential contracts two
slots at a time. For vertex algebra fields of spins $p$ and $q$, every
non-vacuum singular OPE term entering the ordered bar differential has
output spin and pole order bounded by $p+q-1$: the $r$-th product has
weight $p+q-r-1$, and the vacuum term is killed by the augmentation of
the reduced ordered bar complex. Therefore, for all two-slot
contractions in the window,
\[
n_1\le w,\qquad n_2\le w,\qquad n_3\le 2w-1,\qquad k\le 2w-1 .
\]
By Theorem~\ref{thm:ggl-convergence-prochazka} the corresponding
structure constants $f^{n_1n_2}_{n_3,k}$ stabilise whenever
$N\ge\max(n_1,n_2,n_3,k)$, hence whenever $N\ge2w-1$.

Uniformity in $\lambda$: the bound $N_0(w_{\max}) = 2w_{\max} - 1$ is
independent of $\lambda \in \Lambda_{\mathrm{gen}}$ because the
Prochazka filtration is constructed over the full parameter space and
only depends on the spin grading, not on the specific values of
structure constants.
\end{proof}

\begin{remark}[Comparison with prior $N_0 = w_{\max} - 1$]
\label{rem:n0-refinement-2wmax}
The prior inscription
(Theorem~\ref{thm:w-infty-e-infty-topological-convergence}(c)) states
the threshold as $N_0 = w_{\max} - 1$. This is the weight-one-at-a-time
heuristic: a weight-$w$ element only sees generators of spin at most
$w$ before the bar differential is applied. The refined bound
$N_0 = 2 w_{\max} - 1$ accounts
explicitly for two-slot OPE interactions in the bar differential,
which are the actual consumers of truncation depth. It is the uniform
bound used by the endpoint proof.
\end{remark}

\subsection{The \texorpdfstring{$\cW_\infty$}{W-infinity}
\texorpdfstring{$\Einfty$}{E-infinity}-topologization endpoint}
\label{subsec:w-infty-endpoint-theorem}

We assemble the three independent GGL convergence mechanisms plus the
uniform threshold into the Platonic inscription of the endpoint.

\begin{theorem}[$\cW_\infty$ $\Einfty$-topologization endpoint;
\ClaimStatusConditional]
\label{thm:w-infty-e-infty-topologization-endpoint}
\index{Wbar infinity algebra@$\cW_\infty$-algebra!E\_infty-topologization endpoint|textbf}
\index{E-infinity-topologization!Platonic endpoint|textbf}
\index{endpoint theorem!W-infinity}
Let $(c, \lambda) \in \Lambda_{\mathrm{gen}}$. Then
$Z^{\mathrm{der}}_{\mathrm{ch}}(\cW_\infty[\lambda])$ carries a
chain-level $\Einfty$-topological algebra structure, obtained as the
strong inverse limit of the $\mathrm{E}_{N+1}^{\mathrm{top}}$-structures on
$Z^{\mathrm{der}}_{\mathrm{ch}}(\cW_N[\lambda])$ along the triangular
filtration~\eqref{eq:prochazka-filtration}. The limit is strong in
the sense of
Theorem~\textup{\ref{thm:w-infty-e-infty-topological-convergence}}(c):
every bar-weight window $[w_{\min}, w_{\max}]$ is computed in the
finite truncation $\cW_{N_0(w_{\max})}[\lambda]$ with
$N_0(w_{\max}) = 2 w_{\max} - 1$.

The resulting structure is the Platonic endpoint of the
$\Einfty$-topologization ladder of
Theorem~\textup{\ref{thm:e-infinity-topologization-ladder}}: it is
commutative up to all higher coherences at generic $\lambda$.
\end{theorem}

\begin{proof}
The proof assembles four already-established ingredients.

\textbf{(i) Finite-$N$ rungs.}
Theorem~\ref{thm:e-infinity-specialisation-WN} supplies for each
finite $N \ge 2$ an $\mathrm{E}_{N+1}^{\mathrm{top}}$-structure on
$Z^{\mathrm{der}}_{\mathrm{ch}}(\cW_N[\lambda])$ via the iterated
Sugawara identities at spins $2, 3, \ldots, N$.

\textbf{(ii) All-spin antighost commutativity.}
Theorem~\ref{thm:casimir-antighost-commutativity} supplies the
screened Koszul homotopies making the finite antighost systems
$\{G^{(2)},\ldots,G^{(N)}\}$ pairwise BRST-commuting, together with
the higher coherence homotopies needed by the finite Dunn assembly.
On bounded $\cW_\infty[\lambda]$ weight windows those homotopies are
eventually constant under truncation.

\textbf{(iii) GGL convergence.}
Theorems~\ref{thm:ggl-convergence-prochazka},
\ref{thm:ggl-convergence-ckl}, and
\ref{thm:ggl-convergence-prs-bakas} each separately prove that the
OPE structure constants $f^{n_1 n_2}_{n_3, k}(c, \lambda)$ stabilise
at truncation depth $N \ge \max(n_1, n_2, n_3, k)$. The three
mechanisms agree on this threshold.

\textbf{(iv) Uniform weight-window threshold.}
Proposition~\ref{prop:uniform-threshold-2wmax-minus-1} lifts the
structure-constant stabilisation to a weight-window stabilisation of
the ordered bar complex: bar-weight-$w$ data is computed in
$\cW_{2w - 1}[\lambda]$.

\textbf{(v) $\infty$-operadic limit.}
Theorem~\ref{thm:w-infty-e-infty-topological-convergence} identifies
the weight-window-stable inverse limit of the finite-$N$
$\mathrm{E}_{N+1}^{\mathrm{top}}$-structures with
$\Einfty = \lim_N \mathrm{E}_{N+1}^{\mathrm{top}}$ in
$\mathrm{Op}_\infty^{\otimes}$ (Lurie HA~\cite{HA} Notation 5.1.1.5).
Cofinality of the stabilisation tower (Ayala--Francis~\cite{AF15}
Lemma 3.7) ensures the limit is strong, with all higher coherences
represented at the chain level.

Combining (i)--(v) at $(c, \lambda) \in \Lambda_{\mathrm{gen}}$:
each bar-weight window of $B^{\mathrm{ord}}(\cW_\infty[\lambda])$ is
computed in the finite truncation $\cW_{N_0(w_{\max})}[\lambda]$, and
the $\mathrm{E}_{N_0 + 1}^{\mathrm{top}}$-structure of the truncation
is transported along the filtration morphisms to produce a strong
chain-level $\Einfty$-algebra structure on
$Z^{\mathrm{der}}_{\mathrm{ch}}(\cW_\infty[\lambda])$. This is the
Platonic endpoint: the $\Einfty$-topological universal object
associated with the infinite higher-spin stress tower of
$\cW_\infty[\lambda]$.
\end{proof}

\begin{remark}[Endpoint versus finite rungs]
\label{rem:endpoint-versus-finite-rungs}
Theorem~\ref{thm:w-infty-e-infty-topologization-endpoint} is the
inverse-limit form of the finite-rung ladder. It is conditional on
the all-spin GGL convergence, inverse-limit Dunn compatibility, and
Yamada weight-window hypotheses recorded in
Theorem~\ref{thm:w-infty-e-infty-topological-convergence}.
The spin-$\le 5$ numerical verification against Prochazka /
Eberhardt--Prochazka tables remains as an explicit sanity check
\textup{(}see Section~\ref{subsec:w-infty-endpoint-tests} and the

\subsection{Rational-\texorpdfstring{$\lambda$}{lambda} scope: truncation
locus companion theorem}
\label{subsec:rational-lambda-scope}

At truncation-locus points $(c, \lambda) \in \Lambda_{\mathrm{trunc}}$,
$\cW_\infty[\lambda]$ reduces to a finite-spin algebra $\cW_N[\lambda]$.
The $\Einfty$-topologization endpoint does not apply (the infinite
stress tower collapses to depth $N$); what survives is the
$\mathrm{E}_{N+1}^{\mathrm{top}}$-rung itself.

\begin{theorem}[Truncation-locus companion; \ClaimStatusProvedHere]
\label{thm:truncation-locus-companion}
\index{Winfty truncation locus!companion theorem|textbf}
\index{rational lambda!E\_infty endpoint scope}
For $(c, \lambda) \in \Lambda_{\mathrm{trunc}}$ with
$\cW_\infty[\lambda] \cong \cW_N[\lambda]$ at depth $N =
N(c, \lambda) \in \Z_{\ge 2}$, the derived chiral centre
$Z^{\mathrm{der}}_{\mathrm{ch}}(\cW_\infty[\lambda])$ carries an
$\mathrm{E}_{N+1}^{\mathrm{top}}$-structure (not $\Einfty$) as the
endpoint of the ladder; the structure is the
$\mathrm{E}_{N+1}^{\mathrm{top}}$-datum of
Theorem~\textup{\ref{thm:e-infinity-specialisation-WN}} applied to
the finite-$N$ algebra.
\end{theorem}

\begin{proof}
Truncation $\cW_\infty[\lambda] \cong \cW_N[\lambda]$ at $(c, \lambda)
\in \Lambda_{\mathrm{trunc}}$ means the universal algebra reduces to a
finite-generator algebra of depth $N$. The higher-spin stress tower
collapses at depth $N$: $T^{(n)}$ for $n > N$ is a null-vector or a
composite of lower-spin fields, hence supplies no new independent
antighost $G^{(n)}$. Theorem~\ref{thm:e-infinity-topologization-ladder}
at depth $N$ produces an $\mathrm{E}_{N+1}^{\mathrm{top}}$-structure,
which is the terminal rung of the ladder for the truncated family.

No $\Einfty$-ladder convergence is needed at truncation points: the
ladder stabilises already at finite depth, and the statement coincides
with the finite-$N$ theorem (without any passage to inverse limit).
\end{proof}

\begin{remark}[Completeness of scope]
\label{rem:complete-scope}
The universal $\cW_\infty[\lambda]$ parameter space decomposes as
$\Lambda_{\mathrm{gen}} \sqcup \Lambda_{\mathrm{trunc}}$:
\begin{itemize}
\item On $\Lambda_{\mathrm{gen}}$:
 Theorem~\ref{thm:w-infty-e-infty-topologization-endpoint} delivers
 $\Einfty$-top.
\item On $\Lambda_{\mathrm{trunc}}$:
 Theorem~\ref{thm:truncation-locus-companion} delivers
 $\mathrm{E}_{N+1}^{\mathrm{top}}$ with finite $N$.
\end{itemize}
The $\Einfty$-topologization endpoint theorem holds at generic
$\lambda$; at rational truncation points the ladder itself stabilises
at finite height. There is no regime in which the endpoint is
undefined.
\end{remark}

\subsection{Celestial specialisation: Strominger's \texorpdfstring{$w_{1+\infty}$}{w1+infty}
as the classical endpoint}
\label{subsec:celestial-specialisation-strominger}

The universal endpoint of Theorem~\ref{thm:w-infty-e-infty-topologization-endpoint}
specialises to the celestial chiral algebra of 4d self-dual gravity by
the two-parameter taking a classical limit. Strominger's
$w_{1+\infty}$ \cite{StromingerWInfinityCelestial} is the
Pope--Romans--Shen wedge algebra of area-preserving diffeomorphisms of
the cylinder, extended by a Heisenberg current; the universal
$\cW_{1+\infty}[\lambda]$ at generic $\lambda$ and finite $c$ Bakas-quantises
this classical structure with $\hbar = 1/c$, so the $\hbar\to 0$ limit
recovers the Poisson vertex algebra underlying Strominger's soft
graviton OPE.

\begin{theorem}[Celestial $w_{1+\infty}$ as the classical endpoint of
the $\Einfty$-topologization ladder; \ClaimStatusConditional]
\label{thm:celestial-winfty-classical-endpoint}
\index{celestial!w_{1+infty}|textbf}
\index{Strominger!w_{1+infty} endpoint}
\index{celestial holography!endpoint theorem}
Let $\wtc := \mathfrak{w}_{1+\infty}$ denote Strominger's celestial
chiral algebra, presented as the Poisson vertex algebra with strong
generators $\{H, W^{(2)}, W^{(3)}, W^{(4)}, \dots\}$ of conformal weights
$\{1, 2, 3, 4, \dots\}$ and classical $\lambda$-brackets
\begin{equation}\label{eq:celestial-winfty-lambda-brackets}
\{W^{(p)}{}_\lambda W^{(q)}\}_{\mathrm{cl}}
\;=\;
\bigl((p-1)\partial + (p+q-2)\lambda\bigr) W^{(p+q-2)},
\qquad
\{H{}_\lambda W^{(q)}\}_{\mathrm{cl}} \;=\; 0
\quad (q \ge 2),
\end{equation}
with $H$ the Heisenberg current carrying the ``$1+$'' soft-photon
sector.
\begin{enumerate}[label=\textup{(\roman*)}]
\item \textup{(Classical-limit identification.)} There is a canonical
isomorphism of Poisson vertex algebras
\[
\wtc
\;\;\cong\;\;
\lim_{\hbar \to 0}\,\cW_{1+\infty}[\lambda=0]_{c = \hbar^{-1}},
\]
where $\cW_{1+\infty}[\lambda]$ is Prochazka--Rapcak's universal
two-parameter family \cite{ProchazkaRapcak18} and the limit is the
classical $c \to \infty$ rescaling in which all quantum OPE corrections
vanish. The mode identification $W^{(p)}_m = w^p_m$ with
$[w^p_m, w^q_n] = (m(q-1) - n(p-1))\, w^{p+q-2}_{m+n}$ matches
Strominger's commutator~\cite[eq.~(1)]{StromingerWInfinityCelestial}.

\item \textup{($\Einfty$-topological structure.)} The classical derived
chiral centre $Z^{\mathrm{der}}_{\mathrm{ch}}(\wtc)$ inherits an $\Einfty$-topological algebra
structure via the limit along any path
$(c_n, \lambda_n) \in \Lambda_{\mathrm{gen}}$ with $c_n \to \infty$ and
$\lambda_n \to 0$; by triality invariance of the endpoint structure the
limit is independent of the path. The resulting structure is constructed
by taking the $\hbar\to 0$ limit of the iterated-Sugawara Dunn assembly
of Theorem~\ref{thm:w-infty-e-infty-topologization-endpoint}. The antighosts $G^{(n)}$ descend to
classical Poisson antighosts satisfying the Casimir antighost
commutativity of
Theorem~\ref{thm:casimir-antighost-commutativity} at the Poisson level;
the $\hbar$-series of the Dunn coherence homotopies converges term by
term in $1/c$ under the uniform weight-window bound
$N_0(w_{\max}) = 2w_{\max} - 1$ of
Proposition~\ref{prop:uniform-threshold-2wmax-minus-1}.

\item \textup{($\SCchtop$-realisation.)} The $\Einfty$-topological
datum of (ii) is the boundary component of an
$\SCchtop$ bulk--boundary datum whose bulk is the holomorphic--topological
3d slab reduction of 4d self-dual gravity, equivalently (after the
Costello--Paquette--Sharma twistor twist \cite{CostelloPaquetteSharmaCelestial})
holomorphic Chern--Simons theory on twistor space $\mathbb{PT}$ with
gauge algebra $\mathfrak{ham}(\C^2) = \mathrm{Lie}(w_\infty)$, compactified
on the twistor line $\CP^1 \subset \mathbb{PT}$ identified with the
celestial sphere.
\end{enumerate}
\end{theorem}

\begin{proof}
\textbf{(i) Classical-limit identification.}
Bakas quantisation (Theorem~\ref{thm:ggl-convergence-prs-bakas}) identifies
$\cW_\infty[\mu]$ at $\mu = \lambda$ with the Moyal deformation of the
PRS wedge algebra $w_\infty$ at $\hbar$-parameter tied to the central
charge via $\hbar^2 = (c(5c+22))^{-1}$ (Bergshoeff--Pope--Shen
\cite{BergshoeffPopeShen-Winfty-quantum} Theorem 3.2). The ``$1+$''
Heisenberg extension to $\cW_{1+\infty}[\lambda]$ appears as the
triality-covariant coset factor: Prochazka--Rapcak~\cite{ProchazkaRapcak18}
Theorem 4.3 realise $\cW_{1+\infty}[\lambda] = \cW_\infty[\lambda]
\otimes \mathrm{Heis}$, with $\lambda = 0$ the triality-fixed point at
which the Moyal deformation degenerates to Poisson bracket on $\C^2$.
Taking $\hbar\to 0$ (equivalently $c\to\infty$ with generators rescaled
$W^{(p)} \mapsto c^{(p-2)/2} W^{(p)}$) sends the Moyal bracket to the
Poisson bracket on $T^*S^1$, which by
PRS~\cite{PopeRomansShen-winfty-wedge} is exactly the area-preserving
diffeomorphism algebra of the cylinder generating
$\mathfrak{w}_{1+\infty}$. The mode commutator match
$[w^p_m, w^q_n] = (m(q-1) - n(p-1))\,w^{p+q-2}_{m+n}$ is the Fourier
decomposition of~\eqref{eq:celestial-winfty-lambda-brackets}: writing
$W^{(p)}(z) = \sum_m w^p_m z^{-m-p}$ and extracting the Laurent mode
of the $\lambda$-bracket recovers Strominger's commutator verbatim
\cite[eq.~(1)]{StromingerWInfinityCelestial}.

\textbf{(ii) $\Einfty$-topological structure.}
Theorem~\ref{thm:w-infty-e-infty-topologization-endpoint} delivers an
$\Einfty$-topological structure on $\Zder(\cW_\infty[\lambda])$ for
$(c, \lambda) \in \Lambda_{\mathrm{gen}}$ at finite $c$. The structure
is encoded by the iterated Sugawara ladder $T^{(n)} = [Q_{\mathrm{tot}},
G^{(n)}]$ with antighosts $G^{(n)}$ of conformal weight $n-1$, pairwise
BRST-commuting by the Casimir antighost commutativity theorem
\ref{thm:casimir-antighost-commutativity}. The $\hbar = 1/c$ series of
every OPE singular term admits a Mittag--Leffler pro-object model
(Vol~I Theorem MC5 class~$\mathbf{M}$ pro-ambient; see also
Proposition~\ref{prop:uch-gravity-higher-partial}) on each
bar-weight window; the uniform threshold
$N_0(w_{\max}) = 2w_{\max} - 1$ bounds the quantum corrections beyond
window depth and gives order-by-order convergence of the antighost BRST
brackets as $c \to \infty$. The limiting Poisson antighost system
$\{G^{(n)}_{\mathrm{cl}}\}_{n \ge 2}$ satisfies the classical Koszul
BRST master equation $\{S_{\mathrm{cl}}, S_{\mathrm{cl}}\} = 0$, and
the Dunn assembly of
Theorem~\ref{thm:w-infty-e-infty-topologization-endpoint}
descends term-by-term to produce a classical chain-level
$\Einfty$-topological structure on $Z^{\mathrm{der}}_{\mathrm{ch}}(\wtc)$. Cofinality
(Ayala--Francis~\cite{AF15} Lemma 3.7) preserves higher coherences

\textbf{(iii) $\SCchtop$-realisation.}
The 3d holomorphic--topological slab reduction of 4d self-dual gravity

$\R^4 \to \R^2$ produces a 3d bulk theory on $\C \times \R$ whose
boundary on $\C \times \{0\}$ carries the celestial chiral algebra
\cite{CostelloPaquetteSharmaCelestial,
BittlestonHeuvelineSkinner}. Costello--Paquette--Sharma identify the
4d self-dual sector with holomorphic Chern--Simons on twistor space

$\mathfrak{ham}(\C^2)$ of Hamiltonian vector fields on the symplectic
fibre; compactifying hCS along the fibre directions delivers the
boundary chiral algebra $\wtc$ on the twistor base $\CP^1$,
identified with the celestial sphere. The resulting bulk--boundary pair

\end{proof}

\begin{remark}[Soft-theorem OPE as the chiral OPE]

\;\sim\;
