\section{The bar complex as twisting morphism classifier}
\label{sec:bar_cobar}

The bar complex~$\barB(\cA)$ classifies \emph{twisting morphisms}:
universal couplings between the boundary algebra~$\cA$ and its
Koszul dual~$\cA^!$. It is the coalgebraic shadow of the open
factorization category (Part~\ref{part:swiss-cheese},
\S\ref{subsec:categorical-open-sector}), not the universal bulk
(which is the chiral derived center of
\S\ref{sec:chiral_hochschild}). The bar--cobar duality for $\Ainf$
chiral algebras extends the quadratic theory of
Gui--Li--Zeng~\cite{GLZ22} to the non-quadratic setting.

\begin{proposition}[$E_1$ primacy as a categorical adjunction; \ClaimStatusProvedHere]
\label{prop:bar-cobar-e1-primacy-adjunction}
Let $E_1\text{-}\mathsf{Ch}$ denote the $\infty$-category of $E_1$-chiral
algebras (ordered, $\Sigma_n$-free, bar complex $T^c(s^{-1}\bar\cA)$) and
$E_\infty\text{-}\mathsf{Ch}$ the $\infty$-category of $E_\infty$-chiral
algebras (symmetric, $\Sigma_n$-coinvariant, bar complex
$\Sym^c(s^{-1}\bar\cA)$), each enriched over $D$-modules on the
Ran space. Write $U\colon E_\infty\text{-}\mathsf{Ch} \to
E_1\text{-}\mathsf{Ch}$ for the forgetful functor that remembers only the
ordered structure. Then $U$ admits a left adjoint
\begin{equation}
\label{eq:sym-ch-U-adjunction}
\Sym^{\mathrm{ch}} \;\dashv\; U\colon\;
E_\infty\text{-}\mathsf{Ch} \;\rightleftarrows\; E_1\text{-}\mathsf{Ch},
\qquad
\Sym^{\mathrm{ch}}(\cA) \;=\; (\cA)_{\Sigma_\bullet}^{R\text{-tw}},
\end{equation}
given by $R$-twisted $\Sigma_\bullet$-coinvariants along the chiral
tensor filtration. On bar complexes the unit
$\eta_\cA\colon \barB(\cA) \to U\Sym^{\mathrm{ch}}(\barB(\cA))$
is realised by the averaging map
$\mathrm{av}\colon \mathfrak{g}^{E_1}_\cA \to
\mathfrak{g}^{\mathrm{mod}}_\cA$, and the passage
$T^c \twoheadrightarrow \Sym^c$ is the component of $\eta$ at
$\cA$. Consequently the bar--cobar adjunction, twisting morphism
classification (Theorem~\ref{thm:GLZ_MC}), and Koszul inversion
reviewed in this chapter are the image under $U$ of the modular
theory of Volume~I: every $E_\infty$-statement is recovered from
its $E_1$-lift by applying $\Sym^{\mathrm{ch}}$, and no information
is lost because the counit $\epsilon\colon \Sym^{\mathrm{ch}} U \to
\mathrm{id}$ is a natural equivalence on evaluation-type modules.
\end{proposition}

\begin{proof}
The forgetful functor $U$ is strict monoidal (it drops
$\Sigma$-equivariance without changing underlying $D$-modules) and
preserves all limits, so it admits a left adjoint by the adjoint
functor theorem in presentable $\infty$-categories (Lurie, HA 5.5.2.9).
The explicit formula
$\Sym^{\mathrm{ch}}(\cA) = (\cA)_{\Sigma_\bullet}^{R\text{-tw}}$
for the left adjoint is the $R$-twisted coinvariants computed via the
bar-cofree model: the $R$-matrix descent of
Proposition~\ref{prop:r-matrix-descent} provides the $\Sigma_n$-action
compatible with the chiral differential (twisted rather than naive).
The unit $\eta$ on bar complexes factors through
$T^c(s^{-1}\bar\cA) \twoheadrightarrow \Sym^c(s^{-1}\bar\cA)$, which is
the averaging map $\mathrm{av}$. For the counit to be an equivalence on
evaluation modules $V_u$: an $E_\infty$-chiral algebra is (tautologically)
its own $\Sigma$-coinvariants up to $R$-twist, so $\Sym^{\mathrm{ch}}
U \cA \simeq \cA$ on the image of $U$. The claim that $E_\infty$-statements
are images of $E_1$-lifts is the adjunction~\eqref{eq:sym-ch-U-adjunction}
applied to the bar--cobar adjunction, using that both compositions
commute with $\Omega\dashv\barB$ by functoriality of bar constructions.
\end{proof}

\begin{remark}[Locality-hierarchy compatibility]
\label{rem:e1-primacy-v2-ap1-compat}
The adjunction $\Sym^{\mathrm{ch}}\dashv U$
of~\eqref{eq:sym-ch-U-adjunction} avoids the ``are vertex algebras
$E_\infty$ or not?'' controversy by construction: every vertex
algebra is an object of $E_\infty\text{-}\mathsf{Ch}$, hence equals
$\Sym^{\mathrm{ch}}$ of its $E_1$-lift $U\cA$, while genuinely
$E_1$-chiral algebras (Yangians) live in $E_1\text{-}\mathsf{Ch}$
without a preimage under $U$. Pole structure plays no role in the
adjunction; only $\Sigma$-equivariance does.
\end{remark}

\begin{convention}[Strict models versus invariant deformation objects]
\label{conv:vol2-strict-models}
Throughout this volume, concrete complexes such as convolution dg~Lie
algebras, bar-coderivation complexes, and dg endomorphism algebras are
chosen strict coordinates. The deformation theory they control is
invariant only up to filtered $L_\infty$ quasi-isomorphism. In
particular, every Maurer--Cartan equation written in strict dg~Lie
notation should be read as a computation in one chart on an underlying
homotopy deformation object. This is the same strict-versus-homotopy
distinction used throughout Volume~I.
\end{convention}

\subsection{Review: GLZ Quadratic Duality for Chiral Algebras}

\subsubsection{Quadratic Chiral Algebras}

\begin{definition}[Quadratic Chiral Algebra]
\label{def:quadratic_chiral}
A \emph{quadratic chiral algebra} is a chiral algebra $\mathcal{A}$ of the form:
\begin{equation}
\mathcal{A} = T_{\text{ch}}(V) / \langle R \rangle,
\end{equation}
where:
\begin{itemize}
\item $T_{\text{ch}}(V)$ is the free chiral tensor algebra on a graded vertex space $V$;
\item $R \subseteq V \otimes_{\text{ch}} V$ is a subspace of \emph{quadratic relations} (degree-2 elements in the chiral tensor product).
\end{itemize}
\end{definition}

\begin{example}[Heisenberg Vertex Algebra]
\label{ex:Heisenberg_quadratic}
The Heisenberg vertex algebra $\mathcal{H}_k$ at level $k$ is generated by $V = \C \cdot J$ with relation:
\begin{equation}
\{J_\lambda J\} = k\lambda.
\end{equation}

This is a quadratic chiral algebra with:
\begin{equation}
R = \text{span}\big\{\{J_\lambda J\} - k\lambda\big\} \subseteq V \otimes_{\text{ch}} V.
\end{equation}
\end{example}

\subsubsection{GLZ Koszul Dual}

\begin{definition}[Koszul Dual (GLZ)]
\label{def:GLZ_Koszul}
For a quadratic chiral algebra $\mathcal{A} = T_{\text{ch}}(V) / \langle R \rangle$, the \emph{Koszul dual} is:
\begin{equation}
\mathcal{A}^! = T_{\text{ch}}(V^* \otimes \omega_\C[1]) / \langle R^\perp \rangle,
\end{equation}
where:
\begin{itemize}
\item $V^*$ is the graded dual of $V$;
\item $\omega_\C$ is the canonical bundle of $\C$ (shift by 1 in the chiral direction);
\item $R^\perp \subseteq (V^* \otimes \omega_\C[1])^{\otimes 2}$ is the annihilator of $R$ under the natural pairing.
\end{itemize}
\end{definition}

\begin{theorem}[GLZ Maurer--Cartan Control; \ClaimStatusProvedElsewhere]
\label{thm:GLZ_MC}
(Gui--Li--Zeng \cite{GLZ22}, Theorem 4.5) For a Koszul quadratic chiral algebra $\mathcal{A}$, there is a bijection:
\begin{equation}
\Hom_{\text{VA}}(\mathcal{A}, \mathcal{B}) \longleftrightarrow \MC\big(\mathcal{A}^! \widehat{\otimes} \mathcal{B}\big),
\end{equation}
where $\MC(\cdot)$ denotes Maurer--Cartan elements (solutions of $d\alpha + \frac{1}{2}[\alpha, \alpha] = 0$) in the convolution dg Lie algebra (the strict model $\Convstr$ of the homotopy-invariant $L_\infty$-algebra $\Convinf$; Convention~\ref{conv:vol2-strict-models}).
\end{theorem}

This parallels Priddy's Koszul duality for associative algebras and Hinich's for Lie algebras.

\begin{remark}[Homotopy upgrade of the GLZ Maurer--Cartan picture]
\label{rem:glz-homotopy-upgrade}
The GLZ Maurer--Cartan correspondence is a strict quadratic model. As
soon as higher $m_k$ appear, the correct recipient is not merely the dg
Lie algebra $\mathcal{A}^!\widehat\otimes \mathcal{B}$ but the complete
filtered $L_\infty$-object obtained from a bar-cofree model of the two
algebras. The role of the chiral bar--cobar formalism below is
precisely to provide this homotopy-coherent replacement while retaining
GLZ as the genus-zero quadratic shadow.
\end{remark}

\subsection{Extension to $A_\infty$ chiral algebras: the chiral bar construction}

When $m_{k \geq 3} \neq 0$, the quadratic theory extends as follows.

\subsubsection{The Chiral Bar Complex}

\begin{definition}[Chiral Bar Complex with $A_\infty$ Decorations]
\label{def:chiral_bar}
Let $\mathcal{A}$ be an $\OHT$-algebra (Definition~\ref{def:ainfty_chiral}). The \emph{chiral bar complex} is:
\begin{equation}
\overline{B}_{\text{ch}}(\mathcal{A}) = \bigoplus_{n \geq 0} \Omega^\bullet_{\log}\big(\FM_{n+1}(\C)\big) \widehat{\otimes} \mathcal{A}^{\otimes (n+1)},
\end{equation}
equipped with differential:
\begin{equation}
d_{\overline{B}} = d_Q + d_{\text{res}} + d_{A_\infty},
\end{equation}
where:
\begin{enumerate}[label=(\roman*)]
\item $d_Q$ is the internal BV-BRST differential acting on $\mathcal{A}$:
\begin{equation}
d_Q(a_0 \otimes \cdots \otimes a_n) = \sum_{i=0}^n (-1)^{|a_0| + \cdots + |a_{i-1}|} (a_0 \otimes \cdots \otimes Qa_i \otimes \cdots \otimes a_n);
\end{equation}

\item $d_{\text{res}}$ is the \emph{boundary residue differential} from FM collision divisors:
\begin{equation}
d_{\text{res}} = \sum_{\substack{S \subseteq \{0, \ldots, n\} \\ |S| \geq 2}} (-1)^{\varepsilon(S)} \Res_{D_S},
\end{equation}
where $\Res_{D_S}$ extracts the residue at boundary divisor $D_S$ (points in $S$ collide);

\item $d_{A_\infty}$ is the \emph{$A_\infty$ grafting differential}, implementing composition via higher operations $m_k$:
\begin{equation}
d_{A_\infty}(a_0 \otimes \cdots \otimes a_n) = \sum_{k \geq 1} \sum_{0 \leq i < i+k \leq n} (-1)^{\varepsilon(i,k)} (a_0 \otimes \cdots \otimes m_k(a_i, \ldots, a_{i+k-1}) \otimes \cdots \otimes a_n).
\end{equation}
\end{enumerate}
\end{definition}

\begin{remark}[Geometric interpretation]
The three components of $d_{\overline{B}}$ are the three faces of the HT geometry: $d_Q$ is the quantum equation of motion; $d_{\mathrm{res}}$ is holomorphic collision (chiral OPE); $d_{A_\infty}$ is topological concatenation (homotopy coherence).
\end{remark}

\subsubsection{Proof of $d_{\overline{B}}^2 = 0$}

\begin{theorem}[$d_{\overline{B}}^2 = 0$; \ClaimStatusProvedHere]
\label{thm:bar_differential_squared}
The chiral bar differential satisfies $d_{\overline{B}}^2 = 0$.
\end{theorem}

\begin{proof}
We compute:
\begin{equation}
d_{\overline{B}}^2 = (d_Q + d_{\text{res}} + d_{A_\infty})^2 = d_Q^2 + d_{\text{res}}^2 + d_{A_\infty}^2 + \{d_Q, d_{\text{res}}\} + \{d_Q, d_{A_\infty}\} + \{d_{\text{res}}, d_{A_\infty}\}.
\end{equation}

We verify each term vanishes:

\textbf{(i) $d_Q^2 = 0$:} This holds since $Q^2 = 0$ (BV-BRST nilpotency).

\textbf{(ii) $d_{\text{res}}^2 = 0$:} This is the content of the \textbf{Arnold cancellation} (Theorem \ref{thm:Arnold_full_proof}): codimension-2 corners in $\FM_k(\C)$ admit two factorizations (collide $S$ then $T$, or $T$ then $S$), and the residues cancel with appropriate signs:
\begin{equation}
\Res_{D_S} \circ \Res_{D_T} + \Res_{D_T} \circ \Res_{D_S} = 0.
\end{equation}

This is the FM--Stokes--Arnold package developed in Section \ref{sec:FM_calculus} and Appendix \ref{app:FM_Stokes}.

\textbf{(iii) $d_{A_\infty}^2 = 0$:} The full condition $d_{\overline{B}}^2 = 0$ is equivalent to the $A_\infty$ identity. The six-term decomposition shows that this total condition factors into three individual conditions ($d_Q^2 = 0$, $d_{\mathrm{res}}^2 = 0$, $d_{A_\infty}^2 = 0$) and three compatibility conditions. Item~(iii) encodes the purely algebraic part of the $A_\infty$ identity (the composition relations among the $m_k$ (Definition \ref{def:ainfty_chiral}, equation \eqref{eq:ainfty-relation-raw})) while items~(i)--(ii) and (iv)--(vi) encode the compatibility with the differential~$Q$, the residue structure, and the cluster factorization. Explicitly:
\begin{equation}
\sum_{i+j=k+1} (-1)^{\varepsilon(i,j)} m_i \circ m_j = 0.
\end{equation}

\textbf{(iv) $\{d_Q, d_{\text{res}}\} = 0$:} The BV-BRST differential $Q$ commutes with residues because $Q$ acts fiberwise on $\mathcal{A}$, while $\Res_{D_S}$ acts on the FM base space.

\textbf{(v) $\{d_Q, d_{A_\infty}\} = 0$:} This is the statement that $Q = m_1$ is compatible with higher operations: $Q(m_k(a_1, \ldots, a_k)) = \sum (-1)^{\varepsilon} m_k(a_1, \ldots, Qa_i, \ldots, a_k)$, which follows from the $k=2$ case of the $A_\infty$ identity.
\textbf{(vi) $\{d_{\text{res}}, d_{A_\infty}\} = 0$:} This is the \textbf{cluster factorization identity} (Theorem \ref{thm:cluster_factorization}): the $A_\infty$ composition via $m_k$ is compatible with FM boundary collisions. Explicitly, when points $\{i, i+1, \ldots, i+k-1\}$ collide in $\FM_n(\C)$, the boundary contribution is:
\begin{equation}
\Res_{D_{\{i, \ldots, i+k-1\}}} \sim m_k(a_i, \ldots, a_{i+k-1}),
\end{equation}
which is precisely the action of $d_{A_\infty}$.

Summing all contributions:
\begin{equation}
d_{\overline{B}}^2 = 0 + 0 + 0 + 0 + 0 + 0 = 0.
\end{equation}
\end{proof}

\begin{remark}[Identification with Vol~I bar differential]
\label{rem:bar-differential-vol1-identification}
The three-part decomposition $d_{\overline{B}} = d_Q + d_{\mathrm{res}}
+ d_{A_\infty}$ is the geometric presentation of the single bar
coderivation $d_B$ on $T^c(s^{-1}\bar{\cA})$ from Vol~I
(Theorem~A): $d_Q$ encodes the internal differential $m_1$,
$d_{\mathrm{res}}$ the binary product $m_2$ via FM collision
residues, and $d_{A_\infty}$ the higher operations $m_{k \geq 3}$.
The six-term computation above is thus an independent geometric
verification of Vol~I's algebraic $d_B^2 = 0$; the two results
are equivalent, not merely analogous.
\end{remark}

\subsubsection{Structural properties of the $\Ainf$ tower}

The $\Ainf$ operations $\{m_k\}_{k \ge 2}$ extracted from the
bar differential carry two structural properties that hold for
any logarithmic $\SCchtop$-algebra. Both are consequences of
the mechanism by which the central charge enters the $\lambda$-bracket:
as a scalar contact term annihilated by the bracket in both slots.

\begin{proposition}[$c$-linearity of the $\Ainf$ tower; \ClaimStatusProvedHere]
\label{prop:c-linearity-general}
\index{c-linearity@$c$-linearity!general $\Ainf$ tower|textbf}
Let\/ $\cA$ be a logarithmic $\SCchtop$-algebra whose
$\lambda$-bracket has central charge~$c$
\textup{(}i.e.\ a scalar summand proportional to~$c$
in the singular OPE\textup{)}. Then for all $k \ge 2$, the
chain-level $\Ainf$ operation $m_k$ decomposes as
\[
m_k(a_1, \ldots, a_k;\, \lambda_1, \ldots, \lambda_{k-1})
\;=\;
m_k^{\mathrm{state}}(a_1, \ldots, a_k;\, \lambda)
\;+\;
c \cdot Q_k(\lambda_1, \ldots, \lambda_{k-1})\cdot\mathbf{1},
\]
where $m_k^{\mathrm{state}}$ is $c$-independent and
$Q_k$ is a polynomial in the spectral parameters. In particular,
the scalar part of $m_k$ is \textbf{linear} in~$c$.
\end{proposition}

\begin{proof}
Induction on~$k$. At $k = 2$ the claim is the definition:
the central charge enters the $\lambda$-bracket as a scalar
summand $c \cdot q(\lambda) \cdot \mathbf{1}$. For the inductive
step, the Stasheff relation expresses $d\,m_k$ as a sum of
compositions $m_i \circ m_j$ ($i+j = k+1$). Scalars are
annihilated by the $\lambda$-bracket in both slots
($m_2(\mathbf{1}, a; \lambda) = 0 = m_2(a, \mathbf{1}; \lambda)$),
so the only scalar contributions come from composing the
$c$-independent state part of one factor with the scalar part
of another. Both channels produce terms linear in~$c$ by the
inductive hypothesis. The BRST homotopy~$h$, being a
$c$-independent operator on the ambient module, preserves
$c$-linearity.
\end{proof}

\begin{proposition}[Finite presentation of the $\Ainf$ tower; \ClaimStatusProvedHere]
\label{prop:finite-presentation-general}
\index{finite presentation!general $\Ainf$ tower|textbf}
\index{resolvent tree formula!general}
For any logarithmic $\SCchtop$-algebra~$\cA$ realised by
homological perturbation from a dg resolution
$(V, d_0, m_2^0)$ with contracting homotopy~$h$
and perturbation~$\delta = m_2 - m_2^0$,
the entire $\Ainf$ tower $\{m_k\}_{k \ge 3}$ is finitely
presented by the pair $(m_2, h)$: each $m_k$ is the sum
over Catalan-many planar binary trees with $k$ leaves,
internal edges decorated by~$h$, and internal vertices
by~$\delta$ \textup{(}the resolvent tree formula of Volume~I,
Construction~\ref*{V1-constr:htt-alg}\textup{)}.
\end{proposition}

\begin{proof}
This is the standard homological perturbation lemma
applied to the $\SCchtop$ setting.
The transferred operations $m_k^H = p \circ
\sum_{\text{trees}} (\delta \circ h)^{k-2} \circ \delta
\circ i^{\otimes k}$ are determined by the resolvent
$(1 - \delta\,h)^{-1}$, whose expansion over planar
binary trees gives the formula. The sum converges in the
filtration-complete topology ensured by tameness
(Definition~\ref{def:tameness}).
\end{proof}

\begin{remark}[Content]
\label{rem:c-linearity-content}
The $c$-linearity theorem states that the
field-dependent part of $m_k$ (the ``backreaction'') is
universal, independent of the central charge, while the
scalar contact term is controlled by a single coupling
constant. Finite presentation states that one binary
operation and one homotopy produce the entire infinite
tower. The tower may not truncate ($m_k \ne 0$ for all
$k \ge 3$ generically), but the generating data are finite.
Both results specialise to concrete formulas in examples:
for the Virasoro algebra, see
Theorem~\ref{thm:gravity-c-linearity} and
Theorem~\ref{thm:gravity-finite-presentation}.
\end{remark}

\subsection{The Chiral Cobar Construction}
Dually, we define:

\begin{definition}[Chiral Cobar Complex]
\label{def:chiral_cobar}
For a conilpotent chiral coalgebra $\mathcal{C}$ (encoding distributions on configuration spaces), the \emph{chiral cobar complex} is:
\begin{equation}
\Omega_{\text{ch}}(\mathcal{C}) = \bigoplus_{n \geq 0} \text{Dist}\big(\FM_{n+1}(\C)\big) \widehat{\otimes} \mathcal{C}^{\otimes (n+1)},
\end{equation}
with differential:
\begin{equation}
d_{\Omega} = d_{\text{co}Q} + d_{\text{cores}} + d_{\text{co}-A_\infty},
\end{equation}
where each term is the dual/opposite of the corresponding term in $d_{\overline{B}}$.
\end{definition}

\subsection{Bar--Cobar Adjunction and Twisting Morphisms}

\begin{theorem}[Bar--Cobar Adjunction; \ClaimStatusProvedHere]
\label{thm:bar_cobar_adjunction}
There is a natural adjunction:
\begin{equation}
\Omega_{\text{ch}} : (\text{conilpotent chiral coalgebras}) \rightleftarrows (\OHT\text{-algebras}) : \overline{B}_{\text{ch}},
\end{equation}
whose unit and counit are defined by universal twisting morphisms satisfying a Maurer--Cartan equation in the convolution dg Lie algebra $\Convstr$ (strict model; Convention~\ref{conv:vol2-strict-models}).
\end{theorem}

\begin{proof}
The proof applies the standard operadic bar--cobar formalism \cite{BV73,MSS02} to the two-colored HT operad $\OHT = C_\ast(W(\SCchtop))$, verifying the additional structure specific to the chiral-topological setting.

\textbf{Step 1: Conilpotence of the bar construction.}
The bar construction $\overline{B}_{\mathrm{ch}}(\mathcal{A})$ is the cofree conilpotent cooperad on $s^{-1}\overline{\mathcal{A}}$ (the desuspension of the augmentation ideal). Conilpotence means: for every element $x \in \overline{B}_{\mathrm{ch}}(\mathcal{A})$, the iterated coproduct $\Delta^{(n)}(x) = 0$ for $n$ sufficiently large. This holds because:
\begin{enumerate}[label=(\alph*)]
\item The bar construction is graded by tree complexity (number of internal edges). The coproduct $\Delta$ decomposes a tree by cutting an internal edge, reducing the complexity by at least 1.
\item The chiral spectral parameters $\lambda_i$ contribute to a secondary grading by pole order. The tameness condition (Definition~\ref{def:tameness}) ensures finite pole order at each stage, preventing infinite iterations.
\end{enumerate}
Together, these give a filtration by total complexity (tree size + pole order) that is bounded below and exhaustive, ensuring conilpotence.

\textbf{Step 2: Convolution Lie algebra and Maurer--Cartan.}
The convolution dg Lie algebra
$\mathfrak{g} = \Convstr(\overline{B}_{\mathrm{ch}}(\mathcal{A}), \mathcal{B})
= \mathrm{Hom}(\overline{B}_{\mathrm{ch}}(\mathcal{A}), \mathcal{B})$
is the strict model of the homotopy-invariant $L_\infty$-algebra
$\Convinf$, by the operadic homotopy-convolution theorem of Volume~I,
with:
\begin{itemize}
\item Differential $d_{\mathfrak{g}}(f) = d_{\mathcal{B}} \circ f - (-1)^{|f|} f \circ d_{\overline{B}}$;
\item Lie bracket $[f,g] = f \star g - (-1)^{|f||g|} g \star f$, where the convolution $f \star g$ is defined by $\Delta(x) \mapsto \mu(f \otimes g)(\Delta(x))$, using the cooperad structure $\Delta$ on $\overline{B}_{\mathrm{ch}}$ and the operad composition $\mu$ on $\mathrm{End}_{\mathcal{B}}$.
\end{itemize}
A twisting morphism $\tau: \overline{B}_{\mathrm{ch}}(\mathcal{A}) \to \mathcal{A}^!$ is a Maurer--Cartan element: $d_{\mathfrak{g}}\tau + \frac{1}{2}[\tau,\tau] = 0$. The universal twisting morphism is the projection $\tau_{\mathrm{univ}}: \overline{B}_{\mathrm{ch}}(\mathcal{A}) \twoheadrightarrow s^{-1}\overline{\mathcal{A}}$, and the MC equation for $\tau_{\mathrm{univ}}$ encodes exactly the $A_\infty$ relations \eqref{eq:ainfty-relation-raw}.

\textbf{Step 3: Cluster factorization identity.}
The compatibility $\{d_{\mathrm{res}}, d_{A_\infty}\} = 0$ (where $d_{\mathrm{res}}$ is the residue differential from FM boundary faces and $d_{A_\infty}$ is the $A_\infty$ differential) is the cluster factorization identity (Theorem~\ref{thm:cluster_factorization}): when a cluster of points collides in $\FM_k(\C)$, the resulting boundary term decomposes as a composition of lower-degree operations with spectral substitution $\Lambda_I = \sum_{i \in I} \lambda_i$. This identity is verified by Stokes' theorem on $\FM_k(\C)$ (Section~\ref{sec:FM_calculus}).

\textbf{Step 4: Adjunction.}
The bar and cobar functors form an adjunction by the universal property: giving a map $\Omega_{\mathrm{ch}}(C) \to \mathcal{A}$ (from the free operad on $sC$) is equivalent to giving a twisting morphism $C \to \mathrm{End}_{\mathcal{A}}$, which is equivalent to giving a coalgebra map $C \to \overline{B}_{\mathrm{ch}}(\mathcal{A})$. Naturality in both variables is immediate from the universal constructions.

When $\mathcal{A}$ is quadratic ($m_{k \geq 3} = 0$), the bar complex truncates to trees with binary vertices, recovering the GLZ bar construction~\cite{GLZ22}.
\end{proof}

\subsection{GLZ Compatibility: Quadratic Reduction}

\begin{theorem}[GLZ Compatibility; \ClaimStatusProvedHere]
\label{thm:GLZ_compatibility}
Let $\mathcal{A}$ be a quadratic chiral algebra (Definition \ref{def:quadratic_chiral}) with $m_{k \geq 3} = 0$. Then:
\begin{enumerate}[label=(\roman*)]
\item The chiral bar complex $\overline{B}_{\text{ch}}(\mathcal{A})$ reduces to the GLZ bar construction:
\begin{equation}
\overline{B}_{\text{ch}}(\mathcal{A})|_{m_{k \geq 3} = 0} \cong \overline{B}_{\text{GLZ}}(\mathcal{A});
\end{equation}

\item The Koszul dual $\mathcal{A}^! = H^*(\overline{B}_{\text{ch}}(\mathcal{A}))^\vee$
\textup{(}via Verdier duality; note
$\Omega_{\text{ch}}(\overline{B}_{\text{ch}}(\mathcal{A}))
\simeq \mathcal{A}$ by bar-cobar inversion, not~$\mathcal{A}^!$\textup{)}
agrees with the GLZ Koszul dual (Definition \ref{def:GLZ_Koszul});

\item The Maurer--Cartan control (Theorem \ref{thm:GLZ_MC}) lifts to our setting:
\begin{equation}
\Hom_{\OHT}(\mathcal{A}, \mathcal{B}) \longleftrightarrow \MC\big(\overline{B}_{\text{ch}}(\mathcal{A}) \widehat{\otimes} \mathcal{B}\big).
\end{equation}
\end{enumerate}
\end{theorem}

\begin{proof}
When $m_{k \geq 3} = 0$, the differential $d_{A_\infty} = 0$, so:
\begin{equation}
d_{\overline{B}} = d_Q + d_{\text{res}}.
\end{equation}

This is the structure of the GLZ bar construction \cite{GLZ22}, where:
\begin{itemize}
\item $d_Q$ is the internal differential (Chevalley--Eilenberg differential for Lie algebras, or derived bracket for chiral algebras);
\item $d_{\text{res}}$ encodes the chiral OPE via residues at collision divisors.
\end{itemize}

The GLZ Koszul dual is recovered because, in the quadratic case, the cobar construction $\Omega_{\text{ch}}$ produces exactly the free chiral algebra on $V^* \otimes \omega_\C[1]$ modulo the annihilator relations $R^\perp$.

The Maurer--Cartan bijection follows from the standard argument: morphisms $\mathcal{A} \to \mathcal{B}$ correspond to twisting cochains $\tau : \overline{B}_{\text{ch}}(\mathcal{A}) \to \mathcal{B}$ satisfying $d\tau + \frac{1}{2}[\tau, \tau] = 0$.
\end{proof}

\subsection{ChiralBar as classifying space}
\label{subsec:chiralbar-classifying-space}

The bar-cobar adjunction (Theorem~\ref{thm:bar_cobar_adjunction})
and its GLZ specialization (Theorem~\ref{thm:GLZ_compatibility})
produce a bijection between coalgebra maps into the bar complex
and Maurer--Cartan elements in a convolution algebra. This is
the $0$-simplex shadow of a richer statement: the bar complex
$\barB_{\mathrm{ch}}(\cA)$ is the \emph{classifying space} of
$\cA$-deformations, in the sense that it represents the formal
moduli functor of Maurer--Cartan elements as a simplicial set.
The Lurie--Pridham correspondence between formal moduli problems
and dg Lie algebras, when read through the bar-cobar lens,
identifies $\barB_{\mathrm{ch}}(\cA)$ as the universal target
for twisting morphisms, the chiral analogue of the classifying
space $BG$ for principal bundles.

\begin{theorem}[Representability; \ClaimStatusProvedHere]
\label{thm:bar-representability}
\index{bar complex!representability|textbf}
\index{classifying space!chiral bar complex|textbf}
\index{Maurer--Cartan!simplicial set!representability}
For any logarithmic $\SCchtop$-algebra~$\cA$, the bar
construction $\barB_{\mathrm{ch}}(\cA)$ represents the functor
of Maurer--Cartan elements: for any conilpotent
$\SCchtop$-coalgebra~$\cC$,
\begin{equation}\label{eq:bar-representability}
\mathrm{Map}_{\mathrm{CoAlg}}(\cC,\, \barB(\cA))
\;\simeq\;
\MC_\bullet\!\bigl(\mathrm{Conv}(\cC, \cA)\bigr)
\end{equation}
as simplicial sets, where $\MC_\bullet$ is the Dupont--Sullivan
simplicial Maurer--Cartan space. This is the Lurie--Pridham
correspondence: $\barB(\cA)$ represents the formal moduli problem
of $\cA$-deformations.
\end{theorem}

\begin{proof}
The proof proceeds in two stages: the set-level bijection
(already established) and the simplicial upgrade.

\medskip\noindent
\textbf{Stage 1: $0$-simplices.}
By Theorem~\ref{thm:bar_cobar_adjunction}, there is a natural
bijection
\[
\Hom_{\mathrm{CoAlg}}(\cC,\, \barB(\cA))
\;\cong\;
\mathrm{Tw}(\cC, \cA)
\;\cong\;
\MC\!\bigl(\mathrm{Conv}(\cC, \cA)\bigr),
\]
where $\mathrm{Tw}(\cC, \cA)$ denotes twisting morphisms and
$\mathrm{Conv}(\cC, \cA)
= \Hom(\cC, \cA)$ is the convolution dg Lie algebra
(the strict model $\Convstr$ of
Convention~\ref{conv:vol2-strict-models}). A twisting morphism
$\tau\colon \cC \to \cA$ is precisely a degree-$(-1)$ element
satisfying $d\tau + \frac{1}{2}[\tau,\tau] = 0$ in
$\mathrm{Conv}(\cC, \cA)$, hence a Maurer--Cartan element.

\begin{remark}[Two convolution structures]
\label{rem:two-convolutions}
The convolution dg Lie algebra $\mathrm{Conv}(\cC, \cA)$ arises in two equivalent ways: (i)~when $\cC = \barB(\cA)$ is the bar construction, the cooperad structure on $\barB(\cA)$ and the operad structure on $\mathrm{End}_\cA$ give an $L_\infty$-algebra via the operadic convolution of \cite{LV12}, Chapter~6; (ii)~for a general conilpotent coalgebra $\cC$ and algebra $\cA$, the associative convolution product $f \star g = \mu_\cA \circ (f \otimes g) \circ \Delta_\cC$ on $\mathrm{Hom}(\cC, \cA)$ yields a dg Lie algebra by taking graded commutators. When $\cC = \barB(\cA)$, the two constructions agree because the cooperad structure on $\barB(\cA)$ is determined by the coalgebra structure via the cofree property. In the representability theorems above, we use convention~(ii) for general $\cC$ and convention~(i) when $\cC = \barB(\cA)$.
\end{remark}

The universal twisting morphism
$\tau_{\mathrm{univ}}\colon \barB(\cA) \twoheadrightarrow
s^{-1}\bar{\cA}$ witnesses the universality: every
$\tau\colon \cC \to \cA$ factors uniquely through
$\tau_{\mathrm{univ}}$ via a coalgebra map
$f\colon \cC \to \barB(\cA)$. This establishes the bijection on
$0$-simplices.

\medskip\noindent
\textbf{Stage 2: Simplicial enrichment.}
The $n$-simplices of $\MC_\bullet(\mathrm{Conv}(\cC, \cA))$
are Maurer--Cartan elements in
$\mathrm{Conv}(\cC, \cA) \otimes \Omega_n$,
where $\Omega_n = \Omega^\bullet_{\mathrm{poly}}(\Delta^n)$
is the Sullivan algebra of polynomial differential forms
on the standard $n$-simplex
(Dupont~\cite{Dup76}; Robert-Nicoud--Vallette~\cite{RNV20},
\S3). The face and degeneracy maps are induced by the
cosimplicial structure of $\{\Omega_n\}_{n \geq 0}$.

The coalgebra structure on $\barB(\cA) \otimes \Omega_n$ arises from the diagonal on $\Omega_n$: the Sullivan algebra $\Omega_n$ is a commutative dg algebra, hence carries a cocommutative coalgebra structure via the multiplication map $\mu: \Omega_n \otimes \Omega_n \to \Omega_n$ dualized. Tensoring a conilpotent coalgebra with a coalgebra of finite type (over a field of characteristic zero) produces a conilpotent coalgebra. Since $\Omega_n$ is of finite type (finite-dimensional in each degree), $\barB(\cA) \otimes \Omega_n$ is conilpotent.
The bar-cobar adjunction commutes with this
tensoring:
\[
\Hom_{\mathrm{CoAlg}}\!\bigl(\cC,\,
 \barB(\cA) \otimes \Omega_n\bigr)
\;\cong\;
\MC\!\bigl(\mathrm{Conv}(\cC, \cA) \otimes \Omega_n\bigr).
\]
The left-hand side, assembled over all~$n$, gives the simplicial
mapping space
$\mathrm{Map}_{\mathrm{CoAlg}}(\cC, \barB(\cA))$;
the right-hand side gives
$\MC_\bullet(\mathrm{Conv}(\cC, \cA))$.
Naturality in~$n$ (functoriality of the cosimplicial structure)
ensures the isomorphism is simplicial.

\medskip\noindent
\textbf{Formal moduli interpretation.}
By the Lurie--Pridham correspondence
(Lurie~\cite{Lur11}, Theorem~5.3;
Pridham~\cite{Pri10}, Theorem~4.26),
formal moduli problems over a field of characteristic zero are
equivalent to dg Lie algebras up to quasi-isomorphism.
(Completeness of $\mathrm{Conv}(\cC, \cA)$, required for $\MC_\bullet$ to be well-defined, follows from the conilpotence of $\cC$: the coradical filtration dualises to an inverse-limit filtration making $\mathrm{Conv}(\cC, \cA)$ a complete filtered dg Lie algebra; see the proof of Theorem~\ref{thm:formal-moduli-twisting} below for the detailed argument.)
The dg
Lie algebra governing $\cA$-deformations is
$\mathrm{Conv}(\cC, \cA)$, and its associated formal moduli
problem is the functor
$\cC \mapsto \MC_\bullet(\mathrm{Conv}(\cC, \cA))$.
The representability~\eqref{eq:bar-representability} says this
functor is representable by~$\barB(\cA)$: the bar complex
\emph{is} the classifying space of the formal moduli problem.
\end{proof}

\begin{definition}[Chiral $\cA$-torsor]
\label{def:chiral-torsor}
\index{torsor!chiral|textbf}
\index{chiral torsor|textbf}
Let $\cA$ be a logarithmic $\SCchtop$-algebra on a smooth
curve~$X$. A \emph{chiral $\cA$-torsor} on~$X$ is a
factorization module $\cT$ over~$\cA$ on $\Ran(X)$ equipped with:
\begin{enumerate}[label=\textup{(\roman*)}]
\item A free transitive action
 $\mu\colon \cA \boxtimes \cT \to \Delta_!(\cT)$
 compatible with the chiral product: the diagram
 \[
 \begin{gathered}
 \cA \boxtimes \cA \boxtimes \cT
 \xrightarrow{\;\mu_{12} \boxtimes \mathrm{id}\;}
 \Delta_!(\cA) \boxtimes \cT
 \xrightarrow{\;\mu\;}
 \Delta_!(\cT) \\
 \Big\Vert \\
 \cA \boxtimes \cA \boxtimes \cT
 \xrightarrow{\;\mathrm{id} \boxtimes \mu\;}
 \cA \boxtimes \Delta_!(\cT)
 \xrightarrow{\;\mu\;}
 \Delta_!(\cT)
 \end{gathered}
 \]
 commutes, expressing associativity of the action through the
 chiral product $\mu_{12}$.
\item A trivialization: the fiber of~$\cT$ at each point
 $x \in X$ is quasi-isomorphic to~$\cA$ as an
 $\cA$-module.
\item Factorization compatibility: for $x \neq y$,
 $\cT(x,y) \simeq \cT(x) \otimes \cT(y)$.
\end{enumerate}
\end{definition}

\begin{corollary}[Classifying space interpretation;
\ClaimStatusProvedHere]
\label{cor:classifying-space}
\index{classifying space!chiral torsors|textbf}
\index{torsor!classification by bar complex}
The space of isomorphism classes of chiral $\cA$-torsors on~$X$
is
\begin{equation}\label{eq:classifying-space}
\pi_0\!\Big(\MC_\bullet\!\bigl(
 \mathrm{Conv}(\barB(\cA),\, \mathrm{End}_\cA)
\bigr)\Big)
\;\simeq\;
\pi_0\!\Big(
 \mathrm{Map}_{\mathrm{CoAlg}}\!\bigl(
 \barB(\cA),\, \barB(\cA)
 \bigr)
\Big).
\end{equation}
In other words, $\barB(\cA)$ classifies chiral $\cA$-torsors,
analogously to $BG$ classifying principal $G$-bundles.
\end{corollary}

\begin{proof}
A chiral $\cA$-torsor (Definition~\ref{def:chiral-torsor}) is a
factorization module locally isomorphic to~$\cA$, hence a
twisting of the tautological $\cA$-module by a Maurer--Cartan
element $\alpha \in
\MC\!\bigl(\mathrm{Conv}(\barB(\cA), \mathrm{End}_\cA)\bigr)$.
By the representability theorem
(Theorem~\ref{thm:bar-representability}), such a twisting
morphism $\tau\colon \barB(\cA) \to \mathrm{End}_\cA$ corresponds
to a coalgebra map $f_\tau\colon \barB(\cA) \to
\barB(\mathrm{End}_\cA)$.

For \emph{self-torsors} ($\cA$ acting on itself),
since $\cA$ acts on itself via the regular representation,
the bar construction $\barB(\mathrm{End}_\cA)$ is Morita equivalent
to $\barB(\cA)$; on the Koszul locus, this Morita equivalence
promotes to a quasi-isomorphism of the underlying coalgebras. The twisting
morphism $\tau$ thus determines a coalgebra endomorphism
$f_\tau\colon \barB(\cA) \to \barB(\cA)$.

Two torsors $\cT, \cT'$ are isomorphic if and only if their
Maurer--Cartan elements $\alpha, \alpha'$ are gauge-equivalent:
connected by a path in
$\MC_\bullet(\mathrm{Conv}(\barB(\cA), \mathrm{End}_\cA))$.
Passing to $\pi_0$ identifies isomorphism classes of torsors with
path-components of the simplicial set
$\mathrm{Map}_{\mathrm{CoAlg}}(\barB(\cA), \barB(\cA))$.
\end{proof}

\begin{construction}[The tautological chiral module; \ClaimStatusProvedHere]
\label{constr:tautological-bundle}

Over the classifying coalgebra $\barB(\cA)$, we construct the universal chiral $\cA$-module $\cE_{\mathrm{univ}}$ with its flat connection.

\textbf{Step 1: Fibers.} For a bar chain $\phi = [s^{-1}a_1 | \cdots | s^{-1}a_n] \in \barB^n(\cA)$ (a degree-$n$ element of the tensor coalgebra), define the fiber:
$$\cE_{\mathrm{univ}}|_\phi \;:=\; \cA \cdot \{a_1, \ldots, a_n\}$$
the $\cA$-module generated by the insertions $a_1, \ldots, a_n$. Concretely, this is the submodule of $\cA^{\otimes n}$ generated by applying all chiral operations to the tensor $a_1 \otimes \cdots \otimes a_n$.

\textbf{Step 2: Connection.} The connection $\nabla$ on $\cE_{\mathrm{univ}}$ is the bar differential acting on the base:
$$\nabla_\phi(v) \;:=\; d_{\barB}(\phi) \cdot v$$
for $v \in \cE_{\mathrm{univ}}|_\phi$. Here $d_{\barB}(\phi)$ is the bar differential of the base point, and $\cdot$ denotes the module action extending by the Leibniz rule.

\textbf{Step 3: Flatness.} The flatness of the connection is equivalent to $d_{\barB}^2 = 0$:
$$\nabla^2(\phi)(v) = d_{\barB}^2(\phi) \cdot v = 0 \qquad \text{(at genus 0)}$$
This is exactly the content of Volume~I Theorem~A: the bar differential squares to zero, equivalently, the $\Ainf$ relations hold.

\textbf{Step 4: Curvature at genus $g \geq 1$.} At genus $g$, the connection acquires curvature:
$$\nabla^2(\phi)(v) = d_{\barB}^2(\phi) \cdot v = \kappa(\cA) \cdot \omega_g \cdot v$$
The tautological module is no longer flat: it carries the conformal anomaly $\kappa(\cA) \cdot \omega_g$ as its curvature $2$-form. Its sections over $\overline{\cM}_g$ define the corresponding cohomological / derived transport package of the bar family; on the integrable affine comparison surface, $H^0$ specializes to the usual conformal blocks. The curvature measures the failure of that transport package to remain flat, and on the integrable affine surface this is the familiar failure of conformal blocks to form a flat bundle.

\textbf{Step 5: Pullback.} For any twisting morphism $\tau: \cC \to \cA$ (equivalently, a coalgebra map $f_\tau: \cC \to \barB(\cA)$ by Theorem~\ref{thm:bar-terminality}), the pullback $f_\tau^* \cE_{\mathrm{univ}}$ is a chiral $\cA$-module on $\cC$:
$$f_\tau^* \cE_{\mathrm{univ}}|_c \;=\; \cE_{\mathrm{univ}}|_{f_\tau(c)} \;=\; \cA \cdot \{\tau(c^{(1)}), \ldots, \tau(c^{(n)})\}$$
where $\Delta^{(n-1)}(c) = \sum c^{(1)} \otimes \cdots \otimes c^{(n)}$ is the iterated coproduct. The MC equation $\partial\tau + \tau \star \tau = 0$ ensures the pulled-back connection is compatible with the $\cA$-action.
\end{construction}

\begin{example}[Affine Kac--Moody]
\label{ex:affine-classifying-space}
\index{Kac--Moody!classifying space|textbf}
\index{conformal block!as chiral torsor}
For $\cA = \hat{\fg}_k$ (the affine Kac--Moody vertex algebra at
level~$k$), the classifying space interpretation takes a
concrete and classical form.

\medskip\noindent
\textbf{Torsors, conformal blocks, and opers.}
A chiral $\hat{\fg}_k$-torsor on $X$ is a \emph{principal chiral
$\hat{\fg}_k$-module} on $X$: a factorization module over
$\hat{\fg}_k$ on $\Ran(X)$ that is locally isomorphic to
$\hat{\fg}_k$ as a module over itself. In the
Beilinson--Drinfeld framework, this corresponds to a twisted
$\cD$-module on $\operatorname{Bun}_G(X)$. The space of \emph{conformal
blocks} at marked points $(x_1,\ldots,x_n)$ with representations
$(V_1,\ldots,V_n)$ is obtained from a chiral torsor by taking
coinvariants:
$\langle V_1,\ldots,V_n \rangle_X = H_0(X, \hat{\fg}_k \otimes
\cT)$. At the critical level $k = -h^\vee$, where the
Feigin--Frenkel center acts, chiral torsors reduce to opers
(flat $\check{G}$-connections on $X$).

Concretely, the data of
Definition~\ref{def:chiral-torsor} specialize to:
\begin{itemize}
\item The free transitive action $\mu$ is the current algebra
 action $J^a(z) \cdot \cT \to \cT$ (the affine Lie algebra
 $\hat{\fg}$ at level~$k$ acting on the module~$\cT$).
\item The trivialization condition says $\cT$ is locally free as a
 $\hat{\fg}_k$-module: at each $x \in X$, the fiber
 $\cT_x \simeq \hat{\fg}_k$ as a representation.
\item Factorization compatibility is the sewing axiom for
 conformal blocks: insertions at disjoint points factorize.
\end{itemize}
The classifying space $\barB(\hat{\fg}_k)$ parametrizes these
torsors: by Corollary~\ref{cor:classifying-space},
$\pi_0(\mathrm{Map}_{\mathrm{CoAlg}}(\barB(\hat{\fg}_k),
\barB(\hat{\fg}_k)))$ classifies chiral $\hat{\fg}_k$-torsors
on~$X$ up to isomorphism.

\medskip\noindent
\textbf{Koszul dual and dual conformal blocks.}
Restrict now to the affine chirally Koszul locus
\textup{(}for the standard HT gauge realizations, supplied by
Theorem~\textup{\ref{thm:one-loop-koszul}}\textup{)}. On this
locus, the chiral Koszul dual
$\cA^!_{\mathrm{ch}} = \mathrm{CE}^{\mathrm{ch}}(\hat{\fg}_{-k-2h^\vee})$ (the chiral CE algebra at the
dual level) governs the
\emph{dual} conformal blocks: the contragredient representations
at level $-k - 2h^\vee$. The representability theorem
(Theorem~\ref{thm:bar-representability}) gives
\[
\MC_\bullet\!\bigl(\mathrm{Conv}(\cC, \hat{\fg}_k)\bigr)
\;\simeq\;
\mathrm{Map}_{\mathrm{CoAlg}}(\cC, \barB(\hat{\fg}_k))
\]
for any conilpotent coalgebra~$\cC$, and the Koszul duality
involution $(\cA^!)^! \simeq \cA$
(Theorem~\ref{thm:duality-involution}) interchanges
the two classifying spaces:
$\barB(\hat{\fg}_k)^! \simeq \barB(\hat{\fg}_{-k-2h^\vee})$.

\medskip\noindent
\textbf{Genus $g \geq 1$: the conformal anomaly as gerbe.}
At genus~$g \geq 1$, the tautological bundle acquires curvature
$\kappa \cdot \omega_g$ (Construction~\ref{constr:tautological-bundle}).
The modular characteristic $\kappa(\hat{\fg}_k)$ is defined in
Volume~I as the leading coefficient of the bar obstruction; for
the affine lineage it equals the level for the Heisenberg algebra
and involves the Sugawara central charge for non-abelian~$\fg$.
Explicitly, $\kappa(\cH_k) = k$ for the Heisenberg ($\dim\fg = 1$),
while for simple~$\fg$ the modular characteristic is
$\kappa(\hat{\fg}_k) = \dim\fg\cdot(k+h^\vee)\,/\,(2h^\vee)$
(Volume~I, Theorem~D), with
$\kappa(\hat{\fg}_{-k-2h^\vee}) = -\kappa(\hat{\fg}_k)$. The
genus-$g$ bar complex $\barB^{(g)}(\hat{\fg}_k)$ is a
\emph{gerbe} over $\overline{\mathcal{M}}_g$ with
characteristic class $\kappa(\hat{\fg}_k) \cdot \lambda_g$, where
$\lambda_g = c_g(\mathbb{E})$ is the top Chern class of the
Hodge bundle $\mathbb{E} \to \overline{\mathcal{M}}_g$.
Concretely:
\begin{itemize}
\item The curvature $\dfib^{\,2} = \kappa(\hat{\fg}_k) \cdot
 \omega_g$ is the conformal anomaly: the failure of the
 partition function $Z_{g,k}$ to be a function (rather than a
 section of a line bundle) on~$\overline{\mathcal{M}}_g$.
\item The characteristic class
 $\kappa(\hat{\fg}_k) \cdot \lambda_g \in H^2(\overline{\mathcal{M}}_g)$,
 with $\kappa(\hat{\fg}_k) = \dim\fg\cdot(k+h^\vee)\,/\,(2h^\vee)$, is
 the modular characteristic times the Hodge class (Vol~I,
 Theorem~D).
\item At the critical level $k = -h^\vee$, the modular
 characteristic is $\kappa(\hat{\fg}_{-h^\vee}) = 0$:
 the bar complex is \emph{uncurved}.
 The Sugawara construction is undefined at
 $k = -h^\vee$ (not merely singular), and the Feigin--Frenkel
 center $\mathfrak{z}(\hat{\fg}_{-h^\vee})$ provides a large
 commutative subalgebra.
\item The complementarity theorem (Vol~I, Theorem~C) gives
 $\kappa(\hat{\fg}_k) + \kappa(\hat{\fg}_{-k-2h^\vee}) = 0$
 exactly: the Feigin--Frenkel involution
 $k\mapsto -k-2h^\vee$ negates the shifted level $k+h^\vee$,
 hence negates~$\kappa$. The tensor product
 $\barB^{(g)}(\hat{\fg}_k) \otimes
 \barB^{(g)}(\hat{\fg}_{-k-2h^\vee})$ has vanishing total
 curvature at every genus.
\end{itemize}
\end{example}

\subsection{The universal property: initiality, terminality, and formal moduli}
\label{subsec:universal-property-twisting}

The classifying-space theorem
(Theorem~\ref{thm:bar-representability}) identifies the bar complex
as the representing object for a formal moduli functor. The
present subsection refines this in two directions: first, it
isolates the \emph{set-level} universal properties
(terminality of bar, initiality of cobar) that underlie the
representability; second, it promotes the representability to the
full Lurie--Pridham correspondence between formal moduli problems
and complete dg Lie algebras. The result is the platonic form of
the bar--cobar adjunction: a single bifunctor of twisting morphisms,
representable on both sides, whose derived moduli space is the
simplicial Maurer--Cartan space of the convolution algebra.

\begin{theorem}[Terminality of the bar construction;
\ClaimStatusProvedHere]
\label{thm:bar-terminality}
\index{bar construction!terminality|textbf}
\index{twisting morphism!universal property|textbf}
The bar construction $\barB(\cA)$ is terminal in the category of
conilpotent $\SCchtop$-coalgebras equipped with a twisting morphism
to~$\cA$. Explicitly: for any conilpotent coalgebra~$\cC$ and any
twisting morphism $\tau\colon \cC \to \cA$, there exists a unique
coalgebra morphism $f_\tau\colon \cC \to \barB(\cA)$ such that
$\tau = \pi \circ f_\tau$, where
$\pi\colon \barB(\cA) \twoheadrightarrow s^{-1}\bar{\cA}$ is the
universal twisting morphism \textup{(}the bar projection onto
cogenerators\textup{)}.
\end{theorem}

\begin{proof}
The map $f_\tau$ is constructed explicitly. On a coalgebra element
$c \in \cC$, apply the iterated coproduct
$\Delta^{(n-1)}\colon \cC \to \cC^{\otimes n}$, then apply
$\tau^{\otimes n}$, then embed into
$T^c(s^{-1}\bar{\cA}) = \barB(\cA)$. The resulting sum
\begin{equation}\label{eq:bar-terminality-formula}
f_\tau(c)
\;=\;
\sum_{n \geq 1} \tau^{\otimes n} \circ \Delta^{(n-1)}(c)
\end{equation}
converges because $\cC$ is conilpotent: the coradical filtration
$F_0\cC \subset F_1\cC \subset \cdots$ is exhaustive, and for
$c \in F_k\cC$ the iterated coproduct $\Delta^{(n-1)}(c) = 0$ for
$n > k$, so the sum is finite.

\medskip\noindent
\textbf{$f_\tau$ is a coalgebra morphism.}
We must verify $(f_\tau \otimes f_\tau) \circ \Delta_\cC
= \Delta_{\barB} \circ f_\tau$, where $\Delta_{\barB}$ is the
deconcatenation coproduct on $T^c(s^{-1}\bar{\cA})$.
By the coassociativity of~$\Delta_\cC$ and the definition
of~\eqref{eq:bar-terminality-formula}, the left-hand side
decomposes as
\[
\sum_{p+q = n}
(\tau^{\otimes p} \otimes \tau^{\otimes q})
\circ \Delta^{(p-1,q-1)}_\cC(c),
\]
where $\Delta^{(p-1,q-1)}_\cC$ denotes the $(p,q)$-component of
the iterated coproduct. This is the deconcatenation of
the $n$-fold tensor $\tau^{\otimes n} \circ \Delta^{(n-1)}(c)$,
which is $\Delta_{\barB} \circ f_\tau(c)$.

\medskip\noindent
\textbf{$f_\tau$ intertwines the differentials.}
The Maurer--Cartan equation
$\partial\tau + \tau \star \tau = 0$
guarantees that $f_\tau$ commutes with the bar differential.
(The expanded form carries standard desuspension signs
$(-1)^{k(k-1)/2}$ on each $m_k$ term; see
Loday--Vallette~\cite{LV12}, Proposition~1.3.3. We use the
compact form $\partial\tau + \tau\star\tau = 0$, which absorbs
these signs into the convolution product~$\star$.)
Concretely, the MC equation is the condition that the
cogenerator projection of $d_{\barB} \circ f_\tau - f_\tau \circ
d_\cC$ vanishes, and for a map into a cofree coalgebra, vanishing
of the cogenerator projection implies vanishing of the full map
(Loday--Vallette~\cite{LV12}, Theorem~2.2.9).

\medskip\noindent
\textbf{Uniqueness.}
A coalgebra morphism $g\colon \cC \to T^c(s^{-1}\bar{\cA})$ into
a cofree conilpotent coalgebra is uniquely determined by its
projection onto cogenerators
$\pi \circ g\colon \cC \to s^{-1}\bar{\cA}$
(Loday--Vallette~\cite{LV12}, Proposition~1.2.6).
The condition $\tau = \pi \circ f_\tau$ pins down this projection;
hence $f_\tau$ is the unique solution.
\end{proof}

\begin{theorem}[Initiality of the cobar construction;
\ClaimStatusProvedHere]
\label{thm:cobar-initiality}
\index{cobar construction!initiality|textbf}
Dually, $\Omegach(\cC)$ is initial in the category of
$\SCchtop$-algebras equipped with a twisting morphism from~$\cC$.
For any algebra~$\cA$ and any twisting morphism
$\tau\colon \cC \to \cA$, there exists a unique algebra morphism
$g_\tau\colon \Omegach(\cC) \to \cA$ such that
$\tau = g_\tau \circ \iota$, where
$\iota\colon s\bar{\cC} \hookrightarrow \Omegach(\cC)$ is the
universal twisting morphism \textup{(}the cobar inclusion of
generators\textup{)}.
\end{theorem}

\begin{proof}
The cobar construction $\Omegach(\cC) = T(s\bar{\cC})$ is the free
algebra on the suspension of the coaugmentation coideal. By the
universal property of free algebras, the linear map
$\tau\colon s\bar{\cC} \to \cA$ extends uniquely to an algebra
morphism $g_\tau\colon T(s\bar{\cC}) \to \cA$. It remains to
verify that $g_\tau$ intertwines the cobar differential
$d_\Omega$ on $T(s\bar{\cC})$ with the differential $d_\cA$
on~$\cA$.

The cobar differential $d_\Omega$ is the unique derivation of
$T(s\bar{\cC})$ whose restriction to generators $s\bar{\cC}$ is
\[
d_\Omega(sx)
\;=\;
-s(d_\cC\, x)
\;-\;
\sum (-1)^{|x'|}\, sx' \otimes sx'',
\]
where $\bar\Delta(x) = \sum x' \otimes x''$ is the reduced
coproduct (Loday--Vallette~\cite{LV12}, Theorem~2.2.5). The
signs arise from the Koszul rule applied to the suspension~$s$:
the first term carries a sign from $s \circ d_\cC \circ s^{-1}
= -s\, d_\cC$ (since $|s|=1$ and $|d_\cC|=1$), while the
second carries the desuspension sign $(-1)^{|x'|}$ from
commuting $s$ past each tensor factor. The formula encodes
the internal differential and the coproduct of~$\cC$.
The Maurer--Cartan equation for $\tau$ ensures that
$g_\tau \circ d_\Omega = d_\cA \circ g_\tau$ on generators; since
both sides are derivations of $T(s\bar{\cC})$, agreement on
generators implies agreement everywhere.

Uniqueness: an algebra morphism from a free algebra is determined
by its restriction to generators, and the condition
$\tau = g_\tau \circ \iota$ fixes this restriction.
\end{proof}

\begin{corollary}[The bar--cobar adjunction as representability;
\ClaimStatusProvedHere]
\label{cor:bar-cobar-representability}
\index{bar-cobar adjunction!representability|textbf}
\index{twisting morphism!bifunctor|textbf}
The pair $(\barB, \Omegach)$ represents the bifunctor of twisting
morphisms:
\begin{equation}\label{eq:bar-cobar-representability}
\Hom_{\mathrm{Alg}}(\Omegach(\cC), \cA)
\;\cong\;
\mathrm{Tw}(\cC, \cA)
\;\cong\;
\Hom_{\mathrm{CoAlg}}(\cC, \barB(\cA))
\end{equation}
naturally in both\/ $\cC$ \textup{(}conilpotent
$\SCchtop$-coalgebras\textup{)} and\/ $\cA$
\textup{(}$\SCchtop$-algebras\textup{)}. This is the bar--cobar
adjunction: $\Omegach \dashv \barB$.
\end{corollary}

\begin{proof}
The left bijection is
Theorem~\ref{thm:cobar-initiality}: a twisting morphism
$\tau \in \mathrm{Tw}(\cC, \cA)$ determines a unique algebra map
$g_\tau\colon \Omegach(\cC) \to \cA$, and conversely every such
algebra map restricts to a twisting morphism via the inclusion of
generators. The right bijection is
Theorem~\ref{thm:bar-terminality}: $\tau$ determines a unique
coalgebra map $f_\tau\colon \cC \to \barB(\cA)$, and conversely
every such coalgebra map projects to a twisting morphism via the
bar projection onto cogenerators.

Naturality in $\cC$ follows from functoriality of the free algebra
construction (for the left bijection) and from the universal
property of cofree coalgebras (for the right). Naturality in $\cA$
is dual. The composite bijection
$\Hom_{\mathrm{Alg}}(\Omegach(\cC), \cA)
\cong \Hom_{\mathrm{CoAlg}}(\cC, \barB(\cA))$
is the adjunction $\Omegach \dashv \barB$.
\end{proof}

\begin{proposition}[Naturality of the bar--cobar adjunction;
\ClaimStatusProvedHere]
\label{prop:bar-cobar-naturality}
\index{bar-cobar adjunction!naturality|textbf}
\index{Drinfeld--Sokolov reduction!bar intertwining|textbf}
The bar--cobar adjunction $\Omegach \dashv \barB$ is natural in the
following sense:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \textbf{Functoriality in $\cA$:} A morphism
 $f\colon \cA \to \cA'$ of $\SCchtop$-algebras induces a coalgebra
 morphism $\barB(f)\colon \barB(\cA) \to \barB(\cA')$ and an algebra
 morphism
 $\Omegach(f^*)\colon \Omegach(\barB(\cA')) \to \Omegach(\barB(\cA))$.
 These are compatible with the bar--cobar counit:
 $\varepsilon_{\cA'} \circ \Omegach(\barB(f)) = f \circ \varepsilon_{\cA}$.

\item \textbf{Under Drinfeld--Sokolov reduction:} For a nilpotent
 element $\chi \in \fg$ and the DS reduction functor
 $\mathrm{DS}_\chi$, the bar construction commutes:
 $\barB(\mathrm{DS}_\chi(\cA))
 \simeq \mathrm{DS}_\chi(\barB(\cA))$ as $E_1$ dg coassociative coalgebras.
 This is Volume~I, Theorem~\textup{\ref*{V1-thm:ds-koszul-intertwine}}.

\item \textbf{Under level change:} For the affine lineage
 $\cA = \hat{\fg}_k$, the family
 $\{\barB(\hat{\fg}_k)\}_{k \in \mathbb{k} \setminus \{-h^\vee\}}$
 forms a flat family of $E_1$ dg coassociative coalgebras parametrised by the
 affine line $\mathbb{A}^1 \setminus \{-h^\vee\}$. The fiber at~$k$
 is $\barB(\hat{\fg}_k)$; the monodromy around $k = -h^\vee$
 \textup{(}the critical level\textup{)} is the Feigin--Frenkel
 involution.

\item \textbf{Modular characteristic transforms:} Under
 $f\colon \cA \to \cA'$, the modular characteristic satisfies
 $\kappa(\cA') = \kappa(\cA) + \delta\kappa(f)$, where
 $\delta\kappa(f)$ is the anomaly of the morphism \textup{(}zero for
 strict morphisms, nonzero for filtered quasi-isomorphisms that shift
 the central charge\textup{)}.
\end{enumerate}
\end{proposition}

\begin{proof}
Part~(i) follows from the functoriality of the tensor coalgebra:
$\barB(\cA) = T^c(s^{-1}\bar{\cA})$ and $f$ induces
$s^{-1}\bar{f}\colon s^{-1}\bar{\cA} \to s^{-1}\bar{\cA}'$, which
extends uniquely to the cofree coalgebra. The compatibility with the
counit is the commutativity of
\[
\begin{tikzcd}
\Omegach(\barB(\cA)) \ar[r,"\varepsilon_{\cA}"] \ar[d,"\Omegach(\barB(f))"'] & \cA \ar[d,"f"] \\
\Omegach(\barB(\cA')) \ar[r,"\varepsilon_{\cA'}"'] & \cA'
\end{tikzcd}
\]
which holds because the counit is the universal twisting morphism and
$f$ preserves the twisting morphism structure
(Theorem~\ref{thm:cobar-initiality}).

Part~(ii) is Volume~I's DS--bar intertwining theorem: the DS functor
is a morphism of operads (it preserves the $\SCchtop$-algebra
structure), so the bar construction, being functorial, commutes with
it. DS reduction acts on the generators
$s^{-1}\bar{\cA}$ of the cofree coalgebra and the bar differential
$d_{\barB}$ restricts to the BRST differential of $\mathrm{DS}_\chi$.

Part~(iii) follows from the polynomial dependence of the OPE
coefficients on~$k$: the structure constants of $\hat{\fg}_k$ are
polynomial in~$k$, so $\barB(\hat{\fg}_k)$ is a polynomial family of
coalgebras. The singularity at $k = -h^\vee$ is the Sugawara pole
(the Sugawara tensor is \emph{undefined} at the critical level, not
infinite; cf.\ Volume~I). The monodromy statement follows from the
Feigin--Frenkel duality $k \leftrightarrow -k - 2h^\vee$, which acts
on the base $\mathbb{A}^1 \setminus \{-h^\vee\}$ as the reflection
fixing $-h^\vee$ and exchanges $\barB(\hat{\fg}_k)$ with the
Langlands dual bar complex.

Part~(iv) is a consequence of the genus-$1$ obstruction equation:
$\kappa$ appears as the coefficient of $\omega_1$ in $d_{\barB}^2$,
and a morphism that changes the genus-$1$ bar data changes $\kappa$ by
the anomaly~$\delta\kappa$. For a strict morphism (one preserving the
filtration and the central charge), the genus-$1$ data is unchanged,
so $\delta\kappa = 0$. For a filtered quasi-isomorphism that shifts
the central charge by $\delta c$, one has
$\delta\kappa(f)$ is determined by the change in the genus-$1$ bar obstruction class (Volume~I, Theorem~D). For the Virasoro algebra, $\kappa = c/2$ and the anomaly is $\delta\kappa = \delta c/2$; for affine algebras, the relation between $\kappa$ and $c$ is more involved (see the formula $\kappa = \dim\fg \cdot (k+h^\vee)/(2h^\vee)$ above).
\end{proof}

Part~(ii) of the preceding proposition states the genus-$0$
intertwining: DS reduction commutes with the bar differential $D_0$.
The modular bar complex $\barB_{\mathrm{mod}}(\cA)$ carries a second operator
$D_1 = D_{\mathrm{nsep}}$, the genus-raising non-separating clutching
differential. The next theorem extends the intertwining to the full
modular complex $(D_0 + D_1)$.

\begin{theorem}[DS--modular bar intertwining; \ClaimStatusProvedHere]
\label{thm:ds-modular-intertwining}
\index{Drinfeld--Sokolov reduction!modular bar intertwining|textbf}
For $\fg$ simple and $k$ non-critical, the DS reduction functor
extends to the modular bar complex at all genera:
\[
\barB_{\mathrm{mod}}^{\mathcal{W}}(M^{\mathcal{W}})
\;\simeq\;
H^0_{\mathrm{DS}}\bigl(\barB_{\mathrm{mod}}(M)\bigr).
\]
\end{theorem}

\begin{proof}
The three inputs are:

\textup{(i)}~$[Q_{\mathrm{DS}}, D_0] = 0$
\textup{(}Vol~I, Theorem~\textup{\ref{V1-thm:ds-koszul-intertwine}):}
the BRST charge commutes with the genus-preserving bar differential.

\textup{(ii)}~$[Q_{\mathrm{DS}}, D_1] = 0$: The clutching operator
$D_1^{(i,j)}$ contracts flags $i,j$ using the invariant form:
$D_1^{(i,j)}(\phi) = \langle x_i, x_j \rangle_k \cdot
(\text{genus-raised element})$. For $Q_{\mathrm{DS}}$ acting on
$D_1(\phi)$ versus $D_1$ acting on $Q_{\mathrm{DS}}(\phi)$: the
terms from $Q_{\mathrm{DS}}$ acting on vertices other than $i,j$
commute trivially with the contraction. The terms from
$Q_{\mathrm{DS}}$ acting on $x_i$ or $x_j$ produce
$\langle Q_{\mathrm{DS}}(x_i), x_j \rangle +
(-1)^{|x_i|} \langle x_i, Q_{\mathrm{DS}}(x_j) \rangle$, which
vanishes by the self-adjointness of $Q_{\mathrm{DS}}$ with respect
to the level-$k$ Shapovalov--ghost pairing (the standard fact that
$Q_{\mathrm{BRST}}$ of a VOA is BPZ-self-adjoint).

\textup{(iii)}~Arakawa vanishing: the spectral sequence of the
triple complex $(\barB_{\mathrm{mod}} \otimes F_\chi,\, D_0,\, D_1,\,
Q_{\mathrm{DS}})$ degenerates at $E_1$ by $H^{q \ne 0}_{\mathrm{DS}} = 0$.
\end{proof}

\begin{theorem}[Formal moduli interpretation; \ClaimStatusProvedHere]
\label{thm:formal-moduli-twisting}
\index{formal moduli!twisting morphisms|textbf}
\index{Lurie--Pridham correspondence!twisting morphisms|textbf}
The simplicial Maurer--Cartan space
$\MC_\bullet(\mathrm{Conv}(\cC, \cA))$ is the derived moduli space
of twisting morphisms from $\cC$ to $\cA$. It satisfies the
Lurie--Pridham correspondence:
\begin{equation}\label{eq:formal-moduli-twisting}
\MC_\bullet\bigl(\mathrm{Conv}(\cC, \cA)\bigr)
\;\simeq\;
\mathrm{Map}_{\mathrm{CoAlg}}(\cC, \barB(\cA))
\;\simeq\;
\mathrm{Map}_{\mathrm{Alg}}(\Omegach(\cC), \cA)
\end{equation}
as simplicial sets \textup{(}equivalently, as objects of the
$\infty$-category of spaces\textup{)}. The bar construction is
the representing object for the formal moduli problem associated
to the deformation theory of~$\cA$.
\end{theorem}

\begin{proof}
The convolution dg Lie algebra
$\mathrm{Conv}(\cC, \cA)
= \Hom(s^{-1}\bar{\cC},\, \cA)$
is complete: its underlying graded vector space is a product, not
a sum, because $\cC$ is conilpotent. Concretely, the coradical
filtration $F_\bullet\cC$ dualizes to an inverse-limit filtration
$F^\bullet\mathrm{Conv}(\cC, \cA)
= \varprojlim_k \Hom(\cC/F_k\cC, \cA)$, and completeness is the
statement that $\mathrm{Conv}(\cC, \cA)
= \varprojlim_k \mathrm{Conv}(\cC, \cA)/F^k$.

For complete dg Lie algebras, the simplicial Maurer--Cartan functor
$\MC_\bullet$ is well-defined: its $n$-simplices are MC elements
in $\mathfrak{g} \otimes \Omega_n$, where $\Omega_n$ is the
Sullivan algebra of polynomial forms on the standard
$n$-simplex~$\Delta^n$
(Robert-Nicoud--Vallette~\cite{RNV20}, \S3; this is the
$\Linf$-generalization of the Hinich--Goldman--Millson
construction). The resulting simplicial set
$\MC_\bullet(\mathfrak{g})$
is a Kan complex whose homotopy type depends only on the
$\Linf$ quasi-isomorphism class of~$\mathfrak{g}$
(\cite{RNV20}, Theorem~4.1).

The identification with mapping spaces follows from the
representability of
Corollary~\ref{cor:bar-cobar-representability}, extended to the
simplicial enrichment. For each~$n$, the set of
$n$-simplices of
$\mathrm{Map}_{\mathrm{CoAlg}}(\cC, \barB(\cA))$ is
\[
\Hom_{\mathrm{CoAlg}}\!\bigl(\cC,\,
 \barB(\cA) \otimes \Omega_n\bigr)
\;\cong\;
\mathrm{Tw}(\cC, \cA \otimes \Omega_n)
\;\cong\;
\MC\!\bigl(\mathrm{Conv}(\cC, \cA) \otimes \Omega_n\bigr),
\]
which is the set of $n$-simplices of
$\MC_\bullet(\mathrm{Conv}(\cC, \cA))$.
This identification is natural in the simplicial direction
(functoriality of the cosimplicial structure of
$\{\Omega_n\}_{n \geq 0}$), giving the equivalence of simplicial
sets~\eqref{eq:formal-moduli-twisting}. The right-hand equivalence
in~\eqref{eq:formal-moduli-twisting} follows by the same argument
applied to the cobar side, using
Theorem~\ref{thm:cobar-initiality} in place of
Theorem~\ref{thm:bar-terminality}.

By the Lurie--Pridham theorem
(Lurie~\cite{Lur11}, Theorem~5.3;
Pridham~\cite{Pri10}, Theorem~4.26),
the functor $\MC_\bullet$ establishes an equivalence between
complete dg Lie algebras (up to $\Linf$ quasi-isomorphism) and
formal moduli problems (up to equivalence of the associated
deformation functors). The passage from coalgebras to formal moduli
problems uses Koszul duality: a conilpotent coalgebra~$\cC$
corresponds to the complete local cdga~$\cC^\vee$ (the linear dual
with completed tensor product), and
$\MC_\bullet(\mathrm{Conv}(\cC, \cA)) \simeq
\MC_\bullet(\mathfrak{g} \widehat{\otimes} \cC^\vee)$
where $\mathfrak{g}$ is the dg Lie algebra governing the
deformation theory of~$\cA$. The functor
$R \mapsto \MC_\bullet(\mathfrak{g} \widehat{\otimes} \mathfrak{m}_R)$
for Artinian~$R$ is the formal moduli problem in Lurie's sense,
and Pridham's theorem \cite[Theorem~4.26]{Pri10} identifies it
with~$\mathfrak{g}$ up to quasi-isomorphism.
The bar construction $\barB(\cA)$ is
therefore the coalgebra that \emph{corepresents} the formal moduli
problem $\mathrm{Def}_\cA$ of deformations of~$\cA$: the derived
deformation space of~$\cA$ is
$\MC_\bullet(\mathrm{Conv}(\cC, \cA))$ for any cofibrant
replacement $\cC \xrightarrow{\sim} \barB(\cA)$, and this is
independent of the choice by the homotopy invariance of
$\MC_\bullet$.
\end{proof}

\begin{remark}[The modular upgrade]
\label{rem:modular-upgrade-universal}
\index{modular twisting morphism!universal property}
At the modular level, the same universal property holds with the
modular convolution algebra
$\mathrm{Conv}^{\mathrm{mod}}\bigl(
 \cC^{\mathrm{logFM}}_{\mathrm{mod}},\,
 \mathrm{End}_{Ch_\infty}(\cA^{\mathrm{ch}}_\infty)
\bigr)$
replacing the genus-$0$ convolution algebra. The universal modular
twisting morphism~$\Theta_\cA$
(Construction~\ref{constr:vol2-genus-g-mc})
is the base-point of the derived moduli space
$\MC_\bullet\bigl(\mathrm{Def}^{\mathrm{mod}}_{\infty,\log}(\cA)
\bigr)$.

Every structure in this monograph is a shadow of this single
universal object under a projection: the bar differential $d_{\barB}$
is the cogenerator projection of~$\Theta_\cA$; the
$R$-matrix is the closed-color monodromy of the open-color tensor
product parametrized by~$\Theta_\cA$; the genus-$g$ curvature

representability of the bifunctor) is the reason the single
object $\Theta_\cA$ suffices: it is the terminal twisting morphism,
the one through which all others factor.
\end{remark}

\subsection{Example: Heisenberg Vertex Algebra and Its Koszul Dual}
\begin{example}[Heisenberg and Its Koszul Dual]
\label{ex:Heisenberg_Koszul}
Consider the Heisenberg vertex algebra $\mathcal{H}_k$ at level $k$ (Example \ref{ex:Heisenberg_quadratic}), generated by $J$ with relation $\{J_\lambda J\} = k\lambda$.

\textbf{Step 1: Identify as quadratic.}

This is a quadratic chiral algebra:
\begin{equation}
\mathcal{H}_k = T_{\text{ch}}(\C \cdot J) / \langle \{J_\lambda J\} - k\lambda \rangle.
\end{equation}

\textbf{Step 2: Compute the Koszul dual.}

The Koszul dual (GLZ, Definition \ref{def:GLZ_Koszul}) is:
\begin{equation}
\mathcal{H}_k^! = T_{\text{ch}}(\C \cdot J^* \otimes \omega_\C[1]) / \langle R^\perp \rangle.
\end{equation}

The annihilator relation $R^\perp$ is computed from the pairing:
\begin{equation}
\langle \{J_\lambda J\}, J^* \otimes J^* \rangle = k\lambda.
\end{equation}

Therefore:
\begin{equation}
\mathcal{H}_k^! = CE(\text{Heisenberg Lie algebra}),
\end{equation}
the Chevalley--Eilenberg complex of the Heisenberg Lie algebra (not the vertex algebra!).

\textbf{Step 3: Distinguish from quadratic projection.}

\textbf{Warning}: The Koszul dual $\mathcal{H}_k^!$ is \textbf{not} the same as the \emph{quadratic projection} (taking the associated graded with respect to the PBW filtration). The quadratic projection would yield:
\begin{equation}
\text{gr}(\mathcal{H}_k) = \text{Sym}(\C \cdot J),
\end{equation}
a symmetric algebra (commutative).

In contrast, the Koszul dual $\mathcal{H}_k^!$ is the CE complex, which is a \textbf{commutative} dga (an exterior algebra with differential encoding the Lie bracket).

Koszul duality exchanges algebraic structures (commutative $\leftrightarrow$ Lie); the quadratic projection instead forgets higher operations.
\end{example}

\begin{computation}[Bar differential on a degree-2 Heisenberg element;
\ClaimStatusProvedHere]
\label{comp:heisenberg-bar-degree2}
We illustrate the three components of $d_{\barB} = d_Q + d_{\mathrm{res}} + d_{A_\infty}$
acting on a degree-2 bar element (three tensor factors) for the
Heisenberg chiral algebra $\mathcal{H}_k$ at level~$k$, generated
by $J$ with $\{J_\lambda J\} = k\lambda$. Write the element as
\[
\xi \;=\; [J \,|\, J \,|\, J]
 \;\in\; \barB_{\mathrm{ch}}(\mathcal{H}_k),
\]
the bar monomial of bar degree~$2$ (three tensor factors in
$s^{-1}\bar{\mathcal{H}}_k$) and internal degree~$0$.

\medskip\noindent
\textbf{(i) $d_Q[J\,|\,J\,|\,J]$.}
The Heisenberg algebra has $Q = m_1 = 0$ on generators (the BV-BRST
differential vanishes on the physical field~$J$), so $d_Q$ acts by
applying $m_1 = 0$ to each tensor factor:
\[
d_Q[J\,|\,J\,|\,J]
= [QJ\,|\,J\,|\,J]
 + (-1)^{|J|}[J\,|\,QJ\,|\,J]
 + (-1)^{|J|+|J|}[J\,|\,J\,|\,QJ]
= 0.
\]

\medskip\noindent
\textbf{(ii) $d_{\mathrm{res}}[J\,|\,J\,|\,J]$.}
The residue differential extracts OPE poles at collision divisors.
For three factors with labels $\{1,2,3\}$, the relevant subsets
$S$ with $|S| \geq 2$ and \emph{consecutive} indices are
$S = \{1,2\}$ and $S = \{2,3\}$:
\begin{align*}
d_{\mathrm{res}}[J\,|\,J\,|\,J]
&= \Res_{D_{\{1,2\}}}\bigl([J\,|\,J\,|\,J]\bigr)
 + \Res_{D_{\{2,3\}}}\bigl([J\,|\,J\,|\,J]\bigr) \\
&= \bigl[\{J_\lambda J\}\big|_{\mathrm{res}}\,\big|\,J\bigr]
 + (-1)^{|J|}\,\bigl[J\,\big|\,\{J_\lambda J\}\big|_{\mathrm{res}}\bigr] \\
&= [k\,|\,J] + [J\,|\,k]
\;=\; k\,\bigl([1\,|\,J] + [J\,|\,1]\bigr),
\end{align*}
where $\{J_\lambda J\}\big|_{\mathrm{res}} = k$ denotes the OPE residue:
the bar differential encodes the full singular OPE kernel of
$J(z)J(w) \sim k/(z-w)^2$; the spectral parameter version is
$m_2(J,J;\lambda) = k\lambda$, where $k\lambda$ is the Borel transform
of $k\zeta^{-2}$. Here $1$ denotes the vacuum.
In the reduced bar complex $\barB = T^c(s^{-1}\bar{\mathcal{H}})$
(augmentation ideal only), the terms involving the vacuum reduce to
lower-bar-degree elements.

\medskip\noindent
\textbf{(iii) $d_{A_\infty}[J\,|\,J\,|\,J]$.}
Since the Heisenberg algebra is quadratic ($m_{k \geq 3} = 0$), the
$A_\infty$ grafting differential vanishes identically:
\[
d_{A_\infty}[J\,|\,J\,|\,J] = 0.
\]

\medskip\noindent
\textbf{Total.}
Assembling:
\[
d_{\barB}[J\,|\,J\,|\,J]
= k\,\bigl([1\,|\,J] + [J\,|\,1]\bigr).
\]
The result has bar degree~$1$ and records the OPE of adjacent
Heisenberg currents. Note that $d_{\barB}^2 = 0$ is satisfied:
applying $d_{\barB}$ again to either $[1\,|\,J]$ or $[J\,|\,1]$
gives zero (the OPE of $1$ with anything is regular).
\end{computation}

\begin{remark}[Curvature and Non-Flat Connections]
For curved $A_\infty$ algebras (where $m_0 \neq 0$, i.e., a curvature term), the bar--cobar formalism extends by including a \emph{curved differential}:
\begin{equation}
d_{\overline{B}} = m_0 + d_Q + d_{\text{res}} + d_{A_\infty},
\end{equation}
where $m_0 \in \mathcal{A}$ is a degree-2 element (curvature). The Maurer--Cartan equation becomes $\partial\tau + \tau\star\tau = 0$ (the compact form absorbs the standard desuspension signs $(-1)^{k(k-1)/2}$ on each $m_k$ term into the convolution product~$\star$; see Loday--Vallette~\cite{LV12}, Proposition~1.3.3).

Curvature obstructs $m_1$ from being a differential:
$m_1^2(a) = m_2(m_0, a) - m_2(a, m_0) = [m_0, a]_{m_2}$,
so $m_1^2 \neq 0$ whenever $m_0 \neq 0$ (the commutator is ordinary, not graded, because $|m_0| = 2$ is even; the sign $(-1)^{|m_0||a|} = 1$).
This controls \emph{curved deformations} of chiral algebras, relevant for non-flat connections in quantum field theory.
This convention (desuspended, with $m_1^2 = [m_0, -]_{m_2}$) matches Volume~I's signs appendix (equation~(A.2)) and the axioms of Section~\ref{sec:axioms-Ainfty-chiral}.
\end{remark}

\subsection{The $E_1$ coalgebra structure on the bar complex}
\label{subsec:bar-swiss-cheese}

The chiral bar complex $\barB_{\mathrm{ch}}(\cA)$ carries two structures: the bar differential $d_{\barB}$ (from the holomorphic direction) and the deconcatenation coproduct $\Delta$ (from the topological direction). Together they make $\barB(\cA)$ an $E_1$ dg coassociative coalgebra: the differential is a coderivation of the coproduct. The $\SCchtop$ structure does not live on $\barB(\cA)$ itself; it emerges in the chiral derived center pair $(C^\bullet_{\mathrm{ch}}(\cA,\cA),\, \cA)$, where the bulk $C^\bullet_{\mathrm{ch}}(\cA,\cA)$ acts on the boundary~$\cA$ (Section~\ref{sec:derived-center}).

\begin{theorem}[$E_1$ dg coalgebra structure on the bar complex;
\ClaimStatusProvedHere]
\label{thm:bar-swiss-cheese}
Let $\cA$ be a chiral algebra on a smooth algebraic curve~$X$.
The geometric bar complex $\barB_X(\cA)$ of Volume~I, equipped
with:
\begin{enumerate}[label=\textup{(\roman*)}]
\item the bar differential $d_{\barB}$ (residues along collision
 divisors in $\FM_k(X)$), encoding the $A_\infty$ chiral
 structure;
\item the deconcatenation coproduct
 $\Delta[a_1|\cdots|a_n]
 = \sum_{i=0}^{n}
 [a_1|\cdots|a_i]
 \otimes [a_{i+1}|\cdots|a_n]$,
 encoding the $E_1$ (topological) coalgebra structure;
\end{enumerate}
is an $E_1$ dg coassociative coalgebra: $d_{\barB}$ is a
coderivation of $\Delta$, encoding the compatibility between
the holomorphic collision data on $\FM_k(X)$ and the topological
ordering data on $\Conf_k^{<}(\R)$.
\end{theorem}

\begin{proof}
The proof assembles three ingredients, all established earlier in
this volume and in Volume~I.

\medskip\noindent\textbf{Step 1: The bar differential.}
By Volume~I, Theorem~A, the bar differential $d_{\barB}$ satisfies
$d_{\barB}^2 = 0$ (via the Arnold cancellation on $\FM_k(X)$) and
endows $\barB_X(\cA)$ with the structure of a dg coalgebra
governed by $C_\ast(\FM_\bullet(X))$. The differential encodes
the $A_\infty$ chiral structure: the components
$d_k \colon T^c_n \to T^c_{n-k+1}$ extract the higher products
$m_k$ from collision residues on $\FM_k(X)$.

\medskip\noindent\textbf{Step 2: The $E_1$-coalgebra structure.}
The deconcatenation coproduct $\Delta$ is coassociative by
construction (it is the tensor coalgebra coproduct on
$T^c(s^{-1}\bar{\cA})$). Since $\Delta$ splits bar elements into
ordered sub-sequences, it encodes the $\Eone$-coalgebra structure
on the topological direction~$\R$.

\medskip\noindent\textbf{Step 3: Coderivation compatibility.}
The bar differential $d_{\barB}$ is a coderivation of $\Delta$:
the Leibniz rule for residues along collision divisors ensures that
$\Delta \circ d_{\barB}
= (d_{\barB} \otimes \id + \id \otimes d_{\barB}) \circ \Delta$.
This is the defining compatibility of a dg coalgebra: the
differential respects the coproduct.
The coderivation property holds because each component $d_k$
acts on a single tensor factor of the cofree coalgebra
$T^c(s^{-1}\bar{\cA})$, and single-factor operations on a cofree
coalgebra are automatically coderivations by the universal property
of cofree coalgebras (Loday--Vallette~\cite{LV12},

\end{proof}

\begin{proposition}[The bar complex as derived tensor product and
factorisation homology of the interval;
\ClaimStatusProvedHere]
\label{prop:bar-interval-identification}
\index{bar complex!as derived tensor product|textbf}
\index{bar complex!as factorisation homology of the interval|textbf}
\index{factorisation homology!interval vs line|textbf}
Let $A$ be an augmented $\Eone$-algebra over a field~$k$, with
augmentation $\varepsilon \colon A \to k$.  Then:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \textbf{Derived tensor product.}
 The bar construction computes the derived tensor product of
 augmentation modules:
 \[
   B(A) \;\simeq\; k \otimes_A^{\mathbf{L}} k.
 \]
 Here $k$ is a left \textup{(}resp.\ right\textup{)} $A$-module via
 $\varepsilon$, and $\otimes_A^{\mathbf{L}}$ is the derived tensor
 product in $A$-bimodules.

\item \textbf{Factorisation homology of the interval.}
 The bar construction computes the factorisation homology of the
 compact interval $[0,1]$ with augmentation boundary conditions at
 both endpoints:
 \[
   B(A) \;\simeq\; \int_{[0,1]}^{k,\,k} A.
 \]
 The superscript ${}^{k,k}$ denotes the augmentation boundary
 conditions: the left $A$-module $k$ at $\{0\}$ and the right
 $A$-module $k$ at $\{1\}$.

\item \textbf{Factorisation homology of the line is trivial.}
 In contrast, factorisation homology of $\bR$ \textup{(}with no
 boundary\textup{)} recovers $A$ itself:
 \[
   \int_{\bR} A \;\simeq\; A.
 \]
 The bar construction is \textbf{not} factorisation homology of
 the line.

\item \textbf{Coproduct from interval cutting.}
 The deconcatenation coproduct $\Delta$ on $B(A)$ has the
 following geometric interpretation.
 For each $t \in (0,1)$, the decomposition
 $[0,1] = [0,t] \cup_{\{t\}} [t,1]$ induces, via the
 $\otimes$-excision property of factorisation homology, a
 decomposition
 \[
   \int_{[0,1]}^{k,k} A
   \;\simeq\;
   \int_{[0,t]}^{k,A} A
   \;\otimes_A\;
   \int_{[t,1]}^{A,k} A.
 \]
 Here $\int_{[0,t]}^{k,A} A$ is the \emph{one-sided} bar
 construction $B(k, A, A)$ (a right $A$-module quasi-isomorphic
 to~$k$), and $\int_{[t,1]}^{A,k} A = B(A, A, k)$ is the
 corresponding left version.
 The deconcatenation coproduct $\Delta \colon B(A) \to B(A)
 \otimes B(A)$ (a map of \emph{plain} vector spaces, not a
 tensor product over~$A$) is the \emph{cofree} coalgebra
 structure on $T^c(s^{-1}\bar{A})$: it sends
 $[a_1|\cdots|a_n] \mapsto \sum_{i=0}^{n}
 [a_1|\cdots|a_i] \otimes [a_{i+1}|\cdots|a_n]$.
 This is a different operation from excision: excision
 decomposes $B(A)$ as a derived tensor product \emph{over}~$A$
 (recovering one copy), while $\Delta$ maps into a plain tensor
 product (producing two copies). The coassociativity of
 $\Delta$ follows from the cofreeness of~$T^c$.
\end{enumerate}
\end{proposition}

\begin{proof}
\textbf{Part (i).}
The two-sided bar complex $B(A,A,A) = A \otimes T^c(s^{-1}\bar{A})
\otimes A$ is the standard resolution of~$A$ as an $A$-bimodule
(see e.g.\ Loday~\cite{loday1992cyclic}, \S1.1).
Applying $k \otimes_A (-) \otimes_A k$ via the augmentation
$\varepsilon$ yields the reduced bar complex
$B(A) = T^c(s^{-1}\bar{A})$, which by definition computes
$k \otimes_A^{\mathbf{L}} k =
\Tor_\bullet^A(k,k)$.

\textbf{Part (ii).}
By Ayala--Francis~\cite{AF15} (Theorem~3.24; see also
Lurie~\cite{HA}, \S5.5.4), factorisation homology of an
$\Eone$-algebra $A$ on a compact $1$-manifold with boundary
is computed by the two-sided bar construction with the
prescribed boundary modules.
For the interval $[0,1]$ with the augmentation module
$k = A/\bar{A}$ at each boundary component, this gives
\[
  \int_{[0,1]}^{k,\,k} A
  \;\simeq\;
  k \otimes_A^{\mathbf{L}} k
  \;\simeq\;
  B(A),
\]
where the first equivalence is the Ayala--Francis
$\otimes$-excision theorem applied to the collar-gluing
$[0,1] = [0,\tfrac{1}{2}] \cup_{A} [\tfrac{1}{2},1]$,
and the second is Part~(i).

\textbf{Part (iii).}
The line $\bR$ is a $1$-manifold without boundary.
By the $\Eone$ recognition principle
(Lurie~\cite{HA}, \S5.5.4.10), factorisation homology
on $\bR$ is the identity:
$\int_{\bR} A \simeq A$.
This holds because $\bR$ is contractible and has no
boundary where augmentation could impose conditions.
The geometric point: $\bR$ is the \emph{universal cover}
of all compact $1$-manifolds, and factorisation homology
on the universal cover is the algebra itself (no
identifications, no coinvariants, no derived tensor
products).

\textbf{Part (iv).}
The $\otimes$-excision property of
Ayala--Francis~\cite{AF15} (Theorem~3.24) states that
for a collar-gluing $M = M_L \cup_{M_0} M_R$ of manifolds
with boundary, factorisation homology satisfies
$\int_M A \simeq \int_{M_L} A \otimes_{\int_{M_0} A}
\int_{M_R} A$.
Applied to $[0,1] = [0,t] \cup_{\{t\}} [t,1]$, this gives
$B(A) \simeq B(k,A,A) \otimes_A B(A,A,k)$, a derived tensor
product \emph{over}~$A$ of two one-sided bar constructions.
This recovers a \emph{single} copy of~$B(A)$; it does not
produce $B(A) \otimes B(A)$.
The deconcatenation coproduct $\Delta \colon B(A) \to
B(A) \otimes B(A)$ is a separate structure: the cofree
coalgebra map
$[a_1|\cdots|a_n] \mapsto \sum_{i=0}^{n}
[a_1|\cdots|a_i] \otimes [a_{i+1}|\cdots|a_n]$
on $T^c(s^{-1}\bar{A})$, whose coassociativity is
the cofreeness of~$T^c$ (not the associativity of
collar-gluing).
\end{proof}

\begin{remark}[The interval vs.\ the line: boundary conditions
are essential]
\label{rem:interval-vs-line}
The identification $B(A) \simeq \int_{[0,1]}^{k,k} A$ makes
precise why the bar complex is a nontrivial invariant: it is the
compact interval, not the line, that carries the augmentation
boundary conditions forcing the derived tensor product.  Without
boundary conditions, factorisation homology on any contractible
manifold returns~$A$ itself.  This distinction is the source of
common confusion.
The three key $1$-manifolds and their factorisation homology are:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lll}
\hline
\textbf{Manifold} & \textbf{Fact.\ homology} & \textbf{Structure} \\
\hline
$\bR$ (line, no boundary) & $\int_{\bR} A \simeq A$ & trivial \\
$[0,1]$ (interval, $\partial = \{0,1\}$)
 & $\int_{[0,1]}^{k,k} A \simeq B(A)$
 & $\Eone$ coalgebra \\
$S^1$ (circle, no boundary)
 & $\int_{S^1} A \simeq \mathrm{HH}_\ast(A)$
 & cyclic homology \\
\hline
\end{tabular}
\end{center}
The half-line $[0,\infty)$ with a single boundary module $M$
at $\{0\}$ gives the one-sided bar resolution
$B(A,M) \simeq \int_{[0,\infty)}^{M} A$, computing
$\Tor^A_\bullet(k,M)$.
\end{remark}

\begin{remark}[Three bar complexes]
\label{rem:three-bar-complexes}%
\index{bar complex!three variants}%
The $E_1$ dg coalgebra structure of
Theorem~\ref{thm:bar-swiss-cheese} produces the ordered bar complex
$\barB^{\mathrm{ord}}(\cA)$ as the primitive object; three
natural variants play distinct roles
throughout the volume.
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{The Francis--Gaitsgory bar complex}
 $\barB^{\mathrm{FG}}(\cA)$. This retains only the zeroth
 product $a_{(0)}b$ (the chiral Lie bracket), discarding all
 higher OPE poles. It is the bar complex of $\cA$ viewed as a
 chiral Lie algebra; its Koszul dual is the Francis--Gaitsgory
 chiral Lie Koszul dual $\cA^!_{\mathrm{FG}}$.

\item \emph{The symmetric bar complex}
 $\barBch(\cA) = \barB^{\Sigma}(\cA)$. This uses all OPE
 products $\{a_{(n)}b\}_{n \ge 0}$ and takes
 $\Sigma_n$-coinvariants: it is the factorisation coalgebra on
 $\operatorname{Ran}(X)$ constructed in Volume~I, Theorem~A\@.
 Verdier duality $\mathbb{D}_{\operatorname{Ran}}$ produces the
 chiral Koszul dual $\cA^!_{\mathrm{ch}}$.

\item \emph{The ordered bar complex}
 $\barB^{\mathrm{ord}}(\cA)$. This uses all OPE products but
 retains the linear ordering on insertions: no
 $\Sigma_n$-quotient. It is the cofree coalgebra
 $T^c(s^{-1}\bar{\cA})$ with deconcatenation coproduct from
 $\Conf_k^{<}(\R)$; this is the $E_1$ dg coassociative
 coalgebra of Theorem~\ref{thm:bar-swiss-cheese}. Linear duality
 $H^\bullet(-)^\vee$ produces the open-colour Koszul dual
 $\cA^!_{\mathrm{line}}$.
\end{enumerate}
The three complexes are related by two natural maps:
\begin{itemize}[nosep]
\item \emph{Ordered-to-symmetric descent.} Taking
 $\Sigma_n$-coinvariants gives a map
 $\barB^{\mathrm{ord}}(\cA) \to \barBch(\cA)$, but for
 $E_\infty$-chiral algebras with OPE poles (all interesting
 vertex algebras), this descent is $R$-matrix twisted:
 $\barBch(\cA)_n \simeq
 (\barB^{\mathrm{ord}}(\cA)_n)^{R\text{-}\Sigma_n}$
 (Proposition~\ref{prop:r-matrix-descent}). For the pole-free
 subclass, $R(z) = \tau$ and the descent is the naive
 $\Sigma_n$-quotient.

\item \emph{Symmetric-to-FG filtration.} The holomorphic weight
 filtration on $\barBch(\cA)$ has associated graded
 $\barB^{\mathrm{FG}}(\cA)$: retaining only the leading
 (zeroth) pole at each collision recovers the chiral Lie bar
 complex. In particular, $\barB^{\mathrm{FG}}(\cA)$ does not
 see higher poles and therefore misses the curvature
 $\kappa(\cA)$.
\end{itemize}
For the affine Kac--Moody algebra $\widehat{\fg}_k$:
$\barBch$ yields $\cA^!_{\mathrm{ch}} =
\widehat{\fg}_{-k-2h^\vee}$ (level duality),
$\barB^{\mathrm{ord}}$ yields $\cA^!_{\mathrm{line}} =
Y_\hbar(\fg)$ (the Yangian), and $\barB^{\mathrm{FG}}$ yields
the chiral Lie Koszul dual (the $E_\infty$ shadow that forgets
the spectral parameter).

The two-colour Koszul duality architecture is developed in
Part~\ref{part:bbl-core}
(Remark~\ref{rem:two-colour-architecture}); the master theorem
assembling both colours is
Theorem~\ref{thm:two-color-master}.
\end{remark}

\subsection{The essence of Koszulness}
\label{subsec:koszulness-essence}

The Steinberg variety $Z = \widetilde{\mathcal{N}}
\times_{\mathfrak{g}} \widetilde{\mathcal{N}}$ presents the
Hecke algebra: $H_*(Z) \cong H_W$. Every representation-theoretic
fact about~$H_W$ (cells, Kazhdan--Lusztig polynomials, canonical
bases) is a shadow of a geometric fact about~$Z$. The Hecke
algebra does not know it is a convolution algebra; the Steinberg
variety knows nothing else.

The bar complex $\barB(\cA)$ is the Steinberg object of chiral
algebra: the formal bar/Koszul model of the derived Lagrangian
self-intersection $\Steinb_b = \mathscr{L} \times_{\mathscr{M}}^h
\mathscr{L}$, parametrised by operation spaces
$\FM_n(\C) \times \Conf_n^{<}(\R)$. The bar differential is
convolution; the coproduct is restriction; the bar-cobar counit
recovers the algebra. The single object~$\barB(\cA)$ packages the
algebra, the open-colour line-sector operations, the $R$-matrix, the
genus tower, and the PVA shadow as direct data, while the
chirally-Koszul comparison theorems recover the line-category
package.
By operadic Koszul duality
(\S\ref{subsec:dg-yangian-operadic}), the Koszul dual
of an $\SCchtop$-algebra is an $(\SCchtop)^!$-algebra
(closed $\mathsf{Lie}$, open $\mathsf{Ass}$):
on the chirally Koszul locus, $\cA^!_{\mathrm{line}}$ carries the dg-shifted Yangian package
(Theorem~\ref{thm:yangian-recognition}), while the full
two-colour duality involution returns~$\cA$
(Theorem~\ref{thm:duality-involution}).

\begin{proposition}[Koszul duality of the Swiss-cheese operad;
\ClaimStatusProvedElsewhere]
\label{thm:SC-self-duality}
\index{Swiss-cheese operad!Koszul dual|textbf}
\index{Koszul duality!Swiss-cheese operad|textbf}
The Koszul dual cooperad of the chiral--topological Swiss-cheese
operad has closed sector $\mathsf{Lie}^c$
\textup{(}$\dim = (n-1)!$\textup{)},
open sector $\mathsf{Ass}^c$ \textup{(}$\dim = m!$\textup{)},
and mixed sector of dimension $(k-1)!\binom{k+m}{m}$:
\[
(\SCchtop)^! \;=\; (\mathsf{Lie},\, \mathsf{Ass},\,
\text{shuffle-mixed}).
\]
In particular, $\SCchtop$ is \emph{not} self-dual as an operad:
the closed colour $\mathsf{Com}$ \textup{(}$\dim = 1$\textup{)}
is exchanged with $\mathsf{Lie}$ \textup{(}$\dim = (n-1)!$\textup{)}.
The open colour $\mathsf{Ass}$ is self-dual.
The bar-cobar duality \emph{functor} on $\SCchtop$-algebras
is an involution: $(\cA^!)^! \simeq \cA$ on the chirally Koszul
locus \textup{(}Theorem~\textup{\ref{thm:duality-involution})}.
\end{proposition}

\begin{proof}
The classical Swiss-cheese operad is Koszul
by Hoefel and Hoefel--Livernet~\cite{Hoe09,HL12}.
The individual Koszul duals are
$\Etwo^{!} \simeq \Etwo\{1\}$ (Tamarkin self-duality up to
operadic shift~\cite{Tam03}, which degenerates to
$\mathsf{Com}^! = \mathsf{Lie}$ only on the associated graded
for the Poisson filtration) and
$\mathsf{Ass}^! = \mathsf{Ass}$;
the full two-coloured Koszul dual cooperad
$\SCchtop^!$ has the structure stated above.
Homotopy-Koszulity of~$\SCchtop$
(Theorem~\ref{thm:homotopy-Koszul}) promotes this to an
equivalence at the chain level, and the bar-cobar
duality functor is an involution on $\SCchtop$-algebras
(not on the operad itself).
\end{proof}

\begin{remark}[Self-duality is of the functor, not the operad]
\label{rem:SC-duality-functor-vs-operad}
The claim $(\SCchtop)^! \simeq \SCchtop$ that appeared in
earlier versions of this manuscript is \emph{false}: the Koszul
dual exchanges $\mathsf{Com}$ (closed, $\dim = 1$) with
$\mathsf{Lie}$ (closed, $\dim = (n-1)!$).
What IS true: the bar-cobar duality \emph{functor} on
$\SCchtop$-algebras is involutive, $(\cA^!)^! \simeq \cA$
(Theorem~\ref{thm:duality-involution}).
The involutivity is a property of Koszul duality as a functor,
not of the operad being isomorphic to its own Koszul dual.
\end{remark}

The Koszul duality functor on $\SCchtop$-algebras acts on both colours
simultaneously: the closed colour maps
$\cA \mapsto \cA^!_{\mathrm{ch}}$ via Verdier duality
(exchanging $\mathsf{Com}$-structure with $\mathsf{Lie}$-structure),
while the open colour maps
$\cA \mapsto \cA^!_{\mathrm{line}}$ via linear duality on
$\barB^{\mathrm{ord}}(\cA)$
(Remark~\ref{rem:three-bar-complexes}). The full two-colour
master theorem assembling both projections is
Theorem~\ref{thm:two-color-master} in Part~\ref{part:bbl-core}\@.

\begin{theorem}[Duality involution; \ClaimStatusProvedElsewhere]
\label{thm:duality-involution}
\index{duality involution|textbf}
\index{Koszul dual!involutivity|textbf}
Let\/ $\cA$ be a chirally Koszul
\textup{(}Definition~\textup{\ref{def:chiral-koszul-geometric})}
logarithmic $\SCchtop$-algebra. Then the double Koszul dual
is equivalent to the original algebra:
\[
(\cA^!)^! \;\simeq\; \cA.
\]
\end{theorem}

\begin{proof}
On the Koszul locus, the homotopy Koszul dual
$\cA^!_\infty$ is formal
(Theorem~\ref{thm:koszul-dodecahedron}(iii)), so
$\cA^!_\infty \simeq \cA^!$. By
Proposition~\ref{thm:SC-self-duality}, the Koszul dual of an
$\SCchtop$-algebra is an $(\SCchtop)^!$-algebra
(with closed sector $\mathsf{Lie}$, open sector $\mathsf{Ass}$).
The bar-cobar counit
$\Omega(\barB(\cA)) \xrightarrow{\sim} \cA$ is a
quasi-isomorphism on the Koszul locus
(Theorem~\ref{thm:koszul-dodecahedron}(iv)), and
$\barB(\cA^!) \simeq \cA^{!\,\vee}$ by Koszulness.
Composing the two bar-cobar rounds gives
$(\cA^!)^! \simeq \cA$.
\end{proof}

\begin{theorem}[Master projection theorem; \ClaimStatusProvedHere]
\label{thm:master-projection}
\index{master projection theorem|textbf}
\index{Swiss-cheese!master projection|textbf}
Let\/ $\cA$ be a logarithmic $\SCchtop$-algebra with open-sector
category\/~$\cC$
\textup{(}Definition~\textup{\ref{def:oc-factorization-category})}.
The MC element\/ $\alpha_T$ in the Swiss-cheese convolution algebra\/
$\mathfrak{g}^{\mathrm{SC}}_T$
\textup{(}Definition~\textup{\ref{def:sc-convolution})} determines:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Closed face:} the $A_\infty$ chiral algebra
 $A_b = \mathrm{End}_\cC(b)$ for any vacuum object\/~$b$
 \textup{(}Theorem~\textup{\ref{thm:boundary-algebra-from-vacuum})};
\item \emph{Open face:} the open-colour line-sector operations on
 $\cC_{\mathrm{line}}$; on the chirally Koszul locus,
 Theorem~\textup{\ref{thm:lines_as_modules}} identifies this
 sector with
 $\cA^!_{\mathrm{line}}\textup{-}\mathbf{mod}$;
\item \emph{Spectral face:} the $R$-matrix $R(z)$ satisfying the
 quantum Yang--Baxter equation
 \textup{(}Theorem~\textup{\ref{thm:YBE})};
\item \emph{Holographic face:} the bulk
 $= Z_{\mathrm{der}}(\cC) \simeq
 C^\bullet_{\mathrm{ch}}(A_b, A_b)$
 \textup{(}Corollary~\textup{\ref{cor:bulk-derived-center-categorical})};
\item \emph{Modular face:} the genus tower with line bundle
 $\cL_\kappa = (\det\lambda)^{\otimes\kappa}$ and curvature
 $\kappa(\cA) \cdot \omega_g$
 \textup{(}Construction~\textup{\ref{constr:determinant-line-bundle})};
\item \emph{Affine face}
 \textup{(}for the affine lineage\textup{):}
 $\cC_{\mathrm{line}}^{\mathrm{red}}|_{\mathrm{eval}} \simeq
 \mathrm{Rep}_q(\fg)$, computing the Jones polynomial
 \textup{(}Theorem~\textup{\ref{thm:affine-monodromy-identification})}.
\end{enumerate}
The assignment $\cA \mapsto (\mathfrak{g}^{\mathrm{SC}}_T, \alpha_T)$ is
functorial
\textup{(}Proposition~\textup{\ref{prop:bar-cobar-naturality})}, and
the six projections are natural transformations.
\end{theorem}

\begin{proof}
This is the conjunction of the six cited theorems, unified by the
Swiss-cheese convolution algebra framework. The MC element
$\alpha_T$ encodes the full $\SCchtop$-algebra structure by the
representability theorem
(Theorem~\ref{thm:bar-representability}); each projection extracts
one face of this structure by restricting to the appropriate colour
profile.

\smallskip
\noindent\textit{Functoriality.}
By Proposition~\ref{prop:bar-cobar-naturality}(i), a morphism
$f \colon \cA \to \cA'$ of $\SCchtop$-algebras induces a dg
coalgebra map $\barB(f) \colon \barB(\cA) \to \barB(\cA')$,
which transports the MC element via the functorial MC
correspondence: $\barB(f)_\ast(\alpha_T) = \alpha_{T'}$, where
$\alpha_{T'}$ is the MC element of~$\cA'$.

\smallskip
\noindent\textit{Naturality of the projections.}
Each projection $\pi_i$ is defined by restricting the MC element
$\alpha_T$ to a specific colour profile of the $\SCchtop$ operad:
$\pi_i(\alpha_T)$ is the evaluation of $\alpha_T$ at a fixed
colour profile (closed-only for face~(i), open-only for
face~(ii), mixed with spectral parameter for face~(iii), etc.).
Given a morphism $f \colon \cA \to \cA'$, we have
\[
 \pi_i(\alpha_{T'})
 \;=\; \pi_i\!\bigl(\barB(f)_\ast(\alpha_T)\bigr)
 \;=\; \barB(f)_\ast\!\bigl(\pi_i(\alpha_T)\bigr),
\]
where the second equality holds because $\pi_i$ is an evaluation
map at a fixed colour profile: it commutes with the pushforward
$\barB(f)_\ast$, which acts on the coefficient algebra but does
not alter the colour profile at which the evaluation is performed.
This is the naturality square
\[
 \begin{tikzcd}
 (\mathfrak{g}^{\mathrm{SC}}_T, \alpha_T) \ar[r, "\barB(f)_\ast"] \ar[d, "\pi_i"'] &
 (\mathfrak{g}^{\mathrm{SC}}_{T'}, \alpha_{T'}) \ar[d, "\pi_i"] \\
 \mathrm{Face}_i(\cA) \ar[r, "f_i"'] &
 \mathrm{Face}_i(\cA')
 \end{tikzcd}
\]
for each face $i \in \{\mathrm{i},\ldots,\mathrm{vi}\}$, where
$f_i$ is the induced map on the $i$-th face. The commutativity of
this square is the statement that the six projections are natural
transformations from the convolution-algebra functor to the
respective face functors.
\end{proof}

\begin{theorem}[Formal bar model for the Lagrangian self-intersection;
\ClaimStatusProvedHere]
\label{thm:bar-is-self-intersection}
\index{bar complex!as Lagrangian self-intersection|textbf}
\index{Lagrangian!self-intersection!bar complex|textbf}
\index{Steinberg!as bar complex|textbf}
Let $\cM$ be a $(-2)$-shifted symplectic derived stack and
$\cL \hookrightarrow \cM$ a Lagrangian. Write
$B = \cO(\cL)$ for the algebra of functions on~$\cL$.
Then:
\begin{enumerate}[label=\textup{(\roman*)},leftmargin=2em]
\item The derived self-intersection
 $\Steinb := \cL \times_{\cM}^h \cL$
 carries a $(-1)$-shifted symplectic structure
 \textup{(}PTVV \cite{PTVV13}, Theorem~2.9\textup{)}.
\item The derived tensor product
 $\cO(\Steinb)=B\otimes^{\mathbb L}_{\cO(\cM)}B$
 admits, in the formal neighborhood of~$\cL$, a standard
 two-sided bar/Koszul model quasi-isomorphic to
 \[
 \barB(B)
 \;=\;
 T^c(s^{-1}\bar B).
 \]
 The bar differential $d_{\barB}$ encodes the derived
 intersection data \textup{(}the failure of the intersection to be
 transverse\textup{)}, and the cofree coalgebra structure models
 the diagonal
 $\Steinb \to \Steinb \times_{\cL} \Steinb$.
\item Under the HKR identification, Hochschild cochains of~$B$
 are functions on the $(-1)$-shifted cotangent~$T^*[-1]\cL$:
 \[
 \HH^\bullet(B)
 \;\simeq\;
 \cO\bigl(T^*[-1]\cL\bigr).
 \]
 This shifted-cotangent algebra is the formal local model on
 which the formal Darboux theorem expresses the completion of
 $\cM$ along~$\cL$; recovering the actual formal neighborhood
 requires the additional $(-1)$-shifted closed $1$-form of that
 theorem.
\end{enumerate}
\end{theorem}

\begin{proof}
(i) is PTVV \cite{PTVV13}, Theorem~2.9: the derived intersection
of two Lagrangians in a $(-n)$-shifted symplectic stack carries a
$(-(n-1))$-shifted symplectic structure. For $n = 2$ and two
copies of the same Lagrangian, this gives shift~$-1$.

(ii) The derived self-intersection
$\cL \times_\cM^h \cL$ is the derived tensor product
$B \otimes^{\mathbb{L}}_{\cO(\cM)} B$. The Lagrangian condition
gives $\mathbb{L}_{\cL/\cM} \simeq \mathbb{T}_\cL[-1]$
(PTVV~\cite{PTVV13}, \S2; Calaque~\cite{Cal15}, Proposition~2.3).
The two-sided bar resolution models this derived tensor product as
the cofree tensor coalgebra $T^c(s^{-1}\bar B)$
(Calaque~\cite{Cal15}, Theorem~3.2 for the shifted case;
Arinkin--C\u{a}ld\u{a}raru~\cite{AC12} for the unshifted case).

(iii) The HKR theorem gives
$\HH^\bullet(B) \simeq \cO(T^*[-1]\cL)$.
The formal Darboux theorem
(Brav--Bussi--Joyce~\cite{BBJ19}, Theorem~4.1) then implies that
the formal completion $\hat{\cM}_\cL$ is determined by a
$(-1)$-shifted closed $1$-form on $T^*[-1]\cL$.
See Part~\ref{part:frontier} (\S\ref{ssec:bar-as-self-intersection}) for the
detailed proof with all intermediate steps.
\end{proof}

\begin{remark}[The genus tower deforms the presentation]
\label{rem:steinberg-genus-deformation}
\index{genus tower!Steinberg deformation}
At genus~$0$ the Steinberg presentation is the self-intersection
$\Steinb_b = \mathscr{L} \times_{\mathscr{M}}^h \mathscr{L}$ of a
\emph{fixed} Lagrangian. At genus~$g \ge 1$ the Lagrangian
$\mathscr{L}_g$ depends on the moduli: the curvature
$\kappa(\cA) \cdot \omega_g$ is the first-order obstruction to the
displaced Lagrangian remaining isotropic in the geometric reading of
the curved bar equation.

records that the genus-$g$ ``Steinberg variety''

copies of the Lagrangian no longer coincide, and the failure
is controlled by the scalar $\kappa(\cA)$ tensored with the Hodge

coproduct $\Delta$ is a purely combinatorial operation that does
not see the genus of the curve
(Theorem~\ref{thm:topological-e1-rigidity}).
\end{remark}

\begin{theorem}[Topological $E_1$ rigidity; \ClaimStatusProvedHere]
\label{thm:topological-e1-rigidity}

The open-colour $E_1$ coproduct\/ $\Delta$ on the bar complex\/
$\barB(\cA)$ is genus-independent: for all $g \ge 0$,

the canonical identification of the underlying graded vector spaces.
\end{theorem}

\begin{proof}
The deconcatenation coproduct
$\Delta\colon s^{-1}\bar\cA^{\otimes n} \to
\bigoplus_{p+q=n} s^{-1}\bar\cA^{\otimes p}
\otimes s^{-1}\bar\cA^{\otimes q}$
is a purely combinatorial operation on ordered tensor products:
it splits a word into a left factor and a right factor along each
position. This splitting depends only on the linear ordering of
the inputs (the $\Conf_k^{<}(\R)$ factor), not on the holomorphic
collision data (the $\FM_k(\C)$ factor) or on the genus of the
underlying curve. The genus-$g$ deformation
$\dfib^{\,2} = \kappa \cdot \omega_g$
(Construction~\ref{constr:tautological-bundle}) acts entirely on the

\begin{remark}[Perturbative/non-perturbative as clean/excess intersection]
\label{rem:steinberg-perturbative}
\index{perturbative!clean Steinberg intersection}
The Steinberg presentation sharpens the perturbative/non-perturbative
divide. On the Koszul locus the self-intersection is \emph{clean}:
the derived fibers $\operatorname{Tor}^{\cO(\mathscr{M})}_i$
vanish for $i > 0$, the groupoid $\Steinb_b \rightrightarrows
\mathscr{L}$ is \'etale
(Remark~\ref{rem:etale-cross-ref}), and the bar-cobar unit is a
quasi-isomorphism; the theory is perturbatively exact.
Off the Koszul locus the self-intersection has \emph{excess}: the
derived fibers carry higher cohomology, higher $\Ainf$ operations
proliferate, and the homotopy Koszul dual $\cA^!_\infty$ is non-formal
(Remark~\ref{rem:big-vs-small}).
The loop-order spectral sequence
(Theorem~\ref{thm:width-bound}) measures this excess:
the width bound (Theorem~\ref{thm:width-bound}) says the
excess dimension is at most the tree-level width
plus~$\mathfrak{L}(\cA)$. When $\mathfrak{L}(\cA) = 0$, the
self-intersection is clean and Koszulness holds
(Theorem~\ref{thm:one-loop-koszul}).
\end{remark}

\emph{Koszulness is the condition that this presentation is
concentrated.}

\begin{definition}[Chiral Koszulness: the geometric criterion]
\label{def:chiral-koszul-geometric}
A chiral algebra $\cA$ on a smooth curve~$X$ is
\emph{chirally Koszul} if the bar cohomology
$H^\bullet(\barB_X(\cA), d_{\barB})$ is concentrated in bar
degree~$1$:
\begin{equation}\label{eq:koszul-concentration}
H^{p,q}\bigl(\barB_X(\cA)\bigr) = 0 \quad
\text{for } q \neq 0.
\end{equation}
\end{definition}

This is a single condition. Twelve consequences radiate from it, each
illuminating a different face of the same concentration. They are not
independent characterizations assembled by coincidence; they are one
phenomenon seen from twelve angles:

\begin{theorem}[The Koszul dodecahedron;
{\upshape Vol~I, Koszul equivalence theorem}]
\label{thm:koszul-dodecahedron}
For a chiral algebra~$\cA$ with PBW filtration, the following are
equivalent\textup{:}\footnote{Items \textup{(i)--(x)} are
unconditionally equivalent. Item~\textup{(xi)}
\textup{(}Lagrangian transversality\textup{)} is conditional on
perfectness and nondegeneracy hypotheses for the pairing on the
ambient complex. Item~\textup{(xii)} \textup{(}Hodge
purity\textup{)} is fully proved for the semisimple-residue class
\textup{(}Theorem~\textup{\ref{thm:semisimple-purity}):}
Koszulness $\Leftrightarrow$ purity for all
algebras with semisimple effective OPE residues, including the
entire affine lineage. The converse for non-semisimple residues
remains open.}
\begin{enumerate}[label=\textup{(\roman*)}]
\item \textbf{Concentration.}
 Bar cohomology lives in a single degree:
 $H^{p,q}(\barB(\cA)) = 0$ for $q \neq 0$.

\item \textbf{Spectral collapse.}
 The PBW spectral sequence on $\barB(\cA)$ collapses at~$E_2$: no
 higher differentials, no extension problems. The associated graded
 $\gr_F\cA$ is classically Koszul in the sense of Priddy--BGS, and
 the chiral deformation introduces no corrections.

\item \textbf{Formality.}
 The minimal $\Ainf$ model is formal: $m_k = 0$ for $k \geq 3$.
 The binary product $m_2$ determines everything; no genuinely higher
 operations survive. \emph{One-loop exactness}: multi-particle
 collisions on $\FM_k(\C)$ contribute no new cohomology beyond what
 the pairwise OPE already sees.

\item \textbf{Small resolution.}
 The bar-cobar counit
 $\Omegach(\barB(\cA)) \xrightarrow{\;\sim\;} \cA$ is a
 quasi-isomorphism: the cobar of the bar coalgebra resolves~$\cA$.
 This is the ``small'' resolution. The ``big'' resolution is the
 homotopy Koszul dual~$\cA^!_\infty$, the factorization algebra
 whose bar coalgebra is identified with the Verdier dual:
 $\barB(\cA^!_\infty) \simeq \mathbb{D}_{\operatorname{Ran}}\barB(\cA)$
 (Theorem~A intertwining). It exists unconditionally for
 every~$\cA$, Koszul or not. On the Koszul locus, the big
 resolution is \emph{formal}: $\cA^!_\infty \simeq \cA^!$, and
 the small and big constructions agree. Off the Koszul locus,
 $\cA^!_\infty$ carries genuinely higher operations that the
 classical $\cA^!$ does not see.

\item \textbf{Ext diagonal.}
 $\Ext^{p,q}_\cA(\omega_X, \omega_X) = 0$ for $p \neq q$: the
 derived self-Ext of the dualizing module is supported on the diagonal
 of the bigrading. This is the BGS criterion chiralized.

\item \textbf{Hochschild concentration.}
 Chiral Hochschild cohomology $\mathrm{ChirHoch}^n(\cA)$ vanishes
 outside degrees $\{0, 1, 2\}$ at generic level (the critical level
 $k = -h^\vee$ is excluded because $\dim \mathrm{ChirHoch}^0$ can be
 infinite there, see Vol~I Remark~\ref*{rem:critical-level-lie-vs-chirhoch}).
 The polynomial growth of~$\mathrm{ChirHoch}^\bullet$ is the Hochschild
 shadow of bar concentration.

\item \textbf{Factorization homology.}
 $\int_{\Sigma_g} \cA$ is concentrated in degree~$0$ for all genera.
 The higher cohomology of factorization homology measures the failure
 of Koszulness; its vanishing is the global shadow of the local
 concentration.

\item \textbf{Barr--Beck--Lurie.}
 The comparison functor for the bar-cobar monad is an equivalence on
 the fiber: the abstract monadicity theorem applies without
 correction.

\item \textbf{Kac--Shapovalov regularity.}
 The determinant $\det G_h$ of the Shapovalov form is nonvanishing in
 the bar-relevant range: no singular vectors intrude to create
 unexpected cohomology classes.

\item \textbf{Stratum acyclicity.}
 The restriction $i_S^!\barB_n(\cA)$ to every FM boundary stratum~$S$
 is acyclic outside degree~$0$: each collision face contributes
 exactly one cohomology class.

\item \textbf{Lagrangian transversality.}
 The deformation and obstruction complexes $\mathbf{Q}_g(\cA)$ and
 $\mathbf{Q}_g(\cA^!)$ are transverse Lagrangians in the
 $(-1)$-shifted symplectic ambient complex
 $R\Gamma(\overline{\mathcal{M}}_g, \mathcal{Z}(\cA))$
 (Volume~I, Theorem~C).
 \emph{This criterion is conditional on perfectness and nondegeneracy
 hypotheses for the pairing on the ambient complex.}

\item \textbf{Hodge purity.}
 Each $\barB_n(\cA)$ is pure of weight~$n$ as a mixed Hodge module
 on~$\FM_n(X)$: the Hodge filtration and the bar filtration are
 compatible, and no mixing of weights occurs.
 \emph{For the semisimple-residue class,
 the biconditional is proved\textup{:}
 purity $\Leftrightarrow$ Koszulness
 \textup{(}Theorem~\textup{\ref{thm:semisimple-purity}).}
 The converse for non-semisimple residues remains open.}
\end{enumerate}
\end{theorem}

\begin{remark}[The loop-order perspective]
The twelve characterizations above are algebraic and geometric. The
loop-order spectral sequence
(Theorem~\ref{thm:width-bound}) adds a physical
perspective: for logarithmic $\SCchtop$-algebras
with $\mathfrak{L}(\cA)<\infty$, the bar
cohomological width is bounded by the tree-level width plus
$\mathfrak{L}(\cA)$ (Theorem~\ref{thm:width-bound}). When
the tree-level bar complex is concentrated, Koszulness holds iff
$\mathfrak{L}(\cA)=0$.
One-loop exact bulk BV quantization (as in the affine half-space
theorem) implies Koszulness before DS reduction
(Theorem~\ref{thm:one-loop-koszul}); for the principal nilpotent,
reduction breaks Swiss-cheese formality (but preserves Koszulness)
in the Virasoro and $W_N$ families
(Theorem~\ref{thm:ds-koszul-obstruction}).
\end{remark}

\begin{remark}[Classical specialization]
\label{rem:classical-specialization}
When $X = \mathrm{pt}$, the chiral tensor product reduces to the
ordinary tensor product, the FM compactification is the
Stasheff associahedron, and the twelve criteria collapse to
the classical Koszul theory of Priddy--BGG--BGS--Loday--Vallette:
\begin{itemize}
\item Concentration $=$ the classical bar complex
 $B(A) = T^c(s^{-1}\bar{A})$ has cohomology in degree~$1$;
\item Formality $=$ the $\Ainf$ minimal model is formal
 (Keller's classicality);
\item Small resolution $=$ the Koszul resolution
 $A^! \otimes_{A^{!\text{-comod}}} A^!
 \xrightarrow{\sim} A$ (Priddy);
\item Ext diagonal $=$ $\Ext^{p,q}_A(k,k) = 0$ for $p \neq q$
 (the BGS criterion).
\end{itemize}
The chiral theory is not a generalization that weakens these
conditions; it is a \emph{lift} that reveals their geometric origin.
Concentration of bar cohomology, in the classical setting, is a
combinatorial miracle. In the chiral setting, it is a consequence
of the Arnold cancellation on FM compactifications: the boundary
strata of $\FM_k(X)$ produce exact sequences, and Koszulness
is the statement that these exact sequences have no higher
cohomology. The ``miracle'' of classical Koszulness is the shadow
of the geometry of configuration spaces.
\end{remark}

\begin{remark}[Big versus small]
\label{rem:big-vs-small}
The homotopy Koszul dual~$\cA^!_\infty$ is the factorization algebra
determined by the Verdier intertwining
$\mathbb{D}_{\operatorname{Ran}}\barB(\cA) \simeq \barB(\cA^!)$
(Theorem~A): $\cA^!_\infty$ is the algebra whose bar coalgebra is
$\mathbb{D}_{\operatorname{Ran}}\barB(\cA)$.
It exists for \emph{every} chiral algebra, with no Koszul
hypothesis. It is the ``big'' Koszul dual: a factorization algebra on
$\operatorname{Ran}(X)$ carrying the full homotopy type of the bar of
the dual algebra~$\cA^!$.

The classical Koszul dual $\cA^! = T_{\mathrm{ch}}(V^* \otimes
\omega[1])/\langle R^\perp\rangle$ is the ``small'' Koszul dual: a
quadratic algebra defined by generators and relations.

On the Koszul locus, big $=$ small: $\cA^!_\infty$ is formal
(criterion~(iii) above) and its cohomology is~$\cA^!$. Off the
Koszul locus, $\cA^!_\infty$ carries genuinely higher $\Ainf$
operations that $\cA^!$ does not encode. The Virasoro algebra at
$c \neq 13$ is the archetype: $\mathrm{Vir}_c^! = \mathrm{Vir}_{26-c}$
is the small Koszul dual, but the big Koszul dual
$(\mathrm{Vir}_c)^!_\infty$ carries nonvanishing $m_k$ for all
$k \geq 3$, the infinite tower of wheel diagrams in the
Feynman-diagrammatic computation. The bar complex is genuinely
non-formal; the higher operations \emph{are} the curved bar
structure $\dfib^{\,2} = \kappa(\cA) \cdot \omega_1$ of Volume~I.

The Swiss-cheese operad $\SCchtop$ is \emph{homotopy-Koszul}
(Theorem~\ref{thm:homotopy-Koszul}): this is a property of the
operad, not of individual algebras. It means the bar-cobar
adjunction on $\SCchtop$-algebras is a Quillen equivalence: the
``small resolution'' works at the operadic level. Individual algebras may have formal or non-formal Swiss-cheese
structure; the operad is always homotopy-Koszul.
In the Steinberg presentation
(Theorem~\ref{thm:steinberg-presentation}), the distinction is:
part~(ii), the categorical equivalence, uses only operad-level
homotopy-Koszulity and is unconditional; part~(iii), the
identification with $\cA^!_{\mathrm{line}}$-modules and line operators, requires
algebra-level Koszulness so that big $=$ small.
\end{remark}

\begin{theorem}[Width bound for bar cohomology;
\ClaimStatusProvedElsewhere]
\label{thm:width-bound}
\index{width bound|textbf}
\index{bar complex!width bound|textbf}
Let\/ $\cA$ be a logarithmic $\SCchtop$-algebra with loop
complexity\/ $\mathfrak{L}(\cA) < \infty$
\textup{(}the maximal loop order at which the BV-BRST
spectral sequence on the bar complex acquires new
contributions\textup{)}.
Then the bar cohomological width is bounded by the tree-level
width plus\/~$\mathfrak{L}(\cA)$:
\[
\mathrm{width}\bigl(H^\bullet(\barB(\cA))\bigr)
\;\le\;
\mathrm{width}\bigl(H^\bullet(\barB(\cA))_{\mathrm{tree}}\bigr)
\;+\; \mathfrak{L}(\cA).
\]
When the tree-level bar complex is concentrated
\textup{(}width~$1$\textup{)}, chiral Koszulness holds if
and only if\/ $\mathfrak{L}(\cA) = 0$.
\end{theorem}

\begin{proof}
The loop-order filtration on the bar complex induces
a spectral sequence whose $E_1$ page computes tree-level
bar cohomology. Each subsequent page can widen the support
by at most~$1$ in bar degree. After $\mathfrak{L}(\cA)$
pages, the spectral sequence degenerates. The total width
is therefore bounded by the tree-level width plus the number
of non-trivial differentials, which is at
most~$\mathfrak{L}(\cA)$.
\end{proof}

\begin{theorem}[One-loop Koszul criterion;
\ClaimStatusProvedElsewhere]
\label{thm:one-loop-koszul}
\index{one-loop Koszul criterion|textbf}
\index{Koszulness!one-loop criterion|textbf}
\index{BV-BRST!one-loop exactness|textbf}
Let\/ $\cA$ be the boundary chiral algebra of a 3d
holomorphic-topological gauge theory whose BV-BRST
quantization is one-loop exact \textup{(}i.e.\ the
BV master equation is solved at tree level plus one loop,
with no higher-loop corrections to the effective action\textup{)}.
Then $\cA$ is chirally Koszul:
\[
H^{p,q}\bigl(\barB(\cA)\bigr) = 0 \quad \text{for } q \neq 0.
\]
In particular, the bar complex is cogenerated in degree~$1$,
the $\Ainf$ minimal model is formal \textup{(}$m_k = 0$
for $k \ge 3$\textup{)}, and the bar-cobar counit is a
quasi-isomorphism.
\end{theorem}

\begin{proof}
One-loop exactness means the BV effective action $S_{\mathrm{eff}}$
is quadratic plus one-loop determinantal: no Feynman diagrams
beyond one loop contribute. In the bar-complex interpretation,
this forces the loop-order spectral sequence to degenerate at
$E_2$: the tree-level bar cohomology is concentrated (the
free-field propagator gives a Koszul complex), and the one-loop
correction is a scalar determinant that does not shift bar
degrees. By the width bound (Theorem~\ref{thm:width-bound})
with $\mathfrak{L}(\cA) = 0$, the bar cohomology is
concentrated. The remaining Koszul properties follow from the
Koszul dodecahedron (Theorem~\ref{thm:koszul-dodecahedron}).
\end{proof}

\begin{theorem}[DS reduction and Swiss-cheese non-formality;
\ClaimStatusProvedElsewhere]
\label{thm:ds-koszul-obstruction}
\index{Drinfeld--Sokolov reduction!Koszul obstruction|textbf}
\index{Swiss-cheese!non-formality from DS reduction|textbf}
Let\/ $\hat{\fg}_k$ be an affine Kac--Moody vertex algebra
at non-critical level \textup{(}chirally Koszul by
Theorem~\textup{\ref{thm:one-loop-koszul})}, and let\/
$\cW_k(\fg) = H^0_{\mathrm{DS}}(\hat{\fg}_k)$ be the
principal $\cW$-algebra obtained by Drinfeld--Sokolov
reduction.
\begin{enumerate}[label=\textup{(\roman*)}]
\item $\cW_k(\fg)$ is chirally Koszul
 \textup{(}bar cohomology concentrated\textup{)}.
\item The transferred $\Ainf$ operations satisfy
 $m_k^{\mathrm{DS}} \neq 0$ for $k \ge 3$ whenever
 $\fg$ has rank $\ge 2$: DS reduction introduces
 higher-order poles in the OPE that generate genuine
 higher $\Ainf$ operations on the bar complex.
\item The Swiss-cheese structure of\/ $\cW_k(\fg)$ is
 non-formal: the shadow depth is infinite
 \textup{(}class~$\mathbf{M}$\textup{)} for
 $\mathrm{Vir}_c$ and $\cW_N$ at generic parameters.
\end{enumerate}
Chirally Koszul \textup{(}bar concentration\textup{)}
and Swiss-cheese formal \textup{(}$m_{k \ge 3} = 0$\textup{)}
are distinct conditions. DS reduction preserves the former
but breaks the latter.
\end{theorem}

\begin{proof}
Part~(i): the PBW degeneration of\/ $\hat{\fg}_k$
descends through DS reduction
(Volume~I, PBW universality), so the bar spectral
sequence for $\cW_k(\fg)$ still collapses.
Part~(ii): DS reduction transfers the affine OPE through
a BRST complex, and the homological transfer theorem
(HTT) produces non-trivial higher operations $m_k^H$
whenever the BRST differential creates higher-order
poles. For the principal nilpotent in $\mathfrak{sl}_N$
($N \ge 3$), the Virasoro field $T = S(J)$ has a
fourth-order pole in its self-OPE from the Sugawara
construction; this pole is absent from the affine algebra
and is manufactured by reduction.
Part~(iii) follows from (ii) and the shadow depth
classification of Volume~I: non-vanishing $m_k$ for all
$k \ge 3$ forces infinite shadow depth.
\end{proof}

\begin{remark}[The open problem]
\label{rem:koszul-open-problem}
\emph{Chiral Koszulness from physics} is resolved for the affine
lineage: Theorem~\ref{thm:one-loop-koszul} proves it under one-loop
exactness, and Theorem~\ref{thm:ds-koszul-obstruction} identifies the
Drinfeld--Sokolov descent obstruction. What remains open is the
general non-affine case: whether the bridge theorem
(Theorem~\ref{thm:physics-bridge}) by itself, without affine-lineage
input or one-loop exactness, forces the resulting logarithmic
$\SCchtop$-algebra to be chirally Koszul. The three worked examples
(free multiplet, Landau--Ginzburg, abelian Chern--Simons) are all
Koszul; the Virasoro and $\mathcal{W}_3$ algebras are also chirally
Koszul but have non-formal Swiss-cheese structure
($m_{k \geq 3}^{\mathrm{SC}} \neq 0$). The boundary between formal
and non-formal Swiss-cheese structure is controlled by the pole
structure of the OPE: simple poles give formal algebras; higher-order
poles create the higher operations that break formality. The
remaining open problem is to promote this example-by-example control to
a general theorem beyond the affine lineage.
\end{remark}

\begin{remark}[Minimal model non-Koszulness mechanism]
\label{rem:vol2-minimal-model-non-koszul}
\index{minimal model!non-Koszulness mechanism}
While the \emph{universal} Virasoro algebra $V_c$ is chirally Koszul
for all~$c$, the \emph{simple quotient} $L(c_{p,q}, 0)$ at minimal
model values is not. Volume~I
\textup{(}Proposition~\textup{\ref*{V1-prop:minimal-model-non-koszul})}
identifies the mechanism: the null vector at weight
$w_0 = (p{-}1)(q{-}1)$ removes a generator from $\barB^1_{w_0}$
without affecting $\barB^2_{w_0}$, enlarging the kernel of the
differential and producing nonvanishing $H^2$. This is the
structural reason that PBW universality
\textup{(}which applies to freely strongly generated algebras\textup{)}
does not extend to simple quotients.
\end{remark}

\subsection{The purity criterion: Koszulness from the Decomposition Theorem}
\label{subsec:purity-koszulness}

The Beilinson--Ginzburg--Soergel proof that category~$\cO$ is Koszul
proceeds by establishing \emph{purity} of intersection cohomology on
Schubert varieties. The Decomposition Theorem (BBD) supplies the input:
for the proper map $\pi\colon G/B \to G/P$, the direct image
$R\pi_* \IC_{G/B}$ decomposes into shifts of IC complexes, each pure
of a single weight. Concentration of the associated graded, hence
Koszulness, follows. This is the classical mechanism behind the
final row of the Steinberg dictionary
(Theorem~\ref{thm:steinberg-presentation}(iv):
IC purity $\leftrightarrow$ Koszulness).
We propose the chiral analog.

\begin{definition}[OPE-weighted local system and collision purity]
\label{def:rational-smooth-collision}
Let $\cA$ be a chiral algebra on a smooth curve~$X$. For each
$k \geq 2$, the OPE of~$\cA$ determines an \emph{OPE-weighted
local system} $\cL_{\cA}$ on $\Conf_k(\C)$, whose fiber at
$(z_1,\ldots,z_k)$ is $\cA^{\otimes k}$ with flat connection
$\nabla = d - \sum_{i<j} \Omega_{ij}\, d\log(z_i - z_j)$
of KZ type, where $\Omega_{ij} \in \End(\cA^{\otimes k})$ is
the OPE residue acting on factors~$i,j$. Flatness
$\nabla^2 = 0$ follows from OPE associativity
(the singular parts satisfy the infinitesimal braid relations).
The monodromy representation $\pi_1(\Conf_k(\C)) \to
\mathrm{GL}(\cA^{\otimes k})$ defines~$\cL_{\cA}$. The Fulton--MacPherson
compactification $\FM_k(\C)$ carries boundary divisors
$D_{ij} = \{z_i = z_j\}$, along which the local system acquires
poles. Define the \emph{collision filtration} by pole order:
$\cL^{\leq d}_{\cA}$ is the subsheaf of sections with poles of
order at most~$d$ along all boundary divisors. The associated
graded $\mathrm{gr}^d\cL_{\cA} = \cL^{\leq d}/\cL^{\leq d-1}$
lives on the boundary strata of $\FM_k(\C)$.
Say the OPE collision data is \emph{pure} if for each~$d$ and
each~$k$, the cohomology $H^*(\FM_k(\C), \mathrm{gr}^d\cL_{\cA})$
is pure of weight~$d$ (in the sense of mixed Hodge theory).
\end{definition}

\begin{proposition}[Collision purity implies Koszulness;
\ClaimStatusProvedHere]
\label{prop:purity-koszul}
\label{prop:collision-purity-implies-koszul}
\index{Koszulness!purity criterion}
If the OPE collision data of a chiral algebra~$\cA$ is pure in the
sense of Definition~\ref{def:rational-smooth-collision}, then
$\cA$ is chirally Koszul. Specifically, collision purity implies:
\begin{enumerate}[label=\textup{(\alph*)}]
\item the cohomology $H^*(\FM_k(\C), \mathrm{gr}^d\cL_{\cA})$
 is pure of weight~$d$ for all~$k,d$;
\item the weight spectral sequence on $H^*(\FM_k(\C),
 \cL_{\cA})$ degenerates at~$E_1$, giving
 $H^{p,q}(\barB(\cA)) = 0$ for $p + q \neq$ bar degree;
\item the bar cohomology is concentrated, hence~$\cA$ is chirally
 Koszul.
\end{enumerate}
The converse is \emph{false} in general: the Virasoro algebra at
$c \neq 0$ is chirally Koszul
\textup{(}Theorem~\textup{\ref{thm:ds-koszul-obstruction}(i))} but
fails collision purity
\textup{(}Proposition~\textup{\ref{prop:purity-failure-virasoro})}.
The weight spectral sequence has non-trivial differentials~$d_r$
producing non-vanishing higher $\Ainf$ operations $m_k \neq 0$ for
$k \geq 3$, but these differentials cancel perfectly: the total
bar cohomology is concentrated despite the failure of
$E_1$-degeneration. Collision purity characterizes not chiral
Koszulness but rather \emph{Koszulness with Swiss-cheese formality}
\textup{(}Remark~\textup{\ref{rem:purity-sharp-boundary})}.
\end{proposition}

\begin{proof}
The forward direction (collision purity $\Rightarrow$ Koszulness)
follows from the weight spectral sequence argument: purity of each
$\mathrm{gr}^d\cL_{\cA}$ forces $E_1$-degeneration (as in
Step~3 of the proof of Theorem~\ref{thm:semisimple-purity}), giving
bar concentration. The counterexample to the converse is
Proposition~\ref{prop:purity-failure-virasoro}: Virasoro is Koszul
(PBW universality, Vol~I) but not collision-pure (the non-split
MHM extension between weights~$4$ and~$2$ produces a non-trivial
$d_2$).
\end{proof}

\begin{proposition}[Purity for the Koszul examples;
\ClaimStatusProvedHere]
\label{prop:purity-koszul-examples}
The purity criterion holds for the three Koszul examples:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \textbf{Free $\beta\gamma$:} All OPE poles are simple
 ($\beta(z)\gamma(w) \sim (z-w)^{-1}$). The collision strata
 are smooth (codimension-one normal crossings along each
 diagonal $z_i = z_j$). The IC complex is the constant sheaf,
 which is pure.

\item \textbf{Landau--Ginzburg:} The superpotential~$W$ gives
 OPE poles of order at most one in the chiral algebra of the
 LG model. Collision strata are transverse intersections of
 smooth divisors. Purity follows from the smoothness of the
 strata.

\item \textbf{Abelian Chern--Simons:} The abelian current algebra
 $J(z)J(w) \sim k(z-w)^{-2}$ has a double pole, but $k(z-w)^{-2}$
 is a \emph{scalar} OPE coefficient: it takes values in $\C \cdot
 \mathbf{1} \subset \cA$, the center. In the bar complex, scalar
 OPE terms contribute to the curvature $m_0 = \kappa(\cA)$, which
 defines the curving of the $\Ainf$ structure
 (Vol~I, Theorem~D), not to the bar differential $m_k$ for
 $k \geq 2$. Concretely, $\mathrm{gr}^2\cL_{\cA}$ on the
 boundary divisor $D_{ij} \subset \FM_k(\C)$ is the trivial local
 system $\C$ (constant fiber from the scalar residue), whose
 cohomology is pure of weight~$2$. The non-central OPE data
 that feeds the bar differential has pole order at most one:
 for the abelian current algebra $[J,J]$ is central, and there
 are no non-central fields with $\Delta \leq 1$ beyond~$J$ itself
 (the only field of conformal dimension~$1$ is $J$, which is central: $[J,J] = k\cdot\mathbf{1}$).
 Thus $\mathrm{gr}^1\cL_{\cA}$ is also a local system of pure
 weight~$1$, and purity holds.
\end{enumerate}
\end{proposition}

\begin{proof}
In each case, verify that the collision strata relevant to the bar
differential are smooth or have at worst normal-crossing
singularities. For~(i), the $\beta\gamma$ OPE is first-order and
the FM boundary divisors are smooth. For~(ii), the LG chiral
algebra (as computed in Part~\ref{part:e1-core}) has first-order OPE in the
$\Ainf$ generators. For~(iii), the double-pole residue $k \cdot \mathbf{1}$ is scalar,
hence lies in the kernel of the bar differential (which acts
trivially on~$\mathbf{1}$); it contributes to $m_0$ (curvature)
via Vol~I Theorem~D. The associated graded $\mathrm{gr}^2\cL$
is the trivial local system and is pure. The remaining OPE data
(the regular terms $:J^2:$ etc.)\ have pole order~$\leq 1$,
so $\mathrm{gr}^1\cL$ is a local system of pure weight, completing
the verification.
\end{proof}

\begin{proposition}[Purity failure for Virasoro; \ClaimStatusProvedHere]
\label{prop:purity-failure-virasoro}
The purity criterion \emph{fails} for the Virasoro algebra at
generic central charge~$c$. The OPE
\[
T(z)T(w) \;\sim\; \frac{c/2}{(z-w)^4} +
\frac{2T(w)}{(z-w)^2} + \frac{\partial T(w)}{z-w}
\]
creates a fourth-order pole. The space $\FM_2(\C)$ is itself
smooth (it is the real oriented blowup of $\C^2$ along the
diagonal), so the failure is \emph{not} a singularity of the
ambient space but rather a failure of purity of the OPE-weighted
local system. Concretely: $\mathrm{gr}^4\cL_{\mathrm{Vir}}$
on the boundary divisor of $\FM_2(\C)$ has fiber~$\C$
(from the scalar $c/2$ residue), contributing weight~$4$, while
$\mathrm{gr}^2\cL_{\mathrm{Vir}}$ has fiber~$\cA$ (from the
non-scalar $2T(w)$ residue), contributing weight~$2$. The
extension class between these graded pieces in
$H^*(\FM_2(\C), \cL_{\mathrm{Vir}})$ is \emph{non-split}: the
quartic-pole scalar $c/2$ and the quadratic-pole field $2T$
are linked by the Virasoro algebra structure ($L_{-1}$-descendants),
defining a non-trivial class in $\mathrm{Ext}^1$ between weight~$4$
and weight~$2$. This non-split extension obstructs purity of
$\cL_{\mathrm{Vir}}$. At the level of the bar complex,
the higher-pole mixing produces a non-vanishing $m_3$ operation
(computed on $\FM_3(\C)$ via iterated OPE through the quartic
pole), which is precisely the obstruction to collision purity
(Swiss-cheese formality), not to chiral Koszulness itself.
\end{proposition}

\begin{proof}
The OPE-weighted local system $\cL_{\mathrm{Vir}}$ on
$\Conf_2(\C)$ has the collision filtration
$\cL^{\leq 1} \subset \cL^{\leq 2} \subset \cL^{\leq 4} =
\cL_{\mathrm{Vir}}$ (there is no order-$3$ pole in the $TT$ OPE).
Write $F^{-d} = \cL^{\leq d}$ for the corresponding decreasing
filtration, so the non-zero graded pieces sit at
$p = -1, -2, -4$ (weight $d = -p = 1, 2, 4$).
The weight spectral sequence has
$E_1^{p,q} = H^{p+q}(\FM_2, \mathrm{gr}^{-p}\cL)$
with $d_r$ of bidegree $(r, 1-r)$.

The first differential $d_1\colon E_1^{-2,*} \to E_1^{-1,*}$
connects weight~$2$ to weight~$1$: the $(z-w)^{-2}$ residue
$2T(w)$ maps to its $L_{-1}$-descendant $\partial T(w)$ in the
simple-pole piece. This is the conformal-descendant relation
and is determined by the $\mathfrak{sl}_2$-subalgebra of
$\mathrm{Vir}$; it does not by itself obstruct purity.

The key obstruction is the extension between weight~$4$ and
weight~$2$. The $d_2$ differential
$d_2\colon E_2^{-4,*} \to E_2^{-2,*}$
connects weight~$4$ to weight~$2$ (filtration gap~$2$).
Explicitly, the composite
\[
\mathrm{Res}_{z=w}\bigl[(z-w)^{-4} \cdot \tfrac{c}{2}\bigr]
\;\leadsto\; \tfrac{1}{3!}\partial^3_w\bigl[2T(w)\bigr]
\]
arising from the Laurent expansion of the $TT$ OPE expresses
the quartic-pole scalar as a third derivative of the
quadratic-pole field. This composite is non-trivial
(it is the $c \neq 0$ obstruction), so $d_2 \neq 0$ and
the spectral sequence does not degenerate at~$E_1$.

At the bar-complex level, the non-split extension between
weights~$4$ and~$2$ in the two-point local system feeds
(via iterated OPE on $\FM_3(\C)$) into the non-vanishing
of the ternary operation~$m_3$, confirming failure of purity.
\end{proof}

\begin{remark}[The simple-pole principle]
\label{rem:simple-pole-principle}
The purity criterion distills to a single geometric slogan:
\emph{simple poles give pure weights give $E_1$-degeneration give
formal Swiss-cheese structure; higher non-scalar poles give mixed
extensions give non-trivial differentials give non-formal
Swiss-cheese structure.}
This is the Chriss--Ginzburg content of the chiral Koszulness
question. The Decomposition Theorem is the machine that converts
the local geometric data (pole orders) into the global algebraic
conclusion (concentration of bar cohomology). The open problem
reduces to: \emph{prove that the OPE pole structure of a logarithmic
$\SCchtop$-algebra from the bridge theorem has at most simple
effective poles.} Theorem~\ref{thm:semisimple-purity} below
makes this principle precise: for algebras with semisimple effective
OPE residues, purity is equivalent to Koszulness.

The loop-order criterion of
Theorem~\ref{thm:one-loop-koszul} resolves this for the
affine lineage: one-loop exactness of the bulk BV theory implies
simple effective poles in the boundary OPE, hence chiral Koszulness
of $V^k(\fg)$.
Drinfeld--Sokolov reduction introduces higher poles by homological
transfer through the BRST complex
(Theorem~\ref{thm:ds-koszul-obstruction}), explaining the
non-formal Swiss-cheese structure of Virasoro (at $c \neq 0$) and
$\mathcal{W}_N$ from the purity perspective: the transferred
$m_k^{\mathrm{DS}}$ arise from decorated BRST trees that produce
exactly the higher-pole contributions breaking formality.
\end{remark}

\begin{theorem}[The semisimple purity criterion; \ClaimStatusProvedHere]
\label{thm:semisimple-purity}
\index{Koszulness!purity criterion!semisimple case|textbf}
\index{purity!semisimple OPE residue|textbf}
\index{mixed Hodge module!purity of collision data}
Let~$\cA$ be a chiral algebra and let $\Omega_{ij}^{\mathrm{eff}}$
denote the effective OPE residue operators obtained from
$\Omega_{ij}$ after projecting away the scalar\textup{/}curvature
contributions that feed $m_0 = \kappa(\cA)$ rather than the
bar differential~$d_{\barB}$
\textup{(}cf.~Proposition~\textup{\ref{prop:purity-koszul-examples}(iii)).}
If the effective residues $\Omega_{ij}^{\mathrm{eff}}$ are
\emph{semisimple} \textup{(}diagonalizable\textup{)} endomorphisms
of $\cA^{\otimes k}$ for every~$k \geq 2$, then the following
are equivalent\textup{:}
\begin{enumerate}[label=\textup{(\alph*)}]
\item $\cA$ is chirally Koszul.
\item The OPE collision data of~$\cA$ is pure in the sense
 of Definition~\textup{\ref{def:rational-smooth-collision}.}
\end{enumerate}
In particular, Proposition~\textup{\ref{prop:purity-koszul}}
holds for every chiral algebra with semisimple effective OPE
residues. This class includes\textup{:}
\begin{enumerate}[label=\textup{(\roman*)}]
\item the affine Kac--Moody algebras $V^k(\fg)$ at generic level
 \textup{(}the Casimir operator~$\Omega = \sum_a e^a \otimes e_a$
 is semisimple for generic~$k$\textup{);}
\item the free $\beta\gamma$ system
 \textup{(}first-order OPE with scalar residue\textup{);}
\item the Landau--Ginzburg chiral algebras
 \textup{(}first-order effective OPE\textup{);}
\item the abelian Chern--Simons algebra
 \textup{(}scalar double pole $+$ trivial effective residue\textup{).}
\end{enumerate}
\end{theorem}

\begin{proof}
The forward direction (a)$\Rightarrow$(b) is part of K12
(Koszulness implies Hodge purity) in
Theorem~\ref{thm:koszul-dodecahedron}. We prove the converse
(b)$\Rightarrow$(a).

\medskip\noindent\textbf{Step~1: Setup.}
Recall from Definition~\ref{def:rational-smooth-collision}
that the OPE-weighted local system
$\cL_\cA$ on $\Conf_k(\C)$ has a flat connection
$\nabla = d - \sum_{i<j}\Omega_{ij}\,d\log(z_i - z_j)$,
and its collision filtration
$\cL^{\leq 0} \subset \cL^{\leq 1} \subset \cdots$
is indexed by pole order along the boundary divisors of
$\FM_k^{\mathrm{alg}}(\C)$.

By the proof of Proposition~\ref{prop:decomposition-fm} below,
$\cL_\cA$ underlies a regular holonomic algebraic
$\cD$-module on $\Conf_k(\C)$ (the one-forms $d\log(z_i-z_j)$
are algebraic, the residues $\Omega_{ij}$ are constant), and the
collision filtration coincides with the Malgrange--Kashiwara
$V$-filtration along the normal-crossing boundary
divisor $D = \bigcup_{i<j}D_{ij}$. By Saito's functor
(M.~Saito, ``Modules de Hodge polarisables,'' 1988),
$\cL_\cA$ underlies a mixed Hodge module (MHM) on
$\FM_k^{\mathrm{alg}}(\C)$.

\medskip\noindent\textbf{Step~2: Purity of $V$-filtration graded
pieces from semisimplicity.}
Let $D_S$ be a boundary stratum of
$\FM_k^{\mathrm{alg}}(\C)$ corresponding to a collision pattern~$S$.
The effective residue $\Omega_S^{\mathrm{eff}}$ along $D_S$
is a composition of the individual $\Omega_{ij}^{\mathrm{eff}}$.
Since each $\Omega_{ij}^{\mathrm{eff}}$ is semisimple
by hypothesis, the local monodromy $T_S = \exp(2\pi i\,
\Omega_S^{\mathrm{eff}})$ around~$D_S$ is semisimple.

By the Borel--Malgrange theorem (Deligne, ``\'Equations
diff\'erentielles \`a points singuliers r\'eguliers,''
Lecture Notes in Math.\ 163), a regular singular
connection with semisimple monodromy has a canonical extension
to a logarithmic connection whose residue is semisimple with
eigenvalues in $[0,1)$. The $V$-filtration graded pieces
$\gr^V_\alpha\cL_\cA$ (for $\alpha \in [0,1)$) are therefore
the eigenspaces of a semisimple endomorphism.

In Saito's theory of MHM, the nearby cycles functor~$\psi_f$ applied
to a mixed Hodge module whose underlying $\cD$-module has
\emph{semisimple monodromy} produces a \emph{pure} Hodge module
(Saito, ``Mixed Hodge modules,'' Publ.\ RIMS \textbf{26},
Theorem~5.3.1). Concretely: when the monodromy weight filtration
is trivial (as it is for semisimple monodromy, since the logarithm
$N = \log T_u$ of the unipotent part vanishes), $\psi_f(M)$ inherits
purity from~$M$ without monodromy-weight shifting.

Applied to the collision data: the $V$-filtration graded piece
$\gr^d\cL_\cA$ on the boundary stratum~$D_S$ is a pure Hodge module
of weight~$d$. The collision filtration index~$d$ records the OPE
pole order, which under the $V$-filtration${}={}$collision-filtration
identification
(Proposition~\ref{prop:decomposition-fm}, proof)
equals the Hodge-module weight.

\medskip\noindent\textbf{Step~3: Weight shift forces
$\boldsymbol{q = 0}$.}
Each boundary stratum $D_S$ of $\FM_k^{\mathrm{alg}}(\C)$ is a
smooth projective variety (it is a product of lower FM spaces,
each an iterated blowup of projective spaces along smooth centers).
By Saito's purity theorem for smooth projective varieties
(``Mixed Hodge modules,'' Theorem~5.3.1 and Corollary~5.3.8):
for a pure Hodge module~$M$ of weight~$w$ on a smooth projective
variety~$S$,
\begin{equation}\label{eq:saito-weight-shift}
H^q(S, M) \;\text{is pure of weight}\; w + q
\quad\text{for all } q.
\end{equation}

Now apply~\eqref{eq:saito-weight-shift} to
$M = \gr^d\cL_\cA$, which is pure of weight~$w = d$ (Step~2),
on the stratum~$D_S$:
\[
H^q(D_S,\, \gr^d\cL_\cA) \;\text{is pure of weight}\;
d + q.
\]
The purity hypothesis~(b) demands that
$H^q(\FM_k,\, \gr^d\cL_\cA)$ is pure of weight~$d$
(Definition~\ref{def:rational-smooth-collision}). Therefore
$d + q = d$, hence $q = 0$: the only non-vanishing cohomology
group of each graded piece lies in degree zero.

\medskip\noindent\textbf{Step~4: Spectral degeneration and
bar concentration.}
The collision filtration induces a spectral sequence
\[
E_1^{p,q} = H^{p+q}\bigl(\FM_k,\,\gr^{-p}\cL_\cA\bigr)
\;\Longrightarrow\;
H^{p+q}\bigl(\FM_k,\,\cL_\cA\bigr).
\]
By Step~3, $E_1^{p,q} = 0$ unless $p + q = 0$ (i.e.\ $q = -p$).
In particular, the $E_1$~page is concentrated on the anti-diagonal
$p + q = 0$; all differentials~$d_r$ (for $r \geq 1$)
vanish for degree reasons. The spectral sequence degenerates
at~$E_1$, and
\[
H^n\bigl(\FM_k(\C),\,\cL_\cA\bigr)
\;=\;
\begin{cases}
\bigoplus_d H^0(\FM_k,\,\gr^d\cL_\cA) & \text{if } n = 0, \\
0 & \text{if } n \neq 0.
\end{cases}
\]

By the identification $H^{k,q}(\barB(\cA)) =
H^q(\FM_k(\C),\,\cL_\cA)$
(Definition~\ref{def:chiral_bar}), we obtain
$H^{k,q}(\barB(\cA)) = 0$ for $q \neq 0$, which is the
concentration condition~\eqref{eq:koszul-concentration}
defining chiral Koszulness.

\medskip\noindent\textbf{Step~5: Verification for the four families.}
We verify the semisimplicity hypothesis:

(i) \emph{Affine $V^k(\fg)$}: the effective OPE after removing the
scalar Killing-form pole $k\delta^{ab}/(z-w)^2$ has the
simple-pole residue $\Omega_{ij}^{\mathrm{eff}} = \sum_a
e_i^a \otimes e_j^a$ acting via the structure constants of~$\fg$.
For generic~$k$, this is semisimple:
$\Omega_{ij}^{\mathrm{eff}}$ on the representation
$\fg^{\otimes k}$ has eigenvalues that are rational
functions of~$k$, and the discriminant is nonvanishing on
a Zariski-open set.

(ii) \emph{Free $\beta\gamma$}: the OPE
$\beta(z)\gamma(w) \sim 1/(z-w)$ has effective
residue $\Omega_{ij}^{\mathrm{eff}} = \mathrm{Id}|_{\text{fiber}}$,
which is trivially semisimple.

(iii) \emph{LG models}: the effective OPE has at most simple poles
with nilpotent-free residues by the computation in
Proposition~\ref{prop:purity-koszul-examples}(ii);
semisimplicity holds.

(iv) \emph{Abelian CS}: after removing the scalar double pole,
the effective residue is zero (the only field $J$ is central).
The zero operator is semisimple.

In each case the semisimplicity hypothesis is satisfied, so
the biconditional (a)$\Leftrightarrow$(b) holds by the
preceding argument.
\end{proof}

\begin{remark}[Scope and the non-semisimple frontier]
\label{rem:purity-scope}
\index{purity!non-semisimple frontier}
The semisimplicity hypothesis is the natural boundary of the
purity criterion. For algebras with \emph{non-semisimple} effective
monodromy, in particular the Virasoro algebra
(Proposition~\ref{prop:purity-failure-virasoro}) and the
$\mathcal{W}_N$ algebras obtained by DS reduction, the
monodromy weight filtration is non-trivial:
$N = \log T_u \neq 0$, the $V$-filtration graded
pieces carry genuine mixed Hodge module structure, and the
argument of Step~3 fails because the weight shift
becomes $d + q + s$ where $s$ is a correction from
the monodromy weight filtration. This is the MHM-theoretic
origin of the failure of \emph{collision purity} for Virasoro
(Virasoro \emph{is} chirally Koszul, the bar cohomology is
concentrated, but the collision filtration graded pieces carry
non-trivial mixed Hodge structure):
the $L_{-1}$-descendant relation
$\partial T = \mathrm{Res}_{z=w}[(z-w)^{-2}\cdot 2T(w)]$
is a non-split extension between weight~$2$ and weight~$1$
in the category of MHS, and the corresponding non-trivial
$d_2$ differential is the obstruction computed in
Proposition~\ref{prop:purity-failure-virasoro}.

The full Proposition~\ref{prop:purity-koszul} (without the
semisimplicity hypothesis) is equivalent to the claim that
the converse of the weight-shift argument still holds when
the monodromy is non-semisimple: that purity of the cohomology
$H^*(\FM_k,\gr^d\cL_\cA)$ implies concentration even when the
\emph{coefficient} $\gr^d\cL_\cA$ is a mixed (not pure) Hodge
module. This would require additional input beyond Saito's weight
shift, potentially from the specific geometry of FM strata
(e.g.\ the Arnol'd--Stokes relations on codimension-$2$ corners).
The verification for the standard landscape
(Theorem~\ref{thm:semisimple-purity} plus the Virasoro
counterexample of Proposition~\ref{prop:purity-failure-virasoro})
leaves the non-semisimple frontier as the precise remaining gap.
\end{remark}

\begin{remark}[Sharp boundary of the purity criterion:
collision purity versus Koszulness]
\label{rem:purity-sharp-boundary}
\index{purity!sharp boundary|textbf}
\index{Koszulness!purity criterion!sharp boundary|textbf}
\index{collision purity!distinguished from Koszulness}
\index{monodromy weight filtration!role in purity criterion}
\index{shadow depth!purity criterion correlation}
The preceding analysis reveals that \emph{collision purity}
(Definition~\ref{def:rational-smooth-collision}: the cohomology
$H^*(\FM_k, \mathrm{gr}^d\cL_\cA)$ is pure of weight~$d$)
is strictly stronger than chiral Koszulness
(bar cohomology concentrated). The precise relationship is:

\smallskip\noindent
\textbf{Collision purity $=$ Koszulness $+$ Swiss-cheese
formality.}
\smallskip

Collision purity implies $E_1$-degeneration of the weight spectral
sequence, which gives both bar concentration (Koszulness) \emph{and}
vanishing of all higher $\Ainf$ operations $m_k = 0$ for $k \ge 3$
(formality). For chirally Koszul algebras with non-formal
Swiss-cheese structure (class~$\mathbf{M}$), the spectral
sequence has non-trivial differentials~$d_r$ for $r \ge 2$ that
cancel perfectly (the total cohomology is still concentrated)
but the intermediate graded pieces carry non-trivial mixed Hodge
structure. This is what happens for the Virasoro algebra: bar
concentration holds (it is Koszul), but the weight spectral sequence
does \emph{not} degenerate at~$E_1$, and the non-trivial $d_2$
differential is the $m_3$ operation.

\smallskip
The shadow depth classification correlates precisely with the purity
boundary:
\begin{center}
\renewcommand{\arraystretch}{1.25}
\begin{tabular}{@{}ccccc@{}}
\toprule
\textbf{Class} & $r_{\max}$ & \textbf{Eff.\ pole} &
 \textbf{Monodromy} & \textbf{Collision purity} \\
\midrule
$\mathbf{G}$ & $2$ & scalar $\leq 2$ & trivial &
 $\checkmark$ \\
$\mathbf{L}$ & $3$ & non-scalar $\leq 2$ & semisimple &
 $\checkmark$ \\
$\mathbf{C}$ & $4$ & $\leq 1$ (simple poles) & semisimple &
 $\checkmark$ \\
$\mathbf{M}$ & $\infty$ & non-scalar $\geq 4$ &
 non-semisimple & $\times$ \\
\bottomrule
\end{tabular}
\end{center}
For classes $\mathbf{G}$, $\mathbf{L}$, $\mathbf{C}$, the effective
OPE residues are semisimple (Theorem~\ref{thm:semisimple-purity}
applies), so collision purity holds and is equivalent to
Koszulness. For class~$\mathbf{M}$, the effective monodromy is
non-semisimple: the $L_{-1}$-descendant relation linking the
quartic and quadratic poles creates a non-split extension in the
category of mixed Hodge modules. The monodromy weight filtration
$W_\bullet(\psi_f \cL_\cA)$ acquires a non-trivial nilpotent part
$N = \log T_u \neq 0$, so Saito's weight-shift
formula~\eqref{eq:saito-weight-shift} acquires a correction from
the monodromy weight filtration, and the Step~3 argument fails.

\smallskip
One may ask: is semisimplicity of the effective monodromy
\emph{necessary} for collision purity, or merely sufficient? The
answer for the standard landscape is that it is \emph{sharp}: the
non-semisimple effective monodromy of class~$\mathbf{M}$ algebras
produces a non-trivial $d_2$ differential
(Proposition~\ref{prop:purity-failure-virasoro}), which obstructs
$E_1$-degeneration, hence obstructs collision purity. More
precisely, the obstruction class in
$\mathrm{Ext}^1_{\mathrm{MHM}}(\mathrm{gr}^4\cL, \mathrm{gr}^2\cL)$
is non-zero if and only if the quartic pole is non-scalar
(i.e.\ the field $T$ appearing in the quadratic pole is linked to
the scalar $c/2$ in the quartic pole by conformal descent). At
$c = 0$, the quartic pole vanishes and the effective pole order drops
to~$2$ (non-scalar residue $2T$); Virasoro at $c = 0$ is in
class~$\mathbf{L}$, and collision purity holds. This is consistent:
at $c = 0$, the curvature $\kappa = c/2 = 0$ vanishes, and the
extension class in $\mathrm{Ext}^1_{\mathrm{MHM}}$ is trivial.

\smallskip
Two distinct notions of purity appear in the Koszul dodecahedron
and must not be conflated:
\begin{enumerate}[label=\textup{(\alph*)}]
\item \emph{Collision purity}
 (Definition~\ref{def:rational-smooth-collision}): each
 $\mathrm{gr}^d\cL_\cA$ is pure. This is $E_1$-degeneration
 of the weight spectral sequence. Equivalent to Koszulness
 $+$ formality.
\item \emph{$\mathcal{D}$-module purity}
 (Theorem~\ref{thm:koszul-dodecahedron}(xii), Vol~I
 Conjecture~\ref*{V1-conj:d-module-purity-koszulness}):
 each $\barB_n(\cA)$ is pure of weight~$n$ as an MHM. This
 follows from Koszulness (direction (x)$\Rightarrow$(xii)
 in Vol~I), without any formality requirement. The converse
 (purity of $\barB_n$ implies Koszulness) is open.
\end{enumerate}
For semisimple effective monodromy, (a) and (b) coincide
(the monodromy weight filtration is trivial, so purity of the
total object and purity of the graded pieces are the same).
For non-semisimple monodromy, (b) is strictly weaker than (a):
the Virasoro algebra satisfies~(b) (it is Koszul, hence
$\barB_n$ is pure by K12) but fails~(a) (collision purity fails
by Proposition~\ref{prop:purity-failure-virasoro}).
\end{remark}

\begin{proposition}[Decomposition Theorem for FM compactification;
\ClaimStatusProvedHere]
\label{prop:decomposition-fm}
\index{Decomposition Theorem!FM compactification|textbf}
\index{Koszulness!Decomposition Theorem|textbf}
Let $\FM_k^{\mathrm{alg}}(\C)$ denote the \emph{algebraic}
Fulton--MacPherson compactification \textup{(}the iterated blowup of
$(\mathbb{P}^1)^k$ along thin diagonals,
Fulton--MacPherson~\textup{\cite{FM94}),} and let
$\pi_k\colon \FM_k^{\mathrm{alg}}(\C) \to \Conf_k(\C)$ be the
natural map to the open configuration space. This is a proper
morphism of smooth algebraic varieties. By the Decomposition
Theorem \textup{(}Beilinson--Bernstein--Deligne--Gabber
\textup{\cite{BBD82}),} the direct image of the intersection
cohomology complex decomposes:
\begin{equation}
\label{eq:decomposition-fm}
R\pi_{k*}\, \IC_{\FM_k^{\mathrm{alg}}(\C)}
\;\simeq\;
\bigoplus_{\alpha \in S_k}
\IC_{\overline{Y}_\alpha}(\cE_\alpha)[n_\alpha],
\end{equation}
where $\{Y_\alpha\}_{\alpha \in S_k}$ are the strata of a
stratification of $\Conf_k(\C)$ indexed by collision patterns,
$\cE_\alpha$ are semisimple local systems on~$Y_\alpha$, and
$n_\alpha \in \Z$ are cohomological shifts.

The OPE-weighted local system $\cL_\cA$
\textup{(}Definition~\textup{\ref{def:rational-smooth-collision})}
is a twisted coefficient system on $\Conf_k(\C)$. The bar
cohomology $H^{p,q}(\barB(\cA))$ is computed by the
cohomology $H^*(\FM_k^{\mathrm{alg}}(\C),\, \pi_k^*\cL_\cA)$ of the
pullback. By the projection formula,
\begin{equation}
\label{eq:projection-formula-fm}
H^*\bigl(\FM_k^{\mathrm{alg}}(\C),\, \pi_k^*\cL_\cA\bigr)
\;\simeq\;
H^*\bigl(\Conf_k(\C),\,
R\pi_{k*}\,\IC_{\FM_k^{\mathrm{alg}}} \otimes \cL_\cA\bigr)
\;\simeq\;
\bigoplus_\alpha
H^*\bigl(\overline{Y}_\alpha,\,
\IC_{\overline{Y}_\alpha}(\cE_\alpha \otimes \cL_\cA|_{Y_\alpha})
\bigr)[n_\alpha].
\end{equation}

When the IC summands $\IC_{\overline{Y}_\alpha}(\cE_\alpha)$ are
each pure of a single weight \textup{(}in the sense of mixed Hodge
theory\textup{),} each summand in~\eqref{eq:projection-formula-fm}
contributes to a single row of the weight spectral sequence.
The weight spectral sequence then degenerates at~$E_1$, the bar
cohomology $H^{p,q}(\barB(\cA))$ is concentrated along the
diagonal $p + q = $ bar degree, and~$\cA$ is chirally Koszul.

Conversely, when the IC summands are mixed (carrying contributions
of multiple weights), extensions between different weight pieces
produce non-trivial differentials $d_r$ for $r \ge 2$ in the weight
spectral sequence. The bar cohomology spreads off the diagonal:
$H^{p,q}(\barB(\cA)) \neq 0$ for $q \neq 0$, and
the higher $\Ainf$ operations $m_k$ for $k \geq 3$ are
non-vanishing. The Koszul boundary is the boundary between purity
and mixedness of the IC decomposition.
\end{proposition}

\begin{proof}
A clarification on categories is needed before applying BBD.
The real Fulton--MacPherson space $\FM_k(\C)$ is a smooth manifold
with corners (a real-oriented blowup), not an algebraic variety.
The Decomposition Theorem of Beilinson--Bernstein--Deligne--Gabber
\cite{BBD82} applies to proper maps of algebraic varieties; in
the analytic setting, M.~Saito's theory of mixed Hodge modules
\cite{Saito90} extends the decomposition to proper morphisms of
complex analytic spaces. We therefore work with the \emph{algebraic}
Fulton--MacPherson compactification $\FM_k^{\mathrm{alg}}(\C)$,
which is a smooth algebraic variety: it is the iterated blowup of
$(\mathbb{P}^1)^k$ along the thin diagonals
(Fulton--MacPherson~\cite{FM94}). The forgetful map
$\pi_k\colon \FM_k^{\mathrm{alg}}(\C) \to \Conf_k(\C)$ is a proper
morphism of algebraic varieties, and the Decomposition Theorem
applies to give~\eqref{eq:decomposition-fm}. The real manifold with
corners $\FM_k(\C)$ is the real-oriented blowup of
$\FM_k^{\mathrm{alg}}(\C)$; the weight spectral sequence and purity
data are pulled back from the algebraic setting, where they are
defined.

A second categorical point concerns the OPE-weighted local
system~$\cL_\cA$. The connection
$\nabla = d - \sum_{i<j}\Omega_{ij}\,d\log(z_i - z_j)$
is an \emph{algebraic} flat connection on the trivial bundle
$\cA^{\otimes k} \otimes \cO_{\Conf_k(\C)}$: the one-forms
$d\log(z_i - z_j)$ are algebraic (rational with simple poles along
the boundary divisors of $\FM_k^{\mathrm{alg}}$), and the residue
endomorphisms~$\Omega_{ij}$ are constant. This connection has
logarithmic poles along the normal-crossing boundary divisor
$D = \bigcup_{i<j}D_{ij}$, hence is \emph{regular singular} in
the sense of Deligne~\cite{Deligne70}. By the Riemann--Hilbert
correspondence, $\cL_\cA$ corresponds to a regular holonomic
algebraic $\cD$-module on~$\Conf_k(\C)$. The collision filtration
$\cL^{\leq d}_\cA$ of Definition~\ref{def:rational-smooth-collision}
coincides with the Malgrange--Kashiwara $V$-filtration
of this $\cD$-module along~$D$, indexed by pole order. In Saito's
theory of mixed Hodge modules \cite{Saito90}, the $V$-filtration
carries the weight filtration; the associated graded
$\mathrm{gr}^d\cL_\cA$ is therefore a well-defined object of the
abelian category of mixed Hodge modules on the boundary stratum,
with weight determined by the pole order~$d$.
Note that the structure constants of~$\cA$ (the matrix entries
of~$\Omega_{ij}$) may be transcendental; this does not affect
algebraicity of the connection, which depends only on
the algebraic nature of the $d\log$ one-forms and the constancy
of the residues.

The identification of bar cohomology with the cohomology of
$\pi_k^*\cL_\cA$ on $\FM_k(\C)$ is the content of the bar
complex construction
(Definition~\ref{def:chiral_bar}): the bar differential
$d_{\barB}$ is the connection $\nabla$ of the OPE local system
composed with the FM boundary residues. The projection formula
gives~\eqref{eq:projection-formula-fm}.

For the purity implication: since $\cL_\cA$ underlies a mixed
Hodge module (via the preceding paragraph), the tensor product
$\cE_\alpha \otimes \cL_\cA|_{Y_\alpha}$ is again an object of
the category of mixed Hodge modules, and weights are additive
under tensor product in that category. Therefore, when each
$\IC_{\overline{Y}_\alpha}(\cE_\alpha)$ is pure of weight
$w_\alpha$, the summand
$H^n(\overline{Y}_\alpha, \IC(\cE_\alpha \otimes \cL_\cA))$
sits in weight $w_\alpha + d$ (where $d$ is the pole order of
$\cL_\cA$ restricted to~$Y_\alpha$, i.e.\ the $V$-filtration
index). Different summands with
different weights lie in different rows of the weight spectral
sequence. Since the differentials $d_r$ (for $r \geq 1$)
connect rows differing in weight by~$r$, purity (each row
carrying a single weight) forces $d_r = 0$ for all $r \geq 1$
on the off-diagonal terms, giving $E_1$-degeneration and hence
concentration.

For the converse: mixedness means some
$\IC_{\overline{Y}_\alpha}(\cE_\alpha)$ carries an extension
between pieces of weight $w$ and $w + r$. This extension maps,
under the weight spectral sequence, to a non-trivial $d_r$.
Explicitly, the extension class in
$\mathrm{Ext}^1_{\mathrm{MHS}}(H^{w+r}, H^w)$ provides the
differential: this is the standard mechanism by which
non-split mixed Hodge structures produce spectral sequence
differentials (Deligne's theory of weights). The resulting
off-diagonal bar cohomology classes are precisely the higher
$\Ainf$ operations (Proposition~\ref{prop:purity-failure-virasoro}
gives the explicit instance for the Virasoro algebra).
\end{proof}

\begin{remark}[The BGS parallel: from Schubert to FM]
\label{rem:bgs-parallel}
\index{Beilinson--Ginzburg--Soergel!chiral analog}
\index{category O@category $\cO$!Koszulness}
\index{Schubert variety!FM collision stratum analog}
In Beilinson--Ginzburg--Soergel \cite{BGS96}, Koszulness of
category~$\cO$ is proved by applying the Decomposition Theorem to
the proper map $\pi\colon G/B \to G/P$ (the projection from the
full flag variety to a partial flag variety). The IC complexes
$\IC_{\overline{C}_w}$ on Schubert cells $C_w \subset G/B$ are
pure because Schubert varieties are rationally smooth. Purity of
each IC summand in $R\pi_*\IC_{G/B}$ forces the weight spectral
sequence to degenerate, and the Ext algebra
$\mathrm{Ext}^*_{\cO}(L, L)$ is concentrated on the diagonal; this
is Koszulness.

Proposition~\ref{prop:decomposition-fm} is the chiral analog.
The structural dictionary is:

\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{@{}ll@{}}
\toprule
\textbf{BGS / category $\cO$} & \textbf{Chiral bar complex} \\
\midrule
Flag variety $G/B$ & FM compactification $\FM_k(\C)$ \\
Partial flag $G/P$ & Configuration space $\Conf_k(\C)$ \\
Proper map $G/B \to G/P$ & FM forgetful $\pi_k\colon \FM_k \to \Conf_k$ \\
Schubert cell $C_w$ & FM collision stratum $Y_\alpha$ \\
IC on Schubert variety & IC on collision closure $\overline{Y}_\alpha$ \\
Rational smoothness of $\overline{C}_w$ & Smoothness of FM boundary divisors \\
Purity of IC $\Rightarrow$ Koszul & Purity of IC $\Rightarrow$ chirally Koszul \\
Kazhdan--Lusztig polynomials & Bar cohomology Poincaré series \\
\bottomrule
\end{tabular}
\end{center}

The analogy is not merely structural. In both cases, the mechanism
is the same: purity of intersection cohomology on the strata of a
proper map forces the weight spectral sequence to collapse, and the
resulting diagonal concentration \emph{is} Koszulness. The
geometric input differs (rational smoothness of Schubert varieties
versus the OPE pole structure on FM boundary divisors), but the
homological algebra is identical. The simple-pole principle
(Remark~\ref{rem:simple-pole-principle}) is the chiral translation
of rational smoothness: simple OPE poles produce smooth collision
strata, just as simple reflections produce smooth Schubert divisors.
Higher-order non-scalar poles produce singular collision data, just
as longer Weyl group elements produce singular Schubert varieties
(though the Schubert case is saved by rational smoothness, which has
no automatic chiral counterpart; this is why chiral Koszulness is
\emph{harder} than category-$\cO$ Koszulness).
\end{remark}

\begin{remark}[Closed-color purity and open-color purity]
\label{rem:closed-open-purity}
The purity criterion above governs the \emph{closed color} of
$\SCchtop$: bar cohomology concentration is IC purity on $\FM_k(\C)$.
The complete strictification theorem
(Theorem~\ref{thm:complete-strictification}) is its
\emph{open-color} analogue: the spectral Drinfeld class vanishes
because root multiplicity one ($\dim\fg_\alpha=1$) forces the
multilinear Lie space to be one-dimensional at each filtration level.

The two purity statements have different character.
Closed-color collision purity depends on the \emph{specific}
chiral algebra $\cA$: Kac--Moody is collision-pure, Virasoro is
not (though both are chirally Koszul;
Remark~\ref{rem:purity-sharp-boundary}).
Open-color purity (strictification) depends only on the
\emph{root system} of $\fg$: every simple Lie algebra is pure.
The asymmetry reflects the Swiss-cheese directionality: the open
color is controlled by $E_1$ (topology), which sees only the
ordering on $\R$, while the closed color is controlled by
$\FM_k(\C)$ (holomorphic geometry), which sees the full OPE
pole structure.

In the Steinberg analogy: purity of IC complexes on $G/B$ makes the
Hecke algebra arise from the Springer resolution. Purity of root
spaces makes strictification of the spectral quasi-factorization
package inevitable once the dg-shifted Yangian is expressed in that
language. Both are geometric inputs that control algebraic
outputs.
\end{remark}

\subsection{Koszulness as Lagrangian concentration}
\label{subsec:koszul-lagrangian-concentration}

\providecommand{\Steinb}{\mathfrak{S}}
\providecommand{\Mvac}{\mathcal{M}_{\mathrm{vac}}}

The purity criterion attacks chiral Koszulness via the
Decomposition Theorem on FM boundary strata. But purity is a
\emph{consequence} of a deeper geometric fact: Koszulness is the
condition that a Lagrangian meets itself cleanly. This subsection
closes the circuit between the purity criterion
\textup{(}Section~\textup{\ref{subsec:purity-koszulness}}\textup{)} and
the foundational formal identification of the bar complex with the
Lagrangian self-intersection model
\textup{(}Theorem~\textup{\ref{thm:bar-is-self-intersection}}\textup{)}.

\begin{conjecture}[Koszulness and the expected Lagrangian concentration;
\ClaimStatusConjectured]
\label{conj:koszul-lagrangian-concentration}
\index{Koszulness!Lagrangian concentration|textbf}
\index{Lagrangian!concentration!Koszulness|textbf}
Let $\cA$ be a chiral algebra on a smooth curve~$X$, let
$\Mvac$ be the $(-2)$-shifted symplectic vacuum moduli stack of
the associated 3d HT theory, and let
$\cL_b \hookrightarrow \Mvac$ be the Lagrangian boundary condition.
Write $\Steinb_b = \cL_b \times_{\Mvac}^h \cL_b$ for the derived
Lagrangian self-intersection. Then the following are expected to be
equivalent on the geometric surface where the global
self-intersection is governed by the same formal bar/Koszul model as
in Theorem~\ref{thm:bar-is-self-intersection}:
\begin{enumerate}[label=\textup{(\roman*)}]
\item $\cA$ is chirally Koszul
 \textup{(}Definition~\textup{\ref{def:chiral-koszul-geometric})};
\item the Borel--Moore homology $H_*^{\mathrm{BM}}(\Steinb_b)$ is
 concentrated in a single degree.
\end{enumerate}
\end{conjecture}

\begin{remark}[Evidence]
This is a heuristic dictionary rather than a proved theorem at the
current level of the manuscript. Theorem~\ref{thm:bar-is-self-intersection}
supplies the standard formal bar/Koszul model for the derived tensor
product attached to $\Steinb_b$. On that formal surface,
concentration of $H^{p,q}(\barB(\cA))$ is exactly the clean-intersection
condition: no excess classes in the bar model means no higher
$\Ainf$ operations.

would require additional global geometry that is not supplied in this
chapter. The statement is therefore recorded only as the expected
geometric meaning of chiral Koszulness.
\end{remark}

\begin{remark}[The Koszul examples: smooth self-intersection]
\label{rem:koszul-examples-smooth-intersection}
\index{Lagrangian!self-intersection!examples}
On this heuristic geometric picture, the three Koszul examples of
Proposition~\ref{prop:purity-koszul-examples}, the Lagrangian
self-intersection $\Steinb_b$ is explicitly smooth:
\begin{itemize}
\item \textbf{Free $\beta\gamma$:} The vacuum moduli $\Mvac$
 is a $(-2)$-shifted cotangent bundle $T^*[-2]V$ for a vector
 space~$V$. The zero section $\cL_b = V$ is a linear Lagrangian,
 and $\Steinb_b = V \times_{T^*[-2]V}^h V \simeq T^*[-1]V$
 is a $(-1)$-shifted cotangent bundle, smooth. Bar cohomology
 is in degree~$1$: the simple pole $\beta(z)\gamma(w) \sim
 (z-w)^{-1}$ gives a single intersection generator.

\item \textbf{Landau--Ginzburg:} The superpotential~$W$ deforms the
 Lagrangian by a closed $1$-form $dW$. The self-intersection
 $\Steinb_b$ is the derived critical locus of~$W$ on
 $T^*[-1]V$, which is smooth when $W$ has isolated
 singularities. First-order OPE poles ensure clean intersection.

\item \textbf{Abelian Chern--Simons:} The scalar double pole
 $J(z)J(w) \sim k(z-w)^{-2}$ contributes to curvature
 $m_0 = \kappa(\cA)$, not to the bar differential. The
 \emph{effective} intersection data (the non-scalar OPE)
 has at most simple poles: $\Steinb_b$ is smooth after
 curving.
\end{itemize}

For the Virasoro algebra at generic central charge ($c \neq 0$),
by contrast, the fourth-order pole
creates an \emph{excess} self-intersection. The composite
residue $(z-w)^{-4} \cdot \frac{c}{2} \leadsto
\frac{1}{3!}\partial^3_w[2T(w)]$ creates a non-trivial class in
$H^{2,1}(\barB(\mathrm{Vir}))$: the derived self-intersection
is not clean, the excess bundle $\mathcal{E}$ has rank~$> 0$,
and the BM homology of $\Steinb_b$ spreads into an additional
degree. This is the expected geometric content of
Proposition~\ref{prop:purity-failure-virasoro}: the
higher-pole residue that produces the $m_3$ operation \emph{is}
the excess intersection class.
\end{remark}

\begin{remark}[The étale/non-étale dichotomy and the Koszul locus]
\label{rem:etale-dichotomy}
\index{Koszul locus!etale dichotomy@\'etale dichotomy}
Heuristically, the étale groupoid picture sharpens the geometric
meaning of the
Koszul locus. In the moduli of chiral algebras, the Koszul locus
is precisely where the self-intersection groupoid is étale.
Crossing the \emph{formality} boundary (not the Koszul boundary),
as when Drinfeld--Sokolov reduction introduces higher Swiss-cheese
operations while preserving bar concentration
(Theorem~\ref{thm:ds-koszul-obstruction}), is the transition from
étale to non-étale: the source map acquires non-trivial derived
fibers. The Steinberg analogy is instructive: the Springer
resolution $\widetilde{\cN} \to \cN$ is an isomorphism over the
regular nilpotent locus (étale fibers) and has non-trivial fibers
over the singular locus (the Springer fiber). The bar complex
$\barB(\cA)$ has trivial transferred operations ($m_k = 0$ for
$k \ge 3$) over the \emph{formal} locus and non-trivial $\Ainf$
operations over the non-formal locus. Both loci lie within the
Koszul locus: the $\cW$-algebras are chirally Koszul but not
Swiss-cheese formal.
\end{remark}

\begin{remark}[Spectral parameter: operadic origin]
\label{rem:spectral-parameter-operadic}

The spectral parameter $z$ in the $R$-matrix $r(z)$ is the
coordinate on $\FM_2(\C)$ in the closed-color factor of
$\SCchtop$. The $E_1$ structure sees only the ordering on
$\R_t$; the chiral structure sees the holomorphic separation

\[
r(z)
\;=\;
\int_0^\infty e^{-\lambda z}\,
\{\cdot\,{}_\lambda\,\cdot\}\, d\lambda
\]
(Proposition~\ref{prop:field-theory-r}) is the spectral duality
of holomorphic-topological theory: it converts the closed-color
$\lambda$-bracket (singularities in OPE) into the open-color
$R$-matrix (braiding data).
\end{remark}

\begin{proposition}[Two-color canonicity;
\ClaimStatusProvedHere]
\label{prop:two-color-canonicity}
\index{two-color Koszul duality!canonicity}
Given holomorphic factorization data (the bar differential
$d_{\barB}$ on $\FM_k(X)$), the topological factorization
structure (the deconcatenation coproduct $\Delta$ on
$\Conf_k^{<}(\R)$) is uniquely determined by the cofree
conilpotent coalgebra structure on $T^c(s^{-1}\bar{\cA})$
and the coderivation property of $d_{\barB}$. The spectral parameter enters
because the holomorphic propagator creates meromorphic
$z$-dependence in the bar differential; the $R$-matrix is
the systematic extraction of this meromorphic dependence into
an equivariant structure datum for the $E_1$-composition.
\end{proposition}

\begin{proof}
The deconcatenation coproduct $\Delta$ is the unique
$E_1$-coalgebra structure on the bar complex
$\barB(\cA) = T^c(s^{-1}\bar{\cA})$ compatible with the
cofree conilpotent coalgebra structure; this is the standard
result that $T^c(V)$ carries a unique conilpotent coproduct
(Loday--Vallette, Proposition~1.2.6). The
dg coalgebra condition requires that $d_{\barB}$
be a coderivation of $\Delta$, i.e.\ that the bar differential
respect the $E_1$-coalgebra structure. This is the defining
property of an $E_1$ dg coassociative coalgebra
(Theorem~\ref{thm:bar-swiss-cheese}): the $A_\infty$ operations
(the $m_k$ extracted from $d_{\barB}$) are compatible with
the deconcatenation coproduct ($\Delta$). The holomorphic propagator on
$\FM_2(\C)$ creates the meromorphic $z$-dependence; the
$R$-matrix extracts this dependence as an intertwiner
$R(z) \colon \Delta \to \Delta^{\mathrm{op}}$ of the
$E_1$-coproduct.
\end{proof}

\subsection{The genus tower: curved $\R$-factorization}
\label{subsec:genus-tower-R}

At genus~$0$: $d_{\barB}^2 = 0$, $\Delta$ is
coassociative, $\R$-factorization is exact. At genus $g \geq 1$,
the propagator acquires monodromy from the period matrix
of~$\Sigma_g$, and the fiberwise bar differential becomes curved:
\begin{equation}\label{eq:curved-R-fact}
\dfib^{\,2} = \kappa(\cA) \cdot \omega_g,
\end{equation}
where $\kappa(\cA)$ is the modular characteristic and $\omega_g$ is
the Arakelov $(1,1)$-form on $\Sigma_g$.

\begin{proposition}[Curved $\R$-factorization at higher genus;
\ClaimStatusProvedHere]
\label{prop:curved-R-factorization}
Let $\cA$ be a chiral algebra on a genus-$g$ curve $\Sigma_g$.
The genus-$g$ bar complex $\barB^{(g)}(\cA)$ satisfies:
\begin{enumerate}[label=\textup{(\roman*)}]
\item The coproduct $\Delta$ is coassociative
 (independent of genus: it is the tensor coalgebra structure
 and depends on no propagator data).
\item The fibrewise differential $\dfib$ on $\Sigma_g$ has
 curvature
 $\dfib^{\,2} = \kappa(\cA) \cdot \omega_g \in
 H^{2}(\overline{\mathcal{M}}_g)$, where $\omega_g$ is the
 Hodge $\kappa$-class and $\kappa(\cA)$ is the modular
 characteristic of Volume~I, Theorem~D.
\item $\dfib$ is \emph{not} a coderivation of $\Delta$:
 applied at an interior splitting of a length-$n$ bar element,
 $\dfib^{\,2} = \kappa(\cA)\cdot\omega_g$ produces a factor-$2$
 discrepancy between the two sides of the co-Leibniz identity,
 because $\omega_g$ contracts the closed-colour collision
 class against the same $\bR$-interval on both branches.
 This is Remark~\ref{V1-rem:sc-higher-genus}\textup{(ii)} of
 Volume~I, specialized to the universal bar coalgebra.
\item The period-corrected differential
 \[
 \Dg{g}
 \;=\;
 \dfib \;+\;
 \sum_{\alpha,\beta}
 (\operatorname{Im}\Omega)^{-1}_{\alpha\beta}
 \oint_{A_\alpha} \Phi_{\alpha\beta},
 \]
 whose period kernels $\Phi_{\alpha\beta}$ are built from the
 Fay trisecant identity, satisfies
 $\Dg{g}^{\,2} = 0$ \emph{and} is a genuine coderivation of
 $\Delta$. The triple $(\barB^{(g)}(\cA), \Dg{g}, \Delta)$ is
 the $E_1$ dg coassociative coalgebra at genus~$g$
 (Volume~I, Remark~\ref{V1-rem:sc-higher-genus}\textup{(iii)}).
\end{enumerate}
\end{proposition}

\begin{proof}
Part~(i) is algebraic: the deconcatenation coproduct on
$T^c(s^{-1}\bar{\cA})$ is coassociative by definition.
Part~(ii) is Volume~I, Theorem~D, applied to the universal
genus-$g$ bar complex: the period matrix of $\Sigma_g$
contributes the Hodge curvature class to the double residue
of the prime-form propagator.
Part~(iii) is the negative co-Leibniz check: the curvature
$\kappa(\cA)\cdot\omega_g$ acts on the bar element by a class
pulled back from $\overline{\mathcal{M}}_g$, while the
co-Leibniz identity $\Delta\circ\dfib^{\,2} =
(\dfib\otimes\id+\id\otimes\dfib)^{\,2}\circ\Delta$ would
require this class to split additively across an interior
splitting of $\bR$. It does not: $\omega_g$ contracts both
branches against the same Hodge fibre and the right-hand side
is twice the left. This is the genus-$g$ analogue of the
Heisenberg specialization
Theorem~\ref{thm:rosetta-curved}\textup{(c)}.
Part~(iv): by Volume~I's
Theorem~\ref{V1-thm:quantum-diff-squares-zero}, the Fay
trisecant identity provides period kernels
$\Phi_{\alpha\beta}$ on the universal curve such that the
sum $\Dg{g} = \dfib + \delta_{\mathrm{per}}$ satisfies both
$\Dg{g}^{\,2} = 0$ and the Leibniz identity for $\Delta$. The
factor-$2$ discrepancy from~(iii) is exactly the cohomological
obstruction that the Fay correction $\delta_{\mathrm{per}}$
cancels at every interior splitting; a single algebraic
identity in $H^{1}(\Sigma_g)\otimes H^{1}(\Sigma_g)$ controls
both square-zero and the coderivation property.
\end{proof}

\begin{remark}[Physical interpretation]
The curvature $\kappa(\cA) \cdot \omega_g$ is the conformal
anomaly of the 3d HT theory compactified on~$\Sigma_g$. At genus
$0$, there is no anomaly ($\omega_0 = 0$) and the
$\R$-factorization is exact. At genus $g \geq 1$, the anomaly
deforms the $\Eone$ structure into a curved $\Ainf$ structure
whose curvature is controlled by the Hodge bundle
$\mathbb{E} \to \overline{\mathcal{M}}_g$. Volume~I's
complementarity theorem (Theorem~C) constrains the curvatures:
$\kappa(\cA) + \kappa(\cA^!) = \varrho_\cA \, K(\cA)$, where
$K(\cA) = c(\cA) + c(\cA^!)$ is the Trinity conductor of
Vol~I \cref{V1-chap:universal-conductor} (Vir $K = 26$;
$\cW_3$ $K = 100$; affine $V_k(\fg)$ $K = 2\dim(\fg)$) and
$\varrho_\cA$ is the family-specific anomaly ratio
($\varrho_{\mathrm{Vir}} = 1/2$, $\varrho_{\cW_N} = H_N - 1$,
$\varrho_{\mathrm{KM}} = 0$). For the affine lineage (Kac--Moody,
free fields, lattice VOAs, $bc/\beta\gamma$ systems), the scalar
$\kappa$-sum vanishes and the anomalies cancel outright; for
Drinfeld--Sokolov reductions,
$\kappa(\mathrm{Vir}_c) + \kappa(\mathrm{Vir}_{26-c}) = 13$
(Virasoro, self-dual involution $c \mapsto 26 - c$, fixed point
$c = 13$) and $\kappa(\cW_3) + \kappa(\cW_3^!) = 250/3$ for $\cW_3$.
The
generating function for the genus tower is
the $\hat{A}$-genus (Volume~I, Theorem~D\textsubscript{scal}).
\end{remark}

\begin{theorem}[Curved $E_1$ dg coalgebras at higher genus;
\ClaimStatusProvedHere]
\label{thm:curved-swiss-cheese}
At genus $g \geq 1$, the bar complex
$\barB^{(g)}(\cA)$ supports an $E_1$ dg coassociative coalgebra structure
under the period-corrected differential, not under the bare
fibrewise differential:
\begin{enumerate}[label=\textup{(\roman*)}]
\item The fibrewise differential $\dfib$ has curvature
 $\dfib^{\,2} = \kappa(\cA) \cdot \omega_g \neq 0$ in
 $H^{2}(\overline{\mathcal{M}}_g)$, but is \emph{not} a
 coderivation of~$\Delta$
 (Proposition~\ref{prop:curved-R-factorization}\textup{(iii)}).
 The naive triple
 $(\barB^{(g)}(\cA), \dfib, \Delta)$ is therefore not a
 dg coalgebra at higher genus: the coderivation property fails.
\item The period-corrected differential
 $\Dg{g} = \dfib + \delta_{\mathrm{per}}$, where
 $\delta_{\mathrm{per}}$ is built from the Fay trisecant
 identity, satisfies $\Dg{g}^{\,2} = 0$ \emph{and} is a
 genuine coderivation of~$\Delta$. The $E_1$ dg coassociative
 coalgebra structure at genus~$g$ is therefore carried by
 $(\barB^{(g)}(\cA), \Dg{g}, \Delta)$
 (Volume~I, Remark~\ref{V1-rem:sc-higher-genus}\textup{(iii)}).
\item For a Koszul pair $(\cA, \cA^!)$ in the affine lineage,
 the curvatures satisfy $\kappa(\cA) + \kappa(\cA^!) = 0$
 (Volume~I, Theorem~C, Feigin--Frenkel involution); for
 $\mathcal{W}$-algebras the sum
 $\kappa(\cA) + \kappa(\cA^!) = \rho_\cA \cdot K$ is in
 general nonzero (Volume~I, Theorem~D). In the affine
 lineage the residual fibrewise curvatures cancel on
 $\barB^{(g)}(\cA) \otimes \barB^{(g)}(\cA^!)$, while in the
 $\mathcal{W}$-lineage the tensor product retains a residual
 fibrewise curvature. In both cases the period-corrected
 differential supplies the genuine $E_1$ dg coalgebra
 structure on the tensor product.
\end{enumerate}
\end{theorem}

\begin{proof}
\textbf{Part~(i).}
The coproduct $\Delta$ is the deconcatenation on the tensor
coalgebra $T^c(s^{-1}\bar{\cA})$, which is coassociative by
construction and independent of genus
(Proposition~\ref{prop:curved-R-factorization}\textup{(i)}).
The curvature
$\dfib^{\,2} = \kappa(\cA)\cdot\omega_g$ is a class on
$\overline{\mathcal{M}}_g$
(Proposition~\ref{prop:curved-R-factorization}\textup{(ii)});
applied at an interior splitting of a length-$n$ bar element
it produces a factor-$2$ co-Leibniz discrepancy because
$\omega_g$ contracts the closed-colour collision against the
same $\bR$-interval on both branches
(Proposition~\ref{prop:curved-R-factorization}\textup{(iii)}).
The triple $(\barB^{(g)}(\cA), \dfib, \Delta)$ therefore fails
the dg coalgebra axioms (the coderivation property breaks),
even though $\Delta$ itself remains coassociative.

\textbf{Part~(ii).}
The corrected differential $\Dg{g} = \dfib + \delta_{\mathrm{per}}$
adds period corrections $\delta_{\mathrm{per}}$ along $A$-cycles
of $\Sigma_g$, constructed in Volume~I via the period matrix
$(\mathrm{Im}\,\Omega)^{-1}_{\alpha\beta}$ of the Hodge bundle.
By the Volume~I construction, $\Dg{g}^2 = 0$
\textup{(}the Fay trisecant identity ensures cancellation\textup{)},
and the same Fay identity cancels the factor-$2$ co-Leibniz
discrepancy of part~(i): the period correction
$\delta_{\mathrm{per}}$ contributes the missing summand at
every interior splitting and restores
$\Delta\circ\Dg{g} = (\Dg{g}\otimes\id+\id\otimes\Dg{g})\circ\Delta$.

Concretely, $\delta_{\mathrm{per}}$ is a sum over $A$-cycles
\[
\delta_{\mathrm{per}} = \sum_{\alpha=1}^{g}
 (\mathrm{Im}\,\Omega)^{-1}_{\alpha\beta}
 \oint_{A_\alpha} \kappa(\cA) \cdot (\text{insertion along }
 B_\beta);
\]
each summand is a linear operator on $T^c(s^{-1}\bar{\cA})$
that acts on a single tensor factor (the factor that the cycle
punctures), and a single-factor operation on a cofree coalgebra
is automatically a coderivation by the universal property of
cofree coalgebras
(Proposition~\ref{prop:R-canonical-vol2}\textup{(i)}). The
sum $\delta_{\mathrm{per}}$ therefore is a coderivation and
its co-Leibniz contribution at an interior splitting cancels
the factor-$2$ discrepancy from $\dfib^{\,2}$ exactly.
Hence $(\Dg{g}, \Delta)$ defines an uncurved
$E_1$ dg coassociative coalgebra over $\overline{\mathcal{M}}_g$, while
$\dfib$ alone does not.

\textbf{Part~(iii).}
For a Koszul pair, $\kappa(\cA) + \kappa(\cA^!) = K(\cA)$
(Volume~I, Theorem~C), where $K(\cA)$ is the complementarity
constant. The curvature of the tensor product is
$(\kappa(\cA) + \kappa(\cA^!)) \cdot \omega_g
= K(\cA) \cdot \omega_g$. When $K = 0$ (affine lineage), the
fiberwise differential is flat ($d^2 = 0$) on the tensor product,
$\Delta_{\cA} \otimes \Delta_{\cA^!}$ is the coproduct, and the
result is an uncurved $E_1$ dg coassociative coalgebra. When $K \neq 0$
(DS reductions), the tensor product retains residual curvature
$K(\cA) \cdot \omega_g$.
\end{proof}

\begin{corollary}[Genus-$1$ specialization; \ClaimStatusProvedHere]
\label{cor:genus1-curved-swiss-cheese}%
\label{cor:mc5-genus-one-bridge}%
At genus~$1$, the curved $E_1$ dg coalgebra of
