\section{Line Operators, Koszul Duality, and Spectral $R$-Matrices}
\label{sec:line_operators}
\label{sec:line-operators}

Line operators in 3d HT quantum field theories are identified with
modules for the open-colour Koszul dual algebra
$\mathcal{A}^!_{\mathrm{line}}$, and spectral $R$-matrices
controlling their OPE emerge from the bar-cobar adjunction.

\begin{remark}[$E_1$ primacy of line operators]
\label{rem:line-operators-e1-primacy}
Line operators are natively $E_1$ data: they are modules over the
\emph{ordered} (open-colour) Koszul dual
$\cA^!_{\mathrm{line}}$, and their composition is governed by
the deconcatenation coproduct on the ordered bar coalgebra,
not its symmetric-coinvariant quotient.
The spectral $R$-matrix $R(z)$ controlling line-operator OPE is
the arity-$2$ $E_1$ MC datum; the modular characteristic
$\kappa(\cA) = \mathrm{av}(r(z))$ is its
$\Sigma_2$-coinvariant projection.
See Volume~I, \S\textup{E$_1$ as primitive}
for the full statement.
\end{remark}

\begin{remark}[Open sector first, strict presentation second]
\label{rem:open-sector-strict-presentation}
\label{rem:yangian-logical-status}
\index{line operators!strict presentation}
The statement $\mathcal{C}_{\mathrm{line}}\simeq \mathcal{A}^!_{\mathrm{line}}\text{-mod}$
should always be read at the $A_\infty$-module level.  The dg-shifted
Yangian and related algebraic packages are chosen strict presentations
of this meromorphic module theory; their deformation and OPE constraints
are shadows of the same homotopy-coherent deformation object that
controls the boundary algebra
(Convention~\ref{conv:vol2-strict-models}).
\end{remark}

\section{Bar–cobar duality for $\mathsf{SC}^{\mathrm{ch,top}}$ and filtered Koszul duality}
\label{sec:bar-cobar}

\subsection{Colored cooperad and bar construction}
\label{subsec:bar}
Let $\mathsf{SC}^{\mathrm{ch,top}}$ be augmented in each color (units in $\FM_1(\C)$ and $E_1(1)$).
The (colored) bar construction $\mathbf B\mathsf{SC}^{\mathrm{ch,top}}$ is the cofree conilpotent cooperad on the desuspension of the augmentation ideal, with differential induced by partial compositions.

\begin{definition}[Bar construction]
\label{def:bar}
Set $\mathbf B := \mathbf B(\mathsf{SC}^{\mathrm{ch,top}})$; its underlying graded cooperad consists of trees labeled by operations in $\FM_\bullet(\C)$ and $E_1(\cdot)$ with desuspension shifts, and differential $d=d_{\mathrm{int}}+d_{\mathrm{Graft}}$.
\end{definition}

\subsection{Cobar construction and twisting morphisms}
\label{subsec:cobar}
Given a conilpotent colored cooperad $C$, the cobar $\Omega C$ is the free operad on the desuspension $s^{-1}\bar{C}$ with differential determined by the cooperad structure.
A twisting morphism $\tau: C\to \mathsf{End}_A$ determines (co)algebra structures and bar–cobar adjunctions.

\begin{theorem}[Homotopy--Koszulity of $\mathsf{SC}^{\mathrm{ch,top}}$;
\ClaimStatusProvedHere]
\label{thm:homotopy-Koszul}
The two--colored topological operad $\mathsf{SC}^{\mathrm{ch,top}}$
(Definition~\textup{\ref{def:SC-operations}}) is homotopy--Koszul: the canonical
map
\[
  \Omega\mathbf{B}(\mathsf{SC}^{\mathrm{ch,top}})
  \;\xrightarrow{\;\sim\;}
  \mathsf{SC}^{\mathrm{ch,top}}
\]
is a quasi--isomorphism of two--colored dg operads.
Equivalently, the unit and counit of the bar--cobar adjunction
$(\Omega,\mathbf B)$ for $\mathsf{SC}^{\mathrm{ch,top}}$--algebras are weak
equivalences.
\end{theorem}

\begin{proof}
\textbf{Step 1: The classical Swiss--cheese operad is homotopy--Koszul.}
The classical Swiss--cheese operad $\mathsf{SC}$ (with $E_2$ on the closed
color, $E_1$ on the open color, and mixed operations
$E_2(k) \times E_1(m)$) is Koszul as a two--colored operad: it is built
from the individually Koszul operads $E_2$ \cite{GK94} and
$\mathsf{Ass} = E_1$ \cite{GK94} via a distributive law governing the
Swiss--cheese interaction \cite{Vor99,Liv06,Val07}.  For a Koszul operad, the
inclusion of the Koszul dual cooperad into the bar construction is a
quasi--isomorphism \cite{GK94}, so the canonical counit
$\epsilon_{\mathsf{SC}} \colon \Omega\mathbf{B}(\mathsf{SC}) \to
\mathsf{SC}$ is a quasi--isomorphism; that is, $\mathsf{SC}$ is
homotopy--Koszul.

\textbf{Step 2: Quasi--isomorphism of colored operads.}
Kontsevich formality \cite{Kon03} provides a quasi--isomorphism of dg
operads $\phi \colon C_\ast(\FM_\bullet(\C)) \xrightarrow{\sim} E_2$,
where $E_2 = H_\ast(\FM_\bullet(\C))$ is viewed as a dg operad with
zero differential.  Define a map of two--colored dg operads
\[
  \Phi \colon \mathsf{SC}^{\mathrm{ch,top}} \longrightarrow \mathsf{SC}
\]
by setting $\Phi = \phi_k$ on closed operations
$C_\ast(\FM_k(\C)) \to E_2(k)$, $\Phi = \mathrm{id}$ on open operations
$E_1(m) \to E_1(m)$, and $\Phi = \phi_k \times \mathrm{id}$ on mixed
operations
$C_\ast(\FM_k(\C)) \times E_1(m) \to E_2(k) \times E_1(m)$.

We verify that $\Phi$ is a map of two--colored dg operads.  The claim is
that $\Phi$ intertwines all operadic composition maps, including the
mixed--color ones, and we must check this for each of the three
composition types in Definition~\ref{def:SC-operations}.

\smallskip\noindent
\emph{Closed-into-closed:}
Let $\circ_i^{\mathrm{ch}}$ denote the insertion of a closed operation
into the $i$-th slot of another closed operation.
On $\mathsf{SC}^{\mathrm{ch,top}}$, this is the FM operadic
insertion $\circ_i^{\FM}$ on
$C_\ast(\FM_k(\C)) \otimes C_\ast(\FM_l(\C)) \to C_\ast(\FM_{k+l-1}(\C))$.
Since $\phi$ is an operad map,
$\phi_{k+l-1} \circ \circ_i^{\FM} = \circ_i^{E_2} \circ (\phi_k \otimes \phi_l)$,
which is exactly $\Phi \circ \circ_i^{\mathrm{ch}} = \circ_i^{\mathrm{SC}} \circ (\Phi \otimes \Phi)$
on the closed color.

\smallskip\noindent
\emph{Open-into-open:}
The open composition is the $E_1$ operadic composition, and $\Phi$
acts as the identity on open operations, so compatibility is
immediate.

\smallskip\noindent
\emph{Mixed compositions (closed-into-mixed and open-into-mixed):}
By Definition~\ref{def:SC-operations}, the mixed operation spaces
decompose as products:
$\mathsf{SC}^{\mathrm{ch,top}}(\mathsf{ch}^k, \mathsf{top}^m; \mathsf{top})
= C_\ast(\FM_k(\C)) \times E_1(m)$,
and the composition maps decompose correspondingly.  Inserting a
closed operation into a mixed operation acts on the $\FM_k(\C)$
factor by the FM operadic insertion and on the $E_1(m)$ factor by
the identity (the topological ordering is unchanged by a holomorphic
insertion).  Similarly, inserting an open operation acts on $E_1(m)$
by the $E_1$ composition and on $\FM_k(\C)$ by the identity.
Formally, if $\gamma_{\mathrm{mix}} = \gamma^{\FM} \times \gamma^{E_1}$
denotes the mixed composition, then
\[
  \gamma_{\mathrm{mix}} = \gamma^{\FM} \times \mathrm{id}_{E_1}
  \quad\text{(closed insertion)},
  \qquad
  \gamma_{\mathrm{mix}} = \mathrm{id}_{\FM} \times \gamma^{E_1}
  \quad\text{(open insertion)}.
\]
Since $\Phi = \phi \times \mathrm{id}$ on mixed operations and
$\phi$ intertwines $\gamma^{\FM}$, we get
\[
  \Phi \circ \gamma_{\mathrm{mix}}
  = (\phi \circ \gamma^{\FM}) \times (\mathrm{id} \circ \mathrm{id}_{E_1})
  = (\gamma^{E_2} \circ (\phi \otimes \phi)) \times \mathrm{id}_{E_1}
  = \gamma_{\mathrm{mix}}^{\mathrm{SC}} \circ (\Phi \otimes \Phi),
\]
and likewise for open insertions.  The no-open-to-closed rule is
preserved trivially (both sides are empty).

\smallskip\noindent
\emph{Equivariance:}
The $\Sigma_k$-equivariance on the closed color and the identity on
the open color are preserved because $\phi$ is $\Sigma_k$-equivariant
and the product decomposition respects the symmetric group
action on the $\FM$-factor with the trivial action on $E_1$.

\smallskip
Thus $\Phi$ is a map of two--colored dg operads.  The map $\Phi$ is a
quasi--isomorphism in each arity and color profile: on the closed color by
formality, on the open color by identity, and on mixed operations by
K\"unneth (the Eilenberg--Zilber quasi-isomorphism, applicable because
$\FM_k(\C)$ is compact (the Fulton--MacPherson compactification,
a compact manifold with corners) and therefore has finite-dimensional
homology over the ground field of characteristic~$0$; the K\"unneth
isomorphism applies without completion hypotheses).

\textbf{Step 3: Transfer of homotopy--Koszulity.}
The bar functor $\mathbf{B}$ and cobar functor $\Omega$ for colored dg
operads preserve quasi--isomorphisms between operads whose underlying
collections are levelwise cofibrant \cite{BM03,GK94}; since we work over
a field of characteristic zero, this condition is automatic.  The
bar--cobar counit $\epsilon$ is natural, giving a commutative square
\[
\begin{tikzcd}
\Omega\mathbf{B}(\mathsf{SC}^{\mathrm{ch,top}}) \ar[r, "\epsilon"] \ar[d, "\Omega\mathbf{B}(\Phi)"'] & \mathsf{SC}^{\mathrm{ch,top}} \ar[d, "\Phi"] \\
\Omega\mathbf{B}(\mathsf{SC}) \ar[r, "\epsilon"'] & \mathsf{SC}
\end{tikzcd}
\]
in which $\Phi$ is a quasi--isomorphism (Step~2),
$\epsilon_{\mathsf{SC}}$ is a quasi--isomorphism (Step~1), and
$\Omega\mathbf{B}(\Phi)$ is a quasi--isomorphism (preservation).  By the
two--out--of--three property,
$\epsilon_{\mathsf{SC}^{\mathrm{ch,top}}}$ is a quasi--isomorphism.
\end{proof}

\begin{remark}[Three inputs to Theorem~\ref{thm:homotopy-Koszul}]
\label{rmk:hKoszul-evidence}
The proof assembles three known results:
\begin{enumerate}[label=(\roman*)]
\item \emph{Koszulity of the classical Swiss--cheese operad}
  \cite{Liv06}: the ordinary Swiss--cheese operad
  $\mathsf{SC}$ (with $E_2$ on the closed color and $E_1$ on the open color) is Koszul.
\item \emph{Formality of $\FM_\bullet(\mathbb C)$}
  \cite{Kon03}: the chain
  operad $C_\ast(\FM_\bullet(\mathbb C))$ is quasi--isomorphic (as a dg operad) to its
  homology $E_2$, so $\mathsf{SC}^{\mathrm{ch,top}}$ is related to
  $\mathsf{SC}$ by a quasi--isomorphism on the closed color.
  Homotopy--Koszulity is invariant under quasi--isomorphisms of operads
  \cite{Val07}, which gives the result.
\item \emph{Associated graded decomposition} (independent confirmation).
  By Proposition~\ref{prop:gr-chiral} below, the holomorphic weight filtration on
  $\mathsf{SC}^{\mathrm{ch,top}}$ yields
  $\gr\;\mathsf{SC}^{\mathrm{ch,top}}\cong \mathsf P_{\mathrm{ch}}\amalg E_1$,
  where each factor is independently Koszul (Francis--Gaitsgory \cite{FG12} for the chiral
  operad, classical for $E_1$).  This provides a second proof via spectral
  sequence degeneration from the Koszulity of the associated graded.
\end{enumerate}
\end{remark}

\begin{remark}[Homotopy-Koszulity and the two-level convention]
\label{rem:hkoszul-two-level}
Homotopy-Koszulity is exactly the statement that the bar-cobar
adjunction at the $L_\infty$ level ($\Convinf$) is an equivalence,
not merely a quasi-isomorphism at the strict level.
\end{remark}

\label{rmk:hKoszul-dependence}
With homotopy-Koszulity established
(Theorem~\ref{thm:homotopy-Koszul}) and the live algebraic input
packaged by Definition~\ref{def:log-SC-algebra}, the following hold
unconditionally for any logarithmic $\SCchtop$-algebra:
\begin{itemize}
\item Theorem~\ref{thm:bar-cobar-adjunction}:
  the bar--cobar adjunction is a Quillen equivalence;
\item Theorem~\ref{thm:filtered-koszul}:
  the filtered bar--cobar adjunction is a filtered Quillen equivalence.
\end{itemize}
On the chirally Koszul locus, Theorem~\ref{thm:lines_as_modules}
identifies
$\mathcal C_{\mathrm{line}}\simeq \mathcal A^!_{\mathrm{line}}\text{-}\mathbf{mod}$.
For the standard affine HT gauge realization satisfying

Theorem~\ref{thm:Koszul_dual_Yangian} identifies the open-colour

Physical realisations (gauge theories satisfying

\end{remark}

\begin{theorem}[Bar--cobar adjunction; \ClaimStatusProvedHere]
\label{thm:bar-cobar-adjunction}
There is a Quillen adjunction
\[
\Omega: \mathrm{Coalg}_{\mathbf B}\rightleftarrows \mathrm{Alg}_{\mathsf{SC}^{\mathrm{ch,top}}}:\mathbf B,
\]
and it is a Quillen equivalence: the unit
$A\to\Omega\mathbf B(A)$ and counit $\mathbf B\Omega(C)\to C$ are weak equivalences.
\end{theorem}

\begin{proof}
The bar--cobar adjunction for a two-colored dg operad is
constructed by the standard machinery of \cite{GK94,Val07}:
conilpotence of $\mathbf B(\mathsf{SC}^{\mathrm{ch,top}})$ and
cofibrant generation of
$\mathrm{Alg}_{\mathsf{SC}^{\mathrm{ch,top}}}$ give the Quillen
adjunction.  By Theorem~\ref{thm:homotopy-Koszul},
$\mathsf{SC}^{\mathrm{ch,top}}$ is homotopy-Koszul, so the counit
$\Omega\mathbf B(\mathsf{SC}^{\mathrm{ch,top}})\to
\mathsf{SC}^{\mathrm{ch,top}}$ is a quasi-isomorphism of dg operads.
It follows that the adjunction is a Quillen equivalence: the unit and
counit on algebras are weak equivalences.
\end{proof}

\subsection{Filtered comparison with chiral bar--cobar}
\label{subsec:filtered-compare}
Equip $\mathsf{SC}^{\mathrm{ch,top}}$ with a filtration by holomorphic weight (pole order on $\FM_\bullet(\C)$). Then $\mathrm{gr}$ reduces to the chiral operad on the closed color and $E_1$ on the open color (with trivial mixing).

\begin{proposition}[Associated graded; \ClaimStatusProvedHere]
\label{prop:gr-chiral}
$\mathrm{gr}\ \mathsf{SC}^{\mathrm{ch,top}} \cong \mathsf P_{\mathrm{ch}} \amalg E_1$ (free product of colored operads with no mixed operations). Consequently, $\mathrm{gr}$ of bar/cobar reduces to BD chiral bar--cobar on the closed color and $A_\infty$/associative bar--cobar (open color).
\end{proposition}

\begin{proof}
\textbf{Step 1: Definition of the holomorphic weight filtration.}
For each arity profile, define $F^p \mathsf{SC}^{\mathrm{ch,top}}((\mathsf{ch}^k, \mathsf{top}^m); \mathsf{top})$ as the subspace of chains on $\FM_k(\C) \times E_1(m)$ whose form representatives have total pole order $\le p$ along the union of collision divisors $\bigcup_{|S| \ge 2} D_S \subset \FM_k(\C)$. Explicitly, near $D_S$ with local radial coordinate $\varepsilon_S$, a form of pole order $p$ at $D_S$ has a factor $\varepsilon_S^{-p}$ (or equivalently, $p$ copies of $d\log\varepsilon_S$). The total pole order is the sum over all divisors. For the pure closed operations $\mathsf{SC}^{\mathrm{ch,top}}(\mathsf{ch}^k; \mathsf{ch}) = \FM_k(\C)$, the filtration is defined identically.

\textbf{Step 2: Splitting on associated graded.}
The mixed closed-open operations are chains on $\FM_k(\C) \times E_1(m)$. The operadic composition in $\mathsf{SC}^{\mathrm{ch,top}}$ involves simultaneously composing in both factors: inserting a cluster in $\FM_k(\C)$ (which increases pole order at the collision divisor) and composing intervals in $E_1(m)$ (which does not affect pole order). On the associated graded:
\begin{itemize}
\item The $\FM_k(\C)$ composition contributes to $\gr^p$ by the pole order of the collision; passing to $\gr$ extracts the leading polar term, which is governed by the residue at the divisor; this is precisely the BD chiral operad $\mathsf{P}_{\mathrm{ch}}$.
\item The $E_1(m)$ composition is filtered-trivial (it contributes to $F^0$, pole order 0, since interval compositions do not create poles), so it passes unchanged to $\gr^0 = E_1$.
\item Mixed compositions (inserting a cluster that involves both holomorphic collision and topological reordering) have pole order $\ge 1$ from the holomorphic part, but the topological part sits in $F^0$. On the associated graded, the product structure splits: $\gr(\FM_k(\C) \times E_1(m)) = \gr(\FM_k(\C)) \times E_1(m)$, and the mixed operadic compositions factor accordingly.
\end{itemize}
Since no mixed operations survive on $\gr$ (the chiral and topological compositions decouple), we obtain $\gr\;\mathsf{SC}^{\mathrm{ch,top}} \cong \mathsf{P}_{\mathrm{ch}} \amalg E_1$.

\textbf{Step 3: Bar/cobar on associated graded.}
Since bar and cobar constructions are compatible with filtrations (they preserve the grading by tree complexity + pole order), the associated graded of $\mathbf{B}(\mathsf{SC}^{\mathrm{ch,top}})$ is $\mathbf{B}(\mathsf{P}_{\mathrm{ch}}) \amalg \mathbf{B}(E_1)$: the BD chiral bar construction on the closed color, and the classical (associative) bar construction on the open color.
\end{proof}

\begin{theorem}[Filtered homotopy Koszul duality; \ClaimStatusProvedHere]
\label{thm:filtered-koszul}
Under the holomorphic filtration, the bar--cobar adjunction is a filtered Quillen equivalence; the associated graded is the BD chiral bar--cobar adjunction (closed color) and the $A_\infty$/associative bar--cobar (open color). In particular, for filtered $\mathsf{SC}^{\mathrm{ch,top}}$--algebras, $\Omega\mathbf B$ and $\mathbf B\Omega$ are filtered quasi-isomorphisms.
\end{theorem}

\begin{proof}
Theorem~\ref{thm:bar-cobar-adjunction} gives a Quillen equivalence on
the unfiltered level.  Proposition~\ref{prop:gr-chiral} identifies the
associated graded as the product of the chiral bar--cobar adjunction
(closed color) and the classical $A_\infty$/associative bar--cobar
(open color), both of which are Quillen equivalences.  Transferring
model structures across filtered quasi-isomorphisms (as in
\cite[Theorem~2.1]{Val07}) promotes the unfiltered equivalence to a
filtered one.
\end{proof}

\begin{remark}[Quadratic duality checkpoint]
\label{rmk:GLZ}
On the closed color, Theorem~\ref{thm:filtered-koszul} recovers the quadratic duality (GLZ) \cite{GLZ22} in the associated graded; hence our bar--cobar formalism is a filtered deformation extending GLZ to the HT setting.
\end{remark}

\subsection{Line Operators and the Koszul Dual}

\subsubsection{Definition and Basic Properties}

\begin{definition}[Line Operators in 3d HT Theories]
\label{def:line_operators}
A \emph{line operator} in a 3d HT theory on $\R_t \times \C_z$ is a non-local observable supported on a curve $\gamma \subseteq \R_t \times \C_z$. Typically:
\begin{itemize}
\item $\gamma = \{t = \text{const}\} \times \{\text{point in } \C_z\}$ (spatial line);
\item $\gamma = \R_t \times \{\text{point in } \C_z\}$ (temporal line, extending infinitely in the $t$-direction).
\end{itemize}
\end{definition}

\begin{example}[Wilson Lines in Gauge Theory]
In abelian $U(1)$ gauge theory (Example \ref{ex:abelian_CS_complete}), a Wilson line is:
\begin{equation}
W_\gamma[A] = \exp\left(i \int_\gamma A\right),
\end{equation}
where $A$ is the gauge field.
\end{example}

\subsubsection{Category of Line Operators}

\begin{definition}[Category $\mathcal{C}_{\text{line}}$]
\label{def:line_category}
The \emph{category of perturbative line operators} $\mathcal{C}_{\text{line}}$ has:
\begin{itemize}
\item Objects: Line operators $\ell_1, \ell_2, \ldots$ labeled by representations or charges;
\item Morphisms: $\Hom(\ell_1, \ell_2) = $ quantum mechanics of fields stretching between lines $\ell_1$ and $\ell_2$ (topological quantum mechanics in the $t$-direction).
\end{itemize}
\end{definition}

The category $\mathcal{C}_{\text{line}}$ is a \textbf{monoidal category} with tensor product given by:
\begin{equation}
\ell_1 \otimes \ell_2 := \text{(place lines $\ell_1$ and $\ell_2$ parallel in $\C_z$)}.
\end{equation}

\begin{theorem}[Lines as Modules for the Open-Colour Dual;
\ClaimStatusProvedHere]
\label{thm:lines_as_modules}
Let\/ $\cA$ be a chirally Koszul logarithmic $\SCchtop$-algebra
\textup{(}Definition~\textup{\ref{def:log-SC-algebra})}, that is,
$\cA$ satisfies the concentration condition
$H^{p,q}(\barB(\cA))=0$ for $q\neq 0$; see
Definition~\textup{\ref{def:chiral-koszul-geometric}} for the precise
statement.  Let\/ $\cA^!_{\mathrm{line}}$ denote the open-colour
Koszul dual, identified on the chirally Koszul locus with the
ordered-bar cohomology algebra
$H^\bullet(\overline{B}^{\mathrm{ord}}(\cA))^\vee$ by the
open-colour part of Theorem~\ref{thm:two-color-master}.
(Dimofte--Niu--Py \cite{DNP25}, Theorem 3.1) The category of
perturbative line operators is equivalent to the category of modules
for $\mathcal{A}^!_{\mathrm{line}}$:
\begin{equation}
\mathcal{C}_{\text{line}} \simeq \mathcal{A}^!_{\mathrm{line}}\text{-}\mathbf{mod}.
\end{equation}
\end{theorem}

\begin{remark}[Scope within the bulk--boundary--line triangle]
\label{rem:lines-modules-scope}
Theorem~\ref{thm:lines_as_modules} is proved for any chirally Koszul logarithmic $\SCchtop$-algebra: the identification $\cC_{\mathrm{line}} \simeq \cA^!_{\mathrm{line}}\text{-}\mathbf{mod}$ uses the open-colour bar--cobar adjunction and requires only chiral Koszulness.  However, composing this with the bulk--Hochschild identification (Theorem~\ref{thm:bulk_hochschild}) to obtain the full corrected bulk--boundary--line triangle requires additional hypotheses (compact generation and the derived center quasi-isomorphism) that are verified rigorously only in the boundary-linear exact sector (Theorem~\ref{thm:boundary-linear-bulk-boundary}).  The global triangle for all HT theories is conjectural; see Section~\ref{sec:ht-bulk-boundary-line}.
\end{remark}

\begin{proof}
\textbf{Step 1: Dimensional reduction near a line.}

Consider a line operator $\ell$ extending along $\R_t$ at a fixed point $z_0 \in \C$. The tubular neighborhood $U_\ell = D_\epsilon(z_0) \times \R_t$ (a solid cylinder) supports a restriction of the 3d HT theory. Since the theory is topological along $\R_t$ and holomorphic along $\C$, the restriction to $U_\ell$ decomposes as:
\begin{itemize}
\item The \emph{worldline theory}: a 1d topological quantum mechanics along $\R_t$, with state space $V_\ell$ (the space of states of the line $\ell$). Topological invariance means $V_\ell$ carries an $E_1$-algebra (= $A_\infty$-algebra) structure from time-ordered compositions.
\item The \emph{bulk-line coupling}: bulk local operators $\mathcal{O}$ inserted at points $(z, t) \in U_\ell$ near the line $\ell$ act on $V_\ell$ via the OPE. This action is holomorphic in $z$ (with poles as $z \to z_0$) and topological in $t$.
\end{itemize}

\textbf{Step 2: Bulk operators act on lines via the open color.}

The $\mathsf{SC}^{\mathrm{ch,top}}$-algebra structure provides mixed operations
\[
C_\ast(\FM_k(\C) \times E_1(m)) \otimes \mathcal{A}_{\mathrm{bulk}}^{\otimes k} \otimes V_\ell^{\otimes m} \longrightarrow V_\ell,
\]
encoding the insertion of $k$ bulk operators near the line and $m$ line operators along it. This endows $V_\ell$ with the structure of a module over $\mathcal{A}_{\mathrm{bulk}}$ \emph{in the open-color sense}: the action is via the open color of $\mathsf{SC}^{\mathrm{ch,top}}$, with the bulk operators entering through the closed-color inputs and the line operators through the open-color inputs.

\textbf{Step 3: Koszul duality converts bulk-modules to line-modules.}

The bar--cobar adjunction (Theorem~\ref{thm:bar-cobar-adjunction}) provides an equivalence between:
\begin{itemize}
\item Modules over $\mathcal{A}_{\mathrm{bulk}}$ viewed as a $C_\ast(W(\mathsf{SC}^{\mathrm{ch,top}}))$-algebra (the open-color action from Step 2);
\item Comodules over the ordered bar construction $\overline{B}^{\mathrm{ord}}(\mathcal{A}_{\mathrm{bulk}})$, which on the chirally Koszul locus are equivalent to modules over the open-colour Koszul dual
  $\mathcal{A}^!_{\mathrm{line}} := H^\bullet(\overline{B}^{\mathrm{ord}}(\mathcal{A}_{\mathrm{bulk}}))^\vee$.
\end{itemize}
The ordered bar construction $\overline{B}^{\mathrm{ord}}$ converts the open-color $E_1$ structure into a dg coalgebra with deconcatenation coproduct along~$\R$.  The open-color bar--cobar Quillen equivalence in Theorem~\ref{thm:two-color-master} identifies its comodules with modules over~$\mathcal{A}^!_{\mathrm{line}}$, and Theorem~\ref{thm:homotopy-Koszul} ensures this conversion preserves the homotopy type of the line-module category.

\textbf{Step 4: Categorical equivalence (essential surjectivity).}

We construct a functor $\Phi: \mathcal{C}_{\mathrm{line}} \to \mathcal{A}^!_{\mathrm{line}}\text{-}\mathbf{mod}$ sending $\ell \mapsto V_\ell$ (the state space from Step 1, now an $\mathcal{A}^!_{\mathrm{line}}$-module by Step 3). The functor is:
\begin{itemize}
\item \emph{Faithful and full on morphisms}: $\Hom(\ell_1, \ell_2)$ is the space of local operators at the junction of two lines, which is precisely $\Hom_{\mathcal{A}^!_{\mathrm{line}}}(V_{\ell_1}, V_{\ell_2})$ (intertwiners of the $\mathcal{A}^!_{\mathrm{line}}$-action), since the junction operators are determined by their bulk-line couplings.
\item \emph{Essentially surjective}: every $\mathcal{A}^!_{\mathrm{line}}$-module $V$ arises as the state space of some line operator $\ell_V$.  Given $V$, define a worldline theory along $\R_t$ by:
  \begin{enumerate}[label=\textup{(\alph*)}]
  \item State space $V$ with differential $d_V$ given by the $\mathcal{A}^!_{\mathrm{line}}$-module structure maps composed with the ordered-bar quasi-isomorphism $\mathcal{A}^!_{\mathrm{line}} \simeq H^\bullet(\overline{B}^{\mathrm{ord}}(\mathcal{A}))^\vee$;
  \item Bulk-line coupling: $k$ bulk operators at points $z_1, \ldots, z_k$ near the line act on $V$ via the $\mathcal{A}^!_{\mathrm{line}}$-module operations, with holomorphic dependence on $z_i$ inherited from the FM compactification integrals;
  \item BV-BRST closure ($d_V^2 = 0$): the $A_\infty$ module identities for $V$ over $\mathcal{A}^!_{\mathrm{line}}$ are equivalent to the ordered-bar differential condition $d_{\barB}^2 = 0$, so the worldline correlators satisfy the BV master equation, giving a consistent boundary condition.
  \end{enumerate}
  Conversely, any consistent boundary condition on the worldline determines an $\mathcal{A}^!_{\mathrm{line}}$-module structure on its state space by restriction of the BV-BRST action.  The two functors (line operator $\mapsto$ module via $\ell \mapsto V_\ell$, and module $\mapsto$ line operator via $V \mapsto \ell_V$) compose to quasi-isomorphisms: the round-trip $V \mapsto V_{\ell_V}$ is a quasi-isomorphism of $\mathcal{A}^!_{\mathrm{line}}$-modules by the open-color bar-cobar resolution, and the round-trip $\ell \mapsto \ell_{V_\ell}$ is a quasi-isomorphism of line operators by the same ordered/open resolution.
\end{itemize}
On the chirally Koszul locus, the tensor product structure
$\ell_1 \otimes \ell_2$ (placing lines at $z_1, z_2 \in \C$)
corresponds to $V_{\ell_1} \otimes V_{\ell_2}$ with the
$\mathcal{A}^!_{\mathrm{line}}$-action given by the coproduct packaged later by
Theorem~\ref{thm:yangian-recognition}.
\end{proof}

\end{remark}

\subsection{Two module categories from two-color duality}
\label{subsec:two-module-categories}

The $\SCchtop$-algebra structure on the open-colour Koszul dual
$\cA^!_{\mathrm{line}}$
\textup{(}Theorem~\textup{\ref{thm:dual-sc-algebra}}\textup{)}
produces two module
categories, one for each color.

\begin{theorem}[Two module categories;
\ClaimStatusProvedHere]
\label{thm:two-module-categories}
\index{two-color Koszul duality!module categories}
Let $\cA$ be a chirally Koszul logarithmic
$\SCchtop$-algebra
\textup{(}Definition~\textup{\ref{def:chiral-koszul-geometric})}.
The $\SCchtop$-algebra structure
on~$\cA^!_{\mathrm{line}}$ gives rise to two module categories:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Open-color modules.}
  $\cA^!_{\mathrm{line}}$-modules as an $E_1$-algebra model the line
  category $\mathcal{C}_{\mathrm{line}}$
  (Theorem~\ref{thm:lines_as_modules}).  Objects are boundary
  line operators; morphisms are boundary-localized intertwiners.
\item \emph{Closed-color modules.}
  $\cA^!_{\mathrm{line}}$-modules as a chiral algebra form the category of
  \emph{factorization modules} on $\Ran(X)$: families
  $\{L_z\}_{z \in X}$ of line operators parametrized by
  points of the curve, equipped with factorization isomorphisms.
\item \emph{Relationship.}
  An object of~\textup{(ii)} is a family
  $\{L_z\}_{z \in X}$ of objects of~\textup{(i)} carrying a
  flat spectral connection $\nabla_z$ whose monodromy is
  the $R$-matrix $R(z)$
  (Definition~\ref{def:spectral-braiding}).
\end{enumerate}
\end{theorem}

\begin{proof}
Part~(i) is Theorem~\ref{thm:lines_as_modules} applied to the
open-color factor.  For~(ii): the closed-color structure maps
$C_\ast(\FM_k(\C)) \otimes (\cA^!_{\mathrm{line}})^{\otimes k}
\to \cA^!_{\mathrm{line}}$
make $\cA^!_{\mathrm{line}}$ into a chiral algebra, and its module category
is the standard factorization-module category on $\Ran(X)$.
Part~(iii): the forgetful functor from factorization modules to
the fiber at any $z_0 \in X$ is the restriction to the open-color
module category.  The factorization isomorphism at $(z_1, z_2)$
is the $R$-matrix $R(z_1 - z_2)$; flatness of the spectral
connection $\nabla_z$ follows from the Yang--Baxter equation
(Theorem~\ref{thm:YBE}), which is the zero-curvature condition
$[\nabla_{z_1}, \nabla_{z_2}] = 0$ for the KZ-type connection
on configuration space.
\end{proof}

\begin{proposition}[Spectral families as evaluation modules;
\ClaimStatusProvedHere]
\label{prop:spectral-families}
\index{evaluation modules!spectral families}
For $\cA = \widehat{\fg}_k$ with
$\cA^!_{\mathrm{line}} = Y_\hh(\fg)$:
the evaluation homomorphisms
$\mathrm{ev}_z \colon Y_\hh(\fg) \to U(\fg)$ are the fibers
of factorization modules at $z \in X$.  The Yangian coproduct
$\Delta_{z_1,z_2} \colon Y_\hh(\fg) \to
Y_\hh(\fg) \otimes Y_\hh(\fg)$
is the factorization isomorphism for the decomposition at
$(z_1, z_2) \in \Ran^2(X)$.
\end{proposition}

\begin{proof}
The evaluation map $\mathrm{ev}_z$ is the composition
$Y_\hh(\fg) \xrightarrow{T(u)} \End(V)((u^{-1}))
\xrightarrow{u=z} \End(V)$,
which extracts the fiber of the generating matrix $T(u)$ at
$u = z$.  This is the restriction of the factorization-module
structure to the point $z$.  The factorization isomorphism
$Y_\hh(\fg)\text{-mod} |_{\{z_1, z_2\}} \simeq
Y_\hh(\fg)\text{-mod} |_{\{z_1\}} \otimes
Y_\hh(\fg)\text{-mod} |_{\{z_2\}}$
is the Yangian coproduct $\Delta_{z_1,z_2}$, which depends
meromorphically on $z_1 - z_2$ with poles only at $z_1 = z_2$.
\end{proof}

\begin{remark}[Two categories versus one]
\label{rem:two-categories-vs-one}
The forgetful functor from factorization modules~(ii)
to fibers~(i) forgets the flat spectral connection.  The
$R$-matrix is the holonomy datum that reconstructs the family
from its fibers.  Within the $E_\infty$-chiral class, the
pole-free commutative algebras have $R(z) = \tau$: the flat
connection is trivial and the two module categories collapse.
For $E_\infty$-chiral algebras with OPE poles (vertex
algebras such as $\widehat{\fg}_k$ or $\mathrm{Vir}_c$),
$R(z) \neq \tau$ but is derived from the local OPE, so the
two categories are related by an explicit twist.  For genuine
$E_1$-chiral algebras, the $R$-matrix is independent input
the nontrivial datum distinguishing the two categories.
This is why the spectral parameter and quantum group structure
are invisible in the pole-free commutative limit and
progressively more visible as one moves through the three
tiers.
\end{remark}

\subsection{The spectral Maurer--Cartan kernel and modular extension}
\label{subsec:spectral-mc-kernel}

The comparison modeling the line category by
$\mathcal{A}^!_{\mathrm{line}}$-modules becomes computable
once one extracts the \emph{spectral Maurer--Cartan kernel}
governing line-operator OPE.

\begin{definition}[Universal MC kernel]
\label{def:universal-mc-kernel}
Let $\{e_I\}$ be a homogeneous basis for the boundary algebra $\mathcal{A}_\partial$ and
let $\{e_I^\vee\} \subset \mathcal{A}^!_{\mathrm{line}}$ be the dual basis under the ordered
bar--cobar pairing.
The \emph{universal Maurer--Cartan kernel} is the chain-level holographic pairing
\begin{equation}
\label{eq:mc-kernel}
\mu \;:=\; \sum_I e_I^\vee \otimes e_I
\;\in\;
\mathcal{A}^!_{\mathrm{line}} \;\hat\otimes\; \mathcal{A}_\partial.
\end{equation}
\end{definition}

The spectral kernel
$r(z) \in \mathcal{A}^!_{\mathrm{line}} \hat\otimes\, \mathcal{A}^!_{\mathrm{line}}\bigl(\!(z^{-1})\!\bigr)$
is obtained by twisting $\mu$ with the spectral parameter.  It controls the coproduct on
$\mathcal{A}^!_{\mathrm{line}}$ via the twisted formula
\begin{equation}
\label{eq:twisted-coproduct}
\Delta_z:\; \mathcal{A}^!_{\mathrm{line}}
\;\longrightarrow\;
\mathcal{A}^!_{\mathrm{line}}
\;\hat\otimes_{r(z)}\;
\mathcal{A}^!_{\mathrm{line}}\bigl(\!(z^{-1})\!\bigr),
\end{equation}
where the subscript $r(z)$ indicates that the tensor product carries the differential
twisted by $r(z)$.  The consistency of this twisted coproduct is equivalent to the
$A_\infty$ Yang--Baxter equation: it is encoded by the identity $D^2 = 0$ on the
$d_r$-twisted modular bar complex.

\begin{definition}[Modular extension of the spectral kernel]
\label{def:modular-spectral-kernel}
The \emph{modular extension} of the spectral kernel is the formal genus expansion
\begin{equation}
\label{eq:modular-spectral-kernel}
R_T^{\mathrm{mod}}(z;\hbar)
\;=\;
\sum_{g \geq 0} \hbar^{2g}\, r_{T,g}(z),
\end{equation}
with
\[
r_{T,g}(z)
\;\in\;
H^\bullet(\ov{\mathcal{M}}_{g,2})
\;\otimes\;
\mathcal{A}^!_{\mathrm{line}}
\;\hat\otimes\;
\mathcal{A}^!_{\mathrm{line}}\bigl(\!(z^{-1})\!\bigr).
\]
At $g=0$ this recovers the genus-zero spectral kernel
$r(z)$ of Theorem~\ref{thm:spectral_R_YBE}.
\end{definition}

\begin{remark}[From line-category comparison to spectral kernel]
\label{rem:abstract-to-concrete-kernel}
The spectral kernel $r(z)$ makes
Theorem~\ref{thm:lines_as_modules} computable: it converts the categorical statement ``the line category is modeled by $\mathcal{A}^!_{\mathrm{line}}$-modules''
into the formula ``line-operator OPE is governed by a universal $r$-matrix.''
The modular extension $R_T^{\mathrm{mod}}(z;\hbar)$ is a
genus-refined kernel whose coefficients $r_{T,g}$ encode the modular corrections
to the OPE at each loop order.
The dg-shifted Yangian is a strict model of a homotopy-coherent
meromorphic tensor structure, recoverable from the modular spectral
kernel via strictification
(Theorem~\ref{thm:complete-strictification}: the spectral Drinfeld
class vanishes at every filtration for every simple Lie algebra, by
root multiplicity one).
\end{remark}

\begin{remark}[The spectral parameter as $\FM_2(\C)$ coordinate]
\label{rem:spectral-parameter-fm2}
\index{spectral parameter!FM2 coordinate@$\FM_2(\C)$ coordinate}
The variable $z$ in the spectral kernel $r(z)$ is the coordinate
on $\FM_2(\C)$ in the closed color of the Swiss-cheese
correspondence: it is the relative position $z_1 - z_2$ of two
bulk insertion points in the holomorphic plane, compactified by
the Fulton--MacPherson boundary.  The pole at $z = 0$ is the
collision divisor $D_{\{1,2\}} \subset \FM_2(\C)$; the residue
there is the leading OPE coefficient.  In this sense $z$ is
the chiral analog of the $q$-parameter in the Hecke algebra:
both parametrize a one-dimensional family of algebras
interpolating between two boundary conditions (the two strata
of the compactified configuration space), and both degenerate
at $z = 0$ (resp.\ $q = 1$) where the two points (resp.\ the
two Borel orbits) collide.
\end{remark}

\subsection{Boundary flips as morphisms of Koszul dual modules}
\label{subsec:boundary-flips}
\index{flip operation|textbf}
\index{Koszul duality!boundary flips}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

In the CDG framework \cite{CDG2023}, \S4.3, a \emph{flip} is the operation
of coupling a two-dimensional boundary degree of freedom that changes the
boundary condition type, typically exchanging Neumann and Dirichlet
conditions for a given field.

\begin{definition}[Flip as $\mathcal{A}^!_{\mathrm{line}}$-module morphism]
\label{def:boundary-flip}
Let $M_N$ and $M_D$ denote the $\mathcal{A}^!_{\mathrm{line}}$-modules corresponding to
Neumann and Dirichlet boundary conditions respectively (via
Theorem~\ref{thm:lines_as_modules}).  A \emph{boundary flip} for a field
$\varphi$ is a chain map
\begin{equation}\label{eq:flip-morphism}
  \mathrm{Flip}_\varphi \;:\; M_N \;\longrightarrow\; M_D
\end{equation}
in the homotopy category of $\mathcal{A}^!_{\mathrm{line}}$-modules, induced by coupling
the boundary to a $2$d auxiliary system that absorbs the field $\varphi$
and replaces it by its conjugate.
\end{definition}

\begin{proposition}[Flip as natural transformation; \ClaimStatusProvedElsewhere]
\label{prop:flip-natural-transformation}
\textup{(Costello--Dimofte--Gaiotto \cite{CDG2023}, \S4.3.)}
The flip functor acts on the line category
$\cC_T \simeq \mathcal{A}^!_{\mathrm{line}}\text{-}\mathbf{mod}$ as a natural
transformation between the Neumann and Dirichlet module functors.
The induced map on $K$-theory exchanges the corresponding classes of
boundary conditions.
\end{proposition}

\begin{example}[Free chiral flip]
\label{ex:free-chiral-flip}
For the free chiral multiplet with fields $(\phi, \psi)$, flipping
$\phi \leftrightarrow \psi$ exchanges the vortex line families:
$V_{N > 0} \leftrightarrow V_{N < 0}$, where $V_N$ denotes the vortex
line of charge~$N$.  In the $\mathcal{A}^!_{\mathrm{line}}$-module language, this is the
exchange of the positive-charge and negative-charge simple modules.
\end{example}

\begin{remark}[Bar-cobar interpretation of flips]
\label{rem:flip-bar-cobar}
At the chain level, flips are chain maps between different bar complex
models: the Neumann bar complex $\bar{B}(\cA_N)$ and the Dirichlet bar
complex $\bar{B}(\cA_D)$.  The flip morphism
\eqref{eq:flip-morphism} is therefore a morphism in the homotopy category
of $\mathcal{A}^!_{\mathrm{line}}$-modules, and its existence is a consequence of the
Quillen equivalence (Theorem~\ref{thm:homotopy-Koszul}).
\end{remark}

\begin{remark}[Vortex lines as the prototypical example]
\label{rem:vortex-line-prototype}
\index{vortex line!prototypical example}
The free chiral vortex lines $\{V_N\}_{N \in \mathbb{Z}}$ of the
$\beta\gamma$ system, introduced and analyzed in the beta--gamma
chapter of Volume~I, are the simplest non-trivial
line operators.  Their $\cA^!_{\mathrm{line}}$-module structure, explicit $r$-matrix,
and fusion rules are worked out completely there.  The flip exchanging
$V_{N>0} \leftrightarrow V_{N<0}$ of Example~\ref{ex:free-chiral-flip}
is the abelian prototype: the general flip of
Definition~\ref{def:boundary-flip} reduces to the $\beta\gamma$ case
when the gauge group is $\mathrm{U}(1)$ and the boundary is free.
\end{remark}

\begin{example}[Composite flips for gauge theories]
\label{ex:composite-flips}
\index{flip operation!composite}
For a gauge theory with multiple matter fields $\varphi_1, \ldots,
\varphi_r$, flipping a subset $S \subset \{1,\ldots,r\}$ produces a
composite morphism
\begin{equation}\label{eq:composite-flip}
  \mathrm{Flip}_S
  \;=\;
  \prod_{i \in S} \mathrm{Flip}_{\varphi_i}
  \;:\;
  M_{N}^{(S^c)} \otimes M_{D}^{(S)}
  \;\longrightarrow\;
  M_{D}^{(S^c)} \otimes M_{N}^{(S)},
\end{equation}
where the Neumann and Dirichlet modules for each field are exchanged
independently.  The associativity
$\mathrm{Flip}_{S_1} \circ \mathrm{Flip}_{S_2} =
\mathrm{Flip}_{S_1 \triangle S_2}$ (symmetric difference) makes the
flips into a $(\mathbb{Z}/2)^r$-action on the boundary condition
space \cite{CDG2023}, \S4.3.  This $(\mathbb{Z}/2)^r$-symmetry is
the bar-cobar manifestation of the group of boundary condition
changes.  In the Chriss--Ginzburg framework, each flip is a
gauge transformation in $\operatorname{MC}(\mathfrak{g}^{\mathrm{amb}}_{\cA})$:
the MC element $\Theta_{\cA}$ is invariant under the full
$(\mathbb{Z}/2)^r$ but the cross-polarization
(which side is $\cA$, which is $\cA^!$) is exchanged.
\end{example}

\subsection{Heisenberg: free lines}
\label{subsec:line-op-heisenberg}

For $\cA=\cH_k$: the chiral Koszul dual is
$\cA^!_{\mathrm{ch}}=\Sym^{\mathrm{ch}}(V^*)$, while the
open-colour dual is
$\cA^!_{\mathrm{line}}=Y(\mathfrak{u}(1))$
(whose mode algebra $U(\mathfrak{u}(1)[t]) \cong \Bbbk[t]$
coincides with the mode algebra of~$\cH_{-k}$,
but is distinct as a chiral algebra; cf.\ AP33),
$r(z)=k/z$, and $\Theta_{\cH_k}=k\cdot\eta\otimes\Lambda$.
The $r$-matrix is scalar: $r(z)=k/z\in\Bbbk(\!(z^{-1})\!)$.

\begin{computation}[Line-operator category for Heisenberg;
\ClaimStatusProvedHere]
\label{comp:line-op-heisenberg}
\index{Heisenberg algebra!line operators}
$\cC_{\mathrm{line}}(\cH_k)
\simeq\cA^!_{\mathrm{line}}\text{-}\mathbf{mod}
\simeq Y(\mathfrak{u}(1))\text{-}\mathbf{mod}
\simeq\cH_{-k}\text{-}\mathbf{mod}
\simeq\mathrm{Vect}^{\mathbb Z}$:
the category of $\mathbb Z$-graded vector spaces.  The
grading is by charge $q\in\mathbb Z$; the Fock module
$\mathcal F_q$ of charge~$q$ is an irreducible
$\cH_{-k}$-module.  The tensor product is
$\mathcal F_{q_1}\otimes\mathcal F_{q_2}
=\mathcal F_{q_1+q_2}$
(charge addition).  The $r$-matrix acts on
$\mathcal F_{q_1}\otimes\mathcal F_{q_2}$ by the scalar
$k\,q_1\,q_2/z$: the braiding is
\[
R_{q_1,q_2}(z)
\;=\;
z^{k\,q_1\,q_2}\cdot\mathrm{id}
\;\in\;
\operatorname{End}(\mathcal F_{q_1}\otimes\mathcal F_{q_2}).
\]
The CYBE $[r_{12},r_{13}]+[r_{12},r_{23}]+[r_{13},r_{23}]=0$
reduces to $0=0$ (the $r$-matrix is central).
No nontrivial line operators beyond Fock modules.
\end{computation}

\subsection{Affine Kac--Moody: Yangian modules and evaluation sector}
\label{subsec:line-op-affine}

For $\cA=\widehat{\mathfrak{sl}}_2{}_k$:
the chiral Koszul dual is
$\cA^!_{\mathrm{ch}}=\widehat{\mathfrak{sl}}_2{}_{-k-4}$,
while on the chirally Koszul affine locus the open-colour dual is
$\cA^!_{\mathrm{line}}=\Ydg(\mathfrak{sl}_2)$.  Its classical
spectral kernel is
$r(z)=\Omega/z=(e\otimes f+f\otimes e+\tfrac12 h\otimes h)/z$,
and on evaluation modules
$q=e^{i\pi/(k+2)}$.

\begin{computation}[Line-operator category for
$\widehat{\mathfrak{sl}}_2$; \ClaimStatusProvedHere]
\label{comp:line-op-sl2}
\index{Kac--Moody!line operators}
On the chirally Koszul affine locus,
\[
\cC_{\mathrm{line}}(\widehat{\mathfrak{sl}}_2{}_k)
\simeq \Ydg(\mathfrak{sl}_2)\text{-}\mathbf{mod}.
\]
Restricting to evaluation modules and passing to the reduced braided
sector,
\[
\cC_{\mathrm{line}}^{\mathrm{red}}(\widehat{\mathfrak{sl}}_2{}_k)
\big|_{\mathrm{eval}}
\simeq \operatorname{Rep}_q(\mathfrak{sl}_2),
\qquad q=e^{i\pi/(k+2)}.
\]
In this reduced evaluation sector, the simple objects are the usual
quantum-group simples $V_\lambda$ ($\lambda\in\mathbb Z_{\ge 0}$,
with $0\le \lambda\le k$ in the integrable level-$k$ sector).

The $r$-matrix $r(z)=\Omega/z$ acts on
$V_{\lambda_1}\otimes V_{\lambda_2}$ via the Casimir:
\[
R_{\lambda_1,\lambda_2}(z)
\;=\;
\exp\Bigl(\frac{\Omega_{\lambda_1,\lambda_2}}{z}\Bigr)
\;\in\;
\operatorname{End}(V_{\lambda_1}\otimes V_{\lambda_2})
[\![z^{-1}]\!].
\]
For $k=1$ (simplest non-abelian level): two simple objects
$V_0$ (trivial) and $V_1$ (fundamental).  Fusion:
$V_1\otimes V_1=V_0$.  The $R$-matrix at $\lambda_1=\lambda_2=1$
is a $4\times 4$ matrix with eigenvalue $q=e^{i\pi/3}$.

The KZ connection
$\nabla_{0,n}=d-\frac{1}{k+2}\sum_{i<j}\Omega_{ij}\,
d\log(z_i-z_j)$
is, on the affine Kac--Moody comparison surface, the flat
connection on conformal blocks for evaluation modules; its
monodromy representation is compared by the affine
Drinfeld--Kohno package with the braiding on the reduced
evaluation sector.

At $k=-2$ (critical level): $q=e^{i\pi/(k+2)}$ is undefined
(division by zero), and the evaluation-sector quantum-group
description degenerates.  The KZ connection acquires a pole,
so this picture lies outside the chirally Koszul affine locus.
\end{computation}

\begin{computation}[Explicit $R$-matrix for
$\widehat{\mathfrak{sl}}_2$ on $V\otimes V$;
\ClaimStatusProvedHere]
\label{comp:explicit-r-matrix-sl2}
\index{R-matrix!explicit matrix entries!sl2@$\mathfrak{sl}_2$}

The Casimir acts on $V\otimes V$ as
$\Omega=e\otimes f+f\otimes e+\tfrac12 h\otimes h$,
a $4\times 4$ matrix in the basis $\{v_+\otimes v_+,\,
v_+\otimes v_-,\,v_-\otimes v_+,\,v_-\otimes v_-\}$:
\[
\Omega
\;=\;
\begin{pmatrix}
1/2 & 0 & 0 & 0 \\
0 & -1/2 & 1 & 0 \\
0 & 1 & -1/2 & 0 \\
0 & 0 & 0 & 1/2
\end{pmatrix}.
\]
The $R$-matrix at level~$k$ is
$R(z)=1+\Omega/(z(k{+}2))+O(z^{-2})$.  At leading order:
\[
R(z)\big|_{k=1}
\;=\;
\begin{pmatrix}

\subsubsection{The RTT presentation and mode structure}
\label{subsubsec:virasoro-rtt}

The collision residue $r^{\mathrm{coll}}_c(z) = (c/2)/z^3 + 2T/z$
(Proposition~\ref{prop:field-theory-r}) has a \emph{cubic} pole,
not the simple pole of the affine RTT relation.
The cubic pole forces the RTT framework to carry three mode
strata rather than one: while the affine $r$-matrix
$r(z) = \Omega/z$ has a single residue $\Omega_0$, the
Virasoro collision residue has residues $\Omega_0$, $\Omega_1$,
$\Omega_2$ at $z^{-1}$, $z^{-2}$, $z^{-3}$, with the
full series $r(z) = \sum_{m \ge 0} \Omega_m/z^{m+1}$ continuing
to all orders
(Construction~\ref{constr:thqg-V-vir-r-explicit}).

\begin{construction}[Expected Virasoro RTT relation; \ClaimStatusConjectured]
\label{constr:virasoro-rtt-relation}
\index{Virasoro algebra!RTT relation|textbf}
\index{RTT relation!Virasoro}
In the expected Virasoro realization of the line side, the
quantum $R$-matrix $R(z)$ on the modeled line category is the
monodromy of the ordered bar connection
(Construction~\ref{constr:r-matrix-monodromy}).
Define the transfer matrix
$\mathcal{T}(z) \in
\operatorname{End}(V) \otimes
\mathrm{Vir}_{26-c}[\![z^{-1}]\!]$
for any $\mathrm{Vir}_{26-c}$-module $V$ in this expected model by the generating function
\begin{equation}\label{eq:virasoro-transfer}
\mathcal{T}(z)
\;=\;
\sum_{n \ge 0}
z^{-n-1}\,
\pi_V(L_n),
\end{equation}
where $\pi_V \colon \mathrm{Vir}_{26-c} \to \operatorname{End}(V)$
is the representation map and $\{L_n\}_{n \in \mathbb{Z}}$ are the
Virasoro generators at dual central charge $c' = 26 - c$.

The RTT relation takes the form
\begin{equation}\label{eq:virasoro-rtt}
\boxed{
R_{12}(z{-}w)\,
\mathcal{T}_1(z)\,
\mathcal{T}_2(w)
\;=\;
\mathcal{T}_2(w)\,
\mathcal{T}_1(z)\,
R_{12}(z{-}w),
}
\end{equation}
where the subscripts denote tensor factors and
$R_{12}(z)$ is the spectral $R$-matrix of
Corollary~\textup{\ref{cor:gravity-spectral-r}}.
The classical limit $\hbar \to 0$ (equivalently
$c \to \infty$, the semiclassical gravity limit) gives
\begin{equation}\label{eq:virasoro-rtt-classical}
[\mathcal{T}_1(z),\, \mathcal{T}_2(w)]
\;=\;
\frac{1}{c}\bigl[r_{12}(z{-}w),\,
\mathcal{T}_1(z)\,\mathcal{T}_2(w)\bigr]
\;+\; O(c^{-2}),
\end{equation}
with $r_{12}(z) = (c/2)/z^3 + 2T/z$ the classical
collision residue.
\end{construction}

\begin{proof}[Derivation within the expected model]
The RTT relation~\eqref{eq:virasoro-rtt} is the specialisation
of the general RTT framework
(equation~\eqref{eq:rtt-from-bar}) to $\cA = \mathrm{Vir}_c$.
The transfer matrix is the composition of the $R$-matrix
action with the module representation, and the RTT identity
encodes the consistency of the coproduct
$\Delta_z \colon \mathrm{Vir}_{26-c} \to
\mathrm{Vir}_{26-c} \otimes_{r(z)} \mathrm{Vir}_{26-c}$
(Proposition~\ref{prop:virasoro-twisted-coproduct})
with the Yang--Baxter equation for $R(z)$
(Corollary~\ref{cor:gravity-spectral-r}).
\end{proof}

\begin{computation}[Mode decomposition of the Virasoro RTT
relation; \ClaimStatusProvedHere]
\label{comp:virasoro-rtt-modes}
\index{Virasoro algebra!RTT mode relations}
Expanding~\eqref{eq:virasoro-rtt} in modes by substituting
$R(z) = 1 + r(z) + O(r^2)$
with $r(z) = \sum_{m \ge 0} \Omega_m z^{-m-1}$ and
$\mathcal{T}(z) = \sum_n L_n\,z^{-n-1}$,
the coefficient of $z^{-m-1}\,w^{-n-1}$ at leading
classical order gives
\begin{equation}\label{eq:virasoro-rtt-mode-relation}
[L_m,\, L_n]
\;=\;
(m{-}n)\,L_{m+n}
\;+\;
\frac{c'}{12}(m^3{-}m)\,\delta_{m+n,0},
\end{equation}
with $c' = 26 - c$.  This is exactly the
$\mathrm{Vir}_{26-c}$ commutation relation: the RTT
relation does not produce new generators or new relations
beyond the Virasoro algebra itself.

This is the fundamental contrast with the affine case.
For $\widehat{\mathfrak{sl}}_2$, the RTT
relation~\eqref{eq:FRT-components} with $r(z) = \Omega/z$
produces the Yangian $Y(\mathfrak{sl}_2)$, which is a
genuine \emph{deformation} of $U(\mathfrak{sl}_2{[t]})$
with new generators $J^{(1)}_a$ beyond the Lie algebra
generators $J^{(0)}_a$.  For Virasoro, the higher poles
$\Omega_m$ with $m \ge 1$ do not produce new generators:
they are \emph{determined} by the Virasoro algebra
structure through the Casimir component
formula $\Omega_m = \sum_n \binom{n+m-1}{m}\,L_{-n-m}
\otimes L_n$.

The obstruction to a separate ``Yangian layer'' is the
non-formality of $\mathrm{Vir}_c$:
$m_k \ne 0$ for all $k \ge 3$
(Remark~\ref{rem:yangian-virasoro-nonformality}).  For
formal algebras (affine KM), the Yangian is a strict
deformation of $U(\mathfrak{g}{[t]})$ and the modes separate
into level-$0$ (Lie algebra) and level-$\ge 1$ (Yangian).
For non-formal algebras (Virasoro), the entire $A_\infty$
tower is encoded in the algebra itself: there is no
separate ``Yangian direction'' because the higher
$A_\infty$ operations \emph{are} the Virasoro structure.
\end{computation}

\subsubsection{Why the Virasoro has no separate Yangian}
\label{subsubsec:virasoro-no-separate-yangian}

\begin{proposition}[Virasoro: the line algebra is not
a deformation of $U(\mathrm{Witt}{[t]})$;
\ClaimStatusProvedHere]
\label{prop:virasoro-no-yangian-deformation}
\index{Virasoro algebra!absence of Yangian deformation|textbf}
\index{Yangian!does not exist for Virasoro}
The open-colour Koszul dual
$\cA^!_{\mathrm{line}} = \mathrm{Vir}_{26-c}$ is
\emph{not} a filtered deformation of
$U(\mathrm{Witt}{[t]})$ in the sense of a standard Yangian.
The obstruction is twofold:
\begin{enumerate}[label=\textup{(\alph*)}]
\item \emph{Central extension.}\;
  The Virasoro algebra is a non-trivial central extension
  of the Witt algebra $\mathrm{Witt} = \mathrm{Der}(\mathbb{C}[z,z^{-1}])$
  by $\mathbb{C}$.  The extension class
  $[\omega] \in H^2(\mathrm{Witt}, \mathbb{C})$ is the
  Gel'fand--Fuchs cocycle
  $\omega(L_m, L_n) = \tfrac{1}{12}(m^3 - m)\,\delta_{m+n,0}$.
  For a standard Yangian $Y(\mathfrak{g})$,
  the classical limit is $U(\mathfrak{g}{[t]})$: the loop
  algebra of a \emph{finite-dimensional} Lie algebra.
  The Virasoro algebra is infinite-dimensional, and the
  Gel'fand--Fuchs cocycle has no analogue for
  finite-dimensional Lie algebras.
\item \emph{Non-formality.}\;
  The $A_\infty$ structure on $\mathrm{Vir}_c$ has
  $m_k \ne 0$ for all $k \ge 3$
  (Remark~\ref{rem:yangian-virasoro-nonformality}).  For
  a Yangian $Y(\mathfrak{g})$, the associated graded
  $U(\mathfrak{g}{[t]})$ is formal ($m_k = 0$ for $k \ge 3$)
  and the deformation lives in $\HH^2(U(\mathfrak{g}{[t]}))$
  (Proposition~\ref{prop:yangian-deformation-obstruction}).
  The non-vanishing of the Virasoro higher $A_\infty$
  operations means the deformation problem does not
  split into a Yangian direction and a Lie-algebra
  direction.
\end{enumerate}
The correct statement is: $\mathrm{Vir}_{26-c}$ is
simultaneously the boundary algebra, the Koszul dual
algebra, and the line-side algebra.  There is no
separate ``Yangian layer.''
The infinite tower of higher $A_\infty$ operations
$\{m_k\}_{k \ge 3}$ is the Virasoro algebra structure
itself, not a deformation of a simpler object.
\end{proposition}

\begin{proof}
Part~(a): the space of central extensions
$H^2(\mathrm{Witt}, \mathbb{C}) = \mathbb{C}$ is
one-dimensional, generated by the Gel'fand--Fuchs cocycle.
A Yangian $Y(\mathfrak{g})$ is a deformation of
$U(\mathfrak{g}{[t]})$ in the Harrison cohomology class
$\Har^2(U(\mathfrak{g}{[t]}))$
(Proposition~\ref{prop:yangian-deformation-obstruction}).
For $\mathfrak{g}$ infinite-dimensional, the Harrison
cohomology of $U(\mathfrak{g}{[t]})$ is not controlled by
the K\"unneth formula of
Proposition~\ref{prop:HH-Ugt-decomposition}, and the
leading vanishing lemma
(Lemma~\ref{lem:harrison-leading-vanishing}) does not
apply.  In particular, the Virasoro Harrison obstruction
group $\Har^3(U(\mathrm{Witt}{[t]}))$ need not vanish.

Part~(b): the non-formality $m_k \ne 0$ for $k \ge 3$
is proved in the gravity chapter
(Section~\ref{subsec:gravity-ainf}): the quartic OPE pole
forces a non-zero associator $A_3 \ne 0$, which forces
$m_3 \ne 0$, and the quartic contact invariant
$\mathfrak{Q} = 10/[c(5c+22)] \ne 0$ (for generic $c$)
forces $m_4 \ne 0$.  The non-truncation of the shadow
tower (Proposition~\ref{prop:vir-truncation}) then gives
$m_k \ne 0$ for all $k \ge 3$.  For formal algebras, the
Yangian deformation class lives in a separate cohomological
direction from the Lie bracket; for non-formal algebras,
these directions are entangled.
\end{proof}

  $\mathrm{Vir}_{26-c}$.

  The Virasoro algebra is the central extension of the
  Lie algebra of $\operatorname{Diff}(S^1)$.  The
  coadjoint orbits of $\operatorname{Diff}(S^1)$
  (the Kirillov--Kostant classification) provide a
  natural stratification: generic orbits (labeled by
  a single invariant, the Schwarzian derivative) and
  degenerate orbits (the stabilizer jumps).  The
  ``Virasoro Steinberg variety'' would be the
  self-intersection of the universal orbit inside
  the coadjoint representation
  $\mathrm{Vir}^* \simeq
  \{c/(2z^4) + u(z)/(z^2)\,dz^2\}$,
  where $u(z)$ is a projective connection.

\item \emph{DS reduction of the affine Steinberg.}\;
  The Steinberg variety
  $Z_{\mathfrak{sl}_2} = \widetilde{\mathcal{N}} \times_{\mathcal{N}}
  \widetilde{\mathcal{N}}$ for $\mathfrak{sl}_2$ carries a
  $\widehat{\mathfrak{sl}}_2$-action by Kazhdan--Lusztig.
  DS reduction $V_k(\mathfrak{sl}_2) \twoheadrightarrow
  \mathrm{Vir}_c$ induces a BRST functor on modules:
  $H^0_{\mathrm{DS}}(M) = H^0(M, Q_{\mathrm{DS}})$.
  This functor should transport the Steinberg convolution
  from $K^{G \times \mathbb{G}_m}(Z_{\mathfrak{sl}_2})$
  to a ``reduced Steinberg'' convolution algebra acting on
  the expected Virasoro line-side model via
  $\mathrm{Vir}_{26-c}\text{-mod}$.  The key difficulty is
  that BRST reduction does not commute with convolution in
  general, and the correction terms are controlled by the
  non-formality of the Virasoro $A_\infty$ structure.
\end{enumerate}
Route~(iii) is the most promising: it uses the proved
DS intertwining theorem
(Theorem~\ref*{V1-thm:ds-koszul-intertwine}) and the proved
affine Steinberg presentation
(Theorem~\ref{thm:steinberg-presentation}), reducing the
Virasoro geometric realization to a BRST-compatibility
question.  But none of these routes is completed in either
volume.
\end{remark}

\subsection{Comparison with Costello--Dimofte--Gaiotto}

\begin{remark}[CDG line operators = ordered bar cohomology modules]

The perturbative line operators of CDG~\cite{CDG20} are

\subsection{The DNP identification: meromorphic tensor product
from the ordered bar complex}
\label{subsec:dnp-identification}

\begin{theorem}[DNP line-operator package = bar-cobar twisting package;
\ClaimStatusProvedHere]
\label{thm:dnp-bar-cobar-identification}
\index{Dimofte--Niu--Py!bar-cobar identification|textbf}
\index{meromorphic tensor product!ordered bar coproduct|textbf}
Let\/ $\cA$ be a chirally Koszul logarithmic $\SCchtop$-algebra
with open-colour Koszul dual\/ $\cA^!_{\mathrm{line}}$
\textup{(}Theorem~\textup{\ref{thm:lines_as_modules})}.
\begin{enumerate}[label=\textup{(\roman*)}]
\item \textup{(Monoidal identification.)}
  Under the equivalence\/
  $\Phi\colon \cC_{\mathrm{line}} \xrightarrow{\;\sim\;}
  \cA^!_{\mathrm{line}}\text{-}\mathbf{mod}$,
  the meromorphic tensor product\/ $\ell_1\otimes_z\ell_2$
  corresponds to the $R$-twisted tensor product
  $V_{\ell_1}\otimes_{R(z)}V_{\ell_2}$
  via the $R$-twisted coproduct $\Delta_z$ of\/
  Theorem~\textup{\ref{thm:yangian-recognition}}.
  The functor $\Phi$ is monoidal.
\item \textup{(Maurer--Cartan identification.)}
  The DNP $A_\infty$ Yang--Baxter MC element
  $r(z)\in\cA^!\otimes\cA^!$
  of\/ \textup{\cite{DNP25}} is the genus-zero binary collision
  residue of\/ $\Theta_\cA$:
  $r(z) = \operatorname{Res}^{\mathrm{coll}}_{0,2}(\Theta_\cA)$,
  and the MC equation $dr + r\star r = 0$ is the bar-cobar
  adjunction equation $\partial\tau + \tau\cup\tau = 0$.
\item \textup{(Non-renormalization = Koszulness.)}
  The one-loop exactness of line-operator OPE
  \textup{(\cite[Thm.\,4.1]{DNP25})} corresponds to
  $E_2$-collapse of the bar spectral sequence.
  The higher $A_\infty$ operations\/ $m_k$ on\/
  $\cA^!_{\mathrm{line}}$ for $k\geq 3$ are the collision
  residues $\operatorname{Res}^{\mathrm{coll}}_{0,k}(\Theta_\cA)$;
  they vanish on the chirally Koszul locus.
\end{enumerate}
\end{theorem}

\begin{proof}
Part~(i): the functor $\Phi$ sends $\ell$ to its state space $V_\ell$
\textup{(}Theorem~\textup{\ref{thm:lines_as_modules}}, Step~4\textup{)}.
The two-line OPE on the physical side IS the $R$-twisted tensor product
on the algebraic side, both computed from the FM-configuration-space
integral over $\FM_2(\bC)\times E_1(2)$.

Part~(ii): Theorem~\textup{\ref{V1-thm:collision-residue-twisting}} from
Vol~I identifies the collision residue with the twisting morphism
$\pi_\cA$; in Koszul-dual coordinates this is $r_\cA(z)$.  The
MC equation is equivalent to $d^2_{\barB}=0$.

Part~(iii): $E_2$-collapse concentrates $H^{p,q}(\barB(\cA))$ in $q=0$,
so the bar cohomology algebra $\cA^!_{\mathrm{line}}$ is formal: all
$A_\infty$ operations $m_k$ for $k\geq 3$ vanish.  In DNP's framework,
tree-level $r$-matrix determines the full perturbative answer.
\end{proof}

\begin{construction}[Fusion as tensor product of $\cA^!_{\mathrm{line}}$-modules;
\ClaimStatusProvedHere]
\label{constr:fusion-from-bar}

$\Delta\colon\cA^!_{\mathrm{line}}\to

\Delta_z(x)\cdot(v_1\otimes v_2)

R(z_1{-}z_2)\cdot

\]

\[
N^{\ell_3}_{\ell_1\ell_2}
\;=\;

\end{construction}
