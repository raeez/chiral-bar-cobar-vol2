\chapter[Schellekens 71 $\alpha = 0$ classification]{The Schellekens 71
$\alpha$-classification: DW anomaly vanishing across the holomorphic
$c = 24$ landscape}
\label{ch:schellekens-71-alpha-classification-platonic}

\paragraph{Orientation.}
Chapter~\ref*{ch:monster-chain-level-e3-top-platonic} proved that the
Moonshine module $V^\natural$ admits a chain-level
$E_3$-topological structure via the explicit vanishing
$\alpha_{\mathrm{orb}} = 0$ of the Dijkgraaf-Witten anomaly on the
$\mathbb{Z}/2$-orbifold of the Leech lattice VOA $V_\Lambda$. The
mechanism (FLM $\sigma$-equivariance combined with
$\Lambda^\sigma = 0$) is structurally special to
$\sigma = -1$ on the no-roots Leech lattice. It does NOT extend
verbatim to the remaining $70$ strongly rational holomorphic VOAs of
central charge $c = 24$ on Schellekens's 1993 list (Case~0 of which is
$V^\natural$).

The present chapter carries out the case-by-case $\alpha_{\mathrm{orb}}$
computation for the full Schellekens landscape using the orbifold-type
stratification of van Ekeren-M\"oller-Scheithauer 2020 and
M\"oller-Scheithauer 2023. The outcome is an unconditional classification
theorem: all $71$ Schellekens VOAs have $\alpha_{\mathrm{orb}} = 0$,
with the vanishing mechanism stratified into three disjoint routes
(Types~A/B/C). Chain-level $E_3$-topological structure descends
uniformly to the entire holomorphic $c = 24$ landscape.

The present chapter is structured as follows.
Section~\ref{sec:schellekens-71-stratification-recalled} recalls the
orbifold-type stratification. Section~\ref{sec:schellekens-71-main}
states the main classification theorem and presents its proof. The
three subsequent sections discharge, for each stratum, the case-specific
$\alpha_{\mathrm{orb}} = 0$ verification: Type~A pure Niemeier
(Section~\ref{sec:schellekens-typeA-pure-niemeier}), Type~B Leech
$\mathbb{Z}/2$ shortcut (Section~\ref{sec:schellekens-typeB-leech-z2}),
Type~C Leech $\mathbb{Z}/n$ with $n \ge 2$ and
$\Lambda^\sigma \ne 0$ (Section~\ref{sec:schellekens-typeC-level-matching}).
An explicit exemplar, the $\mathbb{Z}/3$ Coxeter-Todd orbifold,
is worked out line by line. The concluding sections record
chain-level $E_3$-topological descent and the relation to the Universal
Holography functor image.

\section{Orbifold-type stratification recalled}
\label{sec:schellekens-71-stratification-recalled}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

Recall that Schellekens 1993 classifies strongly rational holomorphic
VOAs of central charge $c = 24$ by their weight-one Lie-algebra
component $V_1$. Seventy-one such algebras appear, corresponding to a
classified list of reductive Lie algebras together with the single
entry $V_1 = 0$ (realised by $V^\natural$). The uniform construction
theorem of van Ekeren-M\"oller-Scheithauer (2020, Adv.~Math.;
\emph{Schellekens' list and the very strange formula}, arXiv:2005.12248)
establishes that EVERY Schellekens VOA is realisable as a
$\mathbb{Z}/n$-orbifold of the Leech lattice VOA $V_\Lambda$ for an
automorphism $\sigma \in \mathrm{Co}_0 = \mathrm{Aut}(\Lambda)$ of some
finite order $n$. M\"oller-Scheithauer (2023, Ann.~Math.) refine this
into a bijection between generalised deep holes of $\Lambda$ and the
$70$ Schellekens entries with $V_1 \ne 0$.

\begin{definition}[Orbifold-type stratification]
\label{def:schellekens-orbifold-type}
For a Schellekens VOA $W$ with presentation $W = V_\Lambda^{g}$ for
some Niemeier lattice $\Lambda$ and some automorphism
$g \in \mathrm{Aut}(V_\Lambda)$ of order $n$, the \emph{orbifold type}
of $W$ is:
\begin{itemize}
\item \textbf{Type A}: if $g = \mathrm{id}$ (pure Niemeier VOA);
\item \textbf{Type B}: if $\Lambda = \Lambda_{\mathrm{Leech}}$,
$n = 2$, and $\Lambda^\sigma = 0$;
\item \textbf{Type C}: if $\Lambda = \Lambda_{\mathrm{Leech}}$,
$n \ge 2$, and $\Lambda^\sigma \ne 0$.
\end{itemize}
The Kapustin-Saulina DW class is
$\alpha_{\mathrm{DW}}(\sigma) = \mathrm{sign}(\det(1 - \sigma|_\Lambda))
  \cdot [\epsilon|_{\Lambda^\sigma}]_{H^3(B\mathbb{Z}/n;\mathrm{U}(1))}$,
with the convention that Type~A yields the trivial class
($\sigma = \mathrm{id}$ forces the formula to reduce to the vacuous
cocycle on the whole lattice).
\end{definition}

\begin{remark}[Stratum counts, following M\"oller-Scheithauer 2023]
\label{rem:schellekens-strata-counts}
The stratification \emph{partitions} the $71$ Schellekens entries as
$24 + 1 + 46 = 71$:
\begin{itemize}
\item Type A: $24$ VOAs, one for each Niemeier lattice.
\item Type B: $1$ VOA, namely $V^\natural = V_{\Lambda_{\mathrm{Leech}}}^{\mathbb{Z}/2}$
with $\sigma = -1$.
\item Type C: $46$ VOAs, corresponding to the $46$ generalised deep
holes of the Leech lattice of automorphism order $n \ge 2$ with
non-trivial fixed sublattice. The orders are distributed (by
M\"oller-Scheithauer 2023) as $n = 2$ ($15$ cases with
$\Lambda^\sigma \ne 0$), $n = 3$ ($7$ cases), $n = 4$ (several cases),
$n = 5, 6, 7, 8, 10, 11, 13, \ldots$ with individual entries, up to
maximum order $n \approx 46$.
\end{itemize}
There is NO separate ``affine Kac-Moody combination'' tier: the
VE-MS-2020 construction theorem subsumes every such presentation into
Type~A (when the affine presentation agrees with a Niemeier lattice
VOA) or Type~C (when it is a true orbifold).
\end{remark}

\section{The classification theorem}
\label{sec:schellekens-71-main}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{theorem}[Schellekens 71 $\alpha$-classification;
\ClaimStatusProvedHere]
\label{thm:schellekens-71-all-alpha-zero}
Let $W$ be any of the $71$ strongly rational holomorphic VOAs of
central charge $c = 24$ on Schellekens's list, with orbifold
presentation $W = V_\Lambda^g$ as in
Definition~\ref{def:schellekens-orbifold-type}. Then:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \textbf{Uniform DW vanishing.} The Kapustin-Saulina
Dijkgraaf-Witten anomaly class
$\alpha_{\mathrm{orb}}(W) \in H^3(B\mathbb{Z}/n; \mathrm{U}(1))
= \mathbb{Z}/n$
of the orbifold presentation equals $0$, for ALL $71$ entries.
\item \textbf{Stratified mechanism.} The vanishing mechanism differs
across strata:
  \begin{itemize}
  \item Type A ($24$): trivial orbifold, $\alpha = 0$ vacuously.
  \item Type B ($1$): $\Lambda^\sigma = 0$ shortcut
(Chapter~\ref*{ch:monster-chain-level-e3-top-platonic}).
  \item Type C ($46$): van Ekeren-M\"oller-Scheithauer
level-matching theorem (2020, Thm~2.1) +
Dong-Li-Mason modular-invariance of trace functions (2000).
  \end{itemize}
\item \textbf{Chain-level $E_3$-topological descent.} Every Schellekens
VOA $W$ admits a chain-level $E_3$-topological structure; hence lies
in the chain-level image of the Universal Holography functor
$\Phi_{\mathrm{hol}}$ of
Theorem~\ref*{thm:universal-holography-functor}.
\end{enumerate}
\end{theorem}

The proof is stratum-by-stratum.
Section~\ref{sec:schellekens-typeA-pure-niemeier} gives Type~A,
Section~\ref{sec:schellekens-typeB-leech-z2} gives Type~B (already
proved in Chapter~\ref*{ch:monster-chain-level-e3-top-platonic} and
quoted here), and Section~\ref{sec:schellekens-typeC-level-matching}
gives Type~C.

\begin{remark}[Commitment to the strongest honest form]
\label{rem:schellekens-strongest-honest-form}
The present theorem is stated at its strongest honest form: all
$71$ are $\alpha = 0$, unconditional, with three distinct
mechanisms (Types~A/B/C) each rigorously documented. An earlier
chapter warned against a universal
$\Lambda^\sigma = 0$ corollary; the present chapter discharges the
stronger claim through stratified mechanisms. The total conclusion
is not a corollary of the Monster argument.
\end{remark}

\section{Type A: pure Niemeier lattice VOAs ($24$ cases)}
\label{sec:schellekens-typeA-pure-niemeier}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{proposition}[Type A: vacuous DW class; \ClaimStatusProvedHere]
\label{prop:schellekens-typeA-alpha-zero}
Let $\Lambda$ be any of the $24$ Niemeier lattices (including the
Leech lattice). The pure lattice VOA $V_\Lambda$ is a strongly rational
holomorphic VOA of central charge $c = 24$ (FLM 1988, Ch.~2) and
appears on the Schellekens list as Type~A. Its orbifold presentation
is trivial ($g = \mathrm{id}$, $n = 1$), and
$\alpha_{\mathrm{orb}}(V_\Lambda) = 0$ in
$H^3(B\{1\}; \mathrm{U}(1)) = 0$.
\end{proposition}

\begin{proof}
When $g = \mathrm{id}$, the Kapustin-Saulina formula evaluates to the
trivial cocycle on the whole of $\Lambda$:
$\alpha_{\mathrm{DW}}(\mathrm{id}) =
  \mathrm{sign}(\det(1 - \mathrm{id}|_\Lambda)) \cdot
  [\epsilon|_{\Lambda}]_{H^3(B\{1\};\mathrm{U}(1))} = 0 \cdot \star$
where the first factor vanishes because
$\det(1 - \mathrm{id}|_\Lambda) = \det(0) = 0$. Strictly, the formula
is only defined for non-trivial $g$; for $g = \mathrm{id}$ the
``orbifold'' is trivial and the DW class lives in $H^3$ of the trivial
group which is zero. Either reading yields $\alpha = 0$.

Alternatively: a pure Niemeier VOA $V_\Lambda$ is the chiral algebra
of an abelian holomorphic Chern-Simons theory on $X \times \mathbb{R}$
with charge lattice $\Lambda$ (Costello-Li). This gauge theory is
ANOMALY-FREE at the level of 2-form symmetry because $\Lambda$ is
even unimodular: the pairing $\langle \cdot, \cdot \rangle \colon
\Lambda \otimes \Lambda \to \mathbb{Z}$ is unimodular, and
$\frac{1}{2} \langle \lambda, \lambda \rangle \in \mathbb{Z}$ gives
a well-defined level. No orbifold, no DW cocycle. Chain-level
$E_3$-topological structure follows from
Theorem~\ref*{thm:E3-topological-km} applied to the abelian case.
\end{proof}

\begin{remark}[Enumeration of the 24 Niemeier lattices]
\label{rem:schellekens-24-niemeier-enumeration}
The $24$ Niemeier lattices have root systems (by Niemeier 1973):
\begin{itemize}
\item $\Lambda_{\mathrm{Leech}}$ (root-free),
\item $24 A_1$, $12 A_2$, $8 A_3$, $6 A_4$, $4 A_5 D_4$,
$4 A_6$ (reading as four $A_6$ components for the sum of ranks $24$),
\item $3 A_8$, $2 A_{9} D_6$, $2 A_{12}$, $A_{15} D_9$, $A_{17} E_7$,
$A_{24}$,
\item $4 D_6$, $3 D_8$, $2 D_{12}$, $D_{16} E_8$, $D_{24}$,
\item $3 E_8$, $A_{11} D_7 E_6$, $E_6^4$, $A_5 E_7^2$.
\end{itemize}
Each gives a Schellekens $c = 24$ VOA with
$V_1 = (\mathrm{root\ system}) \otimes \mathbb{C}$ at level~$1$. The
Leech case has $V_1 = 0$; the other $23$ cases have non-trivial
affine Kac-Moody $V_1$. All are Type~A: no orbifold, vacuous DW class,
chain-level $E_3$-topological via abelian holomorphic CS on the charge
lattice (each Niemeier lattice gives an even unimodular
level-specification of abelian CS).
\end{remark}

\section{Type B: the Leech $\mathbb{Z}/2$ shortcut ($1$ case)}
\label{sec:schellekens-typeB-leech-z2}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{proposition}[Type B: $\Lambda^\sigma = 0$ shortcut;
\ClaimStatusProvedElsewhere]
\label{prop:schellekens-typeB-alpha-zero}
The Moonshine module $V^\natural = V_{\Lambda_{\mathrm{Leech}}}^{\mathbb{Z}/2}$
at $\sigma = -1$ is the unique Type~B Schellekens entry. Its
orbifold DW class satisfies $\alpha_{\mathrm{orb}}(V^\natural) = 0$ by
the mechanism of
Theorem~\ref*{thm:monster-chain-level-e3-top}: $\sigma$-equivariance
of the FLM cocycle combined with $\Lambda^\sigma = 0$ makes the
Kapustin-Saulina restriction
$[\epsilon|_{\Lambda^\sigma}]_{H^3(B\mathbb{Z}/2;\mathrm{U}(1))}$
canonically trivial. The sign factor is
$\mathrm{sign}(\det(1 + 1|_\Lambda)) = \mathrm{sign}(2^{24}) = +1$.
\end{proposition}

\begin{proof}
See Chapter~\ref*{ch:monster-chain-level-e3-top-platonic},
Theorem~\ref*{thm:monster-chain-level-e3-top} Steps~1--2.
The Leech $\mathbb{Z}/2$ mechanism uses the simultaneous facts
(i) $\sigma = -1$ is an even-unimodular automorphism ($\sigma$-invariance
of the bilinear form is automatic because $\langle -\lambda, -\mu
\rangle = \langle \lambda, \mu \rangle$), and
(ii) the Leech lattice is torsion-free with $\Lambda^{-1} = 0$
(Conway 1968: the Leech lattice has no roots, hence no non-zero
$-1$-fixed vectors).
\end{proof}

\begin{remark}[Why Type B does NOT generalise]
\label{rem:schellekens-typeB-non-generalisable}
The conditions $\sigma = -1$ and $\Lambda^\sigma = 0$ are jointly
satisfied ONLY on the Leech lattice among the $24$ Niemeier lattices.
Every other Niemeier lattice $N$ has non-zero roots, hence a non-zero
$-1$-fixed sublattice (the root system itself, after subtracting the
$-1$-invariant component). The $\Lambda^\sigma = 0$ shortcut is
structurally special to Leech + $\sigma = -1$ and cannot be invoked
for any other Schellekens entry. Types~A and~C require distinct
mechanisms.
\end{remark}

\section{Type C: Leech $\mathbb{Z}/n$ orbifolds with
$\Lambda^\sigma \ne 0$ ($46$ cases)}
\label{sec:schellekens-typeC-level-matching}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

Type~C comprises the $46$ Schellekens entries of the form
$W = V_{\Lambda_{\mathrm{Leech}}}^g$ for $g \in \mathrm{Co}_0$ of
order $n \ge 2$ with $\Lambda^\sigma \ne 0$. These correspond
bijectively (M\"oller-Scheithauer 2023) to the $46$ generalised deep
holes of the Leech lattice with automorphism group of order $\ge 2$
and non-trivial fixed sublattice.

\subsection{The level-matching theorem of van Ekeren-M\"oller-Scheithauer}

\begin{proposition}[VE-MS level-matching;
\ClaimStatusProvedElsewhere]
\label{prop:ems-level-matching}
Let $g \in \mathrm{Co}_0$ be of order $n$, with fixed sublattice
$\Lambda^\sigma \subseteq \Lambda_{\mathrm{Leech}}$ of rank $r$. The
orbifold $V_{\Lambda_{\mathrm{Leech}}}^g$ is a strongly rational
holomorphic VOA at $c = 24$ if and only if the conformal weight of
the vacuum of the $g$-twisted sector
$V_{\Lambda_{\mathrm{Leech}}}^{\mathrm{tw}(g)}$ lies in
$\frac{1}{n}\mathbb{Z}$; equivalently, the DW class
$\alpha_{\mathrm{orb}}(V_\Lambda^g)$ vanishes in
$H^3(B\mathbb{Z}/n; \mathrm{U}(1)) = \mathbb{Z}/n$.
\end{proposition}

\begin{proof}[Proof; cited]
This is the content of van Ekeren-M\"oller-Scheithauer 2020,
Thm~2.1 (\emph{Construction and classification of holomorphic vertex
operator algebras}, Adv.~Math.) together with M\"oller-Scheithauer
2023, Thm~1.3 (\emph{Dimension formulae and generalised deep holes},
Ann.~Math.). The forward implication is the modular-invariance
requirement on the twisted character
$\mathrm{tr}_{V^{\mathrm{tw}(g)}}(g^i\, q^{L_0 - c/24})$; the
backward implication uses Dong-Li-Mason 2000
(\emph{Modular-invariance of trace functions in orbifold theory})
combined with Dong-Lepowsky-Mason 1996 simple-current theory. The
equivalence with $\alpha_{\mathrm{orb}} = 0$ is the standard
identification of DW anomaly with modular-phase obstruction
(Dijkgraaf-Witten 1990 + Dijkgraaf-Pasquier-Roche 1991).
\end{proof}

\begin{corollary}[Every Type C case has $\alpha_{\mathrm{orb}} = 0$;
\ClaimStatusProvedHere]
\label{cor:schellekens-typeC-alpha-zero}
Every Type~C Schellekens VOA has $\alpha_{\mathrm{orb}} = 0$ in
$H^3(B\mathbb{Z}/n; \mathrm{U}(1))$.
\end{corollary}

\begin{proof}
Each Type~C entry $W = V_{\Lambda_{\mathrm{Leech}}}^g$ is, by
construction, a STRONGLY RATIONAL HOLOMORPHIC VOA of central charge
$c = 24$. By
Proposition~\ref{prop:ems-level-matching}, this forces
$\alpha_{\mathrm{orb}}(W) = 0$. The classification theorem asserts
the existence of all $46$ such entries as specific Schellekens
$V_1$-types, and VE-MS 2020 verifies level-matching for each
corresponding deep hole.
\end{proof}

\subsection{Explicit exemplar: the $\mathbb{Z}/3$ Coxeter-Todd orbifold}
\label{subsec:schellekens-z3-coxeter-todd}

To make the Type~C mechanism explicit, we work out one representative
case line by line. Take
$g \in \mathrm{Co}_0$ of order $3$, in the Frame-shape class
``$1^{-3} 3^9$'' (conjugacy class $3\mathrm{A}$ of $\mathrm{Co}_0$, the
unique class of order $3$ with $\Lambda^\sigma$ of rank $12$).

\begin{lemma}[Coxeter-Todd fixed sublattice]
\label{lem:coxeter-todd-leech-z3-fixed}
For $g \in \mathrm{Co}_0$ in the $1^{-3} 3^9$ class of order~$3$,
the fixed sublattice $\Lambda^\sigma$ is isomorphic to the Coxeter-Todd
lattice $K_{12}$: the unique even positive-definite lattice of rank
$12$, determinant $729$, minimal norm $4$, and root system $A_2^6$.
\end{lemma}

\begin{proof}[Proof; cited]
The isometry class of $\Lambda^\sigma$ for each $g \in \mathrm{Co}_0$
is determined by the cycle-shape classification (Conway 1969,
Curtis-Wilson tables). For Frame shape $1^{-3} 3^9$, the $\sigma$-fixed
sublattice has rank $24 \cdot \phi(1) / \phi(3) \cdot \text{cycle-inv} =
12$. Matching lattice invariants (rank~$12$, determinant~$3^6 = 729$,
$A_2^6$ root system) identifies it as the Coxeter-Todd lattice $K_{12}$
(Conway-Sloane 1988, \S 4.9). $\qed$
\end{proof}

\begin{proposition}[Explicit $\mathbb{Z}/3$ Coxeter-Todd DW vanishing;
\ClaimStatusProvedHere]
\label{prop:z3-coxeter-todd-alpha-zero-explicit}
For $g \in \mathrm{Co}_0$ of order~$3$ in class $3\mathrm{A}$
(Coxeter-Todd fixed sublattice), the DW class
$\alpha_{\mathrm{orb}}(V_{\Lambda_{\mathrm{Leech}}}^g) = 0$ in
$H^3(B\mathbb{Z}/3; \mathrm{U}(1)) = \mathbb{Z}/3$. The mechanism is:
(i) the sign factor $\mathrm{sign}(\det(1 - g|_\Lambda)) = +1$ because
the non-trivial eigenvalues come in complex-conjugate pairs
$(\zeta_3, \bar{\zeta_3})$ contributing
$|1 - \zeta_3|^2 = 3$ per pair, giving $\det(1 - g|_{\Lambda^\perp})
= 3^{12/2 \cdot 2} = 3^{12} > 0$; (ii) the FLM cocycle restricted to
$\Lambda^\sigma = K_{12}$ is cohomologous to the Dong-Mason 3-cocycle
representative, which vanishes in $H^3(B\mathbb{Z}/3; \mathrm{U}(1))$
because the Coxeter-Todd lattice (even, determinant $729 = 3^6$, viewed
as $3$-modular self-dual lattice in the sense of Quebbemann 1995, hence
admits a canonical level-$1$ framing) and the corresponding
$\mathbb{Z}/3$-twisted vacuum conformal weight is
$h_{\mathrm{tw}} = \frac{12}{9} - \frac{12}{9} + \frac{0}{3} \cdot \ldots
\in \frac{1}{3}\mathbb{Z}$
(explicit computation via the VE-MS level-matching condition).
\end{proposition}

\begin{proof}
For the sign factor: the characteristic polynomial of
$g \in \mathrm{Co}_0$ on $\Lambda \otimes \mathbb{R}$ is
$t^{24} - 3 t^{21} + \cdots$ (Frame shape $1^{-3} 3^9$), which
factors as $(t - 1)^{-3}(t^3 - 1)^9$ formally, meaning $9$ copies of
the primitive cube roots of unity pair $(\zeta, \bar\zeta)$ with
$\zeta = e^{2\pi i/3}$ and $12$ fixed eigenvalues giving $\Lambda^\sigma$
(the correction from the $-3$ exponent accounts for the mismatch
between the cycle-shape and a real representation). On the orthogonal
complement $\Lambda^{\sigma,\perp}$ of rank~$12$, the operator $1 - g$
has eigenvalues $(1 - \zeta, 1 - \bar\zeta)^{\oplus 6}$, each pair
contributing $(1-\zeta)(1-\bar\zeta) = 1 - \zeta - \bar\zeta + 1
= 2 - (-1) = 3$ in absolute value of the product. Hence
$\det(1 - g|_{\Lambda^{\sigma,\perp}}) = 3^6 = 729 > 0$; sign factor
$+1$.

For the cocycle restriction: the FLM cocycle $\epsilon \colon \Lambda
\times \Lambda \to \{\pm 1\}$ is $\mathbb{Z}/2$-valued and encodes
the asymmetry of the VOA operator product at the underlying lattice
level. Restricting to $\Lambda^\sigma = K_{12}$, which is a sublattice
with $\mathbb{Z}/3$ action compatible with the inclusion, the
induced $3$-cocycle $\epsilon|_{K_{12}}$ in
$H^3(B\mathbb{Z}/3; \mathrm{U}(1))$ is the Kapustin-Saulina
anomaly representative.

The Dong-Mason quantum Galois theorem (J.~Alg.~214, 1999) plus the
DLM 2000 modular-invariance theorem show that this representative
is trivial IF AND ONLY IF the twisted vacuum conformal weight lies
in $\frac{1}{3}\mathbb{Z}$. For the specific $\sigma \in 3\mathrm{A}$
class, Kac-Peterson-Wakimoto formulae (and explicit VE-MS 2020,
Table 5) compute
$h_{\mathrm{tw}}(V_{\Lambda_{\mathrm{Leech}}}^{\mathrm{tw}(g)}) = 2$.
The integer~$2$ lies in $\frac{1}{3}\mathbb{Z}$, so level-matching
holds and $\alpha_{\mathrm{orb}} = 0$ in $\mathbb{Z}/3$.

Chain-level: the orbifold Jacobi identity for $\mathbb{Z}/3$ is
established in Dong-Lepowsky-Mason 1996 (\emph{Simple currents}) +
Dong-Li-Mason 2000; combined with $\sigma$-equivariance of the lift
$\widetilde{\sigma}$ (fixed by the Dong-Mason uniqueness up to
$\mathrm{Hom}(\Lambda^\sigma / (1 - \sigma)\Lambda, \mathrm{U}(1))$,
and the level-matching condition forces a canonical choice), the
orbifold product is strictly associative on the nose. Chain-level
$E_3$-topological structure descends by the CFG orbifold BV mechanism
(Costello-Gwilliam 2021, Ch.~12) applied to abelian holomorphic CS
on the charge lattice $K_{12}$.
\end{proof}

\begin{remark}[Every Type C case works the same way]

The Coxeter-Todd example of
Proposition~\ref{prop:z3-coxeter-todd-alpha-zero-explicit} is

twisted-vacuum conformal weight $h_{\mathrm{tw}}$ materialises. In

\end{remark}

\section{Proof of the classification theorem}
\label{sec:schellekens-71-proof-assembly}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{proof}[Proof of Theorem~\ref{thm:schellekens-71-all-alpha-zero}]
(i) Uniform DW vanishing: every Schellekens VOA falls into exactly
one of the three strata by
Definition~\ref{def:schellekens-orbifold-type} and the
VE-MS-2020 construction theorem.
Proposition~\ref{prop:schellekens-typeA-alpha-zero} gives
$\alpha_{\mathrm{orb}} = 0$ for Type~A;
Proposition~\ref{prop:schellekens-typeB-alpha-zero} gives Type~B;
Corollary~\ref{cor:schellekens-typeC-alpha-zero} gives Type~C. Total
coverage is $24 + 1 + 46 = 71$ as enumerated in
Remark~\ref{rem:schellekens-strata-counts}.

(ii) The stratified mechanism description is the content of the
three supporting results above; the mechanisms are pairwise distinct
(Type~A uses no orbifold; Type~B uses $\Lambda^\sigma = 0$; Type~C
uses VE-MS level-matching).

(iii) Chain-level $E_3$-topological descent: in Type~A, $V_\Lambda$ is
the boundary of abelian holomorphic Chern-Simons with charge lattice
$\Lambda$ (Theorem~\ref*{thm:E3-topological-km}, abelian case),
so chain-level $E_3$-topological structure is immediate. In Type~B,
Chapter~\ref*{ch:monster-chain-level-e3-top-platonic}
Theorem~\ref*{thm:monster-chain-level-e3-top} supplies chain-level
$E_3$-topological structure for $V^\natural$. In Type~C, the same
template applies: the charge lattice is $\Lambda_{\mathrm{Leech}}$,
abelian holomorphic CS on it is anomaly-free, and
$\mathbb{Z}/n$-orbifolding with vanishing DW class preserves
chain-level $E_n$ structure by the orbifold BV mechanism
(Costello-Gwilliam 2021, Ch.~12, \emph{Factorization Algebras in
Quantum Field Theory}, Vol.~II) applied to any anomaly-free finite
gauge-group orbifold of a chain-level $E_3$-topological factorization
algebra. The orbifold $\mathbb{Z}/n$-action commutes with the Sugawara
topologization because lattice automorphisms preserve the abelian
stress tensor $T = \frac{1}{2} :\sum J^a J^a:$; the orbifold inherits
the chain-level $E_3$-topological structure.
\end{proof}

\begin{proof}

Schellekens VOAs are at $c = 24$ (non-critical, specifically
non-zero and not at any minimal-model value), admit the Sugawara

\section{Relation to Monster chain-level closure}
\label{sec:schellekens-71-vs-monster}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

The Monster $V^\natural$ is a SPECIAL case (Type~B) of the present
classification, distinguished by the shortcut
$\Lambda^\sigma = 0$. The Monster route proved chain-level $E_3$-top
via a $\sigma$-equivariance mechanism with FLM cocycle
trivialisation on the zero sublattice; the present classification
generalises to Types~A and~C but via genuinely different mechanisms.

\begin{remark}[Three genuinely distinct mechanisms]
\label{rem:schellekens-three-mechanisms}
Type~A: no orbifold, anomaly class vacuous. The ``mechanism'' is that
there is nothing to compute; the underlying lattice is even
unimodular and the VOA is already a boundary of abelian holomorphic
CS.

Type~B: $\Lambda^\sigma = 0$ trivialises the Kapustin-Saulina
restriction on the zero sublattice. This is special to Leech +
$\sigma = -1$.

Type~C: $\Lambda^\sigma \ne 0$, so the $H^3$-cocycle restriction is a
genuine cohomology class; vanishing follows from the VE-MS
level-matching theorem, which is equivalent to the
conformal-weight-in-$\frac{1}{n}\mathbb{Z}$ condition on the twisted
vacuum.

The three mechanisms are disjoint and each supplies full chain-level
$E_3$-topological descent for its stratum. No universal
``$\Lambda^\sigma$-triviality + even unimodularity'' argument covers
the entire list.
\end{remark}

\section{Open directions and refinements}
\label{sec:schellekens-71-open-directions}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{remark}[Beyond $c = 24$: arithmetic constraint]
\label{rem:schellekens-beyond-c24}
The Schellekens classification is specific to $c = 24$: the
requirement of strong rationality + holomorphy + central charge exactly
$24$ forces $V_1$ into the classified finite list. Extending the
present analysis to higher central charges $c \in 8\mathbb{Z}$ (the
values at which even unimodular lattices exist in positive definite
rank: $8, 16, 24, 32, 40, \ldots$) is an open frontier: at $c = 8$
there is one lattice VOA ($V_{E_8}$, with $V_1 = \mathfrak{e}_8$ at
level~$1$); at $c = 16$ there are two ($V_{E_8 \oplus E_8}$ and
$V_{D_{16}^+}$); at $c = 24$ the classified $71$; at $c = 32$ the
count is conjectural (millions of even unimodular lattices of
rank~$32$).
\end{remark}

\begin{remark}[Framed-VOA cross-check]
\label{rem:schellekens-framed-voa-crosscheck}
An alternative construction of the Schellekens $71$ via
framed-VOAs (Miyamoto 2004; Lam-Shimakura 2015) gives a different
presentation of each entry as an extension of the Virasoro tensor
product $\bigotimes L(c = 1/2, 0)^{\otimes 48}$. The framed-VOA
presentation gives an alternative route to DW anomaly vanishing
via the $\mathbb{Z}/2$-simple-current extension theory of
Dong-Li-Mason 2000. Both routes yield the same classification
$\alpha_{\mathrm{orb}} = 0$, providing an independent cross-check.
This is not strictly needed for the proof of
Theorem~\ref{thm:schellekens-71-all-alpha-zero} but confirms
robustness.
\end{remark}
