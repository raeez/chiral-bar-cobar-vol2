\subsection{The MC5 bridge at genus~$0$}

\begin{theorem}[Feynman $m_k$ = bar differential at genus~$0$;
\ClaimStatusProvedHere]
\label{thm:mc5-genus-zero-bridge}
Assume \textrm{(H1)--(H4)}.  Let $\cA$ be the BV-BRST complex of
a 3d holomorphic-topological theory on $\C \times \R$.  Then the
$A_\infty$ operations $\{m_k\}_{k \geq 1}$ computed from Feynman
diagram integrals over $\FM_k(\C) \times \Conf_k^{<}(\R)$
(Sections~\ref{sec:bv-construction}--\ref{sec:FM_calculus})
coincide with the cogenerator projection of the bar differential
$d_{\barB}$ of Volume~I at genus~$0$.

Precisely: Volume~I's bar complex
$\barB(\cA) = T^c(s^{-1}\bar{\cA})$ is the cofree conilpotent
coalgebra, and $d_{\barB}$ is a coderivation determined by its
cogenerator projection
$\pi_1 \circ d_{\barB} \colon T^c(s^{-1}\bar{\cA}) \to
s^{-1}\bar{\cA}$.  The restriction of this projection to
$\barB^k(\cA) = (s^{-1}\bar{\cA})^{\otimes k}$ is a multilinear
map $\cA^{\otimes k} \to \cA$.  This multilinear map
equals~$m_k$.
\end{theorem}

\begin{proof}
Three ingredients combine.

\textbf{Step 1} (Recognition).
By Theorem~\ref{thm:recognition}, the BV observables of the
3d HT theory form an algebra over
$C_\ast(W(\SCchtop))$.  The bar complex
$\barB(\cA)$ is therefore a coalgebra over the bar cooperad
$\mathbf{B}(\SCchtop)$, and its differential is a coderivation
of the cofree coalgebra
(Theorem~\ref{thm:bar-swiss-cheese}).

\textbf{Step 2} (Cofree rigidity).
On a cofree conilpotent coalgebra $T^c(V)$, every coderivation
$D$ is uniquely determined by the composite
$\pi_1 \circ D \colon T^c(V) \to V$ (the cogenerator projection).
This is a standard fact of coalgebra theory
\textup{(}Volume~I, Proposition~\ref*{prop:cofree-coderivation}\textup{)}.
In particular, $d_{\barB}$ is the unique coderivation whose
projection to cogenerators in arity~$k$ is the residue map
\[
\Res_{D_{1\cdots k}} \colon
\Omega^\bullet_{\log}(\FM_k(\C)) \otimes \cA^{\otimes k}
\;\longrightarrow\; \cA
\]
along the total collision divisor in $\FM_k(\C)$.

\textbf{Step 3} (Physical identification).
Under hypotheses (H1)--(H4), the Feynman diagram integral defining
$m_k(a_1, \ldots, a_k)$ is precisely the integral of the
renormalized propagator product over $\FM_k(\C) \times
\Conf_k^{<}(\R)$.  By (H3), the integrand extends to a
logarithmic form on $\FM_k(\C)$; by (H2), the $\R$-integral
factors as a step-function product encoding time ordering.  The
residue of this logarithmic form along the total collision
divisor $D_{1\cdots k}$ extracts the $k$-point interaction, which
is the coefficient of the OPE singularity.  This residue is
$m_k$ by definition of the $A_\infty$ operations
(Section~\ref{sec:FM_calculus}).  It is also the cogenerator
projection of $d_{\barB}$ by Volume~I's construction of the bar
differential from FM residues
(Volume~I, Definition~\ref*{def:bar-differential}).
Since both sides equal the same FM residue,
$\pi_1 \circ d_{\barB}|_{\barB^k} = m_k$.
\end{proof}

\begin{remark}[Genus~$g \geq 1$ and MC5]
At genus~$g \geq 1$, the bar differential acquires curvature
$d_{\barB}^2 = \kappa(\cA) \cdot \omega_g$ (Volume~I, Theorem~D).
The corresponding statement for the Feynman-diagrammatic
$m_k$ is that the higher $A_\infty$ operations persist at the
chain level and encode the genus-$g$ modular deformation.
Extending the identification of Theorem~\ref{thm:mc5-genus-zero-bridge}
to all genera is the content of MC5 in Volume~I and remains open:
the analytic extension of the propagator comparison from genus~$0$
to genus~$g \geq 1$ requires Costello's renormalization framework
and the Hodge-theoretic clutching data of the genus tower.
\end{remark}

\subsection{The PVA-Coisson bridge}

\begin{theorem}[PVA descent at $X = \mathrm{pt}$ recovers Coisson
structure; \ClaimStatusProvedHere]
\label{thm:pva-coisson-bridge}
Let $\cA$ be an $A_\infty$ chiral algebra satisfying
\textrm{(H1)--(H4)}, and let $H = H^\bullet(\cA, Q)$ be its
$(-1)$-shifted PVA (Theorem~\textup{\ref{thm:cohomology_PVA}}).
Define the \emph{point specialization} by setting all derivatives
$\partial = 0$:
\[
H_{\mathrm{pt}}
:= H / (\partial H),
\qquad
\{a, b\}_{\mathrm{Poisson}} := \{a {}_\lambda b\}|_{\lambda = 0}.
\]
Then $(H_{\mathrm{pt}},\, \cdot,\, \{-,-\}_{\mathrm{Poisson}})$
is a $(-1)$-shifted Poisson algebra \textup(= Coisson algebra in
the sense of Beilinson--Drinfeld\textup).
\end{theorem}

\begin{proof}
The PVA axioms on $H$ specialize as follows.

\emph{Commutativity and associativity.}
The product $\cdot$ on $H$ is graded commutative and associative
(Lemma~\ref{lem:commutativity},
Proposition~\ref{prop:associativity}).  Quotienting by $\partial H$
preserves these properties.

\emph{Lie bracket.}
The $\lambda$-bracket $\{a {}_\lambda b\} = \sum_{n \geq 0}
(a_{(n)} b)\, \lambda^n / n!$ encodes infinitely many products
$a_{(n)} b$.  At $\lambda = 0$, only the $n = 0$ term survives:
$\{a, b\}_{\mathrm{Poisson}} = a_{(0)} b$.  The PVA skewsymmetry
(Proposition~\ref{prop:sesqui-skew}) at $\lambda = 0$ gives
$\{a, b\} = -(-1)^{(|a|+1)(|b|+1)}\{b, a\}$, which is graded
antisymmetry for a $(-1)$-shifted bracket.  The PVA Jacobi
identity (Theorem~\ref{thm:Jacobi}) at $\lambda = \mu = 0$ gives
the Jacobi identity for~$\{-,-\}_{\mathrm{Poisson}}$.

\emph{Leibniz rule.}
The PVA Leibniz rule (Theorem~\ref{thm:Leibniz}) at $\lambda = 0$
gives the Poisson Leibniz rule:
$\{a, b \cdot c\} = \{a, b\} \cdot c +
(-1)^{(|a|+1)|b|}\, b \cdot \{a, c\}$.

\emph{Sesquilinearity collapse.}
PVA sesquilinearity (Proposition~\ref{prop:sesqui-skew}) says
$\{\partial a {}_\lambda b\} = -\lambda\, \{a {}_\lambda b\}$.
At $\lambda = 0$: $\{\partial a, b\}_{\mathrm{Poisson}} = 0$.
Therefore $\partial H \subset \ker\{-,-\}_{\mathrm{Poisson}}$
and the bracket descends to the quotient $H_{\mathrm{pt}}$.
\end{proof}
