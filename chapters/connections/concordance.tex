\subsection{The MC5 bridge at genus~$0$}

\begin{theorem}[Feynman $m_k$ = bar differential at genus~$0$;
\ClaimStatusProvedHere]
% label removed: thm:mc5-genus-zero-bridge
Assume \textrm{(H1)--(H4)}. Let $\cA$ be the BV-BRST complex of
a 3d holomorphic-topological theory on $\C \times \R$. Then the
$A_\infty$ operations $\{m_k\}_{k \geq 1}$ computed from Feynman
diagram integrals over $\FM_k(\C) \times \Conf_k^{<}(\R)$
(Sections~\ref{sec:BV_construction}--\ref{sec:FM_calculus})
coincide with the cogenerator projection of the bar differential
$d_{\barB}$ of Volume~I at genus~$0$.

Precisely: Volume~I's bar complex
$\barB(\cA) = T^c(s^{-1}\bar{\cA})$ is the cofree conilpotent
coalgebra, and $d_{\barB}$ is a coderivation determined by its
cogenerator projection
$\pi_1 \circ d_{\barB} \colon T^c(s^{-1}\bar{\cA}) \to
s^{-1}\bar{\cA}$. The restriction of this projection to
$\barB^k(\cA) = (s^{-1}\bar{\cA})^{\otimes k}$ is a multilinear
map $\cA^{\otimes k} \to \cA$. This multilinear map
equals~$m_k$.
\end{theorem}

\begin{proof}
Three ingredients combine.

\textbf{Step 1} (Recognition).
By Theorem~\ref{thm:recognition}, the BV observables of the
3d HT theory form an algebra over
$C_\ast(W(\SCchtop))$. The bar complex
$\barB(\cA)$ is therefore an $E_1$ dg coassociative coalgebra
whose differential is a coderivation
of the cofree coalgebra
(Theorem~\ref{thm:bar-swiss-cheese}).

\textbf{Step 2} (Cofree rigidity).
On a cofree conilpotent coalgebra $T^c(V)$, every coderivation
$D$ is uniquely determined by the composite
$\pi_1 \circ D \colon T^c(V) \to V$ (the cogenerator projection).
This is a standard fact of coalgebra theory
\textup{(}Volume~I, Proposition~\ref*{prop:cofree-coderivation}\textup{)}.
In particular, $d_{\barB}$ is the unique coderivation whose
projection to cogenerators in degree~$k$ is the residue map
\[
\Res_{D_{1\cdots k}} \colon
\Omega^\bullet_{\log}(\FM_k(\C)) \otimes \cA^{\otimes k}
\;\longrightarrow\; \cA
\]
along the total collision divisor in $\FM_k(\C)$.

\textbf{Step 3} (Physical identification).
Under hypotheses (H1)--(H4), the Feynman diagram integral defining
$m_k(a_1, \ldots, a_k)$ is precisely the integral of the
renormalized propagator product over $\FM_k(\C) \times
\Conf_k^{<}(\R)$. By (H3), the integrand extends to a
logarithmic form on $\FM_k(\C)$; by (H2), the $\R$-integral
factors as a step-function product encoding time ordering. The
residue of this logarithmic form along the total collision
divisor $D_{1\cdots k}$ extracts the $k$-point interaction, which
is the coefficient of the OPE singularity. This residue is
$m_k$ by definition of the $A_\infty$ operations
(Section~\ref{sec:FM_calculus}). It is also the cogenerator
projection of $d_{\barB}$ by Volume~I's construction of the bar
differential from FM residues
(Volume~I, Definition~\ref*{def:bar-differential}).
Since both sides equal the same FM residue,
$\pi_1 \circ d_{\barB}|_{\barB^k} = m_k$.
\end{proof}

\subsection{The MC5 bridge at genus~$1$}

\begin{theorem}[Feynman curvature = modular characteristic at genus~$1$;
\ClaimStatusProvedHere]
% label removed: thm:mc5-genus-one-bridge
Assume \textrm{(H1)--(H4)} on $E_\tau \times \R$, where
$E_\tau = \C/(\Z + \tau\Z)$ is the elliptic curve with period
$\tau \in \mathbb{H}$. Let $\cA$ be the BV-BRST complex of a 3d
holomorphic-topological theory on $E_\tau \times \R$, and let
$\{m_k^{E_\tau}\}_{k \geq 1}$ denote the $A_\infty$ operations
computed from Feynman diagram integrals over
$\FM_k(E_\tau) \times \Conf_k^{<}(\R)$ using the Weierstrass
sigma propagator.

Then:
\begin{enumerate}[label=\textup{(\roman*)}]
\item The cogenerator projection of the fiberwise bar differential
 $\dfib$ coincides with the Feynman-defined operations:
 $\pi_1 \circ \dfib|_{\barB^k(\cA)} = m_k^{E_\tau}$.
\item The fiberwise bar differential satisfies
 \begin{equation}
 % label removed: eq:mc5-genus1-curvature
 \dfib^{\,2} = \kappa(\cA) \cdot \omega_1,
 \end{equation}
 where $\omega_1 = \frac{i}{2\operatorname{Im}(\tau)}\,dz \wedge
 d\bar{z}$ is the Arakelov $(1,1)$-form and $\kappa(\cA)$ is the
 modular characteristic of Volume~I. The curvature arises from
 the Arnold defect on the torus: the quasi-periodicity of the
 Weierstrass sigma function breaks the genus-$0$ Arnold relation.
\item The total corrected differential
 $\Dg{1} = d_0 + F_1(\cA) \cdot d_1$, with period correction
 \[
 F_1(\cA) = \frac{\kappa(\cA)}{24},
 \]
 satisfies $\Dg{1}^{\,2} = 0$.
\end{enumerate}
In particular, the $d' = 1$ formality failure for holomorphic-topological
theories is the modular
characteristic: the higher $A_\infty$ operations at genus~$1$ encode
the universal curvature $\kappa(\cA) \cdot \omega_1$.
\end{theorem}

\begin{proof}
The argument extends the genus-$0$ proof
(Remark~\ref{rem:mc5-genus-zero-bridge}) by replacing the
genus-$0$ propagator with the elliptic propagator and tracking the
Arnold defect.

\textbf{Step~1} (Elliptic propagator).
On $E_\tau$, the propagator $K^{\C}(z_{12}) = dz_1/(z_1 - z_2)$ is
replaced by
\[
K^{E_\tau}(z_{12})
= \zeta(z_1 - z_2 \mid \tau)\,dz_1,
\]
where $\zeta(z|\tau) = \partial_z \log\sigma(z|\tau)$ is the
Weierstrass zeta function. Near $z = 0$,
$\zeta(z|\tau) = 1/z + O(z)$, so the local pole structure is
\emph{identical} to genus~$0$. In particular, the residue
$\Res_{z_1 = z_2} K^{E_\tau}(z_{12}) = 1$, and the FM residue
along collision divisors $D_S \subset \FM_k(E_\tau)$ extracts the
same OPE coefficients as at genus~$0$.

\textbf{Step~2} (Cogenerator identification).
The three ingredients of the genus-$0$ proof carry over verbatim:
\begin{enumerate}[label=(\alph*)]
\item \emph{Recognition:} The bar complex $\barB(\cA)$ is a cofree
 conilpotent coalgebra, and $\dfib$ is a coderivation
 (Theorem~\ref{thm:curved-swiss-cheese}(i)).
\item \emph{Cofree rigidity:} $\dfib$ is determined by its
 cogenerator projection $\pi_1 \circ \dfib$
 (Volume~I, Proposition~\ref*{prop:cofree-coderivation}).
\item \emph{Physical identification:} The FM residue of the
 elliptic propagator product along
 $D_{1\cdots k} \subset \FM_k(E_\tau)$ equals $m_k^{E_\tau}$
 (by~(H2)--(H3) on $E_\tau$), and also equals the cogenerator
 projection of $\dfib$ (by Volume~I's construction).
\end{enumerate}
This establishes~(i).

\textbf{Step~3} (Arnold defect and curvature).
At genus~$0$, the Arnold relation
$\eta_{12} \wedge \eta_{23} + \eta_{23} \wedge \eta_{31}
+ \eta_{31} \wedge \eta_{12} = 0$ (where
$\eta_{ij} = d\log(z_i - z_j)$) ensures $\dfib^{\,2} = 0$.
On $E_\tau$, replace $\eta_{ij}$ by
$\eta_{ij}^{(1)} := d\log\sigma(z_i - z_j|\tau)$. The
quasi-periodicity of the Weierstrass zeta function
($\zeta(z + 1|\tau) = \zeta(z|\tau) + 2\eta_1$,
$\zeta(z + \tau|\tau) = \zeta(z|\tau) + 2\eta_\tau$,
with the Legendre relation
$\tau\eta_1 - \eta_\tau = \pi i$) breaks the Arnold relation:
\[
\eta_{12}^{(1)} \wedge \eta_{23}^{(1)}
+ \eta_{23}^{(1)} \wedge \eta_{31}^{(1)}
+ \eta_{31}^{(1)} \wedge \eta_{12}^{(1)}
= E_2(\tau) \cdot (dz_1 - dz_2) \wedge (dz_2 - dz_3),
\]
where $E_2(\tau) = 1 - 24\sum_{n \geq 1} \sigma_1(n)\,q^n$ is the
weight-$2$ Eisenstein series
(Volume~I, Theorem~\ref*{V1-thm:arnold-genus1}).

This defect enters $\dfib^{\,2}$ when the two-fold composition of
the bar differential produces a three-point integral: the curvature
is the contraction of the Arnold defect with the OPE data of~$\cA$.
The contraction with $(dz_1 - dz_2) \wedge (dz_2 - dz_3)$ extracts
the \emph{highest-pole coefficient} of the OPE, which is, by
definition, the modular characteristic $\kappa(\cA)$
(Volume~I, Theorem~\ref*{V1-thm:universal-generating-function}).
The Arakelov normalization of $E_2(\tau)$ gives the $(1,1)$-form
$\omega_1 = \frac{i}{2\operatorname{Im}(\tau)}\,dz \wedge d\bar{z}$.
This establishes~(ii).

\textbf{Step~4} (Period correction).
The period correction $F_1(\cA) = \kappa(\cA)/24$ and the proof
that $\Dg{1}^{\,2} = 0$ follow from Volume~I
(Theorem~\ref*{V1-thm:genus1-universal-curvature}): the
Faber--Pandharipande constant
$\lambda_1^{FP} = \int_{\overline{\mathcal{M}}_{1,1}} \psi \cdot
\lambda_1 = 1/24$ normalizes the genus-$1$ term.
\end{proof}

\begin{remark}[Curvature-braiding coupling at genus~$1$]
% label removed: rem:curvature-braiding-genus1
The curvature $\dfib^{\,2} = \kappa(\cA)\cdot\omega_1$ interacts
with the spectral $R$-matrix when the latter is lifted from
$\C$ to $E_\tau$. The nature of the interaction is controlled by
the pole structure of the $\lambda$-bracket: decoupling when the
Coisson bracket $\{a{}_0 b\}$ vanishes (Heisenberg, lattice

$B$-cycle quasi-periodicity of the Weierstrass zeta function, which
produces the Arnold defect (curvature) and the braiding monodromy
simultaneously.
See Theorem~\ref{thm:elliptic-spectral-dichotomy} for the precise
statement.
\end{remark}

\end{remark}

structure; \ClaimStatusProvedHere]
% label removed: thm:pva-coisson-bridge
Let $\cA$ be an $A_\infty$ chiral algebra satisfying
\textrm{(H1)--(H4)}, and let $H = H^\bullet(\cA, Q)$ be its
$(-1)$-shifted PVA (Theorem~\textup{\ref{thm:cohomology_PVA}}).
Define the \emph{point specialization} by setting all derivatives
$\partial = 0$:
\[
H_{\mathrm{pt}}
:= H / (\partial H),
\qquad
\{a, b\}_{\mathrm{Poisson}} := \{a {}_\lambda b\}|_{\lambda = 0}.
\]
Then $(H_{\mathrm{pt}},\, \cdot,\, \{-,-\}_{\mathrm{Poisson}})$
is a $(-1)$-shifted Poisson algebra \textup(= Coisson algebra in
the sense of Beilinson--Drinfeld\textup).
\end{theorem}

\begin{proof}
The PVA axioms on $H$ specialize as follows.

\emph{Commutativity and associativity.}
The product $\cdot$ on $H$ is graded commutative and associative
(Proposition~\ref{prop:product-commutative},
Proposition~\ref{prop:product-associative}). Quotienting by $\partial H$
preserves these properties.

\emph{Lie bracket.}
The $\lambda$-bracket $\{a {}_\lambda b\} = \sum_{n \geq 0}
(a_{(n)} b)\, \lambda^n / n!$ encodes infinitely many products
$a_{(n)} b$. At $\lambda = 0$, only the $n = 0$ term survives:
$\{a, b\}_{\mathrm{Poisson}} = a_{(0)} b$. The PVA skewsymmetry
(Proposition~\ref{prop:PVA2_proof}) at $\lambda = 0$ gives
$\{a, b\} = -(-1)^{(|a|+1)(|b|+1)}\{b, a\}$, which is graded
antisymmetry for a $(-1)$-shifted bracket. The PVA Jacobi
identity (Theorem~\ref{thm:Jacobi}) at $\lambda = \mu = 0$ gives
the Jacobi identity for~$\{-,-\}_{\mathrm{Poisson}}$.

\emph{Leibniz rule.}
The PVA Leibniz rule (Theorem~\ref{prop:Leibniz}) at $\lambda = 0$
gives the Poisson Leibniz rule:
$\{a, b \cdot c\} = \{a, b\} \cdot c +
(-1)^{(|a|+1)|b|}\, b \cdot \{a, c\}$.

\emph{Sesquilinearity collapse.}
PVA sesquilinearity (Proposition~\ref{prop:PVA2_proof}) says
$\{\partial a {}_\lambda b\} = -\lambda\, \{a {}_\lambda b\}$.
At $\lambda = 0$: $\{\partial a, b\}_{\mathrm{Poisson}} = 0$.
Therefore $\partial H \subset \ker\{-,-\}_{\mathrm{Poisson}}$
and the bracket descends to the quotient $H_{\mathrm{pt}}$.
\end{proof}

\subsection{The four-stage open/closed architecture}

The passage from a boundary chiral algebra~$\cA$ to its bulk
observables and modular invariants proceeds through four stages,
each depending on its predecessor.

\begin{enumerate}[label=\textup{Stage \arabic*.}, leftmargin=3em]
\item \textbf{Local one-colour theorem.}
 An $\Ainf$ chiral algebra determines, via the brace algebra on
 chiral Hochschild cochains, a chiral derived center
 $\cZ^{\mathrm{der}}_{\mathrm{ch}}(\cA)$ that serves as the
 universal bulk algebra. \emph{Status: proved}
 (Theorem~\ref{thm:bulk_hochschild}).

\item \textbf{Open primitive.}
 The primitive object is the open-sector factorization dg-category
 $\cC_{\mathrm{op}}$ on a tangential log curve $(X,D,\tau)$
 (Definition~\ref{def:bordified-curve}); the boundary algebra
 $A_b = \mathrm{End}(b)$ is a chart, not an essence (Morita
 invariance).
 \emph{Status: proved}
 (Vol~I, Theorem~\ref*{V1-thm:thqg-local-global-bridge}).

\item \textbf{Globalization.}
 Tangential log curves $(X,D,\tau)$ with bordered FM
 compactification provide the geometric arena for the local-global
 bridge.
 \emph{Status: proved locally; global extension modest.}

\item \textbf{Modularization.}
 Trace and clutching on the open sector produce modular invariants.
 The annulus trace
 $\mathrm{Tr}_\cA \simeq \mathrm{HH}_*(\cA)$ is the first modular
 shadow.
 \emph{Status: annulus trace proved; full modular cooperad is a
 programme.}
\end{enumerate}

\noindent
The open/closed MC element
$\Theta^{\mathrm{oc}} = \Theta_\cA + \Sigma\,\mu^{M_j}$
packages all four stages.
Its projections recover every construction in this volume:
the closed-colour projection is the bar differential
(Vol~I, Theorem~A);
the open-colour projection is the $E_1$ coproduct and spectral
$R$-matrix;
the genus tower is its $\hbar$-expansion
(Vol~I, Theorem~D);
and the bulk-boundary-line triangle is its two-colour decomposition.

\noindent\textbf{Seven negative principles.}
\begin{enumerate}[label=\textup{(N\arabic*)}]
\item The bar complex is \emph{not} the bulk.
 $\barB(\cA)$ classifies twisting morphisms; the bulk observables
 are the chiral derived center $\cZ^{\mathrm{der}}_{\mathrm{ch}}(\cA)$.
\item The cobar does \emph{not} produce the Koszul dual.
 $\Omegach(\barB(\cA)) \simeq \cA$ (inversion).
 The Koszul dual $\cA^!$ is obtained by Verdier duality.
\item The boundary algebra is \emph{not} intrinsic.
 $A_b = \mathrm{End}(b)$ is a chart of
 $\cC_{\mathrm{op}}$; different boundary conditions give
 Morita-equivalent charts.
\item The global open sector does \emph{not} live on an ordinary
 curve. It lives on a tangential log curve $(X,D,\tau)$ with
 real-oriented blowup at each puncture
 (Definition~\ref{def:bordified-curve}). Omitting the tangential
 data loses the $E_1$-direction entirely.
\item Modularity does \emph{not} belong to the closed algebra alone.
 It arises from trace and clutching on the open sector:
 the MC equation forces open-closed consistency at every genus
 (Vol~I, Theorem~\ref*{V1-thm:thqg-mc-forced-consistency}).
\item The dg-shifted Yangian is \emph{not} a universal theorem for
 general 3d HT QFT. On the affine lineage it is proved
 (Theorem~\ref{thm:Koszul_dual_Yangian}); on the chirally Koszul
 locus the recognition theorem holds algebraically
 (Theorem~\ref{thm:yangian-recognition}); for general perturbative
 theories the gravitational-Yangian package is conjectural
 (Dimofte--Niu--Py~\cite{DNP25}).
\item The bulk-boundary map is \emph{not} an isomorphism in general.
 The bulk $\cZ^{\mathrm{der}}_{\mathrm{ch}}(\cA)$ maps to the
 boundary, not the reverse.
\end{enumerate}
