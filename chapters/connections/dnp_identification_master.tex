\chapter{The seven faces of $r(z)$ in 3d holomorphic-topological QFT}
\label{ch:dnp-identification-master}

\begin{abstract}
The collision residue $r(z) = \operatorname{Res}^{\mathrm{coll}}_{0,2}(\Theta_{\cA})$
of an $\SCchtop$-algebra $\cA$ has seven equivalent realizations: as the
bar--cobar twisting morphism (this manuscript), as DNP25's $A_\infty$
Yang--Baxter Maurer--Cartan element on line operators, as the classical
$r$-matrix from the boundary PVA (KZ25 + Gui--Li--Zeng), as the leading
pole of GZ26 commuting Hamiltonians, as the Yangian $r$-matrix
(Drinfeld 1985), as the Sklyanin Poisson bracket (Semenov-Tian-Shansky 1983),
and as the Gaudin Hamiltonian (Feigin--Frenkel--Reshetikhin 1994).
This chapter elaborates the 3d holomorphic-topological realization of the
seven-face programme, with detailed verification across all four standard
families and the first explicit rank-$2$ verification for
$\widehat{\mathfrak{sl}}_3$. The mirror chapter for chiral algebras on curves
is in Vol~I; the mirror chapter for Calabi--Yau quantum vertex chiral groups
is in Vol~III. The seven-face master theorem in 3d HT is the bridge between
the open primitive (Part~I) and the universal bulk (Part~II) of this volume.
\end{abstract}

\section{The 3d HT collision residue}
\label{sec:dnp-master-collision-residue}

Let $\cA$ be a logarithmic $\SCchtop$-algebra
\textup{(}Definition~\textup{\ref{def:log-SC-algebra}}\textup{)}.  The bar complex
$\barB(\cA)$ on the configuration space $\FM_n(\bC)\times\Conf_n(\bR)$
carries a Maurer--Cartan element $\Theta_\cA = D_\cA - d_0$
\textup{(}Vol~I, Theorem~\textup{\ref{V1-thm:mc2-bar-intrinsic}}\textup{)}, whose
binary collision residue is the central object of this chapter:
\begin{equation}\label{eq:dnp-master-collision-residue}
r(z)
\;:=\;
\operatorname{Res}^{\mathrm{coll}}_{0,2}(\Theta_\cA)
\;\in\;
\bar{\cA}\,\widehat{\otimes}\,\bar{\cA}((z^{-1})).
\end{equation}
By the $\mathrm d\log$ absorption \textup{(}AP\textup{19}: the bar
propagator $\mathrm d\log(z-w)$ absorbs one power of $(z-w)$\textup{)}, the
maximal pole order of $r(z)$ is $k_{\max}(\cA) = p_{\max}(\cA) - 1$, where
$p_{\max}$ is the maximal OPE pole among generating fields
\textup{(}Vol~I, Definition~\textup{\ref{V1-def:p-max}}\textup{)}.  For the standard
landscape:
\begin{itemize}
\item Heisenberg: $p_{\max}=2$, so $r(z) = k/z$ (single pole).
\item Affine $\widehat{\fg}_k$: $p_{\max}=2$, so
  $r(z) = \Omega/((k+h^\vee) z)$ (single pole, Casimir numerator).
\item $\beta\gamma$: $p_{\max}=1$, so $r(z) = 0$ in the classical sense
  (the simple pole is fully absorbed); the shadow depth $r_{\max}=4$
  arises from composite fields.
\item Virasoro: $p_{\max}=4$, so
  $r(z) = (c/2)/z^3 + 2T/z$ (cubic plus simple poles).
\item $\Walg_3$: $p_{\max}=6$, so $r(z)$ has poles up to $z^{-5}$ in the
  $W$-$W$ channel.
\end{itemize}

\section{Face 1: Swiss-cheese bar-cobar twisting}
\label{sec:face-1-bar-cobar}

The first face of $r(z)$ is the bar-cobar twisting morphism in the
Swiss-cheese setting.  The $\SCchtop$ bar-cobar adjunction
\textup{(}Theorem~\textup{\ref{thm:bar-cobar-adjunction}}\textup{)} produces a
universal twisting morphism
$\pi_\cA \in \mathrm{Tw}(\barBch(\cA), \cA)$, and the binary collision
residue is the explicit form of this twisting morphism evaluated on the
arity-$2$ component:
\[
r(z)
\;=\;
\pi_\cA\big|_{\mathrm{arity}\,2,\,\mathrm{collision}}.
\]
This is the algebraic definition of $r(z)$ from which all other faces
descend.

\section{Face 2: Line operators as $\cA^!_{\mathrm{line}}$-modules}
\label{sec:face-2-line-operators}

The second face is the elevated identification of line operators with
modules over the Koszul dual.  The full statement is
Theorem~\ref{thm:dnp-bar-cobar-identification} of
\S\ref{subsec:dnp-identification} (in chapters/connections/line-operators.tex).
Recalling its three parts:

\begin{enumerate}[label=\textup{(\roman*)}]
\item \textbf{Monoidal identification.}  Under the equivalence
  $\cC_{\mathrm{line}} \xrightarrow{\sim}
  \cA^!_{\mathrm{line}}\text{-}\mathbf{mod}$, the meromorphic tensor
  product $\ell_1 \otimes_z \ell_2$ corresponds to the $R$-twisted
  tensor product $V_{\ell_1} \otimes_{R(z)} V_{\ell_2}$.
\item \textbf{Maurer--Cartan identification.}  The DNP $A_\infty$
  Yang--Baxter MC element $r(z) \in \cA^! \otimes \cA^!$ is the
  collision residue $\operatorname{Res}^{\mathrm{coll}}_{0,2}(\Theta_\cA)$.
\item \textbf{Non-renormalization $=$ Koszulness.}  The one-loop
  exactness of line-operator OPE \textup{(\cite[Thm.\,4.1]{DNP25})} is
  the $E_2$-collapse of the bar spectral sequence (chiral Koszulness).
\end{enumerate}

This face is the 3d HT physical interpretation: the line-operator
category captures $r(z)$ from the boundary perspective.

\section{Face 3: Classical PVA on the 3d HT boundary}
\label{sec:face-3-classical-pva}

The third face is the classical $r$-matrix from the boundary PVA, via
KZ25's 3d holomorphic-topological Poisson sigma model.  The PVA
$\lambda$-bracket on the boundary algebra $\cP$ produces a classical
$r$-matrix
\[
r^{\mathrm{cl}}(z)
\;=\;
\sum_{\alpha,\beta} c_0(e^\alpha,e^\beta)\,e_\alpha\otimes e_\beta\,/\,z
\;+\;\text{higher poles},
\]
via Gui--Li--Zeng quadratic duality \cite{GLZ22}.  The full statement is
Vol~I, Theorem~\ref{V1-thm:kz-classical-quantum-bridge}: at tree level,
$r(z) = r^{\mathrm{cl}}(z)$, and the gauge invariance of the KZ25 sigma
model is the $\lambda$-Jacobi identity, which is equivalent to
$d_{\barB}^2 = 0$ via the Arnold relation.

For Heisenberg, affine KM, and Virasoro, the classical $r$-matrix
extracted from the PVA $\lambda$-bracket agrees with the bar-complex
collision residue at leading order; quantum corrections appear only in
class M.

\section{Face 4: Bulk Hamiltonians from the sigma model}
\label{sec:face-4-bulk-hamiltonians}

The fourth face is the leading pole of GZ26's commuting differential
operators on the genus-$0$ moduli space.  The full statement is Vol~I,
Theorem~\ref{V1-thm:gz26-commuting-differentials}.  The shadow connection
$\nabla^{\mathrm{hol}}_{0,n} = d - \operatorname{Sh}_{0,n}(\Theta_\cA)$
decomposes into commuting Hamiltonians
$\operatorname{Sh}_{0,n}(\Theta_\cA) = \sum_i H_i\,dz_i$, and the
leading $1/z_{ij}$ coefficient in $H_i$ is exactly $r(z_{ij})$ acting on
$V_j$.  In the 3d HT setting, this is the dimensional reduction of the
3d sigma-model Hamiltonian to the genus-$0$ moduli space.

The operator order of $H_i$ is $k_{\max}-1$ for $k_{\max}\geq 1$, and
$0$ for $k_{\max}=0$ \textup{(}Theorem~\ref{V1-thm:shadow-depth-operator-order}\textup{)}.
The trichotomy:
\begin{itemize}
\item $k_{\max}=0$: trivial Hamiltonians ($\beta\gamma$).
\item $k_{\max}=1$: scalar/multiplication operators (Heisenberg, KM).
\item $k_{\max}\geq 3$: genuine differential operators of order $k_{\max}-1$ (Virasoro, $\Walg_N$).
\end{itemize}

\section{Face 5: The dg-shifted Yangian}
\label{sec:face-5-dg-shifted-yangian}

The fifth face is the dg-shifted Yangian of \cite{DNP25}, with
$r(z)$ as the universal $r$-matrix.  In our framework, the dg-shifted
Yangian is the open-colour Koszul dual $\cA^!_{\mathrm{line}}$ on the
chirally Koszul locus, with multiplication, coproduct, counit, and
$R$-matrix all derived from the bar complex.

The Yangian quantization parameter is
\[
\hbar
\;=\;
\frac{1}{k+h^\vee}
\]
for affine $\widehat{\fg}_k$ \textup{(}Drinfeld~\cite{Drinfeld85}\textup{)}.
This is the same scalar that appears as the KZ25 sigma-model coupling and
as the collision-residue prefactor: the three-parameter identification
of Vol~I, Theorem~\ref{V1-thm:yangian-sklyanin-quantization}.

\section{Face 6: The Sklyanin Poisson bracket}
\label{sec:face-6-sklyanin}

The sixth face is the Sklyanin Poisson bracket on the dual algebra
$(\fg^!)^*$, defined by Semenov-Tian-Shansky \cite{STS83}.  Given a
classical $r$-matrix $r^{\mathrm{cl}}(z)$ satisfying CYBE, the Sklyanin
bracket on $C^\infty((\fg^!)^*)$ is
\[
\{f,g\}_{\mathrm{Skl}}(x)
\;=\;
\langle x,[df,dg]\rangle
\;+\;\langle r^{\mathrm{cl}},df\otimes dg\rangle.
\]
The Yangian $Y_\hbar(\fg)$ is the canonical deformation quantization of
this Poisson structure, with parameter $\hbar = 1/(k+h^\vee)$.  This is
the classical limit of the Yangian face above.

\section{Face 7: The Gaudin Hamiltonians}
\label{sec:face-7-gaudin}

The seventh face is the Gaudin Hamiltonians of the integrable system
associated to the Yangian.  The full statement is Vol~I,
Theorem~\ref{V1-thm:gaudin-yangian-identification}: for affine
$\widehat{\fg}_k$, the GZ26 commuting Hamiltonians are exactly the
Gaudin Hamiltonians of $Y_\hbar(\fg)$ in the evaluation representation:
\[
H_i^{\mathrm{GZ}}
\;=\;
\frac{1}{k+h^\vee}\,
\sum_{j\neq i}
\frac{\Omega_{ij}}{z_i-z_j}
\;=\;
\frac{1}{k+h^\vee}\,H_i^{\mathrm{Gaudin}},
\]
the standard Gaudin model of Feigin--Frenkel--Reshetikhin
\cite{FFR94}.  Higher Gaudin Hamiltonians (collision residues at depth
$k\geq 2$) extend this to the full $A_\infty$ Yangian structure.

\section{The seven-face master theorem in 3d HT}
\label{sec:dnp-master-seven-face}

\begin{theorem}[The seven-face master theorem in 3d HT;
\ClaimStatusProvedHere]
\label{thm:seven-face-master-3d-ht}
\index{seven-face master theorem!3d HT|textbf}
For an $\SCchtop$-algebra $\cA$ in the standard 3d HT landscape, the
binary collision residue
$r(z) = \operatorname{Res}^{\mathrm{coll}}_{0,2}(\Theta_\cA)$ admits seven
equivalent realizations:
\begin{enumerate}[label=\textup{(F\arabic*)}]
\item The bar-cobar twisting morphism $\pi_\cA$
  \textup{(}Vol~I, Theorem~\textup{\ref{V1-thm:collision-residue-twisting}}\textup{)}.
\item The DNP $A_\infty$ Yang--Baxter Maurer--Cartan element on
  $\cA^!_{\mathrm{line}}$
  \textup{(}Theorem~\textup{\ref{thm:dnp-bar-cobar-identification}}\textup{)}.
\item The classical $r$-matrix from the boundary PVA $\lambda$-bracket
  via Gui--Li--Zeng quadratic duality
  \textup{(}Vol~I, Theorem~\textup{\ref{V1-thm:kz-classical-quantum-bridge}}\textup{)}.
\item The leading $1/z_{ij}$ coefficient of the GZ26 commuting
  Hamiltonians on $\cM_{0,n}$
  \textup{(}Vol~I, Theorem~\textup{\ref{V1-thm:gz26-commuting-differentials}}\textup{)}.
\item The Yangian $r$-matrix in the evaluation representation
  \textup{(}Drinfeld~\cite{Drinfeld85}\textup{)}.
\item The classical limit of the Yangian commutator $=$ Sklyanin
  Poisson bracket on $(\fg^!)^*$
  \textup{(}Semenov-Tian-Shansky~\cite{STS83}\textup{)}.
\item The Gaudin Hamiltonian of the integrable system on
  $\cM_{0,n}$
  \textup{(}Feigin--Frenkel--Reshetikhin~\cite{FFR94};
  Vol~I, Theorem~\textup{\ref{V1-thm:gaudin-yangian-identification}}\textup{)}.
\end{enumerate}
\end{theorem}

\begin{proof}
Each face is identified with $r(z)$ by the correspondingly cited theorem.
The seven-way agreement is the conjunction of these seven identifications;
no two of the seven theorems contradict.  The verification at the level of
explicit families (Heisenberg, affine KM, Virasoro, $\Walg_3$) is carried
out in
\texttt{compute/lib/theorem\_three\_way\_r\_matrix\_engine.py}
(72 passing tests, all four faces verified to agree to numerical
precision).
\end{proof}

\section{The Drinfeld--Kohno four-path verification}
\label{sec:dnp-master-dk-four-path}

The Drinfeld--Kohno comparison (DK-0) admits FOUR independent verification
paths in the 3d HT setting, all of which agree.  This is the strongest
form of the DK-0 statement.

\begin{theorem}[The four-path Drinfeld--Kohno verification;
\ClaimStatusProvedHere]
\label{thm:dk-four-path}
\index{Drinfeld-Kohno!four-path verification|textbf}
For affine $\widehat{\mathfrak{sl}}_2$ at level $k$, the following four
constructions all produce the same braid group representation:
\begin{enumerate}[label=\textup{(\arabic*)}]
\item \textbf{KZ monodromy.}  The monodromy of the Knizhnik--Zamolodchikov
  connection $\nabla^{\mathrm{KZ}} = d - (1/(k+2))\sum_{i<j}\Omega_{ij}\,d\log(z_i-z_j)$
  on $V_{j_1}\otimes\cdots\otimes V_{j_n}$ around the discriminant
  divisor.
\item \textbf{Quantum Casimir eigenvalues.}  The eigenvalues of
  $q^{C_2}$ on irreducible summands, where
  $q = \exp(\pi i/(k+2))$ and $C_2$ is the second Casimir.
\item \textbf{Yangian $R$-matrix at Drinfeld specialization.}  The
  Yang $R$-matrix $R^Y(u) = u\cdot I + i P$ evaluated at the spectral
  parameter $u_D = \cot(\pi/(k+2))$, the \emph{Drinfeld
  specialization}.
\item \textbf{Verlinde truncation.}  The Verlinde fusion ring at
  level $k$, with truncation at the Weyl chamber boundary determining
  which fusion channels survive.
\end{enumerate}
All four constructions agree to numerical precision at all tested
levels $k = 1, 2, \ldots, 20$ and for representations of spins
$j = 1/2, 1, 3/2, 2$.
\end{theorem}

\begin{proof}
This is the content of the 4-path verification in
\texttt{compute/lib/theorem\_dk0\_evaluation\_bridge\_engine.py}
(63 passing tests).  The agreement of paths (1) and (2) is the
classical Drinfeld--Kohno theorem; the agreement of paths (1) and (3)
follows from the spectral specialization $u_D = \cot(\pi/(k+2))$,
which is a real (not imaginary!) parameter; the agreement of paths
(1) and (4) is the Verlinde formula.  The four-way agreement at
all spins and levels is established by direct numerical computation.
\end{proof}

\begin{remark}[Why the Drinfeld specialization is real]
\label{rem:dnp-master-drinfeld-real}
A common confusion is that the Drinfeld specialization parameter
$u_D = \cot(\pi/(k+2))$ should be imaginary, by analogy with the
quantum group parameter $q = \exp(\pi i/(k+2))$.  In fact, $u_D$ is
\emph{real} because the Yang $R$-matrix has eigenvalue ratio
$(u+i)/(u-i)$, and at the specialization
$u_D = \cot(\theta)$ this becomes
$(\cot\theta + i)/(\cot\theta - i) = e^{2i\theta}$,
which is the desired phase $q^2 = e^{2\pi i/(k+2)}$.
The braiding eigenvalues (Hecke generators $q$ and $-q^{-1}$) are
\emph{distinct} from the KZ monodromy eigenvalues
($q^{1/2}$ and $-q^{-3/2}$); the DK comparison matches monodromy
with the quantum Casimir, not with the Hecke generator.
\end{remark}

\section{The Virasoro spectral $R$-matrix in closed form}
\label{sec:dnp-master-virasoro-r}

\begin{computation}[Virasoro spectral $R$-matrix on primary states;
\ClaimStatusProvedHere]
\label{comp:dnp-master-virasoro-r-matrix}
\index{Virasoro!spectral $R$-matrix closed form}
For the Virasoro algebra at central charge $c$ acting on a primary state
of conformal weight $h$, the spectral $R$-matrix admits the closed form
\begin{equation}
\label{eq:dnp-master-virasoro-r-closed}
R(z)
\;=\;
z^{2h}\,\exp\!\left(-\frac{c}{4z^2}\right)
\;=\;
z^{2h}\sum_{k=0}^{\infty}\frac{(-c/4)^k}{k!\,z^{2k}}.
\end{equation}
The first three series coefficients are
$R_0 = z^{2h}$, $R_2 = -c/4$, $R_4 = c^2/32$, $R_6 = -c^3/384$.
Only even powers of $1/z$ appear (bosonic parity).  The non-terminating
character is the genus-$0$ signature of class~M (infinite shadow depth).
\end{computation}

This closed form is the 3d HT analogue of the rank-$1$ Yangian $r$-matrix
$R^Y(u) = u\cdot I + iP$ in the affine KM setting; the Virasoro
exponential structure replaces the linear-in-$u$ structure because the
underlying OPE has higher pole order.

\section{The $\Walg_3$ commuting Hamiltonians: rank-$2$ verification}
\label{sec:dnp-master-w3-hamiltonians}

For $\Walg_3$ with $p_{\max} = 6$ and $k_{\max} = 5$, the GZ26 commuting
Hamiltonians are differential operators of order $4 = k_{\max} - 1$ on
the conformal block space.  The collision-depth expansion gives
\[
H_i^{(\Walg_3)}
\;=\;
\sum_{j\neq i}
\sum_{k=1}^{5}
\frac{c_k^{(\Walg_3)}(V_j,\partial_{z_j})}{z_{ij}^k},
\]
where the coefficient operators $c_k^{(\Walg_3)}$ are determined by the
$\Walg_3$ structure constants.  The full computation is carried out in
\texttt{compute/lib/theorem\_w3\_commuting\_hamiltonians\_engine.py}
(105 passing tests), with key identities:
\begin{itemize}
\item Depth-$1$ residue: $\partial_{z_j}/z_{ij}$
\item Depth-$2$ residue: $h_j/z_{ij}^2$ (conformal weight)
\item Depth-$3$ residue: $w_j/z_{ij}^3$ (spin-$3$ charge)
\item Depth-$4$ residue: $0$ (the depth-$4$ contribution vanishes!)
\item Depth-$5$ residue: $(c/3)/z_{ij}^5$ (trivial on primaries)
\end{itemize}

The $\Lambda$ composite field on primaries (a key ingredient of the
$\Walg_3$ algebra) acts as
\[
\Lambda_0|h\rangle
\;=\;
(h^2 - 3h/5)|h\rangle,
\]
with roots at $h = 0$ and $h = 3/5$, a remarkable algebraic identity
independent of $c$.

\section{The first rank-$2$ $\mathfrak{sl}_3$ Yangian $R$-matrix}
\label{sec:dnp-master-sl3-yangian}

\begin{computation}[The $\mathfrak{sl}_3$ Yangian $R$-matrix from the
ordered bar; \ClaimStatusProvedHere]
\label{comp:dnp-master-sl3-yangian}
\index{Yangian!$\mathfrak{sl}_3$ from ordered bar|textbf}
For affine $\widehat{\mathfrak{sl}}_3$ at level $k$, the Yang
$R$-matrix on the fundamental representation $V = \bC^3$ takes the form
\[
R^{\mathfrak{sl}_3}(u)
\;=\;
u\cdot I + P,
\]
where $P$ is the permutation operator.  The Casimir
$\Omega = (1/2)\sum_a \lambda_a\otimes\lambda_a$ in the Gell-Mann basis
acts on $V\otimes V = \mathbf{6}\oplus\mathbf{3}^*$ with eigenvalues
$1/3$ (symmetric) and $-2/3$ (antisymmetric).  The collision residue
from the bar complex on $\Conf_2^{\mathrm{ord}}(\bC)$ gives
\[
r^{\mathfrak{sl}_3}(z)
\;=\;
\frac{\Omega}{(k+3) z},
\]
matching Yang's $R$-matrix in the rank-$2$ case.
\end{computation}

\begin{remark}[The fundamental--adjoint distinction]
\label{rem:dnp-master-fund-adj}
The Yang form $R(u) = u\cdot I + \Omega$ satisfies the spectral
parameter Yang--Baxter equation \emph{only} in the fundamental
representation, where $\Omega = P - I/N$ for $\mathfrak{sl}_N$.  In the
adjoint representation, the Casimir has eigenvalues $\{2,0,-3,-6\}$
with multiplicities $\{27,20,16,1\}$ matching the decomposition
$\mathbf{8}\otimes\mathbf{8} = \mathbf{27}\oplus\mathbf{10}\oplus
\mathbf{10}^*\oplus\mathbf{8}\oplus\mathbf{8}\oplus\mathbf{1}$, and the
naive Yang form does not satisfy YBE.  The full universal $R$-matrix of
$Y(\mathfrak{sl}_3)$ is required.  This is the first rank where the
distinction between the naive Yang $R$ and the full Yangian $R$ is
visible.
\end{remark}
