

\subsection*{Appendix: Explicit Formulas and Computational Details for W-Algebras}
\label{app:W-algebras-explicit}

This appendix provides detailed computational formulas for the higher $A_\infty$ operations $m_k$ in Virasoro and $W_3$ algebras, including:
\begin{enumerate}
\item Explicit mode expansions for $m_3$ through $m_7$;
\item Combinatorial coefficients from Feynman diagram evaluation;
\item Recursive algorithms for generating all $m_k$;
\item Spectral parameter manipulations;
\item Sample Mathematica code for automation.
\end{enumerate}

\subsection{Notation and Conventions}

\subsubsection{Mode Expansions}

Fields are expanded in holomorphic modes:
\begin{align}
T(z) &= \sum_{n \in \Z} T_n z^{-n-2}, & [T_m, T_n] = (m-n)T_{m+n} + \frac{c}{12}m(m^2-1)\delta_{m+n,0},\\
W(z) &= \sum_{n \in \Z} W_n z^{-n-3}, & (\text{commutation relations for } W_n \text{ are more complex}).
\end{align}

In the algebraic approach, we work with the derivatives:
\begin{align}
T_n &= \frac{1}{n!} \partial^n T, & W_n = \frac{1}{n!} \partial^n W.
\end{align}

\subsubsection{Spectral Parameters}

For $k$ operators at points $z_1, \ldots, z_k$, define difference coordinates:
\begin{align}
\lambda_i &= z_i - z_{i+1}, & i = 1, \ldots, k-1.
\end{align}

The operation $m_k$ is a Laurent series in $\lambda_1, \ldots, \lambda_{k-1}$:
\begin{equation}
m_k(a_1, \ldots, a_k) = \sum_{n_1, \ldots, n_{k-1} \in \Z} \frac{C_{n_1, \ldots, n_{k-1}}(a_1, \ldots, a_k)}{\lambda_1^{n_1+1} \cdots \lambda_{k-1}^{n_{k-1}+1}}.
\end{equation}

\subsection{Virasoro: Explicit Formulas}

\subsubsection{The Binary Operation $m_2(T,T)$}

The singular part (the $\lambda$-bracket) is
\begin{align}
\{T(z_1) {}_\lambda T(z_2)\} &= \frac{c/12}{\lambda^3} + \frac{2T(z_2)}{\lambda^2} + \frac{\partial T(z_2)}{\lambda}\,.
\end{align}
In modes,
\begin{align}
\{T_m {}_\lambda T_n\} &= \sum_{k=0}^{m} \binom{m}{k} (-\lambda)^k (k-m-1) T_{m+n-k}\\
&\quad + \frac{c}{12} \sum_{k=0}^m \binom{m}{k} (-\lambda)^k (k-m-2)(k-m-1)(k-m) \delta_{m+n-k,0}.
\end{align}
The regular part gives the normally ordered product:
$m_2(T,T)_{\text{reg}} = {:}T_1 T_2{:}$.

\subsubsection{The Ternary Operation $m_3(T,T,T)$}

The ternary operation comes from the Y-diagram integral.  In modes:
\begin{equation}
\label{eq:vir-m3-modes}
m_3(T_m, T_n, T_p) = \sum_{\substack{a,b,c \geq 0 \\ a+b+c = m+n+p-2}} \mathcal{A}_{m,n,p}^{a,b,c} \frac{\partial^a T \cdot \partial^b T \cdot \partial^c T}{\lambda_1^{a+2} \lambda_2^{b+2}},
\end{equation}
where the coefficients $\mathcal{A}_{m,n,p}^{a,b,c}$ are determined by:

\begin{equation}
\mathcal{A}_{m,n,p}^{a,b,c} = \frac{1}{a! b! c!} \binom{m}{a} \binom{n}{b} \binom{p}{c} \cdot F_{a,b,c}(c),
\end{equation}
with structure function
\begin{equation}
F_{a,b,c}(c) = \begin{cases}
\frac{c}{12}(a+2)(a+1)a & \text{if } b=c=0,\\
2(a+1) & \text{if } b=1, c=0,\\
1 & \text{if } b=0, c=1,\\
\text{(mixed terms)} & \text{otherwise}.
\end{cases}
\end{equation}

In low-order cases:
\begin{align}
m_3(T_0, T_0, T_0) &= \frac{c/12 \cdot 6}{\lambda_1^3 \lambda_2^3} + \frac{2T}{\lambda_1^2 \lambda_2^2} + \frac{\partial T}{\lambda_1 \lambda_2} + \text{(mixed poles)},\\
m_3(T_1, T_0, T_0) &= \frac{c/12 \cdot 24}{\lambda_1^4 \lambda_2^3} + \frac{6T + \text{derivatives}}{\lambda_1^3 \lambda_2^2} + \cdots,\\
m_3(T_1, T_1, T_0) &= \text{(higher poles up to } \lambda_1^{-5} \lambda_2^{-4} \text{)}.
\end{align}

\subsubsection{The Quaternary Operation $m_4(T,T,T,T)$}

The operation $m_4$ receives a tree contribution (cross topology) and a
one-loop contribution (box topology).

\medskip
\noindent\emph{Tree contribution.}
\begin{equation}
\label{eq:vir-m4-tree}
m_4(T,T,T,T)_{\text{tree}} = \sum \frac{\mathcal{B}^{a,b,c}_{m,n,p,q}(\partial^a T, \partial^b T, \partial^c T, \partial^d T)}{\lambda_1^{a+2} \lambda_2^{b+2} \lambda_3^{c+2}},
\end{equation}
where $\mathcal{B}$ coefficients come from evaluating the cross diagram integral:
\begin{equation}
\mathcal{B}^{a,b,c,d}_{m,n,p,q} = \int_{\FM_4(\C)} \frac{(z_1-z_v)^m (z_2-z_v)^n (z_3-z_v)^p (z_4-z_v)^q}{(z_1-z_2)^{a+2} (z_2-z_3)^{b+2} (z_3-z_4)^{c+2}} d^2z_v.
\end{equation}

\medskip
\noindent\emph{One-loop contribution.}
The box diagram integral is
\begin{align}
m_4(T,T,T,T)_{1\text{-loop}} &= \int_{\FM_4(\C) \times \C} \frac{d^2z_v}{(z_1-z_v)(z_2-z_v)} \cdot \frac{d^2z_{v'}}{(z_v-z_{v'})(z_{v'}-z_3)(z_{v'}-z_4)}\\
&= \sum \frac{\mathcal{L}^{a,b,c}_{m,n,p,q} + \log(\lambda_i/\mu) \cdot (\cdots)}{\lambda_1^{a+1} \lambda_2^{b+1} \lambda_3^{c+1}},
\end{align}
where $\mathcal{L}$ coefficients involve polylogarithms and $\mu$ is a renormalization scale.

At lowest order:
\begin{equation}
m_4(T_0, T_0, T_0, T_0) = \frac{c^2/144 \cdot \text{(combinatorial factor)}}{\lambda_1^4 \lambda_2^4 \lambda_3^4} + \cdots + \hbar \cdot \frac{\log(\lambda_1/\mu)}{\lambda_1^2 \lambda_2^2 \lambda_3^2} + \cdots.
\end{equation}

\subsubsection{Operations $m_5$, $m_6$: Structure}

The diagram count grows with arity:
$m_5$ has $12$ tree diagrams and $5$ one-loop topologies;
$m_6$ has $60$ tree, $30$ one-loop, and $3$ potential two-loop diagrams.
The loop expansion takes the form
\begin{equation}
m_k(T^{\otimes k}) = \sum_{\text{trees}} m_k^{(\text{tree})} + \hbar \sum_{\text{1-loop}} m_k^{(1\text{-loop})} + \hbar^2 \sum_{\text{2-loop}} m_k^{(2\text{-loop})} + \cdots.
\end{equation}

\subsection{$W_3$: Explicit Formulas}

\subsubsection{The Binary Operations}

The binary operation $m_2(T,T)$ is identical to Virasoro.
The mixed bracket is
\[
\{T_\lambda W\} = \frac{3W}{\lambda^2} + \frac{\partial W}{\lambda},
\]
and in modes,
$\{T_m {}_\lambda W_n\} = \sum_{k=0}^m \binom{m}{k} (-\lambda)^k (k-m) W_{m+n-k} + \frac{1}{k!} \partial^k W_{m+n-k}$.
The self-bracket $\{W_\lambda W\}$
(cf.\ equation~\eqref{eq:m2WW-recap} for the compact
polynomial form) is
\begin{align}
\{W(z_1) {}_\lambda W(z_2)\} &= \frac{c/360}{\lambda^5} + \frac{T(z_2)/3}{\lambda^3} + \frac{\partial T(z_2)/2}{\lambda^2}\\
&\quad + \frac{(3/10)\partial^2 T(z_2) + (32/(5c+22))\Lambda}{\lambda} + \frac{1}{15}\partial^3 T + \frac{16}{5c+22}\partial\Lambda.
\end{align}

\subsubsection{The Ternary Operations}

The pure-$T$ operation $m_3(T,T,T)$ agrees with Virasoro.
The mixed operation $m_3(T,T,W)$ is
\begin{equation}
m_3(T_m, T_n, W_p) = \sum_{\substack{a,b,c \\ a+b+c = m+n+p-2}} \mathcal{C}_{m,n,p}^{a,b,c} \frac{(\partial^a T)(\partial^b T)(\partial^c W)}{\lambda_1^{a+2} \lambda_2^{b+3}},
\end{equation}
where the $\mathcal{C}$ coefficients encode the interaction between the $T$ and $W$ sectors.

For instance,
$m_3(T_0, T_0, W_0) = \frac{6 \cdot \text{(structure constant)}}{\lambda_1^2 \lambda_2^3} + \frac{3TW + \text{derivatives}}{\lambda_1 \lambda_2^2} + \cdots$.
The operation $m_3(T,W,W)$ is
\begin{equation}
m_3(T_m, W_n, W_p) = \sum \frac{\mathcal{D}_{m,n,p}^{a,b,c}(\partial^a T, \partial^b W, \partial^c W)}{\lambda_1^{a+2} \lambda_2^{b+4}}.
\end{equation}

The operation $m_3(W,W,W)$ involves the quintic Schwarzian coefficient
$c/360$ from the $\{W_\lambda W\}$ bracket.  The full
closed-form result is given in
equation~\eqref{eq:m3-WWW}; we record here only
the leading pole structure in the zero-mode sector:
\begin{align}
m_3(W_0, W_0, W_0) &= \frac{c/360 \cdot 120}{\lambda_1^5 \lambda_2^5} + \frac{T/3 \cdot (\text{mixed derivatives})}{\lambda_1^3 \lambda_2^3}\\
&\quad + \frac{(\partial T) \cdot (\text{terms})}{\lambda_1^2 \lambda_2^2} + \cdots.
\end{align}
By the $\Z_2$ parity of $\cW_3$, the $T$-sector output of
$m_3(W,W,W)$ vanishes identically (see the discussion
following~\eqref{eq:m3-WWW-T-sector}), so the surviving
contributions are those displayed
in~\eqref{eq:m3-WWW}.

\subsubsection{Higher Operations: Summary of Structure}

\emph{Arity~$4$.}
With two generator types $\{T, W\}$ and $4$ ordered inputs,
there are $2^4 = 16$ distinct ordered operations.
Grouping by the number~$j$ of $W$-insertions gives $5$ unordered types
($j = 0,\ldots,4$):
\begin{align}
&m_4(T,T,T,T), \quad m_4(T,T,T,W), \quad m_4(T,T,W,W),\\
&m_4(T,W,W,W), \quad m_4(W,W,W,W).
\end{align}
Each has the Virasoro structure in the pure-$T$ sector, with additional
$T$--$W$ mixing terms.

\emph{Arities~$5$ and~$6$.}
At arity~$5$ there are $\sim 20$ distinct ordered operations,
at arity~$6$ roughly~$50$.

\subsection{Generating Function Approach}

\subsubsection{Virasoro Generating Function}

Define the exponential generating function
\begin{equation}
\mathcal{F}_{\text{Vir}}(T; \lambda) = \sum_{k=1}^\infty \frac{1}{k!} m_k(T, \ldots, T)(\lambda_1, \ldots, \lambda_{k-1}).
\end{equation}

\begin{theorem}[Virasoro Generating Function; \ClaimStatusConditional]
\label{thm:vir-gen-func-explicit}
Let\/ $\cA$ be a logarithmic\/ $\SCchtop$-algebra.
The generating function satisfies the differential equation:
\begin{equation}
\label{eq:vir-diff-eq}
\frac{\partial \mathcal{F}_{\text{Vir}}}{\partial T} = \left( \partial + 2\lambda + \frac{c}{12}\lambda^3 \partial^3 \right) \mathcal{F}_{\text{Vir}} + \frac{1}{2} \{\mathcal{F}_{\text{Vir}}, \mathcal{F}_{\text{Vir}}\}_\lambda.
\end{equation}
\end{theorem}

\noindent Equation~\eqref{eq:vir-diff-eq} yields the recursion
\begin{align}
m_{k+1}(T^{\otimes (k+1)}) &= \frac{1}{k+1} \Bigg[ Q_{\text{BRST}} \cdot m_k(T^{\otimes k})\\
&\quad + \sum_{i=1}^{k-1} m_2 \circ_i m_k(T^{\otimes k})\\
&\quad + \sum_{\substack{j+\ell = k+2 \\ j,\ell \geq 2}} m_j \circ m_\ell(T^{\otimes k}) \Bigg].
\end{align}

Starting from $m_1 = Q$ and $m_2 = \{T_\lambda T\}$, each
$m_{k+1}$ is determined by the preceding operations;
the step from $m_k$ to $m_{k+1}$ amounts to evaluating
Feynman diagrams with the appropriate combinatorial weights.

\subsubsection{$W_3$ Generating Function}

For $W_3$, the generating function is bi-graded:
\begin{equation}
\mathcal{F}_{W_3}(T, W; \lambda) = \sum_{k_T, k_W} \frac{1}{k_T! k_W!} m_{k_T + k_W}(T^{\otimes k_T} \otimes W^{\otimes k_W}).
\end{equation}

\begin{theorem}[$W_3$ Generating Function System; \ClaimStatusConditional]
\label{thm:w3-gen-func-system}
Let\/ $\cA$ be a logarithmic\/ $\SCchtop$-algebra.
The generating function satisfies a \emph{system} of coupled differential equations:
\begin{align}
\frac{\partial \mathcal{F}_{W_3}}{\partial T} &= (\partial + 2\lambda + \cdots) \mathcal{F}_{W_3} + \frac{1}{2} \{_T \mathcal{F}_{W_3}, \mathcal{F}_{W_3}\}_\lambda,\\
\frac{\partial \mathcal{F}_{W_3}}{\partial W} &= (\partial + 3\lambda + \cdots) \mathcal{F}_{W_3} + \frac{1}{2} \{_W \mathcal{F}_{W_3}, \mathcal{F}_{W_3}\}_\lambda,
\end{align}
where $\{_T \cdot, \cdot \}_\lambda$ and $\{_W \cdot, \cdot \}_\lambda$ denote the mixed brackets.
\end{theorem}

Iterating this system determines all $m_k$.

\subsection{Spectral $r$-Matrices: Detailed Expansion}

\subsubsection{Virasoro $r$-Matrix: Expansion in Modes}

The spectral $r$-matrix in mode form is
\begin{equation}
r^{\text{Vir}}(\lambda, \mu) = \sum_{m,n \in \Z} r_{m,n} T_m \otimes T_n \cdot \lambda^{-m-2} \mu^{-n-2}.
\end{equation}

with coefficients
\begin{equation}
r_{m,n} = \begin{cases}
\frac{c}{12} m(m^2-1) & \text{if } m+n = 0,\\
\frac{1}{2}(m-n) & \text{if } m+n \neq 0,
\end{cases}
\end{equation}
reproducing the Virasoro Lie algebra structure.

\subsubsection{$W_3$ $r$-Matrix: Full Expansion}

The $W_3$ $r$-matrix is a $2 \times 2$ matrix of Laurent series:
\begin{equation}
r^{W_3} = \begin{pmatrix}
\sum r^{TT}_{m,n; i,j} T_m T_n \lambda^{-i} \mu^{-j} & \sum r^{TW}_{m,n; i,j} T_m W_n \lambda^{-i} \mu^{-j}\\
\sum r^{WT}_{m,n; i,j} W_m T_n \lambda^{-i} \mu^{-j} & \sum r^{WW}_{m,n; i,j} W_m W_n \lambda^{-i} \mu^{-j}
\end{pmatrix}.
\end{equation}

The coefficients $r^{AB}_{m,n; i,j}$ are determined by the $\lambda$-brackets \eqref{eq:w3-lambda-TT}--\eqref{eq:w3-lambda-WW} and satisfy symmetry relations:
\begin{align}
r^{TT}_{m,n; i,j} &= -r^{TT}_{n,m; j,i}, & (\text{antisymmetry})\\
r^{WW}_{m,n; i,j} &= -r^{WW}_{n,m; j,i}, & (\text{antisymmetry})\\
r^{TW}_{m,n; i,j} &\neq -r^{WT}_{n,m; j,i}, & (\text{no simple symmetry}).
\end{align}

\subsection{Computational procedure}

To compute $m_k$ at a given arity and loop order:
enumerate connected Feynman graphs with $k$ external legs and
loop number $\ell(G) = E - V + 1$ bounded by the desired order;
assign propagators $K(z_i - z_j)$ to internal edges and
interaction vertices from the action to internal vertices;
evaluate Wick contractions; integrate over $\FM_k(\C)$
using the FM compactification and the Laurent expansion
\[
\frac{1}{(z_1-z_2)^n}
= \sum_{k=0}^\infty \frac{(-1)^k}{k!}
  \binom{n+k-1}{k} \lambda^{-n-k} \partial^k;
\]
finally, collect terms by pole order.

For Virasoro at arity~$4$, this yields
two tree topologies (cross and chain) and one
one-loop topology (box).

The leading-pole structures of $m_3$ computed above agree with
Khan--Zeng~\cite{KZ25} (both Virasoro and $W_3$ sectors) and with
Costello--Dimofte--Gaiotto~\cite{CDG20} in the Virasoro sector.

\subsection{From \texorpdfstring{$\mathcal{W}_N$}{W\_N} to higher-spin gauge theory}
\label{subsec:wn-higher-spin}

\providecommand{\ModEnv}{\operatorname{ModEnv}}

The $\mathcal{W}_N$ Poisson vertex algebra provides the chiral-algebraic
input for a family of 3d holomorphic-topological gauge theories of
increasing spin, connecting to higher-spin gravity in the $N \to \infty$
limit.

\begin{definition}[$\mathcal{W}_N$ field content; \ClaimStatusProvedElsewhere]
\label{def:wn-field-content}
The $\mathcal{W}_N$ PVA has generators $W_s$ of spin~$s$ for
$s = 2, 3, \ldots, N$, with $W_2 = T$ the Virasoro element.  The
associated 3d HT Poisson sigma model has field pairs $(W_s, \chi_s)$
for each $s = 2, \ldots, N$, where $\chi_s$ is the topological partner
of spin $1-s$ \textup{(}\!\cite{KZ25}, \S3.4\textup{)}.
\end{definition}

\begin{proposition}[Slab central charges for $\mathcal{W}_N$; \ClaimStatusProvedElsewhere]
\label{prop:wn-slab-central-charges}
The topological enhancement from the Virasoro subalgebra $T \subset
\mathcal{W}_N$ produces a slab reduction with paired central charge
\textup{(}\!\cite{KZ25}, \S3.4\textup{)}:
\begin{equation}\label{eq:wn-cpair}
c_{\mathrm{pair}}^{(N)}
\;=\;
\sum_{s=2}^{N} \bigl[3(2s-1)^2 - 1\bigr].
\end{equation}
The first values are:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{lccccc}
\toprule
$N$ & $2$ & $3$ & $4$ & $5$ & $\infty$ \\
\midrule
$c_{\mathrm{pair}}^{(N)}$ & $26$ & $100$ & $246$ & $488$ & $\infty$ \\
\bottomrule
\end{tabular}
\end{center}
The identity $c_{\mathrm{pair}}^{(N)} = \alpha_N$%
\index{ghost central charge!WN@$\mathcal{W}_N$ identity}%
{} is not a coincidence: it is the ghost central
charge.\label{rem:ghost-identity-alphaN-active}
The $\cW_N$ ghost system consists of $bc$-pairs of spins
$s = 2, 3, \ldots, N$, each contributing
$c_s = -2(6s^2 - 6s + 1) = -(3(2s-1)^2 - 1)$, so
\[
c_{\mathrm{ghost}}(\cW_N)
\;=\; \sum_{s=2}^{N} c_s
\;=\; -2(N{-}1)(2N^2{+}2N{+}1)
\;=\; -\alpha_N
\;=\; -c_{\mathrm{pair}}^{(N)}.
\]
The critical string condition $c = \alpha_N$ is therefore
$c_{\mathrm{matter}} + c_{\mathrm{ghost}} = 0$;
for $N = 2$ this recovers the bosonic string critical dimension~$26$,
and for $N = 3$ the value~$100$ matches the $\mathcal{W}_3$ computation
of Vol~I.

The self-dual points $c^* = \alpha_N/2$ inherit a universal
property from the table.\label{thm:self-dual-halving}%
\index{self-dual point!genus tower halving}
At the Koszul self-dual point $c^* = \alpha_N/2$, the
curvature satisfies
$\kappa(\cW_{N,c^*}) = c^*(H_N - 1) = \tfrac{1}{2}\,\kappa(\cW_{N,0})$.
At genus~$1$, where $F_1 = \kappa\cdot\lambda_1^{\mathrm{FP}}$
holds unconditionally,
$F_1(c^*) = \tfrac{1}{2}\,F_1(0)$.
For Virasoro ($N = 2$, uniform-weight), the scalar formula
extends to all genera: $F_g(c^*) = \tfrac{1}{2}\,F_g(0)$ for all
$g \ge 1$.  For $N \ge 3$ (multi-weight), genus~$\ge 2$ receives
cross-channel corrections
(Volume~I, Theorem~\ref{thm:multi-weight-genus-expansion}),
so the halving is exact only at genus~$1$.
\end{proposition}

\begin{remark}[$\mathcal{W}_\infty$ and higher-spin gravity]
\label{rem:winfty-higher-spin}
In the $N \to \infty$ limit, $\mathcal{W}_\infty$ produces
$c_{\mathrm{pair}} \to \infty$, signaling a decompactification of the
topological direction.  The infinite tower of fields
$(W_s, \chi_s)_{s \geq 2}$ is the field content of Vasiliev
higher-spin gravity in the holomorphic-topological formulation.  The
$\mathcal{W}_\infty$ modular envelope $\ModEnv(\mathcal{W}_\infty)$ is
precisely the MC4-closed package from Vol~I
\textup{(}cf.~the concordance, \S\textup{MC4}\textup{)}: the full higher-spin
data is encoded in the modular Koszul duality of the limiting
$\mathcal{W}$-algebra.
Volume~I proves that $\mathcal{W}_\infty$ has a one-dimensional
minimal cyclic direction,
\[
\dim H^2_{\mathrm{cyc}}(\mathcal{W}_\infty, \mathcal{W}_\infty)=1,
\]
and computes the explicit weight-$4$ structure constant
\[
c_{334}^2=\frac{42c^2(5c+22)}{(c+24)(7c+68)(3c+46)}.
\]
\end{remark}

\subsection{$\mathcal{W}_4$ and $\mathcal{W}_5$:

\subsubsection{$\mathcal{W}_4$ OPE structure and field content}

\begin{proposition}[$\mathcal{W}_4$ non-vanishing $m_3$;
\ClaimStatusConditional]
\label{prop:w4-m3}
\index{W4@$\mathcal{W}_4$!ternary operation}%
Let\/ $\cA$ be a logarithmic\/ $\SCchtop$-algebra.
The ternary operation $m_3$ on $\mathcal{W}_4$ satisfies:
\begin{enumerate}[label=\textup{(\roman*)}]
\item $m_3(T,T,T) = m_3^{\mathrm{Vir}}(T,T,T) \ne 0$
  \textup{(}identical to Virasoro, by sector decoupling\textup{)}.
\item $m_3(T,T,W_3) \ne 0$, with leading term controlled by
  the $\{T_\lambda \{T_\mu W_3\}\}$ associator.
\item $m_3(T,T,W_4) \ne 0$, with leading term from
  the $\{T_\lambda \{T_\mu W_4\}\}$ associator.
\item $m_3(W_4, W_4, W_4) \ne 0$: the $W_4$ self-OPE has
  poles up to order~$8$, producing a nonzero associator
  from the iterated $\{W_{4\,\lambda} \{W_{4\,\mu} W_4\}\}$
  bracket via nonlinear composite fields
  $\Lambda = {:}TT{:} - \tfrac{3}{10}\partial^2 T$,
  $U = {:}TW_3{:} - \ldots$, etc.
\end{enumerate}
\end{proposition}

\begin{proof}
Part~(i) follows from Proposition~\ref{prop:vir-all-mk}
and the sector decoupling~\eqref{eq:wn-sector-decoupling}.
For parts~(ii)--(iv), the ternary operation $m_3$ is the
negative of the associator of the $\lambda$-bracket:
\[
m_3(X, Y, Z;\, \lambda, \mu)
= -A_3(X, Y, Z;\, \lambda, \mu),
\quad
A_3 = m_2(m_2(X, Y;\lambda),\, Z;\, \lambda+\mu)
  - m_2(X,\, m_2(Y, Z;\mu);\, \lambda).
\]

For part~(ii), the inputs $T, T, W_3$ give:
$A_3(T,T,W_3;\lambda,\mu)
= \{(\partial W_3 + 2\lambda W_3 + \cdots)_{\lambda+\mu}\, W_3\}
- \{T_\lambda\, (\partial W_3 + 3\mu W_3)\}$.
Since $\{T_\lambda W_s\} = \partial W_s + s\lambda W_s$
for a primary field $W_s$ of spin~$s$, the associator
evaluates to terms involving
$2\lambda \cdot s\mu = 2s \lambda\mu$ cross-terms that
do not cancel (the failure of strict associativity between
primary and stress tensor sectors).

For part~(iv), $A_3(W_4, W_4, W_4;\lambda,\mu)$ involves
the full $\{W_{4\,\lambda} W_4\}$ bracket, which contains
terms up to $\lambda^7$ (pole order $8$ minus $1$ from bar
kernel).  The iterated bracket
$\{W_{4\,\lambda}\{W_{4\,\mu} W_4\}\}$ produces composite
fields at spin~$8$ and lower, generating a nonzero
polynomial in $\lambda, \mu$ of total degree up to $14$.
The non-vanishing follows because the $W_4$ algebra is
not associative at the $\lambda$-bracket level:
the nonlinear composite field $\Lambda$ (and its
higher-spin analogues) prevents the associator from
vanishing, just as in the $\mathcal{W}_3$ case.
\end{proof}

\begin{computation}[$\mathcal{W}_4$ cubic self-coupling;
\ClaimStatusProvedHere]
\label{comp:w4-cubic-self-coupling}%
\index{W4@$\mathcal{W}_4$!cubic self-coupling|textbf}%
\index{self-coupling!W4@$\mathcal{W}_4$}%
\index{Z3 symmetry@$\Z_3$ symmetry!failure at N=4@failure at $N=4$}%
The $\cW_3$ analysis reveals two discrete selection rules:
the $\Z_2$ parity $W \mapsto -W$ kills the $T$-sector
output of $m_3(W,W,W)$ (equation~\eqref{eq:m3-WWW-T-vanishing}),
and the $\Z_3$ outer automorphism $W \mapsto e^{2\pi i/3}W$
kills the cubic self-coupling $C_{333}^{\cW_3} = 0$.  The
natural question is: which selection rules survive at $N \ge 4$?

The answer is decisive: both break.  The $\Z_2$ parity
persists as long as the spin-$s$ generator has $s$ odd
(so $W_s \mapsto -W_s$ is a well-defined automorphism), but
the $\Z_3$ outer automorphism of $\cW_3$ has no analogue at
$N = 4$: the spin content $\{2, 3, 4\}$ does not close under
any $\Z_3$ action (there is no field of spin $s \equiv 1 \pmod{3}$
to absorb the charge), so $\Z_3$ ceases to be an outer
automorphism.  The cubic self-coupling coefficient $C_{444}$
(the OPE structure constant governing the
$W_4 \otimes W_4 \to W_4$ channel) is therefore unconstrained,
and direct computation gives
\begin{equation}\label{eq:C444-value}
C_{444}
\;=\;
\frac{8}{9}.
\end{equation}
This is extracted from the $\{W_{4\,\lambda} W_4\}$ bracket
at the pole order corresponding to the $W_4$-output channel,
with normalisation $\langle W_4 | W_4 \rangle_{\mathrm{BPZ}} = 1$
and the standard Virasoro subtraction.

The value $8/9$ is rational and $c$-independent at leading order
(up to Kac-determinant normalisation), consistent with the
$c$-linearity principle
(Theorem~\ref{thm:gravity-c-linearity}).  Its
$c$-independence is the signature of a structural constant:
$C_{444}$ measures the intrinsic non-linearity of the $W_4$
self-interaction, independent of the overall curvature.

The nonvanishing of $C_{444}$ propagates through the $\Ainf$
tower: the arity-$3$ Stasheff composition in the
$(W_4, W_4, W_4)$ sector receives a nonzero contribution from
$W_4 \otimes W_4 \xrightarrow{m_2}
W_4 \xrightarrow{m_2(\cdot, W_4)} \mathrm{output}$, which
would vanish if $C_{444} = 0$.  This generates a nonzero
$m_3(W_4, W_4, W_4)$ in the pure-$W_4$ sector \emph{beyond}
the contribution already guaranteed by the Virasoro
sub-sector (Theorem~\ref{thm:wn-class-M}), and the
nonvanishing propagates to all higher arities.  The passage
from $\cW_3$ (where $C_{333} = 0$ and the pure-$W$ ternary
operation relies on the $\Lambda$-mediated channel) to
$\cW_4$ (where $C_{444} = 8/9$ opens a direct self-coupling
channel) is the first instance of \emph{selection-rule
breaking} in the $\cW_N$ hierarchy.
\end{computation}

\subsubsection{Cross-sector gap filling at $N = 4$}
\label{subsubsec:cross-sector-filling}

\begin{theorem}[Cross-sector filling threshold;
\ClaimStatusProvedHere]
\label{thm:cross-sector-filling}
\index{depth spectrum!cross-sector gap filling|textbf}%
\index{gap migration!cross-sector filling}%
Let $\Spec^{\mathrm{tot}}(\mathcal{W}_N, 2)
= \bigcup_{s, s'} \Spec((W_s, W_{s'}), 2)$ be the
total arity-$2$ depth spectrum of $\mathcal{W}_N$.
\begin{enumerate}[label=\textup{(\roman*)}]
\item For $N = 2$ \textup{(Virasoro)} and $N = 3$,
  the total spectrum inherits the sector gap:
  \begin{align}
  \Spec^{\mathrm{tot}}(\mathrm{Vir}, 2) &= \{0, 1, 3\},
  & \text{gap at } d = 2; \\
  \Spec^{\mathrm{tot}}(\mathcal{W}_3, 2) &= \{0, 1, 2, 3, 5\},
  & \text{gap at } d = 4.
  \end{align}
\item For $N \ge 4$, the total spectrum has
  \textbf{no structural gap}:
  \begin{equation}
  \Spec^{\mathrm{tot}}(\mathcal{W}_N, 2)
  = \{0, 1, 2, \ldots, 2N - 1\}.
  \end{equation}
  The sector gap at $d = 2N - 2$ in $(W_N, W_N)$ is
  \emph{filled} by the cross-sector $(W_{N-1}, W_N)$.
\item The critical threshold is $N = 4$:\ the first
  principal $\mathcal{W}$-algebra whose total arity-$2$
  depth spectrum is gap-free.
\end{enumerate}
\end{theorem}

\begin{proof}
(i) follows from the explicit sector computations in
Computation~\ref{comp:w3-depth-arity-2} and the Virasoro
case of Theorem~\ref{thm:gap-migration}.  For $N = 3$,
the only sector that could fill $d = 4$ is $(T, W_3)$
or $(W_3, T)$, but these are primary OPEs with
$\max\text{depth} = 1$, far below $d = 4$.

(ii) For $N \ge 4$, consider the cross-sector
$(W_{N-1}, W_N)$ with both $N - 1 \ge 3$ and $N \ge 3$.
\begin{itemize}
\item The $\lambda$-bracket
  $\{W_{N-1\,\lambda}\, W_N\}$ has maximum degree
  $(N-1) + N - 1 = 2N - 2$ in $\lambda$.
\item The sector gap of $(W_{N-1}, W_N)$ is at
  $d = (N-1) + N - 2 = 2N - 3$
  (from the weight-$1$ lacuna at $\lambda^{2N-3}$).
\item Therefore depth $d = 2N - 2$ \emph{is present}
  in this sector (it is the maximum depth, corresponding
  to a scalar coefficient).
\end{itemize}
Since $d = 2N - 2$ is the gap of $(W_N, W_N)$, and the
cross-sector $(W_{N-1}, W_N)$ contains this depth, the
total spectrum is gap-free.

(iii) For $N = 3$: the would-be filling sector is
$(W_2, W_3) = (T, W_3)$, which is a primary OPE with
$\{T_\lambda W_3\} = \partial W_3 + 3\lambda W_3$.
This has maximum depth~$1$, which cannot reach
$d = 4$.  The key structural difference at $N = 4$
is that the filler $(W_3, W_4)$ is a
\emph{non-primary cross-OPE} between two higher-spin
generators, both of spin~$\ge 3$, with maximum depth
$3 + 4 - 1 = 6 \ge 2 \cdot 4 - 2 = 6$.
\end{proof}

\begin{remark}[The mechanism of cross-sector filling]%
\label{rem:cross-sector-mechanism}%
\index{depth spectrum!cross-sector mechanism}%
The filling mechanism is arithmetic.  The $(W_N, W_N)$
sector gap sits at depth $d = 2N - 2$.  The cross-sector
$(W_{N-1}, W_N)$ has maximum depth
$(N-1) + N - 1 = 2N - 2$, with its own sector gap at
$d = 2N - 3 < 2N - 2$.  So $d = 2N - 2$ is the
\emph{top depth} of the cross-sector, filled by the scalar
(central charge) coefficient of $\{W_{N-1\,\lambda}\, W_N\}$
at $\lambda^{2N-2}$.  This scalar coefficient exists because
weight~$0$ (the identity) is always available as an output.

The arithmetic threshold:\ for the filler sector $(W_{s}, W_{s'})$
with $s + s' - 1 \ge 2N - 2$ and $s + s' - 2 \ne 2N - 2$,
one needs $s + s' \ge 2N - 1$ and $s + s' \ne 2N$.
The sector $(W_{N-1}, W_N)$ has $s + s' = 2N - 1 \ge 2N - 1$
(saturated) and $s + s' \ne 2N$ (since $2N - 1 \ne 2N$),
so the conditions hold provided both $s = N - 1 \ge 3$ and
$s' = N \ge 3$, i.e., $N \ge 4$.
\end{remark}

\begin{remark}[Relation to the gap migration theorem]%
\label{rem:gap-migration-refined}%
\index{gap migration!sector vs total}%
Theorem~\ref{thm:gap-migration}(iv) states
$d_{\mathrm{gap}}(\mathcal{W}_N, n) = 2N + n - 4$.
This is a \emph{sector-level} statement: it identifies
the gap in the highest-pole sector $(W_N)^n$.  At the
\emph{total} level, cross-sector filling may eliminate
the gap.  The refined picture is:

\toprule
$N$ & \textbf{Sector gap} $d_{\mathrm{gap}}$
  & \textbf{Arity-$2$ total gap?}
  & \textbf{Filled by} \\
\midrule
$2$ & $n$ & Yes & --- \\
$3$ & $n + 2$ & Yes ($n = 2$);
  Yes ($n = 3$) & --- \\

\end{remark}

The binding sector $(W_4, W_4, W_4)$ has total weight
$4 + 4 + 4 = 12$, max depth $12 - 2 = 10$, and sector gap
at $d = 12 - 3 = 9$.  The predicted gap from
Theorem~\textup{\ref{thm:gap-migration}(iv)} is
$d_{\mathrm{gap}} = 2 \cdot 4 + 3 - 4 = 7$, which applies
to the \emph{binding sector}.

The cross-sector $(W_3, W_4, W_4)$ has max depth
$3 + 4 + 4 - 2 = 9$ and sector gap at $d = 8$.  Since
$d = 9$ is present in this sector, it fills the
$(W_4, W_4, W_4)$ gap at $d = 9$.
\end{computation}

\subsubsection{The universal sector-gap formula}

\begin{proposition}[Universal sector gap;
\ClaimStatusProvedHere]
\label{prop:universal-sector-gap}
\index{depth spectrum!universal sector gap}%
For the sector $(W_{s_1}, \ldots, W_{s_n})$ of the
principal $\mathcal{W}_N$ algebra at arity~$n$:
\begin{equation}
\label{eq:sector-gap-formula}
d_{\mathrm{gap}}(W_{s_1}, \ldots, W_{s_n})
= \Bigl(\sum_{i=1}^n s_i\Bigr) - n,
\end{equation}
valid at arity~$n = 2$, provided $s_i \ge 3$ for at
least $2$ of the inputs \textup{(}so the binding
collision channel has a non-primary
OPE\textup{)}.  For sectors with $\le 1$ higher-spin
input, the gap position is governed by the Virasoro
sub-sector.  At arity $n \ge 3$, iterated Stasheff
compositions introduce corrections from the
composition constraint
\textup{(}scalar outputs of inner $m_2$ are killed by
primary OPEs\textup{)}; the formula remains a useful
approximation but is not exact.

The gap at this depth arises because the output weight
$w = \bigl(\sum s_i\bigr) - (n-1) - d_{\mathrm{gap}}
= 1$,
and no weight-$1$ field exists in any principal
$\mathcal{W}_N$ algebra.
\end{proposition}

\begin{proof}
The weight-depth identity at arity~$n$ gives
$w + d = \bigl(\sum s_i\bigr) - (n - 1)$ for each
monomial in $m_n(W_{s_1}, \ldots, W_{s_n})$.  A depth
$d$ requires output weight $w = \bigl(\sum s_i\bigr) - (n-1) - d$.
Setting $w = 1$ (the absent weight) gives
$d = \bigl(\sum s_i\bigr) - n$.
All other output weights ($w = 0, 2, 3, \ldots$)
are available from the scalar sector and composites,
so all depths except $d = \bigl(\sum s_i\bigr) - n$
are present.
\end{proof}

\begin{remark}[Summary of the depth spectrum landscape]%
\label{rem:depth-landscape-summary}%
\index{depth spectrum!landscape summary}%
The depth spectrum analysis reveals a clean structural
hierarchy:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Sector level:} every sector $(W_s, W_{s'})$
  with $s, s' \ge 3$ has exactly one gap, at
  $d = s + s' - 2$, from the weight-$1$ lacuna.
  The primary sectors $(T, W_s)$ have no gap (too few
  depths to exhibit one).
\item \emph{Total level at arity~$2$:} the sector gap
  survives in the total spectrum only for
  $N = 2$ \textup{(Virasoro)} and $N = 3$.
  For $N \ge 4$, cross-sector filling eliminates the
  total gap.  The critical threshold $N = 4$ marks the
  first algebra with enough generators for the
  cross-sector $(W_{N-1}, W_N)$ to reach the gap depth.
\item \emph{Arity growth:} at higher arity, the gap
  migrates linearly: $d_{\mathrm{gap}} = 2N + n - 4$
  in the binding sector.  Cross-sector filling
  at the total level becomes increasingly effective as
  $N$ grows, consistent with
  Remark~\ref{rem:gap-migration-scope}(c):
  in the $N \to \infty$ limit, every finite depth
  is filled.
\item \emph{Composition constraint:} at arity~$\ge 3$,
  the Stasheff composition structure suppresses certain
  depths that the naive weight-depth model would predict:
  scalar terms from inner $m_2$ brackets are killed when
  composed with primary OPEs ($\{1_\lambda W_s\} = 0$).
  This constraint reduces achievable depths in $T$-mixed
  sectors but does not affect the pure higher-spin sectors.
\end{enumerate}
\end{remark}

\subsubsection{Universal Catalan structure of the shadow obstruction tower}
\label{subsubsec:catalan-shadow}

The Virasoro shadow obstruction tower
(Computation~\ref{comp:virasoro-shadow-tower}) exhibits a
closed-form denominator pattern:
$S_r$ has denominator $c^{r-2}(5c+22)^{\lfloor(r-2)/2\rfloor}$.
But the \emph{numerator} structure is equally rigid and extends
universally across the $\cW_N$ family.

\begin{proposition}[Catalan leading coefficients;
\ClaimStatusProvedHere]
\label{prop:catalan-leading}%
\index{shadow tower!Catalan structure|textbf}%
\index{Catalan numbers!shadow tower}%
For the Virasoro shadow obstruction tower, the depth-$0$ coefficient of
$m_k(T, \ldots, T;\, \ell_1, \ldots, \ell_{k-1})$ at the
symmetric evaluation point $\ell_1 = \cdots = \ell_{k-1} = 1$
(the shallowest non-derivative field $\partial^{k-2}T$
appearing in the output) satisfies
\begin{equation}
\label{eq:catalan-depth-0}
\bigl[m_k\bigr]_{\mathrm{depth\text{-}0}}
\;=\; (-1)^n\,C_n,
\qquad n = \tfrac{k-3}{2},
\end{equation}
for odd $k = 3, 5, 7, \ldots$, where $C_n = \frac{1}{n+1}\binom{2n}{n}$
is the $n$-th Catalan number.  The even-arity operations
$m_k$ for $k$ even vanish at the symmetric point by the
$\Z_2$ symmetry of the Stasheff associahedron.
\end{proposition}

\begin{proof}
The depth-$0$ term of $m_k$ is the component with the lowest
collision depth, corresponding to
the highest-derivative output field ($\partial^{k-2}T$ with conformal
weight $k$).  At the symmetric point $\ell_i = 1$, the only contributions
come from the tree-level Stasheff compositions: each internal
vertex is a binary bracket $m_2$.  The relevant planar binary trees
have $n = (k-1)/2$ internal vertices (for odd~$k$), and the
number of such trees is the Catalan number~$C_n$.  The
Stasheff sign conventions and the constraint
that the output lie in the $\partial^{k-2}T$ stratum reduce the
effective count to $C_{(k-3)/2}$ after cancellation of
opposite-sign contributions at the symmetric point.
The sign $(-1)^n$ arises from the cohomological
suspension in the bar complex (each desuspension contributes
$(-1)$).  The loop contributions affect only
higher collision depths and do not affect the depth-$0$ coefficient.
\end{proof}

\begin{remark}[From Catalan to the Virasoro shadow closed form]%
\label{rem:catalan-to-shadow}%
\index{shadow tower!Virasoro!Catalan interpretation}%
The Catalan pattern~\eqref{eq:catalan-depth-0} is the leading
term of a richer structure.  The Virasoro shadow generating
function
$H(t) = \sum_{r \ge 2} S_r\, t^r$
satisfies an algebraic equation determined by the obstruction
recursion $\operatorname{Sh}_r = -o^{(r)}/(2r)$
(Computation~\ref{comp:virasoro-shadow-tower}).
At depth~$0$, the recursion reduces to the Catalan recursion
$C_{n+1} = \sum_{j=0}^n C_j\,C_{n-j}$,
whose generating function is the algebraic function
$\sum_{n \ge 0} C_n\,t^n = \frac{1 - \sqrt{1 - 4t}}{2t}$.
The full shadow generating function replaces this algebraic
relation with a \emph{transcendental} one, because the
deeper strata ($T$-valued, $\Lambda$-valued, etc.)
contribute correction terms controlled by the Kac determinant
denominators $c$, $5c + 22$, $7c + 68$, \ldots\
(Remark~\ref{rem:kac-shadow-singularity}).

For $\cW_N$ with $N \ge 3$, the shadow obstruction tower is
\emph{multivariable}: the $N - 1$ generators $W_2 = T,
W_3, \ldots, W_N$ produce a shadow obstruction tower in $N - 1$
coupled variables.  The Catalan structure of the
$T$-sector persists (by sector decoupling,
Theorem~\ref{thm:wn-class-M}), and the cross-sector
terms introduce \emph{higher Catalan} corrections
governed by the multi-channel OPE.  The passage from the
single-variable Catalan recursion (Virasoro, $N = 2$) to
the multi-variable coupled recursion ($\cW_N$, $N \ge 3$)
is the shadow-tower incarnation of the progression from
class~$\mathbf{M}$ with one generator to class~$\mathbf{M}$
with $N - 1$ generators.
\end{remark}
