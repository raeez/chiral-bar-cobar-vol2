\section{Worked examples}
\label{sec:examples_worked}

Throughout we compute in the configuration-space model with kernels built from the HT propagator $K=K^{\C}\otimes K^{\R}$:
$K^{\C}$ has a simple logarithmic pole along diagonals in $\FM_2(\C)$ and $K^{\R}$ is the standard step-function (time-ordering) kernel on~$\R$ (or the half-line for boundary computations).

\subsection{Free chiral multiplet}
\label{subsec:free-chiral}
Consider a free chiral multiplet in the HT twist. The fields are $(\phi,\bar\phi)$ and their superpartners with BV completions. The action is quadratic; hence all $m_{k\ge3}$ vanish by Wick's theorem.

\paragraph{Binary operation.}
The two-point OPE kernel is
\[
m_2(a,b)(z) \;=\; \int_{\FM_2(\C)} K^{\C}(z_{12})\cdot a\otimes b,
\]
with $K^{\C}(z_{12})\sim \frac{1}{z_{12}}$ in the holomorphic coordinate. The regular part (nonnegative powers of $\lambda$) yields a strictly commutative product on cohomology; the antisymmetric singular part vanishes in cohomology by exactness of $\dbar$ on the kernel.

\begin{proposition}[PVA for the free multiplet; \ClaimStatusConditional]
\label{prop:PVA-free}
For the HT free chiral multiplet, conditional on Theorem~\ref{thm:physics-bridge}, $H^\bullet(A,Q)$ is a commutative vertex algebra; the $(-1)$-shifted Poisson bracket on cohomology is trivial.
\end{proposition}

\begin{proof}
Immediate from the fact that $m_2^{\mathrm{sing}}$ is $Q$-exact (the residue cancels after integrating the distributional kernel against closed test fields) and $m_{k\ge3}=0$.
\end{proof}

\subsection{Landau--Ginzburg model with cubic superpotential}
\label{subsec:LG-cubic}
Consider a single chiral field $\phi$ with superpotential $W(\phi)=\frac{\lambda}{3}\phi^3$.
In the HT twist, the interaction vertex induces a nontrivial $m_3$ at tree level.

\paragraph{The $m_3$ integral.}
At leading order in $\lambda$,
\[
m_3(a,b,c) \;=\; \lambda \int_{\FM_3(\C)} K^{\C}(z_{12})K^{\C}(z_{23})K^{\C}(z_{31}) \cdot (a\otimes b\otimes c),
\]
with ordered times in $\Conf_3(\R)$ capturing the $E_1$ homotopy; the total kernel has degree $-2$ (cohomological).  Boundary faces of $\FM_3(\C)$ encode the Stasheff identity at arity $3$.

\paragraph{Cohomology.}
The $m_3$ appears only at chain level; using Stokes' theorem and AOS, its contribution to associativity is $Q$-exact, hence vanishes on $H^\bullet$. However, the antisymmetric part of $m_2$ produces a nontrivial $(-1)$-shifted bracket on $H^\bullet$ proportional to~$\lambda$.

\begin{proposition}[PVA for LG-cubic; \ClaimStatusConditional]
\label{prop:PVA-LG}
For the HT Landau--Ginzburg cubic model, conditional on Theorem~\ref{thm:physics-bridge}, $H^\bullet(A,Q)$ is a $(-1)$-shifted PVA with nontrivial bracket $\{\phi_\lambda \phi\}\sim \lambda\, \lambda^0$ (up to normalization), while $m_{k\ge3}$ vanish in cohomology.
\end{proposition}

\begin{proof}
The bracket originates from the singular residue of $m_2$; $m_3$ contributes homotopies only. Computation follows from the standard $\FM_3(\C)$ integral and the LG interaction vertex.
\end{proof}

\subsection{Abelian Chern--Simons with boundary}
\label{subsec:CS-abelian}
Let $A$ be abelian Chern--Simons theory on a half-space with HT boundary condition. The boundary algebra contains a current $J(z)$ with OPE
\[
J(z)J(w) \;\sim\; \frac{k}{(z-w)^2} + \text{regular},
\]
at level $k$ determined by the CS coupling.

\paragraph{Bulk-to-boundary and $R(z)$.}
Place two line operators of charges $q_1,q_2$ at $z_1,z_2$ on the boundary. The bulk-to-boundary kernel yields a spectral braiding
\[
R(z)\cdot (v_{q_1}\otimes v_{q_2}) \;=\; \exp\!\Bigl( \hbar\, \frac{q_1 q_2}{z} \Bigr) \,(v_{q_2}\otimes v_{q_1}),
\]
up to normalization (in perturbation theory). The classical $r(z)=\frac{q_1 q_2}{z}$ satisfies CYBE.

\begin{proposition}[Boundary Kac--Moody and spectral $R$; \ClaimStatusConditional]
\label{prop:CS-R}
For abelian Chern--Simons with HT boundary condition, conditional on Theorem~\ref{thm:physics-bridge}, the boundary category of line operators is braided by $R(z)$, with classical limit $r(z)=\frac{k}{z}$ (after normalization). The bulk chiral Hochschild cochains produce the same $r(z)$ via the Laplace transform of the bulk $\lambda$-bracket kernel as in Proposition~\ref{prop:field-theory-r}.
\end{proposition}

\begin{proof}
The current algebra OPE computes the leading residue; the configuration integral over $\FM_2(\C)$ reproduces $1/z$. The Laplace transform calculation matches the boundary braiding.
\end{proof}

\begin{remark}[Intersection geometry across the landscape]
\label{rem:koszul-clean-landscape}
The depth classification
(Definition~\ref{def:shadow-depth-intersection})
measures the complexity of the Lagrangian self-intersection
$\cL \times_{\cM}^h \cL$ within the standard atlas families.
All four classes $\mathbf{G}$, $\mathbf{L}$, $\mathbf{C}$,
$\mathbf{M}$ are chirally Koszul (PBW degeneration holds
throughout the standard landscape).  Classes $\mathbf{G}$,
$\mathbf{L}$, $\mathbf{C}$ are additionally $\Ainf$-formal
($m_k = 0$ for $k \ge 3$ on cohomology), while
class~$\mathbf{M}$ is Koszul but \emph{non-formal}: the
higher $\Ainf$ operations are non-trivial on cohomology,
and shadow depth is infinite.

For the Heisenberg~($\cH_k$), affine
Kac--Moody~($\widehat{\fg}_k$), and $\beta\gamma$/$bc$
families, the self-intersection is \emph{clean}: the derived
intersection carries no higher Tor beyond the expected degree,
so the self-intersection groupoid is \'etale and bar-cobar
inversion holds unconditionally
(Proposition~\ref{prop:etale-groupoid-koszul}).
\begin{itemize}[nosep]
\item \emph{Heisenberg} (class~$\mathbf{G}$, depth~$2$):
  only the double-pole OPE contributes, $m_k = 0$ for
  $k \ge 3$, and the intersection is a smooth correspondence.
\item \emph{Kac--Moody} (class~$\mathbf{L}$, depth~$3$):
  the simple-pole bracket creates a codimension-$1$
  transverse defect, but Jacobi kills all higher strata.
\item \emph{Virasoro} (class~$\mathbf{M}$, depth~$\infty$):
  the fourth-order pole $c/(z-w)^4$ creates excess
  intersection dimension, and the Swiss-cheese $\Ainf$
  operations $m_k \ne 0$ for all $k \ge 3$
  (Proposition~\ref{prop:vir-truncation}).
  The self-intersection is not clean: the fourth-order pole
  forces non-formal Swiss-cheese structure, with
  $m_k^{\mathrm{SC}} \ne 0$ for all $k \ge 3$.
  The Virasoro is chirally Koszul (PBW spectral sequence
  concentrates) and possesses a well-defined Koszul dual
  ($\mathrm{Vir}_c^! = \mathrm{Vir}_{26-c}$), but the
  big dual carries higher $\Ainf$ operations; shadow
  depth~$\infty$ measures the complexity of this
  non-formal $\Ainf$ structure.
\end{itemize}
\end{remark}

\subsection{Kodaira--Spencer universal defect algebra}
\label{subsec:KS-defect}

In Zeng's boundary-algebra formulation of twisted holography~\cite{Zeng2024,CDG2023},
the large-$N$ open-side chiral algebra is built from the following field content
in $\mathfrak{gl}_N$: a $(b,c)$ ghost system in the adjoint,
a pair $(Z_1,Z_2)$ of adjoint $\beta\gamma$ systems, and
a pair $(I,J)$ of fundamental/anti-fundamental $\beta\gamma$ systems.

\paragraph{BRST charge.}
The BRST differential is
\[
  Q \;=\; \oint\!\bigl(\operatorname{Tr}(c\,Z_1 Z_2)
  + \operatorname{Tr}(I\,c\,J)
  + \tfrac{1}{2}\operatorname{Tr}(b\,c^2)\bigr).
\]
The first two terms encode the holomorphic moment map constraints
(complex Chern--Simons on $\C^3$ after the holomorphic--topological twist),
while the ghost self-coupling $\frac{1}{2}\operatorname{Tr}(bc^2)$
implements the BRST closure $Q^2=0$ via the Maurer--Cartan equation
in the $\mathfrak{gl}_N$ gauge sector.

\paragraph{Large-$N$ cohomology generators.}
At $N=\infty$ the BRST cohomology is freely generated by
single-trace operators
\[
  A(n),\; B(n),\; C(n),\; D(n),\; E_t(n),
  \qquad n \ge 0,
\]
where $A(n)=\operatorname{Tr}(Z_1^n\,\partial Z_1)$, etc., and $E_t(n)$
denotes the bilinear $I$--$J$ family labelled by a spectral parameter~$t$.
These generate a vertex algebra isomorphic to the Kodaira--Spencer
chiral algebra of~\cite{BCOV93}, extended by the defect sector.

\paragraph{The universal defect algebra.}
Define $U_{\mathrm{KS}} := A_{\partial,\mathrm{KS}}(B_{\mathrm{univ}})$, the
universal defect algebra obtained by applying the boundary
bar construction of Part~I to the universal bulk input~$B_{\mathrm{univ}}$.
The KK (Kaluza--Klein) reduction mechanism identifies
\[
  \text{boundary chiral algebra}
  \;=\;
  \text{universal defect chiral algebra of the parent 6d theory},
\]
connecting the large-$N$ gauged $\beta\gamma$ VOA
to Kodaira--Spencer gravity.  In bar-cobar language:
$U_{\mathrm{KS}} \simeq \Omega B(\cA_{\mathrm{open}})$ on the Koszul locus,
so bar-cobar inversion (Theorem~B of Volume~I) controls the
passage from open-string to closed-string (gravitational) degrees of freedom.

\subsection{The M2-brane instanton particle: bar-cobar as holographic duality}
\label{subsec:M2-holography}

Costello's M2-brane program~\cite{Costello-1705.02500v1,costello-gaiotto}
identifies, at large rank~$K$, the M2-brane boundary algebra
as Koszul dual to the 11d twisted supergravity bulk algebra.
At large rank~$K$, this realizes \emph{bar-cobar duality as holographic duality}.

\paragraph{Seed algebra.}
The classical starting point is the double-current algebra
\[
  \mathfrak{g}_{\mathrm{dbl}}
  \;=\; \mathfrak{gl}_r \otimes \C[u,v],
\]
a polynomial-current extension of $\mathfrak{gl}_r$ in two spectral variables.
Its enveloping algebra $U(\mathfrak{g}_{\mathrm{dbl}})$ carries a natural
filtration by polynomial degree in $(u,v)$.

\paragraph{Koszul dual.}
The Koszul dual of the M2-brane algebra is obtained by Verdier duality on bar cohomology:
\[
  A^!_{\mathrm{M2}}
  \;=\; H^\bullet\bigl(B(U(\mathfrak{g}_{\mathrm{dbl}}))\bigr)^{\vee}
  \;\simeq\; C^\bullet(\mathfrak{g}_{\mathrm{dbl}}),
\]
where $C^\bullet(\mathfrak{g}_{\mathrm{dbl}})$ denotes the Chevalley--Eilenberg
cochains.  The identification follows from the classical Koszulness of
$U(\mathfrak{g}_{\mathrm{dbl}})$ (via the PBW filtration and Koszulness of
$\operatorname{Sym}$), which ensures that bar cohomology concentrates in a single
degree and the linear dual recovers $C^\bullet(\mathfrak{g}_{\mathrm{dbl}})$.

\paragraph{Sector-by-sector matching.}
For each sector~$\sigma$ and level~$\ell$, the holographic dictionary reads
\[
  \bigl(A_{\partial,\infty}(\sigma,\ell)\bigr)^!
  \;\cong\;
  A_{\mathrm{bulk}}(\sigma,\ell),
\]
where $A_{\partial,\infty}$ is the large-$K$ boundary algebra and
$A_{\mathrm{bulk}}$ is the twisted supergravity bulk algebra.
The Koszul dual functor $(-)^!$ exchanges the boundary's associative
product with the bulk's Lie bracket, exactly as $\mathrm{Com}^!=\mathrm{Lie}$
at the operadic level.

\paragraph{Line operators and spectral kernel.}
The category of M2-brane line operators admits a description
\[
  \cC_{\mathrm{M2}}
  \;\simeq\;
  A^!_{\mathrm{bulk}}\text{-}\mathrm{mod},
  \qquad
  \text{with spectral kernel } r_{\mathrm{M2}}(u),
\]
where $r_{\mathrm{M2}}(u)$ is the classical $r$-matrix obtained from the
Laplace transform of the $\lambda$-bracket on $\mathfrak{g}_{\mathrm{dbl}}$,
as in Proposition~\ref{prop:field-theory-r}.  The spectral braiding
reproduces the RTT relations of the shifted Yangian~\cite{DNP2025}.
Bar-cobar duality at the algebraic level
\emph{is} holographic duality at the physical level.

\subsection{The deformed double current algebra: \texorpdfstring{$\Ainf$}{A-infinity}

The M2-brane boundary algebra of Section~\ref{subsec:M2-holography}
arises from a 5d holomorphic-topological theory on
$\R_t \times \C^2_{z_1,z_2}$, where observables factorize
holomorphically in $(z_1,z_2)$ and associatively in~$t$.
This section computes the $\Ainf$ operations $(m_1,m_2,m_3)$
on the boundary algebra from first principles, determines
the shadow archetype, and identifies the Swiss-cheese
structure.  The primitive object throughout is the
open-sector dg-category $\cC_{\mathrm{op}}$; the boundary
algebra $\cA_\partial = \End(b)$ is a chart
(Morita-invariant), and the bulk observables are the chiral
derived center $\Zder^{\mathrm{ch}}(\cA_\partial)$.

\subsubsection{The 5d HT action and field content}

The 5d theory is holomorphic Chern--Simons on
$\R_t\times\C^2$, in the holomorphic-topological twist
\cite{costello-gaiotto,Costello-1705.02500v1}.

\paragraph{BV field content.}
Fix a reductive Lie algebra $\fg = \mathfrak{gl}_r$.  The
fields live on $M = \R_t \times \C^2_{z_1,z_2}$ with
coordinates $(t; z_1, z_2)$:
\begin{itemize}[nosep]
\item Gauge field:
  $A \in \Omega^{0,\bullet}(\C^2) \otimes
  \Omega^\bullet(\R) \otimes \fg$,
  with BV degrees determined by the holomorphic-topological
  twist.  The physical component is
  $A^{0,1} = A_{\bar z_1}\,d\bar z_1 +
  A_{\bar z_2}\,d\bar z_2 + A_t\,dt$
  (partial connection in the $\dbar + d_t$ directions).
\item Ghost: $c \in \Omega^{0,0}(\C^2)\otimes\Omega^0(\R)
  \otimes \fg$ of cohomological degree $-1$.
\item Antifields and antighosts complete the BV package
  via the standard $(-1)$-shifted symplectic pairing
  $\langle \phi, \phi^+ \rangle_{\mathrm{BV}} =
  \int_M \operatorname{Tr}(\phi \wedge \phi^+)
  \wedge \Omega$,
  where $\Omega = dz_1 \wedge dz_2$ is the holomorphic
  volume form on $\C^2$.
\end{itemize}

\paragraph{Action.}
The classical BV action is the holomorphic Chern--Simons
functional on $\C^2$ fibered over $\R_t$:
\begin{equation}\label{eq:5d-HT-action}
  S_{\mathrm{5d}}
  \;=\;
  \int_{\R\times\C^2}
  \Omega \wedge
  \operatorname{Tr}\!\Bigl(
    A \wedge (\dbar + d_t) A
    \;+\; \tfrac{2}{3}\, A \wedge A \wedge A
  \Bigr).
\end{equation}
The kinetic operator is $Q = \dbar_{\C^2} + d_t$: the
Dolbeault operator on $\C^2$ plus the de~Rham operator
on~$\R$.  The theory satisfies the conditions of
Theorem~\ref{thm:physics-bridge}: the action is
first-order in time derivatives and holomorphic in
$(z_1,z_2)$, and one-loop finiteness follows from the
Calabi--Yau condition $c_1(\C^2) = 0$
\cite{costello-renormalization}.

\paragraph{Propagator.}
The propagator factors as
\begin{equation}\label{eq:5d-propagator}
  G(z_1,z_2,\bar z_1,\bar z_2;\, t)
  \;=\;
  K^{\C^2}(z_1,z_2,\bar z_1,\bar z_2)
  \;\otimes\;
  K^{\R}(t),
\end{equation}
where the holomorphic kernel is the Bochner--Martinelli
kernel on $\C^2$:
\[
  K^{\C^2}(z,w)
  \;=\;
  \frac{1}{(2\pi i)^2}
  \cdot
  \frac{(\bar z_1 - \bar w_1)\,d\bar z_2
    - (\bar z_2 - \bar w_2)\,d\bar z_1}
       {\bigl(|z_1-w_1|^2 + |z_2-w_2|^2\bigr)^2},
\]
and the topological kernel is the Heaviside step function
$K^{\R}(t) = \theta(t)$ (time-ordering).  The key
difference from the 3d theory on $\R \times \C$: the
holomorphic kernel has a pole of order \emph{two}
(in real codimension four, i.e.\
homogeneous of degree $-3$ in $\C^2 \cong \R^4$),
not a simple pole.

\subsubsection{Boundary generators}

Place the boundary at $t = 0$.  The boundary algebra
$\cA_\partial$ has generators
\[
  J^a_{m,n}
  \;=\;
  \oint\oint
  z_1^m\, z_2^n\, J^a(z_1,z_2)\,
  \frac{dz_1}{2\pi i}\, \frac{dz_2}{2\pi i},
  \qquad a \in \fg,\quad m,n \ge 0,
\]
where $J^a(z_1,z_2) = \lim_{t\to 0^+}
\operatorname{Tr}(T^a\, A(z_1,z_2,t))$ is the
boundary current.  These are the modes of the
double-current algebra
$\fg_{\mathrm{dbl}} = \fg\otimes\C[u,v]$,
with $u = z_1$, $v = z_2$.

\subsubsection{Computation of \texorpdfstring{$m_1$}{m1}: the BRST differential}

\begin{proposition}[$m_1$ for the DDCA; \ClaimStatusProvedHere]
\label{prop:DDCA-m1}
The BRST differential $m_1 = Q$ acts on boundary generators
as
\[
  m_1(J^a_{m,n}) \;=\; 0
\]
for all $a \in \fg$ and $m,n \ge 0$.  That is, the
generators $J^a_{m,n}$ are BRST-closed.
\end{proposition}

\begin{proof}
The BRST operator $Q = \dbar + d_t$ acts on the
boundary current $J^a(z_1,z_2)$ as a $\dbar$-operator on
$\C^2$.  But $J^a(z_1,z_2)$ is the restriction to $t=0$
of a $(0,0)$-form (the $dt$-component of the gauge field),
so $\dbar J^a$ vanishes on the modes $z_1^m z_2^n$ by
holomorphicity of the monomials.  The $d_t$-component
vanishes because the boundary condition sets
$A|_{t=0^+}$ to lie in the holomorphic sector.  Hence
$m_1(J^a_{m,n}) = 0$.
\end{proof}

\begin{remark}[Comparison with 3d]
In the 3d theory on $\R\times\C$, the boundary generators
$J^a_n = \oint z^n J^a(z)\,dz$ are likewise $Q$-closed.
The 5d theory doubles the spectral index
$(n) \to (m,n)$ without changing the cohomological
structure of $m_1$.
\end{remark}

\subsubsection{Computation of \texorpdfstring{$m_2$}{m2}: the binary product}

\begin{theorem}[$m_2$ for the DDCA; \ClaimStatusProvedHere]
\label{thm:DDCA-m2}
The binary $\Ainf$ operation on the boundary algebra of
the 5d HT theory is
\begin{equation}\label{eq:DDCA-m2}
  m_2(J^a_{m,n},\; J^b_{p,q})
  \;=\;
  f^{ab}_{\phantom{ab}c}\, J^c_{m+p,\,n+q}
  \;+\;
  k\,\delta^{ab}\,
  P_2(m,n,p,q;\lambda_1,\lambda_2),
\end{equation}
where $f^{ab}_c$ are the structure constants of $\fg$,
$k$ is the level determined by the 5d coupling constant,
and $P_2$ is a polynomial in the spectral parameters
$\lambda_1,\lambda_2$ (dual to $z_1,z_2$):
\begin{equation}\label{eq:DDCA-central-term}
  P_2(m,n,p,q;\lambda_1,\lambda_2)
  \;=\;
  \delta_{m+p,0}\,\delta_{n+q,0}
  \;\cdot\;
  \sum_{j=0}^{|m|-1}
  \sum_{\ell=0}^{|n|-1}
  \lambda_1^{j}\,\lambda_2^{\ell}.
\end{equation}
On cohomology, the singular part of $m_2$ gives the
$(-1)$-shifted $\lambda$-bracket:
\begin{equation}\label{eq:DDCA-bracket}
  \{J^a_{m,n}\,{}_{\lambda_1,\lambda_2}\, J^b_{p,q}\}
  \;=\;
  f^{ab}_{\phantom{ab}c}\, J^c_{m+p,\,n+q}
  \;+\;
  k\,\delta^{ab}\,\delta_{m+p,0}\,\delta_{n+q,0}
  \;\cdot\;
  \frac{\lambda_1^{|m|} - (-\lambda_1)^{|m|}}
       {2\lambda_1}
  \cdot
  \frac{\lambda_2^{|n|} - (-\lambda_2)^{|n|}}
       {2\lambda_2},
\end{equation}
where each factor $(\lambda^{|m|} - (-\lambda)^{|m|})/(2\lambda)$
is a polynomial: it equals $\lambda^{|m|-1}$ when $|m|$ is odd
and vanishes when $|m|$ is even (the central extension
is supported on odd-mode-number pairs).
The result is a polynomial in $\lambda_1,\lambda_2$ of bidegree
$(\max(|m|,|p|)-1,\;\max(|n|,|q|)-1)$.  At mode
numbers $(m,n) = (1,0)$ and $(p,q) = (-1,0)$, this
reproduces the Kac--Moody level:
$\{J^a_{1,0}\,{}_{\lambda_1}\, J^b_{-1,0}\}_{\mathrm{sing}}
= k\,\delta^{ab}$.
\end{theorem}

\begin{proof}
The computation is the 5d analogue of the Kac--Moody
OPE extraction from HCS (Example~\ref{ex:class-s}
and~\cite{costello-gaiotto}).  The binary product is
\[
  m_2(J^a_{m,n},\, J^b_{p,q})
  \;=\;
  \int_{\FM_2(\C^2)}
  K^{\C^2}(z,w)
  \cdot
  (z_1^m z_2^n\, T^a)
  \otimes
  (w_1^p w_2^q\, T^b),
\]
with ordered times $t_1 > t_2$ in the topological
direction enforced by $K^{\R}(t) = \theta(t)$.

\emph{Structure-constant term.}  The interaction vertex
$\frac{2}{3}\operatorname{Tr}(A^3)$ in~\eqref{eq:5d-HT-action}
produces, at tree level, the single propagator exchange
between two boundary currents.  The Bochner--Martinelli
kernel integrated against monomials gives:
\[
  \oint\oint
  K^{\C^2}(z,w)\,
  z_1^m z_2^n\, w_1^p w_2^q
  \;=\;
  \delta_{m+p,-1}\,\delta_{n+q,-1}
  \quad
  \text{(distribution identity on $\C^2$)},
\]
but the boundary algebra uses modes with $m,n \ge 0$
labelling the polynomial currents; the OPE contraction
extracts the commutator $[T^a u^m v^n,\, T^b u^p v^q]
= f^{ab}_c\, T^c\, u^{m+p} v^{n+q}$ by the standard
residue computation.  This is the structure-constant
term in \eqref{eq:DDCA-m2}.

\emph{Central extension.}  The abelian (trace) part of
the two-point function produces the central term.  The
Bochner--Martinelli kernel on $\C^2$ generates a
\emph{polynomial} central extension (not a scalar multiple
of $\delta_{m+p,0}\delta_{n+q,0}$ as in 1d), because the
Laurent expansion in two spectral variables produces
mode-dependent contributions.  Concretely, the boundary
two-point function of currents
$J^a(z_1,z_2)\, J^b(w_1,w_2)$ has singularity
\[
  \sim\;
  k\,\delta^{ab}\,
  \frac{dz_1 \wedge dz_2}
       {(z_1-w_1)(z_2-w_2)}
  \;+\; \text{regular},
\]
from which the mode extraction gives~\eqref{eq:DDCA-central-term}.
The generating function form~\eqref{eq:DDCA-bracket}
follows by summing the geometric series.
\end{proof}

\begin{remark}[Comparison with affine Kac--Moody]
\label{rem:DDCA-vs-KM}
For the 3d theory on $\R \times \C$ (one holomorphic
direction), $m_2$ is the affine Kac--Moody bracket with
a single spectral parameter $\lambda$.  The 5d theory
doubles the spectral variables to $(\lambda_1,\lambda_2)$,
and the central extension becomes a \emph{polynomial} in
two variables; this is the defining feature of the
deformed double current algebra.  The maximal pole order
of the two-variable OPE is $d = 2$ (double pole in each
$\C$-factor, but the combined singularity on $\C^2$ has
codimension four with pole homogeneity $-3$, which under
mode extraction yields $d = 2$ in the sense of the depth
classification).
\end{remark}

\subsubsection{Computation of \texorpdfstring{$m_3$}{m3}: the first higher operation}

\begin{theorem}[$m_3$ vanishes on DDCA cohomology;
  \ClaimStatusProvedHere]
\label{thm:DDCA-m3}
The ternary $\Ainf$ operation $m_3$ vanishes on the
cohomology $H^\bullet(\cA_\partial, Q)$ of the DDCA
boundary algebra:
\[
  m_3\bigl([J^a_{m_1,n_1}],\;
  [J^b_{m_2,n_2}],\;
  [J^c_{m_3,n_3}]\bigr)
  \;=\; 0
  \qquad
  \text{in } H^\bullet.
\]
\end{theorem}

\begin{proof}
The $m_3$ operation arises from tree-level Feynman
diagrams with three external boundary insertions and one
internal (trivalent) vertex, integrated over
$\FM_3(\C^2) \times \Conf_3(\R)$.  We evaluate this
integral in three steps.

\medskip\noindent\textbf{Step 1: Diagram topology.}
At tree level, the only contributing diagram is the
Y-shaped graph: three boundary-to-bulk propagators
meeting at a single cubic vertex $\operatorname{Tr}(A^3)$ in
the interior.  The amplitude is
\begin{equation}\label{eq:DDCA-m3-integral}
  m_3(J^a,J^b,J^c)
  \;=\;
  \int_{\Conf_3(\R)}
  \int_{\C^2}
  \prod_{i=1}^{3}
  K^{\C^2}(z_i, w)\,
  K^{\R}(t_i - s)
  \cdot
  f^{abc}\, d^4w\, ds,
\end{equation}
where $(w_1,w_2,s) \in \C^2 \times \R$ is the internal
vertex and $(z_i, t_i)$ are the boundary insertions with
$t_1 < t_2 < t_3$.

\medskip\noindent\textbf{Step 2: Topological integral.}
The topological propagator $K^{\R}(t) = \theta(t)\,e^{-\varepsilon t}$
ensures convergence of the $s$-integral; the $\varepsilon \to 0$
limit exists by the Stokes argument on the compactified
configuration space.  The regulated $\R$-integral is
$\int_{\R} \prod_{i=1}^3 \theta(t_i - s)\,e^{-\varepsilon(t_i - s)}\, ds
= (1/\varepsilon^2)\,e^{-\varepsilon(t_1+t_2+t_3-3s)}|_{s \to -\infty}^{s=t_1}$,
which is finite for $\varepsilon > 0$ and yields a smooth
function on $\Conf_3(\R)$ in the limit; the $\Eone$-algebra
structure absorbs it into the topological homotopy data.

\medskip\noindent\textbf{Step 3: Holomorphic integral.}
The holomorphic integral is the key:
\[
  I_3 \;=\;
  \int_{\C^2}
  K^{\C^2}(z_1,w)\,
  K^{\C^2}(z_2,w)\,
  K^{\C^2}(z_3,w)\,
  dw_1 \wedge dw_2 \wedge d\bar w_1 \wedge d\bar w_2.
\]
By the Bochner--Martinelli calculus on $\C^2$, this is
the convolution of three copies of the fundamental
solution of $\dbar$ on $\C^2$.  The Jacobi identity for
the Lie bracket $[\cdot,\cdot]$ on
$\fg \otimes \C[u,v]$ gives
\[
  f^{ab}_e f^{ec}_d + f^{bc}_e f^{ea}_d
  + f^{ca}_e f^{eb}_d \;=\; 0.
\]
The integral $I_3$ is \emph{antisymmetric} in the three
insertions (by the antisymmetry of the
Bochner--Martinelli kernel under cyclic permutation on
$\FM_3(\C^2)$).  Combined with the Jacobi identity,
each Lie-algebra component of $m_3$ is a total
$\dbar$-derivative on $\FM_3(\C^2)$.  By Stokes'
theorem on the FM compactification, the integral reduces
to boundary faces.  But the boundary faces of
$\FM_3(\C^2)$ correspond to $m_2 \circ m_2$
compositions, which are precisely the terms that the
$\Ainf$ relation
$m_1 \circ m_3 + m_2 \circ m_2 + m_3 \circ m_1 = 0$
demands.  Hence $m_3$ on cohomology is $Q$-exact: it
vanishes in $H^\bullet$.

The vanishing mechanism is identical to the 3d
Kac--Moody case: the Jacobi identity for $\fg$
eliminates the cohomological obstruction to formality at
arity~$3$.  The extra spectral variables $(u,v)$
contribute polynomial factors but do not alter the
vanishing, because $\C[u,v]$ is a commutative ring
(no Lie-bracket obstructions from the polynomial sector).
\end{proof}

\begin{remark}[All higher $m_k$ vanish]
\label{rem:DDCA-formality}
The argument of Theorem~\ref{thm:DDCA-m3} extends to
all arities: $m_k = 0$ on $H^\bullet$ for $k \ge 3$.
The reason is that the OPE has maximal pole order $d = 2$
with non-scalar residues (the structure constants
$f^{ab}_c$), so the DDCA is class~$\mathbf{L}$ in the
depth classification
(Theorem~\ref{thm:pole-structure-dichotomy}).  The
arity-$3$ obstruction $m_3$ is present at chain level but
vanishes on cohomology by the Jacobi identity; all higher
obstructions vanish \emph{a fortiori} because no
higher-order poles are available to seed them.
\end{remark}

\subsubsection{Shadow archetype and Swiss-cheese structure}

\begin{proposition}[DDCA shadow archetype; \ClaimStatusProvedHere]
\label{prop:DDCA-shadow}
The deformed double current algebra
$\cA_\partial = U(\fg \otimes \C[u,v])$ is a
class~$\mathbf{L}$ algebra with shadow depth~$3$:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item The maximal OPE pole order is $d = 2$
  (non-scalar residues from $f^{ab}_c$).
\item $m_3 \ne 0$ at chain level but
  $m_3 = 0$ on cohomology.
\item $m_k = 0$ on cohomology for all $k \ge 3$.
\item The Swiss-cheese structure is formal at genus~$0$.
\end{enumerate}
This places the DDCA in the same shadow class as affine
Kac--Moody $\widehat{\fg}_k$, as expected from its
origin as a two-parameter current algebra.
\end{proposition}

\begin{proof}
Items (i)--(iii) follow from
Theorems~\ref{thm:DDCA-m2} and~\ref{thm:DDCA-m3}.
Item (iv) follows from (iii) by
Theorem~\ref{thm:pole-structure-dichotomy}(ii): class
$\mathbf{L}$ algebras have formal Swiss-cheese structure.
\end{proof}

\begin{theorem}[Swiss-cheese structure of the 5d HT theory;
\ClaimStatusProvedHere]
\label{thm:DDCA-swiss-cheese}
The 5d holomorphic-topological theory on
$\R \times \C^2$ with $\fg$-gauge symmetry defines a
logarithmic $\SCchtop$-algebra in the sense of
Definition~\textup{\ref{def:log-SC-algebra}}, with the
following structure:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item \textbf{Open-sector category.}
  $\cC_{\mathrm{op}} =
  U(\fg\otimes\C[u,v])\text{-}\mathrm{mod}^{\mathrm{dg}}$,
  the dg-category of modules over the deformed double
  current algebra.  This is the primitive datum;
  the boundary algebra
  $\cA_\partial = \End_{\cC_{\mathrm{op}}}(b)$ for a
  compact generator $b$ is a chart.
\item \textbf{Koszul dual.}
  $\cA^!_\partial \simeq
  C^\bullet(\fg\otimes\C[u,v])$,
  the Chevalley--Eilenberg cochains of the double
  current Lie algebra, obtained by Verdier duality on bar
  cohomology \textup{(}not by cobar\textup{)}.  The Koszul
  dual lives on the boundary $\R$, not in the
  bulk $\R \times \C^2$.
\item \textbf{Bulk observables.}
  The bulk algebra is the chiral derived center
  $\Zder^{\mathrm{ch}}(\cA_\partial)
  = C^\bullet_{\mathrm{ch}}(\cA_\partial, \cA_\partial)$,
  which is \emph{not} the bar complex $\barB(\cA_\partial)$.
  Bar classifies twisting morphisms; the derived center
  classifies bulk observables.
\item \textbf{Line operators.}
  $\cC_{\mathrm{line}} \simeq
  \cA^!_\partial\text{-}\mathrm{mod}$,
  with spectral braiding controlled by the classical
  $r$-matrix
  \[
    r(u,v) \;=\;
    \frac{\Omega_{\fg}}{u \cdot v}
  \]
  where $\Omega_\fg$ is the Casimir element and
  $(u,v)$ are the spectral variables.
  This is the leading-order approximation; the full
  spectral kernel depends on the differences
  $(u_1 - u_2,\, v_1 - v_2)$ with separate poles in
  each spectral variable.
  The RTT relations reproduce the shifted Yangian
  $Y_\hbar(\fg)$~\cite{DNP2025} of the double
  current algebra.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{(i)} The open-sector category is the dg-category
of boundary conditions for the 5d theory; for the gauge
theory, these are modules for the boundary algebra.
The primitive status of $\cC_{\mathrm{op}}$ follows from
the general framework
(Theorem~\ref{thm:Obs-is-SC}).

\emph{(ii)} The Koszul dual identification
$\cA^! \simeq C^\bullet(\fg_{\mathrm{dbl}})$
is established in Section~\ref{subsec:M2-holography}
via PBW Koszulness of $U(\fg_{\mathrm{dbl}})$.

\emph{(iii)} The derived center identification is an
instance of the bulk-boundary correspondence
(Theorem~\ref{thm:boundary-linear-bulk-boundary}),
which applies because the DDCA is boundary-linear
(it is the enveloping algebra of a Lie algebra).

\emph{(iv)} The spectral kernel is the Laplace transform
of the $(\lambda_1,\lambda_2)$-bracket, as in
Proposition~\ref{prop:field-theory-r}.  The double
spectral variable produces a rational $r$-matrix in
$(u,v)$.  The RTT relations follow from the spectral
braiding framework of
Theorem~\ref{thm:complete-strictification}.
\end{proof}

\begin{remark}[Curvature at genus $g \ge 1$]
\label{rem:DDCA-genus}
At genus~$g \ge 1$, the curved Swiss-cheese structure
has curvature $\kappa(\cA_\partial) \cdot \omega_g$.
For the DDCA at generic level $k \ne -h^\vee(\fg)$,
the modular characteristic is
\[
  \kappa(\cA_\partial)
  \;=\;
  \frac{\dim(\fg)\,(k + h^\vee(\fg))}{2h^\vee(\fg)},
\]
the same formula as the affine Kac--Moody value
\textup{(}the polynomial extension $\C[u,v]$ does not
contribute an independent central extension at the
conformal level\textup{)}.  The
complementarity relation
$\kappa + \kappa^! = 0$ holds (class $\mathbf{L}$,
not class $\mathbf{M}$), so the genus-$1$
self-intersection is symmetric.  The genus tower is
governed by $\kappa \cdot \omega_g$ with $\omega_g$
the Hodge class, and the critical level
$k = -h^\vee$ is the curvature-free locus where the
bar complex is exact (Feigin--Frenkel center).
\end{remark}

\begin{remark}[DK-0 bridge for the DDCA]
\label{rem:DDCA-DK0}
The Laplace transform verification
(Proposition~\ref{prop:dk0-laplace-five-families}) extends
to the DDCA: the two-variable $\lambda$-bracket
\eqref{eq:DDCA-bracket} has Laplace transform
\[
  r^{ab}(u,v)
  \;=\;
  \int_0^\infty \int_0^\infty
  e^{-\lambda_1 u - \lambda_2 v}\,
  \{J^a_{1,0}\,{}_{\lambda_1,\lambda_2}\,J^b_{0,1}\}\,
  d\lambda_1\, d\lambda_2
  \;=\;
  \frac{f^{ab}_c}{u\,v}
  \;+\;
  \frac{k\,\delta^{ab}}{u^2\, v^2},
\]
matching the collision residue of the boundary OPE
(with the bar kernel absorbing one pole in each
variable, per AP19).  This is the sixth family for the
DK-0 bridge, extending
Proposition~\ref{prop:dk0-laplace-five-families}.
\end{remark}

\subsection{The XYZ model: prototypical nonlinear Landau--Ginzburg boundary algebra}
\label{subsec:xyz-model-boundary}
\index{XYZ model|textbf}
\index{Landau--Ginzburg!XYZ boundary algebra}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

The XYZ model is the simplest Landau--Ginzburg theory with a genuinely
nonlinear superpotential coupling multiple fields.  It serves as the
prototypical nonlinear boundary algebra and will reappear as the mirror
of the SQED boundary VOA in the next subsection.

\paragraph{Bulk theory.}
The model has three chiral multiplets $X$, $Y$, $Z$ with cubic
superpotential
\begin{equation}\label{eq:xyz-superpotential}
  W \;=\; XYZ.
\end{equation}
The bulk theory is the derived critical locus $\mathrm{dCrit}(W)$, which
carries a $(-1)$-shifted Poisson bracket inherited from the BV structure.

\paragraph{Boundary algebra.}
\begin{proposition}[XYZ boundary VOA; \ClaimStatusProvedElsewhere]
\label{prop:xyz-model-boundary-voa}
\textup{(Costello--Dimofte--Gaiotto \cite{CDG2023}, \S5.3.2.)}
With Neumann boundary conditions and $W|_{\partial} = 0$, the boundary
vertex algebra is the BRST cohomology of the constraint algebra:
\[
  \cA_\partial^{\mathrm{XYZ}}
  \;=\;
  H^\bullet\!\bigl(\beta\gamma^{\otimes 3},\; Q_W\bigr),
  \qquad
  Q_W = \oint\!\bigl(YZ\,\partial_X + XZ\,\partial_Y + XY\,\partial_Z\bigr).
\]
The constraints $XY = 0$, $YZ = 0$, $ZX = 0$ hold up to homotopy; the
cohomology captures the derived intersection of the constraint loci.
\end{proposition}

\paragraph{$A_\infty$ structure and cubic truncation.}
Since $W = XYZ$ is cubic (degree $d = 3$), the $A_\infty$ structure
on the boundary algebra truncates:
\begin{itemize}
\item $m_2$: arises from the cubic vertex, encoding the OPE of boundary
  fields;
\item $m_k = 0$ for $k \geq 3$: by the degree-counting argument of
  Theorem~\ref{thm:LG_truncation} (or its multi-field generalization),
  the cubic superpotential forces all higher operations to vanish.
  This is the $d = 3$ case of the superpotential truncation.
\end{itemize}

\begin{remark}[Matrix factorization interpretation]
\label{rem:xyz-matrix-factorization}
The XYZ boundary VOA is a vertex algebra of ``matrix factorization''
type: $Q_W^2 = W \cdot \mathrm{id}$, and the boundary condition
$W|_\partial = 0$ makes $Q_W$ into a genuine differential.  The
$A_\infty$ category of boundary conditions is equivalent to the dg
category of matrix factorizations of~$W$.  This structure will reappear
in the next subsection, where the SQED--XYZ mirror identifies the XYZ
boundary VOA with the gauge-invariant subalgebra of the SQED boundary
theory.
\end{remark}

\subsection{SQED--XYZ duality: mirror symmetry as bar-cobar equivalence}
\label{subsec:sqed-xyz-mirror}

Three-dimensional mirror symmetry exchanges gauge-theoretic and
Landau--Ginzburg descriptions of the same physical system.  In the
holomorphic--topological twist, this exchange is visible at the level
of boundary vertex algebras and is controlled by bar-cobar duality.

\paragraph{The SQED side.}
SQED is the $\mathrm{U}(1)$ gauge theory coupled to a single charged
chiral multiplet, placed on a half-space with Neumann boundary
conditions for the gauge field.

\begin{proposition}[SQED boundary VOA; \ClaimStatusProvedElsewhere]
\label{prop:sqed-boundary-voa}
\textup{(Costello--Dimofte--Gaiotto \cite{CDG2023}, \S6.4.)}
The boundary vertex algebra of SQED with Neumann conditions is the
gauge-invariant subalgebra
\[
  \cA_{\partial}^{\mathrm{SQED}}
  \;=\;
  \bigl(\beta\gamma \otimes \widehat{\mathfrak{gl}}_1\bigr)^{\mathrm{U}(1)},
\]
generated by composite operators invariant under the $\mathrm{U}(1)$
gauge action on the charged matter fields.
\end{proposition}

\begin{example}[Explicit generators of the SQED boundary VOA]
\label{ex:sqed-generators}
\index{SQED!boundary generators}
Let $(\beta,\gamma)$ carry $\mathrm{U}(1)$ charges $(+1,-1)$ and let
$J$ be the current of $\widehat{\mathfrak{gl}}_1$.  The
gauge-invariant generators of
$\cA_\partial^{\mathrm{SQED}}$ are:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item The current $J(z)$ itself (neutral under the gauge action);
\item The composite $:\!\beta\gamma\!:(z)$ (the ``meson'');
\item The normal-ordered products $:\!\beta\partial^n\gamma\!:$ and
  their descendants (higher mesons).
\end{enumerate}
The graded character of this gauge-invariant subalgebra is
\begin{equation}\label{eq:sqed-character}
\chi_{\mathrm{SQED}}(q)
\;=\;
\oint \frac{da}{2\pi i\,a}\;
\prod_{n \geq 1}
\frac{1}{(1 - q^n\,a)(1 - q^n\,a^{-1})(1 - q^n)},
\end{equation}
where the $a$-integral projects to $\mathrm{U}(1)$-invariants, the
factors $(1-q^n a)^{-1}$ and $(1-q^n a^{-1})^{-1}$ count $\beta$ and
$\gamma$ modes, and $(1-q^n)^{-1}$ counts $J$-modes.
\end{example}

\paragraph{The XYZ side.}
The XYZ model consists of three chiral multiplets $X$, $Y$, $Z$ with
cubic superpotential $W = XYZ$, again on a half-space with Neumann
boundary conditions.

\begin{proposition}[XYZ boundary VOA; \ClaimStatusProvedElsewhere]
\label{prop:xyz-boundary-voa}
\textup{(Costello--Dimofte--Gaiotto \cite{CDG2023}, \S7.6.)}
The boundary vertex algebra of the XYZ model is the derived critical
locus of $W|_{\partial}$:
\[
  \cA_{\partial}^{\mathrm{XYZ}}
  \;=\;
  H^\bullet\bigl(\beta\gamma^{\otimes 3},\; Q_W\bigr),
  \qquad
  Q_W = \oint\!\bigl(Y Z\, \partial_X + X Z\, \partial_Y
  + X Y\, \partial_Z\bigr),
\]
the algebra of operators satisfying the constraint equations
$YZ = 0$, $XZ = 0$, $XY = 0$ up to homotopy.
\end{proposition}

\begin{theorem}[SQED--XYZ mirror symmetry; \ClaimStatusProvedElsewhere]
\label{thm:sqed-xyz-mirror}
\textup{(Costello--Dimofte--Gaiotto \cite{CDG2023}, \S6--7.)}
The SQED and XYZ boundary vertex algebras are isomorphic:
\[
  \cA_{\partial}^{\mathrm{SQED}}
  \;\cong\;
  \cA_{\partial}^{\mathrm{XYZ}}
\]
as vertex algebras.  This is the holomorphic-twist avatar of $3$d
mirror symmetry.
\end{theorem}

\begin{remark}[Mirror symmetry as Koszul duality]
\label{rem:mirror-as-koszul}
This mirror symmetry is a bar-cobar equivalence.  Both theories
have equivalent bar complexes: their dg coalgebras
$\bar{B}(\cA_{\partial}^{\mathrm{SQED}})$ and
$\bar{B}(\cA_{\partial}^{\mathrm{XYZ}})$ are quasi-isomorphic. So
Theorem~B (bar-cobar inversion) of Volume~I produces the VOA
isomorphism.  Mirror symmetry is not an accident: it is Koszul
duality.
\end{remark}

\begin{remark}[Exchange of boundary conditions]
\label{rem:mirror-bc-exchange}
Under the SQED--XYZ duality, boundary conditions exchange:
Neumann~$\leftrightarrow$~Dirichlet.  In the bar-cobar framework
this is the exchange $\cA \leftrightarrow \cA^!$; Koszul duality
exchanges the two sides of the bar-cobar adjunction
(Theorem~A of Volume~I).  The Neumann algebra of one theory is the
Dirichlet algebra of the mirror, and the exchange is encoded on
the bar side by Verdier duality
$\mathbb{D}_{\Ran}\barB(\cA)\simeq \cA^!_\infty$; the
bar-cobar counit $\Omega \circ B \simeq \id$ only reconstructs
each algebra from its own bar coalgebra.
\end{remark}

\begin{remark}[Mirror symmetry as MC equivalence]
\label{rem:mirror-bar-cobar-explicit}
\index{mirror symmetry!MC equivalence}
\index{Chriss--Ginzburg structure principle!mirror symmetry}
In the Chriss--Ginzburg convolution framework of Volume~I,
mirror symmetry is \emph{MC gauge equivalence} in a strict dg~Lie
chart \textup{(}with the invariant deformation object understood up
to filtered $L_\infty$ quasi-isomorphism, Convention~\ref{conv:vol2-strict-models}\textup{)}:
the master MC
elements $\Theta_{\mathrm{SQED}}$ and $\Theta_{\mathrm{XYZ}}$
in their respective ambient algebras are gauge-equivalent via
\[
  \Theta_{\mathrm{SQED}}
  \;\sim_{\mathrm{MC}}\;
  \Theta_{\mathrm{XYZ}}
  \quad\text{in}\quad
  \operatorname{MC}\!\bigl(\mathfrak{g}^{\mathrm{amb}}\bigr).
\]
The five main theorems of Vol~I appear as projections of this
single equivalence:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item \emph{Theorem~A} ($\Theta\big|_{\hbar=0} = \tau$): the
  SQED bar differential is the gauge BRST differential; the XYZ
  bar differential is the Koszul differential $Q_W$.
\item \emph{Theorem~B} (GK involution): bar-cobar inversion
  recovers both algebras from the shared coalgebra.
\item \emph{Theorem~C} (cross-polarization):
  Neumann~$\leftrightarrow$~Dirichlet is the exchange
  $\cA \leftrightarrow \cA^!$ of the cross-polarized graph sum.
\item \emph{Non-renormalization}: both theories are quasi-linear,
  so $\Theta\big|_{\hbar=0}$ is exact.
\item \emph{Module exchange}: vortex lines of SQED (modules for
  $(\cA^{\mathrm{SQED}})^!$) map to matrix factorizations of XYZ
  (modules for $(\cA^{\mathrm{XYZ}})^!$).
\end{enumerate}
Mirror symmetry is not a coincidence: it is the statement that two
different physical presentations give gauge-equivalent MC elements.
\end{remark}

\subsection{Twisted holography: the Costello--Gaiotto primitive package}
\label{subsec:twisted-holography-primitive}
\index{twisted holography!primitive package|textbf}
\index{Costello--Gaiotto|textbf}
\index{Koszul duality!twisted holography}

The Costello--Gaiotto programme~\cite{costello-gaiotto,Gai19}
places twisted holography inside the bar-cobar framework:
the holographic dictionary is Koszul duality,
the boundary algebra determines the Koszul dual via bar-cobar
and the bulk via the derived center,
and the line-operator category is the module category of the Koszul dual.
This subsection constructs the full primitive package
(Definition~\ref*{V1-def:thqg-completed-platonic-datum}) for the
standard example: 3d holomorphic Chern--Simons on the half-space
$\R_{\ge 0} \times \C$ with gauge algebra $\fg = \mathfrak{gl}_N$.

\subsubsection{The 3d holomorphic Chern--Simons theory}

Fix a reductive Lie algebra $\fg$.
The bulk theory is holomorphic Chern--Simons on $\R_t \times \C_z$:

\begin{equation}\label{eq:tw-hol-hcs-action}
S_{\mathrm{HCS}}
\;=\;
\int_{\R \times \C}
dz \wedge \operatorname{Tr}\!\Bigl(
A \wedge (\dbar + d_t) A
\;+\; \tfrac{2}{3}\, A^3
\Bigr),
\end{equation}
where $A \in \Omega^{0,\bullet}(\C) \otimes \Omega^\bullet(\R) \otimes \fg$
carries both Dolbeault and de~Rham degrees.  The BRST operator is
$Q = \dbar + d_t$, and the propagator factors as
$G(z,t) = K^{\C}(z) \otimes K^{\R}(t)$
with $K^{\C}(z) = (2\pi i)^{-1}\, d\bar z / \bar z$ (the Cauchy kernel)
and $K^{\R}(t) = \theta(t)$ (Heaviside).

\subsubsection{Boundary algebra: affine Kac--Moody}

Place the boundary at $t = 0$.

\begin{proposition}[Boundary algebra of HCS; \ClaimStatusProvedElsewhere]
\label{prop:tw-hol-boundary-KM}
\textup{(Costello--Gaiotto~\cite{costello-gaiotto}, \S3.)}
The boundary algebra of 3d holomorphic Chern--Simons with
$\fg$-gauge symmetry is the affine Kac--Moody vertex algebra at
level $k$ determined by the Chern--Simons coupling:
\[
\cA_\partial \;=\; V_k(\fg),
\qquad
J^a(z)\, J^b(w) \;\sim\;
\frac{k\, \delta^{ab}}{(z-w)^2}
\;+\;
\frac{f^{ab}_{\phantom{ab}c}\, J^c(w)}{z-w}.
\]
\end{proposition}

The boundary OPE has maximal pole order $2$.  By AP19
(the bar kernel absorbs one pole), the collision residue is:
\begin{equation}\label{eq:tw-hol-r-matrix}
r(z)
\;=\;
\Res^{\mathrm{coll}}_{0,2}(\Theta_\cA)
\;=\;
\frac{\Omega_\fg}{z},
\end{equation}
where $\Omega_\fg = \sum_a T^a \otimes T^a$ is the
quadratic Casimir.  This is a single simple pole; the double pole
of the OPE has been absorbed by the $d\log(z-w)$ bar kernel.

\subsubsection{The primitive package}

\begin{theorem}[Primitive package for twisted holography; \ClaimStatusProvedHere]
\label{thm:tw-hol-primitive-package}
Let $\fg$ be a simple Lie algebra and $k \ne -h^\vee$.  The
twisted holographic system $(\fg, k)$ on $\R_{\ge 0} \times \C$
determines a primitive package:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item \textbf{Open-sector category.}
  $\cC_{\mathrm{op}} = V_k(\fg)\text{-}\mathrm{mod}^{\mathrm{dg}}$,
  the dg-category of modules for the affine Kac--Moody algebra at
  level~$k$.

\item \textbf{Boundary object and algebra.}
  $b = V_k(\fg)$ \textup{(}the vacuum module\textup{)},
  $\cA_\partial = \End_{\cC_{\mathrm{op}}}(b) = V_k(\fg)$.

\item \textbf{Koszul dual.}
  $\cA_\partial^! = V_{-k-2h^\vee}(\fg)$, the affine Kac--Moody
  algebra at the Feigin--Frenkel dual level $k' = -k-2h^\vee$
  \textup{(}Theorem~\ref*{V1-thm:universal-kac-moody-koszul}\textup{)}.
  On the associated graded \textup{(}PBW filtration\textup{)},
  $\operatorname{gr}\cA_\partial^! \simeq C^\bullet(\fg[t^{-1}]t^{-1})$.
  The Koszul involution exchanges Neumann~$\leftrightarrow$~Dirichlet
  boundary conditions.

\item \textbf{Derived center \textup{(}universal bulk\textup{)}.}
  $\Zder^{\mathrm{ch}}(\cA_\partial)
  = C^\bullet_{\mathrm{ch}}(V_k(\fg),\, V_k(\fg))$.
  The bulk observables are chiral Hochschild cochains of the boundary
  algebra, not the bar complex.

\item \textbf{Collision residue.}
  $r(z) = \Omega_\fg / z$, satisfying the classical Yang--Baxter
  equation.  The RTT quantization produces the Yangian $Y_\hbar(\fg)$.

\item \textbf{Modular MC element.}
  $\Theta_\cA = D_\cA - d_0 \in \MC(\gAmod)$, with modular
  characteristic $\kappa = \kappa(V_k(\fg))
  = \dim(\fg)\cdot(k + h^\vee)/(2h^\vee)$ and complementarity
  $\kappa(\cA_\partial) + \kappa(\cA_\partial^!) = 0$
  \textup{(}since $\cA^! = V_{-k-2h^\vee}(\fg)$ has
  $\kappa(\cA^!) = -\dim(\fg)\cdot(k+h^\vee)/(2h^\vee)$, by the
  Feigin--Frenkel involution $k \mapsto -k-2h^\vee$\textup{)}.

\item \textbf{Annulus trace.}
  $\operatorname{Tr}_\cA \simeq HH_*(V_k(\fg))$: the
  Hochschild homology of the boundary algebra, computing the
  first modular shadow of the open sector.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{(i)--(ii).}
The open-sector category is the category of boundary
conditions for HCS; for a gauge theory with Neumann
conditions, this is the module category of the boundary
algebra~\cite{costello-gaiotto}.  The vacuum module generates
the category, giving the boundary chart.

\emph{(iii).}
The affine Kac--Moody algebra $V_k(\fg)$ is chirally Koszul
(Volume~I, Proposition~\ref*{V1-prop:pbw-universality}): bar cohomology
concentrates in bar degree~$1$.  The Verdier dual
$\mathbb{D}_{\mathrm{Ran}}(\barB(V_k(\fg))) \simeq
\barB(V_{-k-2h^\vee}(\fg))$ recovers the Koszul dual at the
Feigin--Frenkel dual level
(Theorem~\ref*{V1-thm:universal-kac-moody-koszul}).  On the
associated graded, this is the chiral
$\mathrm{Com}^! = \mathrm{Lie}$ identification.

\emph{(iv).}
The bulk-boundary correspondence
(Theorem~\ref{thm:boundary-linear-bulk-boundary})
applies because $V_k(\fg)$ is boundary-linear.

\emph{(v).}
The $\lambda$-bracket
$\{J^a_\lambda J^b\} = f^{ab}_c J^c + k\, \delta^{ab}\, \lambda$
has Laplace transform
$r^{ab}(z) = \int_0^\infty e^{-\lambda z}\,
(f^{ab}_c J^c + k\, \delta^{ab}\, \lambda)\, d\lambda
= f^{ab}_c J^c / z + k\, \delta^{ab} / z^2$.
The collision residue, after bar-kernel absorption (AP19),
extracts pole orders one less: $r_{\mathrm{coll}}(z)
= \Omega_\fg / z$.  The CYBE follows from the MC equation at
genus~$0$, arity~$3$
(Volume~I, Theorem~\ref*{V1-thm:thqg-g5-yangian}).

\emph{(vi).}
The modular characteristic
$\kappa(V_k(\fg)) = \dim(\fg)(k + h^\vee)/(2h^\vee)$
is proved in Volume~I
(Theorem~\ref*{V1-thm:genus-universality},
Proposition~\ref{prop:dk0-laplace-five-families}).
The complementarity $\kappa + \kappa^! = 0$ follows from
the Feigin--Frenkel involution.

\emph{(vii).}
The annulus trace is the standard Hochschild homology of
$V_k(\fg)$; its identification with the first modular shadow
follows from the annulus trace theorem
(Volume~I, Theorem~\ref*{V1-thm:thqg-annulus-trace}).
\end{proof}

\subsubsection{Primitive package for $K$ M2-branes}

Fix a positive integer $K$ and a non-critical level $k$.
The 5d holomorphic Chern--Simons theory on
$\R_t \times \C^2_{z_1,z_2}$ with gauge algebra
$\fg = \mathfrak{gl}_K$ produces a boundary algebra at $t = 0$.

\begin{theorem}[Primitive package for the M2-brane; \ClaimStatusProvedHere]
\label{thm:twisted-M2-primitive-package}
The twisted M2-brane system with $K$ branes determines a
primitive package:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item \textbf{Open-sector category.}
  $\cC_{\mathrm{op}} = \mathrm{DDCA}_{k}(\mathfrak{gl}_K)\text{-}\mathrm{mod}^{\mathrm{dg}}$.

\item \textbf{Boundary algebra.}
  $\cA_\partial = \mathrm{DDCA}_k(\mathfrak{gl}_K)
  = U(\mathfrak{gl}_K \otimes \C[u,v])$,
  the deformed double current algebra.  Its generators
  $J^a_{m,n}$ with $a \in \mathfrak{gl}_K$, $m,n \ge 0$
  satisfy the binary product of Theorem~\ref{thm:DDCA-m2}:
  \[
  m_2(J^a_{m,n},\; J^b_{p,q})
  \;=\;
  f^{ab}_c\, J^c_{m+p,n+q}
  \;+\;
  k\,\delta^{ab}\,
  P_2(m,n,p,q;\lambda_1,\lambda_2),
  \]
  where $P_2$ is the polynomial central extension
  \eqref{eq:DDCA-central-term}.

\item \textbf{Koszul dual.}
  $\cA_\partial^!
  \simeq C^\bullet(\mathfrak{gl}_K \otimes \C[u,v])$,

  Lie algebra \textup{(}Section~\ref{subsec:M2-holography}\textup{)}.

  $K$-brane sector: Koszul duality exchanges
  boundary~$\leftrightarrow$~bulk.

\item \textbf{Derived center.}
  $\Zder^{\mathrm{ch}}(\cA_\partial)
  = C^\bullet_{\mathrm{ch}}(\mathrm{DDCA}_k,\, \mathrm{DDCA}_k)$:
  the bulk observables of the 5d theory, computed as chiral
  Hochschild cochains of the boundary algebra
  \textup{(}Theorem~\ref{thm:DDCA-swiss-cheese}(iii)\textup{)}.

\item \textbf{Collision residue.}
  The two-variable $r$-matrix
  (Remark~\ref{rem:DDCA-DK0}) is
  \begin{equation}\label{eq:M2-r-matrix}
  r(u_1-u_2,\, v_1-v_2)
  \;=\;
  \frac{\Omega_{\mathfrak{gl}_K}}{(u_1-u_2)(v_1-v_2)},
  \end{equation}
  where $\Omega_{\mathfrak{gl}_K}$ is the Casimir element.
  This is the classical limit of the RTT $R$-matrix
  generating the shifted Yangian
  $Y_\hbar(\mathfrak{gl}_K)$~\cite{DNP2025,Guay2007}.

\item \textbf{Modular MC element.}
  $\Theta_\cA \in \MC(\gAmod)$ with
  $\kappa(\mathrm{DDCA}_k(\mathfrak{gl}_K)) = K(k + K)/2$
  \textup{(}since $\dim(\mathfrak{gl}_K) = K^2$ and
  $h^\vee(\mathfrak{gl}_K) = K$, the formula
  $\kappa = \dim(\fg)(k+h^\vee)/(2h^\vee)$ gives
  $K^2(k+K)/(2K)$\textup{)}.  Complementarity:
  $\kappa + \kappa^! = 0$ \textup{(}class~$\mathbf{L}$\textup{)}.

\item \textbf{Annulus trace.}
  $\operatorname{Tr}_\cA \simeq HH_*(\mathrm{DDCA}_k(\mathfrak{gl}_K))$.
\end{enumerate}
\end{theorem}

\begin{proof}
Items (i)--(iv) and (vii) are established in
Section~\ref{subsec:DDCA-ainfty} and
Theorem~\ref{thm:DDCA-swiss-cheese}.
Item (v) follows from the DK-0 Laplace verification
(Remark~\ref{rem:DDCA-DK0}): the two-variable $\lambda$-bracket
\eqref{eq:DDCA-bracket} has Laplace transform matching the
collision residue, with bar-kernel absorption in both spectral
variables (AP19 applied twice).
Item (vi): the modular characteristic generalizes the affine
Kac--Moody value; the proof of
$\kappa(\mathrm{DDCA}_k(\mathfrak{gl}_K)) = K(k + K)/2$ is the
same computation as for $V_k(\mathfrak{gl}_K)$ because the
polynomial extension $\C[u,v]$ does not contribute to the
genus-$1$ anomaly (it is commutative and carries no independent
central extension at the conformal level).
\end{proof}

\subsubsection{The M2--M5 coupling as twisting morphism}

The M2-brane algebra $\cA_{\mathrm{M2}} = \mathrm{DDCA}_k(\mathfrak{gl}_K)$
couples to the M5-brane algebra
$\cA_{\mathrm{M5}} = \mathcal{W}_\infty(K)$
(Section~\ref{subsec:twisted-M5-primitive}).
The coupling is controlled by a twisting morphism.

\begin{proposition}[M2--M5 coupling; \ClaimStatusConjectured]
\label{prop:M2-M5-coupling}
\index{M2-M5 coupling!twisting morphism}
There exists a twisting morphism
\[
\tau_{\mathrm{M2\text{-}M5}}
\;\in\;
\operatorname{Tw}\!\bigl(
\barB(\cA_{\mathrm{M2}}),\;
\cA_{\mathrm{M5}}
\bigr)
\]
encoding the coupling of $K$ M2-branes to M5-branes.
At tree level \textup{(}genus~$0$\textup{)}, this twisting
morphism restricts to the Miura transformation
$\mathfrak{gl}_K \otimes \C[u,v] \to \mathcal{W}_\infty(K)$
sending double currents to $\mathcal{W}$-algebra generators.
At genus~$g \ge 1$, the twisting morphism acquires curvature
$\kappa_{\mathrm{M2}} \cdot \omega_g$ from the Hodge bundle.
\end{proposition}

\begin{remark}[Status of the M2--M5 coupling]
\label{rem:M2-M5-status}
The tree-level identification (genus~$0$) is proved: the Miura
map from the Heisenberg subalgebra of $\mathrm{DDCA}_k$ to
$\mathcal{W}_\infty(K)$ is a morphism of vertex algebras, and
its compatibility with the bar differential follows from
$m_3 = 0$ on cohomology
(Theorem~\ref{thm:DDCA-m3}).  The higher-genus statement is
conjectural: it requires the twisting morphism to extend to the
curved (genus~$\ge 1$) bar complex, which depends on the
HS-sewing criterion for the DDCA.  The HS-sewing criterion
holds for $\mathrm{DDCA}_k(\mathfrak{gl}_K)$
(Volume~I, Theorem~\ref*{V1-thm:general-hs-sewing}: polynomial OPE growth
and subexponential sector growth), so the obstruction is not
convergence but rather the explicit identification of the
curved twisting morphism with the physical M2--M5 coupling at
higher genus.
\end{remark}

\begin{remark}[Large-$K$ holography]
\label{rem:M2-large-K}
At large $K$, the holographic dictionary reads
$\cA_{\partial,\infty}^! \cong \cA_{\mathrm{bulk}}$
sector by sector (Section~\ref{subsec:M2-holography}).
The primitive package of
Theorem~\ref{thm:twisted-M2-primitive-package} is the
finite-$K$ refinement of this large-$K$ limit.  The shadow
depth classification (class~$\mathbf{L}$,
Proposition~\ref{prop:DDCA-shadow}) is independent of $K$:
the double current algebra has tree-level shadow depth for
all $K$, because the Jacobi identity kills higher operations
at all ranks.
\end{remark}

\subsection{Twisted M5-brane: the matrix \texorpdfstring{$\mathcal{W}_\infty$}{W-infinity} algebra}
\label{subsec:twisted-M5-primitive}
\index{M5-brane!primitive package|textbf}
\index{W-infinity algebra!primitive package|textbf}
\index{matrix W-algebra|textbf}

The M5-brane boundary algebra is the matrix $\mathcal{W}_\infty$
algebra $\mathcal{W}_\infty(K)$, the large-$N$ limit of the
principal $\mathcal{W}$-algebras $\mathcal{W}_N(\mathfrak{gl}_K)$
obtained by quantum Drinfeld--Sokolov reduction.  This is the
dual face of the M2--M5 system: where the M2-brane sees a
current algebra (class~$\mathbf{L}$), the M5-brane sees a
$\mathcal{W}$-algebra (class~$\mathbf{M}$).

\subsubsection{The algebra \texorpdfstring{$\mathcal{W}_\infty(K)$}{W-infinity(K)}}

\begin{definition}[$\mathcal{W}_\infty(K)$; following~\cite{ProchazkRapcak2018}]
\label{def:W-infinity-K}
For each positive integer $K$, the \emph{matrix $\mathcal{W}_\infty$ algebra}
is the inverse limit
\[
\mathcal{W}_\infty(K)
\;:=\;
\varprojlim_{N \to \infty}\;
\mathcal{W}_N(\mathfrak{gl}_K),
\]
where $\mathcal{W}_N(\mathfrak{gl}_K) = H^0_{\mathrm{DS}}(V_k(\mathfrak{gl}_{NK}),
f_{\mathrm{prin},N})$ is the principal $\mathcal{W}$-algebra obtained
by quantum Drinfeld--Sokolov reduction of $V_k(\mathfrak{gl}_{NK})$
with respect to the principal nilpotent
$f_{\mathrm{prin},N} \in \mathfrak{sl}_N \subset \mathfrak{gl}_{NK}$.
\end{definition}

\paragraph{Generators.}
For finite $N$, $\mathcal{W}_N(\mathfrak{gl}_K)$ is freely strongly
generated by $K^2 \cdot N$ fields
$\{W^{(s)}_{a\bar b} : s = 1, \ldots, N,\; a,\bar b = 1, \ldots, K\}$
of conformal weights $s = 1, 2, \ldots, N$.  In the $N \to \infty$ limit,
the algebra acquires generators $W^{(s)}_{a\bar b}$ of all spins
$s \ge 1$, with $a, \bar b$ running over $\mathfrak{gl}_K$ matrix indices.
At $K = 1$, this reduces to the scalar $\mathcal{W}_{1+\infty}$ algebra.

\paragraph{The matrix Miura transformation.}
The embedding of $\mathcal{W}_\infty(K)$ into a product of $K$ copies of the
Heisenberg algebra (the ``free-field realization'') is controlled by the
matrix Miura operator:
\begin{equation}\label{eq:matrix-miura}
L(z)
\;=\;
(\partial + J_1(z))\,(\partial + J_2(z))\, \cdots\, (\partial + J_K(z)),
\end{equation}
where each $J_i(z)$ is a Heisenberg field of level $k_i$.  The
$\mathcal{W}$-algebra generators are extracted as the coefficients of
$L(z) = \partial^K + W^{(2)}(z)\, \partial^{K-2} + \cdots + W^{(K)}(z)$.
The Miura transformation is an injective vertex algebra homomorphism
$\mathcal{W}_\infty(K) \hookrightarrow
\mathcal{H}_{k_1} \otimes \cdots \otimes \mathcal{H}_{k_K}$
whose image consists of the operators invariant under permutations of the
Heisenberg factors (for generic levels).

\subsubsection{Primitive package for $K$ M5-branes}

\begin{theorem}[Primitive package for the M5-brane; \ClaimStatusConjectured]
\label{thm:twisted-M5-primitive-package}
The twisted M5-brane system with $K$ branes determines a
primitive package:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item \textbf{Open-sector category.}
  $\cC_{\mathrm{op}} = \mathcal{W}_\infty(K)\text{-}\mathrm{mod}^{\mathrm{dg}}$,
  the dg-category of modules for the matrix
  $\mathcal{W}_\infty$ algebra.

\item \textbf{Boundary algebra.}
  $\cA_\partial = \mathcal{W}_\infty(K)$,
  with generators $W^{(s)}_{a\bar b}$ for $s \ge 1$,
  $a, \bar b \in \{1, \ldots, K\}$.

\item \textbf{Koszul dual.}
  For finite $N$:
  $\mathcal{W}_N(\mathfrak{gl}_K)^!
  = \mathcal{W}_{N'}(\mathfrak{gl}_K)$
  with dual level determined by the Feigin--Frenkel involution
  applied through DS reduction.  At $K = 1$:
  $\mathcal{W}_N^! = \mathcal{W}_N$ at the dual level
  $k' = -N - k$ \textup{(}self-Koszul-duality up to level shift;
  the complementarity sum $\kappa + \kappa'$ is nonzero
  in general: for $N = 2$, $\kappa + \kappa' = 13$;
  for $N = 3$, $\kappa + \kappa' = 250/3$\textup{)}.

\item \textbf{Derived center.}
  $\Zder^{\mathrm{ch}}(\cA_\partial)
  = C^\bullet_{\mathrm{ch}}(\mathcal{W}_\infty(K),\;
  \mathcal{W}_\infty(K))$.

\item \textbf{Collision residue.}
  For $K = 1$, the Virasoro subalgebra has OPE with poles up to
  order~$4$.  By AP19, the collision residue has poles up to order~$3$:
  \[
  r_{\mathrm{Vir}}(z)
  \;=\;
  \frac{c/2}{z^3} \;+\; \frac{2T}{z}.
  \]
  For general $K$, the $r$-matrix is a $K^2 \times K^2$ matrix-valued
  meromorphic function with poles of order up to $2s-1$ from the
  spin-$s$ generators.

\item \textbf{Modular MC element.}
  For $K = 1$ and finite $N$: $\kappa(\mathcal{W}_N)
  = c(\mathcal{W}_N) \cdot (H_N - 1)$, where $c(\mathcal{W}_N)$ is the
  Fateev--Lukyanov central charge of the principal
  $\mathcal{W}_N$-algebra at level $k$~\cite{FL88} and
  $H_N = \sum_{m=1}^N 1/m$ is the $N$th harmonic number.
  For $N = 2$ \textup{(}Virasoro\textup{)}: $H_2 - 1 = 1/2$,
  so $\kappa = c/2$.
  For $N = 3$ \textup{(}$\mathcal{W}_3$\textup{)}: $H_3 - 1 = 5/6$,
  so $\kappa = 5c/6$.

\item \textbf{Annulus trace.}
  $\operatorname{Tr}_\cA \simeq HH_*(\mathcal{W}_\infty(K))$.
\end{enumerate}
\end{theorem}

\begin{remark}[Status of the M5-brane package]
\label{rem:M5-status}
Items (i), (ii), (iv), and (vii) are structural
consequences of the general framework.  The substantive content is:

\emph{Item (iii):} The Koszul dual of $\mathcal{W}_N(\fg, f_{\mathrm{prin}})$
at the principal nilpotent is controlled by the DS reduction of the
affine Koszul dual.  For $K = 1$ (scalar $\mathcal{W}_N$ algebras),
the principal $\mathcal{W}$-algebra inherits Koszul self-duality
$\mathcal{W}_N^! \cong \mathcal{W}_N$ at the dual level from the
affine lineage.  However, $\kappa + \kappa^! \ne 0$ in general for
$\mathcal{W}$-algebras: the DS reduction changes the anomaly ratio
$\varrho$ from $1/2$ (Kac--Moody) to $H_N - 1$
(Volume~I, landscape census), so the complementarity sum is
$\kappa + \kappa' = \varrho \cdot K_N$ where $K_N = 4N^3 - 2N - 2$.
For $K > 1$ (matrix $\mathcal{W}$-algebras),
the Koszul dual identification is conjectural and depends on the
non-principal DS reduction programme
(Section~\ref{subsec:KS-defect}).

\emph{Item (v):} The $r$-matrix computation requires the
full OPE of $\mathcal{W}_\infty(K)$.  At $K = 1$, the
Virasoro $r$-matrix is proved
(Proposition~\ref{prop:dk0-laplace-five-families}).
At $K > 1$, the matrix $r$-matrix is determined in
principle by the Miura transformation
\eqref{eq:matrix-miura}, but the explicit formula is not
computed here.

\emph{Item (vi):} The central charge formula is standard for principal
$\mathcal{W}$-algebras~\cite{FL88}.  The modular characteristic
$\kappa = c \cdot (H_N - 1)$ follows from the genus universality theorem
\textup{(}Volume~I, Theorem~\textup{\ref*{V1-thm:genus-universality}}\textup{)}.
For $N = 2$: $\kappa = c/2$; for $N = 3$: $\kappa = 5c/6$.
The $N \to \infty$ limit of the anomaly ratio $H_N - 1$ diverges
\textup{(}harmonic series\textup{)}; the modular characteristic of
$\mathcal{W}_\infty(K)$ requires the limiting central charge and the
full generator spectrum.
\end{remark}

\subsubsection{Shadow archetype and the infinite tower}

\begin{proposition}[Shadow archetype of $\mathcal{W}_\infty(K)$; \ClaimStatusProvedHere]
\label{prop:M5-shadow-archetype}
For $K = 1$, $\mathcal{W}_\infty = \mathcal{W}_{1+\infty}$ is
class~$\mathbf{M}$ \textup{(}mixed, $r_{\max} = \infty$\textup{)}.
For all $K \ge 1$, $\mathcal{W}_\infty(K)$ is class~$\mathbf{M}$:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item The spin-$2$ generator \textup{(}Virasoro\textup{)} creates

\item The Swiss-cheese operations $m_k^{\mathrm{SC}} \ne 0$ for

\item The shadow tower does not terminate: the discriminant

\item The shadow growth rate $\rho(\mathcal{W}_\infty(K))$

\begin{proof}
Items (i)--(iii) follow from the pole-structure
dichotomy
(Theorem~\ref{thm:pole-structure-dichotomy}):
the spin-$2$ OPE has a fourth-order pole,
placing $\mathcal{W}_\infty(K)$ in class~$\mathbf{M}$.
The non-vanishing of $S_4$ for $\mathcal{W}$-algebras
is computed in Volume~I:
$S_4^{\mathrm{Vir}} = 10/[c(5c + 22)]$
for the Virasoro subalgebra.
Item (iv): the shadow radius theorem gives
$\rho = \sqrt{9\alpha^2 + 2\Delta}/(2|\kappa|)$;

and convergence holds for $\rho < 1$, i.e.\

\subsubsection{M2--M5 duality as bar-cobar duality}

\begin{conjecture}[M2--M5 bar-cobar duality; \ClaimStatusConjectured]
\label{conj:M2-M5-duality}
\index{M2-M5 duality!bar-cobar}
At each rank $K$ and non-critical level, the M2-brane algebra
$\cA_{\mathrm{M2}} = \mathrm{DDCA}_k(\mathfrak{gl}_K)$
\textup{(}class~$\mathbf{L}$\textup{)} and the M5-brane algebra
$\cA_{\mathrm{M5}} = \mathcal{W}_\infty(K)$
\textup{(}class~$\mathbf{M}$\textup{)} are related by:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item The M2-brane algebra acts on the M5-brane algebra via the
  Miura transformation, which is a twisting morphism
  $\tau \in \operatorname{Tw}(\barB(\cA_{\mathrm{M2}}), \cA_{\mathrm{M5}})$.

\item The line-operator category of the M2-brane system is
  controlled by the Yangian $Y_\hbar(\mathfrak{gl}_K)$; the
  line-operator category of the M5-brane system is controlled by
  the $\mathcal{W}$-algebra $\mathcal{W}_\infty(K)$ acting on its
  own modules.  The M2--M5 coupling identifies modules in the two
  categories.

\item The modular MC elements satisfy
  $\Theta_{\mathrm{M2}} = \pi_{\mathrm{M2}}(\Theta_{\mathrm{M\text{-}theory}})$
  and
  $\Theta_{\mathrm{M5}} = \pi_{\mathrm{M5}}(\Theta_{\mathrm{M\text{-}theory}})$,
  where $\Theta_{\mathrm{M\text{-}theory}}$ is a single MC element
  in the ambient Swiss-cheese algebra of the 11d twisted theory, and
  $\pi_{\mathrm{M2}}, \pi_{\mathrm{M5}}$ are projections to the two
  boundary sectors.

\item At the scalar shadow level:
  $\kappa(\cA_{\mathrm{M2}}) = K(k + K)/2$,
  $\kappa(\cA_{\mathrm{M5}}) = c(\mathcal{W}_\infty(K)) \cdot (H_N - 1)$
  for the $N$th truncation.
  These are not Koszul duals of each other in general, but they
  are projections of the same M-theoretic MC element to different
  boundary sectors.
\end{enumerate}
\end{conjecture}

\begin{remark}[What is proved vs.\ conjectural]
\label{rem:M2-M5-proved-vs-conjectural}
Item (i) at tree level (genus~$0$) is proved: the Miura
transformation is a vertex algebra homomorphism, and
$m_3 = 0$ on DDCA cohomology
(Theorem~\ref{thm:DDCA-m3}) ensures compatibility with
the bar differential.  Items (ii)--(iv) are conjectural:
they require identifying the physical M2--M5 coupling at
all genera with a specific MC element in the modular
convolution algebra.  The key missing ingredient is the
explicit construction of $\Theta_{\mathrm{M\text{-}theory}}$
in the 11d twisted theory.

The shadow depth asymmetry is a structural prediction:
the M2-brane sees class~$\mathbf{L}$ (finite tower, tree-level
shadows), while the M5-brane sees class~$\mathbf{M}$ (infinite
tower).  The DS reduction $V_k(\mathfrak{gl}_{NK}) \to
\mathcal{W}_N(\mathfrak{gl}_K)$ takes class~$\mathbf{L}$ to
class~$\mathbf{M}$: this is the shadow-depth increase under DS
reduction.  The ``extra complexity'' of the M5-brane is not
a new phenomenon but a consequence of the DS functor
introducing higher Swiss-cheese operations
(Theorem~\ref{thm:ds-koszul-obstruction}).
\end{remark}

\begin{remark}[Comparison of the three packages]
\label{rem:three-package-comparison}
\index{twisted holography!package comparison}
The three twisted holographic primitive packages exhibit a
progression of complexity:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\small
\begin{tabular}{lccccc}
\textbf{System} & $\cA_\partial$ & \textbf{Class} & $r_{\max}$
  & $\kappa$ & $\kappa + \kappa^!$ \\
\hline
Twisted holography & $V_k(\fg)$ & $\mathbf{L}$ & $3$
  & $\frac{d(k + h^\vee)}{2h^\vee}$ & $0$ \\
Twisted M2 & $\mathrm{DDCA}_k(\mathfrak{gl}_K)$ & $\mathbf{L}$ & $3$
  & $\frac{K(k + K)}{2}$ & $0$ \\
Twisted M5 & $\mathcal{W}_\infty(K)$ & $\mathbf{M}$ & $\infty$
  & $c \cdot (H_N {-} 1)$ & $\ne 0$
\end{tabular}
\end{center}
The first two are current algebras with tree-level shadows;
the third is a $\mathcal{W}$-algebra with infinite shadow tower.
The first two satisfy $\kappa + \kappa^! = 0$ because they are
affine Kac--Moody algebras.  For the M5 row,
$\kappa + \kappa^! \ne 0$ in general: the DS reduction from the
affine lineage does not preserve the complementarity anti-symmetry
\textup{(}AP24: $\kappa + \kappa' = 13$ for Virasoro,
$250/3$ for $\mathcal{W}_3$\textup{)}.
The M2--M5 coupling is the intertwiner
between a class~$\mathbf{L}$ and a class~$\mathbf{M}$ system:
in the shadow tower language, it is a morphism from a system with
finitely many nonzero shadows to one with infinitely many.
\end{remark}

\subsection{Benchmark global example: \texorpdfstring{$V_k(\mathfrak{sl}_2)$}{V\_k(sl\_2)}
  on \texorpdfstring{$(\mathbb{P}^1, \{0,\infty\})$}{(P\^1, \{0, infty\})}}
\label{subsec:benchmark-global-sl2}
\index{benchmark global example|textbf}
\index{Kac--Moody!benchmark global example|textbf}
\index{projective line!benchmark example|textbf}

The preceding examples compute local data at a single puncture
on the half-space $\R_{\ge 0} \times \C$.  The present section
is the first fully global computation in this monograph: we take
$X = \PP^1$, the marked divisor $D = \{0, \infty\}$, tangential
basepoints $\tau = (\tau_0, \tau_\infty)$ at each puncture, and
the boundary algebra $\cA = V_k(\mathfrak{sl}_2)$ at level
$k \ne -2$ (i.e.\ away from the critical level $k = -h^\vee$).
Every object in the completed platonic datum
(Definition~\ref*{V1-def:thqg-completed-platonic-datum})
is computed explicitly.

The example is the simplest nontrivial test of the global
architecture: the curve $\PP^1$ has genus~$0$; the algebra
$\mathfrak{sl}_2$ has rank~$1$, dimension~$3$, and dual Coxeter
number $h^\vee = 2$.  Yet the full open/closed package has genuine
content: two boundary circles, a factorization category on
$\PP^1 \setminus \{0,\infty\}$, a nontrivial $r$-matrix, and
at integrable levels a finite-dimensional semisimple Verlinde
algebra as the bulk.

\subsubsection{The tangential log curve}
\label{subsubsec:benchmark-tangential-log-curve}

Let $X = \PP^1$ and $D = \{0, \infty\} \subset \PP^1$.
The real oriented blowup
(Definition~\ref{def:bordified-curve})
\[
  \pi \colon \widetilde{\PP}^1_D
  \;\longrightarrow\;
  \PP^1
\]
replaces each marked point by a circle of real tangent directions:
$\pi^{-1}(0) = S^1_0$ and $\pi^{-1}(\infty) = S^1_\infty$.
The resulting smooth manifold with boundary is
\[
  \widetilde{\PP}^1_D
  \;\cong\;
  S^2 \setminus (D^2_0 \sqcup D^2_\infty)
  \;\cong\;
  [0,1] \times S^1,
\]
a cylinder with boundary $\partial \widetilde{\PP}^1_D
= S^1_0 \sqcup S^1_\infty$.

Fix tangential basepoints $\tau_0 \in S^1_0$ (the positive-real
direction at $z = 0$) and $\tau_\infty \in S^1_\infty$ (the
negative-real direction at $w = 0$ in the chart $w = 1/z$ at
$\infty$).  Each punctured boundary circle gives an open interval:
\begin{equation}\label{eq:benchmark-boundary-intervals}
  I_0 := S^1_0 \setminus \{\tau_0\} \cong \R,
  \qquad
  I_\infty := S^1_\infty \setminus \{\tau_\infty\} \cong \R.
\end{equation}
These are the two time axes.  The data
$(\PP^1, \{0,\infty\}, \tau)$ is a pointed curve with tangential
basepoints.

\begin{remark}[Geometric content of the tangential data]
\label{rem:benchmark-tangential-content}
The tangential basepoint $\tau_p$ at a puncture $p$ breaks the
$S^1$ rotational symmetry, producing a contractible interval
$I_p \cong \R$ on which the $E_1$-structure (associative,
time-ordered products) is defined.  Without $\tau_p$, the boundary
is a circle and the algebraic structure is $S^1$-equivariant
(the trace/clutching direction).  Restoring $\tau_p$, i.e.\
gluing the interval back into a circle, is the clutching
operation that produces modular invariants
(Definition~\ref{def:oc-factorization-category}(vi)).
\end{remark}


\subsubsection{The mixed configuration spaces}
\label{subsubsec:benchmark-mixed-config}

The mixed open/closed configuration space
(Definition~\ref{def:mixed-config}) on $(\PP^1, \{0,\infty\}, \tau)$
is
\begin{equation}\label{eq:benchmark-mixed-config}
  \Conf^{\mathrm{oc}}_{I,(m_0,m_\infty)}(\PP^1, D, \tau)
  \;=\;
  \Conf_I(\PP^1 \setminus D)
  \;\times\;
  \Conf^{<}_{m_0}(\R)
  \;\times\;
  \Conf^{<}_{m_\infty}(\R),
\end{equation}
where $\Conf_I(\PP^1 \setminus D)$ parametrises $|I|$ labelled
bulk points in $\PP^1 \setminus \{0,\infty\} \cong \C^\times$,
and $\Conf^{<}_{m_p}(\R) = \{t_1 < \cdots < t_{m_p}\}$
parametrises $m_p$ time-ordered boundary insertions on $I_p$.

The mixed Ran space is the colimit
\[
  \Ran^{\mathrm{oc}}(\PP^1, D, \tau)
  \;=\;
  \varinjlim_{(I,m_0,m_\infty)}
  \Conf^{\mathrm{oc}}_{I,(m_0,m_\infty)}(\PP^1, D, \tau)
  \;/\; \Sigma_I.
\]
A point of $\Ran^{\mathrm{oc}}$ is a finite configuration
consisting of: (a)~an unordered collection of bulk points in
$\C^\times$, (b)~a finite ordered tuple of boundary insertions
on $I_0$, and (c)~a finite ordered tuple of boundary insertions
on $I_\infty$.


\subsubsection{The open sector: generators and relations for
  \texorpdfstring{$\cC_{\mathrm{op}}$}{C\_op}}
\label{subsubsec:benchmark-open-sector}
\index{open-sector category!generators and relations|textbf}
\index{factorization dg-category!SL(2) Chern--Simons|textbf}

The primitive object of the open/closed programme is not the
boundary algebra $V_k(\mathfrak{sl}_2)$, which is a chart,
but the open-sector factorization dg-category
$\cC_{\mathrm{op}}$ on the tangential log curve
$(\PP^1, \{0,\infty\}, \tau)$.  This subsubsection gives its
formal generators-and-relations presentation, the first complete
such construction in the monograph.

\begin{definition}[Open-sector factorization dg-category for
  $\mathrm{SL}(2)$ Chern--Simons at level $k$]
\label{def:benchmark-Cop-sl2}
\index{open-sector category!SL(2)|textbf}
Let $\fg = \mathfrak{sl}_2$, $h^\vee = 2$, and
$k \in \C \setminus \{-2\}$.  The \emph{open-sector
factorization dg-category}
$\cC_{\mathrm{op}} = \cC_{\mathrm{op}}(V_k(\mathfrak{sl}_2))$
has the following data.

\medskip\noindent
\textbf{Objects.}
The objects of $\cC_{\mathrm{op}}$ are
$V_k(\mathfrak{sl}_2)$-modules, i.e.\ smooth modules for the
vertex algebra $V_k(\mathfrak{sl}_2)$:
\begin{equation}\label{eq:benchmark-Cop-objects}
  \Ob(\cC_{\mathrm{op}})
  \;=\;
  V_k(\mathfrak{sl}_2)\text{-}\mathrm{mod}.
\end{equation}
The generating objects are the Weyl modules
(generalized Verma modules):
\begin{equation}\label{eq:benchmark-Cop-generators}
  \mathbb{V}_{k,\lambda}
  \;:=\;
  \operatorname{Ind}_{\mathfrak{sl}_2[t] \oplus \C\mathbf{1}}^{\widehat{\mathfrak{sl}}_2}
  L(\lambda),
  \qquad \lambda \in \mathfrak{h}^*,
\end{equation}
where $L(\lambda)$ is the finite-dimensional
$\mathfrak{sl}_2$-module of highest weight $\lambda$.
The vacuum module $\mathbb{V}_{k,0} = V_k(\mathfrak{sl}_2)$
is the monoidal unit.

At \emph{integrable level} $k \in \Z_{\ge 0}$, the simple
quotients $L_{k,j}$ for $0 \le j \le k$ form a finite
generating set:
\begin{equation}\label{eq:benchmark-Cop-integrable-objects}
  \Ob(\cC_{\mathrm{op}})^{\mathrm{int}}
  \;=\;
  \{L_{k,0},\; L_{k,1},\; \ldots,\; L_{k,k}\},
\end{equation}
with $L_{k,0} = L_k(\mathfrak{sl}_2)$ the simple vacuum module.

\medskip\noindent
\textbf{Morphisms.}
For $M, N \in \Ob(\cC_{\mathrm{op}})$, the morphism space is the
vertex-algebraic Hom complex:
\begin{equation}\label{eq:benchmark-Cop-morphisms}
  \Hom_{\cC_{\mathrm{op}}}(M, N)
  \;=\;
  \RHom_{V_k(\mathfrak{sl}_2)}(M, N),
\end{equation}
computed in the derived category of $V_k(\mathfrak{sl}_2)$-modules.
On the abelian level, the Hom is the space of intertwining operators:
\begin{equation}\label{eq:benchmark-Cop-hom-abelian}
  \Hom_{V_k}(M, N)
  \;=\;
  \bigl\{
  f \colon M \to N
  \;\big|\;
  f(a_{(n)} m) = a_{(n)} f(m)
  \;\text{for all } a \in V_k,\, n \in \Z,\, m \in M
  \bigr\}.
\end{equation}
In particular, the vacuum endomorphism algebra recovers the
boundary chiral algebra:
\begin{equation}\label{eq:benchmark-Cop-vacuum-end}
  \End_{\cC_{\mathrm{op}}}(V_k(\mathfrak{sl}_2))
  \;=\;
  V_k(\mathfrak{sl}_2).
\end{equation}

\medskip\noindent
\textbf{Composition.}
The composition
$\Hom(M,N) \otimes \Hom(L,M) \to \Hom(L,N)$ is given by
the OPE of intertwining operators on the boundary interval
$I_p$: for $f \in \Hom(M,N)$ inserted at time $t_1$ and
$g \in \Hom(L,M)$ at time $t_2 < t_1$, the composition
$f \circ g$ is the time-ordered product
\begin{equation}\label{eq:benchmark-Cop-composition}
  (f \circ g)(v)
  \;=\;
  \lim_{t_1 > t_2}
  f(t_1)\, g(t_2)\, v,
  \qquad v \in L,
\end{equation}
made precise by the configuration-space regularization on
$\Conf^{<}_2(\R)$.  The $A_\infty$-corrections $\mu_n$ for
$n \ge 3$ arise from the codimension-one boundary strata of
the associahedron $K_n$
(Theorem~\ref*{V1-thm:thqg-local-global-bridge}(i)).

\medskip\noindent
\textbf{Factorization structure.}
For disjoint open disks $D_1, \ldots, D_n \subset
\PP^1 \setminus \{0,\infty\}$ and modules
$M_1, \ldots, M_n \in \Ob(\cC_{\mathrm{op}})$, the
factorization isomorphism is
\begin{equation}\label{eq:benchmark-Cop-factorization}
  \cC_{\mathrm{op}}(D_1 \sqcup \cdots \sqcup D_n)
  \;\xrightarrow{\;\sim\;}
  \cC_{\mathrm{op}}(D_1)
  \boxtimes \cdots \boxtimes
  \cC_{\mathrm{op}}(D_n).
\end{equation}
On each formal disk $D_i$, $\cC_{\mathrm{op}}(D_i)$ is the
category of $V_k(\mathfrak{sl}_2)$-modules supported at the
center of $D_i$.  The factorization isomorphism is the tensor
product of modules via fusion:
for modules $M$ at $z_1$ and $N$ at $z_2$ with
$|z_1| < |z_2|$ (in the standard coordinate), the fusion
product is the module
\begin{equation}\label{eq:benchmark-Cop-fusion}
  M \boxtimes_{z_1, z_2} N
  \;:=\;
  H^0\bigl(\PP^1 \setminus \{z_1, z_2, \infty\},\;
  \cF_M \boxtimes \cF_N\bigr),
\end{equation}
where $\cF_M$ and $\cF_N$ are the corresponding
$\widehat{\mathfrak{sl}}_2$-modules viewed as sections of the
factorization algebra on $\PP^1 \setminus D$.  At integrable
levels, this recovers the Verlinde fusion rules:
$L_{k,i} \boxtimes L_{k,j} \cong \bigoplus_l N_{ij}^{\;\;l}\,
L_{k,l}$.

\medskip\noindent
\textbf{Braiding.}
The analytic continuation of the fusion product around
$z_1 = z_2$ defines a braiding on $\cC_{\mathrm{op}}$.
At the perturbative level, this braiding is given by the
monodromy of the Knizhnik--Zamolodchikov connection
\begin{equation}\label{eq:benchmark-KZ}
  \nabla_{\mathrm{KZ}}
  \;=\;
  d
  \;-\;
  \hbar\, \frac{\Omega_{\mathfrak{sl}_2}}{z_1 - z_2}\,
  d(z_1 - z_2),
  \qquad
  \hbar = \frac{1}{k + h^\vee} = \frac{1}{k + 2},
\end{equation}
acting on $M \otimes N$ for any pair of modules $M, N$.
The KZ parameter $\hbar = 1/(k+2)$ is the reciprocal of the
shifted level.  At integrable $k$, the monodromy of
$\nabla_{\mathrm{KZ}}$ gives the braiding matrices of the
modular tensor category
$\mathrm{Rep}(U_q(\mathfrak{sl}_2))$ with
$q = e^{i\pi\hbar} = e^{i\pi/(k+2)}$
(Kazhdan--Lusztig~\cite{KL93},
Drinfeld~\cite{Dri89}).

\medskip\noindent
\textbf{Generating relations.}
The category $\cC_{\mathrm{op}}$ is generated under colimits
and the factorization product by the vacuum module
$V_k(\mathfrak{sl}_2)$.  The defining relations are:
\begin{enumerate}[label=\textup{(R\arabic*)},nosep]
\item \emph{OPE relation.}
  The $\lambda$-bracket on the vacuum endomorphism algebra is
  \begin{equation}\label{eq:benchmark-sl2-ope}
    \{J^a_\lambda J^b\}
    \;=\;
    f^{ab}_{\phantom{ab}c}\, J^c
    \;+\;
    k\, (J^a, J^b)\, \lambda,
  \end{equation}
  where $\{J^a\} = \{e, f, h\}$ are the Chevalley generators,
  $f^{ab}_c$ the structure constants of $\mathfrak{sl}_2$, and
  $(\cdot,\cdot) = \mathrm{tr}_{\mathrm{fund}}$ the normalized
  Killing form.  This determines the vertex algebra
  $V_k(\mathfrak{sl}_2)$ and hence its module category.

\item \emph{Level constraint.}
  The central element $\mathbf{1} \in
  \widehat{\mathfrak{sl}}_2$ acts as
  $\mathbf{1} \cdot m = k \cdot m$ for all $m \in M$ and
  all $M \in \Ob(\cC_{\mathrm{op}})$.  This fixes the level.

\item \emph{Smoothness.}
  For every $m \in M$ and every
  $a \in \mathfrak{sl}_2 \subset \widehat{\mathfrak{sl}}_2$,
  there exists $N \ge 0$ such that
  $a_n \cdot m = 0$ for all $n \ge N$.  This is the smooth-module
  condition, ensuring that the action extends to the vertex
  algebra (not merely the Lie algebra).

\item \emph{Vacuum axiom.}
  The vacuum vector $|0\rangle \in V_k(\mathfrak{sl}_2)$
  satisfies $J^a_n |0\rangle = 0$ for $n \ge 0$, and
  $\mathbf{1} |0\rangle = k |0\rangle$.

\item \emph{Factorization.}
  The fusion product~\eqref{eq:benchmark-Cop-fusion} is
  associative up to coherent homotopy (the factorization
  isomorphisms satisfy the Ran-space sewing identities of
  Definition~\ref{def:oc-factorization-category}).
\end{enumerate}
Relations (R1)--(R4) determine the abelian category
$V_k(\mathfrak{sl}_2)\text{-mod}$.  Relation (R5) upgrades
it to a factorization category on
$(\PP^1, \{0,\infty\}, \tau)$.
\end{definition}

\begin{proposition}[Boundary algebra recovery;
  \ClaimStatusProvedElsewhere]
\label{prop:benchmark-boundary-algebra}
\textup{(Costello--Gaiotto~\cite{costello-gaiotto}.)}
Let\/ $\cC_{\mathrm{op}} = \cC_{\mathrm{op}}(V_k(\mathfrak{sl}_2))$
be the open-sector factorization dg-category of
Definition~\textup{\ref{def:benchmark-Cop-sl2}}, and let\/
$b = V_k(\mathfrak{sl}_2)$ be the vacuum module.  Then:
\begin{enumerate}[label=\textup{(\roman*)}]
\item The boundary algebra at each puncture
  $p \in \{0, \infty\}$ is
  \begin{equation}\label{eq:benchmark-boundary-recovery}
    \cA_{\partial, p}
    \;=\;
    \End_{\cC_{\mathrm{op}}(I_p)}(b)
    \;=\;
    V_k(\mathfrak{sl}_2).
  \end{equation}
\item The vacuum module $b$ is a compact generator of\/
  $\cC_{\mathrm{op}}$: every object is a retract of a
  filtered colimit of finite direct sums of shifts of $b$.
\item The full subcategory generated by $b$ determines the
  Morita equivalence class of\/ $\cC_{\mathrm{op}}$, and
  the derived center
  $\cZ^{\mathrm{der}}_{\mathrm{ch}}(\cC_{\mathrm{op}})$
  depends only on this class
  \textup{(}Theorem~\textup{\ref*{V1-thm:thqg-local-global-bridge}(iii)}\textup{)}.
\end{enumerate}
\end{proposition}

\begin{proof}
(i) The endomorphism algebra $\End(b)$ of the vacuum module in
$V_k(\mathfrak{sl}_2)\text{-mod}$ is $V_k(\mathfrak{sl}_2)$ itself:
every endomorphism of the vacuum module is determined by the
image of $|0\rangle$, which by the vacuum axiom (R4) and
smoothness (R3) lies in $V_k(\mathfrak{sl}_2)$.

(ii) Compact generation follows from the structure theory of
smooth $\widehat{\mathfrak{sl}}_2$-modules at non-critical level:
every smooth module admits a resolution by Weyl modules
$\mathbb{V}_{k,\lambda}$~\eqref{eq:benchmark-Cop-generators},
each of which is a quotient of the vacuum module tensored with
a finite-dimensional $\mathfrak{sl}_2$-representation.
At integrable level $k \in \Z_{\ge 0}$, the finite set
$\{L_{k,0}, \ldots, L_{k,k}\}$ generates the semisimple
quotient category, and $L_{k,0} = b$ generates the full
category since every $L_{k,j}$ appears in the fusion product
$b^{\boxtimes n}$ for $n = j$.

(iii) Morita invariance of the derived center is
Theorem~\ref*{V1-thm:thqg-local-global-bridge}(iii).
\end{proof}

\begin{proposition}[Derived center gives the correct bulk;
  \ClaimStatusProvedHere]
\label{prop:benchmark-bulk-from-Cop}
\index{derived center!SL(2) Chern--Simons|textbf}
The chiral derived center of\/ $\cC_{\mathrm{op}}$ is
\begin{equation}\label{eq:benchmark-bulk-Cop}
  \cZ^{\mathrm{der}}_{\mathrm{ch}}(\cC_{\mathrm{op}})
  \;\simeq\;
  C^\bullet_{\mathrm{ch}}(V_k(\mathfrak{sl}_2),\,
  V_k(\mathfrak{sl}_2)),
\end{equation}
the chiral Hochschild cochain complex of the boundary algebra.
At generic $k$:
\begin{equation}\label{eq:benchmark-bulk-generic-Cop}
  H^0\!\bigl(
  \cZ^{\mathrm{der}}_{\mathrm{ch}}(\cC_{\mathrm{op}})
  \bigr)
  \;\cong\;
  Z\!\bigl(\widetilde{U}_k(\widehat{\mathfrak{sl}}_2)\bigr)
  \;\cong\; \C.
\end{equation}
At integrable level $k \in \Z_{\ge 0}$, the categorical
center is the Verlinde algebra with
$\dim = k + 1$ \textup{(}Proposition~\textup{\ref{prop:benchmark-verlinde}}\textup{)}.
\end{proposition}

\begin{proof}
The identification~\eqref{eq:benchmark-bulk-Cop} follows from
Theorem~\ref*{V1-thm:thqg-local-global-bridge}(ii): the derived
center of $\cC_{\mathrm{op}}$ is computed, in the chart
$b = V_k(\mathfrak{sl}_2)$, by the chiral Hochschild cochains
$C^\bullet_{\mathrm{ch}}(\End(b), \End(b))
= C^\bullet_{\mathrm{ch}}(V_k, V_k)$.
The generic identification~\eqref{eq:benchmark-bulk-generic-Cop}
is the statement that the center of the completed enveloping
algebra at non-critical, non-integrable level is
$\C$ (generated by the identity): the only chiral Hochschild
$0$-cocycle is the identity map $V_k \to V_k$.
At integrable levels, the restriction to the semisimple quotient
category introduces $k+1$ central idempotents (projections onto
simple summands), giving the Verlinde algebra.
\end{proof}

\begin{proposition}[Annulus trace gives the WZW partition function;
  \ClaimStatusProvedHere]
\label{prop:benchmark-annulus-partition}
\index{annulus trace!SL(2) Chern--Simons|textbf}
\index{WZW partition function|textbf}
The annulus trace of the open-sector category recovers the
genus-$1$ data:
\begin{enumerate}[label=\textup{(\roman*)}]
\item The Hochschild homology is
  $HH_*(\cC_{\mathrm{op}}) \simeq
  \int_{S^1} \cC_{\mathrm{op}}$
  \textup{(}Theorem~\textup{\ref*{V1-thm:thqg-annulus-trace}}\textup{)}.
  At integrable level~$k$, the degree-$0$ component is
  $HH_0(\cC_{\mathrm{op}}) \cong
  \bigoplus_{j=0}^k \C \cdot \chi_j$,
  spanned by the $k + 1$ characters of the integrable modules.

\item The non-separating nodal degeneration sends
  $\mathrm{Tr}_\cA \in HH_0(\cC_{\mathrm{op}})$ to the
  genus-$1$ curvature
  \textup{(}Proposition~\textup{\ref*{V1-prop:thqg-annulus-degeneration-kappa}}\textup{)}:
  \begin{equation}\label{eq:benchmark-annulus-kappa}
    \Delta_{\mathrm{ns}}(\mathrm{Tr}_\cA)
    \;=\;
    \kappa(V_k(\mathfrak{sl}_2)) \cdot \lambda_1
    \;=\;
    \frac{3(k+2)}{4}\, \lambda_1.
  \end{equation}

\item At integrable level~$k$, the WZW partition function on the
  torus is
  \begin{equation}\label{eq:benchmark-wzw-partition}
    Z_{\mathrm{WZW}}(\tau)
    \;=\;
    \sum_{j=0}^{k}
    |\chi_j(\tau)|^2,
  \end{equation}
  the diagonal modular invariant.  The characters $\chi_j$
  span the image of the clutching map
  $HH_0(\cC_{\mathrm{op}}) \to
  \mathrm{Fun}(\mathfrak{h})^{W \ltimes Q^\vee}$
  and transform under the modular $S$-matrix
  \textup{(}Proposition~\textup{\ref{prop:benchmark-verlinde}}\textup{)}.
\end{enumerate}

\begin{proof}

homology over $S^1$ is the annulus trace theorem
(Theorem~\ref*{V1-thm:thqg-annulus-trace}).  The degree-$0$ part
at integrable level is spanned by traces of the identity on
each simple module, which are the characters.

Proposition~\ref*{V1-prop:thqg-annulus-degeneration-kappa}.
The value $\kappa = 3(k+2)/4$ is from
Proposition~\ref{prop:benchmark-modular-data}(i).

(iii) The WZW partition function is obtained by pairing the
Hochschild homology of the left and right boundary categories
via the bilinear form on conformal blocks.  The diagonal

compatible with modular invariance when the fusion category
is multiplicity-free (which holds for
$\mathrm{Rep}(V_k(\mathfrak{sl}_2))$ at all integrable $k$

The $S$-matrix transformation $\chi_j(-1/\tau)

\end{proof}

\subsubsection{The Koszul dual}
\label{subsubsec:benchmark-koszul-dual}

The Koszul dual of $V_k(\mathfrak{sl}_2)$ is
$V_{k'}(\mathfrak{sl}_2)$ at the Feigin--Frenkel dual level:
\begin{equation}\label{eq:benchmark-koszul-dual-level}
  k'
  \;=\;
  -k - 2h^\vee
  \;=\;
  -k - 4.
\end{equation}
The Koszul involution $k \mapsto -k-4$ fixes $k = -2$
(the critical level); the self-dual point of
$V_k(\mathfrak{sl}_2)$ under this involution is $k = -2$, where
the Sugawara construction degenerates and the center of the
completed enveloping algebra becomes the algebra of opers on the
formal disc (Feigin--Frenkel~\cite{FBZ04}).  Away from
the critical level, Koszul duality exchanges Neumann and Dirichlet
boundary conditions in the HCS picture.


\subsubsection{The collision residue and $r$-matrix}
\label{subsubsec:benchmark-r-matrix}

The $\lambda$-bracket
\eqref{eq:benchmark-sl2-ope}
of $V_k(\mathfrak{sl}_2)$ has two terms:
\[
  \{J^a_\lambda J^b\}
  \;=\;
  \underbrace{f^{ab}_c\, J^c}_{\text{pole order }1
    \text{ in the OPE}}
  \;+\;
  \underbrace{k\,(J^a, J^b)\,\lambda}_{\text{pole order }2
    \text{ in the OPE}}.
\]
The OPE is
\begin{equation}\label{eq:benchmark-sl2-ope-explicit}
  J^a(z)\, J^b(w)
  \;\sim\;
  \frac{k\,(J^a, J^b)}{(z-w)^2}
  \;+\;
  \frac{f^{ab}_c\, J^c(w)}{z-w},
\end{equation}
with maximal pole order~$2$.

The collision residue extracts pole orders \emph{one less} than
the OPE (AP19: the bar kernel $d\log(z-w)$ absorbs one power of
$(z-w)$):
\begin{equation}\label{eq:benchmark-r-matrix}
  r(z)
  \;=\;
  \Res^{\mathrm{coll}}_{0,2}(\Theta_\cA)
  \;=\;
  \frac{\Omega_{\mathfrak{sl}_2}}{z},
\end{equation}
where
\begin{equation}\label{eq:benchmark-casimir}
  \Omega_{\mathfrak{sl}_2}
  \;=\;
  \sum_a T^a \otimes T^a
  \;=\;
  e \otimes f + f \otimes e
  + \tfrac{1}{2}\, h \otimes h
  \;\in\;
  \mathfrak{sl}_2 \otimes \mathfrak{sl}_2
\end{equation}
is the quadratic Casimir (the inverse of the normalized
Killing form $(J^a, J^b) = \mathrm{tr}_{\mathrm{fund}}(J^a J^b)$,
with $\{T^a\}$ an orthonormal basis).
The $r$-matrix $r(z) = \Omega/z$ is the standard rational
solution of the classical Yang--Baxter equation for
$\mathfrak{sl}_2$.  In the bar complex, it arises as the
arity-$2$, genus-$0$ projection of the universal MC
element~$\Theta_\cA$: the bar kernel $d\log(z-w) = dz/(z-w)$
extracts the leading-order collision data from the OPE, with
pole orders shifted down by one (AP19).

The classical Yang--Baxter equation
\begin{equation}\label{eq:benchmark-cybe}
  [r_{12}(z_1 - z_2),\, r_{13}(z_1 - z_3)]
  +
  [r_{12}(z_1 - z_2),\, r_{23}(z_2 - z_3)]
  +
  [r_{13}(z_1 - z_3),\, r_{23}(z_2 - z_3)]
  \;=\; 0
\end{equation}
for $r(z) = \Omega/z$ follows from the Jacobi identity
in $\mathfrak{sl}_2$: this is the arity-$3$ projection of the
Maurer--Cartan equation $D\Theta + \tfrac{1}{2}[\Theta,\Theta] = 0$
at genus~$0$.  The RTT quantization produces the Yangian
$Y_\hbar(\mathfrak{sl}_2)$.


\subsubsection{The modular characteristic and genus-$1$ free energy}
\label{subsubsec:benchmark-modular-characteristic}

The modular characteristic of $V_k(\mathfrak{sl}_2)$ is
computed from the universal formula
(Theorem~\ref*{V1-thm:genus-universality}):
\begin{equation}\label{eq:benchmark-kappa}
  \kappa\bigl(V_k(\mathfrak{sl}_2)\bigr)
  \;=\;
  \frac{\dim(\mathfrak{sl}_2)\,(k + h^\vee)}{2h^\vee}
  \;=\;
  \frac{3(k + 2)}{4}.
\end{equation}

\begin{proposition}[Modular data for the benchmark example;
  \ClaimStatusProvedHere]
\label{prop:benchmark-modular-data}
The modular invariants of $V_k(\mathfrak{sl}_2)$ are:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item $\kappa = 3(k+2)/4$.
\item The genus-$1$ free energy is
  $F_1 = \kappa/24 = (k+2)/32$.
\item The Koszul dual modular characteristic is
  $\kappa' = \kappa(V_{-k-4}(\mathfrak{sl}_2))
  = 3(-k - 4 + 2)/4
  = -3(k+2)/4 = -\kappa$.
\item The complementarity relation holds:
  $\kappa + \kappa' = 0$.
\item The Sugawara central charge is
  $c = k \cdot \dim(\mathfrak{sl}_2)/(k + h^\vee)
  = 3k/(k+2)$.
\item At the self-dual point $k = -2$
  \textup{(}critical level\textup{)}:
  $\kappa = 0$, $c$ is undefined
  \textup{(}the Sugawara construction degenerates\textup{)}.
\item At $k = 1$ \textup{(}first integrable level\textup{)}:
  $\kappa = 9/4$, $c = 1$, $F_1 = 3/32$.
\end{enumerate}
\end{proposition}

\begin{proof}
Item~(i): direct substitution into
$\dim(\fg)(k+h^\vee)/(2h^\vee)$ with
$\dim(\mathfrak{sl}_2) = 3$, $h^\vee = 2$.
Item~(ii): $F_1 = \kappa/24 = 3(k+2)/(4 \cdot 24) = (k+2)/32$.
Item~(iii): $k' = -k-4$, so $\kappa' = 3(-k-4+2)/4 = 3(-k-2)/4
= -3(k+2)/4$.
Item~(iv): sum.
Item~(v): standard Sugawara formula.
Item~(vi): $\kappa|_{k=-2} = 3 \cdot 0/4 = 0$;
at $k = -h^\vee$, the denominator of $c$ vanishes.
Item~(vii): $\kappa|_{k=1} = 3 \cdot 3/4 = 9/4$;
$c|_{k=1} = 3/(1+2) = 1$;
$F_1|_{k=1} = (1+2)/32 = 3/32$.
\end{proof}

\begin{remark}[Consistency check: $F_1$ from the character]
\label{rem:benchmark-f1-character}
An independent computation of $F_1$ uses the vacuum character.
At level $k$, the vacuum character of $V_k(\mathfrak{sl}_2)$ is
\[
  \chi_0(\tau)
  \;=\;
  q^{-c/24}\,\bigl(1 + 3q + \cdots\bigr),
  \qquad q = e^{2\pi i \tau},
\]
where the leading coefficient $3$ counts the $3$-dimensional
current space.  The genus-$1$ free energy $F_1 = -\log\eta(\tau)
\cdot \kappa$ (at leading order in $q$) gives
$F_1 = \kappa/24 = (k+2)/32$, matching item~(ii).
\end{remark}


\subsubsection{The universal bulk: chiral derived center}
\label{subsubsec:benchmark-derived-center}

The bulk algebra is \emph{not} the bar complex $\barB(V_k(\mathfrak{sl}_2))$
(AP-OC).  It is the chiral derived center
\begin{equation}\label{eq:benchmark-derived-center}
  \cZ^{\mathrm{der}}_{\mathrm{ch}}(\cA_\partial)
  \;=\;
  C^\bullet_{\mathrm{ch}}(V_k(\mathfrak{sl}_2),\,
  V_k(\mathfrak{sl}_2)),
\end{equation}
the chiral Hochschild cochain complex of the boundary algebra.
This is the universal bulk of the corrected Koszul triangle
(Theorem~\ref{thm:boundary-linear-bulk-boundary}).

For generic $k$ (non-integrable), the derived center is
a dg algebra whose cohomology in degree~$0$ is the center of the
completed enveloping algebra $\widetilde{U}(\widehat{\mathfrak{sl}}_2)$
at level~$k$, which is $\C$ (one-dimensional, generated by the
identity):
\begin{equation}\label{eq:benchmark-center-generic}
  H^0\!\bigl(\cZ^{\mathrm{der}}_{\mathrm{ch}}(V_k(\mathfrak{sl}_2))\bigr)
  \;\cong\;
  Z\!\bigl(\widetilde{U}_k(\widehat{\mathfrak{sl}}_2)\bigr)
  \;\cong\;
  \C.
\end{equation}

At \textbf{integrable levels} $k \in \Z_{\ge 0}$, the category
of modules $\mathrm{Rep}(V_k(\mathfrak{sl}_2))$ is a modular
tensor category with $k + 1$ simple objects
$L_0, L_1, \ldots, L_k$ (the integrable highest-weight modules
with highest weights $0, 1, \ldots, k$).  The categorical
derived center is
\begin{equation}\label{eq:benchmark-center-integrable}
  Z\!\bigl(\mathrm{Rep}(V_k(\mathfrak{sl}_2))\bigr)
  \;\cong\;
  \bigoplus_{j=0}^k \C \cdot e_j,
  \qquad \dim = k + 1,
\end{equation}
the \emph{Verlinde algebra}: a commutative semisimple algebra
whose structure constants are the fusion coefficients
$N_{ij}^{\;\;l}$ computed from the modular $S$-matrix.

\begin{proposition}[Verlinde algebra at integrable levels;
  \ClaimStatusProvedElsewhere]
\label{prop:benchmark-verlinde}
\textup{(Verlinde~\cite{Verlinde1988},
Huang~\cite{Huang2005}.)}
At integrable level $k \in \Z_{\ge 0}$, the modular $S$-matrix of
$\mathrm{Rep}(V_k(\mathfrak{sl}_2))$ is the
$(k{+}1) \times (k{+}1)$ matrix
\begin{equation}\label{eq:benchmark-s-matrix}
  S_{ij}
  \;=\;
  \sqrt{\frac{2}{k+2}}\;
  \sin\!\biggl(\frac{\pi(i+1)(j+1)}{k+2}\biggr),
  \qquad
  0 \le i, j \le k.
\end{equation}
The fusion coefficients are $N_{ij}^{\;\;l}
= \sum_{m=0}^k S_{im} S_{jm} \overline{S}_{lm} / S_{0m}$,
and the Verlinde algebra $\mathrm{Ver}_{k}(\mathfrak{sl}_2)$ is
the commutative ring $\Z[L_0, \ldots, L_k] / (\text{fusion
relations})$ with identity $L_0$.  This is the center of the
modular tensor category
$\mathrm{Rep}(V_k(\mathfrak{sl}_2))$.
\end{proposition}

\begin{remark}[Generic vs integrable: algebraic vs categorical center]
\label{rem:benchmark-generic-vs-integrable}
At generic $k$, the algebraic derived center $C^\bullet_{\mathrm{ch}}(V_k, V_k)$ has
$H^0 \cong \C$: the only chiral Hochschild $0$-cocycle is the identity.
At integrable $k$, the \emph{categorical} center of
$\mathrm{Rep}(V_k(\mathfrak{sl}_2))$ has dimension $k+1$.
The passage from algebraic to categorical is the restriction from
all modules to integrable ones, which introduces new central
endomorphisms (projections onto simple summands).  The bar complex
$\barB(V_k(\mathfrak{sl}_2))$ classifies twisting morphisms
(universal couplings between $V_k$ and $V_{-k-4}$); it does not
see the Verlinde algebra directly.
\end{remark}


\subsubsection{The Kazhdan--Lusztig equivalence and the line category}
\label{subsubsec:benchmark-kl-category}

On the chirally Koszul locus, the line-operator category is
modelled by modules for the Koszul dual
(Theorem~\ref{thm:lines_as_modules}):

  \;\simeq\;

At integrable level $k \in \Z_{\ge 0}$, the Kazhdan--Lusztig

  \cO_k^{\mathrm{int}}(\widehat{\mathfrak{sl}}_2)
  \;\simeq\;

\subsubsection{The annulus trace}
\label{subsubsec:benchmark-annulus-trace}

The annulus trace theorem
(Volume~I, Theorem~\ref*{V1-thm:thqg-annulus-trace}) identifies the
Hochschild homology of the open-sector category with the first
modular shadow:
\begin{equation}\label{eq:benchmark-annulus-trace}
  \mathrm{Tr}_\cA
  \;\simeq\;
  HH_*(\cC_{\mathrm{op}}).
\end{equation}

For the benchmark example, the annulus trace at each puncture
$p$ is computed by the character map.
At integrable level $k$, the Hochschild homology of
$\mathrm{Rep}(V_k(\mathfrak{sl}_2))$ in degree~$0$ is spanned
by the characters of the $k + 1$ simple modules:
\begin{equation}\label{eq:benchmark-characters}
  HH_0(\cC_{\mathrm{op}})
  \;\cong\;
  \bigoplus_{j=0}^k \C \cdot \chi_j,
  \qquad
  \chi_j(\tau)
  \;=\;
  \mathrm{tr}_{L_j}(q^{L_0 - c/24}).
\end{equation}
The modular $S$-matrix~\eqref{eq:benchmark-s-matrix} acts on this
space, and modular invariance of the physical partition function
is the statement that $Z(\tau) = \sum_{j=0}^k |\chi_j(\tau)|^2$
is $SL(2,\Z)$-invariant (diagonal modular invariant).

The clutching operation (restoring the tangential basepoint
$\tau_p$ to close the interval $I_p$ back to a circle) is
precisely the trace over the $E_1$-category $\cC_{\mathrm{op}}(I_p)$.
This trace produces the character $\chi_j$ from the module $L_j$.


\subsubsection{Factorization homology on
  \texorpdfstring{$\PP^1 \setminus \{0,\infty\}$}{P\^1 - \{0, infty\}}}
\label{subsubsec:benchmark-factorization-homology}

The factorization homology of the open-sector category over the
open complement $\PP^1 \setminus \{0,\infty\} \cong \C^\times$ is
\begin{equation}\label{eq:benchmark-factorization-homology}
  \int_{\C^\times} \cC_{\mathrm{op}}
  \;\simeq\;
  \mathrm{CB}_{0,2}(V_k(\mathfrak{sl}_2)),
\end{equation}
the space of genus-$0$ conformal blocks with two insertions
(at $0$ and $\infty$).  For a pair of modules
$M_0 \in V_k(\mathfrak{sl}_2)\text{-mod}$ at $0$ and
$M_\infty \in V_k(\mathfrak{sl}_2)\text{-mod}$ at $\infty$,
the conformal block space is
\begin{equation}\label{eq:benchmark-conformal-blocks}
  \mathrm{CB}_{0,2}(M_0, M_\infty)
  \;=\;
  \Hom_{V_k(\mathfrak{sl}_2)}(M_0,\, M_\infty^*),
\end{equation}
the space of $V_k(\mathfrak{sl}_2)$-intertwining operators
from $M_0$ to the contragredient of $M_\infty$.  At integrable
levels, this is $\delta_{j_0, j_\infty}$-dimensional for simple
modules $L_{j_0}$ and $L_{j_\infty}$: conformal blocks on
$\PP^1$ with two insertions are the bilinear pairing, which is
nonzero precisely when the two modules are contragredient.

For the vacuum-to-vacuum conformal block
($M_0 = M_\infty = V_k(\mathfrak{sl}_2)$), the space is
one-dimensional:
\begin{equation}\label{eq:benchmark-vacuum-block}
  \mathrm{CB}_{0,2}(V_k, V_k) \;\cong\; \C.
\end{equation}
This is the partition function of the theory on $\PP^1$ with
vacuum boundary conditions at both punctures.

\begin{remark}[The shadow depth dichotomy in holography]
\label{rem:platonic-shadow-dichotomy}
The five examples exhibit a sharp structural pattern.
The two current-algebra systems (CS and M2) are class~$\mathbf{L}$
with finite shadow tower ($r_{\max} = 3$), complementarity
anti-symmetry $\kappa + \kappa' = 0$, and $r$-matrices with
simple poles satisfying the classical Yang--Baxter equation.
The three $\mathcal{W}$-algebra systems (3d gravity, $\mathcal{W}_3$,
M5) are class~$\mathbf{M}$ with infinite shadow tower,
complementarity $\kappa + \kappa' \ne 0$, and $r$-matrices with
higher-order poles.  The M2--M5 coupling maps
class~$\mathbf{L}$ into class~$\mathbf{M}$: the Drinfeld--Sokolov
functor is the shadow-depth escalator.

At the critical level $k = -h^\vee$, the CS datum degenerates:
$\kappa = 0$, the bar complex is uncurved, the shadow tower
collapses, and the Feigin--Frenkel center replaces the generic
$\cZ^{\mathrm{der}}_{\mathrm{ch}}$.  For the Virasoro at $c = 0$:
$\kappa = 0$, uncurved bar, but the shadow tower remains infinite
(higher-arity components are independent of $\kappa$; AP31).
This is the gravity analogue of anomaly cancellation: the
scalar curvature vanishes, but the nonlinear tower persists.
\end{remark}
