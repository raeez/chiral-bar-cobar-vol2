\section{Worked examples}
\label{sec:examples_worked}

This section computes the bar-cobar package for a systematic sequence
of examples, each teaching a structural lesson unavailable from its predecessors. The progression:
\emph{local genus-$0$ computations} on the half-space
(\S\S\ref{subsec:free-chiral}--\ref{subsec:DDCA-ainfty}), testing the
$\Ainf$ operations and DK-0 bridge; \emph{dualities}
(\S\S\ref{subsec:xyz-model-boundary}--\ref{subsec:sqed-xyz-mirror}),
where mirror symmetry becomes bar-cobar equivalence; \emph{primitive
packages} (\S\S\ref{subsec:twisted-holography-primitive}--\ref{subsec:twisted-M5-primitive}),
assembling full open/closed data; a \emph{benchmark global example}
(\S\ref{subsec:benchmark-global-sl2}), the first complete computation
on a marked curve; the \emph{integrable regime}
(\S\ref{subsec:bar-complex-integrable}), where null vectors truncate the
bar complex; the \emph{extended landscape}
(\S\S\ref{subsec:exceptional-affine}--\ref{subsec:coulomb-coha}),
testing universality across exceptional types, non-principal reductions,
fermionic generators, and non-Lagrangian inputs; and a
\emph{summary table} (\S\ref{subsec:five-platonic-instantiations}),
collecting the five canonical platonic data.

Throughout we compute in the configuration-space model with kernels built from the HT propagator $K=K^{\C}\otimes K^{\R}$:
$K^{\C}$ has a simple logarithmic pole along diagonals in $\FM_2(\C)$ and $K^{\R}$ is the standard step-function (time-ordering) kernel on~$\R$ (or the half-line for boundary computations).

\subsection{Free chiral multiplet}
\label{subsec:free-chiral}
Consider a free chiral multiplet in the HT twist. The fields are $(\phi,\bar\phi)$ and their superpartners with BV completions. The action is quadratic; hence all $m_{k\ge3}$ vanish by Wick's theorem.

\paragraph{Binary operation.}
The two-point OPE kernel is
\[
m_2(a,b)(z) \;=\; \int_{\FM_2(\C)} K^{\C}(z_{12})\cdot a\otimes b,
\]
with $K^{\C}(z_{12})\sim \frac{1}{z_{12}}$ in the holomorphic coordinate. The regular part (nonnegative powers of $\lambda$) yields a strictly commutative product on cohomology; the antisymmetric singular part vanishes in cohomology by exactness of $\dbar$ on the kernel.

\begin{proposition}[PVA for the free multiplet; \ClaimStatusConditional]
\label{prop:PVA-free}
For the HT free chiral multiplet, conditional on Theorem~\ref{thm:physics-bridge}, $H^\bullet(A,Q)$ is a commutative vertex algebra; the $(-1)$-shifted Poisson bracket on cohomology is trivial.
\end{proposition}

\begin{proof}
Immediate from the fact that $m_2^{\mathrm{sing}}$ is $Q$-exact (the residue cancels after integrating the distributional kernel against closed test fields) and $m_{k\ge3}=0$.
\end{proof}

\subsection{Landau--Ginzburg model with cubic superpotential}
\label{subsec:LG-cubic}
Consider a single chiral field $\phi$ with superpotential $W(\phi)=\frac{\lambda}{3}\phi^3$.
In the HT twist, the interaction vertex induces a nontrivial $m_3$ at tree level.

\paragraph{The $m_3$ integral.}
At leading order in $\lambda$,
\[
m_3(a,b,c) \;=\; \lambda \int_{\FM_3(\C)} K^{\C}(z_{12})K^{\C}(z_{23})K^{\C}(z_{31}) \cdot (a\otimes b\otimes c),
\]
with ordered times in $\Conf_3(\R)$ capturing the $E_1$ homotopy; the total kernel has degree $-2$ (cohomological). Boundary faces of $\FM_3(\C)$ encode the Stasheff identity at degree $3$.

\paragraph{Cohomology.}
The $m_3$ appears only at chain level; using Stokes' theorem and AOS, its contribution to associativity is $Q$-exact, hence vanishes on $H^\bullet$. However, the antisymmetric part of $m_2$ produces a nontrivial $(-1)$-shifted bracket on $H^\bullet$ proportional to~$\lambda$.

\begin{proposition}[PVA for LG-cubic; \ClaimStatusConditional]
\label{prop:PVA-LG}
For the HT Landau--Ginzburg cubic model, conditional on Theorem~\ref{thm:physics-bridge}, $H^\bullet(A,Q)$ is a $(-1)$-shifted PVA with nontrivial bracket $\{\phi_\lambda \phi\}\sim \lambda\, \lambda^0$ (up to normalization), while $m_{k\ge3}$ vanish in cohomology.
\end{proposition}

\begin{proof}
The bracket originates from the singular residue of $m_2$; $m_3$ contributes homotopies only. Computation follows from the standard $\FM_3(\C)$ integral and the LG interaction vertex.
\end{proof}

\subsection{Abelian Chern--Simons with boundary}
\label{subsec:CS-abelian}
Let $A$ be abelian Chern--Simons theory on a half-space with HT boundary condition. The boundary algebra contains a current $J(z)$ with OPE
\[
J(z)J(w) \;\sim\; \frac{k}{(z-w)^2} + \text{regular},
\]
at level $k$ determined by the CS coupling.

\paragraph{Bulk-to-boundary and $R(z)$.}
Place two line operators of charges $q_1,q_2$ at $z_1,z_2$ on the boundary. The bulk-to-boundary kernel yields a spectral braiding
\[
R(z)\cdot (v_{q_1}\otimes v_{q_2}) \;=\; \exp\!\Bigl( \hbar\, \frac{q_1 q_2}{z} \Bigr) \,(v_{q_2}\otimes v_{q_1}),
\]
up to normalization (in perturbation theory). The classical $r(z)=\frac{q_1 q_2}{z}$ satisfies CYBE.

\begin{proposition}[Boundary Kac--Moody and spectral $R$; \ClaimStatusConditional]
\label{prop:CS-R}
For abelian Chern--Simons with HT boundary condition, conditional on Theorem~\ref{thm:physics-bridge}, the boundary category of line operators is braided by $R(z)$, with classical limit $r(z)=\frac{k}{z}$ (after normalization). The bulk chiral Hochschild cochains produce the same $r(z)$ via the Laplace transform of the bulk $\lambda$-bracket kernel as in Proposition~\ref{prop:field-theory-r}.
\end{proposition}

\begin{proof}
The current algebra OPE computes the leading residue; the configuration integral over $\FM_2(\C)$ reproduces $1/z$. The Laplace transform calculation matches the boundary braiding.
\end{proof}

\begin{remark}[Koszul duality of tensor products and the string-theory point]
\label{rem:koszul-tensor-string}
\index{Koszul duality!tensor products}%
\index{string theory!ghost system!Koszul dual}%
For mutually regular Koszul algebras $\cA$, $\cB$ (no OPE
singularities between generators of $\cA$ and generators
of $\cB$), the bar complex of the tensor product splits:
$\barB(\cA \otimes \cB) \simeq \barB(\cA) \otimes \barB(\cB)$
as dg coalgebras, because cross-collisions on $\FM_k(\C)$
contribute zero. K\"unneth and dualisation give
$(\cA \otimes \cB)^! \simeq \cA^! \otimes \cB^!$,
so the complementarity constant is additive:
$K(\cA \otimes \cB) = K(\cA) + K(\cB)$.

This is directly verified for the $bc$/$\beta\gamma$
entry in the DK-0 table above. Since
$bc_\lambda^! = \beta\gamma_\lambda$ and
$\beta\gamma_\mu^! = bc_\mu$ (the
$\mathrm{Ext}/\mathrm{Sym}$ exchange of statistics), and
$K(bc_\lambda) = K(\beta\gamma_\mu) = 0$ for all weights, the
tensor product formula gives
\[
(bc_\lambda \otimes \beta\gamma_\mu)^!
\;\simeq\;
\beta\gamma_\lambda \otimes bc_\mu.
\]
At the string-theory point $\lambda = 2$, $\mu = 3/2$: the
ghost system $bc_2 \otimes \beta\gamma_{3/2}$ ($c_{\mathrm{ghost}}
= -26 + 11 = -15$) has Koszul dual
$\beta\gamma_2 \otimes bc_{3/2}$ ($c_{\mathrm{ghost}}^! = 26 - 11
= 15$). The dual reverses statistics: the spin-$2$ ghost becomes
a bosonic spin-$2$ field (a graviton), and the spin-$3/2$ ghost
becomes a fermionic spin-$3/2$ field (a gravitino). For a
free-field BRST system with $c_{\mathrm{matter}} +
c_{\mathrm{ghost}} = 0$ built entirely from Heisenberg, $bc$, and
$\beta\gamma$ factors, the total $K = 0$ ensures that
$c_{\mathrm{matter}}^! + c_{\mathrm{ghost}}^! = 0$: BRST anomaly
cancellation is preserved under Koszul duality.
\end{remark}

\begin{remark}[Intersection geometry across the landscape]
\label{rem:koszul-clean-landscape}
The depth classification
(Definition~\ref{def:shadow-depth-intersection})
measures the complexity of the Lagrangian self-intersection
$\cL \times_{\cM}^h \cL$ within the standard atlas families.
All four Koszul classes $\mathbf{G}$, $\mathbf{L}$, $\mathbf{C}$,
$\mathbf{M}$ are chirally Koszul (PBW degeneration holds
throughout the standard landscape). Classes $\mathbf{G}$,
$\mathbf{L}$, $\mathbf{C}$ are $\Ainf$-formal
($m_k = 0$ for $k \ge 3$ on cohomology), while
class~$\mathbf{M}$ is Koszul but \emph{non-formal}: the
higher $\Ainf$ operations are non-trivial on cohomology,
and shadow depth is infinite. A fifth class~$\mathbf{W}$
(wild) marks the non-Koszul boundary: the Kronecker quiver
$K_m$ for $m \geq 3$ has indefinite Euler form and the bar
spectral sequence does not collapse
(Volume~I, Remark~\ref*{V1-rem:wild-quiver-boundary}).

For the Heisenberg~($\cH_k$), affine
Kac--Moody~($\widehat{\fg}_k$), and $\beta\gamma$/$bc$
families, the self-intersection is \emph{clean}: the derived
intersection carries no higher Tor beyond the expected degree,
so the self-intersection groupoid is \'etale and bar-cobar
inversion holds unconditionally
(Conjecture~\ref{conj:etale-groupoid-koszul}).
\begin{itemize}[nosep]
\item \emph{Heisenberg} (class~$\mathbf{G}$, depth~$2$):
 only the double-pole OPE contributes, $m_k = 0$ for
 $k \ge 3$, and the intersection is a smooth correspondence.
\item \emph{Kac--Moody} (class~$\mathbf{L}$, depth~$3$):
 the simple-pole bracket creates a codimension-$1$
 transverse defect, but Jacobi kills all higher strata.
\item \emph{Virasoro} (class~$\mathbf{M}$, depth~$\infty$):
 the fourth-order pole $c/(z-w)^4$ creates excess
 intersection dimension, and the Swiss-cheese $\Ainf$
 operations $m_k \ne 0$ for all $k \ge 3$
 (Proposition~\ref{prop:vir-truncation}).
 The self-intersection is not clean: the fourth-order pole
 forces non-formal Swiss-cheese structure, with
 $m_k^{\mathrm{SC}} \ne 0$ for all $k \ge 3$.
 The Virasoro is chirally Koszul (PBW spectral sequence
 concentrates) and possesses a well-defined Koszul dual
 ($\mathrm{Vir}_c^! = \mathrm{Vir}_{26-c}$), but the
 big dual carries higher $\Ainf$ operations; shadow
 depth~$\infty$ measures the complexity of this
 non-formal $\Ainf$ structure.
\end{itemize}
\end{remark}

\subsection{Kodaira--Spencer universal defect algebra}
\label{subsec:KS-defect}

In Zeng's boundary-algebra formulation of twisted holography~\cite{Zeng2024,CDG2023},
the large-$N$ open-side chiral algebra is built from the following field content
in $\mathfrak{gl}_N$: a $(b,c)$ ghost system in the adjoint,
a pair $(Z_1,Z_2)$ of adjoint $\beta\gamma$ systems, and
a pair $(I,J)$ of fundamental/anti-fundamental $\beta\gamma$ systems.

\paragraph{BRST charge.}
The BRST differential is
\[
 Q \;=\; \oint\!\bigl(\operatorname{Tr}(c\,Z_1 Z_2)
 + \operatorname{Tr}(I\,c\,J)
 + \tfrac{1}{2}\operatorname{Tr}(b\,c^2)\bigr).
\]
The first two terms encode the holomorphic moment map constraints
(complex Chern--Simons on $\C^3$ after the holomorphic--topological twist),
while the ghost self-coupling $\frac{1}{2}\operatorname{Tr}(bc^2)$
implements the BRST closure $Q^2=0$ via the Maurer--Cartan equation
in the $\mathfrak{gl}_N$ gauge sector.

\paragraph{Large-$N$ cohomology generators.}
At $N=\infty$ the BRST cohomology is freely generated by
single-trace operators
\[
 A(n),\; B(n),\; C(n),\; D(n),\; E_t(n),
 \qquad n \ge 0,
\]
where $A(n)=\operatorname{Tr}(Z_1^n\,\partial Z_1)$, etc., and $E_t(n)$
denotes the bilinear $I$--$J$ family labelled by a spectral parameter~$t$.
These generate a vertex algebra isomorphic to the Kodaira--Spencer
chiral algebra of~\cite{BCOV93}, extended by the defect sector.

\paragraph{The universal defect algebra.}
Define $U_{\mathrm{KS}} := A_{\partial,\mathrm{KS}}(B_{\mathrm{univ}})$, the
universal defect algebra obtained by applying the boundary
bar construction of Volume~I to the universal bulk input~$B_{\mathrm{univ}}$.
The KK (Kaluza--Klein) reduction mechanism identifies
\[
 \text{boundary chiral algebra}
 \;=\;
 \text{universal defect chiral algebra of the parent 6d theory},
\]
connecting the large-$N$ gauged $\beta\gamma$ VOA
to Kodaira--Spencer gravity. Koszul duality
($\cA_{\mathrm{open}} \mapsto \cA^!_{\mathrm{open}}$
via Verdier duality on bar cohomology) exchanges the
boundary algebra with its dual, while the chiral derived
centre $\cZ^{\mathrm{der}}_{\mathrm{ch}}(\cA_{\mathrm{open}})
= C^\bullet_{\mathrm{ch}}(\cA_{\mathrm{open}},
\cA_{\mathrm{open}})$ computes the universal bulk algebra.
Bar-cobar inversion (Theorem~B of Volume~I) recovers the
\emph{original} boundary algebra:
$\Omega B(\cA_{\mathrm{open}}) \simeq \cA_{\mathrm{open}}$.

\subsection{The M2-brane instanton particle: bar-cobar as holographic duality}
\label{subsec:M2-holography}

Costello's M2-brane program~\cite{Costello-1705.02500v1,costello-gaiotto}
identifies, at large rank~$K$, the M2-brane boundary algebra
as Koszul dual to the 11d twisted supergravity bulk algebra.
At large rank~$K$, this realizes \emph{bar-cobar duality as holographic duality}.

\paragraph{Seed algebra.}
The classical starting point is the double-current algebra
\[
 \mathfrak{g}_{\mathrm{dbl}}
 \;=\; \mathfrak{gl}_r \otimes \C[u,v],
\]
a polynomial-current extension of $\mathfrak{gl}_r$ in two spectral variables.
Its enveloping algebra $U(\mathfrak{g}_{\mathrm{dbl}})$ carries a natural
filtration by polynomial degree in $(u,v)$.

\paragraph{Koszul dual.}
The Koszul dual of the M2-brane algebra is obtained by Verdier duality on bar cohomology:
\[
 A^!_{\mathrm{M2}}
 \;=\; H^\bullet\bigl(B(U(\mathfrak{g}_{\mathrm{dbl}}))\bigr)^{\vee}
 \;\simeq\; C^\bullet(\mathfrak{g}_{\mathrm{dbl}}),
\]
where $C^\bullet(\mathfrak{g}_{\mathrm{dbl}})$ denotes the Chevalley--Eilenberg
cochains. The identification follows from the classical Koszulness of
$U(\mathfrak{g}_{\mathrm{dbl}})$ (via the PBW filtration and Koszulness of
$\operatorname{Sym}$), which ensures that bar cohomology concentrates in a single
degree and the linear dual recovers $C^\bullet(\mathfrak{g}_{\mathrm{dbl}})$.

\paragraph{Sector-by-sector matching.}
For each sector~$\sigma$ and level~$\ell$, the holographic dictionary reads
\[
 \bigl(A_{\partial,\infty}(\sigma,\ell)\bigr)^!
 \;\cong\;
 A_{\mathrm{bulk}}(\sigma,\ell),
\]
where $A_{\partial,\infty}$ is the large-$K$ boundary algebra and
$A_{\mathrm{bulk}}$ is the twisted supergravity bulk algebra.
The Koszul dual functor $(-)^!$ exchanges the boundary's associative
product with the bulk's Lie bracket, exactly as $\mathrm{Com}^!=\mathrm{Lie}$
at the operadic level.

\paragraph{Line operators and spectral kernel.}
The category of M2-brane line operators admits a description
\[
 \cC_{\mathrm{M2}}
 \;\simeq\;
 A^!_{\mathrm{bulk}}\text{-}\mathrm{mod},
 \qquad
 \text{with spectral kernel } r_{\mathrm{M2}}(u),
\]
where $r_{\mathrm{M2}}(u)$ is the classical $r$-matrix obtained from the
Laplace transform of the $\lambda$-bracket on $\mathfrak{g}_{\mathrm{dbl}}$,
as in Proposition~\ref{prop:field-theory-r}. The spectral braiding
reproduces the RTT relations of the shifted Yangian~\cite{DNP2025}.
Bar-cobar duality at the algebraic level
\emph{is} holographic duality at the physical level.

\subsection{The deformed double current algebra: \texorpdfstring{$\Ainf$}{A-infinity}
 operations from the 5d HT theory}

The M2-brane boundary algebra of Section~\ref{subsec:M2-holography}
arises from a 5d holomorphic-topological theory on
$\R_t \times \C^2_{z_1,z_2}$, where observables factorize
holomorphically in $(z_1,z_2)$ and associatively in~$t$.
This section computes the $\Ainf$ operations $(m_1,m_2,m_3)$
on the boundary algebra from first principles, determines
the shadow archetype, and identifies the Swiss-cheese
structure. The primitive object throughout is the
open-sector dg-category $\cC_{\mathrm{op}}$; the boundary
algebra $\cA_\partial = \End(b)$ is a chart
(Morita-invariant), and the bulk observables are the chiral
derived center $\Zder^{\mathrm{ch}}(\cA_\partial)$.

\subsubsection{The 5d HT action and field content}

The 5d theory is holomorphic Chern--Simons on
$\R_t\times\C^2$, in the holomorphic-topological twist
\cite{costello-gaiotto,Costello-1705.02500v1}.

\paragraph{BV field content.}
Fix a reductive Lie algebra $\fg = \mathfrak{gl}_r$. The
fields live on $M = \R_t \times \C^2_{z_1,z_2}$ with
coordinates $(t; z_1, z_2)$:
\begin{itemize}[nosep]
\item Gauge field:
 $A \in \Omega^{0,\bullet}(\C^2) \otimes
 \Omega^\bullet(\R) \otimes \fg$,
 with BV degrees determined by the holomorphic-topological
 twist. The physical component is
 $A^{0,1} = A_{\bar z_1}\,d\bar z_1 +
 A_{\bar z_2}\,d\bar z_2 + A_t\,dt$
 (partial connection in the $\dbar + d_t$ directions).
\item Ghost: $c \in \Omega^{0,0}(\C^2)\otimes\Omega^0(\R)
 \otimes \fg$ of cohomological degree $-1$.
\item Antifields and antighosts complete the BV package
 via the standard $(-1)$-shifted symplectic pairing
 $\langle \phi, \phi^+ \rangle_{\mathrm{BV}} =
 \int_M \operatorname{Tr}(\phi \wedge \phi^+)
 \wedge \Omega$,
 where $\Omega = dz_1 \wedge dz_2$ is the holomorphic
 volume form on $\C^2$.
\end{itemize}

\paragraph{Action.}
The classical BV action is the holomorphic Chern--Simons
functional on $\C^2$ fibered over $\R_t$:
\begin{equation}\label{eq:5d-HT-action}
 S_{\mathrm{5d}}
 \;=\;
 \int_{\R\times\C^2}
 \Omega \wedge
 \operatorname{Tr}\!\Bigl(
 A \wedge (\dbar + d_t) A
 \;+\; \tfrac{2}{3}\, A \wedge A \wedge A
 \Bigr).
\end{equation}
The kinetic operator is $Q = \dbar_{\C^2} + d_t$: the
Dolbeault operator on $\C^2$ plus the de~Rham operator
on~$\R$. The theory satisfies the conditions of
Theorem~\ref{thm:physics-bridge}: the action is
first-order in time derivatives and holomorphic in
$(z_1,z_2)$, and one-loop finiteness follows from the
Calabi--Yau condition $c_1(\C^2) = 0$
\cite{costello-renormalization}.

\paragraph{Propagator.}
The propagator factors as
\begin{equation}\label{eq:5d-propagator}
 G(z_1,z_2,\bar z_1,\bar z_2;\, t)
 \;=\;
 K^{\C^2}(z_1,z_2,\bar z_1,\bar z_2)
 \;\otimes\;
 K^{\R}(t),
\end{equation}
where the holomorphic kernel is the Bochner--Martinelli
kernel on $\C^2$:
\[
 K^{\C^2}(z,w)
 \;=\;
 \frac{1}{(2\pi i)^2}
 \cdot
 \frac{(\bar z_1 - \bar w_1)\,d\bar z_2
 - (\bar z_2 - \bar w_2)\,d\bar z_1}
 {\bigl(|z_1-w_1|^2 + |z_2-w_2|^2\bigr)^2},
\]
and the topological kernel is the Heaviside step function
$K^{\R}(t) = \theta(t)$ (time-ordering). The key
difference from the 3d theory on $\R \times \C$: the
holomorphic kernel has a pole of order \emph{two}
(in real codimension four, i.e.\
homogeneous of degree $-3$ in $\C^2 \cong \R^4$),
not a simple pole.

\subsubsection{Boundary generators}

Place the boundary at $t = 0$. The boundary algebra
$\cA_\partial$ has generators
\[
 J^a_{m,n}
 \;=\;
 \oint\oint
 z_1^m\, z_2^n\, J^a(z_1,z_2)\,
 \frac{dz_1}{2\pi i}\, \frac{dz_2}{2\pi i},
 \qquad a \in \fg,\quad m,n \ge 0,
\]
where $J^a(z_1,z_2) = \lim_{t\to 0^+}
\operatorname{Tr}(T^a\, A(z_1,z_2,t))$ is the
boundary current. These are the modes of the
double-current algebra
$\fg_{\mathrm{dbl}} = \fg\otimes\C[u,v]$,
with $u = z_1$, $v = z_2$.

\subsubsection{Computation of \texorpdfstring{$m_1$}{m1}: the BRST differential}

\begin{proposition}[$m_1$ for the DDCA; \ClaimStatusProvedHere]
\label{prop:DDCA-m1}
The BRST differential $m_1 = Q$ acts on boundary generators
as
\[
 m_1(J^a_{m,n}) \;=\; 0
\]
for all $a \in \fg$ and $m,n \ge 0$. That is, the
generators $J^a_{m,n}$ are BRST-closed.
\end{proposition}

\begin{proof}
The BRST operator $Q = \dbar + d_t$ acts on the
boundary current $J^a(z_1,z_2)$ as a $\dbar$-operator on
$\C^2$. But $J^a(z_1,z_2)$ is the restriction to $t=0$
of a $(0,0)$-form (the $dt$-component of the gauge field),
so $\dbar J^a$ vanishes on the modes $z_1^m z_2^n$ by
holomorphicity of the monomials. The $d_t$-component
vanishes because the boundary condition sets
$A|_{t=0^+}$ to lie in the holomorphic sector. Hence
$m_1(J^a_{m,n}) = 0$.
\end{proof}

\begin{remark}[Comparison with 3d]
In the 3d theory on $\R\times\C$, the boundary generators
$J^a_n = \oint z^n J^a(z)\,dz$ are likewise $Q$-closed.
The 5d theory doubles the spectral index
$(n) \to (m,n)$ without changing the cohomological
structure of $m_1$.
\end{remark}

\subsubsection{Computation of \texorpdfstring{$m_2$}{m2}: the binary product}

\begin{theorem}[$m_2$ for the DDCA; \ClaimStatusProvedHere]
\label{thm:DDCA-m2}
The binary $\Ainf$ operation on the boundary algebra of
the 5d HT theory is
\begin{equation}\label{eq:DDCA-m2}
 m_2(J^a_{m,n},\; J^b_{p,q})
 \;=\;
 f^{ab}_{\phantom{ab}c}\, J^c_{m+p,\,n+q}
 \;+\;
 k\,\delta^{ab}\,
 P_2(m,n,p,q;\lambda_1,\lambda_2),
\end{equation}
where $f^{ab}_c$ are the structure constants of $\fg$,
$k$ is the level determined by the 5d coupling constant,
and $P_2$ is a polynomial in the spectral parameters
$\lambda_1,\lambda_2$ (dual to $z_1,z_2$):
\begin{equation}\label{eq:DDCA-central-term}
 P_2(m,n,p,q;\lambda_1,\lambda_2)
 \;=\;
 \delta_{m+p,0}\,\delta_{n+q,0}
 \;\cdot\;
 \sum_{j=0}^{|m|-1}
 \sum_{\ell=0}^{|n|-1}
 \lambda_1^{j}\,\lambda_2^{\ell}.
\end{equation}
On cohomology, the singular part of $m_2$ gives the
$(-1)$-shifted $\lambda$-bracket:
\begin{equation}\label{eq:DDCA-bracket}
 \{J^a_{m,n}\,{}_{\lambda_1,\lambda_2}\, J^b_{p,q}\}
 \;=\;
 f^{ab}_{\phantom{ab}c}\, J^c_{m+p,\,n+q}
 \;+\;
 k\,\delta^{ab}\,\delta_{m+p,0}\,\delta_{n+q,0}
 \;\cdot\;
 \frac{\lambda_1^{|m|} - (-\lambda_1)^{|m|}}
 {2\lambda_1}
 \cdot
 \frac{\lambda_2^{|n|} - (-\lambda_2)^{|n|}}
 {2\lambda_2},
\end{equation}
where each factor $(\lambda^{|m|} - (-\lambda)^{|m|})/(2\lambda)$
is a polynomial: it equals $\lambda^{|m|-1}$ when $|m|$ is odd
and vanishes when $|m|$ is even (the central extension
is supported on odd-mode-number pairs).
The result is a polynomial in $\lambda_1,\lambda_2$ of bidegree
$(\max(|m|,|p|)-1,\;\max(|n|,|q|)-1)$. At mode
numbers $(m,n) = (1,0)$ and $(p,q) = (-1,0)$, this
reproduces the Kac--Moody level:
$\{J^a_{1,0}\,{}_{\lambda_1}\, J^b_{-1,0}\}_{\mathrm{sing}}
= k\,\delta^{ab}$.
\end{theorem}

\begin{proof}
The computation is the 5d analogue of the Kac--Moody
OPE extraction from HCS (Example~\ref{ex:class-s}
and~\cite{costello-gaiotto}). The binary product is
\[
 m_2(J^a_{m,n},\, J^b_{p,q})
 \;=\;
 \int_{\FM_2(\C^2)}
 K^{\C^2}(z,w)
 \cdot
 (z_1^m z_2^n\, T^a)
 \otimes
 (w_1^p w_2^q\, T^b),
\]
with ordered times $t_1 > t_2$ in the topological
direction enforced by $K^{\R}(t) = \theta(t)$.

\emph{Structure-constant term.} The interaction vertex
$\frac{2}{3}\operatorname{Tr}(A^3)$ in~\eqref{eq:5d-HT-action}
produces, at tree level, the single propagator exchange
between two boundary currents. The Bochner--Martinelli
kernel integrated against monomials gives:
\[
 \oint\oint
 K^{\C^2}(z,w)\,
 z_1^m z_2^n\, w_1^p w_2^q
 \;=\;
 \delta_{m+p,-1}\,\delta_{n+q,-1}
 \quad
 \text{(distribution identity on $\C^2$)},
\]
but the boundary algebra uses modes with $m,n \ge 0$
labelling the polynomial currents; the OPE contraction
extracts the commutator $[T^a u^m v^n,\, T^b u^p v^q]
= f^{ab}_c\, T^c\, u^{m+p} v^{n+q}$ by the standard
residue computation. This is the structure-constant
term in \eqref{eq:DDCA-m2}.

\emph{Central extension.} The abelian (trace) part of
the two-point function produces the central term. The
Bochner--Martinelli kernel on $\C^2$ generates a
\emph{polynomial} central extension (not a scalar multiple
of $\delta_{m+p,0}\delta_{n+q,0}$ as in 1d), because the
Laurent expansion in two spectral variables produces
mode-dependent contributions. Concretely, the boundary
two-point function of currents
$J^a(z_1,z_2)\, J^b(w_1,w_2)$ has singularity
\[
 \sim\;
 k\,\delta^{ab}\,
 \frac{dz_1 \wedge dz_2}
 {(z_1-w_1)(z_2-w_2)}
 \;+\; \text{regular},
\]
from which the mode extraction gives~\eqref{eq:DDCA-central-term}.
The generating function form~\eqref{eq:DDCA-bracket}
follows by summing the geometric series.
\end{proof}

\begin{remark}[Comparison with affine Kac--Moody]
\label{rem:DDCA-vs-KM}
For the 3d theory on $\R \times \C$ (one holomorphic
direction), $m_2$ is the affine Kac--Moody bracket with
a single spectral parameter $\lambda$. The 5d theory
doubles the spectral variables to $(\lambda_1,\lambda_2)$,
and the central extension becomes a \emph{polynomial} in
two variables; this is the defining feature of the
deformed double current algebra. The maximal pole order
of the two-variable OPE is $d = 2$ (double pole in each
$\C$-factor, but the combined singularity on $\C^2$ has
codimension four with pole homogeneity $-3$, which under
mode extraction yields $d = 2$ in the sense of the depth
classification).
\end{remark}

\subsubsection{Computation of \texorpdfstring{$m_3$}{m3}: the first higher operation}

\begin{theorem}[$m_3$ vanishes on DDCA cohomology;
 \ClaimStatusProvedHere]
\label{thm:DDCA-m3}
The ternary $\Ainf$ operation $m_3$ vanishes on the
cohomology $H^\bullet(\cA_\partial, Q)$ of the DDCA
boundary algebra:
\[
 m_3\bigl([J^a_{m_1,n_1}],\;
 [J^b_{m_2,n_2}],\;
 [J^c_{m_3,n_3}]\bigr)
 \;=\; 0
 \qquad
 \text{in } H^\bullet.
\]
\end{theorem}

\begin{proof}
The $m_3$ operation arises from tree-level Feynman
diagrams with three external boundary insertions and one
internal (trivalent) vertex, integrated over
$\FM_3(\C^2) \times \Conf_3(\R)$. We evaluate this
integral in three steps.

\medskip\noindent\textbf{Step 1: Diagram topology.}
At tree level, the only contributing diagram is the
Y-shaped graph: three boundary-to-bulk propagators
meeting at a single cubic vertex $\operatorname{Tr}(A^3)$ in
the interior. The amplitude is
\begin{equation}\label{eq:DDCA-m3-integral}
 m_3(J^a,J^b,J^c)
 \;=\;
 \int_{\Conf_3(\R)}
 \int_{\C^2}
 \prod_{i=1}^{3}
 K^{\C^2}(z_i, w)\,
 K^{\R}(t_i - s)
 \cdot
 f^{abc}\, d^4w\, ds,
\end{equation}
where $(w_1,w_2,s) \in \C^2 \times \R$ is the internal
vertex and $(z_i, t_i)$ are the boundary insertions with
$t_1 < t_2 < t_3$.

\medskip\noindent\textbf{Step 2: Topological integral.}
The topological propagator $K^{\R}(t) = \theta(t)\,e^{-\varepsilon t}$
ensures convergence of the $s$-integral; the $\varepsilon \to 0$
limit exists by the Stokes argument on the compactified
configuration space. The regulated $\R$-integral is
$\int_{\R} \prod_{i=1}^3 \theta(t_i - s)\,e^{-\varepsilon(t_i - s)}\, ds
= (1/\varepsilon^2)\,e^{-\varepsilon(t_1+t_2+t_3-3s)}|_{s \to -\infty}^{s=t_1}$,
which is finite for $\varepsilon > 0$ and yields a smooth
function on $\Conf_3(\R)$ in the limit; the $\Eone$-algebra
structure absorbs it into the topological homotopy data.

\medskip\noindent\textbf{Step 3: Holomorphic integral.}
The holomorphic integral is the key:
\[
 I_3 \;=\;
 \int_{\C^2}
 K^{\C^2}(z_1,w)\,
 K^{\C^2}(z_2,w)\,
 K^{\C^2}(z_3,w)\,
 dw_1 \wedge dw_2 \wedge d\bar w_1 \wedge d\bar w_2.
\]
By the Bochner--Martinelli calculus on $\C^2$, this is
the convolution of three copies of the fundamental
solution of $\dbar$ on $\C^2$. The Jacobi identity for
the Lie bracket $[\cdot,\cdot]$ on
$\fg \otimes \C[u,v]$ gives
\[
 f^{ab}_e f^{ec}_d + f^{bc}_e f^{ea}_d
 + f^{ca}_e f^{eb}_d \;=\; 0.
\]
The integral $I_3$ is \emph{antisymmetric} in the three
insertions (by the antisymmetry of the
Bochner--Martinelli kernel under cyclic permutation on
$\FM_3(\C^2)$). Combined with the Jacobi identity,
each Lie-algebra component of $m_3$ is a total
$\dbar$-derivative on $\FM_3(\C^2)$. By Stokes'
theorem on the FM compactification, the integral reduces
to boundary faces. But the boundary faces of
$\FM_3(\C^2)$ correspond to $m_2 \circ m_2$
compositions, which are precisely the terms that the
$\Ainf$ relation
$m_1 \circ m_3 + m_2 \circ m_2 + m_3 \circ m_1 = 0$
demands. Hence $m_3$ on cohomology is $Q$-exact: it
vanishes in $H^\bullet$.

The vanishing mechanism is identical to the 3d
Kac--Moody case: the Jacobi identity for $\fg$
eliminates the cohomological obstruction to formality at
degree~$3$. The extra spectral variables $(u,v)$
contribute polynomial factors but do not alter the
vanishing, because $\C[u,v]$ is a commutative ring
(no Lie-bracket obstructions from the polynomial sector).
\end{proof}

\begin{remark}[All higher $m_k$ vanish]
\label{rem:DDCA-formality}
The argument of Theorem~\ref{thm:DDCA-m3} extends to
all degrees: $m_k = 0$ on $H^\bullet$ for $k \ge 3$.
The reason is that the OPE has maximal pole order $d = 2$
with non-scalar residues (the structure constants
$f^{ab}_c$), so the DDCA is class~$\mathbf{L}$ in the
depth classification
(Theorem~\ref{thm:pole-structure-dichotomy}). The
degree-$3$ obstruction $m_3$ is present at chain level but
vanishes on cohomology by the Jacobi identity; all higher
obstructions vanish \emph{a fortiori} because no
higher-order poles are available to seed them.
\end{remark}

\subsubsection{Shadow archetype and Swiss-cheese structure}

\begin{proposition}[DDCA shadow archetype; \ClaimStatusProvedHere]
\label{prop:DDCA-shadow}
The deformed double current algebra
$\cA_\partial = U(\fg \otimes \C[u,v])$ is a
class~$\mathbf{L}$ algebra with shadow depth~$3$:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item The maximal OPE pole order is $d = 2$
 (non-scalar residues from $f^{ab}_c$).
\item $m_3 \ne 0$ at chain level but
 $m_3 = 0$ on cohomology.
\item $m_k = 0$ on cohomology for all $k \ge 3$.
\item The Swiss-cheese structure is formal at genus~$0$.
\end{enumerate}
This places the DDCA in the same shadow class as affine
Kac--Moody $\widehat{\fg}_k$, as expected from its
origin as a two-parameter current algebra.
\end{proposition}

\begin{proof}
Items (i)--(iii) follow from
Theorems~\ref{thm:DDCA-m2} and~\ref{thm:DDCA-m3}.
Item (iv) follows from (iii) by
Theorem~\ref{thm:pole-structure-dichotomy}(ii): class
$\mathbf{L}$ algebras have finite shadow depth ($r_{\max} = 3$; $m_3^{\mathrm{SC}} \neq 0$, $m_k^{\mathrm{SC}} = 0$ for $k \ge 4$).
\end{proof}

\begin{theorem}[Swiss-cheese structure of the 5d HT theory;
\ClaimStatusProvedHere]
\label{thm:DDCA-swiss-cheese}
The 5d holomorphic-topological theory on
$\R \times \C^2$ with $\fg$-gauge symmetry defines a
logarithmic $\SCchtop$-algebra in the sense of
Definition~\textup{\ref{def:log-SC-algebra}}, with the
following structure:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item \textbf{Open-sector category.}
 $\cC_{\mathrm{op}} =
 U(\fg\otimes\C[u,v])\text{-}\mathrm{mod}^{\mathrm{dg}}$,
 the dg-category of modules over the deformed double
 current algebra. This is the primitive datum;
 the boundary algebra
 $\cA_\partial = \End_{\cC_{\mathrm{op}}}(b)$ for a
 compact generator $b$ is a chart.
\item \textbf{Koszul dual.}
 $\cA^!_\partial \simeq
 C^\bullet(\fg\otimes\C[u,v])$,
 the Chevalley--Eilenberg cochains of the double
 current Lie algebra, obtained by Verdier duality on bar
 cohomology \textup{(}not by cobar\textup{)}. The Koszul
 dual lives on the boundary $\R$, not in the
 bulk $\R \times \C^2$.
\item \textbf{Bulk observables.}
 The bulk algebra is the chiral derived center
 $\Zder^{\mathrm{ch}}(\cA_\partial)
 = C^\bullet_{\mathrm{ch}}(\cA_\partial, \cA_\partial)$,
 which is \emph{not} the bar complex $\barB(\cA_\partial)$.
 Bar classifies twisting morphisms; the derived center
 classifies bulk observables.
\item \textbf{Line operators.}
 $\cC_{\mathrm{line}} \simeq
 \cA^!_\partial\text{-}\mathrm{mod}$,
 with spectral braiding controlled by the classical
 $r$-matrix
 \[
 r(u,v) \;=\;
 \frac{k\,\Omega_{\fg}}{u \cdot v}
 \]
 where $\Omega_\fg$ is the Casimir element and
 $(u,v)$ are the spectral variables.
 This is the leading-order approximation; the full
 spectral kernel depends on the differences
 $(u_1 - u_2,\, v_1 - v_2)$ with separate poles in
 each spectral variable.
 The RTT relations reproduce the shifted Yangian
 $Y_\hbar(\fg)$~\cite{DNP2025} of the double
 current algebra.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{(i)} The open-sector category is the dg-category
of boundary conditions for the 5d theory; for the gauge
theory, these are modules for the boundary algebra.
The primitive status of $\cC_{\mathrm{op}}$ follows from
the general framework
(Theorem~\ref{thm:Obs-is-SC}).

\emph{(ii)} The Koszul dual identification
$\cA^! \simeq C^\bullet(\fg_{\mathrm{dbl}})$
is established in Section~\ref{subsec:M2-holography}
via PBW Koszulness of $U(\fg_{\mathrm{dbl}})$.

\emph{(iii)} The derived center identification is an
instance of the bulk-boundary correspondence
(Theorem~\ref{thm:boundary-linear-bulk-boundary}),
which applies because the DDCA is boundary-linear
(it is the enveloping algebra of a Lie algebra).

\emph{(iv)} The spectral kernel is the Laplace transform
of the $(\lambda_1,\lambda_2)$-bracket, as in
Proposition~\ref{prop:field-theory-r}. The double
spectral variable produces a rational $r$-matrix in
$(u,v)$. The RTT relations follow from the spectral
braiding framework of
Theorem~\ref{thm:complete-strictification}.
\end{proof}

\begin{remark}[Curvature at genus $g \ge 1$]
\label{rem:DDCA-genus}
At genus~$g \ge 1$, the curved $E_1$ coalgebra structure
on the bar complex has curvature $\kappa(\cA_\partial) \cdot \omega_g$.
For the DDCA at generic level $k \ne -h^\vee(\fg)$,
the modular characteristic is
\[
 \kappa(\cA_\partial)
 \;=\;
 \frac{\dim(\fg)\,(k + h^\vee(\fg))}{2h^\vee(\fg)},
\]
the same formula as the affine Kac--Moody value
\textup{(}the polynomial extension $\C[u,v]$ does not
contribute an independent central extension at the
conformal level\textup{)}. The
complementarity relation
$\kappa + \kappa^! = 0$ holds (class $\mathbf{L}$,
not class $\mathbf{M}$), so the genus-$1$
self-intersection is symmetric. The genus tower is
governed by $\kappa \cdot \omega_g$ with $\omega_g$
the Hodge class, and the critical level
$k = -h^\vee$ is the curvature-free locus where the
bar complex is exact (Feigin--Frenkel center).
\end{remark}

\begin{remark}[DK-0 bridge for the DDCA]
\label{rem:DDCA-DK0}
The Laplace transform verification
(Proposition~\ref{prop:dk0-laplace-five-families}) extends
to the DDCA: the two-variable $\lambda$-bracket
\eqref{eq:DDCA-bracket} has Laplace transform
\[
 r^{ab}(u,v)
 \;=\;
 \int_0^\infty \int_0^\infty
 e^{-\lambda_1 u - \lambda_2 v}\,
 \{J^a_{1,0}\,{}_{\lambda_1,\lambda_2}\,J^b_{0,1}\}\,
 d\lambda_1\, d\lambda_2
 \;=\;
 \frac{f^{ab}_c}{u\,v}
 \;+\;
 \frac{k\,\delta^{ab}}{u^2\, v^2},
\]
matching the collision residue of the boundary OPE
(with the bar kernel absorbing one pole in each
variable,). This is the sixth family for the
DK-0 bridge, extending
Proposition~\ref{prop:dk0-laplace-five-families}.
\end{remark}

\subsection{The XYZ model: prototypical nonlinear Landau--Ginzburg boundary algebra}
\label{subsec:xyz-model-boundary}
\index{XYZ model|textbf}
\index{Landau--Ginzburg!XYZ boundary algebra}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

The preceding examples (free multiplet, LG-cubic, abelian CS, DDCA)
are gauge-theoretic: the boundary algebra is built from currents.
In the XYZ model the boundary algebra is determined by a
\emph{superpotential constraint}, producing a qualitatively different
$\Ainf$ structure.
This is the simplest Landau--Ginzburg theory with a genuinely
nonlinear superpotential coupling multiple fields, and serves as the
prototypical nonlinear boundary algebra and will reappear as the mirror
of the SQED boundary VOA in the next subsection.

\paragraph{Bulk theory.}
The model has three chiral multiplets $X$, $Y$, $Z$ with cubic
superpotential
\begin{equation}\label{eq:xyz-superpotential}
 W \;=\; XYZ.
\end{equation}
The bulk theory is the derived critical locus $\mathrm{dCrit}(W)$, which
carries a $(-1)$-shifted Poisson bracket inherited from the BV structure.

\paragraph{Boundary algebra.}
\begin{proposition}[XYZ boundary VOA; \ClaimStatusProvedElsewhere]
\label{prop:xyz-model-boundary-voa}
\textup{(Costello--Dimofte--Gaiotto \cite{CDG2023}, \S5.3.2.)}
With Neumann boundary conditions and $W|_{\partial} = 0$, the boundary
vertex algebra is the BRST cohomology of the constraint algebra:
\[
 \cA_\partial^{\mathrm{XYZ}}
 \;=\;
 H^\bullet\!\bigl(\beta\gamma^{\otimes 3},\; Q_W\bigr),
 \qquad
 Q_W = \oint\!\bigl(YZ\,\partial_X + XZ\,\partial_Y + XY\,\partial_Z\bigr).
\]
The constraints $XY = 0$, $YZ = 0$, $ZX = 0$ hold up to homotopy; the
cohomology captures the derived intersection of the constraint loci.
\end{proposition}

\paragraph{$A_\infty$ structure and cubic truncation.}
Since $W = XYZ$ is cubic (degree $d = 3$), the $A_\infty$ structure
on the boundary algebra truncates:
\begin{itemize}
\item $m_2$: arises from the cubic vertex, encoding the OPE of boundary
 fields;
\item $m_k = 0$ for $k \geq 3$: by the degree-counting argument of
 Theorem~\ref{thm:LG_truncation} (or its multi-field generalization),
 the cubic superpotential forces all higher operations to vanish.
 This is the $d = 3$ case of the superpotential truncation.
\end{itemize}

\begin{remark}[Matrix factorization interpretation]
\label{rem:xyz-matrix-factorization}
The XYZ boundary VOA is a vertex algebra of ``matrix factorization''
type: $Q_W^2 = W \cdot \mathrm{id}$, and the boundary condition
$W|_\partial = 0$ makes $Q_W$ into a genuine differential. The
$A_\infty$ category of boundary conditions is equivalent to the dg
category of matrix factorizations of~$W$. This structure will reappear
in the next subsection, where the SQED--XYZ mirror identifies the XYZ
boundary VOA with the gauge-invariant subalgebra of the SQED boundary
theory.
\end{remark}

\subsection{SQED--XYZ duality: mirror symmetry as bar-cobar equivalence}
\label{subsec:sqed-xyz-mirror}

Three-dimensional mirror symmetry exchanges gauge-theoretic and
Landau--Ginzburg descriptions of the same physical system. In the
holomorphic--topological twist, this exchange is visible at the level
of boundary vertex algebras and is controlled by bar-cobar duality.

\paragraph{The SQED side.}
SQED is the $\mathrm{U}(1)$ gauge theory coupled to a single charged
chiral multiplet, placed on a half-space with Neumann boundary
conditions for the gauge field.

\begin{proposition}[SQED boundary VOA; \ClaimStatusProvedElsewhere]
\label{prop:sqed-boundary-voa}
\textup{(Costello--Dimofte--Gaiotto \cite{CDG2023}, \S6.4.)}
The boundary vertex algebra of SQED with Neumann conditions is the
gauge-invariant subalgebra
\[
 \cA_{\partial}^{\mathrm{SQED}}
 \;=\;
 \bigl(\beta\gamma \otimes \widehat{\mathfrak{gl}}_1\bigr)^{\mathrm{U}(1)},
\]
generated by composite operators invariant under the $\mathrm{U}(1)$
gauge action on the charged matter fields.
\end{proposition}

\begin{example}[Explicit generators of the SQED boundary VOA]
\label{ex:sqed-generators}
\index{SQED!boundary generators}
Let $(\beta,\gamma)$ carry $\mathrm{U}(1)$ charges $(+1,-1)$ and let
$J$ be the current of $\widehat{\mathfrak{gl}}_1$. The
gauge-invariant generators of
$\cA_\partial^{\mathrm{SQED}}$ are:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item The current $J(z)$ itself (neutral under the gauge action);
\item The composite $:\!\beta\gamma\!:(z)$ (the ``meson'');
\item The normal-ordered products $:\!\beta\partial^n\gamma\!:$ and
 their descendants (higher mesons).
\end{enumerate}
The graded character of this gauge-invariant subalgebra is
\begin{equation}\label{eq:sqed-character}
\chi_{\mathrm{SQED}}(q)
\;=\;
\oint \frac{da}{2\pi i\,a}\;
\prod_{n \geq 1}
\frac{1}{(1 - q^n\,a)(1 - q^n\,a^{-1})(1 - q^n)},
\end{equation}
where the $a$-integral projects to $\mathrm{U}(1)$-invariants, the
factors $(1-q^n a)^{-1}$ and $(1-q^n a^{-1})^{-1}$ count $\beta$ and
$\gamma$ modes, and $(1-q^n)^{-1}$ counts $J$-modes.
\end{example}

\paragraph{The XYZ side.}
The XYZ model consists of three chiral multiplets $X$, $Y$, $Z$ with
cubic superpotential $W = XYZ$, again on a half-space with Neumann
boundary conditions.

\begin{proposition}[XYZ boundary VOA; \ClaimStatusProvedElsewhere]
\label{prop:xyz-boundary-voa}
\textup{(Costello--Dimofte--Gaiotto \cite{CDG2023}, \S7.6.)}
The boundary vertex algebra of the XYZ model is the derived critical
locus of $W|_{\partial}$:
\[
 \cA_{\partial}^{\mathrm{XYZ}}
 \;=\;
 H^\bullet\bigl(\beta\gamma^{\otimes 3},\; Q_W\bigr),
 \qquad
 Q_W = \oint\!\bigl(Y Z\, \partial_X + X Z\, \partial_Y
 + X Y\, \partial_Z\bigr),
\]
the algebra of operators satisfying the constraint equations
$YZ = 0$, $XZ = 0$, $XY = 0$ up to homotopy.
\end{proposition}

\begin{theorem}[SQED--XYZ mirror symmetry; \ClaimStatusProvedElsewhere]
\label{thm:sqed-xyz-mirror}
\textup{(Costello--Dimofte--Gaiotto \cite{CDG2023}, \S6--7.)}
The SQED and XYZ boundary vertex algebras are isomorphic:
\[
 \cA_{\partial}^{\mathrm{SQED}}
 \;\cong\;
 \cA_{\partial}^{\mathrm{XYZ}}
\]
as vertex algebras. This is the holomorphic-twist avatar of $3$d
mirror symmetry.
\end{theorem}

\begin{remark}[Mirror symmetry as Koszul duality]
\label{rem:mirror-as-koszul}
This mirror symmetry is a bar-cobar equivalence. Both theories
have equivalent bar complexes: their dg coalgebras
$\bar{B}(\cA_{\partial}^{\mathrm{SQED}})$ and
$\bar{B}(\cA_{\partial}^{\mathrm{XYZ}})$ are quasi-isomorphic. So
Theorem~B (bar-cobar inversion) of Volume~I produces the VOA
isomorphism. Mirror symmetry is not an accident: it is Koszul
duality.
\end{remark}

\begin{remark}[Exchange of boundary conditions]
\label{rem:mirror-bc-exchange}
Under the SQED--XYZ duality, boundary conditions exchange:
Neumann~$\leftrightarrow$~Dirichlet. In the bar-cobar framework
this is the exchange $\cA \leftrightarrow \cA^!$; Koszul duality
exchanges the two sides of the bar-cobar adjunction
(Theorem~A of Volume~I). The Neumann algebra of one theory is the
Dirichlet algebra of the mirror, and the exchange is encoded on
the bar side by Verdier duality
$\mathbb{D}_{\Ran}\barB(\cA)\simeq \cA^!_\infty$; the
bar-cobar counit $\Omega \circ B \simeq \id$ only reconstructs
each algebra from its own bar coalgebra.
\end{remark}

\begin{remark}[Mirror symmetry as MC equivalence]
\label{rem:mirror-bar-cobar-explicit}
\index{mirror symmetry!MC equivalence}
\index{Chriss--Ginzburg structure principle!mirror symmetry}
In the Chriss--Ginzburg convolution framework of Volume~I,
mirror symmetry is \emph{MC gauge equivalence} in a strict dg~Lie
chart \textup{(}with the invariant deformation object understood up
to filtered $L_\infty$ quasi-isomorphism, Convention~\ref{conv:vol2-strict-models}\textup{)}:
the master MC
elements $\Theta_{\mathrm{SQED}}$ and $\Theta_{\mathrm{XYZ}}$
in their respective ambient algebras are gauge-equivalent via
\[
 \Theta_{\mathrm{SQED}}
 \;\sim_{\mathrm{MC}}\;
 \Theta_{\mathrm{XYZ}}
 \quad\text{in}\quad
 \operatorname{MC}\!\bigl(\mathfrak{g}^{\mathrm{amb}}\bigr).
\]
The five main theorems of Vol~I appear as projections of this
single equivalence:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item \emph{Theorem~A} ($\Theta\big|_{\hbar=0} = \tau$): the
 SQED bar differential is the gauge BRST differential; the XYZ
 bar differential is the Koszul differential $Q_W$.
\item \emph{Theorem~B} (GK involution): bar-cobar inversion
 recovers both algebras from the shared coalgebra.
\item \emph{Theorem~C} (cross-polarization):
 Neumann~$\leftrightarrow$~Dirichlet is the exchange
 $\cA \leftrightarrow \cA^!$ of the cross-polarized graph sum.
\item \emph{Non-renormalization}: both theories are quasi-linear,
 so $\Theta\big|_{\hbar=0}$ is exact.
\item \emph{Module exchange}: vortex lines of SQED (modules for
 $(\cA^{\mathrm{SQED}})^!$) map to matrix factorizations of XYZ
 (modules for $(\cA^{\mathrm{XYZ}})^!$).
\end{enumerate}
Mirror symmetry is not a coincidence: it is the statement that two
different physical presentations give gauge-equivalent MC elements.
\end{remark}

\subsection{Twisted holography: the Costello--Gaiotto primitive package}
\label{subsec:twisted-holography-primitive}
\index{twisted holography!primitive package|textbf}
\index{Costello--Gaiotto|textbf}

The Costello--Gaiotto programme~\cite{costello-gaiotto,Gai19}

The full primitive package
(Definition~\ref*{V1-def:thqg-completed-platonic-datum}) for
3d holomorphic Chern--Simons on the half-space
$\R_{\ge 0} \times \C$ with gauge algebra $\fg = \mathfrak{gl}_N$
is computed below.

\subsubsection{The 3d holomorphic Chern--Simons theory}

Fix a reductive Lie algebra $\fg$.
The bulk theory is holomorphic Chern--Simons on $\R_t \times \C_z$:

\begin{equation}\label{eq:tw-hol-hcs-action}
S_{\mathrm{HCS}}
\;=\;
\int_{\R \times \C}
dz \wedge \operatorname{Tr}\!\Bigl(
A \wedge (\dbar + d_t) A
\;+\; \tfrac{2}{3}\, A^3
\Bigr),
\end{equation}
where $A \in \Omega^{0,\bullet}(\C) \otimes \Omega^\bullet(\R) \otimes \fg$
carries both Dolbeault and de~Rham degrees. The BRST operator is
$Q = \dbar + d_t$, and the propagator factors as
$G(z,t) = K^{\C}(z) \otimes K^{\R}(t)$
with $K^{\C}(z) = (2\pi i)^{-1}\, d\bar z / \bar z$ (the Cauchy kernel)
and $K^{\R}(t) = \theta(t)$ (Heaviside).

\subsubsection{Boundary algebra: affine Kac--Moody}

Place the boundary at $t = 0$.

\begin{proposition}[Boundary algebra of HCS; \ClaimStatusProvedElsewhere]
\label{prop:tw-hol-boundary-KM}
\textup{(Costello--Gaiotto~\cite{costello-gaiotto}, \S3.)}
The boundary algebra of 3d holomorphic Chern--Simons with
$\fg$-gauge symmetry is the affine Kac--Moody vertex algebra at
level $k$ determined by the Chern--Simons coupling:
\[
\cA_\partial \;=\; V_k(\fg),
\qquad
J^a(z)\, J^b(w) \;\sim\;
\frac{k\, \delta^{ab}}{(z-w)^2}
\;+\;
\frac{f^{ab}_{\phantom{ab}c}\, J^c(w)}{z-w}.
\]
\end{proposition}

The boundary OPE has maximal pole order $2$. Since the
$d\log$ bar kernel absorbs one pole, the collision residue is:
\begin{equation}\label{eq:tw-hol-r-matrix}
r(z)
\;=\;
\Res^{\mathrm{coll}}_{0,2}(\Theta_\cA)
\;=\;
\frac{k\,\Omega_\fg}{z},
\end{equation}
where $\Omega_\fg = \sum_a T^a \otimes T^a$ is the
quadratic Casimir. This is a single simple pole; the double pole
of the OPE has been absorbed by the $d\log(z-w)$ bar kernel.

\subsubsection{The primitive package}

\begin{theorem}[Primitive package for twisted holography; \ClaimStatusProvedHere]
\label{thm:tw-hol-primitive-package}
Let $\fg$ be a simple Lie algebra and $k \ne -h^\vee$. The
twisted holographic system $(\fg, k)$ on $\R_{\ge 0} \times \C$
determines a primitive package:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item \textbf{Open-sector category.}
 $\cC_{\mathrm{op}} = V_k(\fg)\text{-}\mathrm{mod}^{\mathrm{dg}}$,
 the dg-category of modules for the affine Kac--Moody algebra at
 level~$k$.

\item \textbf{Boundary object and algebra.}
 $b = V_k(\fg)$ \textup{(}the vacuum module\textup{)},
 $\cA_\partial = \End_{\cC_{\mathrm{op}}}(b) = V_k(\fg)$.

\item \textbf{Koszul dual.}
 $\cA_\partial^! \simeq V_{-k-2h^\vee}(\fg)$ via a filtered
 quasi-isomorphism on the Koszul locus
 \textup{(}Theorem~\ref*{V1-thm:universal-kac-moody-koszul}\textup{)};
 more precisely, $\cA_\partial^! = \mathrm{CE}^{\mathrm{ch}}(\widehat\fg_{-k-2h^\vee})$
 \textup{(}chiral CE cochains\textup{)}, which shares
 $\kappa(\cA^!) = -\kappa(\cA)$ with $V_{-k-2h^\vee}(\fg)$
 but is categorically distinct \textup{(}cf.\\textup{)}.
 On the associated graded \textup{(}PBW filtration\textup{)},
 $\operatorname{gr}\cA_\partial^! \simeq C^\bullet(\fg[t^{-1}]t^{-1})$.
 The Koszul involution exchanges Neumann~$\leftrightarrow$~Dirichlet
 boundary conditions.

\item \textbf{Derived center \textup{(}universal bulk\textup{)}.}
 $\Zder^{\mathrm{ch}}(\cA_\partial)
 = C^\bullet_{\mathrm{ch}}(V_k(\fg),\, V_k(\fg))$.
 The bulk observables are chiral Hochschild cochains of the boundary
 algebra, not the bar complex.

\item \textbf{Collision residue.}
 $r(z) = k\,\Omega_\fg / z$, satisfying the classical Yang--Baxter
 equation. The RTT quantization produces the Yangian $Y_\hbar(\fg)$.

\item \textbf{Modular MC element.}
 $\Theta_\cA = D_\cA - d_0 \in \MC(\gAmod)$, with modular
 characteristic $\kappa = \kappa(V_k(\fg))
 = \dim(\fg)\cdot(k + h^\vee)/(2h^\vee)$ and complementarity
 $\kappa(\cA_\partial) + \kappa(\cA_\partial^!) = 0$
 \textup{(}since $\cA^! = V_{-k-2h^\vee}(\fg)$ has
 $\kappa(\cA^!) = -\dim(\fg)\cdot(k+h^\vee)/(2h^\vee)$, by the
 Feigin--Frenkel involution $k \mapsto -k-2h^\vee$\textup{)}.

\item \textbf{Annulus trace.}
 $\operatorname{Tr}_\cA \simeq HH_*(V_k(\fg))$: the
 Hochschild homology of the boundary algebra, computing the
 first modular shadow of the open sector.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{(i)--(ii).}
The open-sector category is the category of boundary
conditions for HCS; for a gauge theory with Neumann
conditions, this is the module category of the boundary
algebra~\cite{costello-gaiotto}. The vacuum module generates
the category, giving the boundary chart.

\emph{(iii).}
The affine Kac--Moody algebra $V_k(\fg)$ is chirally Koszul
(Volume~I, Proposition~\ref*{V1-prop:pbw-universality}): bar cohomology
concentrates in bar degree~$1$. The Verdier dual
$\mathbb{D}_{\mathrm{Ran}}(\barB(V_k(\fg))) \simeq
\barB(V_{-k-2h^\vee}(\fg))$ recovers the Koszul dual at the
Feigin--Frenkel dual level
(Theorem~\ref*{V1-thm:universal-kac-moody-koszul}). On the
associated graded, this is the chiral
$\mathrm{Com}^! = \mathrm{Lie}$ identification.

\emph{(iv).}
The bulk-boundary correspondence
(Theorem~\ref{thm:boundary-linear-bulk-boundary})
applies because $V_k(\fg)$ is boundary-linear.

\emph{(v).}
The $\lambda$-bracket
$\{J^a_\lambda J^b\} = f^{ab}_c J^c + k\, \delta^{ab}\, \lambda$
has Laplace transform
$r^{ab}(z) = \int_0^\infty e^{-\lambda z}\,
(f^{ab}_c J^c + k\, \delta^{ab}\, \lambda)\, d\lambda
= f^{ab}_c J^c / z + k\, \delta^{ab} / z^2$.
The collision residue, after bar-kernel absorption,
extracts pole orders one less: $r_{\mathrm{coll}}(z)
= k\,\Omega_\fg / z$. The CYBE follows from the MC equation at
genus~$0$, degree~$3$
(Volume~I, Theorem~\ref*{V1-thm:thqg-g5-yangian}).

\emph{(vi).}
The modular characteristic
$\kappa(V_k(\fg)) = \dim(\fg)(k + h^\vee)/(2h^\vee)$
is proved in Volume~I
(Theorem~\ref*{V1-thm:genus-universality},
Proposition~\ref{prop:dk0-laplace-five-families}).
The complementarity $\kappa + \kappa^! = 0$ follows from
the Feigin--Frenkel involution.

\emph{(vii).}
The annulus trace is the standard Hochschild homology of
$V_k(\fg)$; its identification with the first modular shadow
follows from the annulus trace theorem
(Volume~I, Theorem~\ref*{V1-thm:thqg-annulus-trace}).
\end{proof}

\begin{proposition}[Shadow archetype of HCS; \ClaimStatusProvedHere]
\label{prop:tw-hol-shadow-archetype}
The affine Kac--Moody algebra $V_k(\fg)$ is class~$\mathbf{L}$
\textup{(}Lie/tree\textup{)} with shadow depth $r_{\max} = 3$:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item The quadratic shadow is
 $\kappa = \dim(\fg)(k + h^\vee)/(2h^\vee)$.
\item The cubic shadow is nonzero at chain level
 \textup{(}from the Lie bracket $f^{ab}_c$\textup{)}, but
 vanishes on cohomology by the Jacobi identity.
\item All higher $m_k^{\mathrm{SC}}$ vanish on cohomology.
\item The shadow metric is $Q_L(t) = (2\kappa + 3\alpha t)^2$
 with $\alpha = f^{ab}_c$ data; the discriminant
 $\Delta = 8\kappa S_4 = 0$ confirms class~$\mathbf{L}$.
\end{enumerate}
\end{proposition}

\begin{proof}
Items (i)--(iii) follow from the pole-structure dichotomy
(Theorem~\ref{thm:pole-structure-dichotomy}): maximal pole order
$d = 2$ with non-scalar leading residue places $V_k(\fg)$ in
class~$\mathbf{L}$. Item (iv): the quartic contact invariant $S_4$ vanishes
for current algebras by the Jacobi identity (the degree-$4$
obstruction class is killed by the same mechanism that kills
$m_3$ on cohomology), so $\Delta = 8\kappa \cdot 0 = 0$.
This is verified for all simple types in Volume~I
(Table~\ref*{V1-tab:exceptional-shadow-data}).
\end{proof}

\subsection{Twisted M2-brane: the deformed double current algebra}
\label{subsec:twisted-M2-primitive}
\index{M2-brane!primitive package|textbf}
\index{deformed double current algebra!primitive package|textbf}

The M2-brane boundary algebra~\cite{Costello-1705.02500v1}
is the deformed double current algebra (DDCA), computed

subsection assembles the primitive package and

\subsubsection{Primitive package for $K$ M2-branes}

Fix a positive integer $K$ and a non-critical level $k$.
The 5d holomorphic Chern--Simons theory on
$\R_t \times \C^2_{z_1,z_2}$ with gauge algebra
$\fg = \mathfrak{gl}_K$ produces a boundary algebra at $t = 0$.

\begin{theorem}[Primitive package for the M2-brane; \ClaimStatusProvedHere]
\label{thm:twisted-M2-primitive-package}
The twisted M2-brane system with $K$ branes determines a
primitive package:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item \textbf{Open-sector category.}
 $\cC_{\mathrm{op}} = \mathrm{DDCA}_{k}(\mathfrak{gl}_K)\text{-}\mathrm{mod}^{\mathrm{dg}}$.

\item \textbf{Boundary algebra.}
 $\cA_\partial = \mathrm{DDCA}_k(\mathfrak{gl}_K)
 = U(\mathfrak{gl}_K \otimes \C[u,v])$,
 the deformed double current algebra. Its generators
 $J^a_{m,n}$ with $a \in \mathfrak{gl}_K$, $m,n \ge 0$
 satisfy the binary product of Theorem~\ref{thm:DDCA-m2}:
 \[
 m_2(J^a_{m,n},\; J^b_{p,q})
 \;=\;
 f^{ab}_c\, J^c_{m+p,n+q}
 \;+\;
 k\,\delta^{ab}\,
 P_2(m,n,p,q;\lambda_1,\lambda_2),
 \]
 where $P_2$ is the polynomial central extension
 \eqref{eq:DDCA-central-term}.

\item \textbf{Koszul dual.}
 $\cA_\partial^!
 \simeq C^\bullet(\mathfrak{gl}_K \otimes \C[u,v])$,
 the Chevalley--Eilenberg cochains of the double current
 Lie algebra \textup{(}Section~\ref{subsec:M2-holography}\textup{)}.
 This is the 11d twisted supergravity algebra in the
 $K$-brane sector: Koszul duality exchanges
 boundary~$\leftrightarrow$~bulk.

\item \textbf{Derived center.}
 $\Zder^{\mathrm{ch}}(\cA_\partial)
 = C^\bullet_{\mathrm{ch}}(\mathrm{DDCA}_k,\, \mathrm{DDCA}_k)$:
 the bulk observables of the 5d theory, computed as chiral
 Hochschild cochains of the boundary algebra
 \textup{(}Theorem~\ref{thm:DDCA-swiss-cheese}(iii)\textup{)}.

\item \textbf{Collision residue.}
 The two-variable $r$-matrix
 (Remark~\ref{rem:DDCA-DK0}) is
 \begin{equation}\label{eq:M2-r-matrix}
 r(u_1-u_2,\, v_1-v_2)
 \;=\;
 \frac{k\,\Omega_{\mathfrak{gl}_K}}{(u_1-u_2)(v_1-v_2)},
 \end{equation}
 where $\Omega_{\mathfrak{gl}_K}$ is the Casimir element.
 This is the classical limit of the RTT $R$-matrix
 generating the shifted Yangian
 $Y_\hbar(\mathfrak{gl}_K)$~\cite{DNP2025,Guay2007}.

\item \textbf{Modular MC element.}
 $\Theta_\cA \in \MC(\gAmod)$ with
 $\kappa(\mathrm{DDCA}_k(\mathfrak{gl}_K)) = K(k + K)/2$
 \textup{(}since $\dim(\mathfrak{gl}_K) = K^2$ and
 $h^\vee(\mathfrak{gl}_K) = K$, the formula
 $\kappa = \dim(\fg)(k+h^\vee)/(2h^\vee)$ gives
 $K^2(k+K)/(2K)$\textup{)}. Complementarity:
 $\kappa + \kappa^! = 0$ \textup{(}class~$\mathbf{L}$\textup{)}.

\item \textbf{Annulus trace.}
 $\operatorname{Tr}_\cA \simeq HH_*(\mathrm{DDCA}_k(\mathfrak{gl}_K))$.
\end{enumerate}
\end{theorem}

\begin{proof}
Items (i)--(iv) and (vii) are established in
Section~\ref{subsec:DDCA-ainfty} and
Theorem~\ref{thm:DDCA-swiss-cheese}.
Item (v) follows from the DK-0 Laplace verification
(Remark~\ref{rem:DDCA-DK0}): the two-variable $\lambda$-bracket
\eqref{eq:DDCA-bracket} has Laplace transform matching the
collision residue, with bar-kernel absorption in both spectral
variables (bar-kernel absorption applied in both spectral variables).
Item (vi): the modular characteristic generalizes the affine
Kac--Moody value; the proof of
$\kappa(\mathrm{DDCA}_k(\mathfrak{gl}_K)) = K(k + K)/2$ is the
same computation as for $V_k(\mathfrak{gl}_K)$ because the
polynomial extension $\C[u,v]$ does not contribute to the
genus-$1$ anomaly (it is commutative and carries no independent
central extension at the conformal level).
\end{proof}

\subsubsection{The M2--M5 coupling as twisting morphism}

The M2-brane algebra $\cA_{\mathrm{M2}} = \mathrm{DDCA}_k(\mathfrak{gl}_K)$
couples to the M5-brane algebra
$\cA_{\mathrm{M5}} = \mathcal{W}_\infty(K)$
(Section~\ref{subsec:twisted-M5-primitive}).
The coupling is controlled by a twisting morphism.

\begin{conjecture}[M2--M5 coupling; \ClaimStatusConjectured]
\label{conj:M2-M5-coupling}
\index{M2-M5 coupling!twisting morphism}
There exists a twisting morphism
\[
\tau_{\mathrm{M2\text{-}M5}}
\;\in\;
\operatorname{Tw}\!\bigl(
\barB(\cA_{\mathrm{M2}}),\;
\cA_{\mathrm{M5}}
\bigr)
\]
encoding the coupling of $K$ M2-branes to M5-branes.
At tree level \textup{(}genus~$0$\textup{)}, this twisting
morphism restricts to the Miura transformation
$\mathfrak{gl}_K \otimes \C[u,v] \to \mathcal{W}_\infty(K)$
sending double currents to $\mathcal{W}$-algebra generators.
At genus~$g \ge 1$, the twisting morphism acquires curvature
$\kappa_{\mathrm{M2}} \cdot \omega_g$ from the Hodge bundle.
\end{conjecture}

\begin{remark}[Status of the M2--M5 coupling]
\label{rem:M2-M5-status}
The tree-level identification (genus~$0$) is proved: the Miura
map from the Heisenberg subalgebra of $\mathrm{DDCA}_k$ to
$\mathcal{W}_\infty(K)$ is a morphism of vertex algebras, and
its compatibility with the bar differential follows from
$m_3 = 0$ on cohomology
(Theorem~\ref{thm:DDCA-m3}). The higher-genus statement is
conjectural: it requires the twisting morphism to extend to the
curved (genus~$\ge 1$) bar complex, which depends on the
HS-sewing criterion for the DDCA. The HS-sewing criterion
holds for $\mathrm{DDCA}_k(\mathfrak{gl}_K)$
(Volume~I, Theorem~\ref*{V1-thm:general-hs-sewing}: polynomial OPE growth
and subexponential sector growth), so the obstruction is not
convergence but rather the explicit identification of the
curved twisting morphism with the physical M2--M5 coupling at
higher genus.
\end{remark}

\begin{remark}[Large-$K$ holography]
\label{rem:M2-large-K}
At large $K$, the holographic dictionary reads
$\cA_{\partial,\infty}^! \cong \cA_{\mathrm{bulk}}$
sector by sector (Section~\ref{subsec:M2-holography}).
The primitive package of
Theorem~\ref{thm:twisted-M2-primitive-package} is the
finite-$K$ refinement of this large-$K$ limit. The shadow
depth classification (class~$\mathbf{L}$,
Proposition~\ref{prop:DDCA-shadow}) is independent of $K$:
the double current algebra has tree-level shadow depth for
all $K$, because the Jacobi identity kills higher operations
at all ranks.
\end{remark}

\subsection{Twisted M5-brane: the matrix \texorpdfstring{$\mathcal{W}_\infty$}{W-infinity} algebra}
\label{subsec:twisted-M5-primitive}
\index{M5-brane!primitive package|textbf}
\index{W-infinity algebra!primitive package|textbf}
\index{matrix W-algebra|textbf}

The M5-brane boundary algebra is the matrix $\mathcal{W}_\infty$
algebra $\mathcal{W}_\infty(K)$, the large-$N$ limit of the
principal $\mathcal{W}$-algebras $\mathcal{W}_N(\mathfrak{gl}_K)$
obtained by quantum Drinfeld--Sokolov reduction. This is the
dual face of the M2--M5 system: where the M2-brane sees a
current algebra (class~$\mathbf{L}$), the M5-brane sees a
$\mathcal{W}$-algebra (class~$\mathbf{M}$).

\subsubsection{The algebra \texorpdfstring{$\mathcal{W}_\infty(K)$}{W-infinity(K)}}

\begin{definition}[$\mathcal{W}_\infty(K)$; following~\cite{ProchazkRapcak2018}]
\label{def:W-infinity-K}
For each positive integer $K$, the \emph{matrix $\mathcal{W}_\infty$ algebra}
is the inverse limit
\[
\mathcal{W}_\infty(K)
\;:=\;
\varprojlim_{N \to \infty}\;
\mathcal{W}_N(\mathfrak{gl}_K),
\]
where $\mathcal{W}_N(\mathfrak{gl}_K) = H^0_{\mathrm{DS}}(V_k(\mathfrak{gl}_{NK}),
f_{\mathrm{prin},N})$ is the principal $\mathcal{W}$-algebra obtained
by quantum Drinfeld--Sokolov reduction of $V_k(\mathfrak{gl}_{NK})$
with respect to the principal nilpotent
$f_{\mathrm{prin},N} \in \mathfrak{sl}_N \subset \mathfrak{gl}_{NK}$.
\end{definition}

\paragraph{Generators.}
For finite $N$, $\mathcal{W}_N(\mathfrak{gl}_K)$ is freely strongly
generated by $K^2 \cdot N$ fields
$\{W^{(s)}_{a\bar b} : s = 1, \ldots, N,\; a,\bar b = 1, \ldots, K\}$
of conformal weights $s = 1, 2, \ldots, N$. In the $N \to \infty$ limit,
the algebra acquires generators $W^{(s)}_{a\bar b}$ of all spins
$s \ge 1$, with $a, \bar b$ running over $\mathfrak{gl}_K$ matrix indices.
At $K = 1$, this reduces to the scalar $\mathcal{W}_{1+\infty}$ algebra.

\paragraph{The matrix Miura transformation.}
The embedding of $\mathcal{W}_\infty(K)$ into a product of $K$ copies of the
Heisenberg algebra (the ``free-field realization'') is controlled by the
matrix Miura operator:
\begin{equation}\label{eq:matrix-miura}
L(z)
\;=\;
(\partial + J_1(z))\,(\partial + J_2(z))\, \cdots\, (\partial + J_K(z)),
\end{equation}
where each $J_i(z)$ is a Heisenberg field of level $k_i$. The
$\mathcal{W}$-algebra generators are extracted as the coefficients of
$L(z) = \partial^K + W^{(2)}(z)\, \partial^{K-2} + \cdots + W^{(K)}(z)$.
The Miura transformation is an injective vertex algebra homomorphism
$\mathcal{W}_\infty(K) \hookrightarrow
\mathcal{H}_{k_1} \otimes \cdots \otimes \mathcal{H}_{k_K}$
whose image consists of the operators invariant under permutations of the
Heisenberg factors (for generic levels).

\subsubsection{Primitive package for $K$ M5-branes}

\begin{conjecture}[Primitive package for the M5-brane; \ClaimStatusConjectured]
\label{conj:twisted-M5-primitive-package}
The twisted M5-brane system with $K$ branes determines a
primitive package:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item \textbf{Open-sector category.}
 $\cC_{\mathrm{op}} = \mathcal{W}_\infty(K)\text{-}\mathrm{mod}^{\mathrm{dg}}$,
 the dg-category of modules for the matrix
 $\mathcal{W}_\infty$ algebra.

\item \textbf{Boundary algebra.}
 $\cA_\partial = \mathcal{W}_\infty(K)$,
 with generators $W^{(s)}_{a\bar b}$ for $s \ge 1$,
 $a, \bar b \in \{1, \ldots, K\}$.

\item \textbf{Koszul dual.}
 For finite $N$:
 $\mathcal{W}_N(\mathfrak{gl}_K)^!
 = \mathcal{W}_{N'}(\mathfrak{gl}_K)$
 with dual level determined by the Feigin--Frenkel involution
 applied through DS reduction. At $K = 1$:
 $\mathcal{W}_N^! = \mathcal{W}_N$ at the dual level
 $k' = -N - k$ \textup{(}self-Koszul-duality up to level shift;
 the complementarity sum $\kappa + \kappa'$ is nonzero
 in general: for $N = 2$, $\kappa + \kappa' = 13$;
 for $N = 3$, $\kappa + \kappa' = 250/3$\textup{)}.

\item \textbf{Derived center.}
 $\Zder^{\mathrm{ch}}(\cA_\partial)
 = C^\bullet_{\mathrm{ch}}(\mathcal{W}_\infty(K),\;
 \mathcal{W}_\infty(K))$.

\item \textbf{Collision residue.}
 For $K = 1$, the Virasoro subalgebra has OPE with poles up to
 order~$4$. Since the $d\log$ bar kernel absorbs one pole order, the collision residue has poles up to order~$3$:
 \[
 r_{\mathrm{Vir}}(z)
 \;=\;
 \frac{c/2}{z^3} \;+\; \frac{2T}{z}.
 \]
 For general $K$, the $r$-matrix is a $K^2 \times K^2$ matrix-valued
 meromorphic function with poles of order up to $2s-1$ from the
 spin-$s$ generators.

\item \textbf{Modular MC element.}
 For $K = 1$ and finite $N$: $\kappa(\mathcal{W}_N)
 = c(\mathcal{W}_N) \cdot (H_N - 1)$, where $c(\mathcal{W}_N)$ is the
 Fateev--Lukyanov central charge of the principal
 $\mathcal{W}_N$-algebra at level $k$~\cite{FL88} and
 $H_N = \sum_{m=1}^N 1/m$ is the $N$th harmonic number.
 For $N = 2$ \textup{(}Virasoro\textup{)}: $H_2 - 1 = 1/2$,
 so $\kappa = c/2$.
 For $N = 3$ \textup{(}$\mathcal{W}_3$\textup{)}: $H_3 - 1 = 5/6$,
 so $\kappa = 5c/6$.

\item \textbf{Annulus trace.}
 $\operatorname{Tr}_\cA \simeq HH_*(\mathcal{W}_\infty(K))$.
\end{enumerate}
\end{conjecture}

\begin{remark}[Status of the M5-brane package]
\label{rem:M5-status}
Items (i), (ii), (iv), and (vii) are structural
consequences of the general framework. The substantive content is:

\emph{Item (iii):} The Koszul dual of $\mathcal{W}_N(\fg, f_{\mathrm{prin}})$
at the principal nilpotent is controlled by the DS reduction of the
affine Koszul dual. For $K = 1$ (scalar $\mathcal{W}_N$ algebras),
the principal $\mathcal{W}$-algebra inherits Koszul self-duality
$\mathcal{W}_N^! \cong \mathcal{W}_N$ at the dual level from the
affine lineage. However, $\kappa + \kappa^! \ne 0$ in general for
$\mathcal{W}$-algebras: the DS reduction changes the anomaly ratio
$\varrho$ from $1/2$ (Kac--Moody) to $H_N - 1$
(Volume~I, landscape census), so the complementarity sum is
$\kappa + \kappa' = \varrho \cdot K_N$ where $K_N = 4N^3 - 2N - 2$.
For $K > 1$ (matrix $\mathcal{W}$-algebras),
the Koszul dual identification is conjectural and depends on the
non-principal DS reduction programme
(Section~\ref{subsec:KS-defect}).

\emph{Item (v):} The $r$-matrix computation requires the
full OPE of $\mathcal{W}_\infty(K)$. At $K = 1$, the
Virasoro $r$-matrix is proved
(Proposition~\ref{prop:dk0-laplace-five-families}).
At $K > 1$, the matrix $r$-matrix is determined in
principle by the Miura transformation
\eqref{eq:matrix-miura}, but the explicit formula is not
computed here.

\emph{Item (vi):} The central charge formula is standard for principal
$\mathcal{W}$-algebras~\cite{FL88}. The modular characteristic
$\kappa = c \cdot (H_N - 1)$ follows from the genus universality theorem
\textup{(}Volume~I, Theorem~\textup{\ref*{V1-thm:genus-universality}}\textup{)}.
For $N = 2$: $\kappa = c/2$; for $N = 3$: $\kappa = 5c/6$.
The $N \to \infty$ limit of the anomaly ratio $H_N - 1$ diverges
\textup{(}harmonic series\textup{)}; the modular characteristic of
$\mathcal{W}_\infty(K)$ requires the limiting central charge and the
full generator spectrum.
\end{remark}

\subsubsection{Shadow archetype and the infinite tower}

\begin{proposition}[Shadow archetype of $\mathcal{W}_\infty(K)$; \ClaimStatusProvedHere]
\label{prop:M5-shadow-archetype}
For $K = 1$, $\mathcal{W}_\infty = \mathcal{W}_{1+\infty}$ is
class~$\mathbf{M}$ \textup{(}mixed, $r_{\max} = \infty$\textup{)}.
For all $K \ge 1$, $\mathcal{W}_\infty(K)$ is class~$\mathbf{M}$:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item The spin-$2$ generator \textup{(}Virasoro\textup{)} creates
 poles of order~$4$ in the OPE, exceeding the class~$\mathbf{C}$
 threshold.
\item The Swiss-cheese operations $m_k^{\mathrm{SC}} \ne 0$ for
 all $k \ge 3$ \textup{(}non-formality\textup{)}.
\item The shadow obstruction tower does not terminate: the discriminant
 $\Delta = 8\kappa S_4 \ne 0$ because $S_4 \ne 0$ for
 $\mathcal{W}$-algebras with $c \ne 0$.
\item The shadow growth rate $\rho(\mathcal{W}_\infty(K))$
 is well-defined and positive, with the shadow obstruction tower
 converging for sufficiently large central charge
 (Volume~I, Theorem~\ref*{V1-thm:shadow-radius}).
\end{enumerate}
\end{proposition}

\begin{proof}
Items (i)--(iii) follow from the pole-structure
dichotomy
(Theorem~\ref{thm:pole-structure-dichotomy}):
the spin-$2$ OPE has a fourth-order pole,
placing $\mathcal{W}_\infty(K)$ in class~$\mathbf{M}$.
The non-vanishing of $S_4$ for $\mathcal{W}$-algebras
is computed in Volume~I:
$S_4^{\mathrm{Vir}} = 10/[c(5c + 22)]$
for the Virasoro subalgebra.
Item (iv): the shadow radius theorem gives
$\rho = \sqrt{9\alpha^2 + 2\Delta}/(2|\kappa|)$;
since $\Delta \ne 0$, the growth rate is positive,
and convergence holds for $\rho < 1$, i.e.\
for $c$ above the critical cubic threshold
$c^* \approx 6.125$.
\end{proof}

\subsubsection{M2--M5 duality as bar-cobar duality}

The central structural claim of the twisted M-theory programme
is that the M2--M5 system realizes bar-cobar duality.

\begin{conjecture}[M2--M5 bar-cobar duality; \ClaimStatusConjectured]
\label{conj:M2-M5-duality}
\index{M2-M5 duality!bar-cobar}
At each rank $K$ and non-critical level, the M2-brane algebra
$\cA_{\mathrm{M2}} = \mathrm{DDCA}_k(\mathfrak{gl}_K)$
\textup{(}class~$\mathbf{L}$\textup{)} and the M5-brane algebra
$\cA_{\mathrm{M5}} = \mathcal{W}_\infty(K)$
\textup{(}class~$\mathbf{M}$\textup{)} are related by:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item The M2-brane algebra acts on the M5-brane algebra via the
 Miura transformation, which is a twisting morphism
 $\tau \in \operatorname{Tw}(\barB(\cA_{\mathrm{M2}}), \cA_{\mathrm{M5}})$.

\item The line-operator category of the M2-brane system is
 controlled by the Yangian $Y_\hbar(\mathfrak{gl}_K)$; the
 line-operator category of the M5-brane system is controlled by
 the $\mathcal{W}$-algebra $\mathcal{W}_\infty(K)$ acting on its
 own modules. The M2--M5 coupling identifies modules in the two
 categories.

\item The modular MC elements satisfy
 $\Theta_{\mathrm{M2}} = \pi_{\mathrm{M2}}(\Theta_{\mathrm{M\text{-}theory}})$
 and
 $\Theta_{\mathrm{M5}} = \pi_{\mathrm{M5}}(\Theta_{\mathrm{M\text{-}theory}})$,
 where $\Theta_{\mathrm{M\text{-}theory}}$ is a single MC element
 in the ambient Swiss-cheese algebra of the 11d twisted theory, and
 $\pi_{\mathrm{M2}}, \pi_{\mathrm{M5}}$ are projections to the two
 boundary sectors.

\item At the scalar shadow level:
 $\kappa(\cA_{\mathrm{M2}}) = K(k + K)/2$,
 $\kappa(\cA_{\mathrm{M5}}) = c(\mathcal{W}_\infty(K)) \cdot (H_N - 1)$
 for the $N$th truncation.
 These are not Koszul duals of each other in general, but they
 are projections of the same M-theoretic MC element to different
 boundary sectors.
\end{enumerate}
\end{conjecture}

\begin{remark}[What is proved vs.\ conjectural]
\label{rem:M2-M5-proved-vs-conjectural}
Item (i) at tree level (genus~$0$) is proved: the Miura
transformation is a vertex algebra homomorphism, and
$m_3 = 0$ on DDCA cohomology
(Theorem~\ref{thm:DDCA-m3}) ensures compatibility with
the bar differential. Items (ii)--(iv) are conjectural:
they require identifying the physical M2--M5 coupling at
all genera with a specific MC element in the modular
convolution algebra. The key missing ingredient is the
explicit construction of $\Theta_{\mathrm{M\text{-}theory}}$
in the 11d twisted theory.

The shadow depth asymmetry is a structural prediction:
the M2-brane sees class~$\mathbf{L}$ (finite tower, tree-level
shadows), while the M5-brane sees class~$\mathbf{M}$ (infinite
tower). The DS reduction $V_k(\mathfrak{gl}_{NK}) \to
\mathcal{W}_N(\mathfrak{gl}_K)$ takes class~$\mathbf{L}$ to
class~$\mathbf{M}$: this is the shadow-depth increase under DS
reduction. The ``extra complexity'' of the M5-brane is not
a new phenomenon but a consequence of the DS functor
introducing higher Swiss-cheese operations
(Theorem~\ref{thm:ds-koszul-obstruction}).
\end{remark}

\label{rem:three-package-comparison}

\subsection{Benchmark global example: \texorpdfstring{$V_k(\mathfrak{sl}_2)$}{V\_k(sl\_2)}
 on \texorpdfstring{$(\mathbb{P}^1, \{0,\infty\})$}{(P\^1, \{0, infty\})}}
\label{subsec:benchmark-global-sl2}
\index{benchmark global example|textbf}
\index{Kac--Moody!benchmark global example|textbf}
\index{projective line!benchmark example|textbf}

The preceding examples compute local data at a single puncture
on the half-space $\R_{\ge 0} \times \C$. The present section
is the first fully global computation in this monograph: we take
$X = \PP^1$, the marked divisor $D = \{0, \infty\}$, tangential
basepoints $\tau = (\tau_0, \tau_\infty)$ at each puncture, and
the boundary algebra $\cA = V_k(\mathfrak{sl}_2)$ at level
$k \ne -2$ (i.e.\ away from the critical level $k = -h^\vee$).
Every object in the completed modular Koszul datum
(Definition~\ref*{V1-def:thqg-completed-platonic-datum})
is computed explicitly.

The example is the simplest nontrivial test of the global
architecture: the curve $\PP^1$ has genus~$0$; the algebra
$\mathfrak{sl}_2$ has rank~$1$, dimension~$3$, and dual Coxeter
number $h^\vee = 2$. Yet the full open/closed package has genuine
content: two boundary circles, a factorization category on
$\PP^1 \setminus \{0,\infty\}$, a nontrivial $r$-matrix, and
at integrable levels a finite-dimensional semisimple Verlinde
algebra as the bulk.

\subsubsection{The tangential log curve}
\label{subsubsec:benchmark-tangential-log-curve}

Let $X = \PP^1$ and $D = \{0, \infty\} \subset \PP^1$.
The real oriented blowup
(Definition~\ref{def:bordified-curve})
\[
 \pi \colon \widetilde{\PP}^1_D
 \;\longrightarrow\;
 \PP^1
\]
replaces each marked point by a circle of real tangent directions:
$\pi^{-1}(0) = S^1_0$ and $\pi^{-1}(\infty) = S^1_\infty$.
The resulting smooth manifold with boundary is
\[
 \widetilde{\PP}^1_D
 \;\cong\;
 S^2 \setminus (D^2_0 \sqcup D^2_\infty)
 \;\cong\;
 [0,1] \times S^1,
\]
a cylinder with boundary $\partial \widetilde{\PP}^1_D
= S^1_0 \sqcup S^1_\infty$.

Fix tangential basepoints $\tau_0 \in S^1_0$ (the positive-real
direction at $z = 0$) and $\tau_\infty \in S^1_\infty$ (the
negative-real direction at $w = 0$ in the chart $w = 1/z$ at
$\infty$). Each punctured boundary circle gives an open interval:
\begin{equation}\label{eq:benchmark-boundary-intervals}
 I_0 := S^1_0 \setminus \{\tau_0\} \cong \R,
 \qquad
 I_\infty := S^1_\infty \setminus \{\tau_\infty\} \cong \R.
\end{equation}
These are the two time axes. The data
$(\PP^1, \{0,\infty\}, \tau)$ is a pointed curve with tangential
basepoints.

\begin{remark}[Geometric content of the tangential data]
\label{rem:benchmark-tangential-content}
The tangential basepoint $\tau_p$ at a puncture $p$ breaks the
$S^1$ rotational symmetry, producing a contractible interval
$I_p \cong \R$ on which the $E_1$-structure (associative,
time-ordered products) is defined. Without $\tau_p$, the boundary
is a circle and the algebraic structure is $S^1$-equivariant
(the trace/clutching direction). Restoring $\tau_p$, i.e.\
gluing the interval back into a circle, is the clutching
operation that produces modular invariants
(Definition~\ref{def:oc-factorization-category}(vi)).
\end{remark}


\subsubsection{The mixed configuration spaces}
\label{subsubsec:benchmark-mixed-config}

The mixed open/closed configuration space
(Definition~\ref{def:mixed-config}) on $(\PP^1, \{0,\infty\}, \tau)$
is
\begin{equation}\label{eq:benchmark-mixed-config}
 \Conf^{\mathrm{oc}}_{I,(m_0,m_\infty)}(\PP^1, D, \tau)
 \;=\;
 \Conf_I(\PP^1 \setminus D)
 \;\times\;
 \Conf^{<}_{m_0}(\R)
 \;\times\;
 \Conf^{<}_{m_\infty}(\R),
\end{equation}
where $\Conf_I(\PP^1 \setminus D)$ parametrises $|I|$ labelled
bulk points in $\PP^1 \setminus \{0,\infty\} \cong \C^\times$,
and $\Conf^{<}_{m_p}(\R) = \{t_1 < \cdots < t_{m_p}\}$
parametrises $m_p$ time-ordered boundary insertions on $I_p$.

The mixed Ran space is the colimit
\[
 \Ran^{\mathrm{oc}}(\PP^1, D, \tau)
 \;=\;
 \varinjlim_{(I,m_0,m_\infty)}
 \Conf^{\mathrm{oc}}_{I,(m_0,m_\infty)}(\PP^1, D, \tau)
 \;/\; \Sigma_I.
\]
A point of $\Ran^{\mathrm{oc}}$ is a finite configuration
consisting of: (a)~an unordered collection of bulk points in
$\C^\times$, (b)~a finite ordered tuple of boundary insertions
on $I_0$, and (c)~a finite ordered tuple of boundary insertions
on $I_\infty$.


\subsubsection{The open sector: generators and relations for
 \texorpdfstring{$\cC_{\mathrm{op}}$}{C\_op}}
\label{subsubsec:benchmark-open-sector}
\index{open-sector category!generators and relations|textbf}
\index{factorization dg-category!SL(2) Chern--Simons|textbf}

The primitive object of the open/closed programme is not the
boundary algebra $V_k(\mathfrak{sl}_2)$, which is a chart,
but the open-sector factorization dg-category
$\cC_{\mathrm{op}}$ on the tangential log curve
$(\PP^1, \{0,\infty\}, \tau)$. This subsubsection gives its
formal generators-and-relations presentation, the first complete
such construction in the monograph.

\begin{definition}[Open-sector factorization dg-category for
 $\mathrm{SL}(2)$ Chern--Simons at level $k$]
\label{def:benchmark-Cop-sl2}
\index{open-sector category!SL(2)|textbf}
Let $\fg = \mathfrak{sl}_2$, $h^\vee = 2$, and
$k \in \C \setminus \{-2\}$. The \emph{open-sector
factorization dg-category}
$\cC_{\mathrm{op}} = \cC_{\mathrm{op}}(V_k(\mathfrak{sl}_2))$
has the following data.

\medskip\noindent
\textbf{Objects.}
The objects of $\cC_{\mathrm{op}}$ are
$V_k(\mathfrak{sl}_2)$-modules, i.e.\ smooth modules for the
vertex algebra $V_k(\mathfrak{sl}_2)$:
\begin{equation}\label{eq:benchmark-Cop-objects}
 \Ob(\cC_{\mathrm{op}})
 \;=\;
 V_k(\mathfrak{sl}_2)\text{-}\mathrm{mod}.
\end{equation}
The generating objects are the Weyl modules
(generalized Verma modules):
\begin{equation}\label{eq:benchmark-Cop-generators}
 \mathbb{V}_{k,\lambda}
 \;:=\;
 \operatorname{Ind}_{\mathfrak{sl}_2[t] \oplus \C\mathbf{1}}^{\widehat{\mathfrak{sl}}_2}
 L(\lambda),
 \qquad \lambda \in \mathfrak{h}^*,
\end{equation}
where $L(\lambda)$ is the finite-dimensional
$\mathfrak{sl}_2$-module of highest weight $\lambda$.
The vacuum module $\mathbb{V}_{k,0} = V_k(\mathfrak{sl}_2)$
is the monoidal unit.

At \emph{integrable level} $k \in \Z_{\ge 0}$, the simple
quotients $L_{k,j}$ for $0 \le j \le k$ form a finite
generating set:
\begin{equation}\label{eq:benchmark-Cop-integrable-objects}
 \Ob(\cC_{\mathrm{op}})^{\mathrm{int}}
 \;=\;
 \{L_{k,0},\; L_{k,1},\; \ldots,\; L_{k,k}\},
\end{equation}
with $L_{k,0} = L_k(\mathfrak{sl}_2)$ the simple vacuum module.

\medskip\noindent
\textbf{Morphisms.}
For $M, N \in \Ob(\cC_{\mathrm{op}})$, the morphism space is the
vertex-algebraic Hom complex:
\begin{equation}\label{eq:benchmark-Cop-morphisms}
 \Hom_{\cC_{\mathrm{op}}}(M, N)
 \;=\;
 \RHom_{V_k(\mathfrak{sl}_2)}(M, N),
\end{equation}
computed in the derived category of $V_k(\mathfrak{sl}_2)$-modules.
On the abelian level, the Hom is the space of intertwining operators:
\begin{equation}\label{eq:benchmark-Cop-hom-abelian}
 \Hom_{V_k}(M, N)
 \;=\;
 \bigl\{
 f \colon M \to N
 \;\big|\;
 f(a_{(n)} m) = a_{(n)} f(m)
 \;\text{for all } a \in V_k,\, n \in \Z,\, m \in M
 \bigr\}.
\end{equation}
In particular, the vacuum endomorphism algebra recovers the
boundary chiral algebra:
\begin{equation}\label{eq:benchmark-Cop-vacuum-end}
 \End_{\cC_{\mathrm{op}}}(V_k(\mathfrak{sl}_2))
 \;=\;
 V_k(\mathfrak{sl}_2).
\end{equation}

\medskip\noindent
\textbf{Composition.}
The composition
$\Hom(M,N) \otimes \Hom(L,M) \to \Hom(L,N)$ is given by
the OPE of intertwining operators on the boundary interval
$I_p$: for $f \in \Hom(M,N)$ inserted at time $t_1$ and
$g \in \Hom(L,M)$ at time $t_2 < t_1$, the composition
$f \circ g$ is the time-ordered product
\begin{equation}\label{eq:benchmark-Cop-composition}
 (f \circ g)(v)
 \;=\;
 \lim_{t_1 > t_2}
 f(t_1)\, g(t_2)\, v,
 \qquad v \in L,
\end{equation}
made precise by the configuration-space regularization on
$\Conf^{<}_2(\R)$. The $A_\infty$-corrections $\mu_n$ for
$n \ge 3$ arise from the codimension-one boundary strata of
the associahedron $K_n$
(Theorem~\ref*{V1-thm:thqg-local-global-bridge}(i)).

\medskip\noindent
\textbf{Factorization structure.}
For disjoint open disks $D_1, \ldots, D_n \subset
\PP^1 \setminus \{0,\infty\}$ and modules
$M_1, \ldots, M_n \in \Ob(\cC_{\mathrm{op}})$, the
factorization isomorphism is
\begin{equation}\label{eq:benchmark-Cop-factorization}
 \cC_{\mathrm{op}}(D_1 \sqcup \cdots \sqcup D_n)
 \;\xrightarrow{\;\sim\;}
 \cC_{\mathrm{op}}(D_1)
 \boxtimes \cdots \boxtimes
 \cC_{\mathrm{op}}(D_n).
\end{equation}
On each formal disk $D_i$, $\cC_{\mathrm{op}}(D_i)$ is the
category of $V_k(\mathfrak{sl}_2)$-modules supported at the
center of $D_i$. The factorization isomorphism is the tensor
product of modules via fusion:
for modules $M$ at $z_1$ and $N$ at $z_2$ with
$|z_1| < |z_2|$ (in the standard coordinate), the fusion
product is the module
\begin{equation}\label{eq:benchmark-Cop-fusion}
 M \boxtimes_{z_1, z_2} N
 \;:=\;
 H^0\bigl(\PP^1 \setminus \{z_1, z_2, \infty\},\;
 \cF_M \boxtimes \cF_N\bigr),
\end{equation}
where $\cF_M$ and $\cF_N$ are the corresponding
$\widehat{\mathfrak{sl}}_2$-modules viewed as sections of the
factorization algebra on $\PP^1 \setminus D$. At integrable
levels, this recovers the Verlinde fusion rules:
$L_{k,i} \boxtimes L_{k,j} \cong \bigoplus_l N_{ij}^{\;\;l}\,
L_{k,l}$.

\medskip\noindent
\textbf{Braiding.}
The analytic continuation of the fusion product around
$z_1 = z_2$ defines a braiding on $\cC_{\mathrm{op}}$.
At the perturbative level, this braiding is given by the
monodromy of the Knizhnik--Zamolodchikov connection
\begin{equation}\label{eq:benchmark-KZ}
 \nabla_{\mathrm{KZ}}
 \;=\;
 d
 \;-\;
 \hbar\, \frac{\Omega_{\mathfrak{sl}_2}}{z_1 - z_2}\,
 d(z_1 - z_2),
 \qquad
 \hbar = \frac{1}{k + h^\vee} = \frac{1}{k + 2},
\end{equation}
acting on $M \otimes N$ for any pair of modules $M, N$.
The KZ parameter $\hbar = 1/(k+2)$ is the reciprocal of the
shifted level. At integrable $k$, the monodromy of
$\nabla_{\mathrm{KZ}}$ gives the braiding matrices of the
modular tensor category
$\mathrm{Rep}(U_q(\mathfrak{sl}_2))$ with
$q = e^{i\pi\hbar} = e^{i\pi/(k+2)}$
(Kazhdan--Lusztig~\cite{KL93},
Drinfeld~\cite{Dri89}).

\medskip\noindent
\textbf{Generating relations.}
The category $\cC_{\mathrm{op}}$ is generated under colimits
and the factorization product by the vacuum module
$V_k(\mathfrak{sl}_2)$. The defining relations are:
\begin{enumerate}[label=\textup{(R\arabic*)},nosep]
\item \emph{OPE relation.}
 The $\lambda$-bracket on the vacuum endomorphism algebra is
 \begin{equation}\label{eq:benchmark-sl2-ope}
 \{J^a_\lambda J^b\}
 \;=\;
 f^{ab}_{\phantom{ab}c}\, J^c
 \;+\;
 k\, (J^a, J^b)\, \lambda,
 \end{equation}
 where $\{J^a\} = \{e, f, h\}$ are the Chevalley generators,
 $f^{ab}_c$ the structure constants of $\mathfrak{sl}_2$, and
 $(\cdot,\cdot) = \mathrm{tr}_{\mathrm{fund}}$ the normalized
 Killing form. This determines the vertex algebra
 $V_k(\mathfrak{sl}_2)$ and hence its module category.

\item \emph{Level constraint.}
 The central element $\mathbf{1} \in
 \widehat{\mathfrak{sl}}_2$ acts as
 $\mathbf{1} \cdot m = k \cdot m$ for all $m \in M$ and
 all $M \in \Ob(\cC_{\mathrm{op}})$. This fixes the level.

\item \emph{Smoothness.}
 For every $m \in M$ and every
 $a \in \mathfrak{sl}_2 \subset \widehat{\mathfrak{sl}}_2$,
 there exists $N \ge 0$ such that
 $a_n \cdot m = 0$ for all $n \ge N$. This is the smooth-module
 condition, ensuring that the action extends to the vertex
 algebra (not merely the Lie algebra).

\item \emph{Vacuum axiom.}
 The vacuum vector $|0\rangle \in V_k(\mathfrak{sl}_2)$
 satisfies $J^a_n |0\rangle = 0$ for $n \ge 0$, and
 $\mathbf{1} |0\rangle = k |0\rangle$.

\item \emph{Factorization.}
 The fusion product~\eqref{eq:benchmark-Cop-fusion} is
 associative up to coherent homotopy (the factorization
 isomorphisms satisfy the Ran-space sewing identities of
 Definition~\ref{def:oc-factorization-category}).
\end{enumerate}
Relations (R1)--(R4) determine the abelian category
$V_k(\mathfrak{sl}_2)\text{-mod}$. Relation (R5) upgrades
it to a factorization category on
$(\PP^1, \{0,\infty\}, \tau)$.
\end{definition}

\begin{proposition}[Boundary algebra recovery;
 \ClaimStatusProvedElsewhere]
\label{prop:benchmark-boundary-algebra}
\textup{(Costello--Gaiotto~\cite{costello-gaiotto}.)}
Let\/ $\cC_{\mathrm{op}} = \cC_{\mathrm{op}}(V_k(\mathfrak{sl}_2))$
be the open-sector factorization dg-category of
Definition~\textup{\ref{def:benchmark-Cop-sl2}}, and let\/
$b = V_k(\mathfrak{sl}_2)$ be the vacuum module. Then:
\begin{enumerate}[label=\textup{(\roman*)}]
\item The boundary algebra at each puncture
 $p \in \{0, \infty\}$ is
 \begin{equation}\label{eq:benchmark-boundary-recovery}
 \cA_{\partial, p}
 \;=\;
 \End_{\cC_{\mathrm{op}}(I_p)}(b)
 \;=\;
 V_k(\mathfrak{sl}_2).
 \end{equation}
\item The vacuum module $b$ is a compact generator of\/
 $\cC_{\mathrm{op}}$: every object is a retract of a
 filtered colimit of finite direct sums of shifts of $b$.
\item The full subcategory generated by $b$ determines the
 Morita equivalence class of\/ $\cC_{\mathrm{op}}$, and
 the derived center
 $\cZ^{\mathrm{der}}_{\mathrm{ch}}(\cC_{\mathrm{op}})$
 depends only on this class
 \textup{(}Theorem~\textup{\ref*{V1-thm:thqg-local-global-bridge}(iii)}\textup{)}.
\end{enumerate}
\end{proposition}

\begin{proof}
(i) The endomorphism algebra $\End(b)$ of the vacuum module in
$V_k(\mathfrak{sl}_2)\text{-mod}$ is $V_k(\mathfrak{sl}_2)$ itself:
every endomorphism of the vacuum module is determined by the
image of $|0\rangle$, which by the vacuum axiom (R4) and
smoothness (R3) lies in $V_k(\mathfrak{sl}_2)$.

(ii) Compact generation follows from the structure theory of
smooth $\widehat{\mathfrak{sl}}_2$-modules at non-critical level:
every smooth module admits a resolution by Weyl modules
$\mathbb{V}_{k,\lambda}$~\eqref{eq:benchmark-Cop-generators},
each of which is a quotient of the vacuum module tensored with
a finite-dimensional $\mathfrak{sl}_2$-representation.
At integrable level $k \in \Z_{\ge 0}$, the finite set
$\{L_{k,0}, \ldots, L_{k,k}\}$ generates the semisimple
quotient category, and $L_{k,0} = b$ generates the full
category since every $L_{k,j}$ appears in the fusion product
$b^{\boxtimes n}$ for $n = j$.

(iii) Morita invariance of the derived center is
Theorem~\ref*{V1-thm:thqg-local-global-bridge}(iii).
\end{proof}

\begin{proposition}[Derived center gives the correct bulk;
 \ClaimStatusProvedHere]
\label{prop:benchmark-bulk-from-Cop}
\index{derived center!SL(2) Chern--Simons|textbf}
The chiral derived center of\/ $\cC_{\mathrm{op}}$ is
\begin{equation}\label{eq:benchmark-bulk-Cop}
 \cZ^{\mathrm{der}}_{\mathrm{ch}}(\cC_{\mathrm{op}})
 \;\simeq\;
 C^\bullet_{\mathrm{ch}}(V_k(\mathfrak{sl}_2),\,
 V_k(\mathfrak{sl}_2)),
\end{equation}
the chiral Hochschild cochain complex of the boundary algebra.
At generic $k$:
\begin{equation}\label{eq:benchmark-bulk-generic-Cop}
 H^0\!\bigl(
 \cZ^{\mathrm{der}}_{\mathrm{ch}}(\cC_{\mathrm{op}})
 \bigr)
 \;\cong\;
 Z\!\bigl(\widetilde{U}_k(\widehat{\mathfrak{sl}}_2)\bigr)
 \;\cong\; \C.
\end{equation}
At integrable level $k \in \Z_{\ge 0}$, the categorical
center is the Verlinde algebra with
$\dim = k + 1$ \textup{(}Proposition~\textup{\ref{prop:benchmark-verlinde}}\textup{)}.
\end{proposition}

\begin{proof}
The identification~\eqref{eq:benchmark-bulk-Cop} follows from
Theorem~\ref*{V1-thm:thqg-local-global-bridge}(ii): the derived
center of $\cC_{\mathrm{op}}$ is computed, in the chart
$b = V_k(\mathfrak{sl}_2)$, by the chiral Hochschild cochains

= C^\bullet_{\mathrm{ch}}(V_k, V_k)$.
The generic identification~\eqref{eq:benchmark-bulk-generic-Cop}

\end{proof}

\subsubsection{The Koszul dual}

 k'
 \;=\;
 -k - 2h^\vee
 \;=\;

\end{equation}

\subsubsection{The collision residue and $r$-matrix}
\label{subsubsec:benchmark-r-matrix}

The $\lambda$-bracket
\eqref{eq:benchmark-sl2-ope}
of $V_k(\mathfrak{sl}_2)$ has two terms:
\[
 \{J^a_\lambda J^b\}
 \;=\;
 \underbrace{f^{ab}_c\, J^c}_{\text{pole order }1
 \text{ in the OPE}}
 \;+\;
 \underbrace{k\,(J^a, J^b)\,\lambda}_{\text{pole order }2
 \text{ in the OPE}}.
\]
The OPE is
\begin{equation}\label{eq:benchmark-sl2-ope-explicit}
 J^a(z)\, J^b(w)
 \;\sim\;
 \frac{k\,(J^a, J^b)}{(z-w)^2}
 \;+\;
 \frac{f^{ab}_c\, J^c(w)}{z-w},
\end{equation}
with maximal pole order~$2$.

The collision residue extracts pole orders \emph{one less} than
the OPE (the bar kernel $d\log(z-w)$ absorbs one power of
$(z-w)$):
\begin{equation}\label{eq:benchmark-r-matrix}
 r(z)
 \;=\;
 \Res^{\mathrm{coll}}_{0,2}(\Theta_\cA)
 \;=\;
 \frac{k\,\Omega_{\mathfrak{sl}_2}}{z},
\end{equation}
where
\begin{equation}\label{eq:benchmark-casimir}
 \Omega_{\mathfrak{sl}_2}
 \;=\;
 \sum_a T^a \otimes T^a
 \;=\;
 e \otimes f + f \otimes e
 + \tfrac{1}{2}\, h \otimes h
 \;\in\;
 \mathfrak{sl}_2 \otimes \mathfrak{sl}_2
\end{equation}
is the quadratic Casimir (the inverse of the normalized
Killing form $(J^a, J^b) = \mathrm{tr}_{\mathrm{fund}}(J^a J^b)$,
with $\{T^a\}$ an orthonormal basis).
The $r$-matrix $r(z) = k\,\Omega/z$ is the standard rational
solution of the classical Yang--Baxter equation for
$\widehat{\mathfrak{sl}}_2$ at level~$k$. In the bar complex, it arises as the
degree-$2$, genus-$0$ projection of the universal MC
element~$\Theta_\cA$: the bar kernel $d\log(z-w) = dz/(z-w)$
extracts the leading-order collision data from the OPE, with
pole orders shifted down by one.

The classical Yang--Baxter equation
\begin{equation}\label{eq:benchmark-cybe}
 [r_{12}(z_1 - z_2),\, r_{13}(z_1 - z_3)]
 +
 [r_{12}(z_1 - z_2),\, r_{23}(z_2 - z_3)]
 +
 [r_{13}(z_1 - z_3),\, r_{23}(z_2 - z_3)]
 \;=\; 0
\end{equation}
for $r(z) = k\,\Omega/z$ follows from the Jacobi identity
in $\mathfrak{sl}_2$ (the factor~$k$ is common and cancels): this is the degree-$3$ projection of the
Maurer--Cartan equation $D\Theta + \tfrac{1}{2}[\Theta,\Theta] = 0$
at genus~$0$. The RTT quantization produces the Yangian
$Y_\hbar(\mathfrak{sl}_2)$.


\subsubsection{The modular characteristic and genus-$1$ free energy}
\label{subsubsec:benchmark-modular-characteristic}

The modular characteristic of $V_k(\mathfrak{sl}_2)$ is
computed from the universal formula
(Theorem~\ref*{V1-thm:genus-universality}):
\begin{equation}\label{eq:benchmark-kappa}
 \kappa\bigl(V_k(\mathfrak{sl}_2)\bigr)
 \;=\;
 \frac{\dim(\mathfrak{sl}_2)\,(k + h^\vee)}{2h^\vee}
 \;=\;
 \frac{3(k + 2)}{4}.
\end{equation}

\begin{proposition}[Modular data for the benchmark example;
 \ClaimStatusProvedHere]
\label{prop:benchmark-modular-data}
The modular invariants of $V_k(\mathfrak{sl}_2)$ are:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item $\kappa = 3(k+2)/4$.
\item The genus-$1$ free energy is
 $F_1 = \kappa/24 = (k+2)/32$.
\item The Koszul dual modular characteristic is
 $\kappa' = \kappa(V_{-k-4}(\mathfrak{sl}_2))
 = 3(-k - 4 + 2)/4
 = -3(k+2)/4 = -\kappa$.
\item The complementarity relation holds:
 $\kappa + \kappa' = 0$.
\item The Sugawara central charge is
 $c = k \cdot \dim(\mathfrak{sl}_2)/(k + h^\vee)
 = 3k/(k+2)$.
\item At the self-dual point $k = -2$
 \textup{(}critical level\textup{)}:
 $\kappa = 0$, $c$ is undefined
 \textup{(}the Sugawara construction degenerates\textup{)}.
\item At $k = 1$ \textup{(}first integrable level\textup{)}:
 $\kappa = 9/4$, $c = 1$, $F_1 = 3/32$.
\end{enumerate}
\end{proposition}

\begin{proof}
Item~(i): direct substitution into
$\dim(\fg)(k+h^\vee)/(2h^\vee)$ with
$\dim(\mathfrak{sl}_2) = 3$, $h^\vee = 2$.
Item~(ii): $F_1 = \kappa/24 = 3(k+2)/(4 \cdot 24) = (k+2)/32$.
Item~(iii): $k' = -k-4$, so $\kappa' = 3(-k-4+2)/4 = 3(-k-2)/4
= -3(k+2)/4$.
Item~(iv): sum.
Item~(v): standard Sugawara formula.
Item~(vi): $\kappa|_{k=-2} = 3 \cdot 0/4 = 0$;
at $k = -h^\vee$, the denominator of $c$ vanishes.
Item~(vii): $\kappa|_{k=1} = 3 \cdot 3/4 = 9/4$;
$c|_{k=1} = 3/(1+2) = 1$;
$F_1|_{k=1} = (1+2)/32 = 3/32$.
\end{proof}

\begin{remark}[Consistency check: $F_1$ from the character]
\label{rem:benchmark-f1-character}
An independent computation of $F_1$ uses the vacuum character.
At level $k$, the vacuum character of $V_k(\mathfrak{sl}_2)$ is
\[
 \chi_0(\tau)
 \;=\;
 q^{-c/24}\,\bigl(1 + 3q + \cdots\bigr),
 \qquad q = e^{2\pi i \tau},
\]
where the leading coefficient $3$ counts the $3$-dimensional
current space. The genus-$1$ free energy $F_1 = -\log\eta(\tau)
\cdot \kappa$ (at leading order in $q$) gives
$F_1 = \kappa/24 = (k+2)/32$, matching item~(ii).
\end{remark}


\subsubsection{The universal bulk: chiral derived center}
\label{subsubsec:benchmark-derived-center}

The bulk algebra is \emph{not} the bar complex $\barB(V_k(\mathfrak{sl}_2))$
(the bar complex classifies twisting morphisms, not bulk observables). It is the chiral derived center
\begin{equation}\label{eq:benchmark-derived-center}
 \cZ^{\mathrm{der}}_{\mathrm{ch}}(\cA_\partial)
 \;=\;
 C^\bullet_{\mathrm{ch}}(V_k(\mathfrak{sl}_2),\,
 V_k(\mathfrak{sl}_2)),
\end{equation}
the chiral Hochschild cochain complex of the boundary algebra.
This is the universal bulk of the corrected Koszul triangle
(Theorem~\ref{thm:boundary-linear-bulk-boundary}).

For generic $k$ (non-integrable), the derived center is
a dg algebra whose cohomology in degree~$0$ is the center of the
completed enveloping algebra $\widetilde{U}(\widehat{\mathfrak{sl}}_2)$
at level~$k$, which is $\C$ (one-dimensional, generated by the
identity):
\begin{equation}\label{eq:benchmark-center-generic}
 H^0\!\bigl(\cZ^{\mathrm{der}}_{\mathrm{ch}}(V_k(\mathfrak{sl}_2))\bigr)
 \;\cong\;
 Z\!\bigl(\widetilde{U}_k(\widehat{\mathfrak{sl}}_2)\bigr)
 \;\cong\;
 \C.
\end{equation}

At \textbf{integrable levels} $k \in \Z_{\ge 0}$, the category
of modules $\mathrm{Rep}(V_k(\mathfrak{sl}_2))$ is a modular
tensor category with $k + 1$ simple objects
$L_0, L_1, \ldots, L_k$ (the integrable highest-weight modules
with highest weights $0, 1, \ldots, k$). The categorical
derived center is
\begin{equation}\label{eq:benchmark-center-integrable}
 Z\!\bigl(\mathrm{Rep}(V_k(\mathfrak{sl}_2))\bigr)
 \;\cong\;
 \bigoplus_{j=0}^k \C \cdot e_j,
 \qquad \dim = k + 1,
\end{equation}
the \emph{Verlinde algebra}: a commutative semisimple algebra
whose structure constants are the fusion coefficients
$N_{ij}^{\;\;l}$ computed from the modular $S$-matrix.

\begin{proposition}[Verlinde algebra at integrable levels;
 \ClaimStatusProvedElsewhere]
\label{prop:benchmark-verlinde}
\textup{(Verlinde~\cite{Verlinde1988},
Huang~\cite{Huang2005}.)}
At integrable level $k \in \Z_{\ge 0}$, the modular $S$-matrix of
$\mathrm{Rep}(V_k(\mathfrak{sl}_2))$ is the
$(k{+}1) \times (k{+}1)$ matrix
\begin{equation}\label{eq:benchmark-s-matrix}
 S_{ij}
 \;=\;
 \sqrt{\frac{2}{k+2}}\;
 \sin\!\biggl(\frac{\pi(i+1)(j+1)}{k+2}\biggr),
 \qquad
 0 \le i, j \le k.
\end{equation}
The fusion coefficients are $N_{ij}^{\;\;l}
= \sum_{m=0}^k S_{im} S_{jm} \overline{S}_{lm} / S_{0m}$,
and the Verlinde algebra $\mathrm{Ver}_{k}(\mathfrak{sl}_2)$ is
the commutative ring $\Z[L_0, \ldots, L_k] / (\text{fusion
relations})$ with identity $L_0$. This is the center of the
modular tensor category
$\mathrm{Rep}(V_k(\mathfrak{sl}_2))$.
\end{proposition}

\begin{remark}[Generic vs integrable: algebraic vs categorical center]
\label{rem:benchmark-generic-vs-integrable}
At generic $k$, the algebraic derived center $C^\bullet_{\mathrm{ch}}(V_k, V_k)$ has
$H^0 \cong \C$: the only chiral Hochschild $0$-cocycle is the identity.
At integrable $k$, the \emph{categorical} center of
$\mathrm{Rep}(V_k(\mathfrak{sl}_2))$ has dimension $k+1$.
The passage from algebraic to categorical is the restriction from
all modules to integrable ones, which introduces new central
endomorphisms (projections onto simple summands). The bar complex
$\barB(V_k(\mathfrak{sl}_2))$ classifies twisting morphisms
(universal couplings between $V_k$ and $V_{-k-4}$); it does not
see the Verlinde algebra directly.
\end{remark}


\subsubsection{The Kazhdan--Lusztig equivalence and the line category}
\label{subsubsec:benchmark-kl-category}

On the chirally Koszul locus, the line-operator category is

 \cC_{\mathrm{line}}

\subsubsection{The annulus trace}

The annulus trace theorem

 \mathrm{Tr}_\cA

 HH_0(\cC_{\mathrm{op}})

\subsubsection{Factorization homology on
 \texorpdfstring{$\PP^1 \setminus \{0,\infty\}$}{P\^1 - \{0, infty\}}}
\label{subsubsec:benchmark-factorization-homology}

On the integrable affine comparison surface, the factorization
homology of the open-sector category over the open complement
$\PP^1 \setminus \{0,\infty\} \cong \C^\times$ identifies with
\begin{equation}\label{eq:benchmark-factorization-homology}
 \int_{\C^\times} \cC_{\mathrm{op}}
 \;\simeq\;
 \mathrm{CB}_{0,2}(V_k(\mathfrak{sl}_2)),
\end{equation}
the genus-$0$ two-point conformal-block package
(at $0$ and $\infty$). For a pair of modules
$M_0 \in V_k(\mathfrak{sl}_2)\text{-mod}$ at $0$ and
$M_\infty \in V_k(\mathfrak{sl}_2)\text{-mod}$ at $\infty$,
the corresponding conformal-block space is
\begin{equation}\label{eq:benchmark-conformal-blocks}
 \mathrm{CB}_{0,2}(M_0, M_\infty)
 \;=\;
 \Hom_{V_k(\mathfrak{sl}_2)}(M_0,\, M_\infty^*),
\end{equation}
the space of $V_k(\mathfrak{sl}_2)$-intertwining operators
from $M_0$ to the contragredient of $M_\infty$. At integrable

modules $L_{j_0}$ and $L_{j_\infty}$: conformal blocks on

nonzero precisely when the two modules are contragredient.

For the vacuum-to-vacuum conformal block
($M_0 = M_\infty = V_k(\mathfrak{sl}_2)$), the space is
one-dimensional:
\begin{equation}\label{eq:benchmark-vacuum-block}
 \mathrm{CB}_{0,2}(V_k, V_k) \;\cong\; \C.

This is the partition function of the theory on $\PP^1$ with
vacuum boundary conditions at both punctures.

\item \textbf{Factorization homology.}

 Vacuum-to-vacuum: one-dimensional.

\end{enumerate}
\end{theorem}

\subsection{The bar complex of the simple quotient
 \texorpdfstring{$L_k(\mathfrak{sl}_2)$}{L\_k(sl\_2)}
 at integrable levels}
\label{subsec:bar-complex-integrable}
\index{simple quotient!bar complex|textbf}
\index{integrable level!bar complex|textbf}
\index{null vector!effect on bar complex|textbf}
\index{bar complex!integrable level|textbf}

The preceding sections compute the bar complex
$\barB^{\mathrm{ch}}(V_k(\mathfrak{sl}_2))$ of the
\emph{universal} affine vertex algebra. At integrable levels
$k \in \Z_{\ge 0}$, the physically relevant object is the
\emph{simple} quotient $L_k(\mathfrak{sl}_2)$, obtained by
dividing out the maximal proper ideal generated by null
vectors. The bar complex of the simple quotient is smaller
than that of the universal algebra, and its structure reflects
the finiteness of the modular tensor category
$\mathrm{Rep}(L_k(\mathfrak{sl}_2))$. The present section
computes this object explicitly at $k = 1$, identifies the
structural mechanism at general integrable $k$, and explains
what the bar complex sees and does not see of the quantum
group at roots of unity.


\subsubsection{The null vector and the surjection
 \texorpdfstring{$V_k \twoheadrightarrow L_k$}{V\_k -> L\_k}}
\label{subsubsec:null-vector-surjection}

At integrable level $k \in \Z_{\ge 0}$, the maximal proper
ideal $\mathcal{J}_k \subset V_k(\mathfrak{sl}_2)$ is
generated by the singular vector
\begin{equation}\label{eq:null-vector-sl2}
 v_{\mathrm{null}}
 \;=\;
 f_{(-1)}^{k+1} |0\rangle
 \;\in\;
 V_k(\mathfrak{sl}_2)_{[k+1]},
\end{equation}
of conformal weight $k + 1$. This vector is primary
($L_n v_{\mathrm{null}} = 0$ for $n > 0$) and of
$\mathfrak{sl}_2$-weight $-2(k+1)$
($h_{(0)} v_{\mathrm{null}} = -2(k+1) v_{\mathrm{null}}$).
The simple quotient is
\begin{equation}\label{eq:simple-quotient-sl2}
 L_k(\mathfrak{sl}_2)
 \;:=\;
 V_k(\mathfrak{sl}_2) / \mathcal{J}_k,
 \qquad
 \mathcal{J}_k = (v_{\mathrm{null}}).
\end{equation}

\begin{remark}[The degree of the null vector relation]
\label{rem:null-vector-degree}
The null vector~\eqref{eq:null-vector-sl2} is a relation of
degree $k + 1$ in the generators $\{e, f, h\}$, where the
degree counts the number of mode applications $J_{(-n)}$ used
to build the vector from the vacuum. For $k = 1$:
$v_{\mathrm{null}} = f_{(-1)}^2 |0\rangle$ is
\emph{quadratic} (degree~$2$). For $k \ge 2$:
$v_{\mathrm{null}}$ is of degree $k + 1 \ge 3$, which is
\emph{not quadratic}. This distinction is critical for
Koszul duality: $L_1(\mathfrak{sl}_2)$ is a quadratic chiral
algebra and the GLZ framework
(Definition~\ref{def:quadratic_chiral}) applies directly,
while $L_k(\mathfrak{sl}_2)$ for $k \ge 2$ requires the full
$\Ainf$ bar construction.
\end{remark}


\subsubsection{Effect on the bar complex: the contravariant
 inclusion}
\label{subsubsec:bar-contravariant}

The surjection
$\pi \colon V_k(\mathfrak{sl}_2) \twoheadrightarrow
L_k(\mathfrak{sl}_2)$ induces a map of bar complexes
\begin{equation}\label{eq:bar-contravariant-inclusion}
 \barB^{\mathrm{ch}}(\pi)
 \colon
 \barB^{\mathrm{ch}}(L_k(\mathfrak{sl}_2))
 \;\hookrightarrow\;
 \barB^{\mathrm{ch}}(V_k(\mathfrak{sl}_2)),
\end{equation}
which is an \emph{inclusion}: the bar construction is
contravariant on surjections (a quotient algebra has fewer
generators and more relations, producing a smaller bar
complex). The cokernel
$\barB^{\mathrm{ch}}(V_k) / \barB^{\mathrm{ch}}(L_k)$ is
generated by bar elements involving the null vector and its
descendants.

The null vector $v_{\mathrm{null}} \in V_k$ produces a bar
element $[s^{-1} v_{\mathrm{null}}]$ in bar degree~$1$. In
$\barB^{\mathrm{ch}}(V_k)$, this element has nontrivial bar
differential (it maps to iterated OPE residues of
$v_{\mathrm{null}}$ with the generators). In
$\barB^{\mathrm{ch}}(L_k)$, the element is \emph{absent}:
$v_{\mathrm{null}} = 0$ in $L_k$, so it does not contribute
to the bar complex. This removal propagates through the bar
complex via the Stasheff/AOS relations, truncating all bar
elements that involve null descendants.

\begin{proposition}[Null vector truncation of the bar complex;
 \ClaimStatusProvedHere]
\label{prop:null-truncation}
\index{null vector!bar complex truncation}
At integrable level $k \in \Z_{\ge 0}$:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item The bar complex $\barB^{\mathrm{ch}}(L_k(\mathfrak{sl}_2))$
 is a sub-dg-coalgebra of\/
 $\barB^{\mathrm{ch}}(V_k(\mathfrak{sl}_2))$, with the
 inclusion~\eqref{eq:bar-contravariant-inclusion} a strict map
 of dg coalgebras.
\item The weight-space decomposition of the bar complex
 truncates: for each bar degree~$n$ and conformal weight~$w$,
 the weight-$w$ component
 $\barB^n(L_k)_{[w]}$ is a quotient of
 $\barB^n(V_k)_{[w]}$ by the sub-coalgebra generated by
 null descendants of weight $\le w$.
\item The collision residue $r(z) = k\,\Omega_{\mathfrak{sl}_2}/z$
 is unchanged: it depends only on the quadratic OPE data,
 which is the same for $V_k$ and $L_k$. In particular, the
 classical Yang--Baxter equation and the Yangian RTT
 presentation survive the null vector truncation at the
 genus-$0$ level.
\item The bar differential acquires additional relations from the
 null vector: the element
 $[s^{-1} v_{\mathrm{null}}] \in \barB^1(V_k)$ is a cycle in
 $\barB^{\mathrm{ch}}(V_k)$ modulo the ideal generated by lower
 bar degrees, and its vanishing in $\barB^{\mathrm{ch}}(L_k)$
 truncates the coalgebra structure.
\end{enumerate}
\end{proposition}

\begin{proof}
(i) The bar construction is a functor from chiral algebras to
dg coalgebras; the surjection $\pi$ is a morphism of chiral
algebras, so $\barB(\pi)$ is a morphism of dg coalgebras.
The map is injective on the bar complex of $L_k$ because
$L_k$ is a quotient: every element of $\barB(L_k)$ is the
image of a bar element of $V_k$ under the quotient map, and
the bar differential on $L_k$ is the restriction of the bar
differential on $V_k$ to the sub-coalgebra generated by
elements of $L_k$.

(ii) The weight-space truncation follows from the fact that
the null vector $v_{\mathrm{null}}$ has conformal weight
$k + 1$: all null descendants have weight $\ge k + 1$, and
bar elements involving them are removed. In bar degree~$n$,
the removed elements are those of the form
$[s^{-1} a_1 | \cdots | s^{-1} a_n]$ where at least one
$a_i$ is a null descendant.

(iii) The collision residue is determined by the degree-$2$,
genus-$0$ bar differential, which uses only the OPE data
$J^a(z) J^b(w) \sim k\,\delta^{ab}/(z-w)^2 + f^{ab}_c\,J^c/(z-w)$.
This OPE is identical in $V_k$ and $L_k$ (the null vector
sits at weight $k+1$, not at the quadratic level).

(iv) In $V_k$, $v_{\mathrm{null}} \ne 0$ is a nonzero
element of conformal weight $k + 1$. Its image in bar
degree~$1$ is $[s^{-1} v_{\mathrm{null}}]$, which is a
cycle in the associated graded of the bar filtration: the
bar differential on $v_{\mathrm{null}}$ produces terms of
lower bar degree (OPE residues of $v_{\mathrm{null}}$ with
generators), and these vanish modulo the ideal. In $L_k$,
$v_{\mathrm{null}} = 0$, so this cycle is absent.
\end{proof}


\subsubsection{Explicit computation at $k = 1$: the minimal
 integrable level}
\label{subsubsec:bar-k-equals-1}

At $k = 1$, the null vector is
$v_{\mathrm{null}} = f_{(-1)}^2 |0\rangle$ at conformal
weight~$2$. The simple quotient $L_1(\mathfrak{sl}_2)$ has
central charge $c = 1$ and two simple modules: $L_0$
(trivial) and $L_1$ (the $2$-dimensional representation).
The fusion rules are $L_1 \boxtimes L_1 \cong L_0$
(the fundamental is self-dual), and the modular tensor category
is $\Z/2\Z$-graded: the simplest nontrivial instance.

\begin{proposition}[$L_1(\mathfrak{sl}_2)$ is quadratic;
 \ClaimStatusProvedHere]
\label{prop:L1-quadratic}
\index{simple quotient!L1(sl2)@$L_1(\mathfrak{sl}_2)$!quadratic}
The simple quotient $L_1(\mathfrak{sl}_2)$ is a quadratic
chiral algebra in the sense of
Definition~\textup{\ref{def:quadratic_chiral}}. Its
presentation is:
\begin{equation}\label{eq:L1-presentation}
 L_1(\mathfrak{sl}_2)
 \;=\;
 T_{\mathrm{ch}}(V) \,/\,
 \bigl\langle
 R_{\mathrm{OPE}} \cup R_{\mathrm{null}}
 \bigr\rangle,
\end{equation}
where $V = \C e \oplus \C f \oplus \C h$,
$R_{\mathrm{OPE}}$ is the space of OPE
relations~\eqref{eq:benchmark-sl2-ope} at level $k = 1$, and
$R_{\mathrm{null}} = \mathrm{span}\{f_{(-1)}^2 |0\rangle\}$
is the null vector relation. Both $R_{\mathrm{OPE}}$ and
$R_{\mathrm{null}}$ are quadratic
\textup{(}degree~$2$ in the chiral tensor product\textup{)}.
\end{proposition}

\begin{proof}
The OPE relations are always quadratic (they express the
binary $\lambda$-bracket). The null vector
$f_{(-1)}^2 |0\rangle$ involves two applications of $f_{(-1)}$
to the vacuum: it is a degree-$2$ element of
$T_{\mathrm{ch}}(V)$, hence a quadratic relation.
\end{proof}

\begin{proposition}[Bar complex of $L_1(\mathfrak{sl}_2)$;
 \ClaimStatusProvedHere]
\label{prop:bar-L1}
\index{bar complex!L1(sl2)@$L_1(\mathfrak{sl}_2)$|textbf}
The bar complex $\barB^{\mathrm{ch}}(L_1(\mathfrak{sl}_2))$
has the following structure.
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item \textbf{Bar degree $1$.}
 $\barB^1(L_1) = s^{-1}\overline{L}_1$, the desuspended
 augmentation ideal of $L_1$, with generators
 $s^{-1}e$, $s^{-1}f$, $s^{-1}h$ at conformal weight~$1$.
 At weight~$2$, the null vector relation removes
 $s^{-1}(f_{(-1)}^2|0\rangle)$: this element is
 \emph{absent} from $\barB^1(L_1)$ because
 $f_{(-1)}^2|0\rangle = 0$ in $L_1$.

\item \textbf{Bar degree $2$.}
 The $\mathfrak{sl}_2$-Casimir cycle survives:
 $[s^{-1}e \,|\, s^{-1}f] + [s^{-1}f \,|\, s^{-1}e]
 + \frac{1}{2}[s^{-1}h \,|\, s^{-1}h]$ has bar
 differential $d_B = \Omega_{\mathfrak{sl}_2} \cdot k = \Omega$
 (nonzero). The new feature at $k = 1$: the additional
 quadratic relation $R_{\mathrm{null}}$ introduces a
 new cycle in bar degree~$2$. The element
 $[s^{-1}f \,|\, s^{-1}f]_{\mathrm{null}}$, which in
 $\barB^2(V_1)$ has $d_B = f_{(1)}f = k = 1$
 \textup{(}nondegenerate\textup{)}, is now \emph{modified}:
 the additional relation $:ff: = 0$ in $L_1$ means that
 the mode product $f_{(0)}f$ vanishes in $L_1$, altering the
 bar differential on degree-$2$ elements.

\item \textbf{Collision residue.}
 Unchanged from $V_1$:
 $r(z) = k\,\Omega_{\mathfrak{sl}_2}/z$.

\item \textbf{Finite representation theory.}
 The bar complex $\barB^{\mathrm{ch}}(L_1)$ has the property
 that its cohomology is finitely generated in each
 weight: the null vector truncation ensures that no
 null descendants contribute, and the weight spaces of
 $L_1$ grow more slowly than those of $V_1$.
\end{enumerate}
\end{proposition}

\begin{proof}
(i) Immediate from $v_{\mathrm{null}} = 0$ in $L_1$.

(ii) The bar differential on $\barB^2(L_1)$ is determined by
the OPE of $L_1$, which includes both the standard affine OPE
and the null vector relation $:ff: = 0$. The Casimir element
$\Omega \in \mathfrak{sl}_2 \otimes \mathfrak{sl}_2$ is
unchanged (it depends only on the Lie algebra, not on the
level or the null vector), so the degree-$2$ collision residue
is unaffected. The modification appears in the full bar
differential (not just the collision residue): the vanishing
of $:ff:$ in $L_1$ removes certain bar elements and alters the
chain-level structure while preserving the cohomological data.

(iii) Proposition~\ref{prop:null-truncation}(iii).

(iv) The null vector truncation kills all descendants of
$f_{(-1)}^2|0\rangle$, which form an infinite-dimensional
ideal in $V_1$. The quotient $L_1 = V_1/\mathcal{J}_1$ has
strictly smaller weight spaces, and correspondingly the bar
complex is smaller.
\end{proof}


\subsubsection{Koszul duality at integrable levels: the
 non-quadratic obstruction}
\label{subsubsec:koszul-integrable-obstruction}
\index{Koszul duality!integrable level!obstruction}

\begin{proposition}[$L_k(\mathfrak{sl}_2)$ is not quadratic
 for $k \ge 2$; \ClaimStatusProvedHere]
\label{prop:Lk-not-quadratic}
For integrable level $k \ge 2$, the simple quotient
$L_k(\mathfrak{sl}_2)$ is \emph{not} a quadratic chiral
algebra: the null vector relation
$f_{(-1)}^{k+1}|0\rangle = 0$ has degree $k + 1 \ge 3$ in
the generators. Consequently:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item The GLZ Koszul dual
 \textup{(}Definition~\textup{\ref{def:GLZ_Koszul}}\textup{)}
 is not defined for $L_k$ at $k \ge 2$: the quadratic
 framework requires all relations to be degree~$2$.
\item The Koszul dual of $L_k$ in the $\Ainf$ sense
 \textup{(}via the full bar-cobar adjunction of
 Theorem~A\textup{)} exists but is an $\Ainf$ chiral
 algebra, not necessarily presentable by generators and
 quadratic relations.
\item The line-operator category
 $\cC_{\mathrm{line}}(L_k)$ is well-defined as the
 MC coupling category
 \textup{(}Remark~\textup{\ref{rem:off-koszul-locus}}\textup{)},
 but its algebraic presentation as modules for a strict
 Koszul dual requires the full $\Ainf$ machinery.
\end{enumerate}
\end{proposition}

\begin{proof}
The null vector $f_{(-1)}^{k+1}|0\rangle$ involves $k+1$
applications of $f_{(-1)}$ to the vacuum. In the chiral
tensor algebra $T_{\mathrm{ch}}(V)$, this is an element of
tensor degree $k + 1$. For a quadratic chiral algebra, all
defining relations must lie in $V \otimes_{\mathrm{ch}} V$
(tensor degree~$2$). For $k \ge 2$, $k + 1 \ge 3$ exceeds
the quadratic bound. The three consequences follow from the
structure theory of Koszul duality for non-quadratic algebras
(cf.\ the discussion in~\S\ref{sec:bar_cobar}).
\end{proof}


\subsubsection{The quantum group at roots of unity}
\label{subsubsec:quantum-group-roots}
\index{quantum group!root of unity|textbf}
\index{restricted quantum group|textbf}
\index{Kazhdan--Lusztig equivalence!bar complex|textbf}

At integrable level~$k$, the Kazhdan--Lusztig equivalence
(Proposition~\ref{prop:benchmark-line-operators}) identifies
the braided tensor category of integrable
$\widehat{\mathfrak{sl}}_2$-modules with the representation
category of the quantum group at a root of unity:
\begin{equation}\label{eq:KL-root-unity}
 \mathrm{Rep}^{\mathrm{int}}(L_k(\mathfrak{sl}_2))
 \;\simeq\;
 \mathrm{Rep}(U_q(\mathfrak{sl}_2)),
 \qquad
 q = e^{i\pi/(k+2)}.
\end{equation}
At this root of unity, the quantum group $U_q(\mathfrak{sl}_2)$
has a finite-dimensional quotient: the \emph{restricted}
(or \emph{small}) quantum group $\overline{U}_q(\mathfrak{sl}_2)$
of Lusztig, which is a finite-dimensional Hopf algebra whose
semisimplified representation category has exactly $k + 1$
simple objects, matching the integrable modules
$L_0, L_1, \ldots, L_k$ of $L_k(\mathfrak{sl}_2)$.

\begin{proposition}[What the bar complex sees and does not
 see; \ClaimStatusProvedHere]
\label{prop:bar-sees-quantum-group}
\index{bar complex!quantum group at root of unity}
Let $k \in \Z_{\ge 0}$.
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item \textbf{The bar complex sees the collision residue.}
 The classical $r$-matrix
 $r(z) = k\,\Omega_{\mathfrak{sl}_2}/z$ is the same for both
 $V_k$ and $L_k$, hence the perturbative quantum group
 structure \textup{(}Yangian $Y_\hbar(\mathfrak{sl}_2)$,
 KZ connection\textup{)} is unchanged at genus~$0$.

\item \textbf{The bar complex sees the truncation of
 representations.}
 The bar complex $\barB^{\mathrm{ch}}(L_k)$ classifies
 twisting morphisms $L_k \to L_k^!$, and the category
 of such morphisms is sensitive to the null vector
 truncation. At the module level, this means that the
 line-operator category
 $\cC_{\mathrm{line}}(L_k) = \operatorname{Line}(L_k)$
 has finitely many simple objects \textup{(}$k + 1$ of them\textup{)},
 in contrast with the infinite family for $V_k$.

\item \textbf{The bar complex does NOT directly produce the
 Verlinde algebra.}
 The Verlinde algebra is the categorical center
 $Z(\mathrm{Rep}(L_k))$
 \textup{(}Proposition~\textup{\ref{prop:benchmark-verlinde}}\textup{)}.
 The bar complex classifies twisting morphisms, not
 categorical centers
 \textup{(}the AP-OC distinction of the standing
 conventions\textup{)}.

\item \textbf{The bar complex does NOT directly produce
 the modular $S$-matrix.}
 The $S$-matrix is genus-$1$ data, arising from the
 annulus trace and clutching
 \textup{(}Proposition~\textup{\ref{prop:benchmark-annulus-partition}}\textup{)}.
 The bar complex is a genus-$0$ object; genus-$1$
 structure requires the modular Swiss-cheese extension.

\item \textbf{The KZ monodromy at root of unity produces
 the quantum group $R$-matrix.}
 The KZ connection $\nabla_{\mathrm{KZ}} = d - \Omega\,
 d\log(z)/(k+2)$ on $V_k$-modules restricts to $L_k$-modules,
 and its monodromy at the integrable level $k$ gives the
 braiding of the modular tensor category
 $\mathrm{Rep}(U_q(\mathfrak{sl}_2))$ with
 $q = e^{i\pi/(k+2)}$. This is the content of the
 Drinfeld--Kohno theorem. The bar complex provides
 the perturbative (genus-$0$) avatar of this structure
 via the collision residue $r(z) = k\,\Omega/z$.
\end{enumerate}
\end{proposition}

\begin{proof}
(i) Proposition~\ref{prop:null-truncation}(iii).

(ii) The category $\operatorname{Line}(L_k)$ is defined as the
MC coupling category of
Remark~\ref{rem:off-koszul-locus}. The null vector relation
in $L_k$ forces the vanishing of all modules whose
highest weight exceeds $k$: the Weyl modules
$\mathbb{V}_{k,\lambda}$ for $\lambda > k$ are zero in
$L_k\text{-mod}$, leaving exactly $k + 1$ simple objects.

(iii)--(iv) These are instances of the fundamental distinction
between the bar complex (classifying twisting morphisms) and
the categorical/derived center (classifying bulk observables):
the AP-OC anti-pattern. The Verlinde algebra and the
$S$-matrix belong to the center/trace side; the bar complex
belongs to the coupling/twisting side.

(v) The KZ connection is the degree-$2$ component of the flat
connection on the genus-$0$ moduli space of marked points on
$\PP^1$. Its monodromy is the braiding of the module category.
At integrable level, the Drinfeld--Kohno theorem
(Drinfeld~\cite{Dri89}, Kazhdan--Lusztig~\cite{KL93})
identifies this monodromy with the $R$-matrix of
$U_q(\mathfrak{sl}_2)$. The bar complex contributes the
classical limit: $r(z) = k\,\Omega/z$ is the classical
$r$-matrix of the Yangian, and the KZ parameter
$\hbar = 1/(k+2)$ is the deformation parameter.
\end{proof}


\subsubsection{The conjectural Koszul dual of $L_k$:
 the restricted quantum group}
\label{subsubsec:conjectural-Lk-koszul-dual}

\begin{conjecture}[$\Ainf$ Koszul dual of $L_k(\mathfrak{sl}_2)$]
\label{conj:Lk-koszul-dual}
\index{Koszul dual!L_k@$L_k(\mathfrak{sl}_2)$!conjectural}
\index{restricted quantum group!as Koszul dual}
For integrable level $k \in \Z_{\ge 1}$, the $\Ainf$ Koszul
dual of the simple quotient $L_k(\mathfrak{sl}_2)$
\textup{(}obtained from the full bar-cobar adjunction, not the
quadratic truncation\textup{)} is quasi-isomorphic to the
restricted quantum group:
\begin{equation}\label{eq:conj-Lk-koszul-dual}
 L_k(\mathfrak{sl}_2)^{!,\Ainf}
 \;\simeq\;
 \overline{U}_q(\mathfrak{sl}_2),
 \qquad
 q = e^{i\pi/(k+2)}.
\end{equation}
More precisely:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item The cohomology of the $\Ainf$ bar complex
 $H^\bullet(\barB^{\mathrm{ch}}(L_k))$ is a finite-dimensional
 coalgebra whose linear dual is the restricted quantum group
 $\overline{U}_q(\mathfrak{sl}_2)$.
\item The $\Ainf$ structure on $L_k^{!,\Ainf}$ encodes the
 higher operations of the non-quadratic null vector relation:
 for $k = 1$ \textup{(}quadratic case\textup{)}, the $\Ainf$
 structure is strict \textup{(}$m_{n \ge 3} = 0$\textup{)};
 for $k \ge 2$, the non-quadratic null vector forces nonzero
 higher operations $m_{n}^!$ for $3 \le n \le k + 1$.
\item The restricted representation category matches:
 $\mathrm{Rep}(\overline{U}_q(\mathfrak{sl}_2))^{\mathrm{ss}}$
 has $k + 1$ simple objects, agreeing with the $k + 1$
 integrable modules of $L_k(\mathfrak{sl}_2)$.
\end{enumerate}
\end{conjecture}

\begin{remark}[Evidence and context]
\label{rem:Lk-koszul-evidence}
The conjecture is supported by:
\begin{enumerate}[label=\textup{(\arabic*)},nosep]
\item The Kazhdan--Lusztig equivalence
 (Proposition~\ref{prop:benchmark-line-operators}), which
 identifies the module categories at the braided tensor level.
\item The $k = 1$ case, where $L_1(\mathfrak{sl}_2)$ is
 quadratic and the Koszul dual can be computed directly: the
 restricted quantum group at $q = e^{i\pi/3}$ has
 $\dim \overline{U}_q = 27$ and $2$ simple representations,
 matching the $2$ integrable modules of $L_1$.
\item The general principle that the bar-cobar adjunction
 (Theorem~A) produces the correct dual at the homotopy level,
 even when the quadratic framework breaks down.
\item The parallel with the classical situation: for a
 finite-dimensional semisimple Lie algebra $\fg$, the
 Koszul dual of $U(\fg)$ is $C^\bullet(\fg)$
 (Chevalley--Eilenberg cochains). For the restricted
 enveloping algebra $u(\fg)$ in characteristic~$p$, the
 Koszul dual is the restricted dual $u(\fg)^!$. The
 passage from $V_k \to L_k$ is the chiral analogue of the
 passage from $U(\fg)$ to its finite-dimensional quotient.
\end{enumerate}
The conjecture is open in the sense that the $\Ainf$
bar-cobar computation has not been carried out for $L_k$ at
$k \ge 2$. The obstruction is precisely the non-quadratic
null vector relation: the bar complex of a non-quadratic
algebra requires the full machinery of the $\Ainf$ bar
construction, which is significantly more involved than the
quadratic case. This is flagged in
Remark~\textup{\ref{rem:off-koszul-locus}} as a known
limitation of the current framework.
\end{remark}

\begin{remark}[The dichotomy: universal versus simple]
\label{rem:universal-vs-simple-dichotomy}
\index{universal vertex algebra!versus simple quotient}
The passage $V_k \to L_k$ at integrable levels exhibits a
fundamental dichotomy in the bar-complex world:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\small
\begin{tabular}{lll}
& \textbf{Universal $V_k$} & \textbf{Simple $L_k$} \\
\hline
Relations & Quadratic (OPE only) & Quadratic + degree-$(k{+}1)$
 null \\
Chirally Koszul & Yes (for $k \ne -2$) & Yes for $k = 1$;
 open for $k \ge 2$ \\
Koszul dual & $V_{-k-4}(\mathfrak{sl}_2)$ (infinite-dim.)
 & $\overline{U}_q(\mathfrak{sl}_2)$
 (finite-dim.; conjectural) \\
Simple modules & $\infty$ & $k + 1$ \\
Line category & $V_{-k-4}\text{-mod}$ & $\mathrm{Rep}(U_q)$
 (Kazhdan--Lusztig) \\
Bar complex & Infinite in each weight & Truncated by null vector \\
Collision residue & $k\,\Omega/z$ & $k\,\Omega/z$ (same) \\
\end{tabular}
\end{center}
The universal algebra $V_k$ has cleaner algebraic properties
(quadratic, Koszul, explicit Koszul dual) but infinite
representation theory. The simple quotient $L_k$ has finite
representation theory (the modular tensor category) but
non-quadratic relations for $k \ge 2$. The bar complex of
$V_k$ is the perturbative input; the bar complex of $L_k$ is
the non-perturbative, root-of-unity completion.
\end{remark}

\subsection{Exceptional affine algebras: uniformity of the Jacobi mechanism}
\label{subsec:exceptional-affine}
\index{exceptional Lie algebra!affine|textbf}
\index{shadow depth!exceptional types}
\index{Jacobi identity!exceptional uniformity}

The preceding examples operate within the classical types
($A_n$, $B_n$, $C_n$, $D_n$) and their small-rank representatives
($\mathfrak{sl}_2$, $\mathfrak{sl}_3$, $\mathfrak{gl}_K$). A natural
question is whether the shadow classification is stable across the
full landscape of simple Lie algebras, including the five exceptional
types $G_2$, $F_4$, $E_6$, $E_7$, $E_8$. The present subsection
answers in the affirmative: every affine Kac--Moody algebra
$V_k(\fg)$ with $\fg$ exceptional is class~$\mathbf{L}$ with shadow
depth $r_{\max} = 3$, for exactly the same reason as the classical
types. The new content is not a new mechanism but a
\emph{verification} that the Jacobi mechanism is universal, and the
verification itself exposes structural invariants (dual Coxeter
numbers, Casimir eigenvalues, complementarity constants) that vary
nontrivially across the exceptional family.

\begin{proposition}[Exceptional affine algebras are class~$\mathbf{L}$;
 \ClaimStatusProvedHere]
\label{prop:exceptional-class-L}
For each exceptional simple Lie algebra
$\fg \in \{G_2, F_4, E_6, E_7, E_8\}$ and each level
$k \ne -h^\vee(\fg)$, the affine Kac--Moody algebra $V_k(\fg)$
has the following properties:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item \textbf{OPE.}
 The $\lambda$-bracket is
 $\{J^a_\lambda J^b\} = f^{ab}_c J^c + k\,\delta^{ab}\,\lambda$,
 with maximal pole order $d_{\max} = 2$
 \textup{(}the Killing-form double pole and the
 structure-constant simple pole\textup{)}.

\item \textbf{Collision residue.}
 The bar-kernel absorption gives
 $r(z) = k\,\Omega_\fg/z$, where $\Omega_\fg$ is the quadratic
 Casimir of $\fg$. The $r$-matrix satisfies the classical
 Yang--Baxter equation.

\item \textbf{Shadow depth.}
 $r_{\max} = 3$, class~$\mathbf{L}$: the ternary operation $m_3$
 is nonzero at chain level and vanishes on cohomology by the Jacobi
 identity, and $m_k = 0$ on cohomology for all $k \ge 3$.

\item \textbf{Koszul dual.}
 $V_k(\fg)^! \simeq V_{-k-2h^\vee}(\fg)$ at the Feigin--Frenkel
 dual level, with
 $\kappa(\fg) + \kappa(\fg)^! = 0$.

\item \textbf{Modular characteristic.}
 $\kappa(V_k(\fg)) = \dim(\fg)(k + h^\vee)/(2h^\vee)$.
\end{enumerate}
The numerical data for the five exceptional types is:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\small
\begin{tabular}{lccccccc}
$\fg$ & $\dim$ & $h^\vee$ & $\mathrm{rank}$
 & $\kappa(V_k(\fg))$
 & $|\Delta^+|$
 & $C_2(\mathrm{adj})$
 & $\dim/h^\vee$ \\
\hline
$G_2$ & $14$ & $4$ & $2$
 & $\frac{7(k+4)}{4}$
 & $6$ & $4$ & $7/2$ \\
$F_4$ & $52$ & $9$ & $4$
 & $\frac{26(k+9)}{9}$
 & $24$ & $9$ & $52/9$ \\
$E_6$ & $78$ & $12$ & $6$
 & $\frac{13(k+12)}{4}$
 & $36$ & $12$ & $13/2$ \\
$E_7$ & $133$ & $18$ & $7$
 & $\frac{133(k+18)}{36}$
 & $63$ & $18$ & $133/18$ \\
$E_8$ & $248$ & $30$ & $8$
 & $\frac{62(k+30)}{15}$
 & $120$ & $30$ & $62/15$
\end{tabular}
\end{center}
Here $C_2(\mathrm{adj})$ is the eigenvalue of the quadratic
Casimir on the adjoint representation,
normalised so that $C_2(\mathrm{adj}) = h^\vee$ for any
simple~$\fg$. This universality
(the Casimir eigenvalue equals the dual Coxeter number)
is the root cause of the uniform shadow classification:
it ensures that the collision residue
$r(z) = k\,\Omega_\fg/z$ has the same analytic structure
for all simple types, and that the Jacobi cancellation
mechanism operates identically.
\end{proposition}

\begin{proof}
The OPE of $V_k(\fg)$ depends only on the structure constants
$f^{ab}_c$ and the Killing form $\kappa^{ab} = k\,\delta^{ab}$
(in an orthonormal basis). For any simple $\fg$, the maximal
pole order is $2$ (from the Killing form), the collision residue
is $k\,\Omega_\fg/z$ (by bar-kernel absorption), and the Jacobi
identity $f^{ab}_e f^{ec}_d + \mathrm{cyc} = 0$ kills $m_3$ on
cohomology. The Jacobi identity is a property of the Lie algebra,
not of its root system: it holds for all simple $\fg$, classical
or exceptional. The vanishing of $m_4$ and all higher operations
follows from the pole-structure dichotomy
(Theorem~\ref{thm:pole-structure-dichotomy}): class~$\mathbf{L}$
algebras have $S_4 = 0$, so the shadow obstruction tower terminates.

The Koszul dual identification is the Feigin--Frenkel involution
$k \mapsto -k - 2h^\vee$, which is valid for all simple $\fg$
(Theorem~\ref*{V1-thm:universal-kac-moody-koszul}).
\end{proof}

\begin{remark}[What the exceptional types teach]
\label{rem:exceptional-lesson}
The table above reveals a structural pattern: the dual Coxeter
number $h^\vee$ grows from $4$ ($G_2$) to $30$ ($E_8$), while
the ratio $\dim(\fg)/h^\vee$ varies from $7/2$ ($G_2$) to
$62/15$ ($E_8$), but the shadow class is uniformly
$\mathbf{L}$ and the complementarity sum $\kappa + \kappa'$
is uniformly $0$. The Jacobi identity is the universal mechanism
that stabilises the shadow obstruction tower for all current algebras.

The lesson is sharper than mere stability: it is that the
shadow classification is controlled by the \emph{Lie-algebraic
structure}, not by the root system combinatorics. The
exceptional root systems have markedly different combinatorics
from the classical types: $E_8$ has $120$ positive roots
and a Weyl group of order $696729600$, while
$G_2$ has a hexagonal root system with two distinct root
lengths, yet the bar complex sees none of this.
What it sees is the single fact that $\fg$ satisfies the
Jacobi identity, which kills $m_3$ on cohomology, and that
$d_{\max} = 2$, which prevents $m_4$ from forming. The
entire shadow obstruction tower is determined by two data: the pole order
and the Lie algebra axiom.

The exceptional types also provide the most stringent test of
the DK-0 Laplace bridge
(Proposition~\ref{prop:dk0-laplace-five-families}). The
$\lambda$-bracket of $V_k(E_8)$ involves $248^2 = 61504$
bracket pairs $(J^a, J^b)$; the Laplace transform
$r^{ab}(z) = f^{ab}_c J^c/z + k\,\delta^{ab}/z^2$ must agree
with the collision residue for each pair. This is guaranteed by
the general theory, but the exceptional verification with its
high-dimensional Casimir provides a nontrivial consistency check
on the normalization conventions.
\end{remark}

\subsection{Bershadsky--Polyakov: non-principal DS reduction
 and the subregular shadow}
\label{subsec:bershadsky-polyakov}
\index{Bershadsky--Polyakov algebra|textbf}
\index{DS reduction!non-principal|textbf}
\index{subregular nilpotent|textbf}
\index{shadow depth!subregular}

Every example so far has used either current algebras (class~$\mathbf{L}$)
or \emph{principal} $\mathcal{W}$-algebras (class~$\mathbf{M}$). The
Bershadsky--Polyakov algebra $\mathcal{W}_3^{(2)}$ is the
\emph{simplest non-principal} $\mathcal{W}$-algebra: it is the quantum
Drinfeld--Sokolov reduction of $V_k(\mathfrak{sl}_3)$ with respect to the
\emph{minimal} (= subregular) nilpotent $f_{\min} \in \mathfrak{sl}_3$,
not the principal nilpotent. This example teaches three things that
no previous example could: (a) the role of the nilpotent orbit in
determining shadow depth, (b) the intermediate shadow class between
$\mathbf{L}$ and $\mathbf{M}$, and (c) the failure of the principal
self-duality pattern for non-principal reductions.

\subsubsection{Definition and generators}

\begin{definition}[Bershadsky--Polyakov algebra]
\label{def:bershadsky-polyakov}
The \emph{Bershadsky--Polyakov algebra}
$\mathcal{W}_3^{(2)}$ is the $\mathcal{W}$-algebra associated to
$\mathfrak{sl}_3$ with the minimal nilpotent embedding
$f_{\min}\colon \mathfrak{sl}_2 \hookrightarrow \mathfrak{sl}_3$:
\[
 \mathcal{W}_3^{(2)}
 \;:=\;
 H^0_{\mathrm{DS}}\bigl(V_k(\mathfrak{sl}_3),\; f_{\min}\bigr).
\]
It has three generators: a Virasoro field $T$ of weight~$2$,
and a pair of bosonic fields $G^\pm$ of weight~$3/2$, subject
to the OPE relations
\begin{align}
 T(z)\,T(w) &\;\sim\; \frac{c/2}{(z-w)^4}
 + \frac{2T(w)}{(z-w)^2} + \frac{\partial T(w)}{z-w},
 \label{eq:BP-TT} \\
 T(z)\,G^\pm(w) &\;\sim\; \frac{(3/2)\,G^\pm(w)}{(z-w)^2}
 + \frac{\partial G^\pm(w)}{z-w},
 \label{eq:BP-TG} \\
 G^+(z)\,G^-(w) &\;\sim\;
 \frac{\alpha}{(z-w)^3}
 + \frac{\beta\, T(w)}{(z-w)}
 + \frac{\gamma\, {:}G^+G^-{:}(w)}{(z-w)}
 + \cdots, \label{eq:BP-GG}
\end{align}
where $\alpha$, $\beta$, $\gamma$ are rational functions of the
central charge
$c = c(k) = -2(2k+3)^2/(k+3) + 8$ determined by the level~$k$ of the
parent $V_k(\mathfrak{sl}_3)$. The $G^+G^-$ OPE has a triple pole
\textup{(}pole order $3$\textup{)}, intermediate between the
double pole of current algebras and the quartic pole of the Virasoro.
\end{definition}

\subsubsection{Shadow archetype: the intermediate case}

\begin{proposition}[Bershadsky--Polyakov shadow archetype;
 \ClaimStatusProvedHere]
\label{prop:BP-shadow}
The Bershadsky--Polyakov algebra $\mathcal{W}_3^{(2)}$ has the
following shadow data:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item \textbf{Maximal OPE pole order:} $d_{\max} = 4$
 \textup{(}from the $TT$ channel: the Virasoro subalgebra has a
 quartic pole\textup{)} and $d = 3$ in the $G^+G^-$ channel.

\item \textbf{Shadow class:} Class~$\mathbf{M}$
 \textup{(}$r_{\max} = \infty$\textup{)}, because the Virasoro
 subsector already forces an infinite shadow obstruction tower. However,
 the $G^+G^-$ sector exhibits a novel \emph{intermediate} behaviour
 distinct from both the current-algebra and the Virasoro cases:
 the triple pole of the $G^+G^-$ OPE generates $m_3 \ne 0$ on
 cohomology \textup{(}unlike the double pole of current algebras,
 where Jacobi kills $m_3$\textup{)}, but the degree-$4$ obstruction
 vanishes by the Jacobi identity of the residual
 $\mathfrak{sl}_2$-triple. The $G^\pm$ sector has shadow depth
 exactly $4$: $m_3$ survives, $m_4$ and higher are killed.
 This is class~$\mathbf{C}$ behaviour \textup{(}finite non-trivial
 tower\textup{)} in the $G^\pm$ sector, embedded within
 the overall class~$\mathbf{M}$ forced by the Virasoro subsector.

\item \textbf{Collision residue:}
 The bar-kernel absorption yields:
 \begin{align*}
 r_{TT}(z) &= \frac{c/2}{z^3} + \frac{2T}{z},
 &\quad&\text{(Virasoro, 3 poles),} \\
 r_{TG^\pm}(z) &= \frac{(3/2)\,G^\pm}{z},
 &&\text{(1 pole each),} \\
 r_{G^+G^-}(z) &= \frac{\alpha}{z^2} + \frac{\beta\,T + \gamma\,{:}G^+G^-{:}}{z},
 &&\text{(2 poles).}
 \end{align*}
 The $G^+G^-$ collision residue has an even-order pole $z^{-2}$:
 this is the bar-kernel absorption of the triple OPE pole.

\item \textbf{Modular characteristic:}
 $\kappa(\mathcal{W}_3^{(2)}) = c/2$
 \textup{(}the Virasoro curvature dominates; the half-integer-weight
 generators $G^\pm$ do not contribute independently to $\kappa$
 at the scalar level\textup{)}.

\item \textbf{Complementarity:}
 The Koszul involution for $\mathcal{W}_3^{(2)}$ is
 $c \mapsto c'_{\mathrm{BP}}$, where $c'_{\mathrm{BP}}$ is determined
 by the non-principal Feigin--Frenkel involution for the minimal
 nilpotent. Unlike the principal $\mathcal{W}_3$
 \textup{(}where $c + c' = 100$\textup{)}, the
 Bershadsky--Polyakov complementarity sum depends on the nilpotent
 orbit: $\kappa + \kappa' = 13$ \textup{(}the Virasoro value,
 since $\kappa = c/2$ for both algebra and dual\textup{)}.
\end{enumerate}
\end{proposition}

\begin{proof}
The OPE data follows from the explicit DS reduction
$V_k(\mathfrak{sl}_3) \to \mathcal{W}_3^{(2)}$
at the minimal nilpotent~\cite{Ara07}.

\emph{Shadow classification.} The Virasoro subalgebra has quartic
poles ($TT$ OPE), placing the algebra in class~$\mathbf{M}$.
Verification: that the $G^+G^-$ channel does not independently
generate an infinite tower. The $G^+G^-$ OPE has a triple pole
($d = 3$), which after bar-kernel absorption produces a
collision residue with a double pole ($d_{\mathrm{coll}} = 2$).
At degree~$3$, the $G^+G^-$ channel contributes a nonzero $m_3$
at chain level. The degree-$4$ obstruction
$S_4^{G^\pm} = 0$ because the $G^\pm$ generators transform
under the residual $\mathfrak{sl}_2$-triple
$(e_\alpha, f_\alpha, h_\alpha)$ underlying the minimal
nilpotent embedding: the structure constants of this triple
satisfy the Jacobi identity, and the degree-$4$ Stasheff
obstruction in the $G^\pm$ sector factorises through the
same contraction that vanishes for affine Kac--Moody algebras.
Concretely, $G^+$ and $G^-$ form a doublet under the
residual $\mathfrak{sl}_2$, and their iterated OPE at
degree~$4$ involves the tensor
$\sum_a f^{ab}_e f^{ec}_d$ of the $\mathfrak{sl}_2$ structure
constants, which is killed by Jacobi. Hence the $G^\pm$
sector alone is class~$\mathbf{L}$ (shadow depth $4$); the
infinite tower is driven entirely by the Virasoro subsector.

\emph{Collision residue.} The computation follows from
bar-kernel absorption applied to each OPE channel:
the $TT$ quartic pole gives collision order $3$; the $G^+G^-$
triple pole gives collision order $2$; the $TG^\pm$
double pole gives collision order $1$.

\emph{Modular characteristic.} Since the Virasoro subsector
is the Sugawara field of the residual $\mathfrak{sl}_2$,
$\kappa = c \cdot \varrho$ with $\varrho = 1/2$
(the anomaly ratio of any Sugawara Virasoro), not
$H_N - 1$ (the anomaly ratio of the principal
$\mathcal{W}_N$ where no residual subalgebra survives).
\end{proof}

\begin{remark}[The nilpotent orbit controls the shadow]
\label{rem:BP-nilpotent-orbit}
The contrast between the principal and minimal reductions of
$\mathfrak{sl}_3$ is sharp:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\small
\begin{tabular}{lccccc}
\textbf{Reduction} & \textbf{Nilpotent} & \textbf{Algebra}
 & \textbf{Generators} & $\kappa + \kappa'$ & \textbf{Source of $\mathbf{M}$} \\
\hline
Principal & $f_{\mathrm{prin}}$ & $\mathcal{W}_3$
 & $T$ (wt~2), $W$ (wt~3) & $250/3$ & $T$--$T$ quartic pole \\
Minimal & $f_{\min}$ & $\mathcal{W}_3^{(2)}$
 & $T$ (wt~2), $G^\pm$ (wt~3/2) & $13$ & $T$--$T$ quartic pole
\end{tabular}
\end{center}
Both algebras are class~$\mathbf{M}$ because both contain a Virasoro
subalgebra with quartic poles. But the complementarity sums differ:
$250/3$ for the principal, $13$ for the minimal. The principal
reduction produces a \emph{purely bosonic} algebra with anomaly
ratio $\varrho = H_3 - 1 = 5/6$, while the minimal reduction
produces an algebra with half-integer-weight generators and
anomaly ratio $\varrho = 1/2$ (the Virasoro value).

The reason $\varrho = 1/2$ for the Bershadsky--Polyakov algebra
(rather than $H_3 - 1 = 5/6$) is that the minimal nilpotent
leaves a residual $\mathfrak{sl}_2$ subalgebra of $\mathfrak{sl}_3$
unconstrainted by the DS reduction. The Virasoro generator $T$
of $\mathcal{W}_3^{(2)}$ is the Sugawara field of this residual
$\mathfrak{sl}_2$, not the full Sugawara of $\mathfrak{sl}_3$.
The anomaly ratio of a Sugawara Virasoro is always $1/2$, regardless
of the parent algebra. By contrast, the principal reduction
consumes the entire $\mathfrak{sl}_3$, producing a Virasoro $T$
that is not Sugawara for any subalgebra, and the anomaly ratio
acquires the non-trivial value $H_3 - 1$.

The moral: the nilpotent orbit changes the anomaly ratio, the
complementarity sum, and the fine structure of the shadow obstruction tower,
even when the coarse shadow class ($\mathbf{M}$) is the same.
This is the DS transport
(Theorem~\ref{thm:ds-koszul-obstruction}) at work: the Sugawara
mechanism creates the Virasoro quartic pole from the affine double
pole, but the remaining generators inherit their pole orders from
the specific nilpotent orbit. The Bershadsky--Polyakov algebra
is the simplest example demonstrating that the shadow class
alone does not determine the complementarity sum: two
class~$\mathbf{M}$ algebras from the same parent $V_k(\mathfrak{sl}_3)$
have complementarity sums differing by a factor
$(250/3)/13 = 250/39$.
\end{remark}

\begin{remark}[$E_N$ level reached by $\mathcal{W}_3^{(2)}$]
\label{rem:bp-EN-level}
\index{Bershadsky--Polyakov algebra!$E_N$ level}
\index{E_N ladder@$E_N$ ladder!Bershadsky--Polyakov}
At generic level $k$ (away from the critical and collapsing
loci), the Bershadsky--Polyakov algebra
$\mathcal{W}_3^{(2)} = H^0_{\mathrm{DS}}(V_k(\mathfrak{sl}_3),
f_{\min})$ reaches the following levels:
\begin{enumerate}[label=\textup{(\roman*)}]
\item $\Eone$-chiral
 (Definition~\ref{def:e1-chiral-algebra}): the $T$-$T$
 OPE~\eqref{eq:BP-TT} has a quartic pole and the
 $G^+G^-$ OPE~\eqref{eq:BP-GG} has a triple pole, giving
 $\mathcal{W}_3^{(2)}$ the structure of an $\Einf$-chiral
 algebra, hence \emph{a fortiori} $\Eone$-chiral.

\item $\Etwo$-chiral on the derived center
 (Definition~\ref{def:E2-chiral-algebra}): automatic from
 the chiral Deligne conjecture.

\item $\Etwo$-topological: $\mathcal{W}_3^{(2)}$ inherits
 a conformal vector $T(z)$ (the Sugawara field of the
 residual $\mathfrak{sl}_2$).
 Construction~\ref{constr:topologization} trivializes the
 complex-structure dependence in cohomology.

\item $\Ethree$-chiral: the 3d holomorphic-topological
 theory is holomorphic Chern--Simons with non-principal
 Drinfeld--Sokolov boundary conditions for
 $\mathfrak{sl}_3$ at the minimal nilpotent
 (Costello--Gaiotto).

\item $\Ethree$-topological
 (Definition~\ref{def:E3-topological-algebra}):
 \textbf{proved}
 (Theorem~\ref{thm:E3-topological-DS-general}).
 The 3d HT theory exists (Costello--Gaiotto), the
 conformal vector exists (Sugawara of the residual
 $\mathfrak{sl}_2$), and the BRST identity
 $T_{\mathrm{DS}} = [Q_{\mathrm{CS}}, G'_{f_{\min}}]$
 holds on $Q_{\mathrm{CS}}$-cohomology. The two
 qualitative differences from the principal case---the
 non-Koszul DS complex and the non-abelian residual Levi
 factor---are irrelevant for the topologization argument,
 which operates entirely in the $3$d bulk
 (Remark~\ref{rem:non-principal-obstructions-illusory}).
 The improvement term involves only Cartan currents
 $\partial J_{h_j}$, all of which are
 $Q_{\mathrm{CS}}$-exact in the bulk regardless of the
 nilpotent orbit.
\end{enumerate}
Shadow class: $\mathbf{M}$ (the Virasoro subsector has a
quartic pole; $m_k \ne 0$ for all $k \ge 3$ by
Theorem~\ref{thm:ds-koszul-obstruction}; the bar complex is
non-formal with infinite $A_\infty$ tower). The $G^\pm$
sector alone is class~$\mathbf{C}$ with shadow depth~$4$
(Proposition~\ref{prop:BP-shadow}).
\end{remark}

\subsection{The \texorpdfstring{$N{=}2$}{N=2} superconformal algebra:
 fermionic generators and spectral flow}
\label{subsec:n2-sca}
\index{superconformal algebra!N=2@$N{=}2$|textbf}
\index{spectral flow|textbf}
\index{shadow depth!superconformal}

Every preceding example has been purely bosonic: the generators have
integer conformal weight and commuting (not anticommuting) OPE
residues. The $N{=}2$ superconformal algebra (SCA) is the simplest
vertex algebra with \emph{fermionic} generators. It teaches three
things: (a) how the bar complex handles generators of half-integer
weight, (b) how spectral flow (an automorphism exchanging Ramond and
Neveu--Schwarz sectors) acts on the Koszul dual, and
(c) how extended supersymmetry constrains the shadow obstruction tower.

\subsubsection{Generators and OPE}

The $N{=}2$ SCA at central charge $c$ has four generators:
\begin{itemize}[nosep]
\item $T(z)$ of conformal weight $2$ (the stress-energy tensor),
\item $J(z)$ of conformal weight $1$ (the $\mathrm{U}(1)$ R-current),
\item $G^+(z)$ of conformal weight $3/2$ (the supercharge, odd),
\item $G^-(z)$ of conformal weight $3/2$ (the conjugate supercharge, odd).
\end{itemize}
The defining OPE relations are:
\begin{align}
 T(z)\,T(w) &\;\sim\; \frac{c/2}{(z-w)^4}
 + \frac{2T}{(z-w)^2} + \frac{\partial T}{z-w},
 \label{eq:N2-TT} \\
 J(z)\,J(w) &\;\sim\; \frac{c/3}{(z-w)^2},
 \label{eq:N2-JJ} \\
 T(z)\,G^\pm(w) &\;\sim\;
 \frac{(3/2)\,G^\pm}{(z-w)^2} + \frac{\partial G^\pm}{z-w},
 \label{eq:N2-TG} \\
 J(z)\,G^\pm(w) &\;\sim\; \frac{\pm G^\pm}{z-w},
 \label{eq:N2-JG} \\
 G^+(z)\,G^-(w) &\;\sim\;
 \frac{c/3}{(z-w)^3} + \frac{J}{(z-w)^2}
 + \frac{T + \frac{1}{2}\partial J}{z-w}.
 \label{eq:N2-GG}
\end{align}
The OPE $G^+(z)\,G^-(w)$ has a triple pole (pole order $3$), while
$T(z)\,T(w)$ has a quartic pole.

\subsubsection{Shadow archetype and collision residue}

\begin{proposition}[$N{=}2$ SCA shadow data; \ClaimStatusProvedHere]
\label{prop:N2-shadow}
The $N{=}2$ superconformal algebra at central charge $c$ has:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item \textbf{Maximal OPE pole order:} $d_{\max} = 4$
 \textup{(}from the $TT$ OPE\textup{)}.

\item \textbf{Shadow class:} Class~$\mathbf{M}$
 \textup{(}$r_{\max} = \infty$\textup{)}. The Virasoro subsector
 alone forces the infinite shadow obstruction tower; the fermionic
 generators do not introduce independent non-formality.

\item \textbf{Multi-channel collision residue:}
 After bar-kernel absorption \begin{align*}
 r_{TT}(z) &= \frac{c/2}{z^3} + \frac{2T}{z},
 &\quad&\text{(Virasoro),} \\
 r_{JJ}(z) &= \frac{c/3}{z},
 &&\text{(Heisenberg-type),} \\
 r_{G^+G^-}(z) &= \frac{c/3}{z^2} + \frac{J}{z},
 &&\text{(2 poles),} \\
 r_{TG^\pm}(z) &= \frac{(3/2)\,G^\pm}{z},
 &&\text{(1 pole each),} \\
 r_{JG^\pm}(z) &= 0,
 &&\text{(simple pole fully absorbed).}
 \end{align*}
 The $JG^\pm$ channel has vanishing collision residue: the simple pole
 $\pm G^\pm/(z-w)$ in the OPE is fully absorbed by the $d\log(z-w)$
 bar kernel, reducing the pole order to zero. This is the first
 example where an OPE channel is nontrivial yet produces
 \emph{zero} collision residue, a phenomenon specific to
 simple-pole channels with no higher-order singularity. The
 threshold for nonzero collision residue is
 $d_{\mathrm{OPE}} \ge 2$.

\item \textbf{Modular characteristic:}
 The modular characteristic is
 \[
 \kappa(N{=}2,\, c) \;=\; \frac{6-c}{2(3-c)}
 \;=\; \frac{k+4}{4},
 \]
 where $c = 3k/(k+2)$. This is proved by the Kazama--Suzuki coset
 decomposition \textup{(}Vol~I, Theorem~\ref{V1-thm:n2-kappa}\textup{)}:
 $\kappa(N{=}2) = \kappa(\widehat{\mathfrak{sl}}_{2,k})
 + \kappa(\psi\bar\psi) - \kappa(\widehat{U(1)})
 = 3(k{+}2)/4 + 1/2 - (k/2{+}1) = (k{+}4)/4$.
 At $c = 1$ \textup{(}$k = 1$\textup{)}: $\kappa = 5/4$.
 At $c = 6$ \textup{(}$k = -4$, critical level\textup{)}:
 $\kappa = 0$.

 \textbf{Caution} the $N{=}2$
 modular characteristic is NOT $c/2$. The $N{=}2$ SCA is a coset,
 and the coset subtraction reduces $\kappa$ below the Virasoro
 value $c/2$.

\item \textbf{Complementarity:}
 The $N{=}2$ Koszul involution is $c \mapsto 6 - c$
 \textup{(}equivalently $k \mapsto -k - 4$, the
 Feigin--Frenkel involution of the underlying
 $\mathfrak{sl}_2$\textup{)}. The complementarity sum is constant:
 $\kappa(c) + \kappa(6-c) = 1$.

\item \textbf{Spectral flow and the Koszul dual.}
 The spectral flow automorphism $\sigma$ acts on the $N{=}2$ SCA
 by $\sigma(G^+_r) = G^+_{r+1}$, $\sigma(G^-_r) = G^-_{r-1}$,
 $\sigma(J_n) = J_n + (c/3)\delta_{n,0}$, exchanging the Ramond
 \textup{(}$r \in \Z$\textup{)} and Neveu--Schwarz
 \textup{(}$r \in \Z + 1/2$\textup{)} sectors. On the bar complex,
 spectral flow acts as a grading-shift automorphism
 $\sigma^* \colon \barB^{\mathrm{ch}}(\cA) \to \barB^{\mathrm{ch}}(\cA)$
 that preserves the dg-coalgebra structure but shifts the
 internal grading: a bar element
 $[s^{-1}G^+_{r_1} \,|\, \cdots \,|\, s^{-1}G^+_{r_n}]$
 is mapped to
 $[s^{-1}G^+_{r_1+1} \,|\, \cdots \,|\, s^{-1}G^+_{r_n+1}]$.
 The bar differential commutes with~$\sigma^*$ because spectral
 flow is a vertex algebra automorphism. Consequently, the Koszul
 dual is equivariant: $\sigma^*(\cA^!) \simeq (\sigma^*\cA)^!$.
 At the module level, spectral flow permutes the NS and R
 sectors of the line-operator category, providing a
 $\Z$-action on $\cC_{\mathrm{line}}$ that is invisible
 in purely bosonic examples.
\end{enumerate}
\end{proposition}

\begin{proof}
The OPE data is standard. The shadow classification follows from
the presence of the Virasoro subalgebra with quartic poles. The
collision residues are computed channel by channel using: the
$d\log$ bar kernel absorbs one pole order from each OPE, so the
$TT$ quartic pole becomes a triple collision pole, the $G^+G^-$
triple pole becomes a double collision pole, the $JJ$ double pole
becomes a simple collision pole, and the $JG^\pm$ simple pole is
fully absorbed. The modular characteristic is computed from the
total Virasoro central charge via
$\kappa = c/2$ (Proposition~\ref{prop:benchmark-modular-data}).
Spectral flow equivariance of the Koszul dual follows from the
functoriality of the bar construction under vertex algebra
automorphisms.
\end{proof}

\begin{remark}[Koszulness of the $N{=}2$ SCA]
\label{rem:N2-koszulness}
\index{superconformal algebra!N=2@$N{=}2$!Koszulness}
\index{Koszulness!N=2 superconformal@$N{=}2$ superconformal}
The $N{=}2$ superconformal algebra is chirally Koszul at all
levels, by PBW universality
(Volume~I, Proposition~\ref{V1-prop:pbw-universality}).
The $N{=}2$ SCA is freely strongly generated
(Adamovi\'c~\cite{Adamovic99}): there are no normally ordered
relations among $T$, $J$, $G^+$, $G^-$ and their derivatives.
By the PBW criterion, free strong generation implies PBW
spectral sequence collapse at $E_2$, which is equivalent to
chiral Koszulness
(Volume~I, Theorem~\ref{V1-thm:koszul-equivalences-meta}).
This holds for the \emph{universal} $N{=}2$ algebra at all
levels; the simple quotient at admissible levels has Koszulness
open for the same bar-Ext vs ordinary-Ext reason as for
$W$-algebras of rank~$\ge 2$.
The rank-dependent picture for affine algebras is now sharp:
$L_k(\mathfrak{sl}_2)$ is chirally Koszul at \emph{all}
admissible levels (structural argument from the single-weight
null vector and the Kac--Wakimoto character formula), while
$L_k(\fg)$ for $\mathrm{rk}(\fg) \geq 2$ at admissible
levels with denominator $q \geq 3$ is \emph{not} chirally
Koszul, with $\dim H^2(\barB(L_k(\fg))) \geq \mathrm{rk}(\fg)$
(Volume~I,
Conjecture~\ref*{V1-conj:admissible-koszul-rank-obstruction}).
\end{remark}

\begin{remark}[What fermionic generators teach about the bar complex]
\label{rem:N2-fermionic-lesson}
The $N{=}2$ SCA is the first example where the bar complex involves
\emph{odd} desuspensions: the elements $s^{-1}G^\pm$ in bar
degree~$1$ have odd total parity (the desuspension shifts by $-1$,
and the generators are odd). The bar differential on
$[s^{-1}G^+ \,|\, s^{-1}G^-]$ in bar degree~$2$ involves the
$G^+G^-$ OPE, which produces even fields ($T$ and $J$) from odd
inputs. The sign conventions for the bar differential on
mixed-parity bar elements are precisely those of
Appendix~\ref{app:brace-signs}, instantiated here for the first
time on a concrete example.

The practical consequence: the $G^+G^-$ channel of the $r$-matrix
has an even-order pole $z^{-2}$ (from the triple OPE pole after
absorption), while the $TT$ channel has only odd-order poles
($z^{-3}$ and $z^{-1}$). The parity of the collision pole order
correlates with the statistics of the generators: even-order
collision poles arise from opposite-statistics pairs. This is a
general pattern: for a fermion-fermion OPE
$\psi(z)\psi(w) \sim \sum c_n/(z-w)^n$, locality requires the
OPE to be symmetric under $z \leftrightarrow w$ (since the
generators anticommute), forcing the OPE pole orders to have a
definite parity. The bar kernel absorption then shifts the
parity, producing even-order collision poles from
odd-statistics pairs. The Bershadsky--Polyakov algebra
\textup{(}\S\ref{subsec:bershadsky-polyakov}\textup{)} confirms
this pattern: its half-integer-weight $G^\pm$ generators produce
an even-order collision pole $z^{-2}$ in the $G^+G^-$ channel.
\end{remark}

\subsection{The small \texorpdfstring{$N{=}4$}{N=4} SCA: extended
 supersymmetry and R-symmetry constraints}
\label{subsec:n4-sca}
\index{superconformal algebra!N=4 small@$N{=}4$ small|textbf}
\index{R-symmetry!shadow constraints}
\index{extended supersymmetry!bar complex}

The small $N{=}4$ superconformal algebra doubles the
supercharges of the $N{=}2$ SCA and promotes the $\mathrm{U}(1)$
R-symmetry to $\mathrm{SU}(2)$. This example teaches how
\emph{extended R-symmetry} provides additional cancellations in the
shadow obstruction tower. Unlike the $N{=}2$ case, where the infinite shadow
tower is driven by the Virasoro subsector alone, the $N{=}4$
R-symmetry current algebra imposes Jacobi-type constraints that
partially tame the higher-degree operations.

\subsubsection{Generators and OPE}

The small $N{=}4$ SCA at central charge $c = 6k$ (with $k$ a
positive integer for unitary representations) has generators:
\begin{itemize}[nosep]
\item $T(z)$: Virasoro, weight $2$.
\item $J^a(z)$, $a = 1,2,3$: $\widehat{\mathfrak{su}}_2$
 R-symmetry currents, weight $1$, level $k$.
\item $G^{\alpha i}(z)$, $\alpha = \pm$, $i = 1,2$:
 four supercharges, weight $3/2$, transforming in the
 $\mathbf{2} \otimes \mathbf{2}$ of
 $\mathrm{SU}(2)_R \times \mathrm{SU}(2)_{\mathrm{outer}}$.
\end{itemize}
The total number of generators is $1 + 3 + 4 = 8$. The defining
OPE includes the $\mathfrak{su}_2$ current algebra at level $k$:
\begin{equation}\label{eq:N4-JJ}
 J^a(z)\,J^b(w)
 \;\sim\;
 \frac{k\,\delta^{ab}}{(z-w)^2}
 + \frac{i\epsilon^{abc}\,J^c(w)}{z-w},
\end{equation}
and the supercharge OPE:
\begin{equation}\label{eq:N4-GG}
 G^{\alpha i}(z)\, G^{\beta j}(w)
 \;\sim\;
 \frac{c/3 \cdot \epsilon^{\alpha\beta}\delta^{ij}}{(z-w)^3}
 + \frac{\text{R-currents}}{(z-w)^2}
 + \frac{T + \text{composites}}{z-w}.
\end{equation}

\subsubsection{Shadow archetype and R-symmetry constraint}

\begin{proposition}[Small $N{=}4$ SCA shadow data;
 \ClaimStatusProvedHere]
\label{prop:N4-shadow}
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item \textbf{Shadow class:} Class~$\mathbf{M}$
 \textup{(}$r_{\max} = \infty$\textup{)}, driven by the Virasoro
 quartic pole.

\item \textbf{R-symmetry constraint on the shadow obstruction tower:}
 The $\mathfrak{su}_2$ R-symmetry at level $k$ satisfies
 the Jacobi identity, which provides cancellations in the
 supercharge sector. Specifically, the degree-$3$ obstruction
 in the $G^{\alpha i}$ sector factorises through the $\mathfrak{su}_2$
 structure constants $\epsilon^{abc}$:
 \[
 m_3(G^{\alpha i},\, G^{\beta j},\, G^{\gamma k})
 \;\propto\;
 \epsilon^{abc}\, \epsilon_{abc}\,
 (\text{R-symmetry Jacobi})
 \;=\; 0
 \quad\text{on cohomology,}
 \]
 by the Jacobi identity of $\mathfrak{su}_2$.
 The supercharge sector is therefore class~$\mathbf{L}$ on its own;
 the class~$\mathbf{M}$ behaviour is driven entirely by the
 Virasoro subsector.

\item \textbf{Collision residue:}
 The R-symmetry channel has collision residue
 $r_{J^aJ^b}(z) = k\,\delta^{ab}/z$
 \textup{(}the double pole is reduced to a simple pole by
 bar-kernel absorption; the structure-constant simple pole
 $i\epsilon^{abc}J^c/(z-w)$ is fully absorbed\textup{)}.
 The supercharge channel has
 $r_{G^{\alpha i}G^{\beta j}}(z)
 = (c/3)\epsilon^{\alpha\beta}\delta^{ij}/z^2 + (\text{R-currents})/z$.

\item \textbf{Modular characteristic:}
 $\kappa(N{=}4,\, c) = c/3 = 2k$, where $c = 6k$.
 At $c = 6$ \textup{(}$k = 1$, the K3 level\textup{)}:
 $\kappa = 2$. This is strictly less than
 $\kappa(\mathrm{Vir}_6) = 3$: the $N{=}4$ supersymmetry
 reduces $\kappa$ by a factor of $2/3$ The formula $\kappa = 2k$ is $5$-path verified:
 geometric \textup{(}$\kappa = \dim_{\C}(K3)$ at $k = 1$\textup{)},
 genus-$1$ character, complementarity, Hodge bundle, and
 $N{=}4$ Ward identity
 \textup{(}Vol~I, \S\ref{ch:toroidal-elliptic}\textup{)}.

\item \textbf{Koszul dual:}
 The Koszul involution preserves the $N{=}4$ structure:
 $(\text{small $N{=}4$ SCA at $c$})^!
 = \text{small $N{=}4$ SCA at $c'$}$,
 where $c' = -c - 24$ is determined by the
 Feigin--Frenkel involution
 $k' = -k - 2h^\vee(\mathfrak{su}_2) = -k - 4$ via
 $c = 6k$. The complementarity sum is
 $\kappa + \kappa' = 2k + 2(-k-4) = -8$
 \textup{(}W-algebra type, not KM-type\textup{)}.
 The Koszul self-dual point is $c^* = -12$
 \textup{(}$k^* = -2 = -h^\vee$, the critical level of
 the R-symmetry current algebra\textup{)}. The unitary
 regime $c \ge 6$ \textup{(}$k \ge 1$\textup{)} maps to
 $c' \le -30$, deep in the non-unitary region.

 \textbf{Caution}: for the K3 sigma model specifically,
 the geometric Verdier duality gives $\kappa' = -\kappa = -2$
 and $\kappa + \kappa' = 0$. This uses the CY sigma-model
 duality $k \mapsto -k$ \textup{(}not the FF involution
 $k \mapsto -k - 4$\textup{)}, and applies only to the
 geometric realization, not to the abstract $N{=}4$ SCA family.
\end{enumerate}
\end{proposition}

\begin{proof}
The shadow class follows from the Virasoro subsector as in the
$N{=}2$ case. The R-symmetry constraint: the degree-$3$
obstruction in the supercharge sector involves the iterated OPE
of three supercharges mediated by the R-symmetry structure
constants. The relevant contraction has the form
$\sum_{a,b} f^{ab}_{\ \ c}\, f^{bc}_{\ \ d}\, f^{ca}_{\ \ e}$
where $f^{ab}_{\ \ c} = i\epsilon^{abc}$ are the
$\mathfrak{su}_2$ structure constants. The Jacobi identity
$f^{ab}_{\ \ e} f^{ec}_{\ \ d} + f^{bc}_{\ \ e} f^{ea}_{\ \ d}
+ f^{ca}_{\ \ e} f^{eb}_{\ \ d} = 0$ kills this contraction.
This is the same mechanism that kills $m_3$ for affine
Kac--Moody algebras
(Proposition~\ref{prop:exceptional-class-L}), now applied to the
R-symmetry subalgebra within the superconformal algebra. The
Koszul dual follows from the Feigin--Frenkel involution
$k \mapsto -k - 2h^\vee(\mathfrak{su}_2) = -k - 4$ applied
to the R-symmetry level, giving $c' = 6(-k-4) = -c - 24$.
\end{proof}

\begin{remark}[Extended supersymmetry as partial formality]
\label{rem:N4-partial-formality}
\index{partial formality!N=4@$N{=}4$}
The small $N{=}4$ SCA illustrates a phenomenon we call
\emph{partial formality}: the algebra as a whole is
class~$\mathbf{M}$ (non-formal), but the supercharge sector
alone is class~$\mathbf{L}$ (formal). The non-formality is
\emph{localised} in the Virasoro subsector. This localisation is a
consequence of extended supersymmetry: the R-symmetry Jacobi
identity provides the same cancellation mechanism as the Lie algebra
Jacobi identity in affine Kac--Moody, but only for the generators
charged under $\mathrm{SU}(2)_R$. The uncharged Virasoro generator
$T$ escapes this constraint and produces the infinite tower.

The contrast with the $N{=}2$ SCA is instructive. In the $N{=}2$
case, the $\mathrm{U}(1)$ R-current has no Jacobi identity (it is
abelian), so the supercharge sector inherits no additional
cancellations. In the $N{=}4$ case, the non-abelian R-symmetry
$\mathfrak{su}_2$ provides the Jacobi mechanism, partially taming
the tower. The general principle: the larger the R-symmetry
algebra, the stronger the constraints on the shadow obstruction tower.
This predicts that algebras with $N \ge 4$ extended supersymmetry
will have increasingly constrained shadow structure, with the
non-formality progressively localised in the Virasoro subsector.

\emph{The moral for the bar complex:} one can decompose the shadow
tower of any vertex algebra with an R-symmetry subalgebra
$\fg_R$ into a ``tame'' part (controlled by the Jacobi identity
of $\fg_R$, class~$\mathbf{L}$) and a ``wild'' part (controlled
by the Virasoro quartic pole, class~$\mathbf{M}$). The tame
part truncates the bar complex in the supercharge sector;
the wild part generates the infinite gravitational tower in
the Virasoro sector. This decomposition is
visible only in examples with non-abelian R-symmetry;
it is invisible in the $N{=}2$ SCA (abelian R-symmetry)
and in all purely bosonic examples.
\end{remark}

\subsection{Coulomb branch VOAs and CoHA: the non-Lagrangian frontier}
\label{subsec:coulomb-coha}
\index{Coulomb branch VOA|textbf}
\index{cohomological Hall algebra|textbf}
\index{BPS algebra|textbf}
\index{non-Lagrangian!bar complex}

Every worked example so far has been defined by a Lagrangian
(an action functional): the boundary algebra arises as the BRST
cohomology of explicit fields with explicit OPE. The present
subsection crosses a qualitative threshold: the Coulomb branch VOAs
and cohomological Hall algebras (CoHAs) are \emph{non-Lagrangian}
in origin. They are defined by their representation theory (as limits
of gauge-theoretic constructions or as cohomological invariants of
moduli spaces), and no closed-form OPE is available in general.

The question for the bar-cobar framework is: \emph{what can be said
about an algebra that is known only through its abstract properties,
not through explicit OPE data?} The answer is that the shadow
classification, the Koszul dual, and the modular characteristic are
all determined by structural data (pole orders, central charge,
Lie-algebraic ancestry) that survive the passage from Lagrangian to
non-Lagrangian. The explicit $r$-matrix and genus tower require
OPE data, but their existence is guaranteed by the general framework.

\subsubsection{Coulomb branch VOAs}

\begin{definition}[Coulomb branch VOA; following~\cite{SV13}]
\label{def:coulomb-branch-voa}
Let $G$ be a reductive group and $\mathbf{N}$ a representation of~$G$.
The \emph{Coulomb branch VOA} $\mathcal{V}_{G,\mathbf{N}}$ is the
vertex algebra associated to the Coulomb branch of the 3d $\mathcal{N}=4$
gauge theory with gauge group $G$ and matter in representation
$\mathbf{N}$, obtained via the holomorphic-topological twist.
\end{definition}

For $G = \mathrm{GL}_n$ and $\mathbf{N} = T^*\mathrm{Rep}(Q)$
(the cotangent bundle of the representation space of a quiver $Q$),
the Coulomb branch VOA is related to the $\mathcal{W}$-algebra
$\mathcal{W}_{1+\infty}$ and its truncations by the
Alday--Gaiotto--Tachikawa correspondence. In the simplest case
$G = \mathrm{GL}_1$, $\mathbf{N} = \C^N$: the Coulomb branch
VOA is the $\beta\gamma$ system of $N$ copies, with
$\kappa = 0$ and class~$\mathbf{G}$ (the freest possible case).

\begin{proposition}[Shadow classification of Coulomb branch VOAs;
 \ClaimStatusProvedHere]
\label{prop:coulomb-shadow}
The Coulomb branch VOA $\mathcal{V}_{G,\mathbf{N}}$ inherits its
shadow class from the vertex-algebraic structure:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item If $\mathcal{V}_{G,\mathbf{N}}$ is a free-field algebra
 \textup{(}$\beta\gamma$ or Heisenberg type\textup{)}:
 class~$\mathbf{G}$, $r_{\max} = 2$.

\item If $\mathcal{V}_{G,\mathbf{N}}$ is a current algebra
 \textup{(}affine Kac--Moody type\textup{)}:
 class~$\mathbf{L}$, $r_{\max} = 3$.

\item If $\mathcal{V}_{G,\mathbf{N}}$ contains a Virasoro
 or $\mathcal{W}$-algebra factor
 \textup{(}which occurs when the Coulomb branch has nontrivial
 geometry\textup{)}: class~$\mathbf{M}$, $r_{\max} = \infty$.
\end{enumerate}
In each case, the Koszul dual $\mathcal{V}_{G,\mathbf{N}}^!$
exists by the general bar-cobar adjunction
\textup{(}Theorem~A of Volume~I\textup{)}, and the modular
characteristic is computed from the central charge via the
standard formula.
\end{proposition}

\begin{proof}
The shadow classification depends only on the maximal OPE pole
order of the vertex algebra, not on the physical origin of the
algebra. A Coulomb branch VOA is a vertex algebra; its OPE data
determines the shadow class by the pole-structure dichotomy
(Theorem~\ref{thm:pole-structure-dichotomy}). The three cases
correspond to $d_{\max} = 1$ (free fields), $d_{\max} = 2$
(current algebras), and $d_{\max} \ge 4$ ($\mathcal{W}$-algebras),
respectively.
\end{proof}

\begin{example}[The ADHM Coulomb branch: from geometry to shadow class]
\label{ex:adhm-coulomb}
\index{ADHM quiver!Coulomb branch}
The ADHM quiver with gauge group $G = \mathrm{GL}_n$ and $N$
fundamental hypermultiplets produces a Coulomb branch whose VOA
is the $\mathcal{W}$-algebra $\mathcal{W}_n$
at a level determined by $N$. This provides the sharpest
illustration of the non-Lagrangian passage: the Coulomb branch
is defined as an algebraic variety
$\mathcal{M}_C = \Spec(\C[\mathcal{M}_H]^G)$
(the categorical quotient of the Higgs branch), but the
associated VOA is the DS reduction
$\mathcal{W}_n = H^0_{\mathrm{DS}}(V_k(\mathfrak{gl}_n))$.
The bar complex of $\mathcal{W}_n$ inherits the shadow data:
class~$\mathbf{M}$ for $n \ge 2$ (the Virasoro quartic pole is
unavoidable once the DS reduction creates a stress tensor).
The passage from the geometric data (moduli space, quiver,
hyperk\"ahler metric) to the algebraic data (shadow class,
Koszul dual, $r$-matrix) is mediated entirely by the VOA
functor.
\end{example}

\subsubsection{Cohomological Hall algebras and BPS algebras}

The cohomological Hall algebra (CoHA)
\cite{kontsevich-soibelman,SV13}
provides a second class of non-Lagrangian inputs to the bar-cobar
framework. The CoHA of a quiver $Q$ is the algebra
$\mathcal{H}_Q = \bigoplus_{d \ge 0} H^*_G(\mathcal{M}_d(Q))$,
where $\mathcal{M}_d(Q)$ is the moduli stack of representations of
$Q$ of dimension vector $d$ and $G = \prod \mathrm{GL}_{d_i}$ is the
gauge group. The CoHA product is the \emph{Hall multiplication},
defined by correspondences in the moduli space.

The connection to the bar-cobar framework passes through the
\emph{BPS algebra}: the primitive part of the CoHA under the
coproduct is a graded Lie algebra
$\mathfrak{g}_Q^{\mathrm{BPS}}$, whose enveloping algebra
reconstructs the full CoHA. The passage
$\mathcal{H}_Q \twoheadrightarrow U(\mathfrak{g}_Q^{\mathrm{BPS}})$
is the CoHA analogue of the PBW theorem, and the bar-cobar
adjunction (Theorem~A) applies to the enveloping algebra.

\begin{proposition}[CoHA and the bar-cobar framework;
 \ClaimStatusHeuristic]
\label{prop:coha-bar-cobar}
Let $Q$ be a symmetric quiver \textup{(}$\dim \mathrm{Ext}^1 = \dim \mathrm{Hom}$
by Serre duality on the Calabi--Yau threefold\textup{)}.
The associated BPS algebra $\mathfrak{g}_Q^{\mathrm{BPS}}$
\textup{(}the primitive part of the CoHA under the coproduct\textup{)}
has the following relationship to the bar-cobar framework:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item The BPS algebra $\mathfrak{g}_Q^{\mathrm{BPS}}$ is the
 \emph{Lie algebra} underlying a vertex algebra
 $\mathcal{V}_Q^{\mathrm{BPS}}$, whose chiral product is
 defined by the Hall multiplication lifted to the chiral level.

\item The shadow class of $\mathcal{V}_Q^{\mathrm{BPS}}$ is
 determined by the quiver: for quivers with only double arrows
 \textup{(}Jordan quiver, ADHM quiver\textup{)}, the BPS vertex
 algebra is class~$\mathbf{L}$; for quivers with loops
 \textup{(}one-loop quiver\textup{)}, the BPS vertex algebra
 is class~$\mathbf{G}$.

\item The Koszul dual of $\mathcal{V}_Q^{\mathrm{BPS}}$ is
 conjectured to be the BPS algebra of the \emph{dual quiver}
 $Q^!$ \textup{(}obtained by reversing all arrows and dualising
 the framing\textup{)}.

\item The $r$-matrix of $\mathcal{V}_Q^{\mathrm{BPS}}$ is the
 rational $r$-matrix of the Maulik--Okounkov Yangian
 $Y_{Q}$~\cite{SV13}, whose RTT presentation is determined by
 the BPS spectrum.
\end{enumerate}
\end{proposition}

\begin{example}[The Jordan quiver CoHA and the affine Yangian]
\label{ex:jordan-coha}
\index{Jordan quiver!CoHA}
\index{affine Yangian!from CoHA}
The Jordan quiver $Q_J$ has a single vertex and a single loop.
Its CoHA is
$\mathcal{H}_{Q_J} = \bigoplus_{n \ge 0}
H^*_{\mathrm{GL}_n}(\mathfrak{gl}_n)$
with Hall multiplication given by the Hecke correspondence.
The BPS Lie algebra is a copy of the positive part
of the affine Yangian $Y_\hbar(\widehat{\mathfrak{gl}}_1)$
\cite{SV13}. The associated vertex algebra is
$\mathcal{W}_{1+\infty}$ at a specific level determined
by the equivariant parameters, placing it squarely in
class~$\mathbf{M}$ (it contains a Virasoro subalgebra with
quartic poles). The $r$-matrix is the rational function
$r(z) = k\,\Omega/z$ of the Yangian, where $\Omega$ is the
Casimir element of $\widehat{\mathfrak{gl}}_1$ acting on
the equivariant cohomology and $k$ is the level.

This example demonstrates the full cycle: a moduli-space
construction (the equivariant cohomology of
$\mathcal{M}_n = \mathfrak{gl}_n /\!\!/ \mathrm{GL}_n$)
produces a Hall algebra, whose primitive part gives a Lie
algebra, whose enveloping algebra is a vertex algebra, whose
bar complex falls into the shadow classification, whose
$r$-matrix is the Yangian's. The bar-cobar framework provides
the algebraic bridge from moduli-space geometry to quantum group
data, without ever requiring an explicit Lagrangian.
\end{example}

\begin{remark}[What non-Lagrangian examples teach]
\label{rem:non-lagrangian-lesson}
\index{non-Lagrangian!lesson for bar-cobar}
The Coulomb branch VOAs and CoHA/BPS algebras demonstrate that
the bar-cobar framework is not limited to field-theoretic inputs.
The essential data is a vertex algebra with its OPE; the shadow
classification, Koszul dual, and modular characteristic are
vertex-algebraic invariants, not field-theoretic ones. The physical
origin determines which vertex algebra appears, but the algebraic
machinery operates independently of the physical construction.

The specific lesson: for non-Lagrangian theories where no explicit
action is available, the bar-cobar framework provides the correct
algebraic substitute for the BV-BRST quantisation. The bar complex
$\barB(\mathcal{V}_{G,\mathbf{N}})$ classifies twisting morphisms
of the Coulomb branch VOA, even when no Lagrangian description of
the 3d theory exists. This is the algebraic core of the
vertex-algebraic approach to non-Lagrangian QFT.

The contrast with the Lagrangian examples is instructive. For
the Lagrangian theories
(\S\S\ref{subsec:free-chiral}--\ref{subsec:twisted-M5-primitive}),
the $\Ainf$ operations $m_k$ are computed from explicit Feynman
integrals, and the shadow class is read off from the interaction
vertices. For the non-Lagrangian theories, the shadow class is
determined by the abstract vertex-algebraic structure (maximal
pole order), and the $\Ainf$ operations exist by the general
bar-cobar adjunction but are not computed from diagrams. The
framework is wider than perturbation theory.
\end{remark}

\subsubsection{Example I: Chern--Simons (affine Kac--Moody)}

\begin{computation}[Completed modular Koszul datum for Chern--Simons;
 \ClaimStatusProvedHere]
\label{comp:platonic-CS}

The individual components are:

 \operatorname{Fun}\!\operatorname{Op}_{{}^L\fg}(D^\times)$

 \kappa\bigl(V_k(\fg)\bigr) = \frac{\dim(\fg)\,(k + h^\vee)}{2h^\vee}.
 \end{equation}
 Complementarity:

 Class $\mathbf{L}$ (Lie/tree), shadow depth $r_{\max} = 3$.
\end{enumerate}

\begin{remark}[The $\Ainf$ truncation dichotomy: KM versus Virasoro]
\label{rem:ainf-truncation-dichotomy}
The explicit computation of the $\Ainf$ tower for
$\widehat{\mathfrak{sl}}_2$ at level~$k$
(Proposition~\ref{prop:su2-truncation}) reveals the sharpest
structural contrast in the landscape.
For affine Kac--Moody algebras, the OPE has at most a double pole
(Killing form) and a simple pole (structure constants).
The ternary operation $m_3$ is nonzero, with explicit formula
$m_3(J^a, J^b, J^c) = f^{ab}{}_d f^{dc}{}_g J^g
+ f^{ab}{}_d\, k\, \delta^{dc}\, \lambda_{23}$
(Proposition~\ref{prop:su2-m3}).
The quartic operation $m_4$ vanishes on cohomology: the
Stasheff obstruction assembles iterated structure constant
contractions $f^{ab}{}_p f^{pc}{}_q f^{qd}{}_r$ whose
Stasheff-signed sum is the generalised Jacobi identity at
depth~$3$. The tower terminates:
$r_{\max} = 3$, class $\mathbf{L}$.

For the Virasoro algebra, the quartic OPE pole
$(c/2)/(z-w)^4$ injects new non-associativity at every
degree. The cubic associator contains the term
$\tfrac{c}{12}\lambda_2^3(2\lambda_1 + \lambda_2)$,
which is a \emph{nontrivial cocycle} on cohomology
(it cannot be written as $d$ of anything), so
$m_3^H \ne 0$ and the tower is infinite:
$r_{\max} = \infty$, class $\mathbf{M}$.

The dichotomy is controlled by the OPE pole order.
At double-pole order ($d_{\max} = 2$), the Jacobi identity
kills the quartic obstruction and the bar complex is
$\Ainf$-formal on cohomology.
At quartic-pole order ($d_{\max} = 4$), the excess
intersection dimension prevents Jacobi cancellation, and
the full perturbative hierarchy of $3$d quantum gravity
emerges.
\end{remark}

\begin{remark}[The shadow depth dichotomy in holography]
\label{rem:platonic-shadow-dichotomy}
The five examples exhibit a sharp structural pattern.
The two current-algebra systems (CS and M2) are class~$\mathbf{L}$
with finite shadow obstruction tower ($r_{\max} = 3$), complementarity
anti-symmetry $\kappa + \kappa' = 0$, and $r$-matrices with
simple poles satisfying the classical Yang--Baxter equation.
The three $\mathcal{W}$-algebra systems (3d gravity, $\mathcal{W}_3$,
M5) are class~$\mathbf{M}$ with infinite shadow obstruction tower,
complementarity $\kappa + \kappa' \ne 0$, and $r$-matrices with
higher-order poles. The M2--M5 coupling maps
class~$\mathbf{L}$ into class~$\mathbf{M}$: the Drinfeld--Sokolov
functor is the shadow-depth escalator.

At the critical level $k = -h^\vee$, the CS datum degenerates:
$\kappa = 0$, the bar complex is uncurved, the shadow obstruction tower
collapses, and the Feigin--Frenkel center replaces the generic
$\cZ^{\mathrm{der}}_{\mathrm{ch}}$. For the Virasoro at $c = 0$:
$\kappa = 0$, uncurved bar, but the shadow obstruction tower remains infinite
(higher-degree components are independent of $\kappa$).
This is the gravity analogue of anomaly cancellation: the
scalar curvature vanishes, but the nonlinear tower persists.
\end{remark}
