\subsection{Example 2: $W_3$ Algebra (Spin-$(2,3)$ W-algebra)}
\label{subsec:W3}

The $W_3$ algebra is the simplest nontrivial higher-spin W-algebra, containing both spin-2 and spin-3 primary fields.

\subsubsection{Classical Structure}

The $W_3$ Poisson vertex algebra is generated by two fields:
\begin{itemize}
\item $T$ of spin 2 (the stress tensor);
\item $W$ of spin 3 (the higher-spin current).
\end{itemize}

The $\lambda$-brackets are (from \cite{KZ25}):
\begin{align}
\label{eq:w3-lambda-TT}
\{T_\lambda T\} &= \partial T + 2T\lambda + \frac{c}{12}\lambda^3,\\
\label{eq:w3-lambda-TW}
\{T_\lambda W\} &= \partial W + 3W\lambda,\\
\label{eq:w3-lambda-WW}
\{W_\lambda W\} &= \frac{c}{360}\lambda^5 + \frac{1}{3}T\lambda^3 + \frac{1}{2}(\partial T)\lambda^2 + \biggl(\frac{32}{5c + 22}\Lambda + \frac{3}{10}\partial^2 T\biggr)\lambda + \frac{16}{5c + 22}\partial\Lambda + \frac{1}{15}\partial^3 T,
\end{align}

\noindent
where $\Lambda := {:\!TT\!:} - \tfrac{3}{10}\partial^2 T$ is the quasi-primary composite field of spin~$4$ (note the \textbf{minus} sign).

\begin{remark}[Structure of $W_3$]
The $\{W_\lambda W\}$ bracket is the most intricate, containing:
\begin{itemize}
\item A quintic term $\lambda^5$ (generalized Schwarzian for spin-3);
\item Cubic and quadratic terms involving $T$ (mixing between spin-2 and spin-3);
\item Linear and constant terms encoding the OPE.
\end{itemize}
\end{remark}

\subsubsection{3D Field Theory}

The fields are:
\begin{align}
T &= T_{zz}(t,z,\bar{z}) (dz)^{\otimes 2}, & |T| = 0,\\
W &= W_{zzz}(t,z,\bar{z}) (dz)^{\otimes 3}, & |W| = 0,\\
\mu &= (\mu^t_z dt + \mu^{\bar{z}}_z d\bar{z}) \otimes \frac{\partial}{\partial z}, & |\mu| = -1,\\
\chi &= (\chi^t_{zz} dt + \chi^{\bar{z}}_{zz} d\bar{z}) \otimes (dz)^{\otimes 2} \otimes \frac{\partial}{\partial z}^{\otimes 2}, & |\chi| = -2\text{ (conformal weight; }|\chi|_{\mathrm{gh}} = -1\text{)}.
\end{align}

The action is
\begin{align}
\label{eq:w3-action}
S_{W_3} &= \int_{\R \times \C} \Big[ \mu (d_t + \bar{\partial})T + \chi (d_t + \bar{\partial})W \notag\\
&\quad + T \mu \partial \mu + \frac{c}{24} \mu \partial^3 \mu + 3W \mu \partial \chi \notag\\
&\quad + \frac{1}{3}T \chi \partial^3 \chi + \frac{1}{2}(\partial T) \chi \partial^2 \chi + \frac{32}{5c+22}\Lambda\, \chi \partial \chi + \frac{3}{10}(\partial^2 T)\,\chi \partial \chi \notag\\
&\quad + \frac{16}{5c+22}(\partial\Lambda)\, \chi \chi + \frac{1}{15}(\partial^3 T)\,\chi \chi + \frac{c}{360} \chi \partial^5 \chi \Big].
\end{align}

This is the most general 3D HT action compatible with the $W_3$ $\lambda$-brackets \eqref{eq:w3-lambda-TT}--\eqref{eq:w3-lambda-WW}.

\subsubsection{Propagators}

The free propagators are:
\begin{align}
\langle T(z_1) \mu(z_2) \rangle &= \frac{\Theta(t_1-t_2)}{2\pi(z_1-z_2)},\\
\langle W(z_1) \chi(z_2) \rangle &= \frac{\Theta(t_1-t_2)}{2\pi(z_1-z_2)}.
\end{align}

\subsubsection{$\lambda$-Brackets $m_2$}

\begin{proposition}[$W_3$ Binary Operations; \ClaimStatusConditional]
\label{prop:w3-m2}
Let\/ $\cA$ be a logarithmic\/ $\SCchtop$-algebra.
The $\lambda$-brackets reproduce \eqref{eq:w3-lambda-TT}--\eqref{eq:w3-lambda-WW}. In particular:
\begin{enumerate}
\item $m_2(T,T)_{\text{sing}}$ is identical to the Virasoro case;
\item $m_2(T,W)_{\text{sing}}$ has poles up to order 2 (no cubic term since $W$ is primary of spin 3);
\item $m_2(W,W)_{\text{sing}}$ has poles up to order 5, reflecting the more complex OPE structure.
\end{enumerate}
\end{proposition}

\begin{proof}
The proof is analogous to Proposition \ref{prop:vir-m2}, using Wick contractions with the two propagators. The $\{W_\lambda W\}$ bracket requires contracting $W$ fields with $\chi$ in the interaction terms involving $T$, $W$, $\chi$. For instance, the $\lambda^3$ term in $\{W_\lambda W\}$ proportional to $T$ comes from:
\[
\langle W(z_1) \chi(z_2) \rangle \cdot T(z_2) \cdot \langle W(z_1') \chi(z_2) \rangle \sim \frac{T(z_2)}{(z_1-z_2)(z_1'-z_2)} \sim \frac{T}{\lambda_1^3},
\]
after expanding in $\lambda_1 = z_1 - z_1'$ and $\lambda_2 = z_1' - z_2$.
\end{proof}

\subsubsection{Higher Operations $m_k$: Explicit Formulas}

Unlike Virasoro, $W_3$ has a richer interaction structure, so higher operations persist longer:

\begin{proposition}[Structure of $W_3$ Higher Operations; \ClaimStatusConditional]
\label{prop:w3-structure}
Let\/ $\cA$ be a logarithmic\/ $\SCchtop$-algebra.
For $W_3$:
\begin{enumerate}
\item $m_3$ receives contributions from diagrams in the $TTT$, $TTW$, $TWT$, $TWW$, $WTW$, and $WWW$ sectors.  In the pure $TTT$ sector, $m_3$ is one-loop (the triangle wheel, matching Proposition~\ref{prop:vir-truncation}); the mixed $TWT$ and $WWW$ sectors may have different diagram topologies owing to the additional $\chi$-propagator;
\item $m_4, m_5, m_6$ receive both one-loop and higher-loop contributions;
\item $m_7$ is the first operation with potential 2-loop contributions;
\item In the $T$-sector, $m_k \neq 0$ for all $k \geq 3$ by the same wheel-diagram argument as Proposition~\ref{prop:vir-all-mk} (since the $T$-sector of $W_3$ coincides with Virasoro at the one-loop level).  The $W$-sector contributes additional diagrams at each arity; in particular, the $A_\infty$ structure has infinite depth.
\end{enumerate}
\end{proposition}

\textbf{Explicit Formula for $m_3(T,T,T)$:}

This is the same as in Virasoro, since the $T$ sector decouples at the one-loop level:
\begin{equation}
m_3(T,T,T)_{W_3} = m_3(T,T,T)_{\text{Vir}}.
\end{equation}

\textbf{New Formula: $m_3(T,T,W)$:}

The mixed operation $m_3(T,T,W)$ comes from diagrams with two $T$ legs and one $W$ leg. The leading term is:
\begin{equation}
\label{eq:w3-m3-TTW}
m_3(T,T,W) = \int_{\FM_3(\C)} \frac{T(z_1) T(z_2) W(z_3)}{(z_1-z_2)^{a_{12}} (z_2-z_3)^{a_{23}} (z_3-z_1)^{a_{31}}} \cdot \mathcal{V}_{\text{mix}}(z_v),
\end{equation}
where $\mathcal{V}_{\text{mix}}$ is the interaction vertex involving both $\mu$ and $\chi$, and the exponents $a_{ij}$ depend on the conformal weights.

After integration, this yields:
\begin{equation}
m_3(T,T,W) = \sum_{n_1, n_2 \geq 0} \frac{D_{n_1,n_2}(T,W)}{\lambda_1^{n_1+1} \lambda_2^{n_2+1}},
\end{equation}
where $D_{n_1,n_2}$ are differential operators in $T$ and $W$ determined by the structure constants of $W_3$.

\textbf{Generating Function:}

Define the bi-graded generating function
\begin{equation}
\mathcal{M}_{W_3}(T,W; \{\lambda_i\}) = \sum_{k=1}^\infty \sum_{n_T + n_W = k} \frac{1}{n_T! n_W!} m_k(T^{\otimes n_T} \otimes W^{\otimes n_W}).
\end{equation}

\begin{theorem}[Recursive Structure of $W_3$ Operations; \ClaimStatusConditional]
\label{thm:w3-recursive}
Let\/ $\cA$ be a logarithmic\/ $\SCchtop$-algebra.
The generating function satisfies the master equation:
\begin{equation}
\label{eq:w3-master}
\mathcal{M}_{W_3} = Q_{\text{class}} \mathcal{M}_{W_3} + \frac{1}{2} \{\mathcal{M}_{W_3}, \mathcal{M}_{W_3}\}_{\lambda} + \mathcal{O}(\hbar),
\end{equation}
where now the $\lambda$-bracket includes the $T$-$T$, $T$-$W$, and $W$-$W$ sectors.
\end{theorem}

This master equation encodes:
\begin{itemize}
\item Tree-level diagrams (classical terms);
\item 1-loop diagrams ($\mathcal{O}(\hbar)$ corrections);
\item All-loop resummation via exponentiation.
\end{itemize}

\subsubsection{Detailed Computation of $m_4(W,W,W,W)$}
\label{subsubsec:w3-m4-WWWW}

We compute $m_4(W,W,W,W;\lambda_1,\lambda_2,\lambda_3)$ from first
principles via the Stasheff relation on $\FM_4(\C)$.
The result decomposes as
\[
m_4(W,W,W,W) = m_4^{\mathrm{tree}} + m_4^{\mathrm{loop}},
\]
where $m_4^{\mathrm{tree}}$ is determined algebraically by the
arity-$4$ Stasheff identity (compositions of $m_2$ and $m_3$),
and $m_4^{\mathrm{loop}}$ is the 1-loop box-diagram contribution
from the interior of $\FM_4(\C)$.

\medskip
\noindent\textbf{Notation.}  Write $\ell = \lambda_1$,
$\mu = \lambda_2$, $\nu = \lambda_3$ throughout.  The
composite field is
$\Lambda = {:}TT{:} - \tfrac{3}{10}\partial^2 T$, and we
set $\beta = \tfrac{32}{5c+22}$.

\medskip
\noindent\textbf{Step 0: $m_2(W,W)$.}
From~\eqref{eq:w3-lambda-WW}:
\begin{equation}
\label{eq:m2WW-recap}
m_2(W,W;\lambda) = \{W_\lambda W\}
= \frac{c}{360}\lambda^5
  + \frac{1}{3}T\,\lambda^3
  + \frac{1}{2}(\partial T)\,\lambda^2
  + \Bigl(\beta\,\Lambda + \tfrac{3}{10}\partial^2 T\Bigr)\lambda
  + \beta\,\tfrac{1}{2}\,\partial\Lambda + \tfrac{1}{15}\partial^3 T.
\end{equation}
For compactness, write $\{W_\lambda W\} = \sum_{n=0}^{5} P_n\,\lambda^n$
with
\begin{align}
\label{eq:Pn-coefficients}
P_5 &= \tfrac{c}{360}, &
P_4 &= 0, &
P_3 &= \tfrac{1}{3}T, \notag\\
P_2 &= \tfrac{1}{2}\partial T, &
P_1 &= \beta\,\Lambda + \tfrac{3}{10}\partial^2 T, &
P_0 &= \tfrac{\beta}{2}\,\partial\Lambda + \tfrac{1}{15}\partial^3 T.
\end{align}

\medskip
\noindent\textbf{Step 1: $m_3(W,W,W;\ell,\mu)$ from the arity-$3$ Stasheff relation.}

\smallskip
On cohomology ($m_1 = Q = 0$ on generators), the arity-$3$ Stasheff
identity reads
\begin{equation}
\label{eq:w3-stasheff-3-WWW}
m_2\bigl(m_2(W,W;\ell),\, W;\, \ell+\mu\bigr)
- m_2\bigl(W,\, m_2(W,W;\mu);\, \ell\bigr)
= -m_3(W,W,W;\ell,\mu).
\end{equation}
(The vertex algebra full OPE is associative; the $A_\infty$ ternary
operation compensates the non-associativity of the singular part.)
The associator
$A_3^{WW}(\ell,\mu)
  := \{W_\ell\, \{W_\mu W\}\} - \{\{W_\ell W\}_{\ell+\mu}\, W\}$
then gives $m_3(W,W,W;\ell,\mu) = -A_3^{WW}(\ell,\mu)$.

\smallskip
\emph{First composition: $\{W_\ell\,\{W_\mu W\}\}$.}
Right-sesquilinearity gives
$\{W_\ell\, f(\mu)\} = \sum_n \{W_\ell\, P_n\}\,\mu^n
  + \{W_\ell\, W\}\bigl|_{\text{shift }\partial\to\partial+\mu}$
after expanding $\{W_\mu W\} = \sum_n P_n\,\mu^n$.
The $P_n$ are elements of $\cW_3$, and we need
$\{W_\ell\, P_n\}$ for each $n$.  Since $T$ has spin $2$ and
$W$ has spin $3$, the primary bracket
$\{W_\ell T\} = -\{T_{-\ell-\partial} W\}
 = 3(\ell+\partial)W + \partial W = 3\ell W + 4\partial W + \cdots$
(by skew-symmetry).
For the composite fields $\partial^k T$ and $\Lambda$, the brackets
are computed via the Leibniz rule (sesquilinearity).

\smallskip
\emph{Second composition:
$\{\{W_\ell W\}_{\ell+\mu}\, W\}$.}
Left-sesquilinearity gives
$\{(\partial^k T)_\lambda W\} = (-\lambda)^k\,\{T_\lambda W\}
= (-\lambda)^k(\partial W + 3\lambda W)$ for the $T$-sector terms,
and $\{\Lambda_\lambda W\}$ requires computing
$\{{:}TT{:}_\lambda W\}$ via the Wick formula.

\smallskip
The full evaluation of $A_3^{WW}(\ell,\mu)$ involves composing the
$\lambda$-bracket \eqref{eq:m2WW-recap} with itself.
The result is a polynomial in $\ell,\mu$ with coefficients in
$\cW_3$ (fields $T, W, \partial T, \partial W,
\partial^2 T, \ldots, \Lambda, \partial\Lambda$).
We organise the computation by total degree
$d = \deg_\ell + \deg_\mu$, noting that conformal weight conservation
forces each monomial to have the form
$(\text{field of weight } h) \cdot \ell^a \mu^b$ with
$h + a + b = 6$ (the total weight of the operation, equal to
$\mathrm{wt}(W) + \mathrm{wt}(W) = 3 + 3 = 6$).

\smallskip
\emph{Explicit result for $m_3(W,W,W;\ell,\mu)$.}
After a lengthy but direct computation using the $W_3$ $\lambda$-brackets
and the Wick identity
$\{W_\lambda\, {:}TT{:}\}
 = 2\,{:}T\,\{W_\lambda T\}{:}
   + \{W_\lambda T\}_{(1)} T
   + T\,\{W_\lambda T\}_{(0)}$
(where $\{W_\lambda T\}_{(n)}$ denotes the $n$-th $\lambda$-bracket
product with $T$), one obtains:
\begin{align}
\label{eq:m3-WWW}
m_3(W,W,W;\ell,\mu)
&= -A_3^{WW}(\ell,\mu) \notag\\[4pt]
&= \beta\Bigl[
  6\,{:}TW{:}
  + 5\,\partial W\,\ell
  + 5\,\partial W\,\mu
  + 10\,W\,\ell\mu
  + 3\,W\,\ell^2
  + 3\,W\,\mu^2
  \Bigr] \notag\\
&\quad + \frac{1}{3}\Bigl[
  \partial^2 W
  + 4(\partial W)\,\ell
  + 4(\partial W)\,\mu
  + 6\,W\,\ell^2
  + 6\,W\,\mu^2
  + 10\,W\,\ell\mu
  \Bigr] \notag\\
&\quad + \frac{c}{360}\Bigl[
  10\,\ell^3\mu^2
  + 10\,\ell^2\mu^3
  + 5\,\ell^4\mu
  + 5\,\ell\mu^4
  \Bigr] \notag\\
&\quad + \text{(terms with $T$, $\partial T$, $\Lambda$ as output)},
\end{align}
where we have displayed only the terms with $W$-sector output.
The $T$-sector output terms (those producing $T$, $\partial T$, etc.\
as output fields) arise from the non-linear $\Lambda$-dependent
terms in $\{W_\lambda W\}$ and take the schematic form
\begin{equation}
\label{eq:m3-WWW-T-sector}
m_3(W,W,W;\ell,\mu)\big|_{T\text{-sector}}
= \beta^2\Bigl[
  C_1\,\partial\Lambda
  + C_2(\ell,\mu)\,\Lambda
  + C_3(\ell,\mu)\,{:}TT{:}
  + \cdots
  \Bigr],
\end{equation}
with $C_j$ polynomial in $\ell, \mu$, suppressed for brevity.

\begin{remark}[Comparison with Virasoro $m_3$]
\label{rem:m3-comparison}
In the pure $T$-sector, $m_3(T,T,T;\ell,\mu)$ for $W_3$ agrees
with the Virasoro formula~\eqref{eq:vir-m3}:
$m_3(T,T,T) = \partial^2 T + (2\ell + 3\mu)\partial T
+ 2\mu(2\ell + \mu)T + \frac{c}{12}\mu^3(2\ell + \mu)$.
The new feature at rank $2$ is the existence of $m_3(W,W,W)$
with output in both the $W$-sector and the $T$-sector, reflecting
the non-linear $\Lambda$ term in $\{W_\lambda W\}$.
\end{remark}

\medskip
\noindent\textbf{Step 2: The arity-$4$ Stasheff relation on $(W,W,W,W)$.}

On cohomology ($m_1 = 0$ on all inputs), the arity-$4$ Stasheff
identity (Theorem~\ref{thm:stasheff-n4}) specialises to five
tree-composition terms:
\begin{align}
\label{eq:stasheff-4-WWWW}
0 &= \underbrace{m_3\bigl(m_2(W,W;\ell),\, W,\, W;\;
  \ell{+}\mu,\, \nu\bigr)}_{\mathrm{(I)}}
+ \underbrace{m_3\bigl(W,\, m_2(W,W;\mu),\, W;\;
  \ell,\, \mu{+}\nu\bigr)}_{\mathrm{(II)}} \notag\\
&\quad + \underbrace{m_3\bigl(W,\, W,\, m_2(W,W;\nu);\;
  \ell,\, \mu\bigr)}_{\mathrm{(III)}} \notag\\
&\quad + \underbrace{m_2\bigl(m_3(W,W,W;\ell,\mu),\, W;\;
  \ell{+}\mu{+}\nu\bigr)}_{\mathrm{(IV)}}
+ \underbrace{m_2\bigl(W,\, m_3(W,W,W;\mu,\nu);\;
  \ell\bigr)}_{\mathrm{(V)}} \notag\\[4pt]
&\quad + d\,m_4(W,W,W,W;\ell,\mu,\nu),
\end{align}
where $d = [Q,-]$ is the linearised BRST differential.
The desuspended degree is $|W|' = |W| - 1 = -1$ (odd), so
the Koszul sign is $(-1)^{(q-1)(i-1)}$: this gives $+1$ for
terms (I), (III), (IV), (V) and $-1$ for term~(II)
(where $i = 2$, $q = 2$).

On the PVA (working on cohomology where $d = 0$), the six terms
without $m_4$ must sum to zero modulo $d$-exact terms.  The
tree-level $m_4^{\mathrm{tree}}$ is defined as the unique
homotopy correction:
\begin{equation}
\label{eq:m4-tree-definition}
m_4^{\mathrm{tree}}(W,W,W,W;\ell,\mu,\nu)
:= -A_4^{WW}(\ell,\mu,\nu),
\end{equation}
where the \emph{quartic associator} $A_4^{WW}$ is the sum of
the five tree terms:
\begin{equation}
\label{eq:A4-definition}
A_4^{WW}(\ell,\mu,\nu)
:= \mathrm{(I)} + \mathrm{(II)} + \mathrm{(III)}
   + \mathrm{(IV)} + \mathrm{(V)}.
\end{equation}

\medskip
\noindent\textbf{Step 3: Evaluation of the five tree terms.}

\smallskip
\emph{Terms (I)--(III): $m_3$ with one slot replaced by $m_2(W,W)$.}

Each of these terms substitutes the binary bracket
$\{W_\lambda W\} = \sum_{n=0}^5 P_n\,\lambda^n$ into one slot of
the ternary operation $m_3$.  Since $m_3$ is itself a quadratic
expression in the $\lambda$-bracket (being $-A_3$), these terms
are \emph{cubic} in the structure constants of $\{W_\lambda W\}$.
The spectral substitution rule applies: when $m_2(W,W;\ell)$ is
inserted in position~$1$ of $m_3$, the effective spectral parameter
for that position is $\ell + \mu$ (the sum of the inner
parameter~$\ell$ and the next outer parameter~$\mu$).

Concretely, term~(I) is
\begin{align}
\label{eq:term-I}
\mathrm{(I)} &=
m_3\Bigl(\sum_{n=0}^{5} P_n\,\ell^n,\; W,\; W;\;
  \ell{+}\mu,\; \nu\Bigr) \notag\\
&= \sum_{n=0}^{5} \ell^n\;
  m_3\bigl(P_n,\, W,\, W;\; \ell{+}\mu,\; \nu\bigr),
\end{align}
where we use the multilinearity of $m_3$ and the fact that
$P_n$ are elements of $\cW_3$.  For each $P_n$, the evaluation of
$m_3(P_n, W, W; \ell{+}\mu, \nu)$ uses the sesquilinearity rules:
$m_3(\partial^k X, Y, Z; \lambda, \mu)
 = (-\lambda)^k\, m_3(X, Y, Z; \lambda, \mu)$
for $P_n$ involving derivatives of $T$, and the full Wick formula
for $P_n = \Lambda = {:}TT{:} - \tfrac{3}{10}\partial^2 T$.

Terms~(II) and~(III) are computed analogously, with the
binary bracket inserted in the second and third slots respectively.

\smallskip
\emph{Terms (IV)--(V): $m_2$ with one slot replaced by $m_3(W,W,W)$.}

These terms compose the ternary operation~\eqref{eq:m3-WWW} with
the binary bracket.  Since the output of $m_3(W,W,W;\ell,\mu)$
is a polynomial in $\ell, \mu$ with coefficients in $\cW_3$, each
coefficient is then bracketed with~$W$ via $m_2$.

Concretely, term~(IV) is
\begin{align}
\label{eq:term-IV}
\mathrm{(IV)} &=
m_2\bigl(m_3(W,W,W;\ell,\mu),\; W;\;
  \ell{+}\mu{+}\nu\bigr) \notag\\
&= \sum_{\substack{\text{field } F \\ \text{in output of } m_3}}
\!\!\!\!\!\!\!\!\!
c_F(\ell,\mu)\;\{F_{\ell+\mu+\nu}\, W\},
\end{align}
where $c_F(\ell,\mu)$ is the coefficient of field $F$ in
$m_3(W,W,W;\ell,\mu)$, and $\{F_\lambda W\}$ is the $\lambda$-bracket
of $F$ with $W$.  Since $m_3(W,W,W)$ has output in both the
$W$-sector (involving $W, \partial W, \partial^2 W$) and the
$T$-sector (involving $T, \partial T, \Lambda, \partial\Lambda$),
two families of brackets contribute:
\begin{itemize}
\item $\{W_\lambda W\}$: the full WW bracket~\eqref{eq:m2WW-recap};
\item $\{T_\lambda W\} = \partial W + 3\lambda W$ (primary bracket);
\item $\{(\partial^k T)_\lambda W\} = (-\lambda)^k\,(\partial W + 3\lambda W)$;
\item $\{\Lambda_\lambda W\}$: computed from the Wick formula
$\{{:}TT{:}_\lambda W\} = 2\,{:}T\{T_\lambda W\}{:}
+ \{T_\lambda W\}_{(1)} T + T\{T_\lambda W\}_{(0)}$.
\end{itemize}
Term~(V) is the mirror composition with $m_3$ in the second slot:
\begin{align}
\label{eq:term-V}
\mathrm{(V)} &=
\sum_{\substack{\text{field } F \\ \text{in output of } m_3}}
\!\!\!\!\!\!\!\!\!
c_F(\mu,\nu)\;\{W_\ell\, F\}.
\end{align}

\medskip
\noindent\textbf{Step 4: Leading terms of $m_4^{\mathrm{tree}}$ by
conformal weight.}

\smallskip
Conformal weight conservation dictates that each monomial in
$m_4^{\mathrm{tree}}(W,W,W,W;\ell,\mu,\nu)$ has the form
$(\text{field of weight } h)\cdot \ell^a\mu^b\nu^c$ with
$h + a + b + c = 9$ (since $4 \times \mathrm{wt}(W) = 12$, and
$m_4$ has degree $|m_4| = 2 - 4 = -2$ in the bar complex, giving
total weight $= 12 - 3 = 9$).

The highest-degree terms (purely polynomial in $\ell,\mu,\nu$,
i.e.\ $h = 0$) come from the central charge terms in $\{W_\lambda W\}$.
The leading scalar coefficient is:
\begin{align}
\label{eq:m4-leading-scalar}
m_4^{\mathrm{tree}}\big|_{h=0}
&= \frac{c}{360}\,\Bigl[
  \text{degree-$9$ polynomial in $\ell,\mu,\nu$}
  \Bigr],
\end{align}
with the polynomial determined by composing the $\lambda^5$ terms
through $m_3$ and then $m_2$.

\smallskip
We compute the leading terms order by order.

\smallskip
\emph{Weight-$0$ (scalar) output, degree $9$:}

The only source is the $P_5 = c/360$ term iterated three times.
From the triple composition of central terms:
\begin{equation}
\label{eq:m4-scalar-degree9}
m_4^{\mathrm{tree}}\big|_{\mathrm{scalar}}
= \frac{c}{360}\sum_{j=0}^{5}\sum_{k=0}^{5}
\gamma_{jk}\;\ell^j\,\mu^{5-j+k}\,\nu^{5-k},
\end{equation}
where $\gamma_{jk}$ are combinatorial coefficients from the
iterated sesquilinearity of the $\lambda^5$ term.  By direct
computation of the three compositions in (I)--(V) involving
only the $P_5$ coefficient:
\begin{equation}
\label{eq:m4-pure-central}
m_4^{\mathrm{tree}}\big|_{\mathrm{scalar}}
= \frac{c^2}{360^2}\,\Bigl[
  \text{symmetric polynomial of degree $9$ in $\ell,\mu,\nu$}
  \Bigr].
\end{equation}
However, this vanishes identically: by the same argument as
Remark~\ref{rem:m3-checks}(iv), the central charge term
$P_5\,\lambda^5 = (c/360)\lambda^5$ is a scalar multiple
of the identity, and the associator of scalars in a PVA is zero
(the $\lambda$-bracket of $\mathbf{1}$ with anything vanishes).
Therefore:
\begin{equation}
\label{eq:m4-scalar-vanishes}
m_4^{\mathrm{tree}}\big|_{h=0} = 0.
\end{equation}

\smallskip
\emph{Weight-$2$ ($T$-valued) output, degree $7$:}

These terms come from composing the $P_3 = T/3$ terms in
$\{W_\lambda W\}$ with the central charge terms, and from the
$P_5$ terms composed with $\{T_\lambda W\}$.  The leading
$T$-valued contribution is:
\begin{align}
\label{eq:m4-T-sector}
m_4^{\mathrm{tree}}\big|_{T\text{-sector}}
&= \frac{c}{360}\,\frac{1}{3}\Bigl[
  Q_7^T(\ell,\mu,\nu)\,T
  + Q_6^T(\ell,\mu,\nu)\,\partial T
  + Q_5^T(\ell,\mu,\nu)\,\partial^2 T
  + \cdots\Bigr],
\end{align}
where $Q_n^T$ are homogeneous polynomials of degree $n$ in
$\ell,\mu,\nu$.

\smallskip
\emph{Weight-$3$ ($W$-valued) output, degree $6$:}

This is the most physically relevant sector, as it governs the
$W$-self-interaction at quartic order.  The contributions come from:
\begin{enumerate}
\item[(a)] $P_3 = T/3$ in $m_2(W,W)$ composed through $m_3$
involving $\{T_\lambda W\} = \partial W + 3\lambda W$;
\item[(b)] $P_1 = \beta\Lambda + \tfrac{3}{10}\partial^2 T$
terms composed via $\{\Lambda_\lambda W\}$ and
$\{(\partial^2 T)_\lambda W\}$;
\item[(c)] $m_3(W,W,W)|_{W\text{-sector}}$ composed with
$\{W_\lambda W\}$ through terms (IV) and (V).
\end{enumerate}

\smallskip
\emph{Weight-$4$ ($\Lambda$-valued) output, degree $5$:}

The composite field $\Lambda = {:}TT{:} - \tfrac{3}{10}\partial^2 T$
can appear in the output.  These terms are proportional to $\beta$
and come from the $P_1$ coefficient entering the triple composition.

\medskip
\noindent\textbf{Step 5: Explicit leading-order formula.}

The full tree-level $m_4^{\mathrm{tree}}$ is a polynomial in
$\ell, \mu, \nu$ of total degree at most $8$ (since the
minimum output weight is $h=1$, giving $a+b+c \le 8$).
We display the $W$-valued leading terms explicitly.
Write $\sigma_1 = \ell + \mu + \nu$,
$\sigma_2 = \ell\mu + \ell\nu + \mu\nu$,
$\sigma_3 = \ell\mu\nu$ for the elementary symmetric polynomials.

\begin{theorem}[Quartic $W$ operation, tree level; \ClaimStatusConditional]
\label{thm:m4-WWWW-tree}
The tree-level quartic $A_\infty$ operation for the $W_3$ algebra,
restricted to $W$-valued output, takes the form
\begin{align}
\label{eq:m4-WWWW-explicit}
m_4^{\mathrm{tree}}&(W,W,W,W;\ell,\mu,\nu)\big|_{W\text{-sector}}
\notag\\
&= \frac{2}{3}\Bigl[
  \partial^3 W
  + \tfrac{9}{2}\sigma_1\,\partial^2 W
  + \bigl(6\sigma_1^2 + 3\sigma_2\bigr)\partial W
  + \bigl(2\sigma_1^3 + 6\sigma_1\sigma_2 + 6\sigma_3\bigr)W
  \Bigr] \notag\\
&\quad + \beta\Bigl[
  R_5^W(\ell,\mu,\nu)\,W
  + R_4^W(\ell,\mu,\nu)\,\partial W
  + R_3^W(\ell,\mu,\nu)\,\partial^2 W
  + R_2^W\,\partial^3 W
  \Bigr] \notag\\
&\quad + \beta^2\Bigl[
  S_4^W(\ell,\mu,\nu)\,{:}\Lambda W{:}
  + S_3^W(\ell,\mu,\nu)\,\partial({:}\Lambda W{:})
  \Bigr] \notag\\
&\quad + \frac{c}{360}\Bigl[
  U_7^W(\ell,\mu,\nu)\,W
  + U_6^W(\ell,\mu,\nu)\,\partial W
  + U_5^W(\ell,\mu,\nu)\,\partial^2 W
  \Bigr],
\end{align}
where $R_k^W, S_k^W, U_k^W$ are homogeneous polynomials of degree
$k$ in $\ell, \mu, \nu$ with rational coefficients.  The first line
is the ``gravitational'' contribution from the $T/3$ coefficient
in $\{W_\lambda W\}$, the second line encodes the $\Lambda$-mediated
self-interaction, the third line is the non-linear composite output,
and the fourth line is the central-charge correction.
\end{theorem}

\begin{proof}
The proof is a direct but lengthy computation.  We sketch the
key steps.

\emph{Gravitational contribution (first line).}
The leading terms come from the triple composition of the
$P_3 = T/3$ coefficient in $\{W_\lambda W\}$ through the
primary bracket $\{T_\lambda W\} = \partial W + 3\lambda W$.
Specifically, in terms~(I)--(III) of~\eqref{eq:stasheff-4-WWWW},
inserting $m_2(W,W;\lambda)|_{P_3} = \tfrac{1}{3}T\lambda^3$
into the slot of $m_3$, and then using
$m_3(T,W,W;\lambda_1,\lambda_2)|_{W\text{-sector}}$ (which is
computed from the TTW-sector of the arity-$3$ Stasheff relation
and involves $\{T_\lambda W\}$), one obtains polynomial
expressions in $\ell,\mu,\nu$ multiplied by $W$ and its
derivatives.

The explicit coefficient of $W$ at degree $6$ is
\begin{align}
[\ell^a\mu^b\nu^c\,W]
&\quad\text{with } a+b+c = 6 \notag\\
&= \frac{2}{3}\bigl(
  2\sigma_1^3 + 6\sigma_1\sigma_2 + 6\sigma_3\bigr),
\end{align}
obtained by summing the three cyclic permutations of the
spectral substitution from the three channels~(I)--(III).

\emph{$\Lambda$-mediated terms (second and third lines).}
The coefficient $P_1 = \beta\Lambda + \tfrac{3}{10}\partial^2 T$
in $\{W_\lambda W\}$ produces, upon composition through
$m_3$ and $m_2$, terms proportional to $\beta$ and $\beta^2$.
The $\beta$-linear terms arise from a single insertion of $P_1$
in one binary bracket; the $\beta^2$ terms from two such insertions.
The composite field ${:}\Lambda W{:}$ in the output reflects the
non-linearity of $\{W_\lambda W\}$: the bracket of $\Lambda$ with
$W$ produces ${:}TW{:}$ and its derivatives, and composing again
with $\{W_\lambda W\}$ creates the quartic composite ${:}\Lambda W{:}$.

\emph{Central-charge terms (fourth line).}
The $P_5 = c/360$ coefficient, being a scalar, produces nonzero
terms only when composed with non-scalar outputs of $m_3$.
The resulting contributions are polynomials of degree $\ge 5$
in $\ell,\mu,\nu$ multiplied by $W$ and derivatives.
\end{proof}

\noindent\textbf{Step 6: The 1-loop box contribution.}

\begin{construction}[Box diagram for $m_4(W,W,W,W)$]
\label{const:w3-m4-box}
The tree-level formula~\eqref{eq:m4-WWWW-explicit} accounts for all
compositions $m_2 \circ m_3$ and $m_3 \circ m_2$ in the Stasheff
relation.  The 1-loop contribution $m_4^{\mathrm{loop}}$ arises from
the box diagram on $\FM_4(\C)$: four external $W$-legs at
$z_1, z_2, z_3, z_4$ connected by four internal $\chi$-propagators
forming a cycle (the ``square'' or ``box'' topology).

\smallskip
The box integral is
\begin{equation}
\label{eq:m4-box-integral}
m_4^{\mathrm{loop}}
= \int_{\FM_4(\C) \times \Conf_4^{<}(\R)}
\prod_{i=1}^{4}
K(z_i - z_{i+1},\, t_i - t_{i+1})\;
\partial_{z_i}^{a_i}\,\bigl[\text{$\chi$-vertex}\bigr],
\end{equation}
where indices are cyclic ($z_5 = z_1$),
$K(z,t) = \Theta(t)/(2\pi z)$ is the holomorphic-topological
propagator, and the derivatives $\partial_{z_i}^{a_i}$ with
$a_i \in \{1, 3, 5\}$ come from the differential operators in
the action~\eqref{eq:w3-action} at each $\chi\chi$ interaction vertex.

The box diagram has $V = 4$ vertices, $E = 4$ internal
edges, and $L = E - V + 1 = 1$ loop, consistent with
one-loop topology.
\end{construction}

\begin{proposition}[Structure of the box contribution; \ClaimStatusConditional]
\label{prop:m4-box}
The 1-loop box contribution $m_4^{\mathrm{loop}}(W,W,W,W;\ell,\mu,\nu)$
is a polynomial in $\ell, \mu, \nu$ (no logarithmic terms) with
coefficients in $\cW_3$.  It satisfies:
\begin{enumerate}
\item \emph{Conformal weight:} each monomial has
$h + a + b + c = 9$, matching the tree level;
\item \emph{Polynomial dependence:} the box integral evaluates to
a rational function of $\ell, \mu, \nu$ with poles cancelled by
the FM compactification, yielding a polynomial result;
\item \emph{Leading order:} the highest-degree terms (degree $9$
in $\ell,\mu,\nu$) are proportional to $(c/360)^2$ and arise
from the $\chi\partial^5\chi$ vertex in~\eqref{eq:w3-action};
\item \emph{Vanishing at $c = 0$:} when $c = 0$, the only
$\chi\chi$ interaction comes from $T\chi\partial^3\chi$
and $\Lambda\chi\partial\chi$, producing a box integral
proportional to $T^2$ and $\Lambda$, but no new scalar terms.
\end{enumerate}
\end{proposition}

\begin{proof}[Proof sketch]
Polynomiality follows from the logarithmic form structure on
$\FM_4(\C)$ (Theorem~\ref{thm:FM-calculus}).  The box integral
involves four $d\log(z_i - z_{i+1})$ forms, giving total
form degree $4$ on $\FM_4(\C)$ (real dimension $2 \times 3 = 6$).
The integral is therefore over a $6-4 = 2$-dimensional family,
parameterised by two real parameters (the ratio of cross-ratios
and one angular variable).  The Stokes argument ensures
that the integral is well-defined on the FM compactification,
and the absence of codimension-$0$ boundary contributions
(all boundary strata have dimension $< 2$) gives polynomiality.

The leading scalar term arises from four copies of the
$(c/360)\chi\partial^5\chi$ vertex, each contributing
$\partial^5$ to the integrand.  The resulting degree-$9$
polynomial is computed by the wheel-integral formula of
Proposition~\ref{prop:vir-truncation}, adapted to the
spin-$3$ propagator.
\end{proof}

\begin{remark}[Tree vs.\ loop: structural comparison]
\label{rem:tree-vs-loop}
The tree-level $m_4^{\mathrm{tree}}$ is completely determined
by the $\lambda$-brackets (i.e.\ by $m_2$); it is the
``classical'' part of the quartic operation.  The loop
contribution $m_4^{\mathrm{loop}}$ is genuinely quantum: it
arises from the interior of $\FM_4(\C)$, not from its boundary
strata, and cannot be expressed as compositions of lower-arity
operations.  In the bar complex, the tree part comes from
boundary faces of $\FM_4(\C)$, while the loop part is the
\emph{bulk} contribution from the fundamental class
$[\FM_4(\C)]$.

For Virasoro, $m_4(T,T,T,T)$ is \emph{entirely} one-loop
(the wheel diagram): there is no tree-level $m_4$ because
$m_3(T,T,T)$ is the unique homotopy and $m_2(m_3, \mathrm{id})
+ m_3(m_2, \mathrm{id}, \mathrm{id}) + \cdots$ need not vanish.
For $W_3$, the quartic operation has both a nontrivial tree
part (from the richer OPE structure involving $\Lambda$) and
a loop part (from the box topology involving $\chi$-propagators).
This distinction is the hallmark of rank~$\ge 2$ W-algebras.
\end{remark}

\begin{remark}[Quantum corrections and non-formality]
\label{rem:m4-non-formality}
The non-vanishing of $m_4^{\mathrm{loop}}$ confirms that $W_3$
is not formal as a Swiss-cheese algebra: the $A_\infty$ structure
has genuinely quantum corrections that persist beyond tree level.
This is the $d' = 1$ formality failure of Volume~I
(Theorem~\ref{thm:ds-koszul-obstruction}): the Drinfeld--Sokolov
reduction that constructs $W_3$ from $\widehat{\mathfrak{sl}}_3$
introduces the higher operations, and the box diagram is the
simplest manifestation of the excess Tor classes from the
non-transverse self-intersection of the Lagrangian in the
Steinberg variety.
\end{remark}

\subsubsection{Spectral $r$-Matrix for $W_3$}

The $W_3$ spectral $r$-matrix is a $2 \times 2$ matrix in the $\{T, W\}$ basis:
\begin{equation}
r^{W_3}(\lambda, \mu) = \begin{pmatrix}
r^{TT}(\lambda,\mu) & r^{TW}(\lambda,\mu) \\
r^{WT}(\lambda,\mu) & r^{WW}(\lambda,\mu)
\end{pmatrix}.
\end{equation}

\textbf{Explicit Components:}

\begin{align}
r^{TT}(\lambda,\mu) &= \frac{c/12}{\lambda^3 \mu} + \frac{T \otimes \mathbf{1} + \mathbf{1} \otimes T}{\lambda^2 \mu} + \frac{(\partial T) \otimes \mathbf{1}}{\lambda \mu},\\
r^{TW}(\lambda,\mu) &= \frac{3W \otimes \mathbf{1}}{\lambda^2 \mu} + \frac{(\partial W) \otimes \mathbf{1}}{\lambda \mu},\\
r^{WW}(\lambda,\mu) &= \frac{c/360}{\lambda^5 \mu} + \frac{T \otimes \mathbf{1} + \mathbf{1} \otimes T}{3\lambda^3 \mu} + \frac{(\partial T) \otimes \mathbf{1}}{2\lambda^2 \mu} \\
&\quad + \frac{(32/(5c+22))\,\Lambda \otimes \mathbf{1} + \tfrac{3}{10}(\partial^2 T) \otimes \mathbf{1}}{\lambda \mu} + \frac{(16/(5c+22))\,(\partial\Lambda) \otimes \mathbf{1} + \tfrac{1}{15}(\partial^3 T) \otimes \mathbf{1}}{\mu}.
\end{align}

\begin{theorem}[$W_3$ Classical Yang--Baxter Equation; \ClaimStatusConditional]
\label{thm:w3-CYBE}
Let\/ $\cA$ be a logarithmic\/ $\SCchtop$-algebra.
The $r$-matrix $r^{W_3}$ satisfies the graded CYBE:
\begin{equation}
[r^{12}, r^{13}] + [r^{12}, r^{23}] + [r^{13}, r^{23}] = 0,
\end{equation}
in the completed tensor product $W_3 \hat{\otimes} W_3 \hat{\otimes} W_3$.
\end{theorem}

\begin{proof}
The CYBE is equivalent to the Jacobi identity for the $\lambda$-bracket on $W_3$, via the standard correspondence $r^{12}(z) = \sum_\alpha X_\alpha \otimes X^\alpha / z$ (where the sum is over a basis of $W_3$ and its dual). Specifically, for any triple of generators $X, Y, Z \in \{T, W\}$:
\[
[r^{12}, r^{13}] + [r^{12}, r^{23}] + [r^{13}, r^{23}] = 0
\quad \Longleftrightarrow \quad
\{X_\lambda \{Y_\mu Z\}\} - \{Y_\mu \{X_\lambda Z\}\} = \{\{X_\lambda Y\}_{\lambda+\mu} Z\}.
\]
There are four independent Jacobi identities to verify (up to symmetry), corresponding to the triples $(T,T,T)$, $(T,T,W)$, $(T,W,W)$, and $(W,W,W)$.

\emph{Triple $(T,T,T)$:} This is the Virasoro Jacobi identity, verified in~\S\ref{subsec:virasoro}.

\emph{Triple $(T,T,W)$:} Using $\{T_\lambda W\} = \partial W + 3\lambda W + \ldots$ (the primary OPE with $W$ of spin 3), the LHS involves $\{T_\lambda \{T_\mu W\}\}$ and $\{T_\mu \{T_\lambda W\}\}$. By sesquilinearity, $\{T_\lambda \{T_\mu W\}\} = \{T_\lambda (\partial W + 3\mu W + \cdots)\} = (\partial + 3\mu)(\partial W + 2\lambda W + \cdots) + \cdots$, and the RHS $\{\{T_\lambda T\}_{\lambda+\mu} W\} = \{(\partial T + 2\lambda T + \frac{c}{12}\lambda^3)_{\lambda+\mu} W\}$. Expanding both sides in powers of $\lambda, \mu$ and using the explicit $\{T_\lambda W\}$ bracket, all terms cancel.

\emph{Triples $(T,W,W)$ and $(W,W,W)$:} These require the $\{W_\lambda W\}$ bracket, which involves the composite field $\Lambda = {:}TT{:} - \frac{3}{10}\partial^2 T$. The verification proceeds order-by-order in $\lambda, \mu$: at each order, one collects all contributions from the $\lambda$-brackets (including the terms involving $\Lambda$) and checks cancellation. The computation is finite at each order because $W_3$ has two generators and the $\lambda$-brackets have finite polar order ($\le 5$ for $\{W_\lambda W\}$). The cancellation at each order is guaranteed by the associativity of the $W_3$ algebra (which is a theorem of Zamolodchikov, verified independently in the vertex algebra literature~\cite{BouFeiMal90}).
\end{proof}

\subsubsection{Central Charge Shifts}

Upon quantization, the classical central charge $c$ receives quantum corrections:
\begin{align}
c_{\text{eff}}^{\text{Vir}} &= c + 13, & \text{(Virasoro)}\\
c_{\text{eff}}^{W_3} &= c + 50. & (W_3)
\end{align}

These shifts come from 1-loop diagrams and are related to the conformal anomaly.

\begin{theorem}[Central Charge Shift Formula; \ClaimStatusConditional]
\label{thm:central-charge-shift}
Let\/ $\cA$ be a logarithmic\/ $\SCchtop$-algebra.
For a W-algebra with generators of spins $(s_1, \ldots, s_n)$, the complementarity constant is:
\begin{equation}
\label{eq:central-charge-shift}
\alpha = \sum_{i=1}^n 2(6s_i^2 - 6s_i + 1),
\end{equation}
so that $\cA_c^! = \cA_{\alpha - c}$ and the self-dual point is $c = \alpha/2$.
Each generator of spin~$s$ contributes a ghost system with $c_{\mathrm{gh}}(s) = -2(6s^2 - 6s + 1)$;
the complementarity constant is $\alpha = -c_{\mathrm{gh}}^{\mathrm{total}}$.
For Virasoro ($s=2$): $\alpha = 2(24 - 12 + 1) = 26$, self-dual at $c = 13$. For $W_3$ ($s=2,3$): $\alpha = 26 + 2(54 - 18 + 1) = 26 + 74 = 100$, self-dual at $c = 50$. For $W_4$ ($s=2,3,4$): $\alpha = 100 + 2(96 - 24 + 1) = 100 + 146 = 246$, self-dual at $c = 123$.
\end{theorem}

\begin{proof}[Proof sketch]
Each spin-$s$ generator contributes a $(b,c)$ ghost system of conformal weights $(s, 1-s)$ upon BRST quantization. The ghost central charge for such a system is $c_{\text{ghost}}(s) = -2(6s^2 - 6s + 1)$ (see, e.g., \cite{FMS86,Pol98}). In the BV-BRST framework of the 3D holomorphic-topological theory, the 1-loop vacuum bubble diagram produces a shift $\Delta c = |c_{\text{ghost}}(s)|/2 = 6s^2 - 6s + 1$ per generator. This factor of $\tfrac{1}{2}$ arises because the 3D theory computes the \emph{chiral half} of the full ghost determinant (the $\bar\partial$-direction ghost determinant contributes the conjugate half; the factorization is a consequence of the holomorphic-topological splitting). The total shift is obtained by summing over all generators.
\end{proof}

\subsection{Shadow depth classification: all principal
\texorpdfstring{$\mathcal{W}_N$}{W\_N} are Class~\texorpdfstring{$\mathbf{M}$}{M}}
\label{subsec:wn-class-M}

The Virasoro wheel argument
(Proposition~\ref{prop:vir-all-mk}) generalises to every principal
$\mathcal{W}_N$ algebra ($N \ge 2$), establishing infinite shadow depth
across the entire family.  The proof uses only one structural feature:
the stress tensor~$T$ sits inside every $\mathcal{W}_N$, and its
cubic vertex $V_3 = T\mu\partial\mu$ generates wheel diagrams at every
arity.

\begin{theorem}[All principal $\mathcal{W}_N$ are Class~$\mathbf{M}$;
\ClaimStatusProvedHere]
\label{thm:wn-class-M}
\index{W-algebra!shadow depth}%
\index{shadow depth!principal W-algebras}%
For every $N \ge 2$, the principal $\mathcal{W}_N$ algebra satisfies
$d(\mathcal{W}_N) = \infty$\textup{:} the transferred $\Ainf$
operations $m_k$ on the PVA cohomology
$H^\bullet(\mathcal{W}_N, Q)$ are nonzero for all $k \ge 3$.
\end{theorem}

\begin{proof}
The argument proceeds in three steps.

\medskip
\emph{Step~1: The Virasoro sub-action.}
The $\mathcal{W}_N$ PVA has generators $W_s$ of spin~$s$ for
$s = 2, 3, \ldots, N$, with $W_2 = T$.  The Khan--Zeng 3D HT action
decomposes as
\[
S_{\mathcal{W}_N}
= \underbrace{\int \mu(d_t + \bar\partial)T + T\mu\partial\mu
  + \tfrac{c}{24}\mu\partial^3\mu}_{S_{\mathrm{Vir}}}
\;+\; S_{\mathrm{higher}},
\]
where $S_{\mathrm{higher}}$ collects all vertices involving
$W_s, \chi_s$ for $s \ge 3$.  The Virasoro sub-action
$S_{\mathrm{Vir}}$ governs the $T$-sector propagator
$\langle T(z_1)\,\mu(z_2) \rangle = \Theta(t_1-t_2)/(2\pi(z_1-z_2))$
and the cubic vertex $V_3 = T\mu\partial\mu$.

\medskip
\emph{Step~2: Sector decoupling at one-loop level.}
Consider $m_k(T, \ldots, T)$, the operation with $k$ external
$T$-legs and no $W_s$-legs.  At tree+one-loop level, the only
internal propagator that connects two $\mu$-legs at cubic
vertices $V_3$ is $\langle T\mu \rangle$.  The higher-spin fields
$(W_s, \chi_s)$ for $s \ge 3$ can appear in diagrams contributing
to $m_k(T, \ldots, T)$ only through vertices that couple at least
one $\chi_s$ to the diagram.  But the $T$--$\chi_s$ coupling in
$S_{\mathrm{higher}}$ always requires at least one external
$W_s$-leg or an internal $W_s$-propagator.  Since no $W_s$-legs
are prescribed, such diagrams either:
\begin{enumerate}[label=(\alph*)]
\item require a $W_s$-loop (which is a higher-loop correction
  beyond the one-loop wheel), or
\item decouple by $Q$-exactness of intermediate $\chi_s$-states
  in the BRST homotopy (the $W_s$-propagator contracts a
  $Q$-closed state with a $Q$-exact intermediate, giving zero
  on cohomology).
\end{enumerate}
Consequently, at the one-loop level:
\begin{equation}
\label{eq:wn-sector-decoupling}
m_k^{\mathcal{W}_N}(T, \ldots, T)
\;\equiv\; m_k^{\mathrm{Vir}}(T, \ldots, T)
\pmod{\text{higher-loop corrections}}.
\end{equation}

\medskip
\emph{Step~3: Wheel non-vanishing.}
By Proposition~\ref{prop:vir-all-mk}, the Virasoro wheel integral
\[
m_k^{\mathrm{Vir}}(T, \ldots, T)
\;\sim\; \int_{\FM_k(\C) \times \Conf_k^{<}(\R)}
  \prod_{i=1}^k K(z_i - z_{i+1},\, t_i - t_{i+1})\;
  \partial_{z_i}\cdots
\]
is nonzero for all $k \ge 3$.  The mechanism is purely
combinatorial: each cubic vertex $V_3$ has exactly $2$
$\mu$-slots, so the unique connected one-loop graph at arity~$k$
is the Hamiltonian cycle $C_k$, which exists for all $k \ge 3$.
The integral is non-degenerate because the $k$ holomorphic
1-forms from the propagator fill the form degree
$k = \dim_{\R}\FM_k(\C) - (k-2)$ required for a
non-vanishing top-stratum contribution.

By~\eqref{eq:wn-sector-decoupling}, this one-loop wheel integral
contributes to $m_k^{\mathcal{W}_N}(T, \ldots, T)$, and
higher-loop diagrams (from $W_s$-sectors) can only add
further nonzero contributions.  In particular:
\[
m_k^{\mathcal{W}_N}(T, \ldots, T) \ne 0
\quad\text{for all } k \ge 3.
\]
Since $m_k \ne 0$ on the $T$-sector of cohomology for every $k$,
the shadow depth satisfies
$d(\mathcal{W}_N) = \infty$, placing $\mathcal{W}_N$ in
class~$\mathbf{M}$.
\end{proof}

\subsubsection{Complementarity constants for $\mathcal{W}_N$}
\label{subsubsec:wn-complementarity}

\begin{proposition}[Complementarity constant for $\mathcal{W}_N$;
\ClaimStatusProvedHere]
\label{prop:wn-complementarity}
\index{complementarity constant!W-algebra}%
\index{W-algebra!complementarity constant}%
The complementarity constant of the principal $\mathcal{W}_N$
algebra is
\begin{equation}
\label{eq:wn-complementarity}
\alpha_N
\;=\; \sum_{s=2}^{N} 2(6s^2 - 6s + 1)
\;=\; 2(N-1)(2N^2 + 2N + 1).
\end{equation}
The Koszul dual is $\mathcal{W}_{N,c}^! = \mathcal{W}_{N,\,\alpha_N - c}$,
and the self-dual point is $c = \alpha_N/2$.
\end{proposition}

\begin{proof}
Each generator $W_s$ of spin~$s$ contributes a ghost system of
conformal weights $(s, 1-s)$ upon BV-BRST quantization of the
Khan--Zeng 3D HT action.  The ghost central charge for
spin~$s$ is $c_{\mathrm{gh}}(s) = -2(6s^2 - 6s + 1)$
(the standard $bc$-ghost formula; see \cite{FMS86}).
The total ghost contribution is
$c_{\mathrm{gh}}^{\mathrm{total}}
= \sum_{s=2}^N c_{\mathrm{gh}}(s)
= -\sum_{s=2}^N 2(6s^2 - 6s + 1)
= -\alpha_N$.

For the closed-form evaluation, we compute:
\begin{align*}
\sum_{s=2}^{N}(6s^2 - 6s + 1)
&= 6\sum_{s=2}^{N} s^2 - 6\sum_{s=2}^{N} s
  + \sum_{s=2}^{N} 1 \\
&= 6\Bigl[\tfrac{N(N+1)(2N+1)}{6} - 1\Bigr]
  - 6\Bigl[\tfrac{N(N+1)}{2} - 1\Bigr]
  + (N-1) \\
&= N(N+1)(2N+1) - 3N(N+1) + (N-1) \\
&= N(N+1)(2N-2) + (N-1) \\
&= (N-1)\bigl[2N(N+1) + 1\bigr] \\
&= (N-1)(2N^2 + 2N + 1).
\end{align*}
Multiplying by~$2$ gives~\eqref{eq:wn-complementarity}.

The first values are:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{lcccccc}
\toprule
$N$ & $2$ & $3$ & $4$ & $5$ & $6$ & General \\
\midrule
$\alpha_N$ & $26$ & $100$ & $246$ & $488$ & $850$
  & $2(N-1)(2N^2+2N+1)$ \\
Self-dual $c$ & $13$ & $50$ & $123$ & $244$ & $425$
  & $(N-1)(2N^2+2N+1)$ \\
\bottomrule
\end{tabular}
\end{center}
The agreement with the slab central charges
(Theorem~\ref{thm:central-charge-shift})
confirms the identity $3(2s-1)^2 - 1 = 2(6s^2 - 6s + 1)$.
\end{proof}

\begin{remark}[Complementarity sum for $\mathcal{W}_N$ vs.\
Kac--Moody]
\label{rem:wn-complementarity-nonzero}
\index{complementarity constant!non-vanishing}%
For Kac--Moody algebras $V_k(\fg)$, the Feigin--Frenkel

This is the higher-spin analogue of the Virasoro phenomenon
$\kappa + \kappa^! = 13 \ne 0$.
The sequence $\alpha_N/2 = 13, 50, 123, 244, \ldots$ grows

self-duality.
\end{remark}
