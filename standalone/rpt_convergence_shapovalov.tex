\title{Level-by-Level Rationality and the Shapovalov Summability Problem\\
for the Virasoro Spectral $R$-Matrix}
\author{Raeez Lorgat}

\begin{document}
\maketitle

\begin{abstract}
Level-by-level rationality of the bar-cobar spectral $R$-matrix
for $\mathrm{Vir}_c$ is proved (bar-cobar $\Ainf$ operations plus
Shapovalov inversion: poles at the Kac determinant zero locus).
Convergence of the level sum on the principal series requires a
summable level majorant.  Fixed-basis condition-number control for the
Shapovalov matrices is one input, but a polynomial lower bound cannot
by itself control exponential numerators and partition growth.  The Kac
determinant supplies a
product formula and finite-level nonvanishing away from the null-vector
locus; it does not supply a least-eigenvalue lower bound.  Numerical
data through level~$8$ are a fixed-PBW-basis conditioning check.  The
full identification with the Ponsot--Teschner fusion kernel also
requires the four-point channel, contour, normalization, and external
Liouville momenta to be matched.  The missing inputs are the
summable majorant and the channel comparison.  Relative to the
five-family theorem of Vol.~I~\cite{VolI}, the Virasoro spectral
$R$-matrix is the adjacent class-$M$ analogue of the
Shapovalov-witness principle carried by reflection-equation Shapovalov
on types~B/C/D; the
rank-$1$ Baxter hyperplane is a type-$A$ rational artifact
(Vol.~I, Proposition ``Baxter hyperplane as a type-$A$ rational
artifact'').
\end{abstract}

\section{Statement and reduction}

\begin{theorem}[Conditional identification with PT]
\label{thm:rpt-conditional}
Fix the real Liouville chamber
$b \in \mathbb{R}_{>0}$, $b^2 \notin \mathbb{Q}$,
$Q=b+b^{-1}$, and $c=1+6Q^2$.  Fix a Ponsot--Teschner four-point
block~\cite{PT01,TeschnerLiouville} with external momenta
$\alpha_i=Q/2+iP_i$, internal $s$- and $t$-channel momenta
$\alpha_s,\alpha_t$, compact real principal-series ranges for the
$P_\bullet$, the standard contour separating the two PT pole families,
and the PT normalization of conformal blocks.  After complexifying the
$P_\bullet$-variables, remove the singular hyperplanes
$P_\bullet=\pm i(rb+s b^{-1})/2$ and keep the contour chamber fixed.
Assume Conjecture~\ref{conj:shapovalov-growth} below on this compact
chamber and assume that the bar-cobar spectral $R$-matrix is expressed
in this same channel and normalization and satisfies the full
$b,b^{-1}$ shift system there.  Then the bar-cobar spectral
$R$-matrix of $\mathrm{Vir}_c$ equals the Ponsot--Teschner fusion
kernel on that principal-series chamber.  Along any complex
$b$-path on which both sides are analytically continued with the same
pole prescription, contour chamber, normalization, and full shift
system, this equality continues by the same uniqueness theorem.
\end{theorem}

The conditional identification has one fixed-level algebraic input and
two proof obligations: a summable level majorant and the normalized PT
channel comparison.

\begin{proposition}[Level-by-level rationality]
\label{prop:rpt-level-rationality}
At fixed levels $(N_1, N_2)$, each matrix element of the bar-cobar
spectral $R^{\mathrm{bar},(N_1,N_2)}$ is a rational function of
the central charge and the weights
$h_i = \alpha_i(Q - \alpha_i)$, with possible poles only along the
finite-level Kac determinant divisors
$\det G_{N_j}(c,h_i)=0$; equivalently, in Liouville coordinates,
along $h_i=h_{r,s}(c)$ for $rs\leq N_j$.
\end{proposition}

\begin{proof}
$\Ainf$ operations $m_k$ are polynomial in spectral parameters and
central charge; evaluation on the Verma basis is polynomial in $h_i$;
homotopy transfer introduces Gram (Shapovalov) inverses, contributing
poles at the Kac determinant zeros $h = h_{r,s}$. At each finite
level the number of operations is finite, and rationality is
preserved.
\end{proof}

\begin{proposition}[Eberhardt uniqueness, normalized full-shift form]
\label{prop:eberhardt-uniqueness}
A meromorphic function of the Liouville momenta satisfying the full
$b$ and $b^{-1}$ shift system, the Ponsot--Teschner symmetry and
normalisation conditions, and the principal-series analyticity
hypotheses is unique up to the stated normalisation.  Under that
normalisation it is the Ponsot--Teschner kernel
\cite[Theorem~1.1]{Eberhardt23}.
\end{proposition}

For the bar-cobar spectral $R$-matrix, one must verify the full
shift system, symmetry, and normalisation before invoking
Proposition~\ref{prop:eberhardt-uniqueness}.  The analytic problem is
convergence of the level sum
$R^{\mathrm{bar}} = \sum_{N=0}^\infty R^{(N)}$ on compact subsets of
the principal-series domain.

\begin{proof}[Proof of Theorem~\ref{thm:rpt-conditional}]
Proposition~\ref{prop:rpt-level-rationality} places every fixed-level
term on the complement of the finite Kac divisor in the chosen
Liouville chamber.  Conjecture~\ref{conj:shapovalov-growth} supplies
a Weierstrass majorant on compact subsets of the chamber complement,
so the level sum defines a meromorphic kernel there with the stated
pole prescription.  The channel, contour, external-momentum, and
normalization hypotheses identify this kernel with a candidate in the
same four-point problem as the PT kernel.  The full $b,b^{-1}$ shift
system, symmetry, and analyticity hypotheses are exactly the inputs of
Proposition~\ref{prop:eberhardt-uniqueness}; hence the two kernels
coincide on the chamber.  Analytic continuation along a path preserving
the same pole prescription and contour chamber carries equality of the
continued solutions.
\end{proof}

\section{The convergence question}

\begin{conjecture}[Summable level bound]
\label{conj:shapovalov-growth}
Fix $b\in\mathbb{R}_{>0}$ with $b^2\notin\mathbb{Q}$ and a compact
subset of the real unitary principal-series chamber
$h = Q^2/4 + P^2$.  In the complexified $P$-plane, remove the Kac
hyperplanes $P=\pm i(rb+s b^{-1})/2$ and work on a compact subset of
one chamber of the complement.  There the level-$N$ contribution to
the bar-cobar spectral $R$-matrix satisfies
\[
  \|R^{(N)}\| \le M_N,
  \qquad \sum_{N\ge0} M_N < \infty .
\]
Equivalently, the Shapovalov denominator estimates, numerator
cancellations, and partition growth combine to give a Weierstrass
majorant.
\end{conjecture}

At $c = 25$, $h = 2$ ($b = 1$, $P = 1$) the fixed partition/PBW-basis
conditioning computation gives:
\[
\begin{array}{c|cccccccc}
N & 1 & 2 & 3 & 4 & 5 & 6 & 7 & 8 \\
\hline
\lambda_{\min} & 4 & 14.8 & 49.2 & 118.5 & 245.4 & 408.9 & 668.8 &
975.7 \\
\lambda_{\min}/p(N) & 4.0 & 7.4 & 16.4 & 23.7 & 35.1 & 37.2 & 44.6 &
44.4
\end{array}
\]
This data point lies at rational $b^2=1$, outside the irrational
$b^2$ hypothesis above.  It is a numerical check on Shapovalov
conditioning, not evidence for the summable majorant by itself.

\section{Analytic proof obligations for Theorem~\ref{thm:rpt-conditional}}

\paragraph{Denominator.}
On the real unitary principal series, after fixing conventions for
$Q=b+b^{-1}$, $h=Q^2/4+P^2$, and
$P_{r,s}=(rb+s b^{-1})/2$,
\[
  h-h_{r,s}=P^2+P_{r,s}^2 .
\]
The Kac determinant gives the finite-level product formula
\cite{KacRaina,FeiginFuchs90}
\[
\det G_N
\;=\;
C_N\prod_{rs \le N}(P^2 + P_{r,s}^2)^{p(N-rs)}
\]
in the chosen PBW normalization, with $C_N\ne0$. This proves
finite-level nonvanishing away from the Kac locus, not
Conjecture~\ref{conj:shapovalov-growth}.  Determinants do not control
least singular values without condition-number information.  The
expected route is a direct condition-number estimate for $G_N$, using
the block decomposition of descendants into $L_{-1}$-chains and
interlacing under partition refinement.

\paragraph{Numerator.}
The level-$N$ contribution to the $R$-matrix involves iterated
$\Ainf$ wheel integrals.  The presently available estimate is only
the naive Wick/tree count: the $\FM_k(\C)$ volume times the OPE
coefficient bound, with no cancellation.  This gives no better than a
$C^N\cdot N!$ envelope for $N$ mode contractions, hence no summable
majorant.

\paragraph{Tree refinement.}
The crude $N!$ is too weak, and even a polynomial lower bound
on $\lambda_{\min}(G_N)$ cannot control an exponential numerator after
partition growth is included.  The needed estimate is the summable
bound of Conjecture~\ref{conj:shapovalov-growth}: either the numerator
has subexponential decay after $\Ainf$ cancellations, or the
Shapovalov inverse has a stronger effective denominator than the
present determinant estimate gives.

\paragraph{Roussillon route.}
Roussillon~\cite{Roussillon24} gives a convergent series
representation of $F^{\mathrm{PT}}$ in the normalization and parameter
regime treated there. To use it here one must first match the PT
four-point channel, contour, normalization, and external momenta, and
then match the bar-cobar series term by term. Only under that
additional comparison does meromorphicity follow from Roussillon's
convergence analysis.

\section{Position inside the MC3 five-family mechanism}

The five-family thick-generation theorem
of Vol.~I~\cite{VolI} supplies the Shapovalov-witness template.  Its
five unconditional evaluation-generated rows are:

\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lll}
\textbf{Family} & \textbf{Shapovalov witness} & \textbf{Thick-generation status} \\
\hline
Type~$A$ rational Yangian & asymptotic prefundamentals (Hernandez--Jimbo) &
unconditional \\
Types~B/C/D twisted Yangian & reflection-equation Shapovalov
($\operatorname{sdet}K(u)$) & unconditional \\
Uniform simple~$\fg$ & Chari--Moura multiplicity-free
$\ell$-weights & unconditional \\
Elliptic $\mathrm{sl}_N$ & theta-divisor complement
(Felder--Varchenko) & unconditional (theta-divisor) \\
Super $\mathrm{gl}_{m|n}$ & parity-balance hyperplane (Nazarov
Berezinian) & unconditional (hyperplane) \\
\hline
Virasoro bar-cobar & summable Shapovalov/level majorant on principal
series & conjectural
\end{tabular}
\end{center}

The Virasoro line is not a sixth case of the five-family theorem.  It
is the class-$M$ analogue whose comparison theorem is
Theorem~\ref{thm:rpt-conditional} and whose missing analytic input is
Conjecture~\ref{conj:shapovalov-growth}.
The five-family mechanism retires the rank-$1$ Baxter hyperplane as a
\emph{type-$A$ rational artifact}
(Vol.~I, Proposition ``Baxter hyperplane as a type-$A$ rational
artifact''): the
Baxter constraint $b = a - \tfrac12$ is visible only in the
rational-form presentation of $Y_\hbar(\mathrm{sl}_N)$ and does not
persist under degeneration from any other chart. In the present
Virasoro setting the analogous codimension-$1$ locus is the Kac
determinant zero locus $h = h_{r,s}$, which is globally intrinsic
(not chart-dependent); the corresponding Shapovalov witness is the
fixed-basis conditioning datum, while the summable level majorant is
still unproved.

\begin{remark}[Silent non-coverage]
\label{rem:silent-non-coverage}
The MC3 full-DK-generation theorem
of Vol.~I is silent on the
following families outside the evaluation-generated core: logarithmic
$\mathcal{W}(p)$-algebras (finite-dimensional Zhu algebra does not
imply a uniform Massey bound; Gurarie and Flohr exhibit logarithmic
amplitudes obstructing that route~\cite{Gurarie93,Flohr96}, and
tempering itself is open), $N = 2$ superconformal algebras, cosets
(beyond irrational cosets), non-rational lattice VOAs, and roots of
unity. The conditional Theorem~\ref{thm:rpt-conditional} applies only
to the chosen irrational principal-series Virasoro chamber, which
excludes the logarithmic $\mathcal{W}(p)$ class entirely.
\end{remark}

\section{Analytic proof obligations}

\begin{enumerate}[nosep]
\item Extend the fixed-PBW Shapovalov conditioning computation to
$N = 12$--$15$ with arbitrary-precision arithmetic and test several
principal-series points, including irrational $b^2$.
\item Prove condition-number estimates for $G_N$ via the block
structure of Virasoro descendants into $L_{-1}$-chains and interlacing
under partition refinement. The Kac determinant itself supplies only
product/nonvanishing data.
\item Replace the crude $N!$ numerator bound by the Stasheff-polytope
bound, using the $\Ainf$ relations to organise the cancellations.
\item Prove a summable majorant for the complete level contribution,
including partition growth. Only then apply the Weierstrass M-test on
compact subsets of the principal series.
\item Fix the PT four-point channel, contour, external momenta, and
normalization; verify the normalized full $b,b^{-1}$ shift system,
symmetry, and normalisation hypotheses of
Proposition~\ref{prop:eberhardt-uniqueness}; then conclude
$R^{\mathrm{bar}} = F^{\mathrm{PT}}$.
\end{enumerate}
