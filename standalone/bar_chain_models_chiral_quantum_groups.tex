\section{The Yangian as $E_1$-chiral quantum group}
\label{sec:yangian}
%% ====================================================================

The Yangian $Y_\hbar(\fsl_2)$ is presented by generators
$t^{(r)}_{ij}$, $1 \le i,j \le 2$, $r \ge 1$, assembled into the
generating series $T(u) = \mathbf{1} + \sum_{r \ge 1} T^{(r)} u^{-r}$
with $T^{(r)} = (t^{(r)}_{ij})$, subject to the RTT relation
\begin{equation}\label{eq:rtt}
R(u-v)\, T_1(u)\, T_2(v) = T_2(v)\, T_1(u)\, R(u-v),
\qquad
R(z) = 1 - \frac{\hbar\, P}{z},
\end{equation}
where $P \in \End(\bC^2 \otimes \bC^2)$ is the permutation.

\smallskip
\noindent\textit{Multiplication.}
The RTT relation~\eqref{eq:rtt} defines a $\bZ$-filtered associative
algebra.  It is \emph{not} commutative: $[t^{(1)}_{ij},\, t^{(1)}_{kl}]
= \hbar(\delta_{kj} t^{(1)}_{il} - \delta_{il} t^{(1)}_{kj})$.
Multiplication is $E_1$ and no more.

\smallskip
\noindent\textit{Coproduct.}
The spectral coproduct
\begin{equation}\label{eq:coproduct}
\Delta_z \colon Y_\hbar(\fsl_2) \to
Y_\hbar(\fsl_2) \,\widehat{\otimes}\, Y_\hbar(\fsl_2),
\qquad
\Delta_z(T(u)) = T(u) \otimes T(u-z),
\end{equation}
is coassociative: $(\id \widehat{\otimes}\, \Delta_w) \circ \Delta_z
= (\Delta_z \widehat{\otimes}\, \id) \circ \Delta_{z+w}$.  At $z=0$
it reduces to the Drinfeld coproduct $\Delta_0(T(u)) = T(u) \otimes T(u)$.

\smallskip
\noindent\textit{$R$-matrix and quasi-triangularity.}
The Yang $R$-matrix $R(z) = 1 - \hbar P/z$ satisfies the Yang--Baxter equation
\begin{equation}\label{eq:ybe}
R_{12}(z_1 - z_2)\, R_{13}(z_1 - z_3)\, R_{23}(z_2 - z_3)
= R_{23}(z_2 - z_3)\, R_{13}(z_1 - z_3)\, R_{12}(z_1 - z_2).
\end{equation}
Quasi-triangularity:
$R(z)\, \Delta_z(a) = \Delta_z^{\op}(a)\, R(z)$ for all $a \in Y_\hbar(\fsl_2)$.

\smallskip
\noindent\textit{Braided module category.}
For $Y_\hbar(\fsl_2)$-modules $M$, $N$, the spectral coproduct
\eqref{eq:coproduct} defines a fusion product $M \otimes_z N$ by
pulling back the $(Y_\hbar \widehat\otimes Y_\hbar)$-action.
The braiding
\[
\beta_{M,N}(z) \colon M \otimes_z N
\xrightarrow{\;\; R(z) \cdot \sigma \;\;}
N \otimes_{z} M
\]
satisfies the braid relation by \eqref{eq:ybe}.

\smallskip
\noindent\textit{What makes $Y_\hbar(\fsl_2)$ genuinely $E_1$.}
The RTT relation \eqref{eq:rtt} is \emph{not} derived from a local OPE
on a curve.  There is no vertex algebra $V$ for which
$Y_\hbar(\fsl_2)$ arises as a chiral envelope.  The ordering of
tensor factors is independent input.  In the hierarchy
\[
E_1\text{-chiral}
\;\supsetneq\;
E_\infty\text{-chiral (vertex algebras)},
\]
the Yangian sits in $E_1 \setminus E_\infty$: genuinely nonlocal.

This example earns the following definition.

\section{$E_1$-chiral quantum groups}
\label{sec:def}
%% ====================================================================

\begin{definition}[Ordered Koszul chiral algebra;
\ClaimStatusProvedElsewhere (Vol~I
\texttt{def:ordered-koszul-chiral-algebra})]%
\label{def:ordered-koszul-chiral-algebra-standalone}%
An $E_1$-chiral algebra $\cA$ on a smooth curve $X$ is
\emph{ordered Koszul} if (i) its ordered bar complex
$\Bord(\cA) = T^c(s^{-1}\bar\cA)$ over the pure braid Orlik--Solomon
operad $\mathrm{OS}(\mathrm{PBr}_n)$ is quadratic-Koszul in the
Shelton--Yuzvinsky sense; (ii) the induced $\Ainf$-structure
$\{m_k^{\mathrm{ord}}\}_{k\ge 2}$ on $\cA$ satisfies the Stasheff
boundary-of-associahedra relations; (iii) the $R$-twisted
$\Sigma_n$-descent identifying $\Bord(\cA)$ with the symmetric-bar
shadow $\BSigma(\cA)$ holds at the level of chain complexes.
\end{definition}

\begin{proposition}[Yangian as ordered-Koszul $E_1$-chiral algebra;
\ClaimStatusProvedElsewhere (Vol~I
\texttt{prop:yangian-ordered-koszul})]%
\label{prop:yangian-ordered-koszul-standalone}%
The Yangian $\Yhbar(\fg)^{\mathrm{ch}}$ in Drinfeld's second
presentation, extended to the $\cD$-module on a smooth curve~$X$
by the factorisation-algebra construction of
\cite{CostelloWittenYamazaki}, is an ordered Koszul chiral algebra
in the sense of
Definition~\ref{def:ordered-koszul-chiral-algebra-standalone}.
Equivalently (Etingof--Kazhdan 2005 + this monograph's
$\Bord$ construction): the pure braid Orlik--Solomon Koszulity
produces a flat $A_\infty$ structure whose symmetrisation recovers
the Etingof--Kazhdan quantum vertex algebra on the Koszul locus.
\end{proposition}

\begin{remark}[Why the Yangian is the archetypal $E_1$-chiral input]%
\label{rem:yangian-archetypal}%
The five notions of $E_1$-chiral algebra (AP161: strict $\mathrm{ChirAss}$,
$A_\infty$ in $\End^{\mathrm{ch}}_\cA$, Etingof--Kazhdan quantum VA,
$A_\infty$ in $E_1$-chiral, factorization on $\Ran^{\mathrm{ord}}(X)$)
coincide on the ordered Koszul locus where the Drinfeld associator
acts freely, through
Proposition~\ref{prop:yangian-ordered-koszul-standalone}. The Yangian
sits in the complement of $E_\infty$-chiral (standard vertex algebras)
and is genuinely nonlocal.
\end{remark}

\begin{definition}[$E_1$-chiral quantum group]%
\label{def:e1-chiral-qg}%
An \textbf{$E_1$-chiral quantum group} on a smooth curve $X$ is a
tuple $(\cA,\, \mu,\, \Delta,\, R,\, S)$ where:
\begin{enumerate}[label=(\roman*)]
\item $\cA$ is an $E_1$-chiral algebra on $X$: an $E_1$-algebra object in
  factorisation algebras on $X$;
\item $\mu$ is the $E_1$-chiral multiplication (the $\Ainf$-structure
  $\{m_k\}_{k \ge 2}$ encoding the ordered OPE);
\item $\Delta \colon \cA \to \cA \otimes \cA$ is a chiral coproduct
  (coassociative coalgebra structure on $\cA$ itself, in the chiral
  tensor category);
\item $R(z) \in \cA^{\widehat\otimes 2}$ is a spectral $R$-matrix
  satisfying the Yang--Baxter equation \eqref{eq:ybe};
\item $R(z)\, \Delta(a) = \Delta^{\op}(a)\, R(z)$ for all $a \in \cA$
  (quasi-triangularity);
\item $S \colon \cA \to \cA$ is an antipode satisfying
  $\mu \circ (\id \otimes S) \circ \Delta = \eta \circ \varepsilon$
  and its mirror.
\end{enumerate}
\end{definition}

The coproduct $\Delta$ is structure on $\cA$.  It is not the
deconcatenation coproduct on the bar complex $B(\cA)$:
deconcatenation is a structural feature of any bar construction
and carries no information about $\cA$ beyond what the product
already encodes.

\begin{theorem}[Categorical level shift]\label{thm:level-shift}
Let $(\cA, \mu, \Delta, R, S)$ be an $E_1$-chiral quantum group.
Then:
\begin{enumerate}[label=\textup{(\alph*)}]
\item $\cA$ is an $E_1$-algebra.
\item $\Mod_\cA$ is an $E_2$-algebra object in $\Cat$\textup: it is a
  braided monoidal category with braiding $\beta_{M,N}(z) = R(z) \cdot \sigma$.
\item The derived chiral centre $Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)
  := \ChirHoch^\bullet(\cA, \cA)$ carries an $E_3$-chiral algebra
  structure via the chiral higher Deligne theorem
  (Vol~II \texttt{thm:chiral-higher-deligne}). On cohomology this
  is $E_2$; the $E_3$ upgrade at chain level is the content of the
  pentagon edges $(3)\leftrightarrow(4)\leftrightarrow(5)$ of the
  $\SCchtop$-heptagon.
\end{enumerate}
$E_3$ lives on $Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)$, not on $\cA$.
\end{theorem}

\begin{proof}
(a) is the definition.

(b) The coproduct $\Delta$ defines the monoidal product
$M \otimes_\Delta N$ on $\Mod_\cA$ by restriction of scalars
along $\Delta$.  Coassociativity of $\Delta$ gives associativity of
$\otimes_\Delta$.  The $R$-matrix $R(z)$ and quasi-triangularity
give the braiding natural transformation
$\beta_{M,N} \colon M \otimes N \to N \otimes M$
satisfying the hexagon axioms (which follow from the YBE
\eqref{eq:ybe}).  The antipode $S$ gives duals.

(c) The chiral higher Deligne theorem (Vol~II
\texttt{thm:chiral-higher-deligne}): for an $E_1$-chiral algebra
$\cA$ on a smooth curve $X$, the chiral Hochschild cochains
$\ChirHoch^\bullet(\cA, \cA)$ carry a canonical $E_3$-chiral
structure delivered by the pentagon edges
$(3)\leftrightarrow(4)\leftrightarrow(5)$ of the
$\SCchtop$-heptagon (Vol~II
\texttt{ch:sc-chtop-heptagon}). On cohomology this reduces to the
Gerstenhaber $E_2$ of the classical chiral Deligne conjecture; the
$E_3$ upgrade is the present chapter's content. Associator discipline
(AP171): the chain-level $E_3$-action depends on a choice of
Drinfeld associator, while the cohomological action is
associator-free.
\end{proof}

\section{The bar complex: what it sees and what it does not}
\label{sec:bar}
%% ====================================================================

Let $\cA$ be an $E_1$-chiral algebra on a smooth curve $X$.

\begin{proposition}[Bar complex as $E_1$ coalgebra
{\cite[Theorem~A]{Vol1}}]\label{prop:bar}
The bar construction
\[
B(\cA) = \bigoplus_{n \ge 0} (s\cA)^{\otimes n},
\qquad
d_B = \sum_{k \ge 2} \sum_i m_k^{(i)},
\qquad
\Delta(a_1|\cdots|a_n) = \sum_{i=0}^{n}
(a_1|\cdots|a_i) \otimes (a_{i+1}|\cdots|a_n),
\]
is an $E_1$ dg coassociative coalgebra on $\Ran(X)$.
The differential $d_B$ is a coderivation of $\Delta$.
The cobar construction $\Omega$ provides a left adjoint:
$\Omega B(\cA) \simeq \cA$ for augmented $\cA$.
\end{proposition}

The bar complex computes $B(\cA) \simeq k \otimes_\cA^{\mathbf{L}} k$,
the factorisation homology of the interval $[0,1]$ with augmentation
boundary conditions.  It is not factorisation homology of the line:
$\int_\bR \cA \simeq \cA$.

\begin{proposition}[Classical $r$-matrix from collision residues]
\label{prop:r-classical}
Let $\cA = V_k(\fg)$ be the Kac--Moody vertex algebra at level $k$.
The genus-$0$, degree-$2$ collision residue of the bar differential gives
\begin{equation}\label{eq:classical-r}
r(z) = \frac{k\,\Omega}{z},
\end{equation}
where $\Omega \in \fg \otimes \fg$ is the Casimir.  This $r(z)$
satisfies the classical Yang--Baxter equation.
\end{proposition}

\begin{proof}
The $m_2$ component of $d_B$ encodes the singular OPE
$J^a(z) J^b(w) \sim k\, \delta^{ab}/(z-w) + f^{ab}_c J^c(w)/(z-w)$
(in the orthonormal basis where the Killing form is $\delta^{ab}$).
The $\dlog$-extraction procedure replaces the OPE pole
$1/(z-w)$ by the logarithmic form $\dlog(z-w)$.  The collision
residue at $z=w$ in degree 2 extracts the coefficient
$k\,\Omega = k \sum_a J^a \otimes J_a$, divided by the spectral
parameter $z = z_1 - z_2$.  The CYBE follows from the Jacobi
identity of $\fg$ (equivalently, from the codimension-2 stratum
of $\mathrm{FM}_3(\bC)$, which imposes the three-term relation).
\end{proof}

\begin{proposition}[Curvature]\label{prop:curvature}
At genus $g \ge 1$, the bar complex has
\begin{equation}\label{eq:curvature}
d_B^2 = \kappa_{\mathrm{ch}} \cdot \omega_g,
\end{equation}
where $\kappa_{\mathrm{ch}}$ is the shadow obstruction class and
$\omega_g$ is the Arakelov $(1,1)$-form on the genus-$g$ surface,
normalised so that $\int_{\Sigma_g} \omega_g = 1$.
\end{proposition}

The bar complex sees the \emph{product} (fusion, encoded in $d_B$)
and the \emph{classical $r$-matrix} (from collision residues).

It does \emph{not} see:
\begin{enumerate}[label=(\roman*)]
\item the chiral coproduct $\Delta \colon \cA \to \cA \otimes \cA$
  (the coproduct encodes splitting, not fusion; it is independent
  algebraic structure);
\item the full quantum $R$-matrix $R(z)$ (only its classical limit
  $r(z)$ appears in the shadow tower);
\item quasi-triangularity or the antipode.
\end{enumerate}

The shadow obstruction tower $\{F_g\}_{g \ge 1}$, extracted from
\eqref{eq:curvature}, detects collision data.  The coproduct encodes
the opposite operation and is invisible to the tower.

Sources of the chiral coproduct: Gaiotto--Rap\v{c}\'ak--Zhou
\cite{GRZ} construct coproducts on deformed double current algebras
of $\fgl_K$ from M2--M5 fusions;
Jindal--Kaubrys--Latyntsev \cite{JKL} construct vertex coproducts
on CoHAs of quivers with potential, recovering Drinfeld's deformed
coproduct for ADE quivers.

\section{The $E_N$ ladder}
\label{sec:ladder}
%% ====================================================================

For an $E_1$-chiral algebra $\cA$ on $X$, the following structures
are obtained by ascending the $E_N$ hierarchy.  Each step adds one
$E_1$ factor via Dunn additivity.

\subsection*{$E_1$: the boundary algebra}

$\cA$ is $E_1$-chiral: an $E_1$-algebra in factorisation algebras
on $X$.  The bar complex $B(\cA)$ is the $E_1$ dg coassociative
coalgebra of Proposition~\ref{prop:bar}.

\emph{Status: proved} (Proposition~\ref{prop:bar}).

\subsection*{$E_1 \to E_2$: Hochschild and Deligne}

\begin{theorem}[$E_2$ on the derived centre]\label{thm:e2}
The chiral derived centre
\[
Z^{\mathrm{der}}_{\mathrm{ch}}(\cA) = C^\bullet_{\mathrm{ch}}(\cA,\cA)
\]
carries an $E_2$-chiral algebra structure.  The $E_2$ decomposition
$E_2 \simeq E_1^{\mathrm{hol}} \otimes E_1^{\mathrm{top}}$
\textup(Dunn additivity\textup) gives:
\begin{enumerate}[label=\textup{(\alph*)}]
\item $E_1^{\mathrm{hol}}$: the chiral algebra structure of
  $Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)$ on $X$;
\item $E_1^{\mathrm{top}}$: the $\Ainf$-algebra structure in the
  transverse direction.
\end{enumerate}
The pair $(Z^{\mathrm{der}}_{\mathrm{ch}}(\cA),\, \cA)$ is the
Swiss-cheese datum: the bulk $E_2$ acts on the boundary $E_1$.
\end{theorem}

\emph{Status: proved.}  The chiral Deligne conjecture follows from
Kontsevich formality and homotopy transfer to the chiral setting.
Dunn additivity applies to
$Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)$, not to $\cA$.

\subsection*{$E_2 \to E_3$-chiral: chiral higher Deligne}

The $E_3$-chiral structure on
$Z^{\mathrm{der}}_{\mathrm{ch}}(\cA) = \ChirHoch^\bullet(\cA, \cA)$
is delivered unconditionally on the Koszul locus by the chiral higher
Deligne theorem (Vol~II \texttt{thm:chiral-higher-deligne}), via the
pentagon edges $(3)\leftrightarrow(4)\leftrightarrow(5)$ of the
$\SCchtop$-heptagon. For chiral algebras of gauge-theoretic origin,
this $E_3$-chiral structure is realised geometrically by the
factorisation algebra of a 3d holomorphic-topological theory on
$X \times \bR$.

\begin{theorem}[Chiral higher Deligne;
\ClaimStatusProvedElsewhere (Vol~II
\texttt{thm:chiral-higher-deligne})]%
\label{thm:chiral-higher-deligne-standalone}%
Let $\cA$ be an $E_1$-chiral algebra on a smooth complex curve $X$ in
the Koszul locus. Then $Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)$ is the
universal $E_3$-chiral algebra acted on by $\SCchtop$ via the
heptagon. Theorem~H concentration $\ChirHoch^{\ge 3}(\cA) = 0$ is
the \emph{consequence} of $E_3$-rigidity-at-a-point plus chiral PBW
collapse, not a hypothesis.
\end{theorem}

\begin{theorem}[$E_3$-chiral realisation for Kac--Moody {\cite{CFG}};
\ClaimStatusProvedElsewhere]%
\label{thm:e3-km}%
For $\cA = V_k(\fg)$ at non-critical level $k \ne -h^\vee$,
holomorphic Chern--Simons theory on $X \times \bR$ provides the
3d HT bulk.  The factorisation algebra of observables on
$X \times \bR$ is an $E_3$-chiral algebra, and its restriction to
the boundary $X \times \{0\}$ recovers $V_k(\fg)$. This geometric
realisation agrees with the chiral higher Deligne
$E_3$-chiral structure of
Theorem~\ref{thm:chiral-higher-deligne-standalone} on cohomology;
the chain-level agreement is associator-dependent
(AP171).
\end{theorem}

\emph{Scope (AP-TOPOLOGIZATION).}
Kac--Moody proved by Costello--Li and
Costello--Francis--Gwilliam. For $\cW$-algebras via DS reduction,
the bulk comes from DS boundary conditions and the BRST identity
$T_{\mathrm{DS}} = [Q_{\mathrm{tot}}, G'_f]$ holds for any nilpotent
$f$ (the improvement term uses only Cartan currents). For general
freely-generated conformal VAs, Khan--Zeng's 3d Poisson sigma model
supplies the bulk. The chiral higher Deligne route gives the
$E_3$-chiral structure unconditionally on the Koszul locus.

\subsection*{$E_3$-chiral $\to$ $E_3$-topological: topologisation}

\begin{theorem}[Topologisation for Kac--Moody;
\ClaimStatusProvedElsewhere]%
\label{thm:topologise}%
Let $\cA = V_k(\fg)$ at non-critical level $k \ne -h^\vee$.
Write $T(z) = \{Q, G(z)\}$ where $T(z)$ is the Sugawara conformal
vector, $Q$ is the BRST operator, and $G(z)$ is the antighost.
Then:
\begin{enumerate}[label=\textup{(\alph*)}]
\item Translations along $X$ are $Q$-exact on the chain-level
  antighost refinement of Vol~I
  \texttt{thm:chain-level-E3-top-class-L}, so the complex structure
  of $X$ is irrelevant in the chain-level BRST cohomology.
\item The $E_3$-chiral structure on
  $Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)$
  (Theorem~\ref{thm:chiral-higher-deligne-standalone},
  Theorem~\ref{thm:e3-km}) descends to an
  $E_3$-topological structure on the original chain-level
  complex (class L via Dunn-additivity of
  $F^{\mathrm{CS}}_k$ on $\bR \times X$).
\item The resulting $E_3$-topological theory is Chern--Simons at
  level $k + h^\vee$.
\end{enumerate}
\end{theorem}

\begin{theorem}[Topologisation tower, all classes;
\ClaimStatusProvedElsewhere (Vol~I \texttt{thm:topologization-tower},
Vol~II \texttt{thm:e-infinity-topologization-ladder})]%
\label{thm:topologisation-tower-standalone}%
On the original complex:
\begin{enumerate}[label=\textup{(\roman*)}]
\item class G (Heisenberg) $E_3^{\mathrm{top}}$ via Dunn on the
  commutative filtration;
\item class L ($V_k(\fg)$, $k \ne -h^\vee$) $E_3^{\mathrm{top}}$
  via Dunn on $F^{\mathrm{CS}}_k$ with explicit
  $\eta_1^{(i)}, \eta_1^{(ii)}$ making
  $[Q, \tilde G_1] = T_{\mathrm{Sug}}$ strict chain-level;
\item class C ($\beta\gamma$, $bc$) $E_3^{\mathrm{top}}$ via FMS
  bosonisation reducing to class G.
\end{enumerate}
In the weight-completed category:
\begin{enumerate}[label=\textup{(\roman*)}]\setcounter{enumi}{3}
\item class M (Virasoro, $\cW_N$) $E_3^{\mathrm{top}}$ via half-BRST
  and the MC5 weight-completed pattern (\texttt{thm:completed-bar-cobar-strong}
  + \texttt{prop:standard-strong-filtration}).
\end{enumerate}
At critical level $k = -h^\vee$, Sugawara degenerates and the
derived centre stops at $E_2^{\mathrm{top}}$ (dimension drop, not
failure).

\noindent
Higher-spin extension (Vol~II
\texttt{thm:e-infinity-topologization-ladder}): $k$ inner
stress tensors yield $E_{k+2}^{\mathrm{top}}$ via Dunn; for
$\cW_N$ this gives $E_{N+1}^{\mathrm{top}}$; for $\cW_\infty$
this is the Platonic endpoint $E_\infty^{\mathrm{top}}$.
\end{theorem}

\begin{conjecture}[Topologisation for general conformal chiral
algebras, original complex class M; AP-TOPOLOGIZATION]%
\label{conj:topologise-general}%
Let $\cA$ be a class-M chiral algebra at non-critical level. The
weight-completed $E_3^{\mathrm{top}}$ structure of
Theorem~\ref{thm:topologisation-tower-standalone}(iv) lifts to an
$E_3^{\mathrm{top}}$ structure on the original complex (not merely
weight-completed). Open for Virasoro and principal $\cW_N$ at generic
central charge on the original complex; BP (minimal $\cW^{(2)}(\fsl_3)$)
has a strict chain-level original-complex proof via branched-cover
integralisation plus Galois descent (Vol~II
\texttt{chap:bp-chain-level-strict-platonic}), indicating the route
for general class~M is branched-cover reduction to class~L upstairs.
\end{conjecture}

At the \emph{critical level} $k = -h^\vee$, the Sugawara
construction degenerates.  No non-critical conformal vector is
available, so topologisation fails: the derived center remains
$E_3$-chiral, but not $E_3$-topological.

\begin{remark}[The generic case is $\mathrm{SC}^{\mathrm{ch,top}}$,
not $E_3$]
$E_3$-topological requires \emph{both} a conformal vector at
non-critical level \emph{and} a 3d holomorphic-topological bulk
theory.  Chiral algebras lacking these inputs---critical-level
Kac--Moody $V_{-h^\vee}(\mathfrak{g})$, $E_1$-chiral algebras
(Yangians), Calabi--Yau functor outputs---carry the generic
first-class structure $\mathrm{SC}^{\mathrm{ch,top}}$, in which
the holomorphic and topological colours remain distinct.

The Heisenberg algebra $\mathcal{H}_k$ ($k \ne 0$) and lattice
VOAs are \emph{not} stuck at $\mathrm{SC}^{\mathrm{ch,top}}$:
$\mathcal{H}_k$ carries the abelian Sugawara conformal vector
$T = \frac{1}{2k}{:}JJ{:}$ ($c = 1$) and abelian holomorphic
Chern--Simons theory provides the 3d HT bulk, so
$Z^{\mathrm{der}}_{\mathrm{ch}}(\mathcal{H}_k)$ reaches
$E_3$-topological (the same proof as for non-abelian
Kac--Moody, Theorem~\ref{thm:topologise}).
\end{remark}

\section{The modular operad: constructed; curved-Dunn: closed}
\label{sec:frontier}
%% ====================================================================

The formerly conjectural chiral-$\Ainf$ modular operad of the
$E_2 \to E_3$ step has been constructed, and curved Dunn additivity
has been closed at all genera. We record the statements.

\begin{theorem}[Chiral-$\Ainf$ modular operad;
\ClaimStatusProvedElsewhere (Vol~II
\texttt{ch:modular-swiss-cheese-operad})]%
\label{thm:chiral-Ainf-modular-operad}%
The modular operad $\mathcal{O}^{\mathrm{ch}\text{-}\Ainf}$ exists
and its algebras consist of:
\begin{enumerate}[label=\textup{(\alph*)}]
\item an $E_1$-chiral algebra structure on $X$
  (governed by $\SCchtop$ on within-surface strata);
\item an $\Ainf$-algebra structure in the transverse direction $\bR$;
\item compatibility between transverse $\Ainf$-operations and the
  $E_1$ ordering on $X$;
\item at genus $g \ge 1$: curvature
  $d^2_{\mathrm{fib}} = \kappa_{\mathrm{ch}}(\cA) \cdot \omega_g$
  as the bar intrinsic obstruction to extending the genus-$0$
  structure to higher genera.
\end{enumerate}
The genus-$0$ part of $\mathcal{O}^{\mathrm{ch}\text{-}\Ainf}$
is $\SCchtop \times E_1^{\mathrm{tr}}$.
\end{theorem}

\begin{theorem}[Curved Dunn additivity, all genera;
\ClaimStatusProvedElsewhere (Vol~II

  \mathbf{B}_{\mathrm{mod}}(\mathcal{O}^{\mathrm{ch}\text{-}\Ainf})

  \;\otimes^{\tau}\;
  \mathbf{B}(E_1^{\mathrm{tr}}),
\]
with twisting cochain $\tau$ built recursively from $R$-matrix

\end{theorem}

\subsection*{The unified chiral quantum group}

The nine families of chiral quantum group (Yangian

four-parameter family $Q_\fg^{k, f, \mu}$
(Vol~II \texttt{ch:unified-chiral-quantum-group}), with uniform

\begin{theorem}[Chiral QG equivalence, ordered Koszul locus;
\ClaimStatusProvedElsewhere (Vol~I

\label{thm:chiral-qg-equiv-ordered-standalone}%
On the ordered Koszul locus
(Definition~\ref{def:ordered-koszul-chiral-algebra-standalone}), the
three structures --- classical $r$-matrix $r(z)$, $A_\infty$ chiral
algebra $\{m_k^{\mathrm{ord}}\}$, and ordered chiral coproduct $\Delta_z$
--- determine each other up to Drinfeld associator. Concretely, the
Yangian triangle $\alpha \circ \gamma \circ \beta = \id$ is verified
explicitly at order $\hbar$ for $\Yhbar(\fsl_2)$. The equivalence
covers $E_1$-chiral input (Yangian) as well as $E_\infty$-chiral
input (vertex algebras on the Koszul locus). The elliptic case is

\end{theorem}

\section{The equivalence triangle for $\mathcal{W}_{1+\infty}$}
\label{sec:equiv-triangle}
%% ====================================================================

The preceding sections establish the $E_1$-chiral quantum group
definition (Definition~\ref{def:e1-chiral-qg}) and note that the
coproduct $\Delta$ is not visible in the shadow tower.  For the
Yangian $Y_\hbar(\fsl_2)$, all three data (multiplication,
coproduct, $R$-matrix) are supplied by the RTT formalism. The
class~M version of the same package is
Theorem~\ref{thm:w-infty-chiral-qg-completed-standalone}: the triangle
closes in the weight-completed category, and fails at the direct-sum
level at weight~$4$ by $S_4(\mathrm{Vir}_c) = 10/[c(5c+22)]$. The
present section records the explicit triangle functors
$\alpha, \beta, \gamma^{-1}$ for $\cW_{1+\infty}$ at self-dual
level $c = 1$; all statements are understood in the weight-completed
category unless noted.

The test case is $\mathcal{W}_{1+\infty}$ at the self-dual level
$c = 1$, where the Maulik--Okounkov (MO) $R$-matrix
$R^{\mathrm{MO}}(u)$ is proved \cite{Vol1, Vol3}.  Strategy~3: use
$R^{\mathrm{MO}}$ as input and reconstruct the remaining data
($\Ainf$ operations and coproduct) via the \emph{equivalence
triangle}
\begin{equation}\label{eq:triangle}
\xymatrix@C=3em{
& \Ainf\text{-structure } \{m_k\} \ar[dr]^{\beta} \\
R\text{-matrix } R(z) \ar[ur]^{\alpha} \ar[rr]_{\gamma^{-1} = \beta \circ \alpha}
&& \text{coproduct } \Delta
}
\end{equation}
whose three functors are:
\begin{enumerate}[label=\textup{(\alph*)}]
\item $\alpha$: \emph{$R$-matrix $\to$ $\Ainf$.}  Restrict the
  spectral $R$-matrix data from
  $\mathrm{FM}_n^{\mathrm{ord}}(\bC)$ to the Stasheff
  associahedra $K_n \hookrightarrow \mathrm{FM}_n^{\mathrm{ord}}(\bC)$.
  The boundary strata of $K_n$ encode the $\Ainf$ operations $m_k$:
  $K_n$ is the Stasheff compactification of the moduli of $n$~points
  on~$\bR$ (a codimension-$0$ submanifold of $\mathrm{FM}_n^{\mathrm{ord}}(\bC)$),
  and the boundary stratification
  $\partial K_n = \bigcup_{p+q=n+1} K_p \times K_q$
  recovers the $\Ainf$ associativity relations $\sum m_p \circ m_q = 0$.
\item $\beta$: \emph{$\Ainf \to$ coproduct.}  Given the $\Ainf$
  operations $\{m_k\}$, form the ordered bar coalgebra
  $B^{\mathrm{ord}}(\cA) = T^c(s^{-1}\bar{\cA})$
  with bar differential $d_B = \sum m_k^{(i)}$.  The cobar construction
  $\Omega(B^{\mathrm{ord}}(\cA))$ produces an $\Ainf$ algebra;
  dualizing the deconcatenation coproduct on $B^{\mathrm{ord}}$
  through Koszul duality gives the twisted coproduct
  $\Delta_z \colon \cA^!_{\mathrm{line}} \to
  \cA^!_{\mathrm{line}} \,\widehat{\otimes}\,
  \cA^!_{\mathrm{line}}$
  on the open-colour Koszul dual.
\item $\gamma^{-1} = \beta \circ \alpha$:
  \emph{$R$-matrix $\to$ coproduct} (composite).
  The MO $R$-matrix $R^{\mathrm{MO}}(u)$ produces the coproduct
  directly, bypassing the intermediate $\Ainf$ structure.
\end{enumerate}

\subsection*{Step $\alpha$: extracting $\Ainf$ from the MO $R$-matrix}

The structure function of $\mathcal{W}_{1+\infty} \simeq
Y(\widehat{\fgl}_1)$ is
\begin{equation}\label{eq:structure-fn}
g(u) = \frac{(u - h_1)(u - h_2)(u - h_3)}{(u + h_1)(u + h_2)(u + h_3)},
\qquad h_1 + h_2 + h_3 = 0.
\end{equation}
The MO $R$-matrix on the charge-$n$ Fock space $\mathcal{F}_n$ has
diagonal eigenvalues
$R_{\lambda\lambda}(u) = \prod_{s \in \lambda} g(u + c(s))$
where $c(s) = h_1 i + h_2 j$ is the content of box $s = (i,j)$
in partition $\lambda$ \cite{Vol3}.

The functor $\alpha$ restricts the $\mathrm{FM}_n(\bC)$
configuration-space data to associahedra:

\begin{construction}[$\alpha$: $R$-matrix to $\Ainf$]
\label{constr:alpha}
At degree $n$, the $\Ainf$ operation $m_n^{\mathrm{ch}}$ is
extracted from the $R$-matrix as
\begin{equation}\label{eq:alpha-mk}
m_n^{\mathrm{ch}}(a_1, \ldots, a_n)
\;=\;
\int_{K_n}
\omega^R_n(z_1, \ldots, z_n; a_1, \ldots, a_n),
\end{equation}
where $\omega^R_n$ is the propagator form on $K_n$ twisted by the
$R$-matrix:
\[
\omega^R_n
\;=\;
\bigwedge_{i < j} \dlog(z_i - z_j)
\;\cdot\;
R_{i,i+1}(z_i - z_{i+1})
\;\cdot\;
a_1(z_1) \otimes \cdots \otimes a_n(z_n).
\]
For $n = 2$: $K_2$ is a point, and $m_2$ is the binary product
$m_2(a,b) = \operatorname{Res}_{z=0}\, a(z)\, b(0)\, \dlog(z)$,
the standard OPE extraction.

For $n = 3$: $K_3 \simeq [0,1]$ is an interval.  The integral
over $K_3$ extracts the first higher product $m_3^{\mathrm{ch}}$
from the triple collision with the $R$-matrix twist:
\begin{equation}\label{eq:m3-from-R}
m_3^{\mathrm{ch}}(a_1, a_2, a_3)
\;=\;
\int_0^1
\Bigl[\operatorname{Res}_{z_{12}=0}\,
R_{12}(z_{12})\, a_1(z_1)\, a_2(z_2)\Bigr]
\;\cdot\; a_3(z_3)\;
\dlog(z_{23})\, dt,
\end{equation}
where the integration variable $t$ parametrises $K_3$ and
$z_{12}, z_{23}$ are the successive differences.  For class~L
(Kac--Moody, $R(z) = 1 + \hbar\, P/z$), the integral evaluates
to zero: $m_3 = 0$, consistent with Koszulness.

For $\mathcal{W}_{1+\infty}$ (class~M): $m_3^{\mathrm{ch}} \neq 0$.
The structure function~\eqref{eq:structure-fn} contributes terms
at every order in $1/z$, producing a nonvanishing $K_3$ integral.
In the spin-$2$ channel (Virasoro at $c = 1$ inside $\mathcal{W}_{1+\infty}$),
the $m_3$ operation encodes the first higher correction to the Virasoro
OPE, consistent with $d_{\mathrm{gen}}(\mathrm{Vir}) = 3$
(generating depth~$3$: $m_3$ generates all higher $m_k$ by
recursive composition).
\end{construction}

\subsection*{Step $\beta$: cobar dualization to the coproduct}

\begin{construction}[$\beta$: $\Ainf$ to coproduct]
\label{constr:beta}
Given the $\Ainf$ operations $\{m_k^{\mathrm{ch}}\}$ from
Construction~\ref{constr:alpha}, form the ordered bar complex
\[
B^{\mathrm{ord}}(\mathcal{W}_{1+\infty})
\;=\;
T^c(s^{-1}\overline{\mathcal{W}_{1+\infty}})
\]
with bar differential $d_B = \sum_{k \ge 2} m_k^{(i)}$ and
deconcatenation coproduct.  The open-colour Koszul dual is
\[
\cA^!_{\mathrm{line}}
\;=\;
H^*\bigl(B^{\mathrm{ord}}(\mathcal{W}_{1+\infty})\bigr)^\vee,
\]
which is $Y(\widehat{\fgl}_1)$ (the affine Yangian of $\fgl_1$).
The twisted coproduct on $\cA^!_{\mathrm{line}}$ is obtained by
dualizing the deconcatenation:
\begin{equation}\label{eq:twisted-coprod}
\Delta_z(f)(\bar a_1 \otimes \cdots \otimes \bar a_n)
\;=\;
\sum_{p=0}^{n}
f(\bar a_1, \ldots, \bar a_p)
\;\otimes_z\;
f(\bar a_{p+1}, \ldots, \bar a_n),
\end{equation}
where $\otimes_z$ denotes the completed tensor product with
spectral parameter~$z$ tracking the position of the cut.  The
twisting by $z$ arises from the holomorphic coordinate:
cutting the ordered sequence at position $p$ separates the
configuration into two clusters whose relative position is~$z$.
\end{construction}

\subsection*{The composite $\gamma^{-1}$: direct $R$-matrix to
coproduct}

\begin{proposition}[Cohomological coproduct from MO $R$-matrix]
\label{prop:gamma-inverse}
At the cohomological level, the composite $\gamma^{-1} = \beta \circ \alpha$
applied to the MO $R$-matrix $R^{\mathrm{MO}}(u)$
of~\eqref{eq:structure-fn} produces a well-defined coproduct
\begin{equation}\label{eq:gamma-coprod}
\Delta_z^{\mathrm{MO}} \colon
Y(\widehat{\fgl}_1)
\;\longrightarrow\;
Y(\widehat{\fgl}_1) \,\widehat{\otimes}\,
Y(\widehat{\fgl}_1)
\end{equation}
satisfying:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Coassociativity at cohomology:}
  $(\Delta_z \widehat{\otimes}\, \id) \circ \Delta_{z+w}
  = (\id \widehat{\otimes}\, \Delta_w) \circ \Delta_z$.
\item \emph{Quasi-triangularity at cohomology:}
  $R^{\mathrm{MO}}(z)\, \Delta_z(a)
  = \Delta_z^{\mathrm{op}}(a)\, R^{\mathrm{MO}}(z)$ for all
  $a \in Y(\widehat{\fgl}_1)$.
\item \emph{Low-spin explicit form:} on the Cartan generators
  $\psi_k$ (spin~$1$):
  \[
  \Delta_z(\psi(u))
  \;=\;
  \psi(u) \otimes \psi(u - z),
  \]
  where $\psi(u) = 1 + \sum_{k \ge 0} \psi_k\, u^{-k-1}$; on the
  positive root generators $e_k$ (spin~$2$):
  \[
  \Delta_z(e(u))
  \;=\;
  e(u) \otimes 1
  \;+\;
  \psi(u) \otimes e(u - z);
  \]
  and on the negative root generators $f_k$:
  \[
  \Delta_z(f(u))
  \;=\;
  1 \otimes f(u)
  \;+\;
  f(u - z) \otimes \psi(u).
  \]
  These match the Schiffmann--Vasserot spectral coproduct at $z = 0$
  and the Drinfeld new coproduct $\Delta_{\mathrm{new}}$ at
  leading spectral order.
\end{enumerate}
\end{proposition}

\begin{proof}[Proof sketch]
The well-definedness at cohomology follows from the general principle:
the bar-cobar adjunction $\Omega \circ B \simeq \id$ holds at the
level of the \emph{coderived category} for all shadow classes,
including class~M \cite[thm:bv-bar-coderived]{Vol1}.  The
obstruction to a chain-level quasi-isomorphism (which exists for
classes G, L, C but fails for class~M) does not affect the
cohomological coproduct, because:
\begin{enumerate}[label=\textup{(\alph*)}]
\item The MO $R$-matrix is a cohomological object: it acts on
  $K_T(\mathrm{Hilb}^n) = H^*(B^{\mathrm{ord}})^\vee$, which is the
  cohomology of the bar complex, not the chain-level complex itself.
\item The functor $\alpha$ (restriction to associahedra) commutes with
  passage to cohomology: the $K_n \hookrightarrow
  \mathrm{FM}_n^{\mathrm{ord}}(\bC)$ embedding respects the
  stratification that defines the bar differential.
\item The functor $\beta$ (Koszul dualization of deconcatenation)
  on cohomology is the classical Koszul duality, which is well-defined
  for any algebra (the coderived correction is invisible at cohomology).
\end{enumerate}

Coassociativity~(i) follows from the tautological coassociativity of
deconcatenation on $T^c$, which survives dualization.
Quasi-triangularity~(ii) follows from the YBE for $R^{\mathrm{MO}}$
(proved in \cite{Vol3}).

The explicit formulas~(iii) are computed as follows.  The Cartan
generator $\psi(u)$ is the diagonal part of the $R$-matrix
action on the charge-$1$ Fock space.  On
$\mathcal{F}_1 = \bigoplus_\lambda \bC \cdot |\lambda\rangle$
(partitions of charge~$1$, i.e.\ a single box), the $R$-matrix
eigenvalue is $g(u) = (u-h_1)(u-h_2)(u-h_3)/((u+h_1)(u+h_2)(u+h_3))$.
The Gauss decomposition $R = F^{-1} \cdot D \cdot E$ (lower
triangular, diagonal, upper triangular) extracts the
Drinfeld-type generators.  The coproduct is read off from the
matrix coproduct $\Delta(T(u)) = T(u) \,\dot\otimes\, T(u-z)$
applied to the Gauss factors.  This is the Schiffmann--Vasserot
coproduct for $Y(\widehat{\fgl}_1)$, obtained here from the
MO $R$-matrix via $\gamma^{-1}$.
\end{proof}

\begin{remark}[The class~M obstruction]
\label{rem:class-M-obstruction}
For class~L algebras (Kac--Moody), $\gamma^{-1}$ produces a coproduct
at chains: the bar-cobar adjunction is a quasi-isomorphism
($\Omega(B(A)) \xrightarrow{\sim} A$), so all three vertices of the
triangle~\eqref{eq:triangle} are equivalent at the chain level.

For class~M algebras ($\mathcal{W}_{1+\infty}$, Virasoro), the
chain-level direct-sum bar-cobar map is \emph{not} a
quasi-isomorphism: the nonvanishing shadow coefficient
$S_4(\mathrm{Vir}_c) = 10/[c(5c+22)]$ forces a bar cohomology class
at weight $4$. The triangle~\eqref{eq:triangle} holds:
\begin{enumerate}[label=\textup{(\alph*)}]
\item \emph{Cohomology:} the coproduct~\eqref{eq:gamma-coprod} is
  well-defined on $H^*(B^{\mathrm{ord}})^\vee = Y(\widehat{\fgl}_1)$.
\item \emph{Coderived category:} the bar-cobar adjunction
  $\Omega \circ B \simeq \id$ holds in the coderived sense
  (Vol~I \texttt{thm:bv-bar-coderived}), producing a coproduct on the
  coderived model of $\mathcal{W}_{1+\infty}$.
\item \emph{Weight-completed chain level (proved):} the MC5 pattern
  of Vol~I \texttt{thm:completed-bar-cobar-strong} +
  \texttt{prop:standard-strong-filtration} lifts $\gamma^{-1}$ to a
  chain-level coproduct on the weight-completed $\Ainf$-model
  $\widehat{\cW}_{1+\infty}$ (Vol~I
  \texttt{thm:w-infty-chiral-qg-completed}).
\item \emph{Direct-sum chain level (genuinely false at generic~$c$).}
  The weight-$4$ shadow obstruction rules out a direct-sum chain-level
  lift; the correct scope is the weight-completed category.
\end{enumerate}
\end{remark}

\begin{theorem}[Chain-level coproduct for class~M, weight-completed;
\ClaimStatusProvedElsewhere (Vol~I
\texttt{thm:w-infty-chiral-qg-completed})]%
\label{thm:chain-level-coprod-weight-completed}%
For $\mathcal{W}_{1+\infty}$ at self-dual level $c = 1$, the
cohomological coproduct $\Delta_z^{\mathrm{MO}}$ of
Proposition~\ref{prop:gamma-inverse} lifts to a chain-level
$\Ainf$-coproduct on the weight-completed $\Ainf$-model
$\widehat{\cW}_{1+\infty}^{\Ainf}$. Making
$(\widehat{\cW}_{1+\infty}^{\Ainf},\, \Delta_z^{\Ainf},\,
R^{\mathrm{MO}}(z))$ an $\Ainf$-bialgebra in the sense of
Markl--Shnider. The Jindal--Kaubrys--Latyntsev vertex coproduct on
the CoHA of $\bC^3$ \cite{JKL} provides a chain-level construction
for the positive half $Y^+(\widehat{\fgl}_1) \simeq
\mathrm{CoHA}(\bC^3)$; the Drinfeld double extension to the full
affine Yangian is absorbed by the weight-completed pattern.
\end{theorem}
