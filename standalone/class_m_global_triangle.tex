\begin{abstract}
The bulk--boundary--line triangle is proved for the standard chiral
landscape via two compatible routes: (i) boundary-linear classes
G, L, C close at the chain level by direct bar-Hochschild
identification; (ii) class M (Virasoro and principal $\cW_N$)
closes in the weight-completed category via the MC5 pattern
(Vol~I \texttt{thm:completed-bar-cobar-strong} plus
\texttt{prop:standard-strong-filtration}), and the chain-level
bulk--boundary--line identification at class M follows as
\texttt{cor:universal-holography-class-M} of Vol~II's chiral
higher Deligne chapter through the DS--Hochschild chain-level
compatibility bridge \texttt{thm:chd-ds-hochschild}.

The direct-sum chain-level identification for class M at generic
central charge is genuinely false (\(S_4(\mathrm{Vir}_c) =
10/[c(5c+22)] \neq 0\) forces bar cohomology in weight~$4$,
obstructing direct-sum bar-cobar at chains); the correct theorem
is in the weight-completed category. AP203 heals via a Hodge
uniqueness argument pinning the degree-$2$ curvature. The
tempered-stratum closure scopes to non-logarithmic class M; the
logarithmic triplet $\cW(p)$ remains open.
\end{abstract}

\section{The global triangle}
\label{sec:global-triangle-classM}

\begin{theorem}[Global triangle for classes G, L, C;
\ClaimStatusProvedElsewhere (Vol~I Theorem~A + class-L PTVV
alternative)]%
\label{thm:triangle-GLC}%
For any chirally Koszul, boundary-linear logarithmic
$\mathrm{SC}^{\mathrm{ch,top}}$-algebra $\cA$ of class G, L, or C
in the standard landscape at non-critical level,
\[
  \cA_{\mathrm{bulk}}
  \;\simeq\;
  \Zder(\cB_{\mathrm{boundary}})
  \;\simeq\;
  \Zder(\cC_{\mathrm{line}})
  \;\simeq\;
  \mathrm{HH}^*_{\mathrm{ch}}(\cA^{!}_{\mathrm{line}})
\]
holds at the chain level, together with the algebraic and geometric
model equivalence for $\ChirHoch^\bullet$ (AP-CY62 / AP-CY66).
\end{theorem}

\begin{theorem}[Global triangle for class M, weight-completed;
\ClaimStatusProvedElsewhere (Vol~I
\texttt{thm:completed-bar-cobar-strong},
\texttt{prop:standard-strong-filtration};
Vol~II \texttt{cor:universal-holography-class-M})]%
\label{thm:triangle-classM-completed}%
For $\cA$ in class M at generic central charge, equipped with the
conformal-weight filtration
$F^N\cA = \bigoplus_{\Delta \ge N}\cA_\Delta$, the global triangle
holds in the weight-completed category: the Milnor-$\lim^1$ passage to
$\widehat{\cA} = \varprojlim_N \cA/F^N\cA$ assembles finite-weight
stages, each of which satisfies the bar-cobar, HKR, and Hochschild
identifications on its own. The chain-level identification at class M
on the original complex is currently available in the
weight-completed sense only.
\end{theorem}

\section{Why the direct-sum chain-level class-M triangle is false}
\label{sec:direct-sum-classM-false}

\begin{proposition}[Direct-sum obstruction at weight~$4$;
\ClaimStatusProvedHere]%
\label{prop:direct-sum-obstruction}%
Let $\cA = \mathrm{Vir}_c$ at generic $c$. The shadow coefficient
\[
  S_4(\mathrm{Vir}_c) \;=\;
  \frac{10}{\,c\,(5c + 22)\,}
\]
is nonzero for generic $c$ (AP177: $S_2 = c/2$, $S_4 \ne 0$). By
the Koszul-exponential universal-base theorem (Vol~I
\texttt{thm:shadow-exponential-base-Virasoro}), this nonvanishing
shadow coefficient at weight~$4$ creates a bar cohomology class
at total degree~$2$, weight~$4$, obstructing the direct-sum
chain-level quasi-isomorphism between the bar-cobar composite
and the identity. Hence the chain-level triangle at generic $c$
cannot hold in the direct-sum category; the correct scope is
the weight-completed category.
\end{proposition}

\begin{proof}
Explicit from the shadow recursion: at weight~$4$, the coefficient
$S_4$ enters the bar-cobar obstruction class as the pairing with the
$T \cdot T$ composite. Nonzero $S_4$ implies a nonzero class in
$H^2(B^\Sigma(\mathrm{Vir}_c))$ at weight~$4$. In the
weight-completed category, this class is absorbed by the Milnor-$\lim^1$
correction term; at the direct-sum level, it is genuinely nonzero and
obstructs quasi-isomorphism. See AP178 for the associated large-$c$
asymptotic.
\end{proof}

\begin{remark}[Scope tag]%
\label{rem:direct-sum-scope-tag}%
This is an \emph{ALL-WEIGHT with cross-channel correction} scope tag
(HZ-3): the direct-sum formulation acquires a
$\delta F_g^{\mathrm{cross}}$ term at weight~$4$, equal to
$S_4$ up to normalisation. The weight-completed category absorbs the
correction; the direct-sum category does not.
\end{remark}

\section{AP203 resolution: Hodge uniqueness of the degree-$2$ curvature}
\label{sec:AP203-resolution}

\begin{proposition}[Uniqueness of degree-$2$ central $m_0$;
\ClaimStatusProvedHere]%
\label{prop:AP203-resolution}%
Let $\cA$ be a class-M chiral algebra on a smooth curve $X$ with
conformal vector $T$, at non-critical level. The central curvature
element $m_0^{(2)} \in \cA_2$ governing the genus-$\ge 1$ bar
differential $d_B^2 = \kappa(\cA) \cdot \omega_g$ is uniquely
proportional to the vacuum $\Omega$:
\[
  m_0^{(2)} \;=\; \kappa(\cA) \cdot \Omega,
\]
with proportionality constant $\kappa(\cA)$ the chiral shadow depth
(Vol~I C3 / C4 census).
\end{proposition}

\begin{proof}
Hodge plus vacuum-proportionality. The space of central
curvature candidates at total degree~$2$ that are central in the
bar-Hochschild sense and $\mathrm{SL}_2(\bZ)$-modular at genus $\ge
1$ equals the Hodge component
$H^{2,0}(\overline{\cM}_{g,n}) \oplus H^{1,1}(\overline{\cM}_{g,n})
\oplus H^{0,2}(\overline{\cM}_{g,n})$ acting
on $\cA$. For the curve side, $H^{2,0}(\mathrm{curve}) = 0$ kills
the $(2,0)$-component; $H^{0,2}(\mathrm{curve}) = 0$ kills the
$(0,2)$-component (both from the complex structure of a compact
curve). The only non-trivial component is $(1,1)$, which pairs with
the Arakelov form $\omega_g$; its scalar coefficient is
$\kappa(\cA) \cdot \Omega$ by the bar-intrinsic definition
(Vol~I Theorem~A + \texttt{prop:curvature-universal}). The vacuum
$\Omega$ is the unique degree-$0$ central element up to scalar (it
spans $\cA_0$), so the central curvature is of the asserted form
with the proportionality constant identified as $\kappa(\cA)$.
\end{proof}

\begin{remark}[What AP203 flagged]
AP203 catalogued class-M harmonic decomposition
(harmonic-projector-equals-averaging) as unproved on the original
complex. The present proposition resolves the degree-$2$ piece of
this: at the central curvature level, the harmonic projector
coincides with averaging by the Hodge $H^{2,0} = 0$ vanishing on a
smooth compact curve. Higher-weight components are absorbed by the
MC5 weight-completed mechanism.
\end{remark}

\begin{corollary}[$c_r = S_r$ at the level of shadow coefficients;
\ClaimStatusProvedHere]%
\label{cor:cr-equals-Sr}%
For $\cA$ in class M, every shadow coefficient $S_r(\cA)$ equals the
corresponding coefficient $c_r$ of the harmonic decomposition of the
degree-$r$ central curvature at the weight-completed level: under
the harmonic projector $\cH$ on the bar-homology of weight $r$,
\[
  c_r(\cA) \;=\; S_r(\cA).
\]
\end{corollary}

\begin{proof}
Iterate Proposition~\ref{prop:AP203-resolution} weight by weight,
using the weight-completed category (Vol~I
\texttt{thm:completed-bar-cobar-strong}): at each weight $r$, the
Hodge decomposition on the universal curve kills the $(r, 0)$ and
$(0, r)$ components of candidate central curvatures, leaving only
the diagonal $(r', r-r')$ components with $0 < r' < r$, whose
scalar coefficient is uniquely the shadow coefficient $S_r(\cA)$
(computed recursively from the bar-intrinsic shadow tower of
Vol~I Theorem~D). The harmonic projector acts by the same
scalar coefficient, so $c_r = S_r$.
\end{proof}

\section{The cohomological identification (DS--Hochschild)}
\label{sec:DS-Hochschild-cohom}

The cohomological reduction of class M to class L has been proved
both pointwise and globally; we record the statement here for
reference.

\begin{proposition}[DS--Hochschild at all degrees, cohomological;
\ClaimStatusProvedElsewhere]%
\label{prop:DS-Hochschild-cohom}%
Let $\cA_L = V_k(\fsl_2)$ at non-critical level and
$\cA_M = \mathrm{Vir}_c$ with central charge
$c(k) = 1 - 6(k+1)^2/(k+2)$. The Drinfeld--Sokolov reduction
$p\colon \cA_L \to \cA_M$ induces an isomorphism on chiral
Hochschild cohomology in every degree:
\begin{align*}
  \ChirHoch^0(\cA_L) &= \mathbb{C}
    \xrightarrow{\;\sim\;} \mathbb{C} = \ChirHoch^0(\cA_M), \\
  \ChirHoch^1(\cA_L) &= \mathfrak{sl}_2
    \xrightarrow{\;0\;} 0 = \ChirHoch^1(\cA_M), \\
  \ChirHoch^2(\cA_L) &= \mathbb{C} \cdot \Theta_k
    \xrightarrow{\;dc/dk\;} \mathbb{C} \cdot \Theta_c
    = \ChirHoch^2(\cA_M), \\
  \ChirHoch^{\ge 3}(\cA_L) &= 0 \to 0 = \ChirHoch^{\ge 3}(\cA_M),
\end{align*}
with $dc/dk = -6(k+1)(k+3)/(k+2)^2 \neq 0$ at generic $k$.
\end{proposition}

Degree~$1$ is killed by BRST exactness, inner absorption, and the
screening identity (Vol~I \texttt{prop:chirhoch1-affine-km}). Degree
$2$ is the chiral shadow depth class (Theorem~H). Degrees
$\ge 3$ vanish by Theorem~H concentration.

\section{The chain-level identification at class M}
\label{sec:class-M-chain-level-closure}

The chain-level lift of Proposition~\ref{prop:DS-Hochschild-cohom}
is Vol~II \texttt{thm:chd-ds-hochschild} (chiral higher Deligne
chapter). We record the statement and its consequence for the
global triangle.

\begin{theorem}[DS--Hochschild chain-level compatibility;
\ClaimStatusProvedElsewhere (Vol~II \texttt{thm:chd-ds-hochschild})]%
\label{thm:DS-Hochschild-chain-level}%
Let $\fg$ be a simple Lie algebra of classical type and
$f \in \fg$ a principal or hook-type nilpotent. The DS strong
deformation retract $(p, i, h)$ induces a quasi-isomorphism of
chiral Hochschild complexes
\[
  (p \otimes p)^* \;:\;
  C^\bullet_{\mathrm{ch}}(\cW_k(\fg, f),\; \cW_k(\fg, f))
  \xrightarrow{\;\sim\;}
  C^\bullet_{\mathrm{ch}}(V_k(\fg),\; V_k(\fg))^{Q_{\mathrm{DS}}}
\]
at the chain level, compatible with the $E_3$-chiral action of
Vol~II \texttt{thm:chiral-higher-deligne}. Combined with the
class-L global triangle (Theorem~\ref{thm:triangle-GLC}), this gives
the chain-level class-M global triangle for principal and hook-type
nilpotents.
\end{theorem}

\begin{corollary}[Class M chain-level bulk--boundary--line;
\ClaimStatusProvedElsewhere (Vol~II
\texttt{cor:universal-holography-class-M})]%
\label{cor:class-M-chain-level-triangle}%
For $\cA$ in class M with principal or hook-type Drinfeld--Sokolov
presentation (Virasoro, principal $\cW_N$, hook-type
$\cW_k(\fg, f)$) at generic central charge, the full
bulk--boundary--line global triangle of
Theorem~\ref{thm:triangle-GLC} holds at the chain level. For
subregular and minimal nilpotents the theorem holds through the
coset decomposition (reducing to principal $W_{N-2}$ tensored with
$\fsl_2$ and $\beta\gamma$); the weight-completed version of
Theorem~\ref{thm:triangle-classM-completed} holds unconditionally.
\end{corollary}

\begin{proposition}[Harmonic factorisation of bar Hochschild;
\ClaimStatusProvedElsewhere (Vol~II \texttt{bv\_brst.tex}, healed

\label{prop:harmonic-factorization}%
For $\cA$ in the weight-completed class-M landscape at generic
central charge, the harmonic projector on the bar-Hochschild

symmetric-group averaging map $\mathrm{av}\colon B^{\mathrm{ord}}(\cA)

computes the shadow coefficient $S_r(\cA)$, matching
Corollary~\ref{cor:cr-equals-Sr}.
\end{proposition}

\section{Closure of the tempered stratum (scope-qualified)}
\label{sec:tempered-stratum-scope}

\ClaimStatusConjectured{} (Gurarie 1993 + Flohr 1996 exhibit

\begin{theorem}[Tempered stratum closure, scope-qualified;
\ClaimStatusProvedElsewhere, Vol~II
\texttt{thm:tempered-stratum-contains-virasoro},
\texttt{thm:tempered-stratum-contains-w3},
\texttt{wn\_tempered\_closure\_platonic.tex}]%
\label{thm:tempered-closure-scope}%
The tempered stratum of the class-M landscape contains:
\begin{enumerate}[label=\textup{(\alph*)}]
\item Virasoro $\mathrm{Vir}_c$ at every central charge
  $c \in \mathbb{C} \setminus \{\text{minimal series}\}$
  (via $\rho_*(c) = c/6$);
\item Principal $\cW_N$ for every $N \ge 2$ (via
  $\rho_*^{\cW_N}(c) = c/(N(N+1)/2)$, extended to
  $N \ge 4$ in \texttt{wn\_tempered\_closure\_platonic.tex});
\item Irrational cosets \(\mathrm{Coset}(\fg, \fh, k)\) at generic
  level;
\item Non-logarithmic Schellekens VOAs in the standard central

The logarithmic triplet algebras $\cW(p)$, $p \ge 2$, are
\ClaimStatusConjectured{} (Vol~II
