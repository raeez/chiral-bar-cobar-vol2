\begin{abstract}
The bulk--boundary--line triangle is proved for the standard chiral
landscape via two compatible routes: (i) boundary-linear classes
G, L, C hold at the chain level by direct bar-Hochschild
identification; (ii) class M (Virasoro and principal $\cW_N$)
holds in the weight-completed category via the MC5 weight-completed
mechanism
(Vol~I \ref*{V1-thm:completed-bar-cobar-strong} plus
\ref*{V1-prop:standard-strong-filtration}), and the chain-level
bulk--boundary--line identification at class M is available in the
principal and hook-type Drinfeld--Sokolov range through the
DS--Hochschild compatibility theorem \ref*{thm:chd-ds-hochschild}.

The direct-sum chain-level identification for class M at generic
central charge is genuinely false (\(S_4(\mathrm{Vir}_c) =
10/[c(5c+22)] \neq 0\) forces bar cohomology in weight~$4$,
obstructing direct-sum bar-cobar at chains); the correct theorem
is in the weight-completed category. The surviving direct-sum
curvature statement is a scalar central-channel statement in the
completed curved bar model; it lives outside the weight-$2$ state
space. The tempered-stratum theorem is restricted to
non-logarithmic class M; the
logarithmic triplet $\cW(p)$ remains open.
\end{abstract}

\section{The global triangle}
\label{sec:global-triangle-classM}

\begin{theorem}[Global triangle for classes G, L, C]%
\label{thm:triangle-GLC}%
For any chirally Koszul, boundary-linear logarithmic
$\mathrm{SC}^{\mathrm{ch,top}}$-algebra $\cA$ of class G, L, or C
in the standard landscape at non-critical level,
\[
  \cA_{\mathrm{bulk}}
  \;\simeq\;
  \Zder(\cB_{\mathrm{boundary}})
  \;\simeq\;
  \Zder(\cC_{\mathrm{line}})
  \;\simeq\;
  \mathrm{HH}^*_{\mathrm{ch}}(\cA^{!}_{\mathrm{line}})
\]
holds at the chain level, together with the algebraic and geometric
model equivalence for $\ChirHoch^\bullet$ (geometric vs algebraic chiral-Hoch model compatibility).
\end{theorem}

\begin{theorem}[Global triangle for class M, weight-completed]%
\label{thm:triangle-classM-completed}%
For $\cA$ in class M at generic central charge, equipped with the
conformal-weight filtration
$F^N\cA = \bigoplus_{\Delta \ge N}\cA_\Delta$, the global triangle
holds in the weight-completed category: the Milnor-$\lim^1$ passage to
$\widehat{\cA} = \varprojlim_N \cA/F^N\cA$ assembles finite-weight
stages, each of which satisfies the bar-cobar, HKR, and Hochschild
identifications on its own. The chain-level identification at class M
on the original complex requires the DS--Hochschild hypotheses stated
below.
\end{theorem}

\section{Why the direct-sum chain-level class-M triangle is false}
\label{sec:direct-sum-classM-false}

\begin{proposition}[Direct-sum obstruction at weight~$4$]%
\label{prop:direct-sum-obstruction}%
Let $\cA = \mathrm{Vir}_c$ at generic $c$. The shadow coefficient
\[
  S_4(\mathrm{Vir}_c) \;=\;
  \frac{10}{\,c\,(5c + 22)\,}
\]
is nonzero for generic $c$ ($S_2 = c/2$, $S_4 \ne 0$). By
the Koszul-exponential universal-base theorem (Vol~I
\ref*{V1-thm:shadow-exponential-base-Virasoro}), this nonvanishing
shadow coefficient at weight~$4$ creates a bar cohomology class
at total degree~$2$, weight~$4$, obstructing the direct-sum
chain-level quasi-isomorphism between the bar-cobar composite
and the identity. Hence the chain-level triangle at generic $c$
fails in the direct-sum category. The correct scope is the
weight-completed category.
\end{proposition}

\begin{proof}
Explicit from the shadow recursion: at weight~$4$, the coefficient
$S_4$ enters the bar-cobar obstruction class as the pairing with the
$T \cdot T$ composite. Nonzero $S_4$ implies a nonzero class in
$H^2(B^\Sigma(\mathrm{Vir}_c))$ at weight~$4$. In the
weight-completed category, the inverse system absorbs the obstruction
through the completed finite-weight stages; at the direct-sum level,
the class remains nonzero and obstructs quasi-isomorphism.
\end{proof}

\begin{remark}[Weight-completed scope]%
\label{rem:direct-sum-scope-tag}%
The obstruction belongs to the direct-sum ambient category.
Completion replaces the isolated
weight-$4$ obstruction by a compatible inverse system of finite-weight
bar-cobar identifications. The direct-sum category has no such
inverse-limit correction term.
\end{remark}

\section{The scalar curvature channel in the completed model}
\label{sec:direct-sum-obstruction-resolution}

\begin{proposition}[Scalar central curvature channel]%
\label{prop:direct-sum-obstruction-resolution}%
Let $\cA$ be a class-M chiral algebra on a smooth curve $X$ with
conformal vector $T$, at non-critical level, in the completed curved
bar model. After projection to the scalar central channel, the
genus-$g$ curvature class is
\[
  [d_{\mathrm{fib}}^{\,2}]_{\mathrm{scal}}
  \;=\;
  \kappa_{\mathrm{ch}}(\cA)\,\omega_g \otimes \mathbf{1}_{\cA}
  \quad\in\quad
  H^2(\overline{\cM}_{g,n})\otimes Z(\cA)_0 ,
\]
with $\omega_g$ the chosen Arakelov representative and
$\kappa_{\mathrm{ch}}(\cA)$ the chiral shadow coefficient of Vol~I.
The statement is a cohomological scalar-channel assertion; it does
not identify a weight-$2$ state of $\cA$ with the vacuum.
\end{proposition}

\begin{proof}
In the completed curved bar construction the genus-$g$ square of the
fibre differential is a degree-two class on the moduli factor with
values in the central channel of the bar-Hochschild complex. Project
to conformal weight zero. The centre in weight zero is
$\mathbb{C}\mathbf{1}_{\cA}$, so this projection is a scalar multiple
of $\mathbf{1}_{\cA}$. The Arakelov normalisation fixes the degree-two
moduli representative as $\omega_g$. Vol~I identifies the scalar
coefficient of this central channel with the chiral shadow coefficient
$\kappa_{\mathrm{ch}}(\cA)$. The argument uses the completed curved
bar model and the scalar projection; it makes no claim about all
higher-weight curvature components in the direct-sum complex.
\end{proof}

\begin{remark}[Direct-sum obstruction remains]
The scalar-channel computation does not remove the weight-$4$
direct-sum obstruction of
Proposition~\ref{prop:direct-sum-obstruction}. It isolates the
cohomological curvature coefficient that survives after completion.
The full original-complex harmonic decomposition remains a separate
chain-level problem.
\end{remark}

\begin{corollary}[$c_r = S_r$ at the level of shadow coefficients]%
\label{cor:cr-equals-Sr}%
For $\cA$ in class M, assume the harmonic factorisation hypothesis for
the completed bar-Hochschild complex and the scalar-channel
identification of
Proposition~\ref{prop:direct-sum-obstruction-resolution} at weight
$r$. Then the shadow coefficient $S_r(\cA)$ equals the corresponding
harmonic coefficient $c_r(\cA)$:
\[
  c_r(\cA) \;=\; S_r(\cA).
\]
\end{corollary}

\begin{proof}
Under the harmonic factorisation hypothesis, the harmonic projector
selects the scalar central channel at weight $r$. The shadow tower of
Vol~I computes the same scalar through the bar-intrinsic obstruction
recursion. Hence the two coefficients agree. The harmonic
factorisation input is the proof obligation; Hodge theory alone does
not supply it.
\end{proof}

\section{The cohomological identification (DS--Hochschild)}
\label{sec:DS-Hochschild-cohom}

The cohomological reduction of Virasoro class M to class L is a
relative DS comparison.  The unreduced affine Hochschild complex has
extra gauge directions.

\begin{proposition}[DS--Hochschild relative comparison, cohomological]%
\label{prop:DS-Hochschild-cohom}%
Let $\cA_L = V_k(\fsl_2)$ at non-critical level and
$\cA_M = \mathrm{Vir}_c$ with central charge
$c(k) = 1 - 6(k+1)^2/(k+2)$. The Drinfeld--Sokolov reduction
$p\colon \cA_L \to \cA_M$ identifies the reduced DS-relative chiral
Hochschild cohomology with the Virasoro chiral Hochschild cohomology.
On unreduced cohomology the visible degrees are:
\begin{align*}
  \ChirHoch^0(\cA_L) &= \mathbb{C}
    \xrightarrow{\;\sim\;} \mathbb{C} = \ChirHoch^0(\cA_M), \\
  \ChirHoch^1(\cA_L) &= \mathfrak{sl}_2
    \xrightarrow{\;0\;} 0 = \ChirHoch^1(\cA_M), \\
  \ChirHoch^2(\cA_L) &= \mathbb{C} \cdot \Theta_k
    \xrightarrow{\;dc/dk\;} \mathbb{C} \cdot \Theta_c
    = \ChirHoch^2(\cA_M), \\
  \ChirHoch^{\ge 3}(\cA_L) &= 0 \to 0 = \ChirHoch^{\ge 3}(\cA_M),
\end{align*}
with $dc/dk = -6(k+1)(k+3)/(k+2)^2 \neq 0$ at generic $k$.
\end{proposition}

Degree~$1$ is killed in the DS-relative complex by BRST exactness,
inner absorption, and the screening identity (Vol~I
\ref*{V1-prop:chirhoch1-affine-km}).  Thus the arrow
$\mathfrak{sl}_2\to0$ is a quotient by gauge directions. Degree~$2$ is the chiral
shadow depth class (Theorem~H). Degrees $\ge 3$ vanish by Theorem~H
concentration.

\section{The chain-level identification at class M}
\label{sec:class-M-chain-level-scope}

The chain-level lift of Proposition~\ref{prop:DS-Hochschild-cohom}
is the DS--Hochschild comparison in Vol~II
\ref*{thm:chd-ds-hochschild}. The statement is used only in its
declared principal and hook-type range.

\begin{theorem}[DS--Hochschild chain-level compatibility]%
\label{thm:DS-Hochschild-chain-level}%
Let $\fg$ be a simple Lie algebra of classical type and
$f \in \fg$ a principal or hook-type nilpotent. The DS strong
deformation retract $(p, i, h)$ induces a quasi-isomorphism of
chiral Hochschild complexes
\[
  (p \otimes p)^* \;:\;
  C^\bullet_{\mathrm{ch}}(\cW_k(\fg, f),\; \cW_k(\fg, f))
  \xrightarrow{\;\sim\;}
  C^\bullet_{\mathrm{ch}}(V_k(\fg),\; V_k(\fg))^{Q_{\mathrm{DS}}}
\]
at the chain level, under the SDR and mixed-action hypotheses of
Vol~II \ref*{thm:chd-ds-hochschild}, compatible with the
$E_3$-chiral action of Vol~II \ref*{thm:chiral-higher-deligne}.
Combined with the
class-L global triangle (Theorem~\ref{thm:triangle-GLC}), this gives
the chain-level class-M global triangle for principal and hook-type
nilpotents.
\end{theorem}

\begin{corollary}[Class M chain-level bulk--boundary--line]%
\label{cor:class-M-chain-level-triangle}%
For $\cA$ in class M with principal or hook-type Drinfeld--Sokolov
presentation (Virasoro, principal $\cW_N$, hook-type
$\cW_k(\fg, f)$) at generic central charge, and satisfying the
DS--Hochschild hypotheses of
Theorem~\ref{thm:DS-Hochschild-chain-level}, the full
bulk--boundary--line global triangle of
Theorem~\ref{thm:triangle-GLC} holds at the chain level. For
subregular and minimal nilpotents the theorem holds through the
coset decomposition only when the same comparison hypotheses are
verified for the coset factors; the weight-completed version of
Theorem~\ref{thm:triangle-classM-completed} holds in the completed
ambient category.
\end{corollary}

\section{Harmonic factorisation}
\label{sec:harmonic-factorization}

\begin{conjecture}[Harmonic factorisation of bar Hochschild]%
\label{prop:bar-hochschild-harmonic-factorization}%
For $\cA$ in the weight-completed class-M landscape at generic
central charge, the harmonic projector on the bar-Hochschild
complex $B^\Sigma(\cA)$ coincides at each weight with the
symmetric-group averaging map $\mathrm{av}\colon B^{\mathrm{ord}}(\cA)
\to B^\Sigma(\cA)$ composed with the canonical inclusion:
\[
  \mathcal{H} \;=\; \mathrm{av} \circ \iota.
\]
Under this hypothesis, the degree-$r$ component of the harmonic
projector computes the shadow coefficient $S_r(\cA)$, matching
Corollary~\ref{cor:cr-equals-Sr}.
\end{conjecture}

\section{The tempered stratum, scope-qualified}
\label{sec:tempered-stratum-scope}

The non-logarithmic class-M tempered range is the declared generic or
admissible exact-sector range below. Logarithmic triplet algebras
remain outside the proved range:
Gurarie 1993 and Flohr 1996 exhibit logarithmic amplitudes with
unbounded Massey-product behaviour despite finite-dimensional Zhu
algebras.

\begin{theorem}[Tempered stratum, scope-qualified]%
\label{thm:tempered-scope}%
The tempered stratum of the class-M landscape contains:
\begin{enumerate}[label=\textup{(\alph*)}]
\item Virasoro $\mathrm{Vir}_c$ at generic central charge
  $c \in \mathbb{C} \setminus \{\text{minimal series}\}$
  in the completed exact-sector ambient (via $\rho_*(c) = c/6$);
\item Principal $\cW_N$ for every $N \ge 2$ on the finite-envelope
  locus; the exact radius
  $\rho_*^{\cW_N}(c)=|c|/[12(H_N-1)]$ is available only under the
  harmonic-radius identification;
\item Irrational cosets \(\mathrm{Coset}(\fg, \fh, k)\) at generic
  level;
\item Non-logarithmic Schellekens VOAs in the standard central
  charge $c = 24$ catalogue.
\end{enumerate}
The logarithmic triplet algebras $\cW(p)$, $p \ge 2$, are outside
this theorem; their temperedness requires Adamovi\'c--Milas
character-amplitude bounds.
\end{theorem}

\begin{remark}[Non-tempered stratum on the non-logarithmic subset]
\label{rem:non-tempered-empty}
The non-tempered stratum is empty on the non-logarithmic exact-sector
standard landscape covered by Theorem~\ref{thm:tempered-scope}. The
C\textsubscript{2}-cofinite part is the admissible/minimal sublocus;
generic Virasoro and generic cosets live in the completed generic
ambient. The logarithmic $\cW(p)$ subcase is open.
\end{remark}

\section{Summary}
\label{sec:summary-classM}

\begin{enumerate}[label=\textup{(\arabic*)}]
\item \textbf{Classes G/L/C (chain level).} Direct bar-Hochschild
  identification; Theorem~\ref{thm:triangle-GLC}.
\item \textbf{Class M weight-completed.} MC5 weight-completed mechanism via
  \ref*{V1-thm:completed-bar-cobar-strong} +
  \ref*{V1-prop:standard-strong-filtration};
  Theorem~\ref{thm:triangle-classM-completed}. This is the
  weight-completed statement on the declared generic exact-sector
  class-M lanes; it is not a direct-sum chain-level theorem.
\item \textbf{Class M chain-level (principal / hook-type DS).} Vol~II
  \ref*{thm:chd-ds-hochschild} bridges DS cohomology at chain
  level under its SDR and mixed-action hypotheses;
  Corollary~\ref{cor:class-M-chain-level-triangle}.
\item \textbf{Scalar curvature channel.} The completed curved bar
  model has scalar central curvature
  $\kappa_{\mathrm{ch}}(\cA)\omega_g\otimes\mathbf{1}_{\cA}$
  (Proposition~\ref{prop:direct-sum-obstruction-resolution}); the harmonic
  projector equals averaging under the harmonic-factorisation
  hypothesis
  (Corollary~\ref{cor:cr-equals-Sr}).
\item \textbf{Direct-sum class-M genuinely false.} Nonzero
  $S_4(\mathrm{Vir}_c)$ obstructs chain-level quasi-isomorphism
  at weight~$4$ in the direct-sum category
  (Proposition~\ref{prop:direct-sum-obstruction}).
\item \textbf{Tempered stratum scope.} Virasoro, principal $\cW_N$
  finite-envelope loci, irrational cosets, non-logarithmic
  Schellekens; $\cW(p)$ open
  (Theorem~\ref{thm:tempered-scope}).
\end{enumerate}

In the class-M sector, the bulk algebra of 3d quantum gravity is the
chiral Hochschild cohomology of the boundary conformal algebra in the
completed category. On the original complex this identification is the
DS--Hochschild theorem under its stated hypotheses.
