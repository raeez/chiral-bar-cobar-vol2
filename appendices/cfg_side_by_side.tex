\section*{Prologue}

Costello--Francis--Gwilliam 2026~\cite{CFG2026} construct a filtered
$E_3$-algebra by BV quantisation of ordinary topological
Chern--Simons theory on a three-manifold, together with factorisation
homology traces that recover Reshetikhin--Turaev link invariants. They
do not construct six-dimensional holomorphic Chern--Simons on a
Calabi--Yau threefold, a K3 compactification, or a Borcherds--Kac--Moody
boundary algebra. This appendix therefore has one permissible use: it
compares the CFG topological-Chern--Simons lane with the Vol~II
holomorphic-topological lane through named functors. The chiral side is
presented by (a) the Chevalley--Eilenberg cochain algebra of a chiral
factorisation algebra on a reference curve $C$ and (b) the Sugawara
topologisation ladder that promotes the chiral direction to
$E_3^{\mathrm{top}}$ when the non-critical-level hypotheses hold.

The two lanes agree only after the relevant comparison theorem supplies
a functor and its hypotheses: Costello--Gwilliam BV observables and
Weiss descent for factorisation algebras; Lurie recognition for locally
constant factorisation algebras; Beilinson--Drinfeld/Francis--Gaitsgory
for the chiral/factorisation comparison on a curve; and the Vol~III
two-stage specialisation
$\Phi_d = {\SpCh}_{\Sigma_{d-1}, C} \circ \PhiFA_d$ for the
Calabi--Yau--to--chiral passage. The strict CE
generators-and-relations model and the Costello--Gwilliam cosheaf are
not equal objects before those functors are applied.

Each row below records the object being compared, the functorial bridge
used in the comparison, and the hypotheses under which the bridge is a
quasi-isomorphism or an equivalence in the relevant
$(\infty,1)$-category. When no such functor is present, the row records
an analogy of BV templates or Feynman weights, not an equality of
factorisation algebras.

\section{Row 1: the classical action}
\label{sec:cfg-row-1-classical-action}

\begin{theorem}[Row 1: cubic BV Chern--Simons templates]
\label{thm:cfg-row-1}
\index{Costello--Francis--Gwilliam!classical action}
Let $M$ be an oriented three-manifold and let
$A\in\Omega^1(M,\fg)$ be an ordinary Chern--Simons field. The CFG action
is the de~Rham cubic BV functional
\[
 S_{\mathrm{CS}}[A] =
 \int_M \Bigl(\tfrac12\langle A\wedge dA\rangle
 + \tfrac16\langle A\wedge[A\wedge A]\rangle\Bigr).
\]
Let $Y$ be a Calabi--Yau threefold with holomorphic volume form
$\Omega_Y$ and let $\cA\in\Omega^{0,1}(Y,\fg)$. The Vol~II hCS action is
\[
 S_{\hCS}[\cA] = \int_Y \Omega_Y \wedge
 \Bigl( \tfrac12\langle \cA\wedge \dbar\cA\rangle
 + \tfrac16\langle\cA\wedge[\cA\wedge\cA]\rangle\Bigr).
\]
The comparison is the replacement of the de~Rham elliptic complex
$(\Omega^\bullet(M),d)$ by the Dolbeault elliptic complex
$(\Omega^{0,\bullet}(Y),\dbar)$ together with integration against
$\Omega_Y$. It is not an equality of field spaces or factorisation
algebras.
\end{theorem}

\begin{proof}
Both functionals are generated by the same invariant pairing on $\fg$
and the same cubic Lie bracket tensor. The differential and integration
theory differ: CFG uses $d$ on a real three-manifold; hCS uses $\dbar$
on a complex threefold and the holomorphic volume form to produce a
top-degree integral. This gives a comparison of BV templates, not a
strict identification of observables.
\end{proof}

\begin{remark}[Chiral-avatar first-principles derivation]
\label{rem:cfg-row-1-avatar}
On a smooth curve $C$, the avatar replaces $\Omega^{0,\bullet}(Y)$ by
$\CE^\bullet(\fg\otimes\cO_C)$ and the cubic term by the Sugawara
quadratic current bracket. The avatar is witnessed at
\texttt{chapters/connections/six\_d\_hcs\_e3\_chiral\_avatar\_platonic.tex}
Definition~\ref{def:hcs-fields}. The comparison says that the
holomorphic Chern--Simons cubic tensor is represented, after the
Vol~II chiralisation functor and its hypotheses, by the integrated
Sugawara pairing on the $\widehat\fg_k$ current algebra, with level
$k$ proportional to the trace-form normalisation~\cite{CostelloYangian}.
\end{remark}

\section{\texorpdfstring{Row 2: the $L_\infty$-algebra of fields}{Row 2: the L-infinity algebra of fields}}
\label{sec:cfg-row-2-linfty}

\begin{theorem}[Row 2: $L_\infty$ fields in the de~Rham and Dolbeault lanes]
\label{thm:cfg-row-2}
\index{Costello--Francis--Gwilliam!$L_\infty$-algebra of fields}
The CFG topological Chern--Simons field algebra is the dg Lie algebra
$(\Omega^\bullet(M,\fg),d,[-,-])$ on a three-manifold $M$. The Vol~II
hCS field algebra is the Dolbeault dg Lie algebra
$(\Omegabar(Y,\fg),\dbar,[-,-])$ on a Calabi--Yau threefold $Y$, namely
the classical $L_\infty$-algebra $\cE_{\hCS}$ of
\texttt{chapters/connections/six\_d\_hcs\_e3\_chiral\_avatar\_platonic.tex}
Theorem~\ref{thm:hcs-classical-linfty}. These are parallel cubic
dg-Lie models. They become comparable only through a chosen elliptic
replacement or dimensional-reduction functor; there is no equality
$\Omega^\bullet(M,\fg)=\Omegabar(Y,\fg)$.
\end{theorem}

\begin{proof}
The two dg Lie algebras have the same algebraic form: unary
differential, binary pointwise Lie bracket, no higher brackets in the
classical cubic theory. Their Maurer--Cartan equations are different
elliptic equations, $dA+\tfrac12[A,A]=0$ in the topological lane and
$\dbar\cA+\tfrac12[\cA,\cA]=0$ in the holomorphic lane. A comparison
must therefore name the functor replacing $d$ by $\dbar$ or integrating
out the transverse directions. Without that functor the statement is a
formal analogy, not an isomorphism of field algebras.
\end{proof}

\begin{remark}[First-principles verification via cubic bound]
\label{rem:cfg-row-2-cubic-bound}
The form-degree bound that kills $\ell_{n\geq 3}$ is the statement
$\Omega^{0,q}(Y) = 0$ for $q\geq 4$ on a complex threefold, which on
the chiral-avatar side becomes the triviality of the bar-length-$\geq 4$
contribution to the OPE cubic vertex; the current-algebra OPE
$J^a(z)J^b(w)$ has at most quadratic poles, so the triple contraction
$J^a J^b J^c$ produces no pole of order $\geq 3$. This is a parallel
degree-bound computation in the two lanes; it is not an isomorphism of
the de~Rham and Dolbeault field complexes.
\end{remark}

\section{Row 3: BV ghost-antifield content}
\label{sec:cfg-row-3-ghost-antifield}

\begin{theorem}[Row 3: BV shifted-cotangent template]
\label{thm:cfg-row-3}
\index{ghost!BV!six-dimensional holomorphic Chern--Simons}
The CFG BV phase space for ordinary Chern--Simons and the hCS BV phase
space have the same shifted-cotangent form: fields plus
Poincare/Serre-dual antifields. In the hCS lane this graded complex is
\[
 \cE^{\BV}_Y
 = \Omegabar(Y,\fg)[1]\oplus \Omegabar(Y,\fg)^\vee[-2]
\]
of
\texttt{chapters/connections/six\_d\_hcs\_e3\_chiral\_avatar\_platonic.tex}
Definition~\ref{def:hcs-bv-brst}, and with
\texttt{chapters/connections/bv\_brst.tex}
Definition~\ref{def:6dhcs-shifted-cotangent}. The four Dolbeault
components $(0,0), (0,1), (0,2), (0,3)$ give the hCS BV-degree classes
(ghost $c$ at $(0,0)$, field $\cA$ at $(0,1)$, antifield $\cA^*$ at
$(3,2)$, antighost-of-ghost $c^*$ at $(3,3)$).
\end{theorem}

\begin{proof}
The graded complex $\cE^{\BV}_Y = \cL_Y[1]\oplus \cL_Y^\vee[-2]$ is the
$(-1)$-shifted cotangent of $\cL_Y[1]$. Serre duality on $Y^6$ with
holomorphic volume $\Omega_Y$ identifies
$(\Omega^{0,p}(Y,\fg))^\vee\simeq \Omega^{3,3-p}(Y,\fg^\vee)$. The
four components thus become
$\Omega^{0,0}(Y,\fg),\Omega^{0,1}(Y,\fg),\Omega^{3,2}(Y,\fg^\vee),\Omega^{3,3}(Y,\fg^\vee)$
at BV-degrees $0, 1, 2, 3$. Setting
ghost number $\mathrm{gh}(c) = 1$, $\mathrm{gh}(\cA) = 0$,
$\mathrm{gh}(\cA^*) = -1$, $\mathrm{gh}(c^*) = -2$ and antifield degree
$\mathrm{af}(c) = 0$, $\mathrm{af}(\cA) = 0$, $\mathrm{af}(\cA^*) = 1$,
$\mathrm{af}(c^*) = 2$ yields the shifted-cotangent degree pattern.
The CFG topological model has the analogous pattern with de~Rham
degrees in place of Dolbeault degrees.
\end{proof}

\section{Row 4: the BV symplectic form}
\label{sec:cfg-row-4-symplectic}

\begin{theorem}[Row 4: $(-1)$-shifted BV symplectic template]
\label{thm:cfg-row-4}
\index{symplectic form!shifted, six-dimensional holomorphic Chern--Simons}
The ordinary Chern--Simons $(-1)$-shifted BV symplectic form has
Poincare-duality form. Its hCS Dolbeault analogue is
\[
 \omega^{\hCS}_{\BV}
 = \int_Y\Omega_Y\wedge\bigl(\delta\cA\wedge\delta\cA^*
 +\delta c\wedge \delta c^*\bigr),
\]
and it is the form $\omega_{\BV}$
of~\eqref{eq:6dhcs-symplectic} in
\texttt{chapters/connections/bv\_brst.tex}
Definition~\ref{def:6dhcs-antibracket}. Non-degeneracy, closedness,
and ghost degree $-1$ are witnessed at
\texttt{chapters/connections/bv\_brst.tex}
Proposition~\ref{prop:6dhcs-shifted-symplectic}.
\end{theorem}

\begin{proof}
The hCS pairing is the Serre-duality pairing on $Y^6$ composed with
$\langle-,-\rangle$ on $\fg$, integrated against $\Omega_Y$. Closedness
follows from constant-coefficient bilinearity on the linear space
$\cE^{\BV}_Y$; non-degeneracy from the non-vanishing of $\Omega_Y$ and
the perfectness of $\langle-,-\rangle$; ghost degree $-1$ from the
pairing of $\cL_Y[1]$ with $\cL_Y^\vee[-2]$ landing in degree
$-1$. The only normalisation ambiguity is an overall sign, which is
fixed by the convention
$\omega_{\BV}(\cA,\cA^*) = +\int_Y\Omega_Y\wedge\langle\cA,\cA^*\rangle$.
The ordinary topological lane uses the corresponding Poincare pairing
on $M$.
\end{proof}

\section{Row 5: the classical master equation}
\label{sec:cfg-row-5-classical-master}

\begin{theorem}[Row 5: classical master equation in the two lanes]
\label{thm:cfg-row-5}
\index{classical master equation!Costello--Francis--Gwilliam comparison}
The ordinary Chern--Simons classical master equation and the hCS
classical master equation have the same Jacobi-Leibniz proof. In the
hCS lane this is the statement of
\texttt{chapters/connections/six\_d\_hcs\_e3\_chiral\_avatar\_platonic.tex}
Proposition~\ref{prop:hcs-classical-master} and with
\texttt{chapters/connections/bv\_brst.tex}
Theorem~\ref{thm:6dhcs-bv-master-action}. The antibrackets are compared
by the shifted-cotangent template of Row~4; they are not literally the
same pairing before choosing the de~Rham-to-Dolbeault comparison.
\end{theorem}

\begin{proof}
Both proofs reduce to three identities: (i) $d^2=0$ or $\dbar^2 = 0$;
(ii) Jacobi on $\fg$; (iii) the Leibniz rule for the relevant
differential.
The Calabi--Yau condition $\dbar\Omega_Y = 0$ eliminates boundary
terms. The chiral-avatar version is the Jacobi identity of the
$\widehat\fg_k$ OPE on $C$. It has the same Jacobi--Leibniz pattern
only after the Serre-dual/Ran-space comparison has replaced
$\Omega^{0,\bullet}(Y)$ by the chiral factorisation model on the
curve.
\end{proof}

\section{Row 6: the regularised BV Laplacian}
\label{sec:cfg-row-6-bv-laplacian}

\begin{theorem}[Row 6: BV Laplacian and regularisation comparison]
\label{thm:cfg-row-6}
\index{BV Laplacian!six-dimensional holomorphic Chern--Simons}
The ordinary Chern--Simons BV Laplacian $\Delta_L$, defined formally as
$\frac{\delta^2}{\delta\cA\,\delta\cA^*}
+ \frac{\delta^2}{\delta c\,\delta c^*}$
regularised by a heat-kernel parametrix, has the same BV role as the
regularised hCS Laplacian $\Delta_{\BV}^L$ of
\texttt{chapters/connections/bv\_brst.tex}
Definition~\ref{def:6dhcs-bv-laplacian} and as the
$\Delta_L = \{P_L,-\}$ of
\texttt{chapters/connections/thqg\_perturbative\_finiteness.tex}
\S\ref{subsec:hcs-qme}. The comparison is through the choice of
propagator and renormalisation scheme; it is not an equality of
operators on the same field complex.
\end{theorem}

\begin{proof}
Both operators are second-order variational operators pairing fields
with antifields, but they regularise different elliptic complexes. In
the CFG topological lane the heat kernel is the de~Rham heat kernel on
the three-manifold $M$. In the hCS lane the bare distributional kernel
on $Y\times Y$ is replaced by
$P_L^{\hCS}=\int_0^L K_t^{\dbar}\,dt$, with $K_t^{\dbar}$ the heat
kernel of the Dolbeault Laplacian. The chiral-avatar version uses the
Dolbeault heat kernel on the curve $C$. Costello's renormalisation group
identifies changes of scale within each lane by BV-exact counterterms.
A comparison between the hCS parameter $L$ and the curve-length
parameter is therefore a one-loop statement after the specified
pushforward reduction, not an equality of the CFG and chiral
regularisation operators.
\end{proof}

\begin{conjecture}[All-loop regularisation comparison between hCS and
chiral-avatar schemes; \ClaimStatusConjectured]
\label{conj:cfg-row-6-all-loop}
\index{regularisation!all-loop bijection}
The one-loop identification of Theorem~\ref{thm:cfg-row-6} extends to
an all-loop comparison between the hCS Dolbeault heat-kernel parameter
tower $\{P_L^{\hCS}\}_{L>0}$ on $Y$ and the curve-length parameter
tower on the chiral avatar reference curve $C$. Concretely: for every
$n\geq 1$ and every choice of hCS scale $L$, there exists a curve-length scale
$\ell = \ell_n(L)$ and a regulator-matching chain-homotopy
$h_n\colon\Delta_L^{(n)}\to\Delta_\ell^{(n),\mathrm{ch}}$ such that
the two $n$-loop effective actions $S_L^{(n)}$ and $S_\ell^{(n),\mathrm{ch}}$
differ by a $\dbar$-exact BV counterterm.
\end{conjecture}

\begin{remark}[Scope of Conjecture~\ref{conj:cfg-row-6-all-loop}]
\label{rem:cfg-row-6-scope}
At one loop, the comparison $\ell_1(L)$ follows from
Theorem~\ref{thm:cfg-row-6}. At two loops, partial progress is
established by Costello--Li~\cite{CostelloLi2020} Appendix~C for the
$\mathfrak{sl}_2$ case; higher-loop matching is open. The conjecture
is consistent with the Feynman-coefficient comparison of Rows~13 and
the loop-expansion matching of
\texttt{chapters/theory/six\_d\_hcs\_feynman\_coefficients.tex}
Proposition~\ref{prop:cfg-6d-comparison}. Anomaly-cancellation
computations at one loop (Rows~7--8) are unaffected by the conjectural
status: the one-loop anomaly class is regularisation-independent up to
BV-exact counterterms.
\end{remark}

\section{Row 7: the quantum master equation}
\label{sec:cfg-row-7-qme}

\begin{theorem}[Row 7: quantum master equation as BV template]
\label{thm:cfg-row-7}
\index{quantum master equation!Costello--Francis--Gwilliam comparison}
The ordinary Chern--Simons quantum master equation
$\tfrac12\{S_L,S_L\} + \hbar\Delta_L S_L = 0$ and the hCS quantum master
equation have the same BV form. In the hCS lane this is
\texttt{chapters/connections/six\_d\_hcs\_e3\_chiral\_avatar\_platonic.tex}
Theorem~\ref{thm:hcs-QME-one-loop} and with
\texttt{chapters/connections/thqg\_perturbative\_finiteness.tex}
equation~\eqref{eq:qme}. The comparison is made after choosing a
renormalisation scheme and passing through the factorisation-algebra
comparison functor.
\end{theorem}

\begin{proof}
Expand both sides in $\hbar$. The $\hbar^0$ piece is Row~5. The
$\hbar^1$ piece equates the one-loop anomaly to the
$\{S_{\mathrm{cl}},T_L\}$-ambiguity modulo BV-exact terms, which is
Row~8. At all higher orders, Costello's renormalisation-group flow
of~\eqref{eq:rg} fixes $S_L$ recursively inside a chosen BV theory. CFG
verify this order-by-order statement for ordinary topological
Chern--Simons. The hCS and chiral-avatar statements use their own
Dolbeault parametrices; they are compared only after the
factorisation-algebra comparison functor and the regularisation
homotopies of Row~6 have been chosen.
\end{proof}

\section{Row 8: the one-loop anomaly and Deligne cancellation}
\label{sec:cfg-row-8-anomaly}

\begin{theorem}[Row 8: one-loop anomaly comparison]
\label{thm:cfg-row-8}
\index{one-loop anomaly!Deligne exceptional series}
The hCS one-loop anomaly is represented by the quartic trace
\[
 \mathrm{Obs}_1
 \;=\;
 \frac{1}{(2\pi i)^3}\,\Tr_{\mathrm{adj}}
 \int_Y\Omega_Y\wedge c\wedge\dbar c\wedge\dbar c\wedge\dbar c,
\]
and is computed in \texttt{chapters/connections/bv\_brst.tex}
Theorem~\ref{thm:6dhcs-one-loop-anomaly}. The Deligne-exceptional
cancellation in the Vogel family
$A_1, A_2, G_2, D_4, F_4, E_6, E_7, E_8$ is the hCS statement
\texttt{chapters/connections/bv\_brst.tex}
Corollary~\ref{cor:6dhcs-deligne-cancellation}.
\end{theorem}

\begin{proof}
The anomaly cocycle lives in $H^1_{\BRST}(\Omega^{3,\bullet}(Y)\otimes
\Lambda^\bullet\fg^\vee)$ at ghost number $+1$ with coefficient the
symmetric quartic Casimir
$\Tr_{\mathrm{adj}}(T^a T^b T^c T^d)^{\mathrm{sym}}$. The Vogel-plane
factorisation
$\Tr_{\mathrm{adj}}T^4 = \alpha_\fg\cdot(\Tr_{\mathrm{adj}}T^2)^2$ for
$\fg$ on the Deligne series forces the anomaly into a
$(F^2)^2$-square, which is absorbed by Green--Schwarz counterterms
constructed from the twistor transform of the
Kodaira--Spencer gauge field. The chiral-avatar first-principles
derivation is the scalar-curvature term
$c_2(\fg)/24\cdot\int_Y\Omega_Y\wedge \Tr(\cA\wedge\dbar\cA\wedge\dbar\cA)$
of~\eqref{eq:one-loop-anomaly}: vanishing of this term on
$\fg = \fu(1)$ (where $h^\vee = 0$) is the Heisenberg
archetype, and on each Deligne-series member the $(F^2)^2$ piece
matches the Vogel-plane reduction of
Corollary~\ref{cor:6dhcs-deligne-cancellation}.
\end{proof}

\begin{remark}[Chiral-avatar dichotomy: abelian vs non-abelian]
\label{rem:cfg-row-8-dichotomy}
On the chiral avatar, the anomaly vanishes unconditionally for
$\fg = \fu(1)$ (Remark~\ref{rem:hcs-anomaly-locus} of
\texttt{thqg\_perturbative\_finiteness.tex}); for simple $\fg$ the
anomaly absorbs through the level-shift
$k\mapsto k + h^\vee$, delivering $\widehat\fg_{-h^\vee}$ at the
critical twist. The Deligne-cancellation recorded here is the Vol~II
hCS non-abelian refinement: the anomaly is cancelled, not absorbed, for
exactly those $\fg$ on the Vogel plane. It is not imported from
CFG2026.
\end{remark}

\section{\texorpdfstring{Row 9: the $\Disk_3$-action on classical observables}{Row 9: the Disk 3 action on classical observables}}
\label{sec:cfg-row-9-disk3}

\begin{theorem}[Row 9: $\Disk_3$-action after local constancy]
\label{thm:cfg-row-9}
\index{little disks!action on classical observables}
The CFG $\Disk_3$-action belongs to locally constant ordinary
Chern--Simons observables on a real three-manifold. The hCS/chiral
avatar carries the corresponding $\Disk_3$-action only after the
topological-reduction hypotheses of
\texttt{chapters/connections/celestial\_holography.tex}
Proposition~\ref{prop:disk3-action-classical-observables}. In that
reduced setting:
the classical observable sheaf $\Fobs^{\mathrm{cl}}$ on $Y = \C^3$
(resp.\ on the chiral avatar $\mathrm{K3}\times C\times\R$) carries an
action of the little $3$-disks operad $\Disk_3$, induced by the
three-fold-product factorisation structure on $Y$ (resp.\ the three
Dolbeault directions plus the topological $\R$-direction after K3
reduction).
\end{theorem}

\begin{proof}
Costello--Gwilliam~\cite[Ch.~4, Thm.~4.4.1]{CG17} establish the
$\Disk_{n}$-action on classical observables of a holomorphic-cotangent
BV theory on a real-$n$-dimensional target. For $Y = \C^3$, real
dimension is $6$; after $\dbar$-Dolbeault reduction (which contracts
three cohomologically-trivial antiholomorphic directions), the
effective $\Disk$-dimension drops to $3$. The chiral-avatar construction
uses $X = \mathrm{K3}\times C$ (complex dimension $3$, i.e., real
dimension $6$) plus a topological $\R$-direction; after K3 reduction
(which contracts the K3 factor), the residual operadic dimension is
$1 + 2 = 3$. After these reductions the operads
$\Disk_3(n)\simeq \Conf_n(\R^3)$ and the operadic composition maps
are compared by Lurie's locally-constant factorisation-algebra
recognition theorem.
\end{proof}

\section{\texorpdfstring{Row 10: the $E_3$-structure from $\R^6$-translation}{Row 10: the E3 structure from R6 translation}}
\label{sec:cfg-row-10-e3-translation}

\begin{theorem}[Row 10: $E_3$-algebra from local constancy]
\label{thm:cfg-row-10}
\index{$E_3$-algebra!six-dimensional holomorphic Chern--Simons}
Lurie $\mathrm{HA}.5.5.4.10$ extracts an $E_3$-algebra from a locally
constant factorisation algebra on $\R^3$. CFG apply this mechanism to
ordinary topological Chern--Simons observables. The hCS avatar reaches
the same operadic type only after Dolbeault reduction and Sugawara
topologisation, as in
\texttt{chapters/connections/six\_d\_hcs\_e3\_chiral\_avatar\_platonic.tex}
Theorem~\ref{thm:hcs-e3-from-translation} and with
\texttt{chapters/connections/bv\_ht\_physics.tex}
Theorem~\ref{thm:E3-action-6d-hCS}, Theorem~\ref{thm:dunn-E3-6d-hCS}.
\end{theorem}

\begin{proof}
Lurie $\mathrm{HA}.5.5.4.10$ extracts an $E_n$-algebra from any
translation-equivariant locally constant factorisation algebra on
$\R^n$. One may not apply this recognition theorem to the unreduced hCS
factorisation algebra as if it were already locally constant in all six
real directions. After the Dolbeault/topological reduction has killed
the antiholomorphic directions and supplied local constancy in the
remaining three directions, the reduced hCS output is an $E_3$-algebra.
The chiral-avatar version uses
$\SCchtop$ on a curve $C$ (open colour $E_1$ on $\R$, closed colour
$E_2$ on $\mathrm{FM}(\C)$) promoted to $E_3^{\mathrm{top}}$ by the
Sugawara-ladder topologisation at non-critical level
(Chapter~\ref{ch:topologization}). The bridge is Dunn additivity after
the stated reductions:
$E_3\simeq E_2\otimes_{\mathrm{BV}} E_1$ of
Theorem~\ref{thm:dunn-E3-6d-hCS}.
\end{proof}

\begin{remark}[Dunn splitting convention]
\label{rem:cfg-row-10-dunn}
The ordinary CFG side starts with a real three-manifold and therefore
does not factorise an $\R^6$ holomorphic target. The reduced hCS side
may be written as $\C^3=\C^2\times\C$, pairing the $E_2$ part with the
holomorphic $\C^2$-plane (Fulton--MacPherson configuration space) and
the $E_1$ part with the residual $\C$-direction (the spectral curve
$C$). Dunn additivity is used only after the hCS and chiral-avatar
reductions have produced the same operadic type
$E_3\simeq E_2\otimes E_1$. The
Dunn-reduction machinery is Lurie $\mathrm{HA}.5.1.2$ via
Francis~\cite{francis2013tangent} Theorem~2.4; no conflict arises
between the two citations, because Francis 2013 specialises Lurie
$\mathrm{HA}.5.5$ to the tangent-complex-and-Hochschild-cohomology
setting that produces the Gerstenhaber bracket.
\end{remark}

\section{\texorpdfstring{Row 11: the $P_3$-Poisson structure}{Row 11: the P3-Poisson structure}}
\label{sec:cfg-row-11-p3-poisson}

\begin{theorem}[Row 11: $P_3$-Poisson structure after reduction]
\label{thm:cfg-row-11}
\index{Poisson algebra!$P_3$, six-dimensional holomorphic Chern--Simons}
The $P_3$-Poisson structure is the homology shadow of a reduced
$E_3$-algebra: the pair $(m_2,\{-,-\}_{\BV})$ is a
graded-commutative algebra with a degree-$(-2)$ Lie bracket satisfying
Leibniz. In the hCS lane this is
\texttt{chapters/connections/bv\_ht\_physics.tex}
Theorem~\ref{thm:P3-obs-6d-hCS}. The Koszul-self-duality $P_3^!\simeq P_3$
of Ginzburg--Kapranov is the operadic input on both lanes after
reduction.
\end{theorem}

\begin{proof}
The $E_3$-bracket of Row~10 is of degree $-2$; pairing with the
commutative multiplication from the factorisation product yields a
$P_3$-structure. The Leibniz identity
$[a, bc]_{E_3} = [a,b]_{E_3}c + (-1)^{|a||b|}b[a,c]_{E_3}$
follows from the $E_3$-cotangent-pairing interpretation of

\end{proof}

\section{Row 13: OPE relations from Feynman integrals}
\label{sec:cfg-row-13-ope}

\begin{theorem}[Row 13: OPE relations from Feynman integrals]
\label{thm:cfg-row-13}
\index{OPE!six-dimensional holomorphic Chern--Simons!Feynman}
The six-dimensional hCS OPE generators-and-relations,
\[
 O^a_{(0,0,0)}(z)\,O^b_{(0,0,0)}(w)
 \;\sim\;
 \frac{f^{ab}_c\,O^c_{(0,0,0)}(w)}{\|z-w\|^4}
 + \frac{\langle T^a,T^b\rangle\,\mathbb 1}{\|z-w\|^6}
 + \mathrm{regular},
\]
are compared with
\texttt{chapters/connections/six\_d\_hcs\_e3\_chiral\_avatar\_platonic.tex}
Theorem~\ref{thm:hcs-ope-relations} under the Mellin-transform
identification of six-dimensional-Euclidean propagator
$\|z-w\|^{-4}$ with the one-dimensional chiral propagator
$(z-w)^{-2}$.
\end{theorem}

\begin{proof}
The six-dimensional Bochner--Martinelli propagator in the hCS lane is
$G_{\C^3}(z-w) = \Omega^{-1}\cdot \dbar^\vee\delta^{(6)}(z-w)$ with
Fourier transform $\|k\|^{-2}$, giving coincidence-limit pole
$\|z-w\|^{-4}$. Restricted to the spectral-curve direction after
Dunn splitting $\C^3 = \C^2\times\C$, the transverse $\C^2$
integration converts
$\|z-w\|^{-4}\to C\cdot(w_1-w_2)^{-2}$ by the residue of the
transverse integral against the Fourier-transform kernel (standard
Mellin-transform computation on the two-plane):
\[
 \int_{\C^2}\frac{d^2z_{\perp}}{(|w|^2+|z_\perp|^2)^2}
 = \frac{\pi}{|w|^2}.
\]
The central term $\langle T^a,T^b\rangle/\|z-w\|^6$ carries the extra
factor $\|z-w\|^{-2}$ on the six-dimensional side; after Mellin, this
becomes the residue factor that fixes the level convention for the
chiral central term $k\delta^{ab}/(w_1-w_2)^2$. Thus the comparison is
one of residue-normalised OPE coefficients, not equality of raw
six-dimensional and one-dimensional kernels. The level $k$ proportional
to the trace-form normalisation is~\cite{CostelloYangian}.
\end{proof}

\end{theorem}

\begin{proof}
Stepwise. Step~1: the pushforward $\int_{K3}$ converts the
6d theory into a $4$d theory on $\R^3\times\C^2$ by the

producing the twisted $\Sigma_{0,24}$-defect geometry. Step~2:

Vol~III. Step~3: bar cohomology of $\mathbf H_{\Delta_5}$ equals
$\fg_{\Delta_5}$ by the Borcherds universal-enveloping
identification. CFG2026 supplies the ordinary Chern--Simons
factorisation-homology technology in a different source lane; it does
not supply the K3/BKM input. The quasi-isomorphism here is conditional
on the Vol~II K3 pushforward and the Vol~III Borcherds denominator
theorem, not on CFG2026.
\end{proof}

\begin{remark}[First-principles BKM-denominator check]
\label{rem:cfg-row-14-denominator}

  \Delta_5
  =
  \prod_{\lambda\in\Pi^{3,2}_+}(1-e^{-\lambda})^{m(\lambda)} ,

with multiplicities \(m(\lambda)\) read from the Borcherds lift of the

  \kappa_{\mathrm{BKM}}(\Delta_5)=5 .
\]
Thus the apparent half-weight computation \(10/2=5\) is only a square-root
check for \(\Phi_{10}^{\mathrm{un}}\), not the definition of

\begin{theorem}[Row 15: factorisation-pushforward and two-stage $\Phi_d$]
\label{thm:cfg-row-15}
\index{two-stage factorisation!Costello--Francis--Gwilliam comparison}
The Vol~III factorisation-pushforward
$\int_{K3}\int_{\R^3}$ from the 6d bulk to the 3d/2d boundary is the

\begin{proof}

\end{proof}

\section{Koszul--Tate resolution vs bar--cobar adjunction}
\label{sec:cfg-kt-bar-cobar}

\begin{theorem}[hCS Koszul--Tate complex and chiral bar complex under
Serre duality; \ClaimStatusProvedElsewhere\,\cite{CG17}]
\label{thm:cfg-kt-bar-cobar}
\index{Koszul--Tate resolution!chiral bar identification}
The Vol~II hCS Koszul--Tate complex, carrying the Koszul--Tate
differential $\delta_\KT$ of \eqref{eq:6dhcs-kt-differential} and
defined on the graded space
$\Omega^{\bullet,\bullet}(Y)\otimes\Lambda^\bullet[\cA^*,c^*]
\otimes\fg^{\otimes\bullet}$, is compared by Serre duality and the
chiralisation functor to the chiral bar complex
$B^{\mathrm{ch}}(\widehat\fg_k)$ of the current-algebra chiral algebra
at level $k$ under the hypotheses of
\texttt{chapters/connections/bv\_brst.tex}
Theorem~\ref{thm:bv-bar-geometric}. Under this quasi-isomorphism, the
BV differential $Q_{\BV}=\{S_{\BV},-\}$ corresponds to the chiral bar
differential $d_{\mathrm{bar}}$, and the higher antifields
$A^*,c^*$ correspond to the higher bar-length generators
$s^{-1}\bar\cA^{\otimes n}$ of $B^{\mathrm{ch}}(\widehat\fg_k)$ for
$n\geq 2$.
\end{theorem}

\begin{proof}
Theorem~\ref{thm:bv-bar-geometric} is the statement
$C_\BV(\cA)\simeq \bar B^{\mathrm{ch}}(\cA)$ for any chiral algebra
$\cA$ on $\P^1$; the proof constructs an explicit generator-matching
between bar generators (operator insertions $\phi_i\in\cA$ paired
with logarithmic $1$-forms $\eta_{ij}$ on
$\overline C_{n+1}(\P^1)$) and Koszul--Tate generators (antifields at
shifted form-degrees on the Serre-dual side). The BV-bracket is a
Poincar\'e residue on the log-de~Rham complex of the compactified
configuration space, which by the Arnold--Brieskorn theorem generates
$H^1(C_n(X);\C)$; the residue map $\Res_{D_{ij}}$ is the chiral-bar
differential of~\eqref{eq:chiral-bar-differential-hcs}, witnessed at
\texttt{chapters/connections/six\_d\_hcs\_e3\_chiral\_avatar\_platonic.tex}
Theorem~\ref{thm:hcs-chiral-bar-differential}.

The hCS Koszul--Tate resolution of~\eqref{eq:6dhcs-kt-differential}
has $\delta_\KT\cA^* = F^{0,2}(\cA) = \dbar\cA + \tfrac12[\cA,\cA]$ and
$\delta_\KT c^* = \dbar\cA^* + [\cA,\cA^*]$; the bar-differential
counterpart is $d_{\mathrm{bar}}(a_1\otimes a_2) = \mu_{\mathrm{ch}}(a_1,a_2)$
at bar-length $2$ and
$d_{\mathrm{bar}}(a_1\otimes a_2\otimes a_3) = \mu(a_1,a_2)\otimes a_3
 - a_1\otimes\mu(a_2,a_3)$ at length $3$. The map
$\cA^*\leftrightarrow s^{-1}\bar\cA\otimes s^{-1}\bar\cA$ sends the
quadratic Koszul--Tate generator to the bar-length-$2$ generator, and
similarly $c^*\leftrightarrow s^{-1}\bar\cA\otimes s^{-1}\bar\cA\otimes
s^{-1}\bar\cA$ at bar-length $3$. Nilpotency of $\delta_\KT$
(Bianchi identity $\dbar F^{0,2} + [\cA,F^{0,2}] = 0$) matches
nilpotency of $d_{\mathrm{bar}}$ (OPE associativity up to chain
homotopy supplied by the Arnold $\eta_{ij}\wedge\eta_{jk} +
\mathrm{cyclic} = 0$).

The bar--cobar adjunction Theorem~A of Vol~I, $\Omega^{\mathrm{ch}}
\dashv B^{\mathrm{ch}}$, is the universal inversion $\Omega\circ B
= \mathrm{id}$. On the BV side, the Koszul--Tate resolution is a
quasi-isomorphism onto the on-shell algebra
$C^\infty(\mathcal F)/\mathcal I$; the converse $\Omega^{\mathrm{ch}}$
(the chiral cobar) reconstructs $\cA$ from
$B^{\mathrm{ch}}(\cA)$ by the inverse construction, which on the 6d
side is the Costello regularisation of the
Koszul--Tate resolution's antifield tower into a local action.
\end{proof}
