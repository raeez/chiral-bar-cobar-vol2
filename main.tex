\title{\textit{Modular Homotopy Theory for Factorization Algebras on Curves}\\[0.5em]
\large Volume~2: $\Ainf$~Chiral Algebras and 3D Holomorphic--Topological QFT}

\begin{abstract}
\itshape

\noindent
The bar complex $\barBch(\cA)$ of a chiral algebra on a curve
carries two algebraic structures.  The differential $D_\cA$
extracts OPE residues on $\FM_k(\C)$: holomorphic factorisation.
The coproduct
\[
\Delta[s^{-1}a_1|\cdots|s^{-1}a_n]
\;=\;
\sum_{i=0}^{n}
[s^{-1}a_1|\cdots|s^{-1}a_i]
\otimes
[s^{-1}a_{i+1}|\cdots|s^{-1}a_n]
\]
from ordered deconcatenation.  This coproduct encodes topological
factorisation along~$\R$: place $n$ fields at ordered positions
$t_1 < \cdots < t_n$ on~$\R$, choose a cut point
$t_i < t_* < t_{i+1}$; the fields on $(-\infty, t_*)$ are
$a_1, \ldots, a_i$, the fields on $(t_*, +\infty)$ are
$a_{i+1}, \ldots, a_n$, and the $i$-th summand of~$\Delta$ is
exactly this factorisation.  The coderivation property
$\Delta \circ D_\cA = (D_\cA \otimes \id + \id \otimes D_\cA)
\circ \Delta$ says that holomorphic collisions commute with
topological cuts: a collision $z_i \to z_j$ in $\C$ does not move
the cut point $t_*$ in~$\R$.

A degree-$k$ bar element therefore lives on
$\FM_k(\C) \times \Conf_k^<(\R)$: the differential sees the first
factor, the coproduct sees the second.  Together they make
$(\barBch(\cA),\, D_\cA,\, \Delta)$ an algebra over the
holomorphic--topological Swiss-cheese operad
$\SCchtop$ on $\C \times \R$.

The genus-zero $L_\infty$ brackets $\ell_k^{(0)}$ are
residues over tree-type boundary strata of $\FM_k^{\log}(\C)$;
the genus-raising operator $\ell_1^{(1)}$ is the non-separating
clutching trace.  The $\Ainf$ chiral algebra structure is the twisting
morphism
\[
\alpha_T \in \MC\bigl(\mathfrak{g}^{\mathrm{SC}}_T\bigr),
\qquad
\mathfrak{g}^{\mathrm{SC}}_T :=
\Hom_\Sigma\bigl(\mathbf{B}(\SCchtop),\,
\End_{(B_\partial,\,\cC_{\mathrm{line}})}\bigr),
\]
whose closed-colour projection $(m_k)_{k\ge 1}$
encodes the $\Ainf$ chiral algebra:
each~$m_k\colon A^{\otimes k}\to A((\lambda_1))\cdots((\lambda_{k-1}))$
is a configuration-space integral on
$\FM_k(\C) \times \Conf_k(\R)$, and the Maurer--Cartan equation
$\partial\alpha_T + \alpha_T \star \alpha_T = 0$
is the spectral Stasheff identity.

\medskip
\textsc{Theorem~F} (Homotopy-Koszulity and Quillen equivalence).
Kontsevich formality gives a quasi-isomorphism
$\SCchtop \xrightarrow{\sim} \mathsf{SC}$
to the classical Swiss-cheese operad, which is Koszul; bar-cobar
transfer yields
\[
\Omega\mathbf{B}(\SCchtop) \;\xrightarrow{\;\sim\;}\; \SCchtop.
\]
The bar-cobar adjunction on $\SCchtop$-algebras is a Quillen
equivalence.  Under axioms~\textup{(H1)--(H4)} on a 3d
holomorphic--topological QFT, Weiss cosheaf descent produces the
$\SCchtop$-algebra from BV-BRST data.

\textsc{Theorem~G} (PVA descent).
Write $m_2(a,b;\zeta) = m_2^{\mathrm{reg}}(a,b;\zeta)
+ \sum_{n \geq 0} a_{(n)}b\;\zeta^{-n-1}$.
On $H^\bullet(A,Q)$:
\[
\mu(a,b) := m_2^{\mathrm{reg}}(a,b;0),
\qquad
\{a_\lambda b\} := \sum_{n \geq 0}
\frac{\lambda^n}{n!}\,a_{(n)}b,
\qquad
m_{k \geq 3}\big|_{H^\bullet} = 0.
\]
The pair $(\mu,\, \{-_\lambda -\})$ is a $(-1)$-shifted Poisson
vertex algebra: Jacobi from Arnold relations on $\FM_3(\C)$;
Leibniz from three-face Stokes on $\partial\FM_3(\C)$;
vanishing of $m_{k \geq 3}$ by topological contractibility of~$\R$.
At chain level, $\ell_3^{(0)}$ and $\ell_4^{(0)}$ are residue
formulas over trivalent and tetravalent trees on rigid log-FM
boundary strata.

\textsc{Theorem~J} (Bulk-boundary-line triangle).
\[
\cA_{\mathrm{bulk}}
\;\simeq\; Z_{\mathrm{der}}(\mathcal{B}_\partial)
\;\simeq\; \ChirHoch^\bullet(\cA^!),
\qquad
\cA^!\text{-}\mathsf{mod} \;\simeq\; \{\text{line operators}\}.
\]
The Koszul dual~$\cA^!$ is a dg-shifted Yangian.
The spectral braiding
$R(z) \colon L_1 \otimes L_2 \xrightarrow{\sim} L_2 \otimes L_1$
satisfies the Yang--Baxter equation: Stokes' theorem on $\FM_3(\C)$
applied to the triple convolution $\alpha \star \alpha \star \alpha$
decomposes the boundary of~$\FM_3(\C)$ into pairwise collision
divisors, one for each side of the YBE.
For a 3d holomorphic--topological theory~$T$, the
bulk-boundary-line structure assembles into
$\mathcal{H}(T)
= (\cA,\, \cA^!,\, \mathcal{C},\, r(z),\,
\Theta_\cA,\, \nabla^{\mathrm{hol}})$,
a modular Koszul datum whose five theorem targets are
projections of the single MC equation
$D_\cA(\Theta_\cA) + \tfrac{1}{2}[\Theta_\cA, \Theta_\cA] = 0$.
The colour grading of $\SCchtop$ decomposes $\alpha_T
= \alpha_T^{\mathrm{cl}} + \alpha_T^{\mathrm{mix}}
+ \alpha_T^{\mathrm{op}}$; the MC equation splits into the Vol~I
modular MC equation ($\alpha_T^{\mathrm{cl}} = \Theta_\cA$),
a twisted cocycle condition on the $r$-matrix and PVA bracket
($d_\Theta\,\alpha_T^{\mathrm{mix}} = 0$), and the closure of
the line-category structure under the combined twisting.

\textsc{Theorem~K} (Curved Swiss-cheese and the modular shadow tower).
At genus~$g \geq 1$ the curved bar complex
$(\bar{B}^{\mathrm{ch},(g)}(\cA),\, \dfib,\, \Delta)$ with
$\dfib^{\,2} = \kappa(\cA) \cdot \omega_g$ is a curved
$\SCchtop$-algebra.  The coproduct remains coassociative; the total
corrected differential~$\Dg{g}$ restores flatness
over~$\overline{\mathcal{M}}_g$ by coupling the $\R$-ordering to
the Hodge bundle.
The universal MC element $\Theta_\cA$ admits a
genus expansion
$\Theta_\cA
= \Theta_\cA^{\mathrm{tree}}
+ \hbar\,\Theta_\cA^{(1)}
+ \hbar^2\,\Theta_\cA^{(2)}
+ O(\hbar^3)$.
At genus~$2$, the three shells are
$\Theta_\cA^{(2)}
= \Theta_{\cA,\mathrm{loop} \circ \mathrm{loop}}^{(2)}
+ \Theta_{\cA,\mathrm{sep} \circ \mathrm{loop}}^{(2)}
+ \Theta_{\cA,\mathrm{pf}}^{(2)}$
--- respectively iterated non-separating self-gluing, mixed
separating/non-separating clutching, and the first rigid
planted-forest correction.  Mok's cutting map
$\FM_n(W/B)(\rho) \to \prod_{v} \FM_{I_v}(Y_v \mathbin{|} D_v)$
turns each clutching identity into a push--pull factorization on
rigid logarithmic boundary strata.

\textsc{Theorem~L} (Deformation brace algebra).
Chiral Hochschild cochains on $\FM(\C) \times \Conf(\R)$ carry a
brace algebra governing deformations of the $\SCchtop$-algebra.
At genus~$0$,
$\ChirHoch^\bullet(\cA) \simeq
\ChirHoch^\bullet_{\mathrm{bulk}}(\cA)$.

\textsc{Theorem~M} (Half-space BV quantization).
The BV-BRST complex of a 3d holomorphic--topological gauge theory
on $\C \times \R_{\geq 0}$ with affine boundary condition
$\widehat{\fg}_k$ is quasi-isomorphic to the bar complex
$\barBch(\widehat{\fg}_k)$.  The quantization is \emph{one-loop
exact}: only genus-$0$ and genus-$1$ Feynman diagrams contribute,
and the boundary vertex algebra is $\widehat{\fg}_{k+h^\vee}$ at
the shifted level.  Principal Drinfeld--Sokolov reduction
transports the result to $\mathrm{Vir}_{c(k)}$ and $\mathcal{W}_3$.

\textsc{Theorem~N} (Modular PVA quantization).
The stable-graph modular bar coalgebra
$\barB^{\mathrm{mod}}(\cA)$ carries a filtered strict modular
deformation dg~Lie algebra
$\mathrm{Def}^{\mathrm{mod}}(\cA^{\mathrm{cl}}(V))$ whose MC
elements are lifts of the genus-$0$ classical PVA to all genera.
The universal obstruction at genus~$g$ is a class
$\mathrm{Ob}_g \in H^2(\mathrm{Def}^{(g)})$, computed recursively
from lower genera.  At genus~$1$: the obstruction is
$\kappa(\cA) \cdot \omega_1$ and the variational modular class
$[\varpi_1(\cA)]$ measures the infinitesimal variation of the
genus-$1$ lift with parameters.  The planted-forest $L_\infty$
model on $\FM_3(X)$ isolates the first nonlinear obstruction: the
residue-splitting operation on planted forests produces the
$L_\infty$ brackets of the modular deformation complex, and the
nested-collision sector is contractible.

\textsc{Theorem~P} (3d gravity from $SL(2,\R)$ Chern--Simons).
$SL(2,\R)$ CS at level $k = \ell/(4G)$ determines
$\alpha_{\mathrm{grav}} \in \mathrm{MC}(\mathfrak{g}^{\mathrm{SC}}_T)$
with boundary $B_\partial = \mathrm{Vir}_{6k}$ (Brown--Henneaux),
Koszul dual $B_\partial^! = \mathrm{Vir}_{26-6k}$, spectral braiding
$R(z) = $ Virasoro fusion kernel, $\kappa = (6k-26)/2$, quartic
contact $\mathfrak{Q} = 10/[6k(30k+22)]$.  The genus expansion of
the curved bar complex is the gravitational loop expansion; the
infinite-depth shadow tower is the perturbative quantum-gravity
series.  BTZ black holes of mass~$M$ and angular momentum~$J$ are
MC deformations with $L_0 = (M\ell + J)/2$; the Cardy formula
$S = 2\pi\sqrt{cL_0/6}$ is the asymptotic density of MC
deformations at high weight.

\textsc{Theorem~Q} (Anomaly-completed Koszul duality).
The transgression algebra
$B_\Theta = (B * \Bbbk\langle\eta\rangle)
/ (\eta b - (-1)^{|b|} b\eta,\; d\eta = \Theta)$ packages the
curvature obstruction into a single dg~algebra.  The secondary
anomaly $u = \eta^2$ is closed and central.  The genus-$g$
Clifford completion $\mathfrak{G}_g(B_\Theta)$ adjoins handle
operators with
$\alpha_i\beta_j + \beta_j\alpha_i = \delta_{ij}\,u$,
$\alpha_i^2 = \beta_j^2 = 0$.  Away from $u = 0$:
$\mathfrak{G}_g[u^{-1}] \cong
\mathrm{Mat}_{2^g}(B_\Theta[u^{-1}])$---higher genus is
Morita-invisible.  On $u = 0$:
$\mathfrak{G}_g/(u) \cong B/(\Theta) \otimes \Lambda^{2g+1}$.

\textsc{Theorem~R} (Logarithmic HT monodromy).
The shifted KZ connection
$\nabla = d - \hbar\,r(z)\,dz/z$ on
$\FM_k(\C) \times \Conf_k(\R)$ extends to an $\Ainf$
superconnection whose monodromy is the spectral $R$-matrix.
The holomorphic--topological associator
$\Phi_{\mathrm{HT}} \in \exp(\hat{\mathfrak{t}}_3)$
satisfies the pentagon identity.  The connection curvature at
genus~$g$ is $\kappa(\cA) \cdot \omega_g$; flat sections are
the modular corrections of Volume~I\@.

\medskip
\noindent\textsc{Yang--Mills boundary theory.}\enspace
The YM boundary algebra $B_{\partial,\mathrm{YM}}$ on $\C$ admits an
instanton-completed visible centre, a visible defect category of
screening charges, and a screening Laplacian whose spectral gap
$\lambda_1 > 0$ reduces the mass-gap question to a screening
domination estimate internal to the formalism.

\noindent\textsc{Celestial holography.}\enspace
Kaluza--Klein reduction from $6$d $\to$ $3$d
holomorphic--topological produces a celestial chiral algebra
$B_\partial^{\mathrm{cel}}$ on $\C P^1$; celestial OPE coefficients
are projections $\alpha_T|_{(n,0)}$ on $\FM_n(\C P^1)$.
Soft theorems are the leading spectral terms of $r(z)$:
$r(z) \sim S_0/z + S_1/z^2 + \cdots$, where $S_0$ is the
Weinberg soft factor and $S_1$ is the subleading soft factor of
Cachazo--Strominger.

\noindent\textsc{Factorization quantum groups.}\enspace
Three languages---homotopy dg-shifted Yangians, spectral
factorization quantum groups, and shifted quantum groups via
critical stable envelopes---are bridged by bar-horizontal
strictification.  The quasi-factorization theorem identifies the
spectral Drinfeld obstruction class.

\medskip
The archetype classification in boundary coordinates:
Heisenberg is Gaussian
($\ell_3^{(0)} = 0$, shadow tower terminates at weight~$2$);
affine Kac--Moody is Lie/tree
($\ell_3^{(0)} \neq 0$, $\ell_4^{(0)} = 0$,
terminates at weight~$3$);
$\beta\gamma$ is contact/quartic
($\ell_3^{(0)} = 0$ on the weight-changing line but
$\Theta_4^{\mathrm{contact}} \neq 0$,
terminates at weight~$4$);
Virasoro and $\mathcal{W}_N$ are mixed
($\ell_3^{(0)} \neq 0$, $\ell_4^{(0)} \neq 0$,
infinite tower).
For $\mathcal{W}_3$, the quartic quasi-primary
$\Lambda = {:}TT{:} - \tfrac{3}{10}\,\partial^2 T$
enters the $WW$ OPE with coefficient $16/(22+5c)$; this is a
genuinely new bar cogenerator whose divergence at $c = -22/5$
forces the planted-forest shell into the genus-two
decomposition.

\begin{itemize}[leftmargin=2em]
\item All algebraic structures are over $\C$ unless stated otherwise.

\item $\Ainf$ denotes the $A_\infty$ operad; $\Eone$, $\Etwo$, $\Einf$ the little-disks operads.

\item $\FM(X)$ denotes the Fulton--MacPherson compactification of $\Conf(X)$.
\item $m_k$ are the $\Ainf$ chiral operations; $\lambda_i$ are spectral parameters.
\item Volume~I notation: $\cA$ for chiral algebras, $\barB(\cA)$ for bar, $\Omegach(C)$ for cobar, $\cA^!$ for Koszul dual.

\subsection*{What this volume assumes from Volume~I}

Volume~I proves five structural theorems for chiral algebras on
curves.  This volume depends on all five.

\begin{enumerate}[label=\textbf{(\Alph*)},leftmargin=2.2em]
\item \textbf{Bar--cobar adjunction.}
  The geometric bar complex $\barB_X(\cA)$ and cobar complex
  $\Omegach(\barB_X(\cA))$ form an adjoint pair of functors on
  chiral algebras.  Volume~II reinterprets the bar complex as an
  algebra over the Swiss-cheese operad
  $\mathsf{SC}^{\mathrm{ch,top}}$; the adjunction is assumed, not
  reproved.
\item \textbf{Bar--cobar inversion.}
  On the Koszul locus the counit
  $\Omegach(\barB_X(\cA))\to \cA$ is a quasi-isomorphism.
  Volume~II lifts this to the raviolo vertex algebra setting and to
  completed towers via weightwise stabilization.
\item \textbf{Complementarity.}
  The deformation and obstruction complexes $\mathbf Q_g(\cA)$ and
  $\mathbf Q_g(\cA^!)$ are complementary Lagrangians inside the
  shifted-symplectic ambient complex
  $R\Gamma(\overline{\mathcal M}_g,\mathcal Z(\cA))$.  The
  $(-1)$-shifted symplectic structure is the genus-tower input for
  the curved Swiss-cheese algebras of Part~I.
\item \textbf{Modular characteristic.}
  The scalar $\kappa(\cA)$ and the $\hat A$-genus control the
  curvature $\dfib^{\,2}=\kappa(\cA)\cdot\omega_g$ of the genus
  tower.  The spectral discriminant $\Delta_\cA$ is the first
  non-scalar invariant.
\item \textbf{Hochschild duality.}
  The chiral Hochschild cochain complex
  $\mathrm{ChirHoch}^*(\cA)$ is polynomial on the Koszul locus and
  dual to $\mathrm{ChirHoch}^*(\cA^!)$.  It is the natural candidate
  for the bulk algebra in the $3$d holomorphic-topological theory.
\end{enumerate}
Volume~II computes the three-dimensional content of these five
theorems through the Swiss-cheese lens.

\subsection*{Conditional and unconditional results}
\label{subsec:conditional-roadmap}

All operadic foundations (Weiss descent, recognition theorem,
homotopy-Koszulity) are proved unconditionally.
Both the \textbf{Recognition} statement
(Theorem~\ref{thm:recognition}, proved via product decomposition
and retract) and \textbf{Homotopy--Koszulity}
(Theorem~\ref{thm:homotopy-Koszul}, proved via Kontsevich formality
and transfer from the classical Swiss--cheese operad) are now
established.  The full development includes claims conditional on
the standing analytic hypotheses~\textrm{(H1)--(H4)}, as
detailed below.

\begin{remark}[Status of results]
\label{rem:conditional-list}
All algebraic results in this volume are unconditional.  The only
conditional inputs are the four standing analytic hypotheses
\textrm{(H1)--(H4)}, which are verified in the three worked
examples but not proved in general:
